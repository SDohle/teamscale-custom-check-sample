006package com.teamscale.index.utils;
007
008import org.conqat.engine.index.shared.CommitDescriptor;
009import org.conqat.engine.service.shared.ServiceUtils;
010import org.conqat.engine.service.shared.client.ServiceClientUtils;
011import org.conqat.lib.commons.string.StringUtils;
012
013/** Utility methods for the creation of Teamscale links. */
014public class LinkCreationUtils {
015
016        /**
017         * Returns a link to the delta perspective for the given commit. In case the
018         * base url is null, the returned link is the empty {@link String}.
019         */
020        public static String createDeltaPerspectiveLink(String baseUrl, CommitDescriptor firstCommit,
021                        CommitDescriptor lastCommit, String projectIdOrAlias, boolean mergeRequestMode) {
022                if (baseUrl == null) {
023                        return "";
024                }
025                baseUrl = StringUtils.ensureEndsWith(baseUrl, "/");
026                String result = baseUrl + "delta.html#findings/" + ServiceUtils.encodePathSegment(projectIdOrAlias)
027                                + ServiceClientUtils.createOptionString("from", firstCommit.toServiceCallFormat(), "to",
028                                                lastCommit.toServiceCallFormat());
029                if (mergeRequestMode) {
030                        result += "&showMergeFindings=true";
031                }
032                return result;
033        }
034
035        /** Returns a link to the issue perspective for the given issue. */
036        public static String createIssuePerspectiveLink(String baseUrl, String issueId, String projectAlias) {
037                baseUrl = StringUtils.ensureEndsWith(baseUrl, "/");
038                return baseUrl + "issues.html#/" + ServiceUtils.encodePathSegment(projectAlias) + "/"
039                                + ServiceUtils.encodePathSegment(issueId);
040        }
041
042        /** Returns a link to the findings detail view for the given finding. */
043        public static String createFindingsDetailLink(String baseUrl, String projectAlias, String findingId,
044                        CommitDescriptor commit) {
045                baseUrl = StringUtils.ensureEndsWith(baseUrl, "/");
046                String optionString;
047                if (commit == null) {
048                        optionString = ServiceClientUtils.createOptionString("id", findingId);
049                } else {
050                        optionString = ServiceClientUtils.createOptionString("id", findingId, "t", commit.toServiceCallFormat());
051                }
052                return baseUrl + "findings.html#details/" + ServiceUtils.encodePathSegment(projectAlias) + optionString;
053        }
054
055        /** Returns the URL for the static findings badge. */
056        public static String createStaticFindingsBadgeUrl(String baseUrl, int addedFindings, int inChangedCodeFindings,
057                        int removedFindings) {
058                baseUrl = StringUtils.ensureEndsWith(baseUrl, "/");
059                String badgeServiceUrl = "api/badges/findings/static";
060                return baseUrl + badgeServiceUrl + "?added=" + addedFindings + "&changed=" + inChangedCodeFindings + "&removed="
061                                + removedFindings;
062        }
063
064}
