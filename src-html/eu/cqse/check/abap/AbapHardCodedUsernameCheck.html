006package eu.cqse.check.abap;
007
008import java.util.regex.Pattern;
009
010import eu.cqse.check.base.HardCodedLiteralsCheckBase;
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckGroupNames;
013import eu.cqse.check.framework.core.ECheckParameter;
014import eu.cqse.check.framework.core.EFindingEnablement;
015import eu.cqse.check.framework.core.option.ACheckOption;
016import eu.cqse.check.framework.scanner.ELanguage;
017import eu.cqse.check.framework.scanner.IToken;
018
019/**
020 * Checks for hard-coded user names. The check searches for occurrences of
021 * character literals which occur in context of an identifier which may indicate
022 * the it stores a user name, e.g. variables named like 'user_name', 'user',
023 * 'sys_uname', ...
024 */
025@ACheck(name = AbapHardCodedUsernameCheck.CHECK_NAME, description = "Hard-coded user names should be avoided, since hard-coded "
026                + "credentials like user name or password are a potential security risk. In any case it should be avoided that the "
027                + "program flow depends on the name of the currently active user."
028                + "", groupName = CheckGroupNames.SECURITY, languages = ELanguage.ABAP, defaultEnablement = EFindingEnablement.YELLOW, parameters = {
029                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
030public class AbapHardCodedUsernameCheck extends HardCodedLiteralsCheckBase {
031
032        /** Name of check */
033        /* package */ static final String CHECK_NAME = "Hard-coded user name";
034
035        /** {@ConQAT.Doc} */
036        @ACheckOption(name = CHECK_NAME
037                        + " - Ignore in context of SY-UNAME", description = "Ignores potential hard-coded usernames if they "
038                                        + "are used together with SY-UNAME. Used to suppress findings which are already identified with checks for "
039                                        + "" + UsageOfSystemVariablesCheck.CHECK_NAME + " or "
040                                        + ControlFlowDependingOnSystemVariableCheck.CHECK_NAME + ".")
041        private boolean ignoreSyUname = true;
042
043        /** {@ConQAT.Doc} */
044        @ACheckOption(name = CHECK_NAME
045                        + " - Restrict search to following user names", description = "Regular expressions to restrict search"
046                                        + "to typical user names - should be set to avoid false positive findings.")
047        private String userNameRegex = "[A-Z0-9_]{3,12}";
048
049        /** {@ConQAT.Doc} */
050        @ACheckOption(name = CHECK_NAME
051                        + " - Report all matching literals", description = "If set, all literals which match the user name pattern "
052                                        + "will be reported as finding. Should be only enabled if the user name pattern on its own is "
053                                        + "sufficient to identify user names.")
054        private boolean reportAllMatchingLiterals = false;
055
056        /** Pattern which should match for identifier to indicate user name */
057        private static final Pattern IDENTIFIER_PATTERN = Pattern.compile("(?i).*(uname|usr|user)(.?name|.?id)?$");
058
059        /** {@inheritDoc} */
060        @Override
061        protected String getFindingsMessageText() {
062                return "Hard-coded username";
063        }
064
065        /** {@inheritDoc} */
066        @Override
067        protected boolean isMatchingIdentifier(IToken identifier) {
068                String identifierText = identifier.getText();
069                if (ignoreSyUname && identifierText.equalsIgnoreCase("sy-uname")) {
070                        return false;
071                }
072                return IDENTIFIER_PATTERN.matcher(identifierText).matches();
073        }
074
075        /** {@inheritDoc} */
076        @Override
077        protected String getLiteralRegex() {
078                return userNameRegex;
079        }
080
081        /** {@inheritDoc} */
082        @Override
083        protected boolean findAllMatchingLiteral() {
084                return reportAllMatchingLiterals;
085        }
086
087}
