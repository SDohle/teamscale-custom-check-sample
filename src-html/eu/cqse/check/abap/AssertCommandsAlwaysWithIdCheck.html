008package eu.cqse.check.abap;
009
010import java.util.List;
011
012import eu.cqse.check.base.EntityCheckBase;
013import eu.cqse.check.framework.core.ACheck;
014import eu.cqse.check.framework.core.CheckException;
015import eu.cqse.check.framework.core.CheckGroupNames;
016import eu.cqse.check.framework.core.ECheckParameter;
017import eu.cqse.check.framework.scanner.ELanguage;
018import eu.cqse.check.framework.scanner.ETokenType;
019import eu.cqse.check.framework.scanner.IToken;
020import eu.cqse.check.framework.shallowparser.SubTypeNames;
021import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
022import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
023
024/**
025 * Checks whether ASSERT commands have an ID.
026 */
027@ACheck(name = "Assert commands should have ID", description = "Assert should always have an ID.", groupName = CheckGroupNames.BAD_PRACTICE, languages = ELanguage.ABAP, parameters = {
028                ECheckParameter.ABSTRACT_SYNTAX_TREE })
029public class AssertCommandsAlwaysWithIdCheck extends EntityCheckBase {
030
031        /** {@inheritDoc} */
032        @Override
033        protected String getXPathSelectionString() {
034                return "//STATEMENT[subtype('" + SubTypeNames.ASSERT + "')]";
035
036        }
037
038        /** {@inheritDoc} */
039        @Override
040        protected void processEntity(ShallowEntity entity) throws CheckException {
041                // check whether ID token exists, create finding if not
042                List<IToken> tokens = entity.ownStartTokens();
043                if (!TokenStreamUtils.containsAny(tokens, ETokenType.ID)) {
044                        createFinding("ASSERT statement without ID", entity);
045                }
046        }
047
048}
