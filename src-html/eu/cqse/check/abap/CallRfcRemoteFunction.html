017package eu.cqse.check.abap;
018
019import java.util.Set;
020
021import org.conqat.lib.commons.collections.CollectionUtils;
022
023import eu.cqse.check.framework.core.ACheck;
024import eu.cqse.check.framework.core.CheckException;
025import eu.cqse.check.framework.core.CheckGroupNames;
026import eu.cqse.check.framework.core.ECheckParameter;
027import eu.cqse.check.framework.scanner.ELanguage;
028import eu.cqse.check.framework.scanner.ETokenType;
029import eu.cqse.check.framework.util.abap.AbapCheckUtils;
030
031/**
032 * Checks for calls of function modules RFC_REMOTE_EXEC and RFC_REMOTE_PIPE.
033 * Here, the list of suspicious functions is fixed and not a parameter, since
034 * function calls are usual in ABAP. Calling other (unwanted) functions won't be
035 * a security issue in most cases but an architectural issue and should be
036 * better handled with architecture conformance analysis.
037 */
038@ACheck(name = "Suspicious call of function modules RFC_REMOTE_EXEC or RFC_REMOTE_PIPE", description = "<p>The function modules RFC_REMOTE_EXEC and RFC_REMOTE_PIPE allow "
039                + "to execute operating system commands. In custom code, these function modules should not be used. "
040                + "The <a href=\"https://help.sap.com/abapdocu_750/en/abenabap_system_commands_appl_sxpg.htm\">SXPG framework</a> "
041                + "should be used instead.</p>", groupName = CheckGroupNames.SECURITY, languages = ELanguage.ABAP, parameters = {
042                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
043public class CallRfcRemoteFunction extends CallStatementCheckBase {
044
045        /**
046         * Set of suspicious function for which a finding should be created
047         */
048        private static final Set<String> SUSPICIOUS_FUNCTIONS = CollectionUtils.asHashSet("RFC_REMOTE_EXEC",
049                        "RFC_REMOTE_PIPE");
050
051        /** {@inheritDoc} */
052        @Override
053        protected void processCallStatement() throws CheckException {
054                // search for token streams [CALL, FUNCTION, 'RFC_REMOTE_EXEC']
055                // and [CALL, FUNCTION, 'RFC_REMOTE_PIPE']
056
057                if (getCallStatmentTokens().size() < 3) {
058                        return;
059                }
060
061                String function = AbapCheckUtils.getUnquotedTextForCharcterLiteral(getCallStatmentTokens().get(2));
062
063                if (SUSPICIOUS_FUNCTIONS.contains(function)) {
064                        createFinding("Call of function " + function, getCallStatement());
065                }
066        }
067
068        /** {@inheritDoc} */
069        @Override
070        protected boolean isTargetedCallStatement(ETokenType typeOfsecondToken) {
071                return typeOfsecondToken == ETokenType.FUNCTION;
072        }
073
074}
