017package eu.cqse.check.abap;
018
019import org.conqat.lib.commons.collections.CollectionUtils;
020import org.conqat.lib.commons.collections.UnmodifiableList;
021
022import eu.cqse.check.base.EntityCheckBase;
023import eu.cqse.check.framework.core.CheckException;
024import eu.cqse.check.framework.scanner.ETokenType;
025import eu.cqse.check.framework.scanner.IToken;
026import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
027
028/**
029 * Base class for checks related to statements which are introduced by a CALL
030 * statement in ABAP, see http://help.sap.com/abapdocu_751/en/abapcall.htm.
031 * Check based on this class may only be implemented for statements which
032 * require a leading CALL keyword, for example CALL cfunc, CALL FUNCTION, CALL
033 * TRANSACTION. The specific statement which should be checked by a check of
034 * this calls is determined by the second token. The match condition for the
035 * second token has to be implemented by sub-classes in
036 * {@link #isTargetedCallStatement(ETokenType)}.
037 *
038 * Note that the this base class is in general not suitable for checks on method
039 * calls, since there is the alternative functional syntax without preceding
040 * CALL keyword. (Method calls should be the only statements where an
041 * alternative syntax without CALL exists).
042 */
043public abstract class CallStatementCheckBase extends EntityCheckBase {
044
045        /** Tokens of the CALL statement */
046        private UnmodifiableList<IToken> callStatementTokens = CollectionUtils.emptyList();
047
048        /** The selected CALL ... statement */
049        private ShallowEntity callStatement;
050
051        /** {@inheritDoc} */
052        @Override
053        protected String getXPathSelectionString() {
054                return "//STATEMENT[subtype('call')]";
055        }
056
057        /**
058         * Determines if the call statement is the targeted kind of the CALL ....
059         * statement.
060         *
061         * @param typeOfsecondToken
062         *            type of token following the CALL keyword
063         * @return <code>true</code> if
064         */
065        protected abstract boolean isTargetedCallStatement(ETokenType typeOfsecondToken);
066
067        /** {@inheritDoc} */
068        @Override
069        protected final void processEntity(ShallowEntity entity) throws CheckException {
070                callStatement = entity;
071                callStatementTokens = entity.ownStartTokens();
072
073                // fist token is the CALL keyword (ensured by subtype),
074                // second token must be that of target type, otherwise nothing to check
075                if (callStatementTokens.size() < 2 || !isTargetedCallStatement(callStatementTokens.get(1).getType())) {
076                        return;
077                }
078                processCallStatement();
079        }
080
081        /**
082         * Processes the selected CALL ... statement entity. This method is only
083         * called from {@link #processEntity(ShallowEntity)} if the first token is
084         * the CALL keyword and the second token is of a type which is accepted by
085         * {@link #isTargetedCallStatement(ETokenType)}.
086         */
087        protected abstract void processCallStatement() throws CheckException;
088
089        /**
090         * @see #callStatement
091         */
092        protected ShallowEntity getCallStatement() {
093                return callStatement;
094        }
095
096        /**
097         * @return the tokens of the selected CALL statement. First token is the
098         *         CALL keyword, second token is of a type accepted by
099         *         {@link #isTargetedCallStatement(ETokenType)}.
100         */
101        protected UnmodifiableList<IToken> getCallStatmentTokens() {
102                return callStatementTokens;
103        }
104
105}
