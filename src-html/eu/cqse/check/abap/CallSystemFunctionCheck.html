033/**
034 * Checks for usages of CALL cfunc - see
035 * <a href="http://help.sap.com/abapdocu_740/en/ABAPCALL-.htm">ABAP keyword
036 * documentation</a>. Names of functions are configurable by blacklist or
037 * whitelist. When CALL 'SYSTEM' is detected, allowed operating system commands
038 * can be specified in a specific whitelist.
039 */
040// TODO (AR) The markup in the description will break the
041// CustomChecksLatexExportService since that can only handle HTML.
042@ACheck(name = CallSystemFunctionCheck.CHECK_NAME, description = "The internal statement `CALL cfunc` executes a system "
043                + "function `cfunc`. Some system functions, e.g. `SYSTEM` or `ThWpInfo` allow to execute any command of the operating system "
044                + "and should be avoided. \n\n"
045                + "The [SXPG framework]((https://help.sap.com/abapdocu_750/en/abenabap_system_commands_appl_sxpg.htm))"
046                + "should be used instead.", groupName = CheckGroupNames.SECURITY, languages = ELanguage.ABAP, parameters = {
047                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
048public class CallSystemFunctionCheck extends CallStatementCheckBase {
049
050        /** Name of this check */
051        /* package */ static final String CHECK_NAME = "Suspicious usage of CALL system function";
052
053        /**
054         * Set of {@link ETokenType}s which may follow CALL and indicate a different
055         * statement than CALL system function, see
056         * http://help.sap.com/abapdocu_751/en/abapcall.htm and
057         * http://help.sap.com/abapdocu_751/en/abenabap_index.htm#C
058         */
059        private static final EnumSet<ETokenType> OTHER_CALL_TOKEN_TYPES = EnumSet.of(BADI, CUSTOMER_FUNCTION, DATABASE,
060                        DIALOG, FUNCTION, METHOD, SCREEN, SELECTION_SCREEN, SUBSCREEN, TRANSACTION, TRANSFORMATION);
061
062        /** {@ConQAT.Doc} */
063        @ACheckOption(name = CHECK_NAME
064                        + " - System function names", description = "Comma separated list of system function names. "
065                                        + "Depending on the blacklist/whitelist flag, this specifies the unwanted or allowed functions.")
066        private Set<String> functionNames = new HashSet<String>(CollectionUtils.asHashSet("SYSTEM", "ThWpInfo"));
067
068        /** {@ConQAT.Doc} */
069        @ACheckOption(name = CHECK_NAME
070                        + " - Use as Whitelist", description = "If enabled, function names specify allowed functions, "
071                                        + "all other usages of 'CALL cfunc' are detected.")
072        private boolean isWhitelist = false;
073
074        /** {@ConQAT.Doc} */
075        @ACheckOption(name = CHECK_NAME
076                        + " - Allowed commands in CALL 'SYSTEM'", description = "Comma separated list (regular expressions) for "
077                                        + "operating system commands which are allowed in CALL 'SYSTEM'. If system function 'SYSTEM' is whitelisted, "
078                                        + "this option has no effect.")
079        private Set<String> osCommandWhitelist = Collections.emptySet();
080
081        /**
082         * {@inheritDoc} Here, the test is if the second token must not be any of
083         * {@link #OTHER_CALL_TOKEN_TYPES}. This specific list of other token types is
084         * used since checking only for literals and identifiers is not sufficient to
085         * cover also syntactic sugar, for example parentheses.
086         */
087        @Override
088        protected boolean isTargetedCallStatement(ETokenType typeOfsecondToken) {
089                return !OTHER_CALL_TOKEN_TYPES.contains(typeOfsecondToken);
090        }
091
092        /** {@inheritDoc} */
093        @Override
094        protected void processCallStatement() throws CheckException {
095
096                String calledFunction = CheckUtils.getUnquotedTextForCharacterLiteral(getCallStatmentTokens().get(1));
097
098                if (calledFunction == null) {
099                        createFinding("Usage of `CALL cfunc` (dynamic call)", getCallStatement());
100                        return;
101                }
102
103                if (functionNames.contains(calledFunction) == isWhitelist) {
104                        return;
105                }
106
107                if (calledFunction.equals("SYSTEM") && isWhitlistedOsCommand()) {
108                        return;
109                }
110
111                createFinding("Usage of `CALL '" + calledFunction + "'`", getCallStatement());
112        }
113
114        /**
115         * Checks if the OS command is whitelisted when used in CALL 'SYSTEM'.
116         */
117        private boolean isWhitlistedOsCommand() {
118                if (CollectionUtils.isNullOrEmpty(osCommandWhitelist)) {
119                        return false;
120                }
121
122                String command = findCommand(getCallStatmentTokens());
123
124                if (command == null) {
125                        return false;
126                }
127
128                return osCommandWhitelist.stream().anyMatch(s -> command.matches(s));
129        }
130
131        /**
132         * Searches for the OS command in the token stream which gets executed, which is
133         * given by the token sequence "ID 'COMMAND' FIELD command". While syntactically
134         * possible to specify more than one OS command, the first one gets executed
135         * only (tested on SAP NW ABAP 7.50SP2).
136         *
137         * @return the OS command string or <code>null</code> if no command is found.
138         */
139        private static String findCommand(UnmodifiableList<IToken> tokens) {
140
141                // search for token sequence "ID 'COMMAND' FIELD command"
142                for (int i = 0; i < tokens.size() - 3; i++) {
143                        if (tokens.get(i).getType() == ETokenType.ID && tokens.get(i + 1).getText().equals("'COMMAND'")
144                                        && tokens.get(i + 2).getType() == ETokenType.FIELD) {
145                                return CheckUtils.getUnquotedTextForCharacterLiteral(tokens.get(i + 3));
146                        }
147                }
148                return null;
149        }
