006package eu.cqse.check.abap;
007
008import eu.cqse.check.framework.core.ACheck;
009import eu.cqse.check.framework.core.CheckException;
010import eu.cqse.check.framework.core.CheckGroupNames;
011import eu.cqse.check.framework.core.EFindingEnablement;
012import eu.cqse.check.framework.core.option.ACheckOption;
013import eu.cqse.check.framework.scanner.ELanguage;
014import eu.cqse.check.framework.scanner.IToken;
015
016/**
017 * Checks for usages of system variables, eg. SY-UNAME, SY-MANDT, ...
018 */
019@ACheck(name = ControlFlowDependingOnSystemVariableCheck.CHECK_NAME, description = "<p>A system variable, e.g., SY-UNAME "
020                + "(user name of current user) or SY-SYSID (SAP system id) is misused to control the program flow. "
021                + "This should should be avoided, since it may lead to different behavior depending on logon "
022                + "data or platform.</p>", groupName = CheckGroupNames.SECURITY, parameters = {}, languages = ELanguage.ABAP, defaultEnablement = EFindingEnablement.RED)
023public class ControlFlowDependingOnSystemVariableCheck extends SystemVariablesCheckBase {
024
025        /** Name of this check */
026        static final String CHECK_NAME = "Control flow depending on system variable (SY-...)";
027
028        /** {@ConQAT.Doc} */
029        @ACheckOption(name = CHECK_NAME + " - relevant system fields", description = ""
030                        + "Fields of SY structure which should be searched.")
031        private String fieldsToCheck = "UNAME, MANDT, SYSID, HOST, OPSYS, DBSYS";
032
033        /** {@inheritDoc} */
034        @Override
035        protected String getFieldsToCheck() {
036                return fieldsToCheck;
037        }
038
039        /** {@inheritDoc} */
040        @Override
041        protected void performSystemVariableCheck(IToken token) throws CheckException {
042                if (isInControlStatement(token)) {
043                        createFinding("Control flow depends on system variable " + token.getText().toUpperCase(), token);
044
045                }
046        }
047
048}
