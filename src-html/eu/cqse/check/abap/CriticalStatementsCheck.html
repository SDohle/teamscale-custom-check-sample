006package eu.cqse.check.abap;
007
008import java.util.HashSet;
009import java.util.Set;
010
011import org.conqat.lib.commons.collections.CollectionUtils;
012
013import eu.cqse.check.framework.core.ACheck;
014import eu.cqse.check.framework.core.CheckGroupNames;
015import eu.cqse.check.framework.core.ECheckParameter;
016import eu.cqse.check.framework.core.EFindingEnablement;
017import eu.cqse.check.framework.core.option.ACheckOption;
018import eu.cqse.check.framework.scanner.ELanguage;
019
020/**
021 * Check which searches for critical statements.
022 */
023@ACheck(name = CriticalStatementsCheck.CHECK_NAME, description = "Critical statements should be avoided since the danger to introduce "
024                + "security flaws is high. Usually these statements should be never used in production code. These statements might be necessary e.g., for "
025                + "code generators or other development related activities, but then the containing objects should be placed in packages which "
026                + " will not be transported to the production system. Usages of critical statements should properly document why the statement is needed.\n\n"
027                + "The statements which are regarded as critical can be specified in the analysis profile.\n\n"
028                + "Statements which are considered as critical by default are:\n"
029                + "* `EDITOR-CALL` - allows to modify ABAP programs regardless if development authorization is granted to the executing user \n"
030                + "* `GENERATE SUBROUTINE POOL` - generates FORM routines at runtime, security critical code could be obfuscated in this way. "
031                + "A severe vulnerability exists if the code is generated from an external input. \n"
032                + "* `INSERT REPORT` -  generates programs (REPORTS) at runtime, as with `GENERATE SUBROUTINE POOL`, security critical code could "
033                + "be obfuscated in this way. A severe vulnerability exists if the code is generated from external input.", groupName = CheckGroupNames.SECURITY, defaultEnablement = EFindingEnablement.RED, languages = ELanguage.ABAP, parameters = {
034                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
035public class CriticalStatementsCheck extends CriticalStatementsCheckBase {
036
037        /** Name of the check */
038        /* package */ static final String CHECK_NAME = "Critical Statement";
039
040        /** {@ConQAT.Doc} */
041        @ACheckOption(name = CHECK_NAME
042                        + " - List of Statements", description = "Comma separted list of keywords which introduce critical statemetns.")
043        private Set<String> criticalStatements = new HashSet<String>(
044                        CollectionUtils.asHashSet("EDITOR-CALL", "GENERATE SUBROUTINE POOL", "INSERT REPORT"));
045
046        /**
047         * @return the set of statements which should be
048         */
049        @Override
050        protected Set<String> getStatements() {
051                return criticalStatements;
052        }
053
054        /** {@inheritDoc} */
055        @Override
056        protected String getFindingsMessage() {
057                return "Critical statement: ";
058        }
059}
