006package eu.cqse.check.abap;
007
008import java.util.HashMap;
009import java.util.Iterator;
010import java.util.Map;
011import java.util.Set;
012
013import org.conqat.lib.commons.string.StringUtils;
014
015import eu.cqse.check.base.EntityCheckBase;
016import eu.cqse.check.framework.core.CheckException;
017import eu.cqse.check.framework.scanner.ETokenType;
018import eu.cqse.check.framework.shallowparser.SubTypeNames;
019import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
020import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
021import eu.cqse.check.framework.shallowparser.languages.abap.AbapShallowParser;
022
023/**
024 * Base class of searches for critical statements.
025 */
026public abstract class CriticalStatementsCheckBase extends EntityCheckBase {
027
028        /** String for {@link #getXPathSelectionString()} */
029        private String xPathSelectonString;
030
031        /**
032         * Maps a sub-type to the expected tokens if the sub-type alone is not
033         * sufficient to identify the statement because the statement is introduced
034         * by a sequence of keywords.
035         */
036        private final Map<String, ETokenType[]> subTypeToTokensMap = new HashMap<>();
037
038        /** {@inheritDoc} */
039        @Override
040        protected String getXPathSelectionString() {
041                parseStatementsOption();
042                return xPathSelectonString;
043        }
044
045        /**
046         * Parses the option list of critical statements, there
047         * {@link #xPathSelectonString} is build and {@link #subTypeToTokensMap} is
048         * filled. The first keyword is in general assumed to be also the sub-type
049         * name of the statement (in lower-case). The only exception is EXEC SQL,
050         * where the sub-type name is {@link SubTypeNames#NATIVE_SQL}. This should
051         * be the only relevant exception, see
052         * {@link AbapShallowParser#createStatementRules()}
053         */
054        @SuppressWarnings("javadoc")
055        private void parseStatementsOption() {
056                StringBuilder xPathSelection = new StringBuilder("//STATEMENT[subtype('");
057                for (Iterator<String> iterator = getStatements().iterator(); iterator.hasNext();) {
058                        String[] keywords = iterator.next().trim().split("\\s+");
059                        if (StringUtils.isEmpty(keywords[0])) {
060                                continue;
061                        }
062                        String subType = keywords[0].toLowerCase();
063                        if (subType.equals("exec")) {
064                                subType = SubTypeNames.NATIVE_SQL;
065                        }
066                        xPathSelection.append(subType);
067                        if (iterator.hasNext()) {
068                                xPathSelection.append("') or subtype('");
069                        }
070                        if (keywords.length > 1) {
071                                addToSubTypesToTokensMap(subType, keywords);
072                        }
073                }
074                xPathSelection.append("')]");
075                xPathSelectonString = xPathSelection.toString();
076        }
077
078        /**
079         * Adds the given array of keywords to {@link #subTypeToTokensMap}.
080         */
081        private void addToSubTypesToTokensMap(String subType, String[] keywords) {
082                ETokenType[] tokenTypes = new ETokenType[keywords.length];
083                for (int i = 0; i < keywords.length; i++) {
084                        tokenTypes[i] = ETokenType.valueOf(keywords[i].toUpperCase());
085                }
086                subTypeToTokensMap.put(subType, tokenTypes);
087        }
088
089        /** {@inheritDoc} */
090        @Override
091        protected void processEntity(ShallowEntity entity) throws CheckException {
092                String subtype = entity.getSubtype();
093                String statement;
094                if (subTypeToTokensMap.containsKey(subtype)) {
095                        ETokenType[] startTokens = subTypeToTokensMap.get(subtype);
096                        if (!TokenStreamUtils.startsWith(entity.ownStartTokens(), startTokens)) {
097                                return;
098                        }
099                        statement = StringUtils.concat(startTokens, StringUtils.SPACE);
100                } else {
101                        statement = subtype.toUpperCase();
102                }
103                createFinding(getFindingsMessage() + statement, entity);
104        }
105
106        /**
107         * @return the set of statements which should be searched, this list is
108         *         stated by an option of the concrete check class
109         */
110        protected abstract Set<String> getStatements();
111
112        /**
113         * @return the findings message, will be appended with the name of the
114         *         statement.
115         */
116        protected abstract String getFindingsMessage();
117}
