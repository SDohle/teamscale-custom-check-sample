006package eu.cqse.check.abap;

008import java.util.List;
009import java.util.stream.Collectors;
010
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckException;
013import eu.cqse.check.framework.core.CheckGroupNames;
014import eu.cqse.check.framework.core.CheckImplementationBase;
015import eu.cqse.check.framework.core.EFindingEnablement;
016import eu.cqse.check.framework.core.phase.ECodeViewOption;
017import eu.cqse.check.framework.scanner.ELanguage;
018import eu.cqse.check.framework.scanner.ETokenType;
019import eu.cqse.check.framework.scanner.IToken;
020import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
021
022/**
023 * Checks for usages of CLIENT SPECIFIED addition which can be used in Open SQL
024 * statements. The check searches for any occurrence of keyword sequence CLIENT
025 * specified since these should only occur within the Open SQL statements.
026 */
027@ACheck(name = "Cross-client database access", description = CrossClientDatabaseAccessCheck.FINDING_DESCRIPTION, groupName = CheckGroupNames.SECURITY, languages = ELanguage.ABAP, defaultEnablement = EFindingEnablement.RED)
028public class CrossClientDatabaseAccessCheck extends CheckImplementationBase {
029
030        /** The finding description. */
031        protected final static String FINDING_DESCRIPTION = "The addition `CLIENT SPECIFIED` for the "
032                        + "Open SQL statements `SELECT`, `INSERT`, `UPDATE`, `MODIFY`, `DELETE` and `OPEN CURSOR` allows access "
033                        + "to data of an other client than the client to which the current user is logged in. "
034                        + "Since there is no authority check performed, a user could possibly access data "
035                        + "in a client where he has no authorization.\n\n"
036                        + "Note that there is no need to add a condition to the `WHERE` clause for restricting "
037                        + "the selection to the current logon client (e.g. `MANDT = SY-MANDT`) together with "
038                        + "`CLIENT-SPECIFIED`. Without `CLIENT SPECIFIED` this is ensured by the Open SQL statements.\n\n"
039                        + "See also  [ABAP Security Note on Cross-Client Database Access]((http://help.sap.com/abapdocu_751/en/abenclient_dependent_scrty.htm)).";
040
041        /** {@inheritDoc} */
042        @Override
043        public void execute() throws CheckException {
044                List<IToken> keywordTokens = context.getTokens(ECodeViewOption.FILTERED_PREPROCESSED).stream()
045                                .filter(token -> token.getType().isKeyword()).collect(Collectors.toList());
046                for (int i : TokenStreamUtils.firstTokenOfTypeSequences(keywordTokens, 0, ETokenType.CLIENT,
047                                ETokenType.SPECIFIED)) {
048                        createFinding("Cross-client database access (usage of `CLIENT SPECIFIED`)", keywordTokens.get(i),
049                                        keywordTokens.get(i + 1));
050                }
051        }
052}
