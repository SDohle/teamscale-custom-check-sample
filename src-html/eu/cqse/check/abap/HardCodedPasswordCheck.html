006package eu.cqse.check.abap;
007
008import java.util.regex.Pattern;
009
010import eu.cqse.check.framework.core.ACheck;
011import eu.cqse.check.framework.core.CheckGroupNames;
012import eu.cqse.check.framework.core.ECheckParameter;
013import eu.cqse.check.framework.core.EFindingEnablement;
014import eu.cqse.check.framework.core.option.ACheckOption;
015import eu.cqse.check.framework.scanner.ELanguage;
016import eu.cqse.check.framework.scanner.IToken;
017
018/**
019 * Checks for hard-coded user names. The check searches for occurrences of
020 * character literals which occur in context of an identifier which is may
021 * indicate the it stores a user name, e.g. variables named like 'user_name',
022 * 'user', 'sys_uname', ...
023 */
024@ACheck(name = HardCodedPasswordCheck.CHECK_NAME, description = "<p>Hard-coded passwords should be avoided, since hard-coded "
025                + "credentials like user name or password are a potential security risk. Anybody who has access to the source code "
026                + "(regardless if on the development or production system) would be able to read the password.</p>"
027                + "", groupName = CheckGroupNames.SECURITY, languages = ELanguage.ABAP, defaultEnablement = EFindingEnablement.YELLOW, parameters = {
028                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
029public class HardCodedPasswordCheck extends HardCodedLiteralsCheckBase {
030
031        /** Name of check */
032        /* package */ static final String CHECK_NAME = "Hard-coded password";
033
034        /** {@ConQAT.Doc} */
035        @ACheckOption(name = CHECK_NAME
036                        + " - Restrict to password rule", description = "Regular expressions for password rule which must match, "
037                                        + "e.g. '^(?=.*[A-Za-z])(?=.*\\d)\\S{8,}$' for 'at least one alphabet letter, at least one "
038                                        + "digit, no white spaces and minimum length of 5 characters'.")
039        private String passwordRegex = "^(?=.*[A-Za-z])(?=.*\\d)\\S{5,}$";
040
041        /** Pattern which should match for identifier to indicate user name */
042        private static final Pattern IDENTIFIER_PATTERN = Pattern
043                        .compile("(?i).*(pass.?word|passwd|auth.*token|passwort|kennwort).*");
044
045        /** {@inheritDoc} */
046        @Override
047        protected String getFindingsMessageText() {
048                return "Hard-coded password";
049        }
050
051        /** {@inheritDoc} */
052        @Override
053        protected boolean isMatchingIdentifier(IToken identifier) {
054                return IDENTIFIER_PATTERN.matcher(identifier.getText()).matches();
055        }
056
057        /** {@inheritDoc} */
058        @Override
059        protected String getLiteralRegex() {
060                return passwordRegex;
061        }
062}
