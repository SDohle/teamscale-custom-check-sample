005import java.util.Optional;
006import java.util.Set;

009import eu.cqse.check.framework.core.CheckException;

012import eu.cqse.check.framework.core.EFindingEnablement;

014import eu.cqse.check.framework.core.util.CheckUtils;
015import eu.cqse.check.framework.scanner.ELanguage;
016import eu.cqse.check.framework.scanner.ETokenType;
017import eu.cqse.check.framework.scanner.IToken;
018import eu.cqse.check.framework.shallowparser.SubTypeNames;
019import eu.cqse.check.framework.shallowparser.TokenStreamUtils;

026/**
027 * Checks for usages of CALL TRANSACTION where no authority check is done right
028 * before. This check does not identify situations where an authority check is
029 * done, but checking the return code sy-subrc is missing. This is handled by a
030 * separate check.
031 */
032@ACheck(name = MissingAuthorityCheckForCallTransactionCheck.CHECK_NAME, description = "`CALL TRANSACTION` executes an transaction with "
033                + "the given transaction code. Until SAP_BASIS release 7.40 no authority check was performed for `CALL TRANSACTION`, "
034                + "from SAP_BASIS release 7.40 on, a authority check is only performed, if the addition `WITH AUTHORITY CHECK` is used.\n\n"
035                + "To solve this issue, the addition `WITH AUTHORITY CHECK` should be added (if SAP_BASIS version is 7.40 or higher)"
036                + "or the function module `AUTHORITY_CHECK_TCODE` should be called before.", groupName = CheckGroupNames.SECURITY, languages = ELanguage.ABAP, parameters = {
037                                ECheckParameter.ABSTRACT_SYNTAX_TREE }, defaultEnablement = EFindingEnablement.RED)
038public class MissingAuthorityCheckForCallTransactionCheck extends CallStatementCheckBase {
039
040        /**
041         * Function name of function module to check authorization of CALL TRANSACTION
042         */
043        /* package */ static final String AUTHORITY_CHECK_TCODE = "AUTHORITY_CHECK_TCODE";
044
045        /** Name of the check */
046        /* package */ static final String CHECK_NAME = "Missing authority check at CALL TRANSACTION";
047
048        /** {@ConQAT.Doc} */
049        @ACheckOption(name = CHECK_NAME
050                        + " - Non-critical transactions", description = "Comma separated list of transaction codes"
051                                        + "(regular expressions) which are non-critical and thus not require an authority-check.")
052        private Set<String> nonCriticalTransactions = Collections.emptySet();
053
054        /** {@inheritDoc} */
055        @Override
056        protected boolean isTargetedCallStatement(ETokenType typeOfsecondToken) {
057                return typeOfsecondToken == ETokenType.TRANSACTION;
058        }
059
060        /** {@inheritDoc} */
061        @Override
062        protected void processCallStatement() throws CheckException {
063                if (hasWithAuthorityCheckAddition()) {
064                        return;
065                }
066                checkAuthorityCheckBeforeAndCreateFindings();
067        }
068
069        /**
070         * @return <code>true</code> if the CALL TRANSACTION statement contains the
071         *         addition WITH AUTHORITY-CHECK.
072         */
073        private boolean hasWithAuthorityCheckAddition() {
074                return TokenStreamUtils.firstTokenOfTypeSequence(getCallStatmentTokens(), 0, ETokenType.WITH,
075                                ETokenType.AUTHORITY_CHECK) != TokenStreamUtils.NOT_FOUND;
076        }
077
078        /**
079         * Checks if a proper authority check is stated before the CALL TRANSACTION.
080         */
081        private void checkAuthorityCheckBeforeAndCreateFindings() throws CheckException {
082                IToken calledTransaction = getCallStatmentTokens().get(2);
083                if (isNoncriticalTransaction(calledTransaction)) {
084                        return;
085                }
086                ShallowEntity expectedAuthorityCheck = findStatementWhichShouldBeAuthorityCheck();
087                if (isAuthorityCheck(expectedAuthorityCheck, calledTransaction)) {
088                        return;
089                }
090                createFinding("Missing authority check before CALL TRANSACTION", getCallStatement());
091        }
092
093        /**
094         * @return <code>true</code> if the called transaction is stated as literal and
095         *         matches any of {@link #nonCriticalTransactions}.
096         */
097        private boolean isNoncriticalTransaction(IToken calledTransaction) {
098                String transactionCode = CheckUtils.getUnquotedTextForCharacterLiteral(calledTransaction);
099                if (transactionCode == null) {
100                        return false;
101                }
102                return nonCriticalTransactions.stream().anyMatch(s -> transactionCode.matches(s));
103        }
104
105        /**
106         * Checks if the given entity is an authority check for CALL transaction. In
107         * case of ineffective authority checks the methods will still return
108         * <code>true</code> but create specific findings.
109         *
110         * @param entity
111         *            which should be checked, can be <code>null</code>
112         * @return <code>true</code> if the entity is an authority check for CALL
113         *         TRANSACTION, <code>false</code> if entity is others or
114         *         <code>null</code>
115         */
116        private boolean isAuthorityCheck(ShallowEntity entity, IToken calledTransaction) throws CheckException {
117                if (entity == null) {
118                        return false;
119                }
120                Optional<FunctionCallInfo> functionCallInfo = LanguageFeatureParser.ABAP.getFunctionCallInfo(entity);
121                if (!functionCallInfo.isPresent()) {
122                        return false;
123                }
124                if (!AUTHORITY_CHECK_TCODE.equals(functionCallInfo.get().getFunctionName())) {
125                        return false;
126                }
127                createFindingsInCaseOfIneffectiveAuthorityCheck(entity, calledTransaction, functionCallInfo.get());
128                return true;
129        }
130
131        /**
132         * Checks if the authority check is ineffective and creates findings in that
133         * case. This is given if e.g. no transaction code is provided, wrong
134         * transaction is checked or error code for exception "not_ok" is not set.
135         */
136        private void createFindingsInCaseOfIneffectiveAuthorityCheck(ShallowEntity entity, IToken calledTransaction,
137                        FunctionCallInfo authortiyCheckCallInfo) throws CheckException {
138                Optional<IToken> authorizedTransaction = authortiyCheckCallInfo.getPassedExportingToken("tcode");
139
140                if (!authorizedTransaction.isPresent()) {
141                        createFinding("No transaction code passed to authority check before CALL TRANSACTION", entity);
142                } else if (!TokenUtils.isEqualTypeAndText(authorizedTransaction.get(), calledTransaction)) {
143                        createFinding(
144                                        "Ineffective authority check before CALL TRANSACTION: Checks for transaction code "
145                                                        + authorizedTransaction.get().getText() + " but called is " + calledTransaction.getText(),
146                                        authorizedTransaction.get());
147                }
148
149                if (!authortiyCheckCallInfo.isSettingErrorCodeForException("not_ok")) {
150                        createFinding(
151                                        "Ineffective authority check before CALL TRANSACTION: No error code set for exception 'not_ok'",
152                                        entity);
153                }
154        }
155
156        /**
157         * Searches for the statement that should be an authority check. There are three
158         * expected cases:
159         * <ol>
160         * <li>The CALL TRANSACTION statement is the first statement within an IF
161         * checking the return code sy-subrc. Then, the statement before the IF and with
162         * the same parent as the IF is relevant.
163         * <li>The CALL TRANSACTION statement is the first statement within the ELSE
164         * branch of an IF checking the return code sy-subrc. Again, the statement
165         * before the IF and with the same parent as the IF is relevant.
166         * <li>THE CALL TRANSACTION is not in the IF or ELSE branch of checking sy-subrc
167         * return code. In that case it is expected that the statement before CALL
168         * TRANSACTION is an IF checking sy-subrc and the relevant statement is that
169         * before this IF.
170         * </ol>
171         */
172        private ShallowEntity findStatementWhichShouldBeAuthorityCheck() throws CheckException {
173                ShallowEntity parent = getCallStatement().getParent();
174                if (MissingCheckOfSySubrcCheckBase.isConditionalOnSySubrc(parent) && isFirstChildStatement(getCallStatement())) {
175                        return getPrecedingStatement(parent);
176                }
177
178                if (parent.getType() == EShallowEntityType.STATEMENT && SubTypeNames.ELSE.equals(parent.getSubtype())
179                                && isFirstChildStatement(getCallStatement())) {
180                        ShallowEntity ifStatement = findSiblingIfForElse(parent);
181                        if (MissingCheckOfSySubrcCheckBase.isConditionalOnSySubrc(ifStatement)) {
182                                return getPrecedingStatement(ifStatement);
183                        }
184                }
185
186                ShallowEntity precedingEntity = getPrecedingStatement(getCallStatement());
187                if (MissingCheckOfSySubrcCheckBase.isConditionalOnSySubrc(precedingEntity)) {
188                        precedingEntity = getPrecedingStatement(precedingEntity);
189                }
190                return precedingEntity;
191        }
192
193        /**
194         * @return an IF statement which is a sibling of the given entity and is next
195         *         before the given entity. <code>null</code> if no such entity exists.
196         */
197        private ShallowEntity findSiblingIfForElse(ShallowEntity entity) throws CheckException {
198                List<ShallowEntity> ifSiblings = select(entity, "preceding-sibling::STATEMENT[subtype('if')]");
199                if (ifSiblings.size() > 0) {
200                        return ifSiblings.get(ifSiblings.size() - 1);
201                }
202                return null;
203        }
204
205        /**
206         * @return <code>true</code> if the given entity is the first statement of its
207         *         parent.
208         */
209        private static boolean isFirstChildStatement(ShallowEntity entity) {
210                List<ShallowEntity> children = entity.getParent().getChildrenOfType(EShallowEntityType.STATEMENT);
211                if (children.size() < 1) {
212                        return false;
213                }
214                return entity.equals(children.get(0));
215        }
216
217        /**
218         * @return the statement before the given entity within the same parent.
219         *         <code>null</code> if no such entity exists.
220         */
221        private static ShallowEntity getPrecedingStatement(ShallowEntity entity) {
222                List<ShallowEntity> siblings = entity.getParent().getChildrenOfType(EShallowEntityType.STATEMENT);
223                int entityIndex = siblings.indexOf(entity);
224                if (entityIndex > 0) {
225                        return siblings.get(entityIndex - 1);
226                }
227                return null;
228        }
