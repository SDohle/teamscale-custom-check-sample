006package eu.cqse.check.abap;
007
008import java.util.ArrayList;
009import java.util.Collection;
010import java.util.List;
011import java.util.Set;
012
013import eu.cqse.check.base.EntityCheckBase;
014import eu.cqse.check.framework.core.CheckException;
015import eu.cqse.check.framework.scanner.ETokenType;
016import eu.cqse.check.framework.scanner.IToken;
017import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
018import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
019import eu.cqse.check.framework.util.abap.AbapCheckUtils;
020
021/**
022 * Base class for custom checks for checking if the object has an authority
023 * check before executing any than declarative / trival statements.
024 */
025public abstract class MissingAuthorityCheckInObjectBase extends EntityCheckBase {
026
027        /**
028         * Gets the option value for the alternative authority check statements.
029         */
030        protected abstract Set<String> getAlternativeAuthorityChecksOption();
031
032        /**
033         * Checks if the given entity is starting with an authority check.
034         */
035        protected boolean isStartingWithAuthorityCheck(ShallowEntity entity) throws CheckException {
036                ShallowEntity firstStatement = getFirstNonDeclarativeStatement(entity);
037
038                if (firstStatement == null) {
039                        return false;
040                }
041                if (isStandardAuthorityCheck(firstStatement)) {
042                        return true;
043                }
044
045                String statementText = AbapCheckUtils.getNormalizedStartTokensText(firstStatement, context);
046
047                return getAlternativeAuthorityChecksOption().stream().anyMatch(s -> statementText.matches("(?i)" + s));
048        }
049
050        /**
051         * Checks if the given statement performs a standard authority check.
052         * Sub-classes may overwrite this method for specific checks.
053         */
054        protected boolean isStandardAuthorityCheck(ShallowEntity firstStatement) {
055                return firstStatement.getSubtype().equals("authority-check");
056        }
057
058        /**
059         * Gets the first statement which is a descendant of the given statement and
060         * which is not just a declarative statement or other statement which should be
061         * ignored (e.g. empty statement, TRY).
062         *
063         * @return first statement found or <code>null</code> if no statement is found
064         */
065        private static ShallowEntity getFirstNonDeclarativeStatement(ShallowEntity entity) throws CheckException {
066                List<ShallowEntity> statements = select(entity,
067                                ".//STATEMENT[not(" + "subtype('data') or subtype('constants')  or subtype('statics') "
068                                                + "or subtype('tables') or subtype('nodes') or subtype('types') "
069                                                + "or subtype('field-symbols') or subtype('selection-screen') "
070                                                + "or subtype('controls') or subtype('infotypes') "
071                                                + "or subtype('empty statement') or subtype('try') )]");
072
073                statements = removeTrivialAssignmentsAtStart(statements);
074
075                if (statements.isEmpty()) {
076                        return null;
077                }
078                return statements.get(0);
079        }
080
081        /**
082         * Removes trivial assignments (of literals or identifiers) from the beginning
083         * of the given statements list. Since these are often trivial initializations,
084         * these can occur before an authority check.
085         *
086         * @return the given statements where leading trivial assignments where removed
087         */
088        private static List<ShallowEntity> removeTrivialAssignmentsAtStart(Collection<ShallowEntity> statements) {
089                List<ShallowEntity> result = new ArrayList<>(statements);
090
091                for (ShallowEntity statement : statements) {
092                        if (isTrivialAssignment(statement.ownStartTokens())) {
093                                result.remove(statement);
094                        } else {
095                                return result;
096                        }
097                }
098                return result;
099        }
100
101        /**
102         * Checks if the given token sequence is a trivial assignment statement, e.g. of
103         * the from x = y. / x = 'abc'.
104         *
105         * @return <code>true</code> if the token sequence matches to [IDENTIFIER, EQ,
106         *         any, DOT]
107         */
108        private static boolean isTrivialAssignment(List<IToken> tokens) {
109                return (tokens.size() == 4 && TokenStreamUtils.startsWith(tokens, ETokenType.IDENTIFIER, ETokenType.EQ));
110        }
111
112}
