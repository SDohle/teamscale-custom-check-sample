006package eu.cqse.check.abap;
007
008import java.util.Collections;
009import java.util.List;
010import java.util.Set;
011
012import eu.cqse.check.framework.core.ACheck;
013import eu.cqse.check.framework.core.CheckException;
014import eu.cqse.check.framework.core.CheckGroupNames;
015import eu.cqse.check.framework.core.ECheckParameter;
016import eu.cqse.check.framework.core.EFindingEnablement;
017import eu.cqse.check.framework.core.option.ACheckOption;
018import eu.cqse.check.framework.core.phase.ECodeViewOption;
019import eu.cqse.check.framework.scanner.ELanguage;
020import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
021
022/**
023 * Checks for missing authority checks in reports (executable programs),
024 * introduced with keyword REPORT or PROGRAM.
025 */
026@ACheck(name = MissingAuthorityCheckInReportCheck.CHECK_NAME, description = "Executable programs (reports) ---"
027                + "introduced with statement `REPORT` or `PROGRAM`--- should contain an explicit `AUTHORITY-CHECK` before "
028                + "executing any statement. This is especially relevant since programs may be executed not only via transaction codes but "
029                + "by other programs using the SUBMIT statement. In these cases no implicit authority check is performed by the system.\n\n"
030                + "The `AUTHORITY-CHECK` is expected to be placed at the beginning of the `INITIALIZATION` or "
031                + "`START-OF-SELECTION` event block, or before the first executable statement if none of these blocks "
032                + "is explicitly specified.\n\n"
033                + "Custom authority check procedures which encapsulate the `AUTHORITY-CHECK` statement can be specified "
034                + "in the analysis profile as alternative authority check statements to avoid false positives.", groupName = CheckGroupNames.SECURITY, languages = ELanguage.ABAP, parameters = {
035                                ECheckParameter.ABSTRACT_SYNTAX_TREE }, defaultEnablement = EFindingEnablement.RED)
036public class MissingAuthorityCheckInReportCheck extends MissingAuthorityCheckInObjectBase {
037
038        /** Name of check */
039        /* package */ static final String CHECK_NAME = "Missing AUTHORITY-CHECK in reports/programs";
040
041        /** Findings Message */
042        private static final String FINDINGS_MESSAGE = "Missing authority check at start of program";
043
044        /** {@ConQAT.Doc} */
045        @ACheckOption(name = CHECK_NAME
046                        + " - Alternative authority check statements", description = "Comma separated list (regular expressions) for "
047                                        + "alternative authority check statements, e.g. call to custom methods which encapsulate AUTHORITY-CHECK.")
048        private Set<String> alternativeAuthorityChecks = Collections.emptySet();
049
050        /** {@inheritDoc} */
051        @Override
052        protected String getXPathSelectionString() {
053                return "//TYPE[subtype('report') or subtype('program')]";
054        }
055
056        /** {@inheritDoc} */
057        @Override
058        protected void processEntity(ShallowEntity entity) throws CheckException {
059                if (checkInitializationBlocks() || checkStartOfSelectionBlocks()) {
060                        return;
061                }
062                createFinding(FINDINGS_MESSAGE, entity.getStartLine());
063        }
064
065        /** {@inheritDoc} */
066        @Override
067        protected ECodeViewOption getCodeViewOption() {
068                return ECodeViewOption.FILTERED_PREPROCESSED;
069        }
070
071        /**
072         * Checks if INITIALIZATION blocks exist and if the first one starts with an
073         * authority check.
074         *
075         * @return <code>true</code> if the check is completed, e.g. authority check is
076         *         stated during initialization, <code>false</code> if check should be
077         *         continued with checking other sections
078         *
079         */
080        private boolean checkInitializationBlocks() throws CheckException {
081                List<ShallowEntity> initializationBlocks = select("//METHOD[subtype('initialization')]");
082
083                return (!initializationBlocks.isEmpty() && isStartingWithAuthorityCheck(initializationBlocks.get(0)));
084        }
085
086        /**
087         * Checks if START-OF-SELECTION blocks exist and if the first one starts with an
088         * authority check.
089         *
090         * @return <code>true</code> if the check is completed, e.g. an explicit
091         *         start-of-selection blocks exist, then the authority check is expected
092         *         to be there. <code>false</code> if check should be continued with
093         *         checking other sections
094         */
095        private boolean checkStartOfSelectionBlocks() throws CheckException {
096                List<ShallowEntity> startOfSelectionBlocks = select("//METHOD[subtype('start-of-selection')]");
097                if (startOfSelectionBlocks.isEmpty()) {
098                        return false;
099                }
100                ShallowEntity startOfSelection = startOfSelectionBlocks.get(0);
101                if (!isStartingWithAuthorityCheck(startOfSelection)) {
102                        createFinding(FINDINGS_MESSAGE, startOfSelection.getStartLine());
103                }
104                return true;
105        }
106
107        /** {@inheritDoc} */
108        @Override
109        protected Set<String> getAlternativeAuthorityChecksOption() {
110                return alternativeAuthorityChecks;
111        }
112
113}
