006package eu.cqse.check.abap;
007
008import java.util.ArrayList;
009import java.util.Collections;
010import java.util.List;
011import java.util.Set;
012
013import eu.cqse.check.base.EntityCheckBase;
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckGroupNames;
017import eu.cqse.check.framework.core.ECheckParameter;
018import eu.cqse.check.framework.core.EFindingEnablement;
019import eu.cqse.check.framework.core.option.ACheckOption;
020import eu.cqse.check.framework.scanner.ELanguage;
021import eu.cqse.check.framework.scanner.ETokenType;
022import eu.cqse.check.framework.scanner.IToken;
023import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
024import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
025import eu.cqse.check.framework.util.abap.AbapCheckUtils;
026
027/**
028 * Checks for missing authority checks in reports (executable programs),
029 * introduced with keyword REPORT or PROGRAM.
030 */
031@ACheck(name = MissingAuthorityCheckInReportCheck.CHECK_NAME, description = "<p>Executable programs (reports)&mdash;"
032                + "introduced with statement REPORT or PROGRAM&mdash;should contain an explicit AUTHORITY-CHECK before "
033                + "executing any statement. This is especially relevant since programs may be executed not only via transaction codes but "
034                + "by other programs using the SUBMIT statement. In these cases no implicit authority check is performed by the system.</p>"
035                + "<p>The AUTHORITY-CHECK is expected to be placed at the beginning of the INITIALIZATION or "
036                + "START-OF-SELECTION event block, or before the first executable statement if none of these blocks "
037                + "is explicitly specified.</p>"
038                + "<p>Custom authority check procedures which encapsulate the AUTHORITY-CHECK statement can be specified "
039                + "in the analysis profile as alternative authority check statements to avoid false positives.</p>", groupName = CheckGroupNames.SECURITY, languages = ELanguage.ABAP, parameters = {
040                                ECheckParameter.ABSTRACT_SYNTAX_TREE }, defaultEnablement = EFindingEnablement.RED)
041public class MissingAuthorityCheckInReportCheck extends EntityCheckBase {
042
043        /** Name of check */
044        /* package */ static final String CHECK_NAME = "Missing AUTHORITY-CHECK in reports/programs";
045
046        /** Findings Message */
047        private static final String FINDIGNS_MESSAGE = "Missing authority check at start of program";
048
049        /** {@ConQAT.Doc} */
050        @ACheckOption(name = CHECK_NAME
051                        + " - Alternative authority check statemetns", description = "Comma separated list (regular expressions) for "
052                                        + "alternative authority check statemetns, e.g. call to custom methods which encapsulate AUTHORITY-CHECK.")
053        private Set<String> alternativeAuthorityChecks = Collections.emptySet();
054
055        /** {@inheritDoc} */
056        @Override
057        protected String getXPathSelectionString() {
058                return "//METHOD[subtype('report') or subtype('program')]";
059        }
060
061        /** {@inheritDoc} */
062        @Override
063        protected void processEntity(ShallowEntity entity) throws CheckException {
064                if (checkInitializationBlocks()) {
065                        return;
066                }
067
068                if (checkImplicitStartOfSelection(entity)) {
069                        return;
070                }
071
072                if (checkStartOfSelectionBlocks()) {
073                        return;
074                }
075
076                createFinding(FINDIGNS_MESSAGE, entity.getStartLine());
077
078        }
079
080        /**
081         * Checks if INITIALIZATION blocks exist and if the first one starts with an
082         * authority check.
083         *
084         * @return <code>true</code> if the check is completed, e.g. authority check
085         *         is stated during initialization, <code>false</code> if check
086         *         should be continued with checking other sections
087         *
088         */
089        private boolean checkInitializationBlocks() throws CheckException {
090                List<ShallowEntity> initializationBlocks = select("//METHOD[subtype('initialization')]");
091                if (!initializationBlocks.isEmpty() && isStartingWithAuthorityCheck(initializationBlocks.get(0))) {
092                        return true;
093                }
094                return false;
095        }
096
097        /**
098         * Checks implicit start of selection blocks, thus statements which are
099         * directly stated after REPORT / PROGRAM without explicitly stated
100         * START-OF-SELECTION section.
101         *
102         * @return <code>true</code> if the check is completed, e.g. an implicit
103         *         start-of-selection blocks exist, then the authority check is
104         *         expected to be there. <code>false</code> if check should be
105         *         continued with checking other sections
106         */
107        private boolean checkImplicitStartOfSelection(ShallowEntity entity) throws CheckException {
108
109                ShallowEntity firstStatement = getFirstNonDeclarativeStatement(entity);
110
111                if (firstStatement == null) {
112                        return false;
113                }
114
115                if (!isStartingWithAuthorityCheck(entity)) {
116                        createFinding(FINDIGNS_MESSAGE, firstStatement.getStartLine());
117                }
118                return true;
119        }
120
121        /**
122         * Checks if START-OF-SELECTION blocks exist and if the first one starts
123         * with an authority check.
124         *
125         * @return <code>true</code> if the check is completed, e.g. an explicit
126         *         start-of-selection blocks exist, then the authority check is
127         *         expected to be there. <code>false</code> if check should be
128         *         continued with checking other sections
129         */
130        private boolean checkStartOfSelectionBlocks() throws CheckException {
131                List<ShallowEntity> startOfSelectionBlocks = select("//METHOD[subtype('start-of-selection')]");
132                if (startOfSelectionBlocks.isEmpty()) {
133                        return false;
134                }
135                ShallowEntity startOfSelection = startOfSelectionBlocks.get(0);
136                if (!isStartingWithAuthorityCheck(startOfSelection)) {
137                        createFinding(FINDIGNS_MESSAGE, startOfSelection.getStartLine());
138                }
139                return true;
140        }
141
142        /**
143         * Checks if the given entity is starting with an authority check.
144         */
145        private boolean isStartingWithAuthorityCheck(ShallowEntity entity) throws CheckException {
146                ShallowEntity firstStatement = getFirstNonDeclarativeStatement(entity);
147
148                if (firstStatement == null) {
149                        return false;
150                }
151                if (firstStatement.getSubtype().equals("authority-check")) {
152                        return true;
153                }
154
155                String statementText = AbapCheckUtils.getNormalizedStartTokensText(firstStatement, context);
156
157                return alternativeAuthorityChecks.stream().anyMatch(s -> statementText.matches("(?i)" + s));
158        }
159
160        /**
161         * Gets the first statement which is a descendant of the given statement.
162         *
163         * @return first statement found or <code>null</code> if no statement is
164         *         found
165         */
166        private ShallowEntity getFirstNonDeclarativeStatement(ShallowEntity entity) throws CheckException {
167                List<ShallowEntity> statemetns = select(entity,
168                                ".//STATEMENT[not(" + "subtype('data') or subtype('constants')  or subtype('statics') "
169                                                + "or subtype('tables') or subtype('nodes') or subtype('types') "
170                                                + "or subtype('field-symbols') or subtype('selection-screen') "
171                                                + "or subtype('controls') or subtype('infotypes') " + "or subtype('empty statement'))]");
172
173                statemetns = removeTrivialAssignmentsAtStart(statemetns);
174
175                if (statemetns.isEmpty()) {
176                        return null;
177                }
178                return statemetns.get(0);
179        }
180
181        /**
182         * Removes trivial assignments (of literals or identifiers) from the
183         * beginning of the given statements list. Since these are often trivial
184         * initializations, these can occur before an authority check.
185         *
186         * @return the given statements where leading trivial assignments where
187         *         removed
188         */
189        private static List<ShallowEntity> removeTrivialAssignmentsAtStart(List<ShallowEntity> statemetns) {
190                List<ShallowEntity> result = new ArrayList<>(statemetns);
191
192                for (ShallowEntity statement : statemetns) {
193                        if (isTrivialAssignment(statement.ownStartTokens())) {
194                                result.remove(statement);
195                        } else {
196                                return result;
197                        }
198                }
199                return result;
200        }
201
202        /**
203         * Checks if the given token sequence is a trivial assignment statement,
204         * e.g. of the from x = y. / x = 'abc'.
205         *
206         * @return <code>true</code> if the token sequence matches to [IDENTIFIER,
207         *         EQ, any, DOT]
208         */
209        private static boolean isTrivialAssignment(List<IToken> tokens) {
210                if (tokens.size() != 4 || !TokenStreamUtils.startsWith(tokens, ETokenType.IDENTIFIER, ETokenType.EQ)) {
211                        return false;
212                }
213                return true;
214        }
