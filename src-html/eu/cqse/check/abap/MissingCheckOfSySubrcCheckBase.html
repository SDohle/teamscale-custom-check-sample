006package eu.cqse.check.abap;
007
008import java.util.Set;
009
010import org.conqat.lib.commons.collections.CollectionUtils;
011
012import eu.cqse.check.base.EntityCheckBase;
013import eu.cqse.check.framework.core.CheckException;
014import eu.cqse.check.framework.scanner.ETokenType;
015import eu.cqse.check.framework.scanner.IToken;
016import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
017import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
018import eu.cqse.check.framework.shallowparser.framework.ShallowEntityTraversalUtils;
019import eu.cqse.check.framework.util.abap.AbapLanguageFeatureParser;
020
021/**
022 * Base class for checks which search for missing checks on the value of
023 * sy-subrc return code which are expected to follow specific statements. These
024 * statements are defined in the implementation of sub-classes. Examples of such
025 * statements are AUTHORITY-CHECK, CALL FUNCTION, OPEN SQL statements ... see
026 * e.g. https://wiki.scn.sap.com/wiki/display/ABAP/Sy-Subrc for a list of
027 * statements which set sy-subrc.
028 */
029public abstract class MissingCheckOfSySubrcCheckBase extends EntityCheckBase {
030
031        /** Sub-types of statements for conditional control flow. */
032        private static final Set<String> CONDITIONAL_SUBTYPES = CollectionUtils.asHashSet("if", "check", "case");
033
034        /** {@inheritDoc} */
035        @Override
036        protected void processEntity(ShallowEntity entity) throws CheckException {
037                if (!isEntityToCheck(entity)) {
038                        return;
039                }
040                if (containsSySubrc(ShallowEntityTraversalUtils.getSubsequentEntity(entity))) {
041                        return;
042                }
043                createFinding(getFindingsMessage(entity), entity);
044        }
045
046        /**
047         * Gets the findings message for the given entity.
048         */
049        protected abstract String getFindingsMessage(ShallowEntity entity);
050
051        /**
052         * Checks if the given entity expects an authority check. May be overwritten by
053         * sub-classes if selection on sub-type of statement is not precise enough. By
054         * default, this method returns always <code>true</code>.
055         */
056        @SuppressWarnings("unused")
057        protected boolean isEntityToCheck(ShallowEntity entity) throws CheckException {
058                return true;
059        }
060
061        /**
062         * @return <code>true</code> if the given entity is an IF statement which uses
063         *         the SAP return code system variable sy-subrc in its condition.
064         */
065        /* package */ static boolean isConditionalOnSySubrc(ShallowEntity entity) {
066                return containsSySubrc(entity) && CONDITIONAL_SUBTYPES.contains(entity.getSubtype());
067        }
068
069        /**
070         * @return <code>true</code> if the given entity is a statement operating on the
071         *         sy-subrc variable, and <code>false</code> if it is any other entity,
072         *         or <code>null</code>.
073         */
074        private static boolean containsSySubrc(ShallowEntity entity) {
075                return entity != null && entity.getType() == EShallowEntityType.STATEMENT
076                                && CollectionUtils.anyMatch(entity.ownStartTokens(), MissingCheckOfSySubrcCheckBase::isSySubrcToken);
077        }
078
079        /**
080         * @return Whether the given token is an identifier for <code>sy-subrc</code>.
081         */
082        private static boolean isSySubrcToken(IToken token) {
083                return token.getType() == ETokenType.IDENTIFIER
084                                && "sy-subrc".equals(AbapLanguageFeatureParser.normalizeVariable(token.getText()));
085        }
086}
