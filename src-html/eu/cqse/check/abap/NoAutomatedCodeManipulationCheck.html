008package eu.cqse.check.abap;
009
010import java.util.List;
011
012import eu.cqse.check.base.EntityCheckBase;
013import eu.cqse.check.framework.core.ACheck;
014import eu.cqse.check.framework.core.CheckException;
015import eu.cqse.check.framework.core.CheckGroupNames;
016import eu.cqse.check.framework.core.ECheckParameter;
017import eu.cqse.check.framework.scanner.ELanguage;
018import eu.cqse.check.framework.scanner.ETokenType;
019import eu.cqse.check.framework.scanner.IToken;
020import eu.cqse.check.framework.shallowparser.SubTypeNames;
021import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
022import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
023
024/**
025 * Checks whether code is manipulated automatically. <br>
026 *
027 * The following criteria are checked:
028 * <ul>
029 * <li>Manipulation of source code via ABAP statements (INSERT REPORT, GENERATE
030 * SUBROUTINE POOL, ...)
031 * </ul>
032 *
033 * The following criteria are Not checked:
034 * <ul>
035 * <li>Manipulation of source code via write access to code tables (not sure if
036 * this is possible and if it should be checked)
037 * </ul>
038 *
039 * Information Sources:
040 * <ul>
041 * <li>SAP Help at
042 * <a href="http://help.sap.com/abapdocu_750/de/abenabap_generic_program.htm">
043 * http://help.sap.com/abapdocu_750/de/abenabap_generic_program.htm</a>
044 * <li>"Top 20 Sicherheitsrisiken in ABAP Anwendungen" from BSI/ VirtualForge
045 * </ul>
046 */
047@ACheck(name = "No automated code manipulation", description = "Programs should not create, delete, or change coding.", groupName = CheckGroupNames.BAD_PRACTICE, languages = ELanguage.ABAP, parameters = {
048                ECheckParameter.ABSTRACT_SYNTAX_TREE })
049public class NoAutomatedCodeManipulationCheck extends EntityCheckBase {
050
051        /** Format string used as findings message when creating findings. **/
052        private static final String FINDING_MESSAGE = "Found automated manipulation of ABAP code by statement '%s'";
053
054        /** {@inheritDoc} */
055        @Override
056        protected String getXPathSelectionString() {
057                return "//STATEMENT[subtype('" + SubTypeNames.INSERT + "')" + " or subtype('" + SubTypeNames.GENERATE + "')"
058                                + " or subtype('" + SubTypeNames.DELETE + "')]";
059        }
060
061        /** {@inheritDoc} */
062        @Override
063        protected void processEntity(ShallowEntity entity) throws CheckException {
064                List<IToken> tokens = entity.ownStartTokens();
065
066                if (TokenStreamUtils.startsWith(tokens, ETokenType.INSERT, ETokenType.REPORT)) {
067                        createFinding(String.format(FINDING_MESSAGE, "INSERT REPORT"), entity);
068                }
069                if (TokenStreamUtils.startsWith(tokens, ETokenType.GENERATE, ETokenType.SUBROUTINE, ETokenType.POOL)) {
070                        createFinding(String.format(FINDING_MESSAGE, "GENERATE SUBROUTINE POOL"), entity);
071                }
072                if (TokenStreamUtils.startsWith(tokens, ETokenType.GENERATE, ETokenType.REPORT)) {
073                        createFinding(String.format(FINDING_MESSAGE, "GENERATE REPORT"), entity);
074                }
075                if (TokenStreamUtils.startsWith(tokens, ETokenType.DELETE, ETokenType.REPORT)) {
076                        createFinding(String.format(FINDING_MESSAGE, "DELETE REPORT"), entity);
077                }
078        }
079
080}
