008package eu.cqse.check.abap;
009
010import java.util.List;
011
012import eu.cqse.check.base.EntityCheckBase;
013import eu.cqse.check.framework.core.ACheck;
014import eu.cqse.check.framework.core.CheckException;
015import eu.cqse.check.framework.core.CheckGroupNames;
016import eu.cqse.check.framework.core.ECheckParameter;
017import eu.cqse.check.framework.scanner.ELanguage;
018import eu.cqse.check.framework.scanner.ETokenType;
019import eu.cqse.check.framework.scanner.IToken;
020import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
021
022/**
023 * Checks whether a breakpoint without breakpoint ID exists. <br>
024 * Assumption: Breakpoint is encoded with statements BREAK or BREAK-POINT.
025 */
026@ACheck(name = "No static breakpoints in code", description = "Coded breakpoints should not be used.", groupName = CheckGroupNames.BAD_PRACTICE, languages = ELanguage.ABAP, parameters = {
027                ECheckParameter.ABSTRACT_SYNTAX_TREE })
028public class NoBreakpointCheck extends EntityCheckBase {
029
030        /** {@inheritDoc} */
031        @Override
032        protected String getXPathSelectionString() {
033                return "//STATEMENT[subtype('break-point') or subtype('break')]";
034        }
035
036        /**
037         * {@inheritDoc}
038         *
039         * Checks whether break point has an ID (then breakpoint is allowed), if not
040         * creates a finding.
041         */
042        @Override
043        protected void processEntity(ShallowEntity entity) throws CheckException {
044                List<IToken> tokens = entity.ownStartTokens();
045                if (tokens.size() < 2) {
046                        return;
047                }
048                IToken potentialIdToken = tokens.get(1);
049                if (!potentialIdToken.getType().equals(ETokenType.ID)) {
050                        createFinding("Coded breakpoints should not be used", entity);
051                }
052        }
053}
