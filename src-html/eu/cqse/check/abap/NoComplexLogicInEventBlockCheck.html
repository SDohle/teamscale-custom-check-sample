008package eu.cqse.check.abap;
009
010import java.util.List;
011
012import org.conqat.lib.commons.collections.CollectionUtils;
013
014import eu.cqse.check.base.EntityCheckBase;
015import eu.cqse.check.framework.core.ACheck;
016import eu.cqse.check.framework.core.CheckException;
017import eu.cqse.check.framework.core.CheckGroupNames;
018import eu.cqse.check.framework.core.ECheckParameter;
019import eu.cqse.check.framework.core.option.ACheckOption;
020import eu.cqse.check.framework.scanner.ELanguage;
021import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
022import eu.cqse.check.framework.util.abap.AbapCheckUtils;
023
024/**
025 * Checks whether complex logic is contained in an event block. "Complex logic"
026 * is approximated by the length of the event block (in lines).
027 */
028@ACheck(name = "No complex logic in event block", description = "Complicated statement blocks should not be implemented directly in event blocks but in appropriate modularization units.", groupName = CheckGroupNames.BAD_PRACTICE, languages = ELanguage.ABAP, parameters = {
029                ECheckParameter.ABSTRACT_SYNTAX_TREE })
030public class NoComplexLogicInEventBlockCheck extends EntityCheckBase {
031
032        /** {@ConQAT.Doc} */
033        @ACheckOption(name = "Maximum lines in event block", description = "Maximum length of an event block body in number of lines (approximates event block complexity; default: 25).")
034        public int maxEventBlockLines = 25;
035
036        /** {@inheritDoc} */
037        @Override
038        protected String getXPathSelectionString() {
039                return "//METHOD";
040        }
041
042        /** {@inheritDoc} */
043        @Override
044        protected List<ShallowEntity> filterEntities(List<ShallowEntity> entities) {
045                return CollectionUtils.filter(entities, e -> AbapCheckUtils.EVENT_TYPES.contains(e.getSubtype()));
046        }
047
048        /** {@inheritDoc} */
049        @Override
050        protected void processEntity(ShallowEntity entity) throws CheckException {
051                int length = entity.getEndLine() - entity.getStartLine();
052                if (length > maxEventBlockLines) {
053                        createFinding("Event block is longer than " + maxEventBlockLines + " lines.", entity);
054                }
055        }
056}
