030/**
031 * Checks whether a macro contains database operations.<br>
032 * <br>
033 *
034 * Implemented checks:
035 * <ul>
036 * <li>Open SQL database operations
037 * <li>Native SQL database operations
038 * <li>Database operations via Object Services (the custom check for Object
039 * Services does not check all possible usages, but the occurrence of
040 * "characteristic" methods).
041 * </ul>
042 *
043 * NOT checked database operations:
044 * <ul>
045 * <li>Database operations via SQLScript
046 * <li>Database operations via ABAP Managed Database Procedures (AMDP)
047 * </ul>
048 *
049 * Source: http://help.sap.com/abapdocu_750/de/index.htm (Accessed 08.12.15)
050 */
051@ACheck(name = "No database operation in macro", description = "A macro should not contain database operations.", groupName = CheckGroupNames.BAD_PRACTICE, languages = ELanguage.ABAP, parameters = {
052                ECheckParameter.ABSTRACT_SYNTAX_TREE, ECheckParameter.TYPE_RESOLUTION })
053public class NoDbOperationInMacroCheck extends EntityCheckBase {
054
055        /** General finding message used in all findings. **/
056        private static final String GENERAL_MESSAGE = "A macro should not contain database operations";
057
058        /**
059         * Format string for finding messages of object service findings (i.e.
060         * detected database operations via Object Services).
061         */
062        private static final String FINDING_MESSAGE_OBJECT_SERVICES = GENERAL_MESSAGE
063                        + " but found string '%s' indicating a database operation using Object Services.";
064
065        /** Format string for finding messages of Native SQL Statements. */
066        private static final String FINDING_MESSAGE_NATIVE_SQL_STATEMENT = GENERAL_MESSAGE
067                        + " but found Native SQL statement '%s'.";
068
069        /** Format string for finding messages of Native SQL Classes. */
070        private static final String FINDING_MESSAGE_NATIVE_SQL_CLASS = GENERAL_MESSAGE
071                        + " but found Native SQL class '%s'.";
072
073        /** Format string for finding messages of OpenSQL Statements. */
074        private static final String FINDING_MESSAGE_OPEN_SQL_STATEMENT = GENERAL_MESSAGE
075                        + " but found Open SQL statement '%s'.";
076
077        /**
078         * Regex 1 used to detect database operations via Persistence service.
079         * Matches methods CREATE_PERSISTENT, GET_PERSISTENT*, DELETE_PERSISTENT.
080         */
081        private static final String REGEX_PERSISTENCE_SERVICE_1 = "(create|get|delete)_persistent";
082
083        /**
084         * Regex 2 used to detect database operations via Persistence service.
085         * Matches methods CREATE_TRANSIENT and GET_TRANSIENT.
086         */
087        private static final String REGEX_PERSISTENCE_SERVICE_2 = "(create|get)_transient";
088
089        /**
090         * Regex 1 used to detect database operations via Transaction service.
091         * Matches method GET_TRANSACTION_MANAGER.
092         */
093        private static final String REGEX_TRANSACTION_SERVICE_1 = "get_transaction_manager";
094
095        /**
096         * Regex 2 used to detect database operations via Transaction service.
097         * Matches methods TRANSACTION_MANAGER~CREATE_TRANSACTION,
098         * TRANSACTION_MANAGER~GET_CURRENT_TRANSACTION,
099         * TRANSACTION_MANAGER~GET_TOP_TRANSACTION.
100         */
101        private static final String REGEX_TRANSACTION_SERVICE_2 = "(create|get_current|get_top)_transaction";
102
103        /**
104         * Regex 1 used to detect database operations via Query service. Matches
105         * method GET_QUERY_MANAGER.
106         */
107        private static final String REGEX_QUERY_SERVICE_1 = "get_query_manager";
108
109        /**
110         * Regex 2 used to detect database operations via Query Service. Matches
111         * method CREATE_QUERY.
112         */
113        private static final String REGEX_QUERY_SERVICE_2 = "create_query";
114
115        /**
116         * Regex 3 used to detect database operations via Query Service. Matches
117         * method GET_EXPR_FACTORY.
118         */
119        private static final String REGEX_QUERY_SERVICE_3 = "get_expr_factory";
120
121        /**
122         * Regex 4 used to detect database operations via Query Service. Matches
123         * methods SET_FILTER_EXPR, SET_PARAMETERS_EXPR, SET_ORDERING_EXPR.
124         */
125        private static final String REGEX_QUERY_SERVICE_4 = "set_(filter|parameters|ordering)_expr";
126
127        /**
128         * Regex 5 used to detect database operations via Query Service. Matches
129         * methods CREATE_OPERATOR_EXPR, CREATE_LIKE_EXPR, CREATE_ISNULL_EXPR,
130         * CREATE_REF_EXPR, CREATE_AND_EXPR, CREATE_NOT_EXPR, CREATE_OR_EXPR,
131         * CREATE_PARAMETERS_EXPR, CREATE_ORDERING_EXPR.
132         */
133        private static final String REGEX_QUERY_SERVICE_5 = "create_(operator|like|isnull|ref|and|not|or|parameters|ordering)_expr";
134
135        /**
136         * Pattern for detection of database operations via Object Services.
137         */
138        private static final Pattern PATTERN_OBJECT_SERVICE_USE = Pattern.compile(
139                        REGEX_PERSISTENCE_SERVICE_1 + "|" + REGEX_PERSISTENCE_SERVICE_2 + "|" + REGEX_QUERY_SERVICE_1 + "|"
140                                        + REGEX_QUERY_SERVICE_2 + "|" + REGEX_QUERY_SERVICE_3 + "|" + REGEX_QUERY_SERVICE_4 + "|"
141                                        + REGEX_QUERY_SERVICE_5 + "|" + REGEX_TRANSACTION_SERVICE_1 + "|" + REGEX_TRANSACTION_SERVICE_2,
142                        Pattern.CASE_INSENSITIVE);
143
144        /**
145         * Regex used to detect database operations via Native SQL. Matches classes
146         * CL_SQL_*, CX_SQL_*.
147         */
148        private static final Pattern PATTERN_NATIVE_SQL_CLASSES = Pattern.compile("^C[LX]_SQL_.*",
149                        Pattern.CASE_INSENSITIVE);
150
151        /** {@inheritDoc} */
152        @Override
153        protected String getXPathSelectionString() {
154                return "//META[subtype('macro')]";
155        }
156
157        /** {@inheritDoc} */
158        @Override
159        protected void processEntity(ShallowEntity entity) throws CheckException {
160                if (AbapCheckUtils.isIgnoredForDatabaseChecks(entity, context.getAbstractSyntaxTree(),
161                                context.getUniformPath())) {
162                        return;
163                }
164                CCSMAssert.isFalse(entity.hasChildren(), "Assumption 'Macro entity has no children' is violated.");
165                List<IToken> tokens = entity.ownStartTokens();
166                List<List<IToken>> tokenLists = TokenStreamUtils.split(tokens, ETokenType.DOT);
167
168                checkForOpenSqlOperations(tokenLists, entity);
169                checkForNativeSQLOperation(tokenLists);
170                checkForObjectServiceOperation(tokenLists);
171        }
172
173        /**
174         * Check whether a macro contains a database operation via Object Services.
175         *
176         * @param macroStatements
177         *            A list of macro statements where each statement is represented
178         *            as token list.
179         * @throws CheckException
180         *             Raised in case of exception during finding creation.
181         */
182        private void checkForObjectServiceOperation(List<List<IToken>> macroStatements) throws CheckException {
183                for (List<IToken> currentTokenList : macroStatements) {
184                        createFindingsForIdentifiersContainingPattern(currentTokenList, PATTERN_OBJECT_SERVICE_USE,
185                                        FINDING_MESSAGE_OBJECT_SERVICES);
186                }
187        }
188
189        /**
190         * Check whether a macro contains a Native SQL database operation.
191         *
192         * @param macroStatements
193         *            A list of macro statements where each statement is represented
194         *            as token list.
195         */
196        private void checkForNativeSQLOperation(List<List<IToken>> macroStatements) throws CheckException {
197                for (List<IToken> currentTokenList : macroStatements) {
198                        findTokenTypeSequenceInTokenListAndCreateFinding(currentTokenList, FINDING_MESSAGE_NATIVE_SQL_STATEMENT,
199                                        ETokenType.EXEC, ETokenType.SQL);
200                        findTokenTypeSequenceInTokenListAndCreateFinding(currentTokenList, FINDING_MESSAGE_NATIVE_SQL_STATEMENT,
201                                        ETokenType.ENDEXEC);
202                        findTokenTypeSequenceInTokenListAndCreateFinding(currentTokenList, FINDING_MESSAGE_NATIVE_SQL_STATEMENT,
203                                        ETokenType.EXIT, ETokenType.FROM, ETokenType.SQL);
204
205                        // Use of classes CL_SQL_*, CX_SQL_*.
206                        createFindingsForIdentifiersContainingPattern(currentTokenList, PATTERN_NATIVE_SQL_CLASSES,
207                                        FINDING_MESSAGE_NATIVE_SQL_CLASS);
208                }
209        }
210
211        /**
212         * Attempts to find the token type sequence {@code tokenTypes} in token list
213         * {@code tokens} and creates a finding if found.
214         *
215         * @param tokens
216         *            A list of tokens.
217         * @param formatString
218         *            The format of the finding message with one placeholder for
219         *            textual representation of token list.
220         * @param tokenTypes
221         *            A sequence of token types.
222         * @throws CheckException
223         *             Raised in case of exception during finding creation.
224         */
225        private void findTokenTypeSequenceInTokenListAndCreateFinding(List<IToken> tokens, String formatString,
226                        ETokenType... tokenTypes) throws CheckException {
227                int position = TokenStreamUtils.firstTokenOfTypeSequence(tokens, 0, tokenTypes);
228                if (position != TokenStreamUtils.NOT_FOUND) {
229                        String tokenTypesAsString = Stream.of(tokenTypes).map(ETokenType::toString)
230                                        .collect(Collectors.joining(StringUtils.SPACE));
231                        String findingMessage = String.format(formatString, tokenTypesAsString);
232                        createFinding(findingMessage, tokens.get(position));
233                }
234        }
235
236        /**
237         * Check whether a macro contains an OpenSQL database operation.
238         *
239         * @param macroStatements
240         *            A list of macro statements where each statement is represented
241         *            as token list.
242         * @throws CheckException
243         *             Raised in case of exception during finding creation.
244         */
245        private void checkForOpenSqlOperations(List<List<IToken>> macroStatements, ShallowEntity macroEntity)
246                        throws CheckException {
247
248                for (List<IToken> currentTokenList : macroStatements) {
249                        if (currentTokenList == null || currentTokenList.isEmpty()) {
250                                continue;
251                        }
252                        if (TokenStreamUtils.containsAny(currentTokenList, 0, 1, ETokenType.INSERT, ETokenType.UPDATE,
253                                        ETokenType.MODIFY, ETokenType.DELETE)) {
254                                OpenSqlStatement statement = OpenSqlStatement.create(macroEntity, currentTokenList,
255                                                context.getTypeResolution());
256                                if (statement != null && !statement.targetsInternalTable()) {
257                                        createFinding(String.format(FINDING_MESSAGE_OPEN_SQL_STATEMENT,
258                                                        statement.getStatementStartToken().getText()), currentTokenList.get(0));
259                                }
260                        }
261                        findTokenTypeSequenceInTokenListAndCreateFinding(currentTokenList, FINDING_MESSAGE_OPEN_SQL_STATEMENT,
262                                        ETokenType.SELECT);
263                        findTokenTypeSequenceInTokenListAndCreateFinding(currentTokenList, FINDING_MESSAGE_OPEN_SQL_STATEMENT,
264                                        ETokenType.END, ETokenType.SELECT);
265                        findTokenTypeSequenceInTokenListAndCreateFinding(currentTokenList, FINDING_MESSAGE_OPEN_SQL_STATEMENT,
266                                        ETokenType.OPEN, ETokenType.CURSOR);
267                        findTokenTypeSequenceInTokenListAndCreateFinding(currentTokenList, FINDING_MESSAGE_OPEN_SQL_STATEMENT,
268                                        ETokenType.FETCH, ETokenType.NEXT, ETokenType.CURSOR);
269                        findTokenTypeSequenceInTokenListAndCreateFinding(currentTokenList, FINDING_MESSAGE_OPEN_SQL_STATEMENT,
270                                        ETokenType.CLOSE, ETokenType.CURSOR);
271                        findTokenTypeSequenceInTokenListAndCreateFinding(currentTokenList, FINDING_MESSAGE_OPEN_SQL_STATEMENT,
272                                        ETokenType.COMMIT, ETokenType.WORK);
273                        findTokenTypeSequenceInTokenListAndCreateFinding(currentTokenList, FINDING_MESSAGE_OPEN_SQL_STATEMENT,
274                                        ETokenType.ROLLBACK, ETokenType.WORK);
275                }
276        }
277
278        /**
279         * Attempts to find the {@code pattern} in each {@code IDENTIFIER} token
280         * text in token list and create a finding with message
281         * {@code messageTemplate} for each find.
282         *
283         * @param tokens
284         *            A list of tokens.
285         * @param pattern
286         *            The pattern representing a regex to be matched.
287         * @param messageTemplate
288         *            A format string used as message if a finding is created.
289         */
290        private void createFindingsForIdentifiersContainingPattern(List<IToken> tokens, Pattern pattern,
291                        String messageTemplate) throws CheckException {
292                for (IToken currentToken : tokens) {
293                        if (currentToken.getType().equals(ETokenType.IDENTIFIER)
294                                        && pattern.matcher(currentToken.getText()).find()) {
295                                createFinding(String.format(messageTemplate, currentToken.getText()), currentToken);
296                        }
297                }
298        }
