092
093        /** {@inheritDoc} */
094        @Override
095        protected String getXPathSelectionString() {
096                return "//STATEMENT[subtype('" + CALL + "')]";
097        }
098
099        /** {@inheritDoc} */
100        @Override
101        protected void processEntity(ShallowEntity entity) throws CheckException {
102
103                List<IToken> tokens = entity.ownStartTokens();
104                if (tokens.size() < 3 || !TokenStreamUtils.startsWith(tokens, ETokenType.CALL, ETokenType.FUNCTION)) {
105                        return;
106                }
107
108                String functionName = tokens.get(2).getText();
109
110                if (ALL_THFB_FUNCTIONS.contains(functionName.toUpperCase())) {
111                        createFinding("Found use of task handler function `" + functionName + "` which is forbidden.", entity);
112                }
113        }
