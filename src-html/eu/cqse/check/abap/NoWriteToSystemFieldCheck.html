008package eu.cqse.check.abap;
009
010import java.util.List;
011import java.util.regex.Pattern;
012
013import eu.cqse.check.base.EntityCheckBase;
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckGroupNames;
017import eu.cqse.check.framework.core.ECheckParameter;
018import eu.cqse.check.framework.scanner.ELanguage;
019import eu.cqse.check.framework.scanner.ETokenType;
020import eu.cqse.check.framework.scanner.IToken;
021import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
022import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
023
024/**
025 * Checks whether write accesses to system fields (sy-xxx or syst-xxx) exist.
026 * <br>
027 * <br>
028 * There are many ways to write to system fields. The following lists give an
029 * overview which of them are checked.<br>
030 * <br>
031 *
032 * The following criteria are checked:
033 * <ul>
034 * <li>Write access via assignment
035 * <li>write access via MOVE statement
036 * </ul>
037 *
038 * The following criteria are NOT checked:
039 * <ul>
040 * <li>Indirect write access by defining a variable of system field type, e.g.
041 * DATA: ok_code TYPE sy-ucomm (because this style is often used in cases where
042 * no write access to system field occurrs).
043 * <li>Using System Fields as Actual Parameters (system field is passed to
044 * method as call by reference) as described in <a href=
045 * "http://help.sap.com/abapdocu_750/en/abenuse_actual_parameters_guidl.htm">
046 * http://help.sap.com/abapdocu_750/en/abenuse_actual_parameters_guidl.htm</a>
047 * (Check logic would require data flow analysis and therefore become too
048 * complex).
049 * <li>Using Elements in Operand Positions (system field is passed as operand to
050 * statement which manipulates operand) as described in <a href=
051 * "http://help.sap.com/abapdocu_750/en/abenuse_operand_position_guidl.htm">http
052 * ://help.sap.com/abapdocu_750/en/abenuse_operand_position_guidl.htm</a> (Check
053 * logic would have to know syntax of each statement and therefore become too
054 * complex).
055 * </ul>
056 */
057@ACheck(name = "No write access to system fields", description = "System fields (Sy-xxxx, fields of structure SYST) should only be read (as they are set by the runtime environment).", groupName = CheckGroupNames.BAD_PRACTICE, languages = ELanguage.ABAP, parameters = {
058                ECheckParameter.ABSTRACT_SYNTAX_TREE })
059public class NoWriteToSystemFieldCheck extends EntityCheckBase {
060
061        /** Format string used as finding message. */
062        private static final String FINDINGS_TEXT = "Write to system fields are forbidden but write to system field '%s' found.";
063
064        /**
065         * Regex used to detect a system field by its name.
066         */
067        private static final Pattern PATTERN_SYSTEM_FIELD = Pattern.compile("!?(sy|syst)-.*", Pattern.CASE_INSENSITIVE);
068
069        /** {@inheritDoc} */
070        @Override
071        protected String getXPathSelectionString() {
072                return "//STATEMENT";
073        }
074
075        /** {@inheritDoc} */
076        @Override
077        protected void processEntity(ShallowEntity entity) throws CheckException {
078                List<IToken> tokens = entity.ownStartTokens();
079
080                if (TokenStreamUtils.containsAny(tokens, ETokenType.EQ)) {
081                        checkAssignmentStatement(entity);
082                } else if (TokenStreamUtils.containsAny(tokens, ETokenType.MOVE)) {
083                        checkMoveStatement(entity);
084                } else if (TokenStreamUtils.containsAny(tokens, ETokenType.CLEAR)) {
085                        checkClearStatement(entity);
086                } else if (TokenStreamUtils.containsAny(tokens, ETokenType.WRITE)) {
087                        checkWriteStatement(entity);
088                } else if (TokenStreamUtils.containsAny(tokens, ETokenType.INTO)) {
089                        checkStatementWithIntoToken(entity);
090                }
091
092        }
093
094        /**
095         * Checks whether a MOVE statement writes to a system field and creates a
096         * finding if it does.
097         *
098         * @param entity
099         *            An entity representing a MOVE statement.
100         * @throws CheckException
101         *             Signals an error during finding creation.
102         */
103        private void checkMoveStatement(ShallowEntity entity) throws CheckException {
104                // Example: MOVE identifier1 TO identifier2.
105                List<IToken> tokens = entity.ownStartTokens();
106                if (tokens.size() < 4) {
107                        return;
108                }
109                checkIdentifier(tokens.get(3));
110        }
111
112        /**
113         * Checks whether the passed IDENTIFIER token represents a system field and
114         * creates finding if it does.
115         *
116         * @param currentToken
117         *            A token representing an IDENTIFIER.
118         * @throws CheckException
119         *             Signals an error during finding creation.
120         */
121        private void checkIdentifier(IToken currentToken) throws CheckException {
122                String identifierText = currentToken.getText();
123                if (PATTERN_SYSTEM_FIELD.matcher(identifierText).matches()) {
124                        createFinding(String.format(FINDINGS_TEXT, identifierText), currentToken);
125                }
126        }
127
128        /**
129         * Checks whether the passed assignment writes into a system field and
130         * creates a finding if it does.
131         *
132         * @param entity
133         *            An entity representing a variable assignment.
134         * @throws CheckException
135         *             Signals an error during finding creation.
136         */
137        private void checkAssignmentStatement(ShallowEntity entity) throws CheckException {
138                List<IToken> tokens = entity.ownStartTokens();
139
140                // Exclude assertion or condition statements
141                // Necessary because '=' and 'EQ' are mapped to token type EQ
142                if (TokenStreamUtils.startsWith(tokens, ETokenType.IF, ETokenType.ELSEIF, ETokenType.CHECK, ETokenType.WHILE,
143                                ETokenType.ASSERT)) {
144                        return;
145                }
146
147                // Example: sy-uname = 'mister x'.
148                if (tokens.size() < 1) {
149                        return;
150                }
151                checkIdentifier(tokens.get(0));
152        }
153
154        /**
155         * Checks whether a CLEAR statement resets a system field and creates a
156         * finding if it does.
157         *
158         * @param entity
159         *            An entity representing a CLEAR statement.
160         * @throws CheckException
161         *             Signals an error during finding creation.
162         */
163        private void checkClearStatement(ShallowEntity entity) throws CheckException {
164                // Example: CLEAR sy-msgid1, sy-msgid2.
165                List<IToken> tokens = entity.ownStartTokens();
166                if (tokens.size() >= 2) {
167                        checkIdentifier(tokens.get(1));
168                }
169        }
170
171        /**
172         * Checks whether a WRITE statement writes to a system field and creates a
173         * finding if it does.
174         *
175         * @param entity
176         *            An entity representing a WRITE statement.
177         * @throws CheckException
178         *             Signals an error during finding creation.
179         */
180        private void checkWriteStatement(ShallowEntity entity) throws CheckException {
181                // Example: WRITE source TO sy-*.
182                List<IToken> tokens = entity.ownStartTokens();
183                int positionOfToToken = TokenStreamUtils.firstTokenOfType(tokens, ETokenType.TO); // Ignore
184                                                                                                                                                                // WRITE
185                                                                                                                                                                // statements
186                                                                                                                                                                // without
187                                                                                                                                                                // TO
188                if (positionOfToToken == TokenStreamUtils.NOT_FOUND || tokens.size() < positionOfToToken + 2) {
189                        return;
190                }
191                checkIdentifier(tokens.get(positionOfToToken + 1));
192        }
193
194        /**
195         * Checks whether a statement featuring a INTO token writes to a system
196         * field and creates a finding if it does.
197         *
198         * @param entity
199         *            An entity representing a statement with an INTO token.
200         * @throws CheckException
201         *             Signals an error during finding creation.
202         */
203        private void checkStatementWithIntoToken(ShallowEntity entity) throws CheckException {
204                // Example: CONCATENATE 'foo' 'bar' INTO sy-xxx.
205                List<IToken> tokens = entity.ownStartTokens();
206                int positionOfIntoToken = TokenStreamUtils.firstTokenOfType(tokens, ETokenType.INTO);
207                if (positionOfIntoToken == TokenStreamUtils.NOT_FOUND || tokens.size() < positionOfIntoToken + 2) {
208                        return;
209                }
210                checkIdentifier(tokens.get(positionOfIntoToken + 1));
211        }
212
213}
