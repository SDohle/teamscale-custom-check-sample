004import java.util.List;

007import eu.cqse.check.base.EntityCheckBase;

018import eu.cqse.check.framework.util.abap.AbapCheckUtils;

020/**
021 * Checks for usages of OPEN DATASET with FILTER addition - see
022 * <a href="http://help.sap.com/abapdocu_740/en/abapopen_dataset.htm">ABAP
023 * keyword documentation</a>. Allowed operating system commands can be specified
024 * in a whitelist.
025 */
026@ACheck(name = OpenDatasetWithFilterAdditonCheck.CHECK_NAME, description = "<p>When used with the addtion FILTER, OPEN DATASET allows "
027                + "to execute any operating system command. Thus, the FILTER addtion should be avoided in general. "
028                + "The <a href=\"https://help.sap.com/abapdocu_750/en/abenabap_system_commands_appl_sxpg.htm\">SXPG framework</a> "
029                + "should be used for OS calls instead.</p>", groupName = CheckGroupNames.SECURITY, languages = ELanguage.ABAP, parameters = {
030                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
031
032public class OpenDatasetWithFilterAdditonCheck extends EntityCheckBase {
033
034        /** Name of this check */
035        /* package */ static final String CHECK_NAME = "Suspicious usage of OPEN DATASET";
036
037        /** {@ConQAT.Doc} */
038        @ACheckOption(name = CHECK_NAME + " - Allowed OS commands", description = "Comma separted list of operating system"
039                        + "commands (regular expressions) which are allowed for FILTER addtion in OPEN DATASET.")
040        private Set<String> osCommandWhitelist = Collections.emptySet();
041
042        /** {@inheritDoc} */
043        @Override
044        protected String getXPathSelectionString() {
045                return "//STATEMENT[subtype('open')]";
046        }
047
048        /** {@inheritDoc} */
049        @Override
050        protected void processEntity(ShallowEntity entity) throws CheckException {
051
052                List<IToken> tokens = entity.ownStartTokens();
053
054                if (tokens.size() < 2 || tokens.get(1).getType() != ETokenType.DATASET) {
055                        return;
056                }
057
058                int filterTokenIndex = TokenStreamUtils.firstTokenOfType(tokens, ETokenType.FILTER);
059                if (filterTokenIndex == TokenStreamUtils.NOT_FOUND || tokens.size() <= filterTokenIndex + 1) {
060                        return;
061                }
062
063                String findingsMessage = "Usage of OPEN DATASET with FILTER addition";
064
065                String osCommand = AbapCheckUtils.getUnquotedTextForCharcterLiteral(tokens.get(filterTokenIndex + 1));
066
067                if (osCommand != null) {
068                        if (osCommandWhitelist.stream().anyMatch(s -> osCommand.matches(s))) {
069                                return;
070                        }
071                        findingsMessage = findingsMessage + " '" + osCommand + "'";
072                }
073
074                createFinding(findingsMessage, entity);
075        }
076
077}
