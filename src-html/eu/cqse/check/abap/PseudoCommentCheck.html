006package eu.cqse.check.abap;
007
008import java.util.regex.Matcher;
009import java.util.regex.Pattern;
010
011import eu.cqse.check.base.CommentCheckBase;
012import eu.cqse.check.framework.core.ACheck;
013import eu.cqse.check.framework.core.CheckException;
014import eu.cqse.check.framework.core.CheckGroupNames;
015import eu.cqse.check.framework.core.ECheckParameter;
016import eu.cqse.check.framework.core.EFindingEnablement;
017import eu.cqse.check.framework.core.option.ACheckOption;
018import eu.cqse.check.framework.core.util.CheckUtils;
019import eu.cqse.check.framework.scanner.ELanguage;
020import eu.cqse.check.framework.scanner.IToken;
021
022/**
023 * {@ConQAT.Doc}
024 */
025@ACheck(name = PseudoCommentCheck.CHECK_NAME, description = "Pseudo comments ``\"#EC ...`` to suppress messages from the "
026                + "Extended Program Check or from SAP Code Inspector should not be used. Instead, invalid findings may be "
027                + "blacklisted in Teamscale.", groupName = CheckGroupNames.FINDINGS_SUPPRESSION, defaultEnablement = EFindingEnablement.YELLOW, languages = {
028                                ELanguage.ABAP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
029public class PseudoCommentCheck extends CommentCheckBase {
030
031        /** {@ConQAT.Doc} */
032        @ACheckOption(name = CHECK_NAME
033                        + " - Pseudo comment selection", description = "Regular expression for selecting pseudo comments, for example 'CI_.*' to match all Code Inspector pseudo comments. Matches only against the pseudo comment name without '\"EC'. "
034                                        + "If it is specified, the check option 'Use pseudo comment selection as whitelist' specifies whether it is treated as a white list or a black list."
035                                        + "If it is not specified, all pseudo comments are reported as finding.")
036        private String selectedPseudoCommentsRegex = "";
037
038        /** {@ConQAT.Doc} */
039        @ACheckOption(name = CHECK_NAME
040                        + " - Use pseudo comment selection as whitelist", description = "If enabled, 'Pseudo comment selection' is treated as white list (all pseudo comments not matching 'Pseudo comment selection' are reported as finding). "
041                                        + "If not enabled, 'Pseudo comment selection' is treated as black list (all pseudo comments matching 'Pseudo comment selection' are reported as finding).")
042        private boolean isWhitelist = false;
043
044        /** Name of this check */
045        public static final String CHECK_NAME = "Pseudo comments should not be used";
046
047        /** Pattern for pseudo comments */
048        private static final Pattern PSEUDO_COMMENT_PATTERN = Pattern.compile("^\"#EC\\s+(\\S*)");
049
050        private Pattern selectedPseudoCommentsPattern;
051
052        /** {@inheritDoc} */
053        @Override
054        public void initialize() throws CheckException {
055                selectedPseudoCommentsPattern = CheckUtils.compilePatternFromCheckOptionRegex(selectedPseudoCommentsRegex,
056                                CHECK_NAME, "pseudo comment selection");
057        }
058
059        /** {@inheritDoc} */
060        @Override
061        protected void processComment(IToken token, int startLine, int endLine) throws CheckException {
062                String comment = token.getText();
063                Matcher commentMatcher = PSEUDO_COMMENT_PATTERN.matcher(comment);
064                if (commentMatcher.find() && commentMatcher.start() == 0 && !isAllowedPseudoComment(commentMatcher.group(1))) {
065                        createFinding("Pseudo comment ``" + comment + "`` should not be used", startLine);
066                }
067        }
068
069        /**
070         * Checks if the given comment is an allowed pseudo comment depending on
071         * {@link #selectedPseudoCommentsRegex} and {@link #isWhitelist}.
072         */
073        private boolean isAllowedPseudoComment(String comment) {
074                if (selectedPseudoCommentsPattern == null) {
075                        return false;
076                }
077                if (selectedPseudoCommentsPattern.matcher(comment).matches()) {
078                        return isWhitelist;
079                }
080                return !isWhitelist;
081        }
082}
