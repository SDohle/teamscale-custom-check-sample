006package eu.cqse.check.abap;
007
008import java.util.EnumSet;
009import java.util.List;
010import java.util.regex.Pattern;
011
012import org.conqat.lib.commons.string.StringUtils;
013
014import eu.cqse.check.framework.core.CheckException;
015import eu.cqse.check.framework.core.CheckImplementationBase;
016import eu.cqse.check.framework.core.phase.ECodeViewOption;
017import eu.cqse.check.framework.scanner.ETokenType;
018import eu.cqse.check.framework.scanner.IToken;
019
020/**
021 * Base class for check on system variables.
022 */
023public abstract class SystemVariablesCheckBase extends CheckImplementationBase {
024
025        /** Set of token types which refer to control statements. */
026        private static final EnumSet<ETokenType> CONTROL_STATEMENT_TOKENS = EnumSet.of(ETokenType.IF, ETokenType.ELSEIF,
027                        ETokenType.CASE, ETokenType.WHEN, ETokenType.WHILE);
028
029        /** Set of token types which refer to type declarations. */
030        private static final EnumSet<ETokenType> TYPE_DECLARATION_TOKENS = EnumSet.of(ETokenType.TYPE, ETokenType.LIKE);
031
032        /** Pattern for the token text which identifies a system variable */
033        protected Pattern tokenTextPattern = Pattern.compile("(?i)sy-[a-z]{2,5}");
034
035        /** {@inheritDoc} */
036        @Override
037        public void execute() throws CheckException {
038                buildTokenTextPattern();
039
040                List<IToken> tokens = context.getTokens(ECodeViewOption.FILTERED_PREPROCESSED);
041                for (int i = 0; i < tokens.size(); i++) {
042                        if (isSystemVariable(tokens, i)) {
043                                performSystemVariableCheck(tokens, i);
044                        }
045                }
046        }
047
048        /**
049         * Builds {@link #tokenTextPattern} form {@link #getFieldsToCheck()}, e.g.
050         * replacing comma or spaces for separation of list items with the vertical bar
051         * used in the regex and possible added prefixes of "sy-" are removed. If
052         * {@link #getFieldsToCheck()} does not contain any text, it is left to its
053         * initial value to match any occurrence of system fields.
054         */
055        protected void buildTokenTextPattern() {
056                String fieldsToCheck = getFieldsToCheck().trim();
057                if (!StringUtils.isEmpty(fieldsToCheck)) {
058                        String patternForFieldsToCheck = fieldsToCheck.replaceAll("(?i)sy-", StringUtils.EMPTY_STRING)
059                                        .replaceAll(",?\\s+", "|");
060
061                        tokenTextPattern = Pattern.compile("(?i)sy-(" + patternForFieldsToCheck + ")");
062                }
063        }
064
065        /**
066         * @return <code>true</code> if the given token is an identifier for a system
067         *         variable (sy-...) whose value is accessed in the statement.
068         */
069        private boolean isSystemVariable(List<IToken> tokens, int index) {
070                if (!hasSystemVariableName(tokens.get(index))) {
071                        return false;
072                }
073                if (index == 0) {
074                        return true;
075                }
076                // The system variable's value is not accessed when used in a type declaration.
077                // It's just a reference to declare a data type for another variable.
078                return !TYPE_DECLARATION_TOKENS.contains(tokens.get(index - 1).getType());
079        }
080
081        /**
082         * @return <code>true</code> if the given token is an identifier for a system
083         *         variable.
084         */
085        private boolean hasSystemVariableName(IToken token) {
086                if (!token.getType().isIdentifier()) {
087                        return false;
088                }
089                return tokenTextPattern.matcher(token.getText()).matches();
090        }
091
092        /**
093         * Checks if the given token is in the condition of a control statement.
094         */
095        protected boolean isInControlStatement(List<IToken> tokens, int index) {
096                for (int tokenIndex = index - 1; tokenIndex >= 0; tokenIndex--) {
097                        ETokenType tokenType = tokens.get(tokenIndex).getType();
098                        if (tokenType == ETokenType.DOT) {
099                                return false;
100                        }
101                        if (CONTROL_STATEMENT_TOKENS.contains(tokenType)) {
102                                return true;
103                        }
104                }
105                return false;
106        }
107
108        /**
109         * Gets the string of the options for the fields to check.
110         */
111        protected abstract String getFieldsToCheck();
112
113        /**
114         * Perform the check for the given token which is an identifier of a system
115         * variable.
116         */
117        protected abstract void performSystemVariableCheck(List<IToken> tokens, int index) throws CheckException;
118}
