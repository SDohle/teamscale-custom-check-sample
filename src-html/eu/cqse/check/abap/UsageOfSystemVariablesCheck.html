006package eu.cqse.check.abap;
007
008import java.util.List;
009
010import eu.cqse.check.framework.core.ACheck;
011import eu.cqse.check.framework.core.CheckException;
012import eu.cqse.check.framework.core.CheckGroupNames;
013import eu.cqse.check.framework.core.EFindingEnablement;
014import eu.cqse.check.framework.core.option.ACheckOption;
015import eu.cqse.check.framework.scanner.ELanguage;
016import eu.cqse.check.framework.scanner.IToken;
017
018/**
019 * Checks for usages of system variables, eg. SY-UNAME, SY-MANDT, ...
020 */
021@ACheck(name = UsageOfSystemVariablesCheck.CHECK_NAME, description = "A system variable, e.g., `SY-UNAME` "
022                + "(user name of current user) or `SY-SYSID` (SAP system id) is accessed. System variables which "
023                + "contain user, installation or client specific values should be avoided, since these may lead "
024                + "to different behavior depending on logon data or platform."
025                + "", groupName = CheckGroupNames.SECURITY, parameters = {}, languages = ELanguage.ABAP, defaultEnablement = EFindingEnablement.YELLOW)
026public class UsageOfSystemVariablesCheck extends SystemVariablesCheckBase {
027
028        /** Name of this check */
029        static final String CHECK_NAME = "Usage of system variables (SY-...)";
030
031        /** {@ConQAT.Doc} */
032        @ACheckOption(name = CHECK_NAME + " - relevant system fields", description = ""
033                        + "Fields of SY structure which should be searched.")
034        private String fieldsToCheck = "UNAME, SYSID";
035
036        /** {@ConQAT.Doc} */
037        @ACheckOption(name = CHECK_NAME + " - ignore control statements", description = ""
038                        + "If true, control statements are ignored (when checked with "
039                        + ControlFlowDependingOnSystemVariableCheck.CHECK_NAME + ").")
040        private boolean ignoreControlStatements = true;
041
042        /** {@inheritDoc} */
043        @Override
044        protected String getFieldsToCheck() {
045                return fieldsToCheck;
046        }
047
048        /**
049         * {@inheritDoc}
050         */
051        @Override
052        protected void performSystemVariableCheck(List<IToken> tokens, int index) throws CheckException {
053                if (ignoreControlStatements && isInControlStatement(tokens, index)) {
054                        return;
055                }
056                IToken token = tokens.get(index);
057                createFinding("Usage of system variable `" + token.getText().toUpperCase() + "`", token);
058        }
059
060}
