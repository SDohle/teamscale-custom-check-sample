008package eu.cqse.check.base;
009
010import java.util.List;
011
012import eu.cqse.check.framework.core.CheckException;
013import eu.cqse.check.framework.scanner.IToken;
014import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
015
016/**
017 * Base class for checks that select entities via xPath and process the tokens
018 * of each selected entity.
019 * 
020 * @author $Author: hummelb $
021 * @version $Rev: 52968 $
022 * @ConQAT.Rating GREEN Hash: F973E21B7008F6976B4AD44513FED2E5
023 */
024public abstract class EntityTokenCheckBase extends EntityCheckBase {
025
026        /** {@inheritDoc} */
027        @Override
028        protected void processEntity(ShallowEntity entity) throws CheckException {
029                processTokens(entity.ownStartTokens());
030                if (entity.hasChildren()) {
031                        processTokens(entity.ownEndTokens());
032                }
033        }
034
035        /** Process the start or end tokens for a selected entity. */
036        protected abstract void processTokens(List<IToken> tokens)
037                        throws CheckException;
038}
