001package eu.cqse.check.base;
002
003import java.util.regex.MatchResult;
004import java.util.regex.Matcher;
005import java.util.regex.Pattern;
006
007import eu.cqse.check.framework.core.CheckException;
008import eu.cqse.check.framework.scanner.IToken;
009
010/**
011 * Base class for checks which find unwanted expressions in any comment in a
012 * resource.
013 *
014 * @author $Author: kinnen $
015 * @version $Rev: 53936 $
016 * @ConQAT.Rating GREEN Hash: 64580A2E415DE0D0A9CB40A3F4DEB502
017 */
018public abstract class UnwantedExpressionInCommentCheckBase
019                extends CommentCheckBase {
020
021        /** Stores the unwanted expression regex. */
022        private Pattern unwantedRegex;
023
024        /** {@inheritDoc} */
025        @Override
026        public void initialize() {
027                super.initialize();
028                unwantedRegex = getUnwantedRegex();
029        }
030
031        /** {@inheritDoc} */
032        @Override
033        protected void processComment(IToken token, int startLine, int endLine)
034                        throws CheckException {
035                Matcher commentMatcher = unwantedRegex.matcher(token.getText());
036
037                // The pattern might appear anywhere in the token text. Therefore, we
038                // use find() instead of matches() to search for it.
039                if (commentMatcher.find()) {
040                        createFinding(getFindingText(token, commentMatcher.toMatchResult()),
041                                        startLine, endLine);
042                }
043        }
044
045        /**
046         * Provides the Regex pattern that specifies which expressions are unwanted
047         * in comments. Note that this method is only called once at the
048         * initialization of the check for performance reasons.
049         */
050        protected abstract Pattern getUnwantedRegex();
051
052        /**
053         * Gets the text for the finding.
054         *
055         * @param token
056         *            The token where the unwanted pattern was found.
057         * @param matchResult
058         *            The {@link MatchResult} which allows providing more details in
059         *            the findings text.
060         * @return The finding text.
061         */
062        protected abstract String getFindingText(IToken token,
063                        MatchResult matchResult);
064
065}
