055/**
056 * Base class for c-like avoid variable assignment to itself checks.
057 */
058@ACheck(description = AvoidVariableAssignmentToItselfCheck.DESCRIPTION, groupName = BAD_PRACTICE, languages = { JAVA,
059                CPP, CS, JAVASCRIPT, PYTHON, PHP, RUST, KOTLIN, SWIFT,
060                OBJECTIVE_C }, name = "Assignment of a variable to itself", parameters = ABSTRACT_SYNTAX_TREE)
061public class AvoidVariableAssignmentToItselfCheck extends EntityCheckBase {
062
063        /** The check's description. */
064        public static final String DESCRIPTION = "A variable is assigned to itself. Such assignments can hint at bugs.";
065
066        /**
067         * Mapping from programming languages to the token type representing references
068         * to the current object.
069         */
070        private static final Map<ELanguage, ETokenType> SELF_REFERENCE_TOKEN_TYPES = CollectionUtils.asMap(
071                        createPair(JAVA, THIS), createPair(CPP, THIS), createPair(JAVA, THIS), createPair(CPP, THIS),
072                        createPair(CS, THIS), createPair(JAVASCRIPT, THIS), createPair(KOTLIN, THIS), createPair(PHP, THIS),
073                        createPair(SWIFT, SELF), createPair(OBJECTIVE_C, SELF), createPair(PYTHON, SELF));
074
075        /** The token types that are used to access members of an object. */
076        private static final EnumSet<ETokenType> MEMBER_ACCESS_TOKEN_TYPES = EnumSet.of(DOT, POINTERTO, ARROW);
077
078        /**
079         * Set of tokens which mark the end of the scan area. In general the scan area
080         * is a statement. However, it may also be the closing parenthesis of a loop or
081         * a comma separator. For simplicity, currently the same tokens are used for all
082         * languages. If this proves to cause issues, it might be turned into a method
083         * parameterized with the language. (e.g. if in some language a semicolon
084         * actually modifies the value of the preceding variable).
085         */
086        private static final ETokenType[] END_OF_SCAN_AREA = { SEMICOLON, EOL, // end-of-line (EOL) tokens are only needed
087                                                                                                                                                        // for some languages, e.g. Swift
088                        COMMA, // e.g. separator in for-loop
089                        RPAREN // e.g. closing parenthesis in for-loop
090        };
091
092        /**
093         * Maps programming language to tokens which do not modify the value of the
094         * preceding variable, i.e. should not prevent the "assigned to itself" finding.
095         * The post-increment operator '++' is an exception as it only temporarily
096         * modifies the value of the rhs, but not the value of the lhs.
097         *
098         * @param language
099         *            currently analyzed programming language
100         * @return set of whitelisted tokens for the selected language
101         */
102        private static Set<ETokenType> getWhitelistTokensForLanguage(ELanguage language) {
103                Set<ETokenType> whitelistedTokens = new HashSet<>();
104
105                if (language == KOTLIN) {
106                        whitelistedTokens.add(DOUBLE_EXCLAMATION); // not-null assertion
107                }
108                whitelistedTokens.add(PLUSPLUS); // post-increment operator does not change the value of lhs in 'a = a++'
109                // Add additional tokens which should be whitelisted here
110
111                return whitelistedTokens;
112        }
113
114        /** {@inheritDoc} */
115        @Override
116        protected String getXPathSelectionString() {
117                return "//STATEMENT";
118        }
119
120        /** {@inheritDoc} */
121        @Override
122        protected void processEntity(ShallowEntity entity) throws CheckException {
123                List<IToken> tokens = entity.ownStartTokens();
124                List<Integer> equalsSignTokenIndices = TokenStreamUtils.findAll(tokens, EnumSet.of(EQ));
125
126                for (int equalsSignTokenIndex : equalsSignTokenIndices) {
127                        List<IToken> wholeLhs = tokens.subList(0, equalsSignTokenIndex);
128
129                        // An expression should consist of a sequence of identifiers/self-references
130                        // separated by member access tokens.
131                        // During scanning we toggle the following flag for each scanned token,
132                        // e.g. to prevent two consecutive identifiers as in 'String a = a'
133                        boolean identifierExpected = true;
134
135                        // Scan backwards from the equals sign until the expression is considered
136                        // complete
137                        int startIndexOfLhs = 0;
138                        for (int i = wholeLhs.size() - 1; i >= 0; i--) {
139                                IToken currentToken = wholeLhs.get(i);
140                                ETokenType currentTokenType = currentToken.getType();
141
142                                boolean reachedStartOfExpression;
143                                if (identifierExpected) {
144                                        reachedStartOfExpression = !getValidExpressionTokenTypes().contains(currentTokenType);
145                                } else {
146                                        reachedStartOfExpression = !MEMBER_ACCESS_TOKEN_TYPES.contains(currentTokenType);
147                                }
148
149                                if (reachedStartOfExpression) {
150                                        startIndexOfLhs = i + 1;
151                                        break;
152                                }
153
154                                identifierExpected = !identifierExpected;
155                        }
156
157                        List<IToken> lhs = tokens.subList(startIndexOfLhs, equalsSignTokenIndex);
158
159                        // Get rhs until end of scan area
160                        List<IToken> wholeRhs = tokens.subList(equalsSignTokenIndex + 1, tokens.size());
161                        int endIndexOfRhs = TokenStreamUtils.firstTokenOfType(wholeRhs, END_OF_SCAN_AREA);
162                        if (endIndexOfRhs == -1) {
163                                // If no end token is found, we just take all remaining tokens
164                                endIndexOfRhs = wholeRhs.size();
165                        }
166                        List<IToken> rhs = wholeRhs.subList(0, endIndexOfRhs);
167                        processAssignment(lhs, rhs, tokens, equalsSignTokenIndex);
168                }
169        }
170
171        /**
172         * Process the given assignment of lhs and rhs. If both sides are considered
173         * equal, a finding is created.
174         * 
175         * @param lhs
176         *            tokens which are on the lhs of the equals sign
177         * @param rhs
178         *            tokens which are on the rhs of the equals sign
179         */
180        private void processAssignment(List<IToken> lhs, List<IToken> rhs, List<IToken> tokens, int equalsSignTokenIndex)
181                        throws CheckException {
182                if (lhs.isEmpty() || rhs.size() < lhs.size()) {
183                        // If lhs is empty, there is no valid assignment
184                        // If the rhs is shorter then the lhs, they cannot be equal.
185                        // However, rhs may be larger than lhs as it can contain remaining tokens
186                        return;
187                }
188
189                // Compare tokens from rhs and lhs
190                for (int i = 0; i < lhs.size(); i++) {
191                        IToken currentLhsToken = lhs.get(i);
192                        IToken currentRhsToken = rhs.get(i);
193
194                        if (!currentLhsToken.getText().equals(currentRhsToken.getText())) {
195                                // Mismatching token detected -> sides are not considered equal
196                                return;
197                        }
198                }
199
200                List<IToken> remainingTokens = rhs.subList(lhs.size(), rhs.size());
201                if (!remainingTokensWhitelisted(remainingTokens)) {
202                        return;
203                }
204
205                if (isOptionalArgumentAssignmentInsideFunctionCall(tokens, equalsSignTokenIndex)) {
206                        return;
207                }
208
209                String variableName = TokenStreamTextUtils.concatTokenTexts(lhs);
210                createFinding("`" + variableName + "` is assigned to itself", lhs.get(0), lhs.get(lhs.size() - 1));
211        }
212
213        /**
214         * Check whether the remaining tokens of the expression on rhs are empty or
215         * whitelisted for the current language. We only want to create findings for 'a
216         * = a', 'this.a = this.a', ... We need to prevent false positives for something
217         * like 'a = a + 1' (TS-18709) For this case we check whether the remaining
218         * characters are whitelisted.
219         *
220         * @param remainingTokens
221         *            remaining tokens after the rhs expression
222         * @return {@code true} if remaining tokens of the statement are empty or
223         *         whitelisted, {@code false} otherwise
224         */
225        private boolean remainingTokensWhitelisted(List<IToken> remainingTokens) {
226                Set<ETokenType> whitelistedTokens = getWhitelistTokensForLanguage(context.getLanguage());
227
228                // If remainingTokens is empty, this will also return true
229                return whitelistedTokens.containsAll(CollectionUtils.map(remainingTokens, IToken::getType));
230        }
231
232        /**
233         * Returns the token types that are valid for expressions (either identifier or
234         * self reference keyword).
235         * 
236         * @return the list of valid token types
237         */
238        private EnumSet<ETokenType> getValidExpressionTokenTypes() {
239                EnumSet<ETokenType> types = EnumSet.of(IDENTIFIER);
240
241                Optional<ETokenType> selfReferenceKeyword = Optional
242                                .ofNullable(SELF_REFERENCE_TOKEN_TYPES.get(context.getLanguage()));
243                selfReferenceKeyword.ifPresent(types::add);
244
245                return types;
246        }
247
248        /**
249         * Checks whether the assignment is an optional argument assignment inside a
250         * function call by checking for an opening bracket on the left, an identifier
251         * in front of the opening parenthesis and a closing bracket on the right. The
252         * intended format is: methodName( ... , a = a , ... ).
253         * 
254         * @param tokens
255         *            the list of all tokens
256         * @param equalsSignTokenIndex
257         *            the index of the equals sign token inside this list
258         */
259        private static boolean isOptionalArgumentAssignmentInsideFunctionCall(List<IToken> tokens,
260                        int equalsSignTokenIndex) {
261                int currPos = equalsSignTokenIndex - 2;
262                int lparenTakenCounter = 0;
263                // Checks if there is a LPAREN token on the left of the equalsSignToken.
264                // Since we are looking for the opening parenthesis of the method we ignore any
265                // other pairs of parentheses in between.
266                while (currPos >= 0 && (lparenTakenCounter > 0 || tokens.get(currPos).getType() != LPAREN)) {
267                        ETokenType currentType = tokens.get(currPos).getType();
268                        if (currentType == LPAREN) {
269                                --lparenTakenCounter;
270                        } else if (currentType == RPAREN) {
271                                ++lparenTakenCounter;
272                        }
273                        --currPos;
274                }
275                if (currPos <= 0) {
276                        return false;
277                }
278                // Check if the token in front of the LPAREN token is an IDENTIFIER.
279                if (tokens.get(currPos - 1).getType() != IDENTIFIER) {
280                        return false;
281                }
282                // Check if there is a RPAREN token on the right of the equalsSignToken.
283                currPos = equalsSignTokenIndex + 2;
284                int rparenTakenCounter = 0;
285                while (currPos < tokens.size() && (rparenTakenCounter > 0 || tokens.get(currPos).getType() != RPAREN)) {
286                        ETokenType currentType = tokens.get(currPos).getType();
287                        if (currentType == RPAREN) {
288                                --rparenTakenCounter;
289                        } else if (currentType == LPAREN) {
290                                ++rparenTakenCounter;
291                        }
292                        ++currPos;
293                }
294                if (currPos < tokens.size()) {
295                        return true;
296                }
297                return false;
298        }
