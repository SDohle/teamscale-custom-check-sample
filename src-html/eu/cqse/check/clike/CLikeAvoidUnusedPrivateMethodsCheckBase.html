052/**
053 * Base class for c-like avoid unused private methods checks.
054 */
055public abstract class CLikeAvoidUnusedPrivateMethodsCheckBase extends CheckImplementationBase {
056
057        /** The general check description. */
058        public static final String DESCRIPTION = "This rule is reported when a private method exists but is never called by any code path.";
059
060        /** The internally used language utilities. */
061        protected CLikeLanguageFeatureParserBase languageFeatureParser;
062
063        /** Constructor. */
064        public CLikeAvoidUnusedPrivateMethodsCheckBase(CLikeLanguageFeatureParserBase languageFeatureParser) {
065                CCSMAssert.isNotNull(languageFeatureParser);
066                this.languageFeatureParser = languageFeatureParser;
067        }
068
069        /** {@inheritDoc} */
070        @Override
071        public void execute() throws CheckException {
072                SetMap<String, Integer> methodCalls = retrieveMethodUses();
073
074                // here, we use the filtered rootEntity since we want to execute this
075                // check only for methods that were not filtered in the project
076                // settings.
077                List<ShallowEntity> methods = select(context.getRootEntity(ECodeViewOption.FILTERED),
078                                getXPathMethodSelectionString());
079                for (ShallowEntity method : CollectionUtils.filter(methods, this::isMethodIncluded)) {
080                        String methodName = method.getName();
081                        Set<Integer> callArities = methodCalls.getCollection(methodName);
082                        if (callArities == null || !isCalled(method, callArities)) {
083                                createFindingOnFirstLine("Private method `" + methodName + "` is unused", method);
084                        }
085                }
086        }
087
088        /** Returns a xPath selection string for selecting all included methods. */
089        private String getXPathMethodSelectionString() {
090                StringBuilder selectionString = new StringBuilder("//TYPE");
091                getTypeSelectorPredicate().ifPresent(predicate -> selectionString.append("[").append(predicate).append("]"));
092                selectionString.append("/METHOD[not(subtype('").append(SubTypeNames.CONSTRUCTOR)
093                                .append("')) and modifiers('private')]");
094                return selectionString.toString();
095        }
096
097        /**
098         * Returns a selector predicate that is applied on the methods parent type.
099         */
100        protected Optional<String> getTypeSelectorPredicate() {
101                return Optional.empty();
102        }
103
104        /**
105         * Retrieves all method calls and references within the analyzed token element.
106         * A mapping from method names to the called method arities is returned.
107         */
108        private SetMap<String, Integer> retrieveMethodUses() throws CheckException {
109                SetMap<String, Integer> calledMethods = new SetMap<>();
110                for (List<IToken> tokens : retrieveTokensForMethodUseAnalysis()) {
111                        retrieveMethodUses(tokens, calledMethods);
112                }
113                return calledMethods;
114        }
115
116        /**
117         * Retrieves a list of token lists that shall be analysed for method uses.
118         */
119        private List<List<IToken>> retrieveTokensForMethodUseAnalysis() throws CheckException {
120                // here, we use the unfiltered rootEntity since we want to find calls of
121                // methods even if they were filtered (we don't generate findings if we
122                // find a call)
123                List<List<IToken>> tokens = CollectionUtils.map(selectUnfilteredPreProcessed("//STATEMENT | //ATTRIBUTE"),
124                                ShallowEntity::includedTokens);
125                tokens.addAll(retrieveAdditionalTokensForMethodUseAnalysis());
126                return tokens;
127        }
128
129        /**
130         * Returns additional tokens that shall be analysed for method uses. This method
131         * may be overridden by subclasses.
132         * 
133         * @throws CheckException
134         *             if retrieving additional tokens fails
135         */
136        protected List<List<IToken>> retrieveAdditionalTokensForMethodUseAnalysis() throws CheckException {
137                return CollectionUtils.emptyList();
138        }
139
140        /** Retrieves all method calls from the given tokens. */
141        private void retrieveMethodUses(List<IToken> tokens, SetMap<String, Integer> calledMethods) {
142                retrieveMethodCalls(tokens, calledMethods);
143                retrieveMethodReferences(tokens, calledMethods);
144        }
145
146        /**
147         * Retrieves all method calls from the given tokens and adds them to the given
148         * SetMap.
149         */
150        protected void retrieveMethodCalls(List<IToken> tokens, SetMap<String, Integer> calledMethods) {
151                List<Integer> callIndices = TokenStreamUtils.firstTokenOfTypeSequences(tokens, 0, IDENTIFIER, LPAREN);
152                for (int callIndex : callIndices) {
153                        List<IToken> parameters = TokenStreamUtils.tokensBetweenWithNesting(tokens, callIndex + 1, LPAREN, RPAREN);
154                        int methodCallArity = 0;
155                        if (parameters.size() > 0) {
156                                methodCallArity = TokenStreamUtils.splitWithNesting(parameters, COMMA,
157                                                Arrays.asList(LPAREN, LBRACK, LBRACE, LT), Arrays.asList(RPAREN, RBRACK, RBRACE, GT)).size();
158                        }
159                        calledMethods.add(tokens.get(callIndex).getText(), methodCallArity);
160                }
161        }
162
163        /**
164         * Retrieves all method references from the given tokens and adds them to the
165         * given SetMap.
166         */
167        protected abstract void retrieveMethodReferences(List<IToken> tokens, SetMap<String, Integer> calledMethods);
168
169        /**
170         * Checks whether a method is called based on the length of its argument list.
171         */
172        private boolean isCalled(ShallowEntity method, Set<Integer> callArities) {
173                PairList<Integer, Boolean> methodCallArities = getMethodCallArities(
174                                languageFeatureParser.getSplitParameterTokens(method));
175                return callArities.contains(UNKNOWN_ARITY)
176                                || !Collections.disjoint(methodCallArities.getFirstList(), callArities)
177                                || isCalledWithHigherArityAndHasVariableLengthArgumentList(methodCallArities, callArities);
178        }
179
180        /**
181         * Returns a PairList of possible method call arities for the given parameter
182         * tokens. The first value represents the minimum arity of the method call
183         * (excluding variable-length parameter) and the second value is a boolean
184         * representing whether the method has variable length parameters. For example,
185         * method foo(int x) has arity (1,false), method foo(int x, String... y) has
186         * arity (1,true) and method foo(String... x) has arity (0,true).
187         *
188         * This is necessary to represent the different possible arities for extension
189         * methods.
190         */
191        protected PairList<Integer, Boolean> getMethodCallArities(List<List<IToken>> parameterTokens) {
192                PairList<Integer, Boolean> result = new PairList<>();
193                int methodCallArity = parameterTokens.size();
194                boolean variableLengthArguments = parameterTokens.stream()
195                                .anyMatch(tokens -> languageFeatureParser.hasVariableLengthArgumentList(tokens));
196                // Excluding variable-length parameter from the minimum arity.
197                if (variableLengthArguments) {
198                        methodCallArity -= 1;
199                }
200                result.add(methodCallArity, variableLengthArguments);
201                return result;
202        }
203
204        /**
205         * Returns whether this method accepts a variable length argument list and is
206         * called with more arguments than expected.
207         */
208        private static boolean isCalledWithHigherArityAndHasVariableLengthArgumentList(
209                        PairList<Integer, Boolean> methodCallArities, Set<Integer> callArities) {
210                return methodCallArities.stream()
211                                .anyMatch(pair -> pair.getSecond() && containsValueHigherOrEqualThan(callArities, pair.getFirst()));
212        }
213
214        /**
215         * Tests if a given set contains at least one value that is higher or equal to
216         * the comparison value.
217         */
218        private static boolean containsValueHigherOrEqualThan(Set<Integer> values, int comparisonValue) {
219                return values.stream().anyMatch(value -> value >= comparisonValue);
220        }
221
222        /** Returns whether this methods shall be included by this analysis. */
223        protected boolean isMethodIncluded(ShallowEntity method) {
224                Set<Integer> parameterCounts = getExcludedMethods().getCollection(method.getName());
225                return parameterCounts == null
226                                || !parameterCounts.contains(ShallowParsingUtils.extractParameterNameTokens(method).size());
227        }
228
229        /**
230         * Returns methods that are excluded from the check. The methods are returned as
231         * mapping from method names to arities.
232         */
233        protected SetMap<String, Integer> getExcludedMethods() {
234                return new SetMap<>();
235        }
236}
