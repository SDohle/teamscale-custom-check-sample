006package eu.cqse.check.clike;
007
008import java.util.ArrayList;
009import java.util.List;
010
011import org.conqat.lib.commons.collections.CollectionUtils;
012import org.conqat.lib.commons.collections.ListMap;
013import org.conqat.lib.commons.collections.PairList;
014
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckImplementationBase;
017import eu.cqse.check.framework.core.phase.ECodeViewOption;
018import eu.cqse.check.framework.scanner.IToken;
019import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
020import eu.cqse.check.framework.util.clike.ConditionalBlock;
021import eu.cqse.check.framework.util.clike.ConditionalBlockGroupBase;
022import eu.cqse.check.framework.util.clike.ConditionalBlockGroupExtractor;
023
024/**
025 * The base class for all checks on the grouped conditional blocks.
026 */
027public abstract class CLikeConditionalBlockCheckBase extends CheckImplementationBase {
028
029        /** Corresponding blocks of conditional statements */
030        protected List<ConditionalBlockGroupBase> blockGroups;
031
032        /** {@inheritDoc} */
033        @Override
034        public void execute() throws CheckException {
035                blockGroups = new ArrayList<>();
036
037                for (ShallowEntity entity : context.getAbstractSyntaxTree(ECodeViewOption.FILTERED_PREPROCESSED)) {
038                        blockGroups.addAll(ConditionalBlockGroupExtractor.findCorrespondingBlockGroups(entity));
039                }
040                processBlockGroups();
041        }
042
043        /**
044         * Groups the blocks according to their content hash (see
045         * {@link #hashBlockContent(ConditionalBlock)}.
046         * 
047         * @param blockGroup
048         *            the block group for which the blocks are getting checked for
049         *            equality
050         * @return a map of identical block content hashes to their corresponding blocks
051         */
052        private ListMap<Integer, ConditionalBlock> groupBlocksByContentHash(ConditionalBlockGroupBase blockGroup) {
053                ListMap<Integer, ConditionalBlock> blocksByContentHash = new ListMap<>();
054
055                for (ConditionalBlock block : blockGroup.getCorrespondingBlocks()) {
056                        // Ignore empty content blocks for if and switch (there is a separate check for
057                        // this)
058                        if (block.hasContent()) {
059                                blocksByContentHash.add(hashBlockContent(block), block);
060                        }
061                }
062
063                return blocksByContentHash;
064        }
065
066        /**
067         * Checks if there are duplicates in the grouped blocks and creates findings if
068         * duplicates are found
069         * 
070         * @param blocksByContentHash
071         *            the blocks which get checked for duplicates
072         */
073        private void createFindingsForGroupedBlocks(ListMap<Integer, ConditionalBlock> blocksByContentHash)
074                        throws CheckException {
075                for (Integer contentHash : blocksByContentHash.getKeys()) {
076                        List<ConditionalBlock> groupedBlock = blocksByContentHash.getCollection(contentHash);
077                        if (groupedBlock.size() > 1 && !areEmptyBlocks(groupedBlock)) {
078                                createFindingsForSiblingBlocks(groupedBlock);
079                        }
080                }
081        }
082
083        /**
084         * Two ore more empty conditional bodies needn't be reported as identical
085         * bodies, they already create findings themselves.
086         *
087         * @param blocks
088         *            List of identical blocks which are checked for emptiness.
089         */
090        private static boolean areEmptyBlocks(List<ConditionalBlock> blocks) {
091                return blocks.get(0).getContent().isEmpty();
092        }
093
094        /**
095         * For given blocks extract the start and end tokens and create findings for
096         * each block with the other blocks as siblings
097         * 
098         * @param blocks
099         *            the blocks that are duplicates of each other
100         */
101        private void createFindingsForSiblingBlocks(List<ConditionalBlock> blocks) throws CheckException {
102                PairList<IToken, IToken> startAndEndTokens = new PairList<>();
103                for (ConditionalBlock block : blocks) {
104                        startAndEndTokens.add(getStartToken(block), getEndToken(block));
105                }
106
107                createFindingForSiblings(getFindingsMessage(), startAndEndTokens);
108        }
109
110        /**
111         * Extracts the start token from the given block of statements
112         * 
113         * @param block
114         *            the block of statements from which the start token gets extracted
115         * @return the start token
116         */
117        private static IToken getStartToken(ConditionalBlock block) {
118                return block.getContent().get(0).includedTokens().get(0);
119        }
120
121        /**
122         * Extracts the end token from the given block of statements
123         * 
124         * @param block
125         *            the block of statements from which the end token gets extracted
126         * @return the end token
127         */
128        private static IToken getEndToken(ConditionalBlock block) {
129                List<IToken> lastTokens = CollectionUtils.getLast(block.getContent()).includedTokens();
130                return CollectionUtils.getLast(lastTokens);
131        }
132
133        /**
134         * Override this method to set the findings message for the sibling findings.
135         */
136        protected abstract String getFindingsMessage();
137
138        /**
139         * Converts a block into the needed representation to compare it with other
140         * blocks. Override it for the use in
141         * {@link #groupBlocksByContentHash(ConditionalBlockGroupBase)}.
142         */
143        protected abstract int hashBlockContent(ConditionalBlock block);
144
145        /**
146         * Groups the blocks in the block groups according to
147         * {@link #groupBlocksByContentHash(ConditionalBlockGroupBase)} and creates
148         * findings for grouped blocks in
149         * {@link #createFindingsForGroupedBlocks(ListMap)}. Override this method if the
150         * findings creation is not done on grouped blocks or doesn't need grouping.
151         */
152        protected void processBlockGroups() throws CheckException {
153                for (ConditionalBlockGroupBase blockGroup : blockGroups) {
154                        ListMap<Integer, ConditionalBlock> groupedBlocks = groupBlocksByContentHash(blockGroup);
155                        createFindingsForGroupedBlocks(groupedBlocks);
156                }
157        }
158}
