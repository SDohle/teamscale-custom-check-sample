017package eu.cqse.check.clike;
018
019import java.util.List;
020
021import org.conqat.lib.commons.assertion.CCSMAssert;
022
023import eu.cqse.check.framework.core.CheckException;
024import eu.cqse.check.framework.core.CheckImplementationBase;
025import eu.cqse.check.framework.scanner.IToken;
026import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
027import eu.cqse.check.framework.util.CLikeLanguageFeatureParserBase;
028
029/**
030 * Base class for checks that analyze the usage of fields in entities. By
031 * default if no usage is found, a finding is created. This behavior can be
032 * inverted by overriding {@link #createFindingOnFound()} and returning true.
033 *
034 * @author $Author: kupka$
035 * @version $Rev: 55686 $
036 * @ConQAT.Rating GREEN Hash: 7056956A7BB484D88CCA14A029B722F5
037 */
038public abstract class CLikeFieldUsageCheckBase extends CheckImplementationBase {
039
040        /** The internally used language utilities. */
041        protected CLikeLanguageFeatureParserBase languageFeatureParser;
042
043        /** Constructor. */
044        public CLikeFieldUsageCheckBase(CLikeLanguageFeatureParserBase utils) {
045                CCSMAssert.isNotNull(utils);
046                this.languageFeatureParser = utils;
047        }
048
049        /** {@inheritDoc} */
050        @Override
051        public void execute() throws CheckException {
052                List<ShallowEntity> selectedFields = select(
053                                context.getUnfilteredRootEntity(), getFieldSelectionXPath());
054                for (ShallowEntity field : selectedFields) {
055                        processField(field);
056                }
057        }
058
059        /** Processes the given field. */
060        protected void processField(ShallowEntity field) throws CheckException {
061                List<ShallowEntity> processedEntities = select(field.getParent(),
062                                getProcessedEntitiesXPath());
063                processedEntities.remove(field);
064
065                // If someone declares multiple attributes like this:
066                // private int x, y;
067                // it is necessary to create multiple findings, because otherwise it
068                // is ambiguous, which field is unused.
069                for (IToken fieldNameToken : languageFeatureParser
070                                .getVariableNamesFromTokens(field.ownStartTokens())) {
071                        if (createFindingOnFound() == analyze(processedEntities, field,
072                                        fieldNameToken.getText())) {
073                                createFinding(getFindingMessage(fieldNameToken.getText()),
074                                                fieldNameToken);
075                        }
076                }
077        }
078
079        /**
080         * Walks through all given entities and checks their start and end tokens
081         * for uses of the field with the given field name. This is done executing
082         * the {@link #analyzeTokens(List, String, boolean)} method. If it returns
083         * true for any entity, true is returned, otherwise false.
084         */
085        private boolean analyze(List<ShallowEntity> entities, ShallowEntity field,
086                        String fieldName) throws CheckException {
087
088                for (ShallowEntity entity : entities) {
089                        boolean isShadowed = context.getUnfilteredTypeResolution()
090                                        .getTypeLookup(entity)
091                                        .isShadowingVariable(fieldName, field);
092
093                        boolean foundInTokens = analyzeTokens(entity.ownStartTokens(),
094                                        fieldName, isShadowed);
095                        if (!foundInTokens && entity.hasChildren()) {
096                                foundInTokens = analyzeTokens(entity.ownEndTokens(), fieldName,
097                                                isShadowed);
098                        }
099                        if (foundInTokens) {
100                                return true;
101                        }
102                }
103                return false;
104        }
105
106        /**
107         * Analyzes the given tokens for field usages. Returns true if
108         *
109         * @param fieldName
110         *            the name of the field
111         * @param isShadowed
112         *            whether the field is currently shadowed by a local variable
113         */
114        protected abstract boolean analyzeTokens(List<IToken> tokens,
115                        String fieldName, boolean isShadowed);
116
117        /**
118         * Returns a xPath string for selecting all fields that should be analyzed.
119         */
120        protected abstract String getFieldSelectionXPath();
121
122        /**
123         * Returns a xPath string for selecting all entities that are walked through
124         * and checked for field usages.
125         */
126        protected abstract String getProcessedEntitiesXPath();
127
128        /** Returns a finding message for the given field name. */
129        protected abstract String getFindingMessage(String fieldName);
130
131        /**
132         * Returns true if a finding should be created, if a field usage is found.
133         * Otherwise it returns false.
134         */
135        protected boolean createFindingOnFound() {
136                return false;
137        }
138}
