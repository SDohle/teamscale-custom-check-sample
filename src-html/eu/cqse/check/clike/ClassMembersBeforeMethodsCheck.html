001package eu.cqse.check.clike;
002
003import static eu.cqse.check.framework.scanner.ELanguage.CPP;
004import static eu.cqse.check.framework.scanner.ELanguage.CS;
005import static eu.cqse.check.framework.scanner.ELanguage.GOSU;
006import static eu.cqse.check.framework.scanner.ELanguage.JAVA;
007import static eu.cqse.check.framework.shallowparser.framework.EShallowEntityType.ATTRIBUTE;
008import static eu.cqse.check.framework.shallowparser.framework.EShallowEntityType.META;
009import static eu.cqse.check.framework.shallowparser.framework.EShallowEntityType.METHOD;
010
011import java.util.ArrayList;
012import java.util.List;
013
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckGroupNames;
017import eu.cqse.check.framework.core.CheckImplementationBase;
018import eu.cqse.check.framework.core.ECheckParameter;
019import eu.cqse.check.framework.core.EFindingEnablement;
020import eu.cqse.check.framework.core.option.ACheckOption;
021import eu.cqse.check.framework.shallowparser.SubTypeNames;
022import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
023import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
024
025/** {@ConQAT.Doc} */
026@ACheck(name = "Order of class members", description = "In a class, member variables should"
027                + " be declared before member methods.", groupName = CheckGroupNames.BAD_PRACTICE, languages = { GOSU, JAVA,
028                                CPP,
029                                CS }, defaultEnablement = EFindingEnablement.OFF, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
030public class ClassMembersBeforeMethodsCheck extends CheckImplementationBase {
031
032        /** {@ConQAT.Doc} */
033        @ACheckOption(name = "Exemption: constructors", description = "Constructors can be in front of class members")
034        private boolean exemptConstructors = false;
035
036        /** {@inheritDoc} */
037        @Override
038        public void execute() throws CheckException {
039                for (ShallowEntity classEntity : select("//TYPE[subtype('class')]")) {
040                        if (context.getLanguage().equals(CPP)) {
041                                checkMemberOrderForCpp(classEntity);
042                        } else {
043                                checkMemberOrder(select(classEntity, "./ATTRIBUTE | ./METHOD"));
044                        }
045                }
046        }
047
048        /**
049         * Checks the order of the given list of shallow entities; If there are member
050         * variables defined below any method declaration it will create a finding for
051         * each one.
052         */
053        private void checkMemberOrder(List<ShallowEntity> entities) throws CheckException {
054                boolean methodDefinitionStarted = false;
055                // xpath always returns the order of the nodes in the document
056                for (ShallowEntity entity : entities) {
057                        if (!(exemptConstructors && entity.getSubtype().equals(SubTypeNames.CONSTRUCTOR))) {
058                                if (!methodDefinitionStarted && entity.getType().equals(METHOD)) {
059                                        methodDefinitionStarted = checkForStaticInitializer(entity);
060                                } else if (methodDefinitionStarted && entity.getType().equals(ATTRIBUTE)) {
061                                        createFinding("`" + entity.getName() + "` should be before any method declaration", entity);
062                                }
063                        }
064                }
065        }
066
067        /**
068         * Checks whether the analyzed entity is a static initializer or the like.
069         */
070        private boolean checkForStaticInitializer(ShallowEntity entity) {
071                if (context.getLanguage().equals(JAVA) && entity.getSubtype().equals(SubTypeNames.STATIC_INITIALIZER)) {
072                        return false;
073                } else if (context.getLanguage().equals(CS) && entity.getSubtype().equals(SubTypeNames.STATIC_CONSTRUCTOR)) {
074                        return false;
075                }
076                return true;
077        }
078
079        /**
080         * In C++ it is allowed to have access modifier in the classes, for example
081         * "public:". We want to check the member order inside every access modifier
082         * separately. Therefore we extract those shallow entities and give it the
083         * checkMemberOrder function.
084         */
085        private void checkMemberOrderForCpp(ShallowEntity classEntity) throws CheckException {
086                List<ShallowEntity> membersList = new ArrayList<>();
087                for (ShallowEntity child : classEntity.getChildren()) {
088                        EShallowEntityType type = child.getType();
089                        if (type.equals(META) && child.getSubtype().matches("public|protected|private")) {
090                                checkMemberOrder(membersList);
091                                membersList.clear();
092                        } else if (type.equals(ATTRIBUTE) || type.equals(METHOD)) {
093                                membersList.add(child);
094                        }
095                }
096                // when there are no children left, analyze the rest
097                checkMemberOrder(membersList);
098        }
099}
