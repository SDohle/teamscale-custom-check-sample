006package eu.cqse.check.clike;
007
008import java.util.ArrayList;
009import java.util.List;
010
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckGroupNames;
013import eu.cqse.check.framework.core.ECheckParameter;
014import eu.cqse.check.framework.core.EFindingEnablement;
015import eu.cqse.check.framework.core.option.ACheckOption;
016import eu.cqse.check.framework.scanner.ELanguage;
017import eu.cqse.check.framework.scanner.ETokenType;
018import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
019import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
020import eu.cqse.check.framework.util.clike.ConditionalBlock;
021
022/**
023 * Check for equal statements inside conditional expressions. The statements
024 * inside conditional expressions are checked if they consist of the same
025 * statements. Conditional expressions checked are: if/else/else if and
026 * switch/case constructs.
027 */
028@ACheck(name = ConditionalStatementsEqualCheck.CHECK_NAME, description = ConditionalStatementsEqualCheck.DESCRIPTION, languages = {
029                ELanguage.JAVA, ELanguage.CS, ELanguage.CPP, ELanguage.OPEN_CL, ELanguage.JAVASCRIPT, ELanguage.XTEND,
030                ELanguage.SWIFT, ELanguage.GROOVY, ELanguage.GOSU, ELanguage.KOTLIN, ELanguage.OBJECTIVE_C }, parameters = {
031                                ECheckParameter.ABSTRACT_SYNTAX_TREE }, groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW)
032public class ConditionalStatementsEqualCheck extends CLikeConditionalBlockCheckBase {
033
034        /** The findings description for this check. */
035        public static final String DESCRIPTION = "Statements inside the body of `if, else, else-if` or "//
036                        + "`switch/case` constructs should not be the same, because it introduces unnecessary duplicated code. "//
037                        + "A way to deal with equal statements in conditional expressions is to merge the condition and keep the body once.\n\n"//
038                        + "``` js \n"//
039                        + "// The following piece of code:\n"//
040                        + "   if (condition1) { \n"//
041                        + "     foo(); \n"//
042                        + "   } else if (condition2) { \n"//
043                        + "     foo(); \n"//
044                        + "   } else { \n"//
045                        + "     bar(); \n"//
046                        + "   } \n\n"//
047                        + "// can be rewritten as: \n"//
048                        + "   if (condition1 || condition2) { \n"//
049                        + "     foo(); \n"//
050                        + "   } else { \n"//
051                        + "     bar(); \n"//
052                        + "   }\n"//
053                        + "```";
054
055        @ACheckOption(name = CHECK_NAME
056                        + ": Use character equality.", description = "The check can search for equality of the tokens or the string representation of the checked conditional statements. "
057                                        + "If checked, the character equality is used which is much more precise and only creates findings if the conditional statements are exactly the same on the character level. "
058                                        + "If not checked, equality of the tokens is used which will generate findings that are likely to be equal in meaning.")
059        private boolean useCharacterEquality = true;
060
061        /** The name of this check. */
062        public static final String CHECK_NAME = "Statements in Conditional Expression are equal";
063
064        @Override
065        protected String getFindingsMessage() {
066                return CHECK_NAME;
067        }
068
069        @Override
070        protected int hashBlockContent(ConditionalBlock block) {
071                if (useCharacterEquality) {
072                        StringBuilder representation = new StringBuilder();
073                        for (ShallowEntity statement : block.getContent()) {
074                                representation.append(TokenStreamTextUtils.concatTokenTexts(statement.includedTokens()));
075                        }
076                        return representation.toString().hashCode();
077                }
078
079                List<ETokenType> tokenTypes = new ArrayList<>();
080                for (ShallowEntity statement : block.getContent()) {
081                        statement.includedTokens().forEach(token -> tokenTypes.add(token.getType()));
082                }
083                return tokenTypes.hashCode();
084        }
085}
