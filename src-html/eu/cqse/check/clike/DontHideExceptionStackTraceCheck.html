054/**
055 * {@ConQAT.Doc}
056 */
057@ACheck(name = "Exception stacktrace is lost", groupName = CheckGroupNames.EXCEPTION_HANDLING, defaultEnablement = EFindingEnablement.YELLOW, languages = {
058                JAVA, CS, ELanguage.OBJECTIVE_C }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE }, description = ""
059                                + "If you catch an exception and then throw a different exception, you should preserve the original stacktrace by "
060                                + "passing the caught exception to the constructor of the thrown exception. "
061                                + "Failing to do so will cause the original stacktrace to be lost, which will hinder debugging the root cause.")
062public class DontHideExceptionStackTraceCheck extends EntityCheckBase {
063
064        /** Name of the StackTrace property in C#. */
065        private static final String STACK_TRACE = "StackTrace";
066
067        /** Name of the getStackTrace method in Java. */
068        private static final String GET_STACK_TRACE = "getStackTrace";
069
070        /** Name of the ExceptionUtils class in Java. */
071        private static final String EXCEPTION_UTILS = "ExceptionUtils";
072
073        private static final String DOT = ".";
074        private static final String LEFT_PARANTHESIS = "(";
075        private static final String RIGHT_PARANTHESIS = ")";
076
077        /**
078         * The message of all findings created by this check.
079         */
080        private static final String FINDING_MESSAGE = "Exception stacktrace is lost";
081
082        /**
083         * Matches the name and type of the caught exception in a catch entity. Group 1
084         * contains the type, group 2 contains the name.
085         */
086        private static final TokenPattern EXCEPTION_VARIABLE_NAME_PATTERN = new TokenPattern().sequence(IDENTIFIER).group(1)
087                        .sequence(IDENTIFIER).group(2).sequence(RPAREN, LBRACE);
088
089        /**
090         * Matches a single variable being used as a constructor argument. Might also
091         * match more than that, but we would rather have less false positives than a
092         * higher precision, thus this is OK. Group 1 contains the name of the matched
093         * variable.
094         */
095        private static final TokenPattern VARIABLE_AS_CONSTRUCTOR_ARGUMENT_PATTERN = new TokenPattern()
096                        .sequence(EnumSet.of(LPAREN, COMMA)).sequence(IDENTIFIER).group(1).sequence(EnumSet.of(RPAREN, COMMA));
097
098        /** {@ConQAT.Doc} */
099        @ACheckOption(name = "Exception stacktrace is lost: Ignored exception types", description = "A list of exception types for which not to check whether the stacktrace is passed on.")
100        private Set<String> ignoredExceptionTypes = CollectionUtils.asHashSet("NumberFormatException");
101
102        /** {@inheritDoc} */
103        @Override
104        protected String getXPathSelectionString() {
105                return "//STATEMENT[subtype('" + SubTypeNames.CATCH + "')]";
106        }
107
108        /** {@inheritDoc} */
109        @Override
110        protected void processEntity(ShallowEntity entity) throws CheckException {
111                if (context.getLanguage() == CS && isAnonymousCsCatchBlock(entity)) {
112                        checkThrows(entity, this::checkThrowsStatementInAnonymousCsCatchBlock);
113                        return;
114                }
115
116                TokenPatternMatch exceptionVariableMatch = EXCEPTION_VARIABLE_NAME_PATTERN
117                                .findFirstMatch(entity.ownStartTokens());
118                if (exceptionVariableMatch == null) {
119                        return;
120                }
121
122                if (ignoredExceptionTypes.contains(exceptionVariableMatch.groupString(1))) {
123                        return;
124                }
125
126                String exceptionVariableName = exceptionVariableMatch.groupString(2);
127                checkThrows(entity, throwEntity -> checkThrowsStatement(exceptionVariableName, throwEntity));
128        }
129
130        /**
131         * Returns <code>true</code> if the given entity is an anonymous C# catch block:
132         * <code>catch {</code> or <code>catch (ExceptionClass) {</code>
133         */
134        private static boolean isAnonymousCsCatchBlock(ShallowEntity entity) {
135                return TokenStreamUtils.hasTypes(entity.ownStartTokens(), CATCH, LBRACE)
136                                || TokenStreamUtils.hasTypes(entity.ownStartTokens(), CATCH, LPAREN, IDENTIFIER, RPAREN, LBRACE);
137        }
138
139        /**
140         * Check all throw statements in the given catch entity by applying the given
141         * check function to them.
142         */
143        private void checkThrows(ShallowEntity catchEntity, ThrowEntityChecker checkFunction) throws CheckException {
144                for (ShallowEntity throwEntity : select(catchEntity,
145                                ".//STATEMENT[subtype('" + SubTypeNames.SIMPLE_STATEMENT + "') and name-matches('throw')]")) {
146                        checkFunction.check(throwEntity);
147                }
148        }
149
150        /**
151         * Checks a throw statement inside an anonymous C# catch block.
152         * 
153         * @see #isAnonymousCsCatchBlock(ShallowEntity)
154         */
155        private void checkThrowsStatementInAnonymousCsCatchBlock(ShallowEntity throwEntity) throws CheckException {
156                if (!TokenStreamUtils.hasTypes(throwEntity.includedTokens(), THROW, SEMICOLON)) {
157                        createFinding(throwEntity);
158                }
159        }
160
161        /** Checks a normal throw statement. */
162        private void checkThrowsStatement(String exceptionVariableName, ShallowEntity throwEntity) throws CheckException {
163                if (!TokenStreamUtils.containsAny(throwEntity.ownStartTokens(), NEW)) {
164                        return;
165                }
166
167                List<TokenPatternMatch> matches = VARIABLE_AS_CONSTRUCTOR_ARGUMENT_PATTERN
168                                .findAll(throwEntity.includedTokens());
169
170                for (TokenPatternMatch match : matches) {
171                        if (match.groupString(1).equals(exceptionVariableName)) {
172                                return;
173                        }
174                }
175
176                ELanguage fileLanguage = context.getLanguage();
177                UnmodifiableList<ShallowEntity> siblings = throwEntity.getParent().getChildren();
178                for (ShallowEntity sibling : siblings) {
179                        if ((fileLanguage == CS && entityContainsSequence(sibling, exceptionVariableName, DOT, STACK_TRACE))
180                                        || (fileLanguage == JAVA
181                                                        && (entityContainsJavaGetStackTraceSequence(sibling, exceptionVariableName)))) {
182                                return;
183                        }
184                }
185
186                createFinding(throwEntity);
187        }
188
189        /**
190         * Checks whether shallow entity contains tokens with the given sequence of
191         * text.
192         */
193        private static boolean entityContainsSequence(ShallowEntity entity, String... sequence) {
194                return TokenStreamTextUtils.findSequence(entity.includedTokens(), 0, IDENTIFIER, sequence) != NOT_FOUND;
195        }
196
197        /**
198         * Checks whether shallow entity contains tokens with one of the two given
199         * sequences of text (each includes `getStackTrace` Java method and name of
200         * exception variable).
201         */
202        private static boolean entityContainsJavaGetStackTraceSequence(ShallowEntity entity, String exceptionVariableName) {
203                return (entityContainsSequence(entity, exceptionVariableName, DOT, GET_STACK_TRACE)
204                                || entityContainsSequence(entity, EXCEPTION_UTILS, DOT, GET_STACK_TRACE, LEFT_PARANTHESIS,
205                                                exceptionVariableName, RIGHT_PARANTHESIS));
206        }
207
208        /**
209         * Creates the finding produced by this check for the given throw entity.
210         */
211        private void createFinding(ShallowEntity throwEntity) throws CheckException {
212                createFindingOnFirstLine(FINDING_MESSAGE, throwEntity);
213        }
214
215        /**
216         * Checks a throw entity and may create findings for it. Needed so we can throw
217         * {@link CheckException}.
218         */
219        @FunctionalInterface
220        private static interface ThrowEntityChecker {
221
222                /** Performs the check and may create a findings for the entity. */
223                public void check(ShallowEntity throwEntity) throws CheckException;
224
225        }
