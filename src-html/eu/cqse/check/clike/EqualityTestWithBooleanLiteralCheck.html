006package eu.cqse.check.clike;

008import java.util.ArrayList;
009import java.util.List;
010import java.util.Set;
011
012import org.conqat.lib.commons.collections.CollectionUtils;
013import org.conqat.lib.commons.collections.UnmodifiableList;
014
015import eu.cqse.check.base.EntityCheckBase;
016import eu.cqse.check.framework.core.ACheck;
017import eu.cqse.check.framework.core.CheckException;
018import eu.cqse.check.framework.core.CheckGroupNames;
019import eu.cqse.check.framework.core.ECheckParameter;
020import eu.cqse.check.framework.core.EFindingEnablement;
021import eu.cqse.check.framework.core.phase.ECodeViewOption;
022import eu.cqse.check.framework.scanner.ELanguage;
023import eu.cqse.check.framework.scanner.ETokenType;
024import eu.cqse.check.framework.scanner.IToken;
025import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
026import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
027import eu.cqse.check.framework.typetracker.ITypeResolution;
028import eu.cqse.check.framework.typetracker.ScopedTypeLookup;
029import eu.cqse.check.framework.typetracker.TypedVariable;
030
031/**
032 * {@ConQAT.Doc}
033 */
034@ACheck(name = "Equality check with boolean literal", description = "Checking equality with boolean literals is unnecessarily verbose. For instance, conditions like `if (value == true)` can be simplified to `if (value)`.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
035                ELanguage.CPP, ELanguage.CS,
036                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE, ECheckParameter.TYPE_RESOLUTION })
037public class EqualityTestWithBooleanLiteralCheck extends EntityCheckBase {
038
039        /** {@inheritDoc} */
040        @Override
041        protected String getXPathSelectionString() {
042                return "//STATEMENT";
043        }
044
045        /** {@inheritDoc} */
046        @Override
047        protected void processEntity(ShallowEntity entity) throws CheckException {
048                UnmodifiableList<IToken> tokens = entity.ownStartTokens();
049                List<List<IToken>> literals = TokenStreamUtils.split(tokens, ETokenType.ANDAND, ETokenType.OROR);
050
051                ITypeResolution typeResolution = context.getTypeResolution(ECodeViewOption.FILTERED);
052                ScopedTypeLookup scopedTypeLookup = typeResolution.getTypeLookup(entity);
053                Set<String> variableNames = scopedTypeLookup.getAllVariables();
054
055                for (List<IToken> literalTokens : literals) {
056                        List<Integer> literalIndexes = new ArrayList<>();
057                        if (containsEqualityCheckAgainstBooleanLiteral(literalTokens)) {
058                                // don't create a finding if the '?' operator in C# is used. If so,
059                                // explicit casting from bool? to bool is needed, e.g. via "== true"
060                                if (context.getLanguage() == ELanguage.CS
061                                                && containsNullConditionalOperatorCSharp(literalTokens, variableNames, scopedTypeLookup)) {
062                                        continue;
063                                }
064
065                                addIndexOfTokenSequences(literalIndexes, literalTokens, ETokenType.EQEQ, ETokenType.BOOLEAN_LITERAL);
066                                addIndexOfTokenSequences(literalIndexes, literalTokens, ETokenType.NOTEQ, ETokenType.BOOLEAN_LITERAL);
067                                addIndexOfTokenSequences(literalIndexes, literalTokens, ETokenType.BOOLEAN_LITERAL, ETokenType.EQEQ);
068                                addIndexOfTokenSequences(literalIndexes, literalTokens, ETokenType.BOOLEAN_LITERAL, ETokenType.NOTEQ);
069
070                                for (Integer literalIndex : literalIndexes) {
071                                        createFinding("Avoid comparisons to boolean literals", literalTokens.get(literalIndex));
072                                }
073                        }
074                }
075        }
076
077        /**
078         * Checks if the null conditional operator '?' in C# is used. Either it is used
079         * directly in the condition or the variable used in the condition was declared
080         * by using the '?' operator. If so, return true. Otherwise, return false.
081         */
082        private static boolean containsNullConditionalOperatorCSharp(List<IToken> tokens, Set<String> variableNames,
083                        ScopedTypeLookup scopedTypeLookup) {
084
085                // check if the operator '?' is used in the condition statement
086                if (TokenStreamUtils.contains(tokens, ETokenType.QUESTION)) {
087                        return true;
088                }
089
090                // also check if the variable in the condition was declared using the '?'
091                // operator. If so, return true
092                for (String variableName : variableNames) {
093                        if (variableIsUsedInLiteral(tokens, variableName)) {
094                                TypedVariable typedVariable = scopedTypeLookup.getTypeInfo(variableName);
095
096                                // Check if unknown nullable type like CustomClass? or Nullable<X>
097                                UnmodifiableList<IToken> declaringEntityOfVariableTokens = typedVariable.getDeclaringEntity()
098                                                .ownStartTokens();
099
100                                if (typedVariable.getTypeName().equals("Nullable")) {
101                                        return true;
102                                }
103
104                                if (typedVariable.getTypeName().equals(TypedVariable.UNKNOWN_TYPE_NAME)
105                                                && TokenStreamUtils.contains(declaringEntityOfVariableTokens, ETokenType.QUESTION)) {
106                                        return true;
107                                }
108                        }
109                }
110                return false;
111        }
112
113        /**
114         * Checks if the passed variable is contained in the list of tokens.
115         */
116        private static boolean variableIsUsedInLiteral(List<IToken> literalTokens, String variableName) {
117                return TokenStreamUtils.getTokenByTypeAndText(literalTokens, variableName,
118                                CollectionUtils.asHashSet(ETokenType.IDENTIFIER)) != null;
119        }
120
121        private static boolean containsEqualityCheckAgainstBooleanLiteral(List<IToken> tokens) {
122                return TokenStreamUtils.containsSequence(tokens, ETokenType.EQEQ, ETokenType.BOOLEAN_LITERAL)
123                                || TokenStreamUtils.containsSequence(tokens, ETokenType.NOTEQ, ETokenType.BOOLEAN_LITERAL)
124                                || TokenStreamUtils.containsSequence(tokens, ETokenType.BOOLEAN_LITERAL, ETokenType.EQEQ)
125                                || TokenStreamUtils.containsSequence(tokens, ETokenType.BOOLEAN_LITERAL, ETokenType.NOTEQ);
126        }
127
128        /** Adds the index of token sequences as specified */
129        private static void addIndexOfTokenSequences(List<Integer> indexes, List<IToken> tokens, ETokenType... tokenTypes) {
130                indexes.addAll(TokenStreamUtils.firstTokenOfTypeSequences(tokens, 0, tokenTypes));
131        }
