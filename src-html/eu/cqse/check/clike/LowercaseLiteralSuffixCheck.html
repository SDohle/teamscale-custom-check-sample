009import static eu.cqse.check.framework.scanner.ETokenType.FLOATING_POINT_LITERAL;
010import static eu.cqse.check.framework.scanner.ETokenType.INTEGER_LITERAL;

012import java.util.EnumSet;

014import eu.cqse.check.cpp.misra.TokenFilterCheckBase;
015import eu.cqse.check.framework.core.ACheck;
016import eu.cqse.check.framework.core.CheckGroupNames;
017import eu.cqse.check.framework.core.ECheckParameter;

021import eu.cqse.check.framework.scanner.ETokenType;
022import eu.cqse.check.framework.scanner.IToken;

024/**
025 * Checks that no lowercase 'L' is used as literal suffix for integer and
026 * floating-point literals, because 1 (one) and l (lowercase L) can be easily
027 * confused.
028 */
029@ACheck(name = "Lowercase literal suffix", description = LowercaseLiteralSuffixCheck.DESCRIPTION
030                + ". Because lowercase `L` could be mistaken for `1`, `L` should be used to provide clarification.", groupName = CheckGroupNames.READABILITY, defaultEnablement = EFindingEnablement.YELLOW, languages = {
031                                ELanguage.CPP, ELanguage.JAVA,
032                                ELanguage.OBJECTIVE_C }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
033@ACheckRuleset(ruleId = "7.3", ruleset = MISRA)
034public class LowercaseLiteralSuffixCheck extends TokenFilterCheckBase {
035
036        /** Only INTEGER_LITERAL and FLOATING_POINT_LITERAL tokens should be checked. */
037        private static final EnumSet<ETokenType> CHECKED_TOKENTYPES = EnumSet.of(INTEGER_LITERAL, FLOATING_POINT_LITERAL);
038
039        public static final String DESCRIPTION = "Usage of lowercase `L` in literal suffixes is not allowed";
040
041        /** {@inheritDoc} */
042        @Override
043        protected String getFindingMessage(IToken token) {
044                return DESCRIPTION;
045        }
046
047        /** {@inheritDoc} */
048        @Override
049        protected boolean isValidToken(IToken token) {
050                if (!CHECKED_TOKENTYPES.contains(token.getType())) {
051                        return true;
052                }
053                return !token.getText().contains("l");
054        }
055}
