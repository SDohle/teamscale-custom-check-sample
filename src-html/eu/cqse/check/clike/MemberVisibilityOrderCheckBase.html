001package eu.cqse.check.clike;
002
003import static eu.cqse.check.framework.shallowparser.SubTypeNames.CLASS;
004
005import java.util.List;
006import java.util.stream.Collectors;
007
008import org.conqat.lib.commons.collections.Pair;
009import org.conqat.lib.commons.collections.PairList;
010
011import eu.cqse.check.framework.core.CheckException;
012import eu.cqse.check.framework.core.CheckImplementationBase;
013import eu.cqse.check.framework.scanner.ETokenType;
014import eu.cqse.check.framework.scanner.IToken;
015import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
016
017/**
018 * Base class for checking the ordering of class members regarding their
019 * visibility or access modifier.
020 */
021public abstract class MemberVisibilityOrderCheckBase extends CheckImplementationBase {
022
023        /** Base name of the check. */
024        protected static final String CHECK_BASE_NAME = "Ordering of members by visibility";
025
026        /** Base description of the check. */
027        public static final String CHECK_BASE_DESC = "In every class, the order of the access modifier should be as follows: ";
028
029        /** List of how the modifier should be ordered in the class */
030        protected List<ETokenType> MODIFIER_ORDER = getOderedModifier();
031
032        /** Returns a list with the order of all available modifier. */
033        protected abstract List<ETokenType> getOderedModifier();
034
035        /** {@inheritDoc} */
036        @Override
037        public void execute() throws CheckException {
038                for (ShallowEntity entity : select(getClassEntitiesXPath())) {
039                        PairList<ETokenType, ShallowEntity> members = getMembersVisibilityOrder(entity);
040                        int currentModifierIndex = 0;
041                        for (Pair<ETokenType, ShallowEntity> member : members) {
042                                ETokenType modifier = member.getFirst();
043                                int modifierIndex = MODIFIER_ORDER.indexOf(modifier);
044                                if (modifierIndex >= currentModifierIndex) {
045                                        currentModifierIndex = modifierIndex;
046                                } else {
047                                        createFinding(getFindingsMessage(member, MODIFIER_ORDER.get(currentModifierIndex)),
048                                                        member.getSecond());
049                                }
050                        }
051                }
052        }
053
054        /** Returns the message for the finding. */
055        protected String getFindingsMessage(Pair<ETokenType, ShallowEntity> member, ETokenType precedingModifier) {
056                return member.getFirst().toString().toLowerCase() + " member `" + member.getSecond().getName()
057                                + "` should not occur after " + precedingModifier.toString().toLowerCase() + " member. Expected order: "
058                                + getOrderString(MODIFIER_ORDER);
059        }
060
061        /** Returns all visibility tokens of all members in the given class entity. */
062        protected PairList<ETokenType, ShallowEntity> getMembersVisibilityOrder(ShallowEntity classEntity)
063                        throws CheckException {
064                List<ShallowEntity> members = select(classEntity, getMemberXPathSelection());
065                PairList<ETokenType, ShallowEntity> pairList = new PairList<ETokenType, ShallowEntity>();
066                for (ShallowEntity member : members) {
067                        pairList.add(getAccessModifierFromMember(member), member);
068                }
069                return pairList;
070        }
071
072        /**
073         * Returns the access modifier of the given member. If no modifier can be found
074         * <code>null</code> is returned instead.
075         */
076        protected ETokenType getAccessModifierFromMember(ShallowEntity member) {
077                List<IToken> tokens = member.ownStartTokens();
078                if (tokens.size() > 0) {
079                        ETokenType firstType = tokens.get(0).getType();
080                        if (MODIFIER_ORDER.contains(firstType)) {
081                                return firstType;
082                        }
083                }
084                return null;
085        }
086
087        /**
088         * XPath selection of members inside of a class.
089         * <p>
090         * Note that root node for the xpath is a class and not the document.
091         */
092        protected String getMemberXPathSelection() {
093                return "./ATTRIBUTE";
094        }
095
096        /** Returns the entities of all classes in the file. */
097        protected String getClassEntitiesXPath() {
098                return "//TYPE[subtype('" + CLASS + "')]";
099        }
100
101        /**
102         * Returns the the elements of the given ETokenType list as a String with all
103         * elements separated by a comma.
104         */
105        protected static String getOrderString(List<ETokenType> list) {
106                return list.stream().map(x -> x.toString().toLowerCase()).collect(Collectors.joining(", "));
107        }
108
109}
