030/**
031 * Checks whether `throw` keyword is missing in the statement with new
032 * exception.
033 */
034@ACheck(name = "`New` exception is created but not thrown", description = "Keyword `throw` before `new **Exception(*)` is missing. Either throw the exception or remove the unnecessary constructor.", groupName = CheckGroupNames.EXCEPTION_HANDLING, defaultEnablement = EFindingEnablement.RED, languages = {
035                ELanguage.JAVA, ELanguage.CS, ELanguage.OBJECTIVE_C }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
036public class MissingThrowBeforeNewExceptionCheck extends EntityCheckBase {
037
038        /** Pattern for exception creation, e.g. new Exception() */
039        public static final TokenPattern EXCEPTION_CREATION_PATTERN = new TokenPattern().notPrecededBy(RETURN).sequence(NEW)
040                        .repeated(IDENTIFIER, DOT).regex("Exception").sequence(LPAREN).skipTo(RPAREN).notFollowedBy(DOT);
041
042        /** {@inheritDoc} */
043        @Override
044        protected String getXPathSelectionString() {
045                return "//STATEMENT[not(subtype('" + LAMBDA_EXPRESSION + "'))]";
046        }
047
048        /** {@inheritDoc} */
049        @Override
050        protected void processEntity(ShallowEntity entity) throws CheckException {
051                UnmodifiableList<IToken> statementStartTokens = entity.ownStartTokens();
052                TokenPatternMatch sequenceMatch = EXCEPTION_CREATION_PATTERN.matchAtStartOf(statementStartTokens);
053                if (sequenceMatch != null) {
054                        createFinding("Keyword `throw` is missing before `new " + statementStartTokens.get(1).getText() + "`",
055                                        entity.getStartLine());
056                }
057        }
058}
