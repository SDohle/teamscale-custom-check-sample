006package eu.cqse.check.clike;
007
008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009import static eu.cqse.check.framework.scanner.ETokenType.INTEGER_LITERAL;
010
011import java.util.regex.Pattern;
012
013import eu.cqse.check.cpp.misra.TokenFilterCheckBase;
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckGroupNames;
016import eu.cqse.check.framework.core.ECheckParameter;
017import eu.cqse.check.framework.core.EFindingEnablement;
018import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
019import eu.cqse.check.framework.scanner.ELanguage;
020import eu.cqse.check.framework.scanner.IToken;
021
022/**
023 * Checks that no octal constant is used as integer literal except the special
024 * case 0.
025 */
026@ACheck(name = "Octal constant usage", description = "Usage of octal constants is not allowed, because it may be misinterpreted as decimal constant. Use decimal or hexadecimal format to avoid confusion.", groupName = CheckGroupNames.READABILITY, defaultEnablement = EFindingEnablement.YELLOW, languages = {
027                ELanguage.CPP, ELanguage.JAVA, ELanguage.OBJECTIVE_C }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
028@ACheckRuleset(ruleId = "7.1", ruleset = MISRA)
029public class OctalContstantCheck extends TokenFilterCheckBase {
030
031        /**
032         * Pattern for non-zero octal constants. We check for 0 followed by a non-0
033         * digit (other characters would be suffixes).
034         */
035        private static final Pattern NON_ZERO_OCTAL_PATTERN = Pattern.compile("^0+[1-9]");
036
037        /** {@inheritDoc} */
038        @Override
039        protected String getFindingMessage(IToken token) {
040                String text = token.getText();
041                return "Usage of octal constant `" + text + "` is not allowed, because it may be misinterpreted as `"
042                                + text.substring(1) + '`';
043        }
044
045        /** {@inheritDoc} */
046        @Override
047        protected boolean isValidToken(IToken token) {
048                return token.getType() != INTEGER_LITERAL || !NON_ZERO_OCTAL_PATTERN.matcher(token.getText()).find();
049        }
050}
