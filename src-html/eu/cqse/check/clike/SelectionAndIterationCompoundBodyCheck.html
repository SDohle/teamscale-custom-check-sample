054/**
055 * Check creates finding if enclosing braces are missed to a compound body
056 */
057@ACheck(name = "Missing braces for block statements", description = "Creates findings for statement blocks in loops "
058                + "and conditional statements that are not enclosed in braces - '{' and '}'.", groupName = CheckGroupNames.FORMATTING, defaultEnablement = EFindingEnablement.YELLOW, languages = {
059                                ADA, CPP, CS, JAVA, JAVASCRIPT, XTEND, SWIFT, RUST, PHP, OPEN_CL, GOSU, GROOVY, DELPHI, OSCRIPT,
060                                OBJECTIVE_C }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
061@ACheckRuleset(ruleId = "15.6", ruleset = MISRA)
062public class SelectionAndIterationCompoundBodyCheck extends EntityCheckBase {
063
064        /** {@ConQAT.Doc} */
065        @ACheckOption(name = "Ifs w/o braces if a return follows", description = "Allow ifs without braces if a return follows on the same line")
066        private boolean allowIfs = false;
067
068        /**
069         * Set of start token types indicating statements to be skipped. This is needed
070         * to handle C preprocessed code correctly.
071         */
072        private static final Set<ETokenType> NEGATIVE_START_SET = EnumSet.of(LPAREN, IDENTIFIER);
073
074        private static final String IF_CONDITION_SUB_TYPE_NAMES_CONDITION = "subtype('"
075                        + join("') | subtype('", IF_CONDITION_SUB_TYPE_NAMES) + "')";
076        private static final String LOOP_SUB_TYPE_NAMES_CONDITION = "subtype('"
077                        + join("') | subtype('", LOOP_SUB_TYPE_NAMES) + "')";
078
079        /** {@inheritDoc} */
080        @Override
081        protected String getXPathSelectionString() {
082                return "//STATEMENT[" + IF_CONDITION_SUB_TYPE_NAMES_CONDITION + " | " + LOOP_SUB_TYPE_NAMES_CONDITION + "]";
083        }
084
085        /** {@inheritDoc} */
086        @Override
087        protected void processEntity(ShallowEntity entity) throws CheckException {
088                if (!entity.hasChildren()) {
089                        // EmptyBlocks are checked elsewhere
090                        return;
091                }
092                // ignore if preprocessor statements in C++ are found
093                if (entity.getChildren().get(0).getType() == EShallowEntityType.META) {
094                        return;
095                }
096                // protect from out of bounds
097                if (entity.getChildren().get(0).getRelativeStartTokenIndex() == 0) {
098                        return;
099                }
100
101                List<IToken> startTokens = entity.ownStartTokens();
102                CCSMAssert.isTrue(0 < startTokens.size(), "No tokens to check.");
103
104                if (NEGATIVE_START_SET.contains(startTokens.get(0).getType())) {
105                        return;
106                }
107
108                if (allowIfs && entity.getChildren().size() == 1 && checkReturnStatement(entity.getChildren().get(0))) {
109                        return;
110                }
111
112                if (checkLastTokenType(startTokens, EnumSet.of(RPAREN, ELSE, DO))) {
113                        createFinding(getStatementName(startTokens) + " statement without braces", entity.getStartLine());
114                }
115        }
116
117        /** Checks whether the last token has some type contained by types */
118        private static boolean checkLastTokenType(List<IToken> tokens, EnumSet<ETokenType> types) {
119                return types.contains(tokens.get(tokens.size() - 1).getType());
120        }
121
122        /**
123         * Checks whether the entity is a return statement only and is in the same line
124         * like the parent entity
125         */
126        private static boolean checkReturnStatement(ShallowEntity entity) {
127                return !entity.hasChildren() && entity.ownStartTokens().get(0).getType().equals(RETURN)
128                                && entity.getParent().getStartLine() == entity.getStartLine()
129                                && entity.getStartLine() == entity.getEndLine();
130        }
131
132        /**
133         * Receive the name of the used statement to generate a more specific finding
134         * message.
135         */
136        private static String getStatementName(List<IToken> tokens) {
137                String statementName = "block";
138                int indexFirstLParen = TokenStreamUtils.firstTokenOfType(tokens, ETokenType.LPAREN);
139                int indexElseOrDo = TokenStreamUtils.firstTokenOfType(tokens, ETokenType.ELSE, ETokenType.DO);
140                if (indexFirstLParen > 0) {
141                        statementName = tokens.get(indexFirstLParen - 1).getText();
142                } else if (indexElseOrDo != TokenStreamUtils.NOT_FOUND) {
143                        statementName = tokens.get(indexElseOrDo).getText();
144                }
145                return statementName;
146        }
147}
