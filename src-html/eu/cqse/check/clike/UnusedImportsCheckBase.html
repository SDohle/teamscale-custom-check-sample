006package eu.cqse.check.clike;
007
008import org.conqat.lib.commons.collections.ListMap;
009
010import eu.cqse.check.framework.core.CheckException;
011import eu.cqse.check.framework.core.CheckImplementationBase;
012import eu.cqse.check.framework.core.phase.ECodeViewOption;
013import eu.cqse.check.framework.scanner.ETokenType.ETokenClass;
014import eu.cqse.check.framework.scanner.IToken;
015import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
016import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
017
018/** Base class for checks analyzing unused and duplicated imports. */
019public abstract class UnusedImportsCheckBase extends CheckImplementationBase {
020
021        /** {@inheritDoc} */
022        @Override
023        public void execute() throws CheckException {
024
025                // Stores for type names the import entities. The keys are unqualified type
026                // names
027                ListMap<String, ShallowEntity> importsBySimpleNames = new ListMap<>();
028
029                // Stores for type names the import entities. The keys are fully qualified type
030                // names
031                ListMap<String, ShallowEntity> importsByFullNames = new ListMap<>();
032
033                // Stores the duplicates in imports.
034                ListMap<String, ShallowEntity> duplicates = new ListMap<>();
035
036                // As imports are top-level it is sufficient to only process direct children.
037                // Get all imports first
038                context.getRootEntity(ECodeViewOption.UNFILTERED_PREPROCESSED).getChildren().stream()
039                                .filter(entity -> entity.getType() == EShallowEntityType.META && isImportStatement(entity))
040                                .forEach(entity -> recordImport(entity, importsBySimpleNames, importsByFullNames, duplicates));
041                // Then check which imports are used
042                context.getRootEntity(ECodeViewOption.UNFILTERED_PREPROCESSED).getChildren().stream()
043                                .filter(entity -> !(entity.getType() == EShallowEntityType.META && isImportStatement(entity)))
044                                .forEach(entity -> processEntity(entity, importsBySimpleNames));
045
046                context.getTokens(ECodeViewOption.UNFILTERED_PREPROCESSED).stream()
047                                .filter(token -> token.getType().getTokenClass() == ETokenClass.COMMENT)
048                                .forEach(token -> processComment(token, importsBySimpleNames));
049
050                createFindings(importsBySimpleNames, duplicates);
051        }
052
053        /**
054         * Recursively process all child entities and marks all identifiers found in the
055         * given token stream as used types.
056         */
057        protected abstract void processEntity(ShallowEntity entity, ListMap<String, ShallowEntity> imports);
058
059        /**
060         * Returns whether the entity's subtype {@link ShallowEntity#getSubtype()}
061         * equals the import type the check is written for.
062         */
063        protected abstract boolean isImportStatement(ShallowEntity entity);
064
065        /** Records an import statement. */
066        private void recordImport(ShallowEntity entity, ListMap<String, ShallowEntity> importsBySimpleNames,
067                        ListMap<String, ShallowEntity> importsByFullNames, ListMap<String, ShallowEntity> duplicates) {
068                String unqualifiedTypeName = getSimpleNameFromImport(entity);
069                String fullTypeName = getFullyQualifiedNameFromImport(entity);
070                if (unqualifiedTypeName != null) {
071                        if (!importsByFullNames.getKeys().contains(fullTypeName)) {
072                                importsBySimpleNames.add(unqualifiedTypeName, entity);
073                                importsByFullNames.add(fullTypeName, entity);
074                        } else {
075                                duplicates.add(unqualifiedTypeName, entity);
076                        }
077                }
078        }
079
080        /** Processes a single comment token and marks included types as used. */
081        protected abstract void processComment(IToken token, ListMap<String, ShallowEntity> imports);
082
083        /** Marks a type as used by removing the import from the imports. */
084        protected void useType(String typeName, ListMap<String, ShallowEntity> imports) {
085                imports.removeCollection(typeName);
086        }
087
088        /** Create findings for unused and duplicated imports. */
089        private void createFindings(ListMap<String, ShallowEntity> imports, ListMap<String, ShallowEntity> duplicates)
090                        throws CheckException {
091                createFindings(imports, "Unused");
092                createFindings(duplicates, "Duplicated");
093        }
094
095        /** Create findings for unused/duplicated imports. */
096        private void createFindings(ListMap<String, ShallowEntity> map, String description) throws CheckException {
097                for (String typeName : map.getKeys()) {
098                        for (ShallowEntity entity : map.getCollection(typeName)) {
099                                createFindingOnUnfilteredCode(
100                                                description + " import: `" + getFullyQualifiedNameFromImport(entity) + "`", entity);
101                        }
102                }
103        }
104
105        /**
106         * Returns the simple name for an import statement (e.g. "String" for
107         * java.lang.String). If no specific type is found the method returns
108         * <code> null </code>.
109         */
110        protected abstract String getSimpleNameFromImport(ShallowEntity entity);
111
112        /** Returns the fully qualified name for an import statement. */
113        protected abstract String getFullyQualifiedNameFromImport(ShallowEntity entity);
114
115}
