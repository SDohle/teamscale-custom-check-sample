006package eu.cqse.check.clike;
007
008import java.util.List;
009
010import org.conqat.lib.commons.string.StringUtils;
011
012import eu.cqse.check.framework.core.ACheck;
013import eu.cqse.check.framework.core.CheckException;
014import eu.cqse.check.framework.core.CheckGroupNames;
015import eu.cqse.check.framework.core.CheckImplementationBase;
016import eu.cqse.check.framework.core.EFindingEnablement;
017import eu.cqse.check.framework.core.option.ACheckOption;
018import eu.cqse.check.framework.core.phase.ECodeViewOption;
019import eu.cqse.check.framework.scanner.ELanguage;
020
021/**
022 * This check allows for reporting various invisible whitespace anomalies
023 */
024@ACheck(name = "Consistent whitespace usage", description = "Checks that usage of whitespace conforms to formatting guidelines  \n"
025                + "Depending on the set options this check marks:  \n" + "* the usage of tabulator characters  \n"
026                + "* spaces behind the last visible character of a line  \n"
027                + "* files that do not end with a linebreak", groupName = CheckGroupNames.FORMATTING, defaultEnablement = EFindingEnablement.OFF, languages = {
028                                ELanguage.JAVA, ELanguage.CPP, ELanguage.JAVASCRIPT, ELanguage.CS,
029                                ELanguage.OBJECTIVE_C }, parameters = {})
030public class WhitespaceFormattingCheck extends CheckImplementationBase {
031
032        @ACheckOption(name = "Do not use tabs", description = "Mark use of tabulator characters")
033        private boolean checkTabs = true;
034
035        @ACheckOption(name = "Do not use trailing spaces", description = "Mark spaces behind the last visible character of a line. These spaces are not visible in editors.")
036        private boolean checkSpaces = true;
037
038        @ACheckOption(name = "File must end with a linebreak", description = "Files that do not end with a linebreak can confuse programs")
039        private final boolean checkEndLineBreak = true;
040
041        @Override
042        public void execute() throws CheckException {
043                if (checkTabs || checkSpaces) {
044                        List<String> lines = StringUtils
045                                        .splitLinesAsList(context.getTextContent(ECodeViewOption.ETextViewOption.FILTERED_CONTENT));
046                        for (int i = 0; i < lines.size(); ++i) {
047                                if (checkTabs && lines.get(i).contains("\t")) {
048                                        createFinding("Tab Character violates Coding Guideline", i);
049                                }
050                                if (checkSpaces && lines.get(i).endsWith(" ")) {
051                                        createFinding("Trailing Whitespace in Line", i);
052                                }
053                        }
054                }
055                if (checkEndLineBreak
056                                && !context.getTextContent(ECodeViewOption.ETextViewOption.FILTERED_CONTENT).endsWith("\n")) {
057                        createFinding("No Linebreak at End of File");
058                }
059        }
060}
