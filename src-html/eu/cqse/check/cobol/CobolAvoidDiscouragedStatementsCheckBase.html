006package eu.cqse.check.cobol;
007
008import java.util.Arrays;
009import java.util.List;
010
011import eu.cqse.check.framework.core.CheckException;
012import eu.cqse.check.framework.core.CheckImplementationBase;
013import eu.cqse.check.framework.core.phase.ECodeViewOption;
014import eu.cqse.check.framework.scanner.ETokenType;
015import eu.cqse.check.framework.scanner.IToken;
016import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
017
018/**
019 * Check whether discouraged statements are not used in Cobol.
020 */
021public abstract class CobolAvoidDiscouragedStatementsCheckBase extends CheckImplementationBase {
022        /** The TokenType sequences that we scan for here. */
023        private List<ETokenType[]> discouragedSequences = Arrays.asList();
024
025        /** Constructor. */
026        public CobolAvoidDiscouragedStatementsCheckBase(List<ETokenType[]> discouragedSequences) {
027                this.discouragedSequences = discouragedSequences;
028        }
029
030        /** {@inheritDoc} */
031        @Override
032        public void execute() throws CheckException {
033                List<IToken> tokens = context.getTokens(ECodeViewOption.FILTERED);
034                for (ETokenType[] discouragedSequence : discouragedSequences) {
035                        int findingIndex = -1;
036                        while ((findingIndex = TokenStreamUtils.firstTokenOfTypeSequence(tokens, findingIndex + 1,
037                                        discouragedSequence)) != TokenStreamUtils.NOT_FOUND) {
038                                List<IToken> statementTokens = tokens.subList(findingIndex, findingIndex + discouragedSequence.length);
039                                createDiscouragedStatementFinding(statementTokens);
040                        }
041                }
042        }
043
044        /**
045         * Generates a finding for the given discouraged-token sequence.
046         */
047        protected abstract void createDiscouragedStatementFinding(List<IToken> statementTokens) throws CheckException;
048}
