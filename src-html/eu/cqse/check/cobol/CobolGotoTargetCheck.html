017package eu.cqse.check.cobol;
018
019import static eu.cqse.check.framework.scanner.ETokenType.GO;
020import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
021import static eu.cqse.check.framework.scanner.ETokenType.TO;
022import static eu.cqse.check.framework.shallowparser.languages.cobol.CobolShallowParser.PARAGRAPH_SUBTYPE_NAME;
023import static eu.cqse.check.framework.shallowparser.languages.cobol.CobolShallowParser.SECTION_SUBTYPE_NAME;
024
025import java.util.List;
026import java.util.OptionalInt;
027
028import org.conqat.lib.commons.collections.CollectionUtils;
029import org.conqat.lib.commons.collections.PairList;
030
031import eu.cqse.check.framework.core.ACheck;
032import eu.cqse.check.framework.core.CheckException;
033import eu.cqse.check.framework.core.CheckGroupNames;
034import eu.cqse.check.framework.core.CheckImplementationBase;
035import eu.cqse.check.framework.core.ECheckParameter;
036import eu.cqse.check.framework.core.EFindingEnablement;
037import eu.cqse.check.framework.core.phase.ECodeViewOption;
038import eu.cqse.check.framework.scanner.ELanguage;
039import eu.cqse.check.framework.scanner.IToken;
040import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
041import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
042import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
043
044/**
045 * Checks whether GO TO statements in Cobol always target the last paragraph of
046 * their section. Also checks whether there is a paragraph with the target name
047 * in the section.
048 */
049@ACheck(name = "GO TO not targeting final paragraph of section", description = "GO TO statements should target only the final paragraph of a section.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
050                ELanguage.COBOL }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
051public class CobolGotoTargetCheck extends CheckImplementationBase {
052
053        /** {@inheritDoc} */
054        @Override
055        public void execute() throws CheckException {
056                List<IToken> tokens = context.getTokens(ECodeViewOption.FILTERED);
057                List<Integer> gotoIndices = TokenStreamUtils.firstTokenOfTypeSequences(tokens, 0, GO, TO, IDENTIFIER);
058                if (gotoIndices.isEmpty()) {
059                        return;
060                }
061                PairList<Integer, String> paragraphs = getParagraphs(context.getRootEntity(ECodeViewOption.FILTERED));
062                List<Integer> sectionIndices = getSectionIndices(context.getRootEntity(ECodeViewOption.FILTERED));
063                for (int gotoIndex : gotoIndices) {
064                        String gotoLabelName = tokens.get(gotoIndex + 2).getText().toLowerCase();
065
066                        OptionalInt nextSectionStartIndex = getNextSectionIndex(gotoIndex, sectionIndices);
067                        OptionalInt previousSectionStartIndex = getPreviousSectionIndex(gotoIndex, sectionIndices);
068
069                        PairList<Integer, String> paragraphsInSection = getEntriesBetween(previousSectionStartIndex,
070                                        nextSectionStartIndex, paragraphs);
071                        if (paragraphsInSection.isEmpty()) {
072                                continue;
073                        }
074                        String lastParagraphLabel = CollectionUtils.getLast(paragraphsInSection.extractSecondList());
075                        if (paragraphsInSection.extractSecondList().contains(gotoLabelName)
076                                        && !lastParagraphLabel.equals(gotoLabelName)) {
077                                createFinding("GO TO does not target last paragraph \"" + lastParagraphLabel + "\" of current section",
078                                                tokens.get(gotoIndex), tokens.get(gotoIndex + 2));
079                        }
080                }
081        }
082
083        /**
084         * Returns the PairList entries which have a first value (the token index here)
085         * between beginIndex and endIndex. If beginIndex or endIndex are
086         * Optional.empty(), 0 or Integer.max is used, respectively.
087         */
088        private static PairList<Integer, String> getEntriesBetween(OptionalInt previousSectionStartIndex,
089                        OptionalInt nextSectionStartIndex, PairList<Integer, String> paragraphs) {
090                return paragraphs.stream()
091                                .filter(pair -> (!previousSectionStartIndex.isPresent()
092                                                || previousSectionStartIndex.getAsInt() < pair.getFirst())
093                                                && (!nextSectionStartIndex.isPresent() || pair.getFirst() < nextSectionStartIndex.getAsInt()))
094                                .collect(PairList.toPairList());
095        }
096
097        /**
098         * Returns the smallest index in the given list that is greater than the given
099         * index or Optional.empty if there is no such index in the list.
100         */
101        private static OptionalInt getNextSectionIndex(int index, List<Integer> sectionIndices) {
102                return sectionIndices.stream().mapToInt(Integer::intValue).filter(i -> i > index).min();
103        }
104
105        /**
106         * Returns the largest index in the given list that is smaller than the given
107         * index or Optional.empty if there is no such index in the list.
108         */
109        private static OptionalInt getPreviousSectionIndex(int index, List<Integer> sectionIndices) {
110                return sectionIndices.stream().mapToInt(Integer::intValue).filter(i -> i < index).max();
111        }
112
113        /**
114         * Returns the indices of Section tokens in the tokens of the given shallow
115         * entity.
116         */
117        private List<Integer> getSectionIndices(ShallowEntity rootEntity) throws CheckException {
118                List<ShallowEntity> sectionEntities = select(rootEntity,
119                                "//*[type('" + EShallowEntityType.METHOD + "') and subtype('" + SECTION_SUBTYPE_NAME + "')]");
120                return CollectionUtils.map(sectionEntities, ShallowEntity::getStartTokenIndex);
121        }
122
123        /**
124         * Returns a {@link PairList} of the token indices and names of paragraphs in
125         * the given shallow entity.
126         */
127        private PairList<Integer, String> getParagraphs(ShallowEntity rootEntity) throws CheckException {
128                PairList<Integer, String> paragraphIndices = new PairList<>();
129                List<ShallowEntity> paragraphEntities = select(rootEntity,
130                                "//*[type('" + EShallowEntityType.STATEMENT + "') and subtype('" + PARAGRAPH_SUBTYPE_NAME + "')]");
131                for (ShallowEntity paragraph : paragraphEntities) {
132                        paragraphIndices.add(paragraph.getStartTokenIndex(), paragraph.getName());
133                }
134                return paragraphIndices;
135
136        }
