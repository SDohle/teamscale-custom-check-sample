001package eu.cqse.check.cpp;
002
003import java.util.List;
004
005import org.conqat.lib.commons.string.StringUtils;
006
007import eu.cqse.check.framework.core.ACheck;
008import eu.cqse.check.framework.core.CheckException;
009import eu.cqse.check.framework.core.CheckGroupNames;
010import eu.cqse.check.framework.core.CheckImplementationBase;
011import eu.cqse.check.framework.core.ECheckParameter;
012import eu.cqse.check.framework.core.EFindingEnablement;
013import eu.cqse.check.framework.core.option.ACheckOption;
014import eu.cqse.check.framework.scanner.ELanguage;
015import eu.cqse.check.framework.shallowparser.SubTypeNames;
016import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
017
018/** {@ConQAT.Doc} */
019@ACheck(name = "Avoid top level anonymous namespaces", description = "There should be no top level anonymous namespaces."
020                + "", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
021                                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
022public class AnonymousNamespaceCheck extends CheckImplementationBase {
023
024        /** {@ConQAT.Doc} */
025        @ACheckOption(name = "No anonymous namespace in header", description = "There should be no anonymous namespace in a header at all.")
026        private boolean noAnonymousNSInHeader = true;
027
028        /** {@inheritDoc} */
029        @Override
030        public void execute() throws CheckException {
031                boolean isHeader = HeaderFileCheckBase.isHeader(context.getUniformPath());
032
033                List<ShallowEntity> allNamespaces = select(
034                                "//MODULE[subtype('" + SubTypeNames.NAMESPACE + "') and not(name-matches('.+'))]");
035                for (ShallowEntity namespace : allNamespaces) {
036                        String name = namespace.getName();
037                        if (StringUtils.isEmpty(name)) {
038                                boolean topLevel = (namespace.getParent().getSubtype() == "document-root");
039                                if (noAnonymousNSInHeader && isHeader) {
040                                        createFinding("Headers should not contain anonymous namespaces", namespace);
041                                } else if (topLevel) {
042                                        createFinding("Avoid using top level anonymous namespaces", namespace);
043                                }
044                        }
045                }
046        }
047}
