017package eu.cqse.check.cpp;
018
019import java.util.ArrayList;
020import java.util.Arrays;
021import java.util.List;
022import java.util.Set;
023import java.util.stream.Collectors;
024
025import org.conqat.lib.commons.collections.CollectionUtils;
026
027import eu.cqse.check.base.EntityCheckBase;
028import eu.cqse.check.framework.core.ACheck;
029import eu.cqse.check.framework.core.CheckException;
030import eu.cqse.check.framework.core.CheckGroupNames;
031import eu.cqse.check.framework.core.ECheckParameter;
032import eu.cqse.check.framework.core.EFindingEnablement;
033import eu.cqse.check.framework.core.option.ACheckOption;
034import eu.cqse.check.framework.scanner.ELanguage;
035import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
036import eu.cqse.check.framework.typetracker.ITypeInfoExtractor;
037import eu.cqse.check.framework.typetracker.TypeInfoExtractorFactory;
038import eu.cqse.check.framework.typetracker.TypedVariable;
039import eu.cqse.check.framework.util.LanguageFeatureParser;
040
041/**
042 * {@ConQAT.Doc}
043 */
044@ACheck(name = "Do not use forbidden types (C++)", description = "Reports findings if a type from a user-defined list is "
045                + "used.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
046                                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE, ECheckParameter.TYPE_RESOLUTION })
047public class CppForbiddenTypesCheck extends EntityCheckBase {
048
049        /** {@ConQAT.Doc} */
050        @ACheckOption(name = "Forbidden types", description = "The list of types that are forbidden to use.")
051        private List<String> forbiddenTypes = new ArrayList<>();
052
053        /**
054         * The internally used type info extractor. As the C++ type info extractor
055         * does not make use of the passed root entities and logger, we can safely
056         * pass null.
057         */
058        private final ITypeInfoExtractor extractor = TypeInfoExtractorFactory.getTypeInfoExtractor(ELanguage.CPP, null,
059                        null);
060
061        /** {@inheritDoc} */
062        @Override
063        protected String getXPathSelectionString() {
064                return "//STATEMENT | //METHOD | //ATTRIBUTE";
065        }
066
067        /** {@inheritDoc} */
068        @Override
069        protected void processEntity(ShallowEntity entity) throws CheckException {
070                switch (entity.getType()) {
071                case METHOD:
072                        handleMethod(entity);
073                        break;
074                case ATTRIBUTE:
075                case STATEMENT:
076                        handleDeclaringEntity(entity);
077                        break;
078                }
079        }
080
081        /** Handles the given method entity. */
082        private void handleMethod(ShallowEntity entity) throws CheckException {
083                String returnTypeName = LanguageFeatureParser.CPP.getReturnType(entity);
084                handleDeclaringEntity(entity, returnTypeName);
085        }
086
087        /** Handles an entity that is declaring variables and thus using types. */
088        private void handleDeclaringEntity(ShallowEntity entity, String... additionalTypeNames) throws CheckException {
089                Set<String> importedNamespaces = context.getTypeResolution().getTypeLookup(entity).getImportedNamespaces();
090                List<TypedVariable> typedVariables = extractor.extract(entity);
091
092                List<String> otherNames = LanguageFeatureParser.CPP.getGenericTypeNames(entity.ownStartTokens());
093                otherNames.addAll(Arrays.asList(additionalTypeNames));
094
095                for (String name : reduceToPlainTypeSet(typedVariables, otherNames)) {
096                        checkTypeName(entity, importedNamespaces, name);
097                }
098        }
099
100        /** Checks whether the given type name is forbidden. */
101        private void checkTypeName(ShallowEntity entity, Set<String> importedNamespaces, String typeName)
102                        throws CheckException {
103                for (String forbiddenTypeName : forbiddenTypes) {
104                        if (LanguageFeatureParser.CPP.matchesFullQualifiedTypeName(typeName, importedNamespaces,
105                                        forbiddenTypeName)) {
106                                createFindingOnFirstLine("Type '" + forbiddenTypeName + "' must not be used.", entity);
107                        }
108                }
109        }
110
111        /**
112         * Reduces the given list of typed variables to a set of unique plain type
113         * names. If additional names are given they are converted to plain type
114         * names and added to the result set.
115         */
116        private static Set<String> reduceToPlainTypeSet(List<TypedVariable> typedVariables,
117                        List<String> additionalTypeNames) {
118                List<String> names = CollectionUtils.map(typedVariables, variable -> variable.getTypeName());
119                names.addAll(additionalTypeNames);
120                return names.stream().filter(name -> name != null).map(name -> LanguageFeatureParser.CPP.getPlainTypeName(name))
121                                .collect(Collectors.toSet());
122        }
123}
