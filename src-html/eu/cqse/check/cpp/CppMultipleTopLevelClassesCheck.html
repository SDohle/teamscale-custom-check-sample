017package eu.cqse.check.cpp;
018
019import java.util.List;
020
021import eu.cqse.check.framework.core.ACheck;
022import eu.cqse.check.framework.core.CheckException;
023import eu.cqse.check.framework.core.CheckGroupNames;
024import eu.cqse.check.framework.core.ECheckParameter;
025import eu.cqse.check.framework.core.EFindingEnablement;
026import eu.cqse.check.framework.scanner.ELanguage;
027import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
028
029/**
030 * {@ConQAT.Doc}
031 */
032@ACheck(name = "Multiple classes in one file (C++)", description = "Defining more than one class in a file makes "
033                + "finding the file for a given class very hard. Therefore each file should contain only one top-level "
034                + "class.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
035                                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
036public class CppMultipleTopLevelClassesCheck extends CppTopLevelClassCheckBase {
037
038        /** {@inheritDoc} */
039        @Override
040        protected void analyzeTopLevelTypes(List<ShallowEntity> topLevelTypeEntities) throws CheckException {
041                if (topLevelTypeEntities.size() > 1) {
042                        for (ShallowEntity entity : topLevelTypeEntities) {
043                                createFindingOnFirstLine("Multiple top-level class definitions in one file.", entity);
044                        }
045                }
046        }
047}
