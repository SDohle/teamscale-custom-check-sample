017package eu.cqse.check.cpp;
018
019import static eu.cqse.check.framework.scanner.ETokenType.PRIVATE;
020import static eu.cqse.check.framework.scanner.ETokenType.PROTECTED;
021import static eu.cqse.check.framework.scanner.ETokenType.PUBLIC;
022
023import eu.cqse.check.base.EntityCheckBase;
024import eu.cqse.check.framework.core.ACheck;
025import eu.cqse.check.framework.core.CheckException;
026import eu.cqse.check.framework.core.CheckGroupNames;
027import eu.cqse.check.framework.core.ECheckParameter;
028import eu.cqse.check.framework.core.EFindingEnablement;
029import eu.cqse.check.framework.scanner.ELanguage;
030import eu.cqse.check.framework.scanner.ETokenType;
031import eu.cqse.check.framework.shallowparser.SubTypeNames;
032import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
033import eu.cqse.check.framework.util.CppLanguageFeatureParser;
034
035/**
036 * {@ConQAT.Doc}
037 */
038@ACheck(name = "Public class attribute (C++)", description = "Classes should not have public attributes as this violates encapsulation. "
039                + "Instead access should be provided using public methods. This rule does not apply to constants and to attributes in "
040                + "structs.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
041                                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
042public class CppPublicClassAttributeCheck extends EntityCheckBase {
043
044        /** {@inheritDoc} */
045        @Override
046        protected String getXPathSelectionString() {
047                return "//TYPE[subtype('class')]";
048        }
049
050        /** {@inheritDoc} */
051        @Override
052        protected void processEntity(ShallowEntity entity) throws CheckException {
053                // private is the default visibility within C++ classes.
054                ETokenType currentVisibility = PRIVATE;
055                for (ShallowEntity child : entity.getChildren()) {
056                        switch (child.getType()) {
057                        case META:
058                                currentVisibility = updateVisibility(child, currentVisibility);
059                                break;
060                        case ATTRIBUTE:
061                                if (currentVisibility == PUBLIC && !CppLanguageFeatureParser.isConstant(child)) {
062                                        createFindingOnFirstLine("Attribute `" + child.getName() + "` should not be public", child);
063                                }
064                                break;
065                        default:
066                                // ignore other entity types
067                                break;
068                        }
069                }
070        }
071
072        /**
073         * Returns the new member visibility within a C++ class depending on the given
074         * meta entity and the old visibility.
075         */
076        private static ETokenType updateVisibility(ShallowEntity metaEntity, ETokenType oldVisibility) {
077                switch (metaEntity.getSubtype()) {
078                case SubTypeNames.PUBLIC:
079                        return PUBLIC;
080                case SubTypeNames.PROTECTED:
081                        return PROTECTED;
082                case SubTypeNames.PRIVATE:
083                        return PRIVATE;
084                default:
085                        return oldVisibility;
086                }
087        }
088}
