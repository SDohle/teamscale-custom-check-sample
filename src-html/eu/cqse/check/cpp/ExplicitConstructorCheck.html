001package eu.cqse.check.cpp;

003import static eu.cqse.check.framework.core.CheckGroupNames.BAD_PRACTICE;
004import static eu.cqse.check.framework.core.ECheckParameter.ABSTRACT_SYNTAX_TREE;
005import static eu.cqse.check.framework.core.EFindingEnablement.OFF;
006import static eu.cqse.check.framework.scanner.ETokenType.AND;
007import static eu.cqse.check.framework.scanner.ETokenType.ANDAND;
008import static eu.cqse.check.framework.scanner.ETokenType.EQ;
009import static eu.cqse.check.framework.scanner.ETokenType.EXPLICIT;
010import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
011import static eu.cqse.check.framework.shallowparser.SubTypeNames.CONSTRUCTOR_DECLARATION;
012import static eu.cqse.check.framework.util.LanguageFeatureParser.CPP;
013
014import java.util.List;
015import java.util.ListIterator;
016import java.util.regex.Pattern;
017
018import eu.cqse.check.framework.core.ACheck;
019import eu.cqse.check.framework.core.CheckException;
020import eu.cqse.check.framework.core.CheckImplementationBase;
021import eu.cqse.check.framework.scanner.ELanguage;
022import eu.cqse.check.framework.scanner.ETokenType;
023import eu.cqse.check.framework.scanner.IToken;
024import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
025import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
026import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
027
028/** {@ConQAT.Doc} */
029@ACheck(name = "Only constructor with one parameter should be explicit", description = "Constructors which only have one "
030                + "parameter must be declared explicit, with the exception of [copy](http://en.cppreference.com/w/cpp/language/copy_constructor)&"
031                + "[move](http://en.cppreference.com/w/cpp/language/move_constructor), as well as constructors with `std::initializer_list` as an argument."
032                + "Those should not be declared explicit."
033                + "", groupName = BAD_PRACTICE, defaultEnablement = OFF, languages = {
034                                ELanguage.CPP }, parameters = { ABSTRACT_SYNTAX_TREE })
035public class ExplicitConstructorCheck extends CheckImplementationBase {
036        public static final Pattern LIST_INIT_PATTERN = Pattern.compile(".*?(std::)?initializer_list.*?");
037
038        /** {@inheritDoc} */
039        @Override
040        public void execute() throws CheckException {
041                for (ShallowEntity constructor : select("//TYPE/METHOD[subtype('" + CONSTRUCTOR_DECLARATION + "')]")) {
042                        checkConstructor(constructor);
043                }
044        }
045
046        /** Checks if the given constructor should be explicit or not */
047        private void checkConstructor(ShallowEntity constructor) throws CheckException {
048                List<List<IToken>> parameterList = CPP.getSplitParameterTokens(constructor);
049                String className = constructor.getParent().getName();
050
051                boolean oneParameter = parameterList.size() == 1;
052                boolean isExplicit = TokenStreamUtils.contains(constructor.ownStartTokens(), EXPLICIT);
053                boolean isMove = isMoveConstructor(parameterList, className);
054                boolean isCopy = isCopyConstructor(parameterList, className);
055                boolean isListInit = isListInitializer(parameterList);
056                boolean isAllowedType = isListInit || isMove || isCopy;
057
058                // should be explicit
059                if (oneParameter && !(isExplicit || isAllowedType)) {
060                        createFinding("Constructor `" + className + "` should be declared explicit", constructor);
061                }
062
063                // shouldn't be explicit
064                if (isExplicit && isAllowedType) {
065                        createFinding("Constructor `" + className + "` shouldn't be declared explicit", constructor);
066                }
067        }
068
069        /** Checks if the */
070        private static boolean isListInitializer(List<List<IToken>> parameterList) {
071                if (parameterList.size() == 1) {
072                        String tokensString = TokenStreamTextUtils.concatTokenTexts(parameterList.get(0));
073                        return LIST_INIT_PATTERN.matcher(tokensString).matches();
074                }
075                return false;
076        }
077
078        /** Checks if the parameter given as a list of tokens has a default value. */
079        private static boolean hasDefaultValue(List<IToken> parameterToken) {
080                return TokenStreamUtils.contains(parameterToken, EQ);
081        }
082
083        /**
084         * Checks if the constructor is a so-called move constructor by analyzing its
085         * parameter.
086         * 
087         * More information on:
088         * http://en.cppreference.com/w/cpp/language/move_constructor
089         */
090        private static boolean isMoveConstructor(List<List<IToken>> parameterList, String className) {
091                return isMoveOrCopyConstructor(parameterList, className, ANDAND);
092        }
093
094        /**
095         * Checks if the constructor is a so-called move constructor by analyzing its
096         * parameter.
097         * 
098         * More information on:
099         * http://en.cppreference.com/w/cpp/language/copy_constructor
100         */
101        private static boolean isCopyConstructor(List<List<IToken>> parameterList, String className) {
102                return isMoveOrCopyConstructor(parameterList, className, AND);
103        }
104
105        /**
106         * Method that checks if the given parameterList is for a copy or move
107         * constructor. Because they are so similar and only differ in the first
108         * parameter and basically one token, we can do this in one method and just give
109         * the different token as a parameter.
110         */
111        private static boolean isMoveOrCopyConstructor(List<List<IToken>> parameterList, String className,
112                        ETokenType separator) {
113                ListIterator<List<IToken>> iterator = parameterList.listIterator();
114                boolean isMoveOrCopy = false;
115                if (iterator.hasNext()) {
116                        List<IToken> firstParameter = iterator.next();
117                        int sequenceIndex = TokenStreamUtils.firstTokenOfTypeSequence(firstParameter, 0, IDENTIFIER, separator);
118                        boolean sequenceFound = sequenceIndex != TokenStreamUtils.NOT_FOUND;
119                        isMoveOrCopy = sequenceFound && firstParameter.get(sequenceIndex).getText().equals(className);
120
121                        if (isMoveOrCopy) {
122                                while (iterator.hasNext()) {
123                                        isMoveOrCopy = hasDefaultValue(iterator.next());
124                                }
125                        }
126                }
127                return isMoveOrCopy;
128        }
129}
