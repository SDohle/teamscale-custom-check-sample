001package eu.cqse.check.cpp;
002
003import static eu.cqse.check.framework.scanner.ETokenType.PREPROCESSOR_INCLUDE;
004
005import java.io.File;
006import java.util.EnumSet;
007import java.util.List;
008import java.util.regex.Pattern;
009
010import org.conqat.lib.commons.filesystem.FileSystemUtils;
011
012import eu.cqse.check.framework.core.ACheck;
013import eu.cqse.check.framework.core.CheckException;
014import eu.cqse.check.framework.core.CheckGroupNames;
015import eu.cqse.check.framework.core.CheckImplementationBase;
016import eu.cqse.check.framework.core.ECheckParameter;
017import eu.cqse.check.framework.core.EFindingEnablement;
018import eu.cqse.check.framework.core.phase.ECodeViewOption;
019import eu.cqse.check.framework.scanner.ELanguage;
020import eu.cqse.check.framework.scanner.IToken;
021import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
022
023@ACheck(name = "Own header should be included first", description = "When including its own header, "
024                + "the include should be on the first position", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.OFF, languages = {
025                                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
026public class FirstIncludeOwnHeaderCheck extends CheckImplementationBase {
027        /**
028         * Regex which checks if an #include is for the filename defined by the
029         * placeholder %s. Note that you need to format the string and insert the actual
030         * filename.
031         */
032        private static final String FILE_NAME_INCLUDE = "#(\\s*)include(\\s+)(\"|<)%s(\\..*|>)";
033
034        @Override
035        public void execute() throws CheckException {
036                String fileName = FileSystemUtils
037                                .getFilenameWithoutExtension(new File(context.getRootEntity(ECodeViewOption.FILTERED).getName()));
038                Pattern fileIncludePattern = Pattern.compile(String.format(FILE_NAME_INCLUDE, fileName),
039                                Pattern.CASE_INSENSITIVE);
040
041                List<IToken> allIncludes = TokenStreamUtils.findAllTokens(context.getTokens(ECodeViewOption.FILTERED),
042                                EnumSet.of(PREPROCESSOR_INCLUDE));
043
044                if (allIncludes.size() == 0) {
045                        return;
046                }
047                for (IToken include : allIncludes.subList(1, allIncludes.size())) {
048                        if (fileIncludePattern.matcher(include.getText()).matches()) {
049                                createFinding("Own header file `" + fileName + "` should be included first", include);
050                                break;
051                        }
052                }
053        }
054}
