001package eu.cqse.check.cpp;
002
003import static eu.cqse.check.framework.core.CheckGroupNames.BAD_PRACTICE;
004import static eu.cqse.check.framework.core.ECheckParameter.ABSTRACT_SYNTAX_TREE;
005import static eu.cqse.check.framework.core.EFindingEnablement.OFF;
006import static eu.cqse.check.framework.scanner.ELanguage.CPP;
007import static eu.cqse.check.framework.scanner.ETokenType.FRIEND;
008import static eu.cqse.check.framework.shallowparser.SubTypeNames.CLASS;
009import static eu.cqse.check.framework.shallowparser.SubTypeNames.STRUCT;
010
011import java.util.ArrayList;
012import java.util.List;
013
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckImplementationBase;
017import eu.cqse.check.framework.shallowparser.SubTypeNames;
018import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
019import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
020
021/** 
022 * For an explanation on the friend concept, see here: 
023 * http://en.cppreference.com/w/cpp/language/friend 
024 */
025/** {@ConQAT.Doc} */
026@ACheck(name = "Friend declarations should be private", description = "Any member of a class preceeded by the "
027                + "`friend` keyword should be private.", groupName = BAD_PRACTICE, defaultEnablement = OFF, languages = {
028                                CPP }, parameters = { ABSTRACT_SYNTAX_TREE })
029public class FriendDeclarationCheck extends CheckImplementationBase {
030        /** {@inheritDoc} */
031        @Override
032        public void execute() throws CheckException {
033                // Only structs and classes can have friends. Better restrict it to them before
034                // the inevitable Shallow parser inadequacy comes along.
035                for (ShallowEntity classEntity : select("//TYPE[subtype('" + CLASS + "') or subtype('" + STRUCT + "')]")) {
036                        List<ShallowEntity> entities = collectPublicAndProtectedMembers(classEntity);
037                        checkForFriends(entities);
038                }
039        }
040
041        /**
042         * Collects every public and protected method and attribute in the given entity,
043         * which should either be a struct or a class.
044         */
045        private static List<ShallowEntity> collectPublicAndProtectedMembers(ShallowEntity classOrStructEntity) {
046                List<ShallowEntity> entities = new ArrayList<>();
047                // Outside of any access modifier the members in a class are private,
048                // but public in a struct
049                boolean add = classOrStructEntity.getSubtype().equals(STRUCT);
050                for (ShallowEntity entity : classOrStructEntity.getChildren()) {
051                        if (add) {
052                                entities.add(entity);
053                        }
054
055                        if (entity.getSubtype().matches(SubTypeNames.PUBLIC + "|" + SubTypeNames.PROTECTED)) {
056                                add = true;
057                        } else if (entity.getSubtype().equals(SubTypeNames.PRIVATE)) {
058                                add = false;
059                        }
060                }
061
062                return entities;
063        }
064
065        /**
066         * Checks the given list of shallow entities if any of them is a friend
067         * declaration. If there is one, a new finding will be created.
068         */
069        private void checkForFriends(List<ShallowEntity> entities) throws CheckException {
070                for (ShallowEntity entity : entities) {
071                        if (TokenStreamUtils.contains(entity.ownStartTokens(), FRIEND)) {
072                                createFinding("The friend declaration `" + entity.getName() + "` should be private", entity);
073                        }
074                }
075        }
076}
