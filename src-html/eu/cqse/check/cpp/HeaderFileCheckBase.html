006package eu.cqse.check.cpp;
007
008import java.io.File;
009import java.util.Set;
010import java.util.stream.Collectors;
011
012import org.conqat.lib.commons.collections.CollectionUtils;
013import org.conqat.lib.commons.filesystem.FileSystemUtils;
014
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckImplementationBase;
017
018/**
019 * Base class for any check specific to header files. If the analyzed file is
020 * not a header it will be ignored
021 */
022public abstract class HeaderFileCheckBase extends CheckImplementationBase {
023        /** File extensions that are used to identify header files. */
024        private static Set<String> headerExtensions = CollectionUtils.asHashSet("h", "hh", "hpp", "hxx");
025
026        /** {@inheritDoc} */
027        @Override
028        public void initialize() {
029                headerExtensions = headerExtensions.stream().map(x -> x.toLowerCase()).collect(Collectors.toSet());
030        }
031
032        /** {@inheritDoc} */
033        @Override
034        public void execute() throws CheckException {
035                if (isHeader(context.getUniformPath())) {
036                        executeHeaderCheck();
037                }
038        }
039
040        /**
041         * Executes the custom check with the given context only if the file is a
042         * header-file
043         */
044        public abstract void executeHeaderCheck() throws CheckException;
045
046        /**
047         * Checks if the currently analyze files is a header by checking if its
048         * extension matches any of the ones given by the header extension option
049         */
050        public static boolean isHeader(String uniformPath) {
051                String extension = FileSystemUtils.getFileExtension(new File(uniformPath));
052                return (extension != null && headerExtensions.contains(extension.toLowerCase()));
053        }
054}
