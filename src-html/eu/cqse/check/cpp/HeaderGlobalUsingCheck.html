001package eu.cqse.check.cpp;
002
003import java.util.List;
004
005import eu.cqse.check.framework.core.ACheck;
006import eu.cqse.check.framework.core.CheckException;
007import eu.cqse.check.framework.core.CheckGroupNames;
008import eu.cqse.check.framework.core.ECheckParameter;
009import eu.cqse.check.framework.core.EFindingEnablement;
010import eu.cqse.check.framework.scanner.ELanguage;
011import eu.cqse.check.framework.shallowparser.SubTypeNames;
012import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
013
014/** {@ConQAT.Doc} */
015@ACheck(name = "No global `using`", description = "`using` commands should only be used in classes, methods or functions."
016                + "", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
017                                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
018public class HeaderGlobalUsingCheck extends HeaderFileCheckBase {
019        /** {@inheritDoc} */
020        @Override
021        public void executeHeaderCheck() throws CheckException {
022                List<ShallowEntity> violatingEntities = select(
023                                "//META[subtype('" + SubTypeNames.USING + "') and not(ancestor::TYPE or ancestor::METHOD)]");
024                for (ShallowEntity entity : violatingEntities) {
025                        createFinding("Do not use `using` outside of functions/methods/classes in headers", entity);
026                }
027        }
028}
