001package eu.cqse.check.cpp;
002
003import static eu.cqse.check.framework.scanner.ETokenType.INLINE;
004import static eu.cqse.check.framework.shallowparser.SubTypeNames.FUNCTION;
005import static eu.cqse.check.framework.shallowparser.SubTypeNames.TEMPLATE;
006
007import eu.cqse.check.framework.core.ACheck;
008import eu.cqse.check.framework.core.CheckException;
009import eu.cqse.check.framework.core.CheckGroupNames;
010import eu.cqse.check.framework.core.ECheckParameter;
011import eu.cqse.check.framework.core.EFindingEnablement;
012import eu.cqse.check.framework.core.phase.ECodeViewOption;
013import eu.cqse.check.framework.scanner.ELanguage;
014import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
015
016/** {@ConQAT.Doc} */
017@ACheck(name = "Methods should not be defined in a header file", description = "Do not put method "
018                + "definitions inside the class declaration, except for inline or template functions."
019                + "", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
020                                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
021public class MethodDefinitionInHeaderCheck extends HeaderFileCheckBase {
022        /**
023         * xpath for nodes that are not a c++ template. This is done by checking if
024         * preceding sibling is of the type TEMPLATE or not.
025         */
026        private static final String IS_NOT_TEMPLATE = "not(preceding-sibling::*[1][subtype('" + TEMPLATE + "')])";
027
028        /** xpath selection for nodes that do not contain the INLINE token. */
029        private static final String IS_NOT_INLINE = "not(anytoken('" + INLINE.name() + "'))";
030
031        /**
032         * xpath selection for function definition nodes. This means that function
033         * declaration, for example, will not match.
034         */
035        private static final String IS_FUNCTION = "subtype('" + FUNCTION + "')";
036
037        /** xpath selection for function definitions in types */
038        private static final String WRONG_TYPE_FUNCTION = "//TYPE[" + IS_NOT_TEMPLATE + "]/METHOD[" + IS_FUNCTION + " and "
039                        + IS_NOT_TEMPLATE + " and " + IS_NOT_INLINE + "]";
040
041        /**
042         * xpath selection for a simple function definition outside of the class, which
043         * are not templates or inline
044         */
045        private static final String WRONG_FUNCTION = "./METHOD[" + IS_FUNCTION + "]";
046
047        @Override
048        protected ECodeViewOption getCodeViewOption() {
049                return ECodeViewOption.FILTERED_PREPROCESSED;
050        }
051
052        /** {@inheritDoc} */
053        @Override
054        public void executeHeaderCheck() throws CheckException {
055                for (ShallowEntity entity : select(WRONG_TYPE_FUNCTION + "|" + WRONG_FUNCTION)) {
056                        createFinding("`" + entity.getName() + "` should not be defined in the header", entity);
057                }
058        }
059}
