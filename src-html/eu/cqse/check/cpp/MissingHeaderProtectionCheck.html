017package eu.cqse.check.cpp;
018
019import java.util.List;
020import java.util.Set;
021import java.util.function.Predicate;
022import java.util.regex.Pattern;
023
024import org.conqat.lib.commons.collections.CollectionUtils;
025import org.conqat.lib.commons.string.StringUtils;
026
027import eu.cqse.check.framework.core.ACheck;
028import eu.cqse.check.framework.core.CheckException;
029import eu.cqse.check.framework.core.CheckGroupNames;
030import eu.cqse.check.framework.core.CheckImplementationBase;
031import eu.cqse.check.framework.core.EFindingEnablement;
032import eu.cqse.check.framework.core.option.ACheckOption;
033import eu.cqse.check.framework.core.phase.ECodeViewOption;
034import eu.cqse.check.framework.scanner.ELanguage;
035import eu.cqse.check.framework.scanner.ETokenType;
036import eu.cqse.check.framework.scanner.ETokenType.ETokenClass;
037import eu.cqse.check.framework.scanner.IToken;
038import eu.cqse.check.framework.scanner.ScannerUtils;
039import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
040import eu.cqse.check.framework.util.CppLanguageConstants;
041
042/**
043 * {@ConQAT.Doc}
044 */
045@ACheck(name = "Missing header protection", description = "Header files must protect from multiple inclusion using the preprocessor.", groupName = CheckGroupNames.CORRECTNESS, defaultEnablement = EFindingEnablement.YELLOW, languages = {
046                ELanguage.CPP })
047public class MissingHeaderProtectionCheck extends CheckImplementationBase {
048
049        /** Pattern for pragma once (microsoft style). */
050        private static final Pattern PRAGMA_ONCE_PATTERN = Pattern.compile("\\s*#\\s*pragma\\s+once\\s*");
051
052        /** {@ConQAT.Doc} */
053        @ACheckOption(name = "Header extensions", description = "File extensions that are used to identify header files.")
054        private Set<String> headerExtensions = CollectionUtils.asHashSet("h", "hh", "hpp", "hxx");
055
056        /** {@inheritDoc} */
057        @Override
058        public void execute() throws CheckException {
059
060                String extension = StringUtils.getLastPart(context.getUniformPath(), '.');
061                if (extension == null || (!headerExtensions.isEmpty() && !headerExtensions.contains(extension.toLowerCase()))) {
062                        // skip non-headers
063                        return;
064                }
065
066                List<IToken> tokens = ScannerUtils.getTokens(
067                                context.getTextContent(ECodeViewOption.ETextViewOption.UNFILTERED_CONTENT), context.getLanguage());
068                // pragma once is Microsoft style
069                if (tokens.stream().anyMatch(token -> token.getType() == ETokenType.PREPROCESSOR_DIRECTIVE
070                                && PRAGMA_ONCE_PATTERN.matcher(token.getText()).matches())) {
071                        return;
072                }
073                Predicate<IToken> isRelevantToken = token -> token.getType().getTokenClass() != ETokenClass.COMMENT;
074                List<Integer> relevantIndices = TokenStreamUtils.indicesOfByPredicate(tokens, isRelevantToken);
075                // we check if there are at least 3 tokens, namely #ifndef, #define
076                // and #endif for the header protection
077                if (relevantIndices.size() < 3) {
078                        createHeaderProtectFinding();
079                        return;
080                }
081                String checkedConstant = checkAndExtractIfndef(tokens.get(relevantIndices.get(0)));
082                String definedConstant = extractDefineConstant(tokens.get(relevantIndices.get(1)));
083
084                if (checkedConstant == null || definedConstant == null
085                                || !checkClosingEndIf(tokens.get(CollectionUtils.getLast(relevantIndices)))) {
086                        createHeaderProtectFinding();
087                } else if (!checkedConstant.equals(definedConstant)) {
088                        createFinding("Header protection defines constant `" + definedConstant + "` but is is checked for `"
089                                        + checkedConstant + "`", tokens.get(relevantIndices.get(1)).getLineNumber() + 1);
090                }
091        }
092
093        /**
094         * The first macro must be either "#if !defined" or "#ifndef". Returns the
095         * constant checked for or null if constraint not satisfied.
096         */
097        private static String checkAndExtractIfndef(IToken token) {
098                if (token.getType() != ETokenType.PREPROCESSOR_DIRECTIVE) {
099                        return null;
100                }
101
102                String[] parts = getPreprocessorDirectiveText(token).split("(\\s|[()])+");
103                if (parts.length >= 2 && parts[0].equals(CppLanguageConstants.PREPROCESSOR_IFNDEF)) {
104                        return parts[1];
105                }
106
107                if (parts.length < 3 || !parts[0].equals(CppLanguageConstants.PREPROCESSOR_IF)) {
108                        return null;
109                }
110
111                if (parts[1].equals("!defined")) {
112                        return parts[2];
113                }
114
115                if (parts[1].equals("!") && parts[2].equals("defined") && parts.length > 3) {
116                        return parts[3];
117                }
118
119                return null;
120        }
121
122        /**
123         * Returns the text of a preprocessor directive with surrounding whitespace
124         * trimmed and any whitespace after the initial '#' removed.
125         */
126        private static String getPreprocessorDirectiveText(IToken token) {
127                return token.getText().trim().replaceFirst("#\\s+", "#");
128        }
129
130        /**
131         * The second macro must be a "#define". Returns the name of the defined
132         * constant or null if constraint not satisfied.
133         */
134        private static String extractDefineConstant(IToken token) {
135                if (token.getType() != ETokenType.PREPROCESSOR_DIRECTIVE) {
136                        return null;
137                }
138
139                String[] parts = getPreprocessorDirectiveText(token).split("\\s+");
140                if (parts.length < 2 || !parts[0].equals(CppLanguageConstants.PREPROCESSOR_DEFINE)) {
141                        return null;
142                }
143
144                return parts[1];
145        }
146
147        /** Last non-comment token must be "#endif". */
148        private static boolean checkClosingEndIf(IToken lastToken) {
149                return lastToken.getType() == ETokenType.PREPROCESSOR_DIRECTIVE
150                                && getPreprocessorDirectiveText(lastToken).startsWith(CppLanguageConstants.PREPROCESSOR_ENDIF);
151        }
152
153        /** Creates the finding. */
154        private void createHeaderProtectFinding() throws CheckException {
155                createFinding("Header protection missing or not implemented correctly");
156        }
157}
