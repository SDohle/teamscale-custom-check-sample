001package eu.cqse.check.cpp;
002
003import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
004import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
005import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
006import static eu.cqse.check.framework.util.CppLanguageFeatureParser.VALID_IDENTIFIER_PATTERN;
007
008import java.util.regex.Pattern;
009
010import org.conqat.lib.commons.collections.UnmodifiableList;
011
012import eu.cqse.check.base.EntityCheckBase;
013import eu.cqse.check.framework.core.ACheck;
014import eu.cqse.check.framework.core.CheckException;
015import eu.cqse.check.framework.core.CheckGroupNames;
016import eu.cqse.check.framework.core.ECheckParameter;
017import eu.cqse.check.framework.scanner.ELanguage;
018import eu.cqse.check.framework.scanner.IToken;
019import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
020import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
021
022/**
023 * {@ConQAT.Doc}
024 */
025@ACheck(name = "No C-Style casts", description = "C-style casting should not be used. Contrary to C++ casting, e.g. `static_cast<>()`, "
026                + "c-style casting, e.g. `(int*) variable`, cannot be checked during the compile time and might fail or "
027                + "give you unexpected behaviour during runtime.", groupName = CheckGroupNames.BAD_PRACTICE, languages = {
028                                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
029public class NoCStyleCastCheck extends EntityCheckBase {
030        /** Pattern which matches c-style casts in a String. */
031        private static final Pattern C_STYLE_CAST_PATTERN = Pattern
032                        .compile("\\(" + VALID_IDENTIFIER_PATTERN + "\\*?\\)\\(?(" + VALID_IDENTIFIER_PATTERN + ")\\)?");
033
034        /** {@inheritDoc} */
035        @Override
036        protected String getXPathSelectionString() {
037                return "//STATEMENT[anytoken('" + LPAREN + "','" + RPAREN + "','" + IDENTIFIER + "')]";
038        }
039
040        /** {@inheritDoc} */
041        @Override
042        public void execute() throws CheckException {
043                if (!isCpp()) {
044                        return;
045                }
046                super.execute();
047        }
048
049        /** {@inheritDoc} */
050        @Override
051        protected void processEntity(ShallowEntity entity) throws CheckException {
052                if (containsCStyleCast(entity)) {
053                        createFinding("Avoid using c-style casts", entity.getStartLine());
054                }
055        }
056
057        /**
058         * Checks whether the given entity contains any sequence of tokens which
059         * constitute a so-called c-style cast.
060         */
061        public boolean containsCStyleCast(ShallowEntity entity) {
062                UnmodifiableList<IToken> tokens = entity.ownStartTokens();
063                return C_STYLE_CAST_PATTERN.matcher(TokenStreamTextUtils.concatTokenTexts(tokens)).find();
064        }
065}
