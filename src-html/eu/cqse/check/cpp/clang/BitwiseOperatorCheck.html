033/**
034 * Finds use of ~ operator on signed integers. Part of
035 * <ul>
036 * <li>MISRA C++:2008, 5-0-21 - Bitwise operators shall only be applied to
037 * operands of unsigned underlying type</li>
038 * <li>MISRA C:2012, 10.1 - Operands shall not be of an inappropriate essential
039 * type</li>
040 * <li>cf. https://rules.sonarsource.com/c/type/Bug/RSPEC-874</li>
041 * </ul>
042 */
043@ACheck(name = "Avoid using the ~ operator on signed integers", description = "Avoid using the ~ operator on signed integers. Part of\n"
044                + "* MISRA C++:2008, 5-0-21 - Bitwise operators shall only be applied to operands of unsigned underlying type\n"
045                + "* MISRA C:2012, 10.1 - Operands shall not be of an inappropriate essential type\n"
046                + "", groupName = CheckGroupNames.EMBEDDED_SOFTWARE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
047                                ELanguage.CPP }, parameters = { ECheckParameter.CLANG })
048@ACheckRuleset(ruleId = "10.1", ruleset = ERuleset.MISRA)
049public class BitwiseOperatorCheck extends CheckImplementationBase {
050
051        /**
052         * {@inheritDoc}
053         */
054        @Override
055        public void execute() throws CheckException {
056                ClangTranslationUnitWrapper clangTranslationUnitWrapper = context.getClangTranslationUnitWrapper();
057                List<Pair<CheckTextRegionLocation, String>> findings = checkBitwiseOperatorUse(clangTranslationUnitWrapper,
058                                context.getTextContent(ECodeViewOption.ETextViewOption.UNFILTERED_CONTENT));
059                for (Pair<CheckTextRegionLocation, String> finding : findings) {
060                        context.createFindingForOffsets(finding.getSecond(), finding.getFirst().rawStartOffset,
061                                        finding.getFirst().rawEndOffset, ECodeViewOption.ETextViewOption.UNFILTERED_CONTENT);
062                }
063        }
064
065        /**
066         * Finds use of ~ operator on signed integers.
067         * 
068         * @param translationUnitWrapper
069         *            Wrapper for the Clang translation unit object
070         * @return A List of Pairs containing location and message of findings to be
071         *         created.
072         */
073        List<Pair<CheckTextRegionLocation, String>> checkBitwiseOperatorUse(
074                        ClangTranslationUnitWrapper translationUnitWrapper, String text) {
075                final SWIGTYPE_p_CXTranslationUnitImpl translationUnit = translationUnitWrapper.getMainTranslationUnit();
076                final CXCursor rootCursor = translationUnitWrapper.getMainRootCursor();
077                final List<Pair<CheckTextRegionLocation, String>> findings = new ArrayList<>();
078
079                ClangBinding.visitChildren(rootCursor, new IClangCursorVisitor() {
080                        @Override
081                        public CXChildVisitResult visit(CXCursor cursor, CXCursor parent) {
082                                if (Clang.clang_getCursorKind(cursor) != CXCursorKind.CXCursor_UnaryOperator) {
083                                        return CXChildVisitResult.CXChildVisit_Recurse;
084                                }
085                                CXToken token = Clang.clang_getToken(translationUnit, Clang.clang_getCursorLocation(cursor));
086                                if (token == null) {
087                                        return CXChildVisitResult.CXChildVisit_Recurse;
088                                }
089                                if (!"~".equals(Clang.clang_getCString(Clang.clang_getTokenSpelling(translationUnit, token)))) {
090                                        return CXChildVisitResult.CXChildVisit_Recurse;
091                                }
092                                List<CXCursor> children = ClangUtils.getDirectChildren(cursor);
093                                if (children.isEmpty()) {
094                                        return CXChildVisitResult.CXChildVisit_Recurse;
095                                }
096                                CXCursor child = children.get(0);
097                                if (Clang.clang_getCursorKind(child) == CXCursorKind.CXCursor_FirstExpr
098                                                && Clang.clang_getCursorType(child).getKind() == CXTypeKind.CXType_Int) {
099                                        Optional<CheckTextRegionLocation> cursorLocation = ClangUtils
100                                                        .createTextRegionLocation(context.getUniformPath(), cursor);
101                                        if (!cursorLocation.isPresent()) {
102                                                return CXChildVisitResult.CXChildVisit_Continue;
103                                        }
104
105                                        Optional<String> nameOrExpression = ClangUtils.getCompleteNodeText(child, context, text);
106                                        findings.add(Pair.createPair(cursorLocation.get(), "Finding: Must not use ~ on signed int "
107                                                        + StringUtils.truncate(nameOrExpression.get(), 8, "...")));
108                                        return CXChildVisitResult.CXChildVisit_Continue;
109                                }
110                                return CXChildVisitResult.CXChildVisit_Recurse;
111                        }
112                });
113                return findings;
114
115        }
