041/**
042 * Checks that each external object or function is only declared once in the
043 * entire project.
044 */
045@ACheck(name = "All declarations should have the same function signature", description = "The signature of the implementation must be equal to the signature of the prototype declaration. "
046                + "The parameter names and types must be equal."
047                + " Identical typedefs are not considered equal in the signature-equality check. Due to method overloading this check will result in many false positives for C++.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.OFF, languages = {
048                                ELanguage.CPP }, parameters = {
049                                                ECheckParameter.ABSTRACT_SYNTAX_TREE }, phases = { CppMethodDeclarationsExtractorPhase.class })
050@ACheckRuleset(ruleId = "8.3", ruleset = MISRA)
051public class AllDeclarationsShallUseSameSignatureCheck extends EntityCheckBase {
052
053        @Override
054        protected ECodeViewOption getCodeViewOption() {
055                return ECodeViewOption.FILTERED_PREPROCESSED;
056        }
057
058        /** {@inheritDoc} */
059        @Override
060        protected String getXPathSelectionString() {
061                return "//METHOD";
062        }
063
064        /** {@inheritDoc} */
065        @Override
066        protected void processEntity(ShallowEntity entity) throws CheckException {
067                List<FunctionDeclaration> declarations = getDifferentDeclarationsFrom(entity);
068                if (declarations == null || CppLanguageFeatureParser.isCppMethod(entity)) {
069                        return;
070                }
071
072                // check number of parameters
073                List<List<IToken>> parameters = LanguageFeatureParser.CPP.getSplitParameterTokens(entity);
074                Map<Boolean, List<FunctionDeclaration>> hasDifferentNumberOfParameters = declarations.stream().collect(
075                                Collectors.partitioningBy(declaration -> declaration.getParameters().size() != parameters.size()));
076                processDeclarationsWithDifferentNumberOfParameters(entity, hasDifferentNumberOfParameters.get(true));
077
078                // check the names and types of parameters for declarations with the same number
079                for (int i = 0; i < parameters.size(); i++) {
080                        List<IToken> implementationParameter = parameters.get(i);
081                        processDeclarationsWithDifferentParameterName(i, implementationParameter,
082                                        hasDifferentNumberOfParameters.get(false));
083                        processDeclarationsWithDifferentParameterType(i, implementationParameter,
084                                        hasDifferentNumberOfParameters.get(false));
085                }
086        }
087
088        /**
089         * Returns a list of declarations (and definitions) with the same name but a
090         * different start location than the given shallow entity.
091         */
092        private List<FunctionDeclaration> getDifferentDeclarationsFrom(ShallowEntity entity) {
093                Predicate<FunctionDeclaration> isDifferentFromEntity = declaration -> entity.ownStartTokens().get(0)
094                                .getOffset() != declaration.getAdditionalInformation().ownStartTokens().get(0).getOffset();
095                Function<String, List<FunctionDeclaration>> declarationToUniformPaths = context
096                                .accessPhaseInvertedResult(CppMethodDeclarationsExtractorPhase.class);
097                return CollectionUtils.filter(declarationToUniformPaths.apply(entity.getName()), isDifferentFromEntity);
098        }
099
100        /**
101         * Creates a finding for a procedure declaration (or definition) if there are
102         * other declarations (or definitions) with a different number of parameters.
103         * For each of these other declarations, a sibling finding is created.
104         * 
105         * @param mainDeclaration
106         *            The declaration to consider.
107         * @param declarationsWithDifferentNumberOfParameters
108         *            The list of other declarations with the same name
109         */
110        private void processDeclarationsWithDifferentNumberOfParameters(ShallowEntity mainDeclaration,
111                        List<FunctionDeclaration> declarationsWithDifferentNumberOfParameters) {
112                if (declarationsWithDifferentNumberOfParameters.size() > 0) {
113                        String message = "Parameter count differs from other declaration(s)";
114                        context.createFindingForSiblings(message, extractSignatureLocationFrom(mainDeclaration),
115                                        CollectionUtils.map(declarationsWithDifferentNumberOfParameters, FunctionDeclaration::getLocation));
116                }
117        }
118
119        /**
120         * Creates a finding for a particular parameter of a procedure when there are
121         * other declarations (or definitions) with a differing parameter name. Creates
122         * sibling findings for each of these other parameters.
123         *
124         * @param parameterIndex
125         *            The index of the parameter
126         * @param mainParameter
127         *            The considered parameter
128         * @param declarations
129         *            The other declarations with the same procedure name
130         * @throws CheckException
131         */
132        private void processDeclarationsWithDifferentParameterName(int parameterIndex, List<IToken> mainParameter,
133                        List<FunctionDeclaration> declarations) throws CheckException {
134                IToken implementationParameterName = LanguageFeatureParser.CPP.getVariableNameFromTokens(mainParameter);
135
136                ArrayList<CheckTextRegionLocation> siblingLocations = new ArrayList<>();
137                for (FunctionDeclaration declaration : declarations) {
138                        List<IToken> declarationParameter = declaration.getParameters().get(parameterIndex);
139                        IToken declarationParameterName = LanguageFeatureParser.CPP.getVariableNameFromTokens(declarationParameter);
140
141                        if (!TokenUtils.isEqualTypeAndText(implementationParameterName, declarationParameterName)) {
142                                siblingLocations.add(getLocationOfParameter(context.getUniformPath(), declarationParameter));
143                        }
144                }
145
146                if (siblingLocations.size() > 0) {
147                        String message = "Parameter name ``" + getTextOrUndefined(implementationParameterName)
148                                        + "`` differs from other declaration(s)";
149                        context.createFindingForSiblings(message, getLocationOfParameter(context.getUniformPath(), mainParameter),
150                                        siblingLocations);
151                }
152        }
153
154        /**
155         * Creates a finding for a particular parameter of a procedure when there are
156         * other declarations (or definitions) with a differing parameter type. Creates
157         * sibling findings for each of these other parameters.
158         *
159         * @param parameterIndex
160         *            The index of the parameter
161         * @param mainParameter
162         *            The considered parameter
163         * @param declarations
164         *            The other declarations with the same procedure name
165         * @param parameterIndex
166         * @throws CheckException
167         */
168        private void processDeclarationsWithDifferentParameterType(int parameterIndex, List<IToken> mainParameter,
169                        List<FunctionDeclaration> declarations) throws CheckException {
170                mainParameter = removeNameIfPresent(mainParameter);
171
172                ArrayList<CheckTextRegionLocation> siblingLocations = new ArrayList<>();
173                for (FunctionDeclaration declaration : declarations) {
174                        List<IToken> declarationParameter = removeNameIfPresent(declaration.getParameters().get(parameterIndex));
175                        if (!removeRedundantTokensAndSort(mainParameter)
176                                        .equals(removeRedundantTokensAndSort(declarationParameter))) {
177                                siblingLocations.add(getLocationOfParameter(context.getUniformPath(), declarationParameter));
178                        }
179                }
180
181                if (siblingLocations.size() > 0) {
182                        String message = "Parameter type ``" + TokenStreamTextUtils.concatTokenTexts(mainParameter, " ")
183                                        + "`` differs from other declaration(s)";
184                        context.createFindingForSiblings(message, getLocationOfParameter(context.getUniformPath(), mainParameter),
185                                        siblingLocations);
186                }
187        }
188
189        /**
190         * Returns the location of a given shallow entity (representing a method
191         * declaration).
192         */
193        private CheckTextRegionLocation extractSignatureLocationFrom(ShallowEntity methodDeclaration) {
194                List<IToken> declarationTokens = methodDeclaration.ownStartTokens();
195                int endOffset = declarationTokens.get(declarationTokens.size() - 1).getOffset();
196                int endLine = declarationTokens.get(declarationTokens.size() - 1).getLineNumber() + 1;
197                return new CheckTextRegionLocation(context.getUniformPath(), methodDeclaration.getStartOffset(), endOffset,
198                                methodDeclaration.getStartLine(), endLine);
199        }
200
201        private static List<IToken> removeNameIfPresent(List<IToken> parameters) {
202                if (parameters.size() > 1) {
203                        return TokenStreamUtils.removeAtEnd(parameters, ETokenType.IDENTIFIER);
204                }
205                return parameters;
206        }
207
208        private static String getTextOrUndefined(IToken token) {
209                if (token == null) {
210                        return "<undefined>";
211                }
212                return token.getText();
213        }
214
215        private static List<String> removeRedundantTokensAndSort(List<IToken> list) {
216                List<String> tokenTexts = TokenStreamTextUtils.getTokenTexts(list);
217                Collections.sort(tokenTexts);
218                if (!tokenTexts.contains("char")
219                                && (tokenTexts.contains("short") || tokenTexts.contains("int") || tokenTexts.contains("long"))) {
220                        // if the type is not char then the signed and int keywords are optional
221                        // see https://pebble.gitbooks.io/learning-c-with-pebble/content/appendixa.html
222                        tokenTexts.remove("int");
223                        tokenTexts.remove("signed");
224                }
225                return tokenTexts;
226        }
