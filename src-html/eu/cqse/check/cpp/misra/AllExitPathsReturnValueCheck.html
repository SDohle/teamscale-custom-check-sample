012import java.util.Set;

014import org.conqat.lib.commons.collections.CollectionUtils;

025import eu.cqse.check.framework.scanner.ETokenType;
026import eu.cqse.check.framework.scanner.IToken;
027import eu.cqse.check.framework.shallowparser.SubTypeNames;
028import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
029import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
030import eu.cqse.check.framework.util.CppLanguageFeatureParser;
031import eu.cqse.check.framework.util.LanguageFeatureParser;

033/**
034 * Checks whether all exit paths from a function with non-void return type have
035 * an explicit return statement with a return value. MISRA Rule 17.4.
036 */
037@ACheck(name = "Non-void function should return a value", description = "Non-void functions should return a value.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.RED, languages = {
038                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
039@ACheckRuleset(ruleId = "17.4", ruleset = MISRA)
040public class AllExitPathsReturnValueCheck extends EntityCheckBase {
041
042        /**
043         * Elements on which to skip this check as it will only produce false positives
044         */
045        private static final Set<String> SUBTYPES_TO_SKIP = CollectionUtils.asHashSet(SubTypeNames.CONSTRUCTOR,
046                        SubTypeNames.CONSTRUCTOR_DECLARATION, SubTypeNames.DESTRUCTOR, SubTypeNames.DESTRUCTOR_DECLARATION,
047                        SubTypeNames.METHOD_DECLARATION, SubTypeNames.FUNCTION_DECLARATION, SubTypeNames.OPERATOR_DECLARATION,
048                        SubTypeNames.LAMBDA_EXPRESSION);
049
050        @Override
051        protected ECodeViewOption getCodeViewOption() {
052                return ECodeViewOption.FILTERED_PREPROCESSED;
053        }
054
055        /**
056         * {@inheritDoc}
057         */
058        @Override
059        protected String getXPathSelectionString() {
060                return "//" + EShallowEntityType.METHOD;
061        }
062
063        /**
064         * {@inheritDoc}
065         */
066        @Override
067        protected void processEntity(ShallowEntity entity) throws CheckException {
068                if (CppLanguageFeatureParser.isFunctionDeclaration(entity) || CppLanguageFeatureParser.isVirtual(entity)
069                                || SUBTYPES_TO_SKIP.contains(entity.getSubtype())) {
070                        return;
071                }
072                if (LanguageFeatureParser.CPP.hasVoidReturnType(entity)) {
073                        return;
074                }
075
076                // Check for no return statements in entire function
077                // Note that is a little more lenient than the check requires, as we could only
078                // return in one if and not in the else branch
079                List<ShallowEntity> returnStatements = select(entity,
080                                "descendant::" + EShallowEntityType.STATEMENT + "[name-matches('" + RETURN + "')]");
081                if (CollectionUtils.isNullOrEmpty(returnStatements)) {
082                        createFindingForEntity(entity, entity);
083                        return;
084                }
085
086                for (ShallowEntity returnStatement : returnStatements) {
087                        checkReturnStatement(returnStatement, entity);
088                }
089        }
090
091        /**
092         * Creates a finding at location of findingsEntity mentioning the name of
093         * functionEntity
094         */
095        private void createFindingForEntity(ShallowEntity functionEntity, ShallowEntity findingsEntity)
096                        throws CheckException {
097                createFindingOnFirstLine("Non-void function `" + functionEntity.getName() + "` should return a value",
098                                findingsEntity);
099        }
100
101        /**
102         * Checks if the provided return statement returns a value. Creates a finding if
103         * it doesn't.
104         */
105        private void checkReturnStatement(ShallowEntity returnStatement, ShallowEntity functionEntity)
106                        throws CheckException {
107                if (returnStatement.hasChildren()) {
108                        // this is something complex, such as lambda, and thus can not be void
109                        return;
110                }
111
112                List<IToken> returnStartTokens = returnStatement.ownStartTokens();
113                if (returnStartTokens.size() < 2 || returnStartTokens.get(1).getType() == ETokenType.SEMICOLON) {
114                        createFindingForEntity(functionEntity, returnStatement);
115                }
116        }
117}
