035/**
036 * Extraction phase for the {@link MultipleObjectOrFunctionDeclarationCheck}.
037 * Extracts and stores all object- or function declarations.
038 */
039public class CppExternDeclarationsExtractorPhase
040                implements IGlobalExtractionPhase<ExternDeclaration, CheckTextRegionLocation> {
041
042        /**
043         * Matches on declarations of extern objects/functions. A match contains the
044         * name of the declared object/function in match group 1.
045         * 
046         * Examples for matched sequences:
047         * <ul>
048         * <li><code>extern bool(*SetPose)(AirSimPose pose, bool ignoreCollision, const char*
049         * vehicleName); </code></li>
050         * <li><code>extern AirSimPose(*GetPose)(const char* vehicleName);</code></li>
051         * <li><code>extern int16_t a;</code></li>
052         * </ul>
053         */
054        private static final TokenPattern EXTERN_DECLARATION_PATTERN = new TokenPattern().sequence(EXTERN)
055                        // Type
056                        .alternative(IDENTIFIER, ETokenClass.KEYWORD)
057                        // Name
058                        .alternative(
059                                        // directly the declared name
060                                        new TokenPattern().sequence(IDENTIFIER).group(1),
061                                        // declared name with pointer in parentheses " ( * name ) "
062                                        new TokenPattern().sequence(LPAREN, MULT).sequence(IDENTIFIER).group(1).sequence(RPAREN))
063                        // parameters
064                        .skipNested(LPAREN, RPAREN, true).sequence(ETokenType.SEMICOLON);
065
066        @Override
067        public List<ExternDeclaration> extract(ITokenElementContext fileContext) throws CheckException {
068
069                List<ExternDeclaration> externDeclarations = new ArrayList<>();
070                List<IToken> tokens = fileContext.getTokens(ECodeViewOption.FILTERED_PREPROCESSED);
071                for (TokenPatternMatch externDeclaration : EXTERN_DECLARATION_PATTERN.findAll(tokens)) {
072                        IToken startToken = externDeclaration.groupTokens(1).get(0);
073                        IToken endToken = CollectionUtils.getLast(externDeclaration.groupTokens(1));
074                        CheckTextRegionLocation location = new CheckTextRegionLocation(fileContext.getUniformPath(),
075                                        startToken.getOffset(), endToken.getEndOffset(), startToken.getLineNumber() + 1,
076                                        endToken.getLineNumber() + 1);
077                        externDeclarations.add(
078                                        new ExternDeclaration(fileContext.getUniformPath(), externDeclaration.groupString(1), location));
079                }
080                return externDeclarations;
081        }
082
083        @Override
084        public Set<ELanguage> getLanguages() {
085                return Collections.singleton(ELanguage.CPP);
086        }
087
088        @Override
089        public ExternDeclaration createValue(String uniformPath, String value,
090                        CheckTextRegionLocation additionalInformation) {
091                return new ExternDeclaration(uniformPath, value, additionalInformation);
092        }
093
094        @Override
095        public boolean needsAccessUniformPathByValue() {
096                return true;
097        }
098
099        /**
100         * Encapsulates the information about one object/function declaration with the
101         * extern keyword.
102         */
103        public class ExternDeclaration implements IExtractedValue<CheckTextRegionLocation> {
104
105                private final String uniformPath;
106                private final String declaredObjectName;
107
108                private final CheckTextRegionLocation location;
109
110                private ExternDeclaration(String uniformPath, String declaredObjectName, CheckTextRegionLocation location) {
111                        this.uniformPath = uniformPath;
112                        this.declaredObjectName = declaredObjectName;
113                        this.location = location;
114                }
115
116                @Override
117                public String getUniformPath() {
118                        return uniformPath;
119                }
120
121                @Override
122                public String getValue() {
123                        return declaredObjectName;
124                }
125
126                @Override
127                public CheckTextRegionLocation getAdditionalInformation() {
128                        return location;
129                }
130
131                public int getStartOffset() {
132                        return location.rawStartOffset;
133                }
134
135                public int getEndOffset() {
136                        return location.rawEndOffset;
137                }
138
139        }
140}
