006package eu.cqse.check.cpp.misra;
007
008import java.util.ArrayList;
009import java.util.Collections;
010import java.util.List;
011import java.util.Set;
012
013import org.conqat.lib.commons.collections.CollectionUtils;
014
015import eu.cqse.check.CheckTextRegionLocation;
016import eu.cqse.check.cpp.misra.CppMethodDeclarationsExtractorPhase.FunctionDeclaration;
017import eu.cqse.check.framework.core.CheckException;
018import eu.cqse.check.framework.core.CheckImplementationBase;
019import eu.cqse.check.framework.core.phase.ECodeViewOption;
020import eu.cqse.check.framework.core.phase.IExtractedValue;
021import eu.cqse.check.framework.core.phase.IGlobalExtractionPhase;
022import eu.cqse.check.framework.core.phase.ITokenElementContext;
023import eu.cqse.check.framework.scanner.ELanguage;
024import eu.cqse.check.framework.scanner.IToken;
025import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
026import eu.cqse.check.framework.util.LanguageFeatureParser;
027
028/**
029 * Extracts all function declarations.
030 */
031public class CppMethodDeclarationsExtractorPhase implements IGlobalExtractionPhase<FunctionDeclaration, ShallowEntity> {
032
033        @Override
034        public List<FunctionDeclaration> extract(ITokenElementContext fileContext) throws CheckException {
035
036                List<FunctionDeclaration> externDeclarations = new ArrayList<>();
037                List<ShallowEntity> declarations = CheckImplementationBase
038                                .select(fileContext.getRootEntity(ECodeViewOption.FILTERED_PREPROCESSED), "//METHOD");
039
040                for (ShallowEntity declaration : declarations) {
041                        externDeclarations.add(new FunctionDeclaration(fileContext.getUniformPath(), declaration));
042                }
043                return externDeclarations;
044        }
045
046        @Override
047        public boolean needsAccessUniformPathByValue() {
048                return true;
049        }
050
051        @Override
052        public Set<ELanguage> getLanguages() {
053                return Collections.singleton(ELanguage.CPP);
054        }
055
056        private static ArrayList<ArrayList<IToken>> extractParameters(ShallowEntity entity) {
057                ArrayList<ArrayList<IToken>> parameters = new ArrayList<>();
058                for (List<IToken> parameter : LanguageFeatureParser.CPP.getSplitParameterTokens(entity)) {
059                        parameters.add(new ArrayList<>(parameter));
060                }
061                return parameters;
062        }
063
064        @Override
065        public FunctionDeclaration createValue(String uniformPath, String value, ShallowEntity entity) {
066                // the value is already contained in entity ( entity.getName() ), no need to
067                // store it twice
068                return new FunctionDeclaration(uniformPath, entity);
069        }
070
071        /**
072         * Returns the location of the given parameter (contains uniform path,
073         * unfiltered-token start/end offset, and start line)
074         */
075        public static CheckTextRegionLocation getLocationOfParameter(String uniformPath, List<IToken> parameter) {
076                int startOffset = parameter.get(0).getOffset();
077                int endOffset = CollectionUtils.getLast(parameter).getOffset();
078                int startLine = parameter.get(0).getLineNumber() + 1;
079                int endLine = CollectionUtils.getLast(parameter).getLineNumber() + 1;
080                return new CheckTextRegionLocation(uniformPath, startOffset, endOffset, startLine, endLine);
081        }
082
083        /**
084         * Encapsulates the information about one function declaration.
085         */
086        public class FunctionDeclaration implements IExtractedValue<ShallowEntity> {
087
088                private final String uniformPath;
089
090                private final ShallowEntity declaration;
091
092                private FunctionDeclaration(String uniformPath, ShallowEntity declaration) {
093                        this.uniformPath = uniformPath;
094                        this.declaration = declaration;
095                }
096
097                @Override
098                public String getUniformPath() {
099                        return uniformPath;
100                }
101
102                @Override
103                public String getValue() {
104                        return declaration.getName();
105                }
106
107                @Override
108                public ShallowEntity getAdditionalInformation() {
109                        return declaration;
110                }
111
112                /**
113                 * Returns the tokens for each parameter of this declaration.
114                 */
115                public ArrayList<ArrayList<IToken>> getParameters() {
116                        return extractParameters(declaration);
117                }
118
119                /**
120                 * Returns the location of this declaration (contains uniform path,
121                 * unfiltered-token start/end offset, and start line)
122                 */
123                public CheckTextRegionLocation getLocation() {
124                        return new CheckTextRegionLocation(this.uniformPath, declaration.getStartOffset(),
125                                        declaration.getEndOffset(), declaration.getStartLine(), declaration.getEndLine());
126                }
127        }
128
129}
