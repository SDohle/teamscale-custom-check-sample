006package eu.cqse.check.cpp.misra;

008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009import static eu.cqse.check.framework.preprocessor.c.MacroHandlingCPreprocessorBase.MACRO_ORIGIN;
010import static eu.cqse.check.framework.shallowparser.SubTypeNames.INLINE_ASSEMBLER_BLOCK;
011import static eu.cqse.check.framework.shallowparser.SubTypeNames.INLINE_ASSEMBLER_CALL;
012import static eu.cqse.check.framework.shallowparser.framework.EShallowEntityType.MODULE;
013
014import eu.cqse.check.base.EntityCheckBase;
015import eu.cqse.check.framework.core.ACheck;
016import eu.cqse.check.framework.core.CheckException;
017import eu.cqse.check.framework.core.CheckGroupNames;
018import eu.cqse.check.framework.core.ECheckParameter;
019import eu.cqse.check.framework.core.EFindingEnablement;
020import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
021import eu.cqse.check.framework.scanner.ELanguage;
022import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
023
024/**
025 * Checks that every inline assembly is encapsulated in a macro or an own
026 * function.
027 */
028@ACheck(name = "Assembler encapsulation", description = "Isolation and encapsulation of assembly language is expected. Therefore, Assembler should be directly enclosed by a (inline) method or capsulated as macro or as extern assembly function.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.RED, languages = {
029                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
030@ACheckRuleset(ruleId = "Dir 4.3", ruleset = MISRA)
031public class EncapsulatedAssemblerCheck extends EntityCheckBase {
032
033        /** {@inheritDoc} */
034        @Override
035        protected String getXPathSelectionString() {
036                return "//*[subtype('" + INLINE_ASSEMBLER_CALL + "')|subtype('" + INLINE_ASSEMBLER_BLOCK + "')]";
037        }
038
039        /** {@inheritDoc} */
040        @Override
041        protected void processEntity(ShallowEntity entity) throws CheckException {
042                boolean isInMethod = entity.getParent().getType() != MODULE; // INLINE_ASSEMBLER_CALL could be also at top level
043                boolean isAlone = entity.getParent().getChildren().size() == 1;
044
045                boolean hasMacroAsOrigin = entity.ownStartTokens().size() <= 1
046                                || entity.ownStartTokens().get(0).getOriginId().equals(MACRO_ORIGIN);
047
048                if (isInMethod && !isAlone && !hasMacroAsOrigin) {
049                        createFinding("Isolation and encapsulation of assembly language is expected", entity.getStartLine(),
050                                        entity.getEndLine());
051                }
052
053        }
054
055}
