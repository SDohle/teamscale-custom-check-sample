027/**
028 * Checks that no flexible array members are used in structures.
029 */
030@ACheck(name = "Flexible array members", description = "Flexible array members should not be used, because `sizeof` may behave in an unexpected way for other programmers.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
031                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
032@ACheckRuleset(ruleId = "18.7", ruleset = MISRA)
033public class FlexibleArrayMemberCheck extends EntityCheckBase {
034
035        /** TokenPattern for recognizing a flexible array member. */
036        private static final TokenPattern FLEXIBLE_ARRAY_MEMBER_PATTERN = new TokenPattern()
037                        .sequence(ETokenType.IDENTIFIER, ETokenType.LBRACK, ETokenType.RBRACK, ETokenType.SEMICOLON).endOfStream();
038
039        /**
040         * {@inheritDoc} Selects for each structure the last member, because flexible
041         * arrays can only be at the last position.
042         */
043        @Override
044        protected String getXPathSelectionString() {
045                return "(//TYPE[subtype('" + SubTypeNames.STRUCT + "')]/child::node()[position()=last()])";
046        }
047
048        /**
049         * {@inheritDoc} If macros are used as constant array length, we need the
050         * unprocessed tokens of the context.
051         */
052        @Override
053        protected void processEntity(ShallowEntity entity) throws CheckException {
054                List<IToken> entityTokens = entity.includedTokens();
055                if (FLEXIBLE_ARRAY_MEMBER_PATTERN.matchesAnywhere(entityTokens)) {
056                        String arrayIdentifier = entityTokens
057                                        .get(TokenStreamUtils.firstTokenOfType(entityTokens, ETokenType.LBRACK) - 1).getText();
058                        createFinding("Flexible array member `" + arrayIdentifier + "` shall not be used", entity.getStartLine());
059                }
060        }
061
062}
