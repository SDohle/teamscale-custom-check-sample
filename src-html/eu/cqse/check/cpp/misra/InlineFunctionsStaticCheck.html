022/**
023 * The meaning of the inline keyword differs between C and C++. See C++: https://en.cppreference.com/w/cpp/language/inline
024 * C: https://en.cppreference.com/w/c/language/inline
025 * the rationale for the test no longer holds in C++, as definitions of functions declared inline must be identical.
026 * Therefore, default enablement is set to OFF, so as not to generate false positives in C++ code
027 */
028@ACheck(name = "Inline functions should be declared with the static storage-class modifier", description = "To prevent undefined behavior, the static storage-class specifier should be used when declaring inline functions: If there exists at program scope a defintion with external linkage of an inline function, as well as an inline definition inside the compilation unit, it is undefined which one will used. C++ compilers don't allow this, so this check should be enabled only on C.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.OFF, languages = {
029                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
030@ACheckRuleset(ruleId = "8.10", ruleset = MISRA)
031public class InlineFunctionsStaticCheck extends EntityCheckBase {
032
033        /** {@inheritDoc} */
034        @Override
035        protected String getXPathSelectionString() {
036            // the use of one '/' here is intentional, as C function definitions must
037            // be top level. Thus we avoid checking cpp member functions
038                return "/METHOD";
039        }
040
041        /** {@inheritDoc} */
042        @Override
043        protected void processEntity(ShallowEntity entity) throws CheckException {
044                boolean staticModifier = TokenStreamUtils.contains(entity.ownStartTokens(), ETokenType.STATIC);
045                boolean inlineModifier = TokenStreamUtils.contains(entity.ownStartTokens(), ETokenType.INLINE);
046                if (inlineModifier && !staticModifier) {
047                        createFinding("`" + entity.getName() + "`" + " is declared inline, but not static", entity);
048                }
049
050        }
051
052}
