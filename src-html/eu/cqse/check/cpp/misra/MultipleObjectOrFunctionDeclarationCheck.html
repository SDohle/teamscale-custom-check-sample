026import eu.cqse.check.framework.shallowparser.TokenStreamUtils;

028/**
029 * Checks that each external object or function is only declared once in the
030 * entire project.
031 */
032@ACheck(name = "Multiple declarations of extern objects/functions", description = "Each external object or function should be declared only once (in the entire project).", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.RED, languages = {
033                ELanguage.CPP }, parameters = {}, phases = { CppExternDeclarationsExtractorPhase.class })
034@ACheckRuleset(ruleId = "8.5", ruleset = MISRA)
035public class MultipleObjectOrFunctionDeclarationCheck extends CheckImplementationBase {
036
037        /** {@inheritDoc} */
038        @Override
039        public void execute() throws CheckException {
040                Function<String, List<ExternDeclaration>> uniformPathToLiterals = context
041                                .accessPhaseResult(CppExternDeclarationsExtractorPhase.class);
042
043                Function<String, List<ExternDeclaration>> literalToSiblings = context
044                                .accessPhaseInvertedResult(CppExternDeclarationsExtractorPhase.class);
045
046                String uniformPath = context.getUniformPath();
047                List<IToken> tokens = context.getTokens(ECodeViewOption.FILTERED_PREPROCESSED);
048                List<ExternDeclaration> externDeclarationsInFile = uniformPathToLiterals.apply(uniformPath);
049                for (ExternDeclaration externDeclaration : externDeclarationsInFile) {
050                        int tokenIndex = TokenStreamUtils.indexOfByOffset(tokens, externDeclaration.getStartOffset());
051                        if (tokenIndex == TokenStreamUtils.NOT_FOUND) {
052                                continue;
053                        }
054
055                        List<ExternDeclaration> sameExternDeclarations = literalToSiblings.apply(externDeclaration.getValue());
056                        List<CheckTextRegionLocation> otherSiblingLocations = CollectionUtils.filterAndMap(sameExternDeclarations,
057                                        literal -> !externDeclaration.getUniformPath().equals(literal.getUniformPath()),
058                                        ExternDeclaration::getAdditionalInformation);
059                        if (otherSiblingLocations.isEmpty()) {
060                                continue;
061                        }
062
063                        String message = "Multiple declarations of " + externDeclaration.getValue();
064                        CheckTextRegionLocation mainLocation = externDeclaration.getAdditionalInformation();
065                        context.createFindingForSiblings(message, mainLocation, otherSiblingLocations);
066                }
067        }
068}
