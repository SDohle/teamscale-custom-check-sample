022import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
023import static eu.cqse.check.framework.scanner.ELanguage.CPP;
024import static eu.cqse.check.framework.scanner.ETokenType.EQ;
025import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
026import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
027import static eu.cqse.check.framework.scanner.ETokenType.MINUSEQ;
028import static eu.cqse.check.framework.scanner.ETokenType.MINUSMINUS;
029import static eu.cqse.check.framework.scanner.ETokenType.PLUSEQ;
030import static eu.cqse.check.framework.scanner.ETokenType.PLUSPLUS;
031import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
032import static eu.cqse.check.framework.scanner.ETokenType.SIZEOF;
033import static eu.cqse.check.framework.shallowparser.TokenStreamUtils.NOT_FOUND;

035import java.util.List;

037import eu.cqse.check.base.EntityCheckBase;
038import eu.cqse.check.framework.core.ACheck;
039import eu.cqse.check.framework.core.CheckException;
040import eu.cqse.check.framework.core.phase.ECodeViewOption;
041import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
042import eu.cqse.check.framework.scanner.IToken;
043import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
044import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
045
046/**
047 * Checks whether a sizeof() call potentially contains side effects. This
048 * implements MISRA Rule 13.6.
049 */
050@ACheck(name = "`sizeof()` operands should not have potential side effects", description = "If the operand used in `sizeof()` has potential side effects, these may have no effect/not be executed. `sizeof()` calls are usually evaluated at compile time and replaced with a constant.", groupName = BAD_PRACTICE, defaultEnablement = RED, languages = {
051                CPP }, parameters = { ABSTRACT_SYNTAX_TREE })
052@ACheckRuleset(ruleId = "13.6", ruleset = MISRA)
053public class SizeOfWithSideEffectsCheck extends EntityCheckBase {
054
055        /** {@inheritDoc} */
056        @Override
057        protected String getXPathSelectionString() {
058                return "//STATEMENT";
059        }
060
061        @Override
062        protected ECodeViewOption getCodeViewOption() {
063                return ECodeViewOption.FILTERED_PREPROCESSED;
064        }
065
066        /** {@inheritDoc} */
067        @Override
068        protected void processEntity(ShallowEntity entity) throws CheckException {
069                List<IToken> tokens = entity.ownStartTokens();
070                List<Integer> offsets = TokenStreamUtils.firstTokenOfTypeSequences(tokens, 0, SIZEOF, LPAREN);
071
072                for (Integer startOffset : offsets) {
073                        int endOffset = TokenStreamUtils.findMatchingClosingToken(tokens, startOffset + 2, LPAREN, RPAREN);
074                        if (endOffset == NOT_FOUND) {
075                                // Could not find end, probably broken code, continue
076                                continue;
077                        }
078                        if (TokenStreamUtils.containsAny(tokens, startOffset + 2, endOffset, PLUSPLUS, MINUSMINUS, PLUSEQ, MINUSEQ,
079                                        EQ)) {
080                                createFindingOnFirstLine("Call to `sizeof()` contains operator with potential side effect", entity);
081                        } else {
082                                findAndCreateFindingsForFunctionCalls(entity, tokens, startOffset, endOffset);
083                        }
084                }
085
086        }
087
088        /** Finds function calls inside of a sizeof() operation. */
089        private void findAndCreateFindingsForFunctionCalls(ShallowEntity entity, List<IToken> tokens, Integer startOffset,
090                        int endOffset) throws CheckException {
091                List<Integer> functionCalls = TokenStreamUtils.firstTokenOfTypeSequences(tokens, startOffset + 2, IDENTIFIER,
092                                LPAREN);
093
094                for (Integer functionOffset : functionCalls) {
095                        if (functionOffset >= endOffset) {
096                                return;
097                        }
098
099                        createFindingOnFirstLine("Call to `sizeof()` contains function call (`"
100                                        + tokens.get(functionOffset).getText() + "(...)`) with potential side effect", entity);
101                }
102        }
103
104}
