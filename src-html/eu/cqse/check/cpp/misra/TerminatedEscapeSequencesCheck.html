006package eu.cqse.check.cpp.misra;
007
008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009
010import java.util.EnumSet;
011import java.util.Optional;
012import java.util.regex.Matcher;
013import java.util.regex.Pattern;
014
015import eu.cqse.check.framework.core.ACheck;
016import eu.cqse.check.framework.core.CheckGroupNames;
017import eu.cqse.check.framework.core.ECheckParameter;
018import eu.cqse.check.framework.core.EFindingEnablement;
019import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
020import eu.cqse.check.framework.scanner.ELanguage;
021import eu.cqse.check.framework.scanner.ETokenType;
022import eu.cqse.check.framework.scanner.IToken;
023
024/**
025 * Checks that octal and hexadecimal sequences are the end of a literal of if
026 * they are directly followed by another escape sequence.
027 */
028@ACheck(name = "Terminate octal and hexadecimal escape sequence", description = "Octal and hexadecimal sequences must be followed by another escape sequence "
029                + "or the end of the literal to prevent confusion about their belonging. Valid example: \"\\xA1BC\" (string literal ends after hex literal). "
030                + "Invalid example: \"\\x3AFX (hex literal is followed by another character which is not part of the hex literal)", groupName = CheckGroupNames.READABILITY, defaultEnablement = EFindingEnablement.RED, languages = {
031                                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
032@ACheckRuleset(ruleId = "4.1", ruleset = MISRA)
033public class TerminatedEscapeSequencesCheck extends TokenFilterCheckBase {
034
035        /** Only STRING_LITERAL and CHARACTER_LITERAL tokens should be checked. */
036        private static final EnumSet<ETokenType> CHECKED_TOKENTYPES = EnumSet.of(ETokenType.STRING_LITERAL,
037                        ETokenType.CHARACTER_LITERAL);
038
039        private static final Pattern UNTERMINATED_OCTAL_OR_HEXADECIMAL_SEQUENCE = Pattern.compile(
040                        "(?<!\\\\)(\\\\[0-7]{1,3}+|\\\\x[0-9a-fA-F]++|\\\\u[0-9a-fA-F]{4}+|\\\\U[0-9a-fA-F]{8})[^'\"\\\\]");
041
042        /** {@inheritDoc} */
043        @Override
044        protected String getFindingMessage(IToken token) {
045                return "`" + getFirstNonterminatedOctalOrHexadecimalEscapeSequence(token).get()
046                                + "` is expected to be terminated by another escape sequence or the end of the literal";
047        }
048
049        /**
050         * Returns the first non-terminated escape-sequence or empty String is no
051         * non-terminated escape sequence was found
052         */
053        private static Optional<String> getFirstNonterminatedOctalOrHexadecimalEscapeSequence(IToken token) {
054                Matcher matcher = UNTERMINATED_OCTAL_OR_HEXADECIMAL_SEQUENCE.matcher(token.getText());
055                if (matcher.find()) {
056                        return Optional.of(matcher.group(1));
057                }
058                return Optional.empty();
059        }
060
061        /** {@inheritDoc} */
062        @Override
063        protected boolean isValidToken(IToken token) {
064                if (!CHECKED_TOKENTYPES.contains(token.getType())) {
065                        return true;
066                }
067                return !getFirstNonterminatedOctalOrHexadecimalEscapeSequence(token).isPresent();
068        }
069
070}
