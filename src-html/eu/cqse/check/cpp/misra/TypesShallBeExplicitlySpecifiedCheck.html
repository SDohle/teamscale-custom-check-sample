046/**
047 *
048 */
049@ACheck(name = "Avoid usage of implicit int", description = "In C type declarations,"
050                + " it it possible to omit the type (implicit type is `int` then). To improve readability, this implicit typing should be avoided.", //
051                groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
052                                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE, ECheckParameter.TYPE_RESOLUTION })
053@ACheckRuleset(ruleId = "8.1", ruleset = MISRA)
054public class TypesShallBeExplicitlySpecifiedCheck extends CheckImplementationBase {
055
056        private static final int TYPEDEF_RETURN_TYPE_GROUP = 0;
057        private static final int TYPEDEF_PARAMETERS_GROUP = 1;
058        /*-
059         * Pattern for typedefs for functions pointers eg
060         * typdef int (*foo)(int a, int b);
061         *       ^                ^             ^                        ^
062         *       0                1     2            3
063         */
064        private static final TokenPattern TYPEDEF_PATTERN = new TokenPattern()
065                        // matches between 0 and 1. This is the return type plus typedef and LPAREN
066                        .skipNested(TYPEDEF, LPAREN, false).group(TYPEDEF_RETURN_TYPE_GROUP)
067                        // matches *foo)
068                        .sequence(MULT, IDENTIFIER, RPAREN)
069                        // matches the parameters between 2 and 3
070                        .skipNested(LPAREN, RPAREN, false).group(TYPEDEF_PARAMETERS_GROUP);
071
072        private static final int VARIABLE_NAME_GROUP = 0;
073
074        /**
075         * Matches assignments to simple variables eg foo = 3; This can either mean that
076         * we assign the value 3 to an existing variable foo or we declare a new
077         * variable with implicit int type.
078         *
079         * We use the typeresolution mechanism to decide if this is a known variable or
080         * a new one.
081         */
082        private static final TokenPattern SIMPLE_ASSIGNMENT_PATTERN = new TokenPattern().beginningOfStream()
083                        .repeated(CppLanguageFeatureParser.PRIMITIVE_TYPE_TOKENS).sequence(IDENTIFIER).group(VARIABLE_NAME_GROUP)
084                        .sequence(EQ);
085
086        /**
087         * {@inheritDoc}
088         */
089        @Override
090        public void execute() throws CheckException {
091                if (isCpp()) {
092                        return;
093                }
094                checkAttributes();
095                checkConstructors();
096                checkFunctionDeclarations();
097                checkTypdefs();
098                checkSimpleStatements();
099        }
100
101        @Override
102        protected ECodeViewOption getCodeViewOption() {
103                return ECodeViewOption.FILTERED_PREPROCESSED;
104        }
105
106        /**
107         * Checks all simple assignments. See {@link #SIMPLE_ASSIGNMENT_PATTERN}.
108         */
109        private void checkSimpleStatements() throws CheckException {
110                for (ShallowEntity entity : select(createSelectionPattern(STATEMENT, SubTypeNames.SIMPLE_STATEMENT))) {
111                        ScopedTypeLookup typeLookup = context.getTypeResolution(getCodeViewOption()).getTypeLookup(entity);
112                        TokenPatternMatch match = SIMPLE_ASSIGNMENT_PATTERN.findFirstMatch(entity.ownStartTokens());
113                        if (match == null || !match.hasGroup(VARIABLE_NAME_GROUP)) {
114                                continue;
115                        }
116                        TypedVariable typedVariable = typeLookup.getTypeInfo(match.groupString(VARIABLE_NAME_GROUP));
117                        if (typedVariable == null) {
118                                createFinding("Type specification missing: Missing declaration type for `" + entity.getName() + "`",
119                                                entity);
120                        }
121                }
122        }
123
124        /**
125         * Checks function pointer typedefs for implicit int usage.
126         */
127        private void checkTypdefs() throws CheckException {
128                for (ShallowEntity entity : select(createSelectionPattern(TYPE, SubTypeNames.TYPEDEF))) {
129                        TokenPatternMatch match = TYPEDEF_PATTERN.findFirstMatch(entity.ownStartTokens());
130                        if (checkTypedefReturnType(entity, match)) {
131                                continue;
132                        }
133                        checkTypedefParameters(entity, match);
134                }
135        }
136
137        /**
138         * Checks the parameters of a function pointer typedef. The match contains the
139         * parameters surrounded by parentheses.
140         */
141        private void checkTypedefParameters(ShallowEntity entity, TokenPatternMatch match) throws CheckException {
142                if (match == null || !match.hasGroup(TYPEDEF_PARAMETERS_GROUP)) {
143                        return;
144                }
145                List<IToken> typeTokens = match.groupTokens(TYPEDEF_PARAMETERS_GROUP);
146                typeTokens = TokenStreamUtils.removeAtFront(typeTokens, LPAREN);
147                typeTokens = TokenStreamUtils.removeAtEnd(typeTokens, RPAREN);
148                checkParameterTypes(entity, LanguageFeatureParser.CPP.splitVariableTokens(typeTokens));
149        }
150
151        /**
152         * Checks the return value of a function pointer typedef. The match contains the
153         * return value surrounded by `typedef` and `(`. See {@link #TYPEDEF_PATTERN}.
154         *
155         * Returns true if a finding was created.
156         */
157        private boolean checkTypedefReturnType(ShallowEntity entity, TokenPatternMatch match) throws CheckException {
158                if (match == null || !match.hasGroup(TYPEDEF_RETURN_TYPE_GROUP)) {
159                        return false;
160                }
161                List<IToken> typeTokens = match.groupTokens(TYPEDEF_RETURN_TYPE_GROUP);
162                typeTokens = TokenStreamUtils.removeAtFront(typeTokens, TYPEDEF);
163                typeTokens = TokenStreamUtils.removeAtEnd(typeTokens, LPAREN);
164                if (typeTokens.isEmpty()) {
165                        createFinding("Type specification missing: Missing return type for typedef `" + entity.getName() + "`",
166                                        entity);
167                        return true;
168                }
169                return false;
170        }
171
172        /**
173         * Checks that all parameters of a function declaration have an explicit type.
174         * To check the return type is not necessary because functions without return
175         * type are parsed as constructors.
176         */
177        private void checkFunctionDeclarations() throws CheckException {
178                for (ShallowEntity entity : select(createSelectionPattern(METHOD, SubTypeNames.FUNCTION_DECLARATION))) {
179                        List<List<IToken>> parameterTokens = LanguageFeatureParser.CPP.getSplitParameterTokens(entity);
180                        checkParameterTypes(entity, parameterTokens);
181                }
182        }
183
184        /**
185         * Checks whether parameters of functions or function pointer definitions have a
186         * valid type. Just `void` is a valid type.
187         */
188        private void checkParameterTypes(ShallowEntity entity, List<List<IToken>> parameterTokens) throws CheckException {
189                for (List<IToken> parameter : parameterTokens) {
190                        if (!isVoidOrEllipsis(parameter)) {
191                                if (checkType(parameter,
192                                                "Type specification missing: Missing parameter type for `" + entity.getName() + "`", entity)) {
193                                        break;
194                                }
195                        }
196                }
197        }
198
199        /**
200         * Functions without explicit type are parsed as constructors, if the function
201         * is not a real constructor (inside a class/struct or scoped with ::) then it
202         * is a function without explicit type and we create a finding.
203         */
204        private void checkConstructors() throws CheckException {
205                for (ShallowEntity entity : select(
206                                createSelectionPattern(METHOD, SubTypeNames.CONSTRUCTOR_DECLARATION, SubTypeNames.CONSTRUCTOR))) {
207                        if (!CppLanguageFeatureParser.isCppMethod(entity)) {
208                                createFinding("Type specification missing: Missing return type for `" + entity.getName() + "`", entity);
209                        }
210                }
211        }
212
213        /**
214         * Check that attributes like global variables `int x;` or members of struct
215         * have a valid type.
216         */
217        private void checkAttributes() throws CheckException {
218                for (ShallowEntity entity : select(
219                                createSelectionPattern(ATTRIBUTE, SubTypeNames.GLOBAL_VARIABLE, SubTypeNames.ATTRIBUTE))) {
220                        List<IToken> tokens = removeAssignmentRHS(entity);
221                        if (isAfterStruct(entity)) {
222                                continue;
223                        }
224                        checkType(tokens, "Type specification missing: Missing declaration type for `" + entity.getName() + "`",
225                                        entity);
226                }
227        }
228
229        /*-
230         * Workaround for a parser bug.
231         * 
232         * struct {
233         *      foo int;
234         * } bla;
235         *
236         * bla is of type struct, but our parser creates an attribute without type.
237         * Therefore, we check here if the attribute follows a struct definition.
238         */
239        private static boolean isAfterStruct(ShallowEntity entity) {
240                ShallowEntity previousEntity = ShallowEntityTraversalUtils.getPreviousEntity(entity);
241                if (previousEntity == null) {
242                        return false;
243                }
244                return previousEntity.getSubtype().equals(SubTypeNames.STRUCT);
245        }
246
247        /**
248         * Removes the right hand side of an assignment eg
249         *
250         * int a = foo();
251         *
252         * will result in
253         *
254         * int a
255         */
256        private static List<IToken> removeAssignmentRHS(ShallowEntity entity) {
257                List<IToken> tokens = entity.ownStartTokens();
258                int equalsOffset = TokenStreamUtils.firstTokenOfType(tokens, EQ);
259                if (equalsOffset != TokenStreamUtils.NOT_FOUND) {
260                        tokens = tokens.subList(0, equalsOffset);
261                }
262                return tokens;
263        }
264
265        /**
266         * Checks if the tokens that describe a type declaration eg. `int x` have a
267         * valid type (in this case `int`). If no type can be found a finding is
268         * created.
269         */
270        private boolean checkType(List<IToken> typeTokens, String message, ShallowEntity entity) throws CheckException {
271                String type = LanguageFeatureParser.CPP.getModifiersAndTypeFromTokens(typeTokens).getSecond();
272                if (type == null) {
273                        createFinding(message, entity);
274                        return true;
275                }
276                return false;
277        }
278
279        private static boolean isVoidOrEllipsis(List<IToken> parameter) {
280                if (parameter.size() == 1) {
281                        ETokenType type = parameter.get(0).getType();
282                        return type == VOID || type == ELLIPSIS;
283                }
284                return false;
285        }
