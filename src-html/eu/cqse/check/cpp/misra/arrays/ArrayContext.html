006package eu.cqse.check.cpp.misra.arrays;
007
008import static eu.cqse.check.framework.scanner.ETokenType.COMMA;
009import static eu.cqse.check.framework.scanner.ETokenType.EQ;
010import static eu.cqse.check.framework.scanner.ETokenType.INTEGER_LITERAL;
011import static eu.cqse.check.framework.scanner.ETokenType.LBRACE;
012import static eu.cqse.check.framework.scanner.ETokenType.LBRACK;
013import static eu.cqse.check.framework.scanner.ETokenType.RBRACE;
014import static eu.cqse.check.framework.scanner.ETokenType.RBRACK;
015import static eu.cqse.check.framework.scanner.ETokenType.SEMICOLON;
016import static eu.cqse.check.framework.scanner.ETokenType.STRING_LITERAL;
017import static eu.cqse.check.framework.scanner.ETokenType.ETokenClass.DELIMITER;
018
019import java.util.ArrayList;
020import java.util.EnumSet;
021import java.util.List;
022import java.util.Optional;
023
024import eu.cqse.check.cpp.CppIntegerLiteralParser;
025import eu.cqse.check.framework.scanner.ETokenType;
026import eu.cqse.check.framework.scanner.IToken;
027import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
028import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
029import eu.cqse.check.framework.util.tokens.TokenPattern;
030
031/**
032 * Describes the array and holds the tokens representing its initialization.
033 */
034public class ArrayContext {
035
036        /** array's dimensions in the same order as found in source code */
037        private final int[] dimensions;
038
039        /** array's maximum capacity */
040        private final int capacity;
041
042        /** capacities of subobjects */
043        private final int[] dimensionCapacities;
044
045        /** holds the tokens representing its initialization */
046        private final List<IToken> initializationTokens;
047
048        /** pattern for recognizing array declaration and initialization */
049        private static final TokenPattern ARRAY_INITIALIZATION_PATTERN = createArrayInitializationPattern();
050
051        /** Type of the array */
052        private final EArrayType type;
053
054        /** Used once for creating {@link #ARRAY_INITIALIZATION_PATTERN}. */
055        private static TokenPattern createArrayInitializationPattern() {
056                TokenPattern bracketPattern = new TokenPattern().sequence(LBRACK, INTEGER_LITERAL, RBRACK);
057                TokenPattern compoundPattern = new TokenPattern().skipNested(LBRACE, RBRACE, true);
058                TokenPattern stringLiteralPattern = new TokenPattern().repeatedAtLeastOnce(STRING_LITERAL);
059                return new TokenPattern().beginningOfStream().skipTo(bracketPattern).repeated(bracketPattern).sequence(EQ)
060                                .alternative(stringLiteralPattern, compoundPattern).optional(SEMICOLON).endOfStream();
061        }
062
063        /** wraps the constructor to check whether the given entity is an array. */
064        public static Optional<ArrayContext> createArrayContext(ShallowEntity entity) {
065                List<IToken> tokens = entity.includedTokens();
066                if (!ARRAY_INITIALIZATION_PATTERN.matchesAnywhere(tokens)) {
067                        return Optional.empty();
068                }
069                int leftBracketIndex = TokenStreamUtils.firstTokenOfType(tokens, LBRACK);
070                EArrayType type = EArrayType.getType(tokens.subList(0, leftBracketIndex - 1));
071                int eqIndex = TokenStreamUtils.firstTokenOfType(tokens.subList(leftBracketIndex, tokens.size()), EQ)
072                                + leftBracketIndex;
073                return extractArrayDimensions(tokens.subList(leftBracketIndex, eqIndex))
074                                .map(dimensions -> new ArrayContext(dimensions, tokens.subList(eqIndex + 1, tokens.size()), type));
075        }
076
077        /** Returns the dimensions of an array as array. */
078        private static Optional<int[]> extractArrayDimensions(List<IToken> dimensionTokens) {
079                List<Integer> dimensions = new ArrayList<>();
080
081                StringBuilder dimensionCache = new StringBuilder();
082                for (IToken token : dimensionTokens) {
083                        switch (token.getType()) {
084                        case LBRACK:
085                                dimensionCache.setLength(0);
086                                break;
087                        case RBRACK:
088                                if (!CppIntegerLiteralParser.isInteger(dimensionCache.toString())) {
089                                        return Optional.empty();
090                                }
091                                dimensions.add(CppIntegerLiteralParser.parse(dimensionCache.toString()));
092                                break;
093                        default:
094                                dimensionCache.append(token.getText());
095                                break;
096                        }
097                }
098                return Optional.ofNullable(dimensions.stream().mapToInt(i -> i).toArray());
099        }
100
101        /** Constructor for ArrayContext additionally calculates capacities */
102        private ArrayContext(int[] dimensions, List<IToken> initializationTokens, EArrayType type) {
103                this.dimensions = dimensions;
104                this.capacity = calculateCapacity();
105                this.dimensionCapacities = calculateDimensionCapacities();
106                this.initializationTokens = initializationTokens;
107                this.type = type;
108        }
109
110        /**
111         * Calculates how many elements fit into the whole array
112         * 
113         * int arr[4][2][3] -> 24
114         * 
115         */
116        private int calculateCapacity() {
117                int capacity = 1;
118                for (int dimension : dimensions) {
119                        capacity *= dimension;
120                }
121                return capacity;
122        }
123
124        /**
125         * Calculates how many elements fit into a dimension of the array.
126         * 
127         * int arr[4][2][3] -> [6,3,1]
128         * 
129         */
130        private int[] calculateDimensionCapacities() {
131                int[] dimensionCapacities = new int[dimensions.length];
132                if (dimensionCapacities.length > 0) {
133                        dimensionCapacities[0] = capacity / dimensions[0];
134                        for (int i = 1; i < dimensions.length; ++i) {
135                                dimensionCapacities[i] = dimensionCapacities[i - 1] / dimensions[i];
136                        }
137                }
138                return dimensionCapacities;
139        }
140
141        /** Maps an offset to the corresponding indexes */
142        public int[] toIndexes(int offset) {
143                int[] indexes = new int[dimensionality()];
144
145                for (int i = 0; i < dimensionality(); i++) {
146                        indexes[i] = offset / dimensionCapacities[i];
147                        offset -= indexes[i] * dimensionCapacities[i];
148                }
149                return indexes;
150        }
151
152        /** Maps indexes to the corresponding offset */
153        public int toOffset(int[] indexes) {
154                int offset = 0;
155                for (int i = 0; i < indexes.length; ++i) {
156                        offset += (dimensionCapacities[i] * indexes[i]);
157                }
158                return offset;
159        }
160
161        /** @see #initializationTokens */
162        public List<IToken> getInitializationTokens() {
163                return initializationTokens;
164        }
165
166        /** @see #capacity */
167        public int getCapacity() {
168                return capacity;
169        }
170
171        /** returns the size of a certain dimension */
172        public int getDimension(int dimensionIndex) {
173                if (dimensionIndex < 0 || dimensionality() <= dimensionIndex) {
174                        return 0;
175                }
176                return dimensions[dimensionIndex];
177        }
178
179        /** returns the number of dimensions */
180        public int dimensionality() {
181                return dimensions.length;
182        }
183
184        /** Return all tokens corresponding to same compound like startToken */
185        public List<IToken> getCompoundTokens(IToken startToken) {
186                int startIndex = getInitializationTokens().indexOf(startToken);
187                EnumSet<ETokenType> endTypes = ETokenType.getTokenTypesByClass(DELIMITER);
188                endTypes.remove(COMMA);
189                int endIndex = TokenStreamUtils.firstTokenOfType(getInitializationTokens(), startIndex + 1,
190                                endTypes.toArray(new ETokenType[0]));
191                if (getInitializationTokens().get(endIndex - 1).getType() == COMMA) {
192                        endIndex--;
193                }
194                return getInitializationTokens().subList(startIndex, endIndex);
195
196        }
197
198        /** @see #type */
199        public EArrayType getType() {
200                return type;
201        }
202}
