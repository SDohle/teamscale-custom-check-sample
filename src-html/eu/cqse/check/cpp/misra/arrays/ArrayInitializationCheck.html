006package eu.cqse.check.cpp.misra.arrays;
007
008import static eu.cqse.check.framework.core.CheckGroupNames.BAD_PRACTICE;
009import static eu.cqse.check.framework.core.ECheckParameter.ABSTRACT_SYNTAX_TREE;
010import static eu.cqse.check.framework.core.EFindingEnablement.RED;
011import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
012import static eu.cqse.check.framework.scanner.ELanguage.CPP;
013import static eu.cqse.check.framework.shallowparser.SubTypeNames.GLOBAL_VARIABLE;
014import static eu.cqse.check.framework.shallowparser.SubTypeNames.LOCAL_VARIABLE;
015
016import java.util.Arrays;
017import java.util.List;
018import java.util.Optional;
019import java.util.stream.Collectors;
020
021import eu.cqse.check.base.EntityCheckBase;
022import eu.cqse.check.framework.core.ACheck;
023import eu.cqse.check.framework.core.CheckException;
024import eu.cqse.check.framework.core.phase.ECodeViewOption;
025import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
026import eu.cqse.check.framework.scanner.IToken;
027import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
028import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
029
030/**
031 * Due to its expense this check combines three MISRA rules. First it parses the
032 * array declaration and initialization. Then it checks whether : - the array is
033 * partial initialized - any element is initialized several times - every sub
034 * object is enclosed by braces
035 */
036@ACheck(name = "Use of enclosing braces, partial array initialization, multiple initialization of array elements", description = ArrayInitializationCheck.DESCRIPTION, groupName = BAD_PRACTICE, defaultEnablement = RED, languages = {
037                CPP }, parameters = { ABSTRACT_SYNTAX_TREE })
038@ACheckRuleset(ruleId = "9.2", ruleset = MISRA)
039@ACheckRuleset(ruleId = "9.3", ruleset = MISRA)
040@ACheckRuleset(ruleId = "9.4", ruleset = MISRA)
041public class ArrayInitializationCheck extends EntityCheckBase {
042
043        protected static final String DESCRIPTION = "1.\n\t"
044                        + "The use of enclosing braces is expected for every subobject.\n" + "2.\n\t"
045                        + "     Partial initialization of arrays is not allowed.\n" + "3.\n\t"
046                        + "     Any element of arrays should only be initialized once.\n" + "\n" + "Example 1:\n" + "``` cpp\n"
047                        + "uint16_t arr1[2][2]={{0},[1][0]=0,1}; //not acceptable\n" + "\n"
048                        + "uint16_t arr2[2][2]={{0,1},{2,3}}; //acceptable\n"
049                        + "uint16_t arr3[2][2]={{0},[1][1]=1,[1][0]=0}; //acceptable\n" + "```\n" + "\n" + "Example 2: \n"
050                        + "``` cpp\n" + "uint16_t arr1[2][2]={{0,1}}; //not acceptable\n"
051                        + "uint16_t arr1[2][2]={{0,1},[1][0]=2}; //not acceptable\n" + "\n"
052                        + "uint16_t arr3[2][2]={{0,1},{0}}; //acceptable\n"
053                        + "uint16_t arr4[2][2]={[1][1]=1,[1][0]=0}; //acceptable\n" + "```\n" + "\n" + "Example 3: \n" + "``` cpp\n"
054                        + "uint16_t arr4[2][2]={[0][0]=1,[0][0]=2}; //not acceptable\n" + "\n"
055                        + "uint16_t arr4[2][2]={[0][0]=1,[1][1]=2}; //acceptable\n" + "```";
056
057        /** {@inheritDoc} */
058        @Override
059        protected String getXPathSelectionString() {
060                return "//*[subtype('" + LOCAL_VARIABLE + "') | subtype('" + GLOBAL_VARIABLE + "')]";
061        }
062
063        @Override
064        protected ECodeViewOption getCodeViewOption() {
065                return ECodeViewOption.FILTERED_PREPROCESSED;
066        }
067
068        /** {@inheritDoc} */
069        @Override
070        protected void processEntity(ShallowEntity entity) throws CheckException {
071                Optional<ArrayContext> arrayContext = ArrayContext.createArrayContext(entity);
072                if (arrayContext.isPresent()) {
073                        checkArray(arrayContext.get(), entity);
074                }
075        }
076
077        /** Parses the array and creates findings if necessary. */
078        private void checkArray(ArrayContext arrayContext, ShallowEntity entity) throws CheckException {
079                Optional<ArrayAutomaton> parsedAutomation = ArrayAutomaton.parse(arrayContext);
080                if (parsedAutomation.isPresent()) {
081                        ArrayAutomaton automaton = parsedAutomation.get();
082                        checkPartialInitialization(automaton, entity);
083                        checkMultipleInitialization(automaton, entity);
084                        checkEnclosingBraces(automaton, arrayContext);
085                } else {
086                        createFinding("Array definition `" + entity.getName() + "` is not valid", entity.getStartLine(),
087                                        entity.getEndLine());
088                }
089        }
090
091        /**
092         * Creates finding if Rule 9.3 is violated: array is partial initialized
093         */
094        private void checkPartialInitialization(ArrayAutomaton automaton, ShallowEntity entity) throws CheckException {
095                if (automaton.isFullyInitialized()) {
096                        return;
097                }
098                createFinding("Array `" + entity.getName() + "` should not be partially initialized", entity.getStartLine(),
099                                entity.getEndLine());
100        }
101
102        /**
103         * Creates finding if Rule 9.4 is violated: any element is initialized multiple
104         * times
105         */
106        private void checkMultipleInitialization(ArrayAutomaton automaton, ShallowEntity entity) throws CheckException {
107                if (!automaton.anyMultipleInitialization()) {
108                        return;
109                }
110                List<int[]> indexes = automaton.getMultipleInitializations();
111                List<String> indexTexts = indexes.stream().map(index -> "`" + Arrays.toString(index).replace(", ", "][") + "`")
112                                .collect(Collectors.toList());
113                String inferredMessageInfo = "Element";
114                if (indexes.size() > 1) {
115                        inferredMessageInfo += "s";
116                }
117                inferredMessageInfo += " " + String.join(", ", indexTexts);
118                createFinding(inferredMessageInfo + " should only be initialized once", entity.getStartLine(),
119                                entity.getEndLine());
120
121        }
122
123        /**
124         * Creates finding if Rule 9.2 is violated: subobject is not enclosed by braces
125         */
126        private void checkEnclosingBraces(ArrayAutomaton automaton, ArrayContext arrayContext) throws CheckException {
127                for (IToken token : automaton.getCompoundFindings()) {
128                        createCompoundFinding(arrayContext, token);
129                }
130        }
131
132        /** Create finding for compound without enclosing braces */
133        private void createCompoundFinding(ArrayContext arrayContext, IToken startToken) throws CheckException {
134                List<IToken> compoundTokens = arrayContext.getCompoundTokens(startToken);
135                String inferredMessage = "`{" + TokenStreamTextUtils.concatTokenTexts(compoundTokens) + "}`";
136                createFinding("The use of enclosing braces is expected for " + inferredMessage, startToken,
137                                compoundTokens.get(compoundTokens.size() - 1));
138        }
139
140}
