006package eu.cqse.check.cpp.misra.libraries;
007
008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009
010import java.util.Set;
011
012import org.conqat.lib.commons.collections.CollectionUtils;
013
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckGroupNames;
016import eu.cqse.check.framework.core.ECheckParameter;
017import eu.cqse.check.framework.core.EFindingEnablement;
018import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
019import eu.cqse.check.framework.scanner.ELanguage;
020
021/**
022 * Checks that no input or output method of stdio.h or wchar.h is used.
023 */
024@ACheck(name = "Forbidden Standard I/O-Functions", description = "The standard input and output functions defined in `<stdio.h>` and `<wchar.h>` should not be used.", groupName = CheckGroupNames.EMBEDDED_SOFTWARE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
025                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
026@ACheckRuleset(ruleId = "21.6", ruleset = MISRA)
027public class ForbiddenStandardIOFunctionsCheck extends UnwantedStatementsCheckBase {
028
029        /** A list of forbidden methods used in {@link #getUnwantedMethods}. */
030        public static final Set<String> FORBIDDEN_METHODS = CollectionUtils.asHashSet("fopen", "freopen", "fclose",
031                        "fflush", "setbuf", "setvbuf", "fwide", "fread", "fwrite", "ftell", "fgetpos", "fseek", "fsetpos", "rewind",
032                        "clearerr", "feof", "ferror", "perror", "remove", "rename", "tmpfile", "tmpnam", "fgetwc", "fgetws",
033                        "getwc", "getwchar", "fwscanf", "wscanf", "vfwscanf", "vwscanf", "fputwc", "fputws", "putwc", "putwchar",
034                        "fwprintf", "wprintf", "vfwprintf", "vwprintf", "ungetwc", "fgetc", "fgets", "fprintf", "fputc", "fputs",
035                        "fread", "fscanf", "fwrite", "getc", "getchar", "gets", "printf", "putc", "putchar", "puts", "scanf",
036                        "ungetc", "vfprintf", "vfscanf", "vprintf", "vscanf");
037
038        /** {@inheritDoc} */
039        @Override
040        protected Set<String> getUnwantedMethods() {
041                return FORBIDDEN_METHODS;
042        }
043
044}
