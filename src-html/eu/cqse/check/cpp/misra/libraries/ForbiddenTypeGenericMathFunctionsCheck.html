006package eu.cqse.check.cpp.misra.libraries;
007
008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009
010import java.util.Set;
011
012import org.conqat.lib.commons.collections.CollectionUtils;
013
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckGroupNames;
017import eu.cqse.check.framework.core.ECheckParameter;
018import eu.cqse.check.framework.core.EFindingEnablement;
019import eu.cqse.check.framework.core.phase.ECodeViewOption;
020import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
021import eu.cqse.check.framework.scanner.ELanguage;
022import eu.cqse.check.framework.scanner.ETokenType;
023
024/**
025 * Checks that no type-generic math functions provided by tgmath.h are used.
026 */
027@ACheck(name = "Forbidden type-generic math functions", description = "Any type-generic macros defined in `<tgmath.h>` should not be used, because undefined behaviour could occur", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
028                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
029@ACheckRuleset(ruleId = "21.11", ruleset = MISRA)
030public class ForbiddenTypeGenericMathFunctionsCheck extends UnwantedStatementsCheckBase {
031
032        /** A list of forbidden methods used in {@link #getUnwantedMethods}. */
033        public static final Set<String> FORBIDDEN_METHODS = CollectionUtils.asHashSet("fabs", "exp", "log", "pow", "sqrt",
034                        "sin", "cos", "tan", "asin", "acos", "atan", "sinh", "cosh", "tanh", "asinh", "acosh", "atanh", "atan2",
035                        "cbrt", "ceil", "copysign", "erf", "erfc", "exp2", "expm1", "fdim", "floor", "fma", "fmax", "fmin", "fmod",
036                        "frexp", "hypot", "ilogb", "ldexp", "lgamma", "llrint", "llround", "log10", "log1p", "log2", "logb",
037                        "lrint", "lround", "nearbyint", "nextafter", "nexttoward", "remainder", "remquo", "rint", "round",
038                        "scalbln", "scalbn", "tgamma", "trunc", "carg", "conj", "creal", "cimag", "cproj");
039
040        @Override
041        public void execute() throws CheckException {
042                if (!includesTgmath()) {
043                        return;
044                }
045
046                super.execute();
047        }
048
049        /** Returns whether the "tgmath.h" header is included. */
050        private boolean includesTgmath() throws CheckException {
051                return CollectionUtils.anyMatch(context.getTokens(ECodeViewOption.FILTERED),
052                                token -> token.getType() == ETokenType.PREPROCESSOR_INCLUDE && token.getText().contains("<tgmath.h>"));
053        }
054
055        @Override
056        protected Set<String> getUnwantedMethods() {
057                return FORBIDDEN_METHODS;
058        }
059
060}
