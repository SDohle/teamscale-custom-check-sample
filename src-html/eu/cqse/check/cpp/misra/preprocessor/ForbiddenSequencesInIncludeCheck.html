006package eu.cqse.check.cpp.misra.preprocessor;
007
008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009
010import java.util.Arrays;
011import java.util.EnumSet;
012import java.util.List;
013import java.util.stream.Collectors;
014
015import org.conqat.lib.commons.string.StringUtils;
016
017import eu.cqse.check.cpp.misra.TokenFilterCheckBase;
018import eu.cqse.check.framework.core.ACheck;
019import eu.cqse.check.framework.core.CheckGroupNames;
020import eu.cqse.check.framework.core.ECheckParameter;
021import eu.cqse.check.framework.core.EFindingEnablement;
022import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
023import eu.cqse.check.framework.scanner.ELanguage;
024import eu.cqse.check.framework.scanner.ETokenType;
025import eu.cqse.check.framework.scanner.IToken;
026
027/**
028 * Checks that included header names do not contain forbidden sequences.
029 */
030@ACheck(name = "Included header names should not contain `'`, `\"`, `\\`, `/*` or `//`", description = "Included header names should not contain `'`, `\"`, `\\`, `/*` or `//`", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.RED, languages = {
031                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
032@ACheckRuleset(ruleId = "20.2", ruleset = MISRA)
033public class ForbiddenSequencesInIncludeCheck extends TokenFilterCheckBase {
034
035        /** Only PREPROCESSOR_INCLUDE tokens have to be checked. */
036        private static final EnumSet<ETokenType> CHECKED_TOKENTYPES = EnumSet.of(ETokenType.PREPROCESSOR_INCLUDE);
037
038        /**
039         * List of forbidden sequences, which are not allowed to appear in include
040         * filenames.
041         */
042        public static final List<String> FORBIDDEN_SEQUENCES = Arrays.asList("'", "\"", "\\", "/*", "//");
043
044        /** {@inheritDoc} */
045        @Override
046        protected String getFindingMessage(IToken token) {
047                String includeFileName = getIncludeFileName(token);
048                String foundForbiddenSequences = FORBIDDEN_SEQUENCES.stream().filter(includeFileName::contains)
049                                .map(sequence -> "`" + sequence + "`").collect(Collectors.joining(", "));
050                return foundForbiddenSequences + " should not be part of `" + token.getText() + "`";
051        }
052
053        /** Returns the included filename */
054        private static String getIncludeFileName(IToken token) {
055            // this splits at whitespace that is not right after the #
056                String[] parts = token.getText().split("[^#\\s]\\s+", 2);
057                if (parts.length < 2) {
058                        return StringUtils.EMPTY_STRING;
059                }
060                String includeFileName = parts[1].trim();
061                return includeFileName.substring(1, includeFileName.length() - 1);
062        }
063
064        /** Checks whether the include filename contains forbidden sequences. */
065        private static boolean containsForbiddenSequences(String includeFileName) {
066                return FORBIDDEN_SEQUENCES.stream().anyMatch(includeFileName::contains);
067        }
068
069        /** {@inheritDoc} */
070        @Override
071        protected boolean isValidToken(IToken token) {
072                if (!CHECKED_TOKENTYPES.contains(token.getType())) {
073                        return true;
074                }
075                return !containsForbiddenSequences(getIncludeFileName(token));
076        }
077
078}
