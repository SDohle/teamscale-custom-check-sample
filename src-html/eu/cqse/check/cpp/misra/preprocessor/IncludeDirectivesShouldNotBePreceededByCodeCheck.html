006package eu.cqse.check.cpp.misra.preprocessor;
007
008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009
010import java.util.List;
011
012import eu.cqse.check.framework.core.ACheck;
013import eu.cqse.check.framework.core.CheckException;
014import eu.cqse.check.framework.core.CheckGroupNames;
015import eu.cqse.check.framework.core.CheckImplementationBase;
016import eu.cqse.check.framework.core.ECheckParameter;
017import eu.cqse.check.framework.core.EFindingEnablement;
018import eu.cqse.check.framework.core.phase.ECodeViewOption;
019import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
020import eu.cqse.check.framework.scanner.ELanguage;
021import eu.cqse.check.framework.scanner.ETokenType;
022import eu.cqse.check.framework.scanner.IToken;
023
024/** Checks that include statements are not preceded by code (Misra 20.1) */
025@ACheck(name = "Include statements may not be preceeded by code", description = "Code may only be preceeded by preprocessor directives or comments.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
026                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
027@ACheckRuleset(ruleId = "20.1", ruleset = MISRA)
028public class IncludeDirectivesShouldNotBePreceededByCodeCheck extends CheckImplementationBase {
029
030        /** {@inheritDoc} */
031        @Override
032        public void execute() throws CheckException {
033                List<IToken> tokens = context.getTokens(ECodeViewOption.FILTERED);
034                boolean hadCode = false;
035                for (IToken token : tokens) {
036                        if (!token.getText().startsWith("#") && !token.getText().startsWith("/")) {
037                                hadCode = true;
038                                continue;
039                        }
040                        if (hadCode && token.getType() == ETokenType.PREPROCESSOR_INCLUDE) {
041                                createFinding("Includes shall only be preceded by preprocessor statements or comments", token);
042                        }
043                }
044        }
045}
