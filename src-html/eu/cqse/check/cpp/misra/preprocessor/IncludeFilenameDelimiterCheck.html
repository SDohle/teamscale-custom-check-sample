006package eu.cqse.check.cpp.misra.preprocessor;
007
008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009
010import java.util.ArrayList;
011import java.util.EnumSet;
012import java.util.Hashtable;
013import java.util.List;
014import java.util.Map;
015import java.util.stream.Collectors;
016
017import org.conqat.lib.commons.string.StringUtils;
018
019import eu.cqse.check.framework.core.ACheck;
020import eu.cqse.check.framework.core.CheckException;
021import eu.cqse.check.framework.core.CheckGroupNames;
022import eu.cqse.check.framework.core.CheckImplementationBase;
023import eu.cqse.check.framework.core.ECheckParameter;
024import eu.cqse.check.framework.core.EFindingEnablement;
025import eu.cqse.check.framework.core.phase.ECodeViewOption;
026import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
027import eu.cqse.check.framework.scanner.ELanguage;
028import eu.cqse.check.framework.scanner.ETokenType;
029import eu.cqse.check.framework.scanner.IToken;
030import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
031import eu.cqse.check.framework.util.CppLanguageConstants;
032
033/**
034 * Checks that all includes use '<...>' or '\"...\"' as delimiter for header
035 * names. Therefore also defined macros have to be processed.
036 */
037@ACheck(name = "Included header names should be delimited by <...> or \"...\".", description = "Included header names should be delimited by `<...>` or `...`.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.RED, languages = {
038                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
039@ACheckRuleset(ruleId = "20.3", ruleset = MISRA)
040public class IncludeFilenameDelimiterCheck extends CheckImplementationBase {
041
042        /**
043         * Works like a cache to prevent unnecessarily calling
044         * {@link eu.cqse.check.framework.core.ICheckContext#getTokens()} to get all
045         * tokens.
046         */
047        private List<IToken> tokens;
048
049        /** {@inheritDoc} */
050        @Override
051        public void execute() throws CheckException {
052                tokens = context.getTokens(ECodeViewOption.FILTERED);
053
054                List<IToken> preprocessedIncludeTokens = getPreprocessedIncludeTokens();
055
056                for (IToken token : preprocessedIncludeTokens) {
057                        String[] parts = token.getText().split("[^#\\s]\\s", 2);
058                        if (parts.length < 2) {
059                                // some include we can not parse cleanly
060                                continue;
061                        }
062
063                        String includeFileName = parts[1].trim();
064                        if (!includeFileName.matches("<[^>]*>") && !includeFileName.matches("\"[^\"]*\"")) {
065                                createFinding("Included header names should be delimited by `<...>` or `\"...\"`",
066                                                token.getLineNumber() + 1);
067                        }
068                }
069        }
070
071        /** Returns a list of all include tokens. */
072        private List<IToken> getPreprocessedIncludeTokens() {
073                List<IToken> includeTokens = getTokensByType(ETokenType.PREPROCESSOR_INCLUDE);
074                List<IToken> directiveTokens = getTokensByType(ETokenType.PREPROCESSOR_DIRECTIVE);
075                includeTokens.addAll(findAndPreprocessIncludeTokens(directiveTokens));
076                return includeTokens;
077        }
078
079        /** Returns a list of tokens filtered by type. */
080        private List<IToken> getTokensByType(ETokenType type) {
081                return TokenStreamUtils.findAll(tokens, EnumSet.of(type)).stream().map(this::mapIndexToToken)
082                                .collect(Collectors.toList());
083        }
084
085        /** Maps index to token */
086        private IToken mapIndexToToken(int index) {
087                return tokens.get(index);
088        }
089
090        /** Returns a list of processed includes. */
091        private static List<IToken> findAndPreprocessIncludeTokens(List<IToken> directiveTokens) {
092                List<IToken> defineTokens = filterTokensByBeginning(directiveTokens, CppLanguageConstants.PREPROCESSOR_DEFINE);
093                Map<String, String> macroMapping = processDefineTokens(defineTokens);
094
095                List<IToken> includeTokens = filterTokensByBeginning(directiveTokens,
096                                CppLanguageConstants.PREPROCESSOR_INCLUDE);
097                return applyMacros(includeTokens, macroMapping);
098        }
099
100        /** Returns a list of tokens whose text starts with given prefix. */
101        private static List<IToken> filterTokensByBeginning(List<IToken> directiveTokens, String prefix) {
102                return directiveTokens.stream().filter(token -> token.getText().replaceAll("\\s", "").startsWith(prefix))
103                                .collect(Collectors.toList());
104        }
105
106        /** Processes the '#define' tokens to create a map for applying macros. */
107        private static Map<String, String> processDefineTokens(List<IToken> defineTokens) {
108                Hashtable<String, String> macroMapping = new Hashtable<>();
109                for (IToken token : defineTokens) {
110                        String[] parts = token.getText().split(StringUtils.SPACE, 3);
111                        if (parts.length == 3) {
112                                macroMapping.put(parts[1], parts[2].trim());
113                        }
114                }
115                return macroMapping;
116        }
117
118        /**
119         * Applies the macros and returns a processed version for all includeTokens.
120         */
121        private static List<IToken> applyMacros(List<IToken> includeTokens, Map<String, String> macroMapping) {
122                List<IToken> processedIncludeTokens = new ArrayList<>();
123
124                for (IToken token : includeTokens) {
125                        String[] wordsInToken = token.getText().split(StringUtils.SPACE);
126                        boolean getsConcatenated = wordsInToken.length > 2;
127
128                        String processedFileName = "";
129                        for (int i = 1; i < wordsInToken.length; i++) {
130                                String replacement = macroMapping.getOrDefault(wordsInToken[i], wordsInToken[i]);
131                                processedFileName += removeDoubleQuotesIfNecessary(replacement, getsConcatenated);
132                        }
133                        if (!processedFileName.isEmpty()) {
134                                processedIncludeTokens
135                                                .add(token.newToken(ETokenType.PREPROCESSOR_INCLUDE, token.getOffset(), token.getLineNumber(),
136                                                                CppLanguageConstants.PREPROCESSOR_INCLUDE + StringUtils.SPACE + processedFileName,
137                                                                token.getOriginId()));
138                        }
139
140                }
141                return processedIncludeTokens;
142        }
143
144        /**
145         * Removes trailing and leading double quotes if replacement gets concatenated.
146         */
147        private static String removeDoubleQuotesIfNecessary(String replacement, boolean getsConcatenated) {
148                if (getsConcatenated && replacement.startsWith("\"") && replacement.endsWith("\"")) {
149                        replacement = replacement.substring(1, replacement.length() - 1);
150                }
151                return replacement;
152        }
