006package eu.cqse.check.cpp.misra.preprocessor;
007
008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009
010import java.util.EnumSet;
011import java.util.HashSet;
012import java.util.List;
013import java.util.Set;
014import java.util.regex.Matcher;
015import java.util.regex.Pattern;
016
017import eu.cqse.check.framework.core.ACheck;
018import eu.cqse.check.framework.core.CheckException;
019import eu.cqse.check.framework.core.CheckGroupNames;
020import eu.cqse.check.framework.core.CheckImplementationBase;
021import eu.cqse.check.framework.core.ECheckParameter;
022import eu.cqse.check.framework.core.EFindingEnablement;
023import eu.cqse.check.framework.core.phase.ECodeViewOption;
024import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
025import eu.cqse.check.framework.scanner.ELanguage;
026import eu.cqse.check.framework.scanner.ETokenType;
027import eu.cqse.check.framework.scanner.IToken;
028import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
029
030/**
031 * Checks if a macro is called with arguments, which contain tokens looking like
032 * preprocessor directives.
033 * 
034 * This check has a few inaccuracies:
035 * 
036 * - we can only handle calls of macros that are defined in the current file
037 * (there is no token stream in which macros from headers are visible). To
038 * improve this, we would need to know which headers are included (costly
039 * without aggregation phase) and have extracted the macros defined in these
040 * headers in a CheckPhase.
041 * 
042 * - The C preprocessor allows use of macros only _after_ their definition. We
043 * would even report them before they may be used.
044 */
045@ACheck(name = "Macro arguments looking like preprocessor directives", description = "Macro arguments should not contain tokens that look like preprocessor directives.", groupName = CheckGroupNames.READABILITY, defaultEnablement = EFindingEnablement.RED, languages = {
046                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
047@ACheckRuleset(ruleId = "20.6", ruleset = MISRA)
048public class MacroArgumentWithDirectivesCheck extends CheckImplementationBase {
049
050        /**
051         * {@link Pattern} on the text of a token that matches on macro definitions that
052         * accept parameters. Match group 1 contains the macro name. This relies on the
053         * fact that the given text contains no line continuation characters.
054         * 
055         * The C Preprocessor defines that there must be no space between the macro name
056         * and the opening parenthesis.
057         */
058        private static final Pattern MACRO_MATCHING_PATTERN = Pattern.compile("#define\\s+(\\w+)\\(.*",
059                        Pattern.CASE_INSENSITIVE);
060
061        /** {@inheritDoc} */
062        @Override
063        public void execute() throws CheckException {
064                List<IToken> fileTokens = context.getTokens(ECodeViewOption.FILTERED);
065                Set<String> macrosWithParameters = findMacrosWithParameters(fileTokens);
066                for (int identifierIndex : TokenStreamUtils.allStartingIndicesOfTypeSequence(fileTokens, 0, fileTokens.size(),
067                                ETokenType.IDENTIFIER, ETokenType.LPAREN)) {
068                        if (macrosWithParameters.contains(fileTokens.get(identifierIndex).getText())) {
069                                int closingIndex = TokenStreamUtils.findMatchingClosingToken(fileTokens, identifierIndex + 2,
070                                                ETokenType.LPAREN, ETokenType.RPAREN);
071                                if (closingIndex == TokenStreamUtils.NOT_FOUND) {
072                                        continue;
073                                }
074                                int foundIndex = TokenStreamUtils.firstTokenOfType(fileTokens, identifierIndex, closingIndex,
075                                                ETokenType.PREPROCESSOR_DIRECTIVE);
076                                if (foundIndex == TokenStreamUtils.NOT_FOUND) {
077                                        continue;
078                                }
079                                createFinding(
080                                                "Token `" + fileTokens.get(foundIndex).getText()
081                                                                + "` in macro argument looks like preprocessor directive",
082                                                fileTokens.get(identifierIndex).getLineNumber(), fileTokens.get(closingIndex).getLineNumber());
083                        }
084                }
085        }
086
087        /**
088         * Scans the given tokens and returns the names of all defined macros that
089         * accept parameters.
090         */
091        private static Set<String> findMacrosWithParameters(List<IToken> fileTokens) {
092                Set<String> macroNames = new HashSet<>();
093                for (IToken preprocessorDirectiveToken : TokenStreamUtils.findAllTokens(fileTokens,
094                                EnumSet.of(ETokenType.PREPROCESSOR_DIRECTIVE))) {
095                        String directiveText = preprocessorDirectiveToken.getText();
096                        // replace line continuation characters
097                        directiveText = directiveText.replaceAll("\\\\\n", " ");
098                        Matcher matcher = MACRO_MATCHING_PATTERN.matcher(directiveText);
099                        if (matcher.find()) {
100                                macroNames.add(matcher.group(1));
101                        }
102                }
103                return macroNames;
104        }
105}
