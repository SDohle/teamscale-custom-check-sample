006package eu.cqse.check.cpp.misra.preprocessor;
007
008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009
010import java.util.EnumSet;
011import java.util.List;
012import java.util.Set;
013import java.util.Stack;
014import java.util.stream.Collectors;
015
016import org.conqat.lib.commons.collections.CollectionUtils;
017import org.conqat.lib.commons.collections.Pair;
018import org.conqat.lib.commons.collections.PairList;
019
020import eu.cqse.check.framework.core.ACheck;
021import eu.cqse.check.framework.core.CheckException;
022import eu.cqse.check.framework.core.CheckGroupNames;
023import eu.cqse.check.framework.core.CheckImplementationBase;
024import eu.cqse.check.framework.core.ECheckParameter;
025import eu.cqse.check.framework.core.EFindingEnablement;
026import eu.cqse.check.framework.core.phase.ECodeViewOption;
027import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
028import eu.cqse.check.framework.scanner.ELanguage;
029import eu.cqse.check.framework.scanner.ETokenType;
030import eu.cqse.check.framework.scanner.IToken;
031
032/**
033 * Checks if the sequence of preprocessor directives is valid per-file.
034 */
035@ACheck(name = PreprocessorDirectiveEndCheck.DESCRIPTION, description = PreprocessorDirectiveEndCheck.DESCRIPTION + ".", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.RED, languages = {
036                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
037@ACheckRuleset(ruleId = "20.14", ruleset = MISRA)
038public class PreprocessorDirectiveEndCheck extends CheckImplementationBase {
039
040        /** Check description and finding message. */
041        public static final String DESCRIPTION = "Every `#else`, `#elif` and `#endif` has to be in the same file with related `#if`, `#ifdef`, `#ifndef`";
042
043        /** OPENING and BRANCHING combined in an EnumSet for faster checking */
044        private static final EnumSet<EDirectiveType> OPENING_BRANCHING = EnumSet.of(EDirectiveType.OPENING,
045                        EDirectiveType.BRANCHING);
046
047        /** {@inheritDoc} */
048        @Override
049        public void execute() throws CheckException {
050                checkDirectives(getDirectiveTokens());
051        }
052
053        /** Returns a list of all preprocessor directives. */
054        private List<IToken> getDirectiveTokens() throws CheckException {
055                return context.getTokens(ECodeViewOption.FILTERED).stream()
056                                .filter(token -> token.getType() == ETokenType.PREPROCESSOR_DIRECTIVE).collect(Collectors.toList());
057        }
058
059        /** Creates a finding for all directives, which are not of the expected type. */
060        private void checkDirectives(List<IToken> directives) throws CheckException {
061                PairList<IToken, EDirectiveType> directivesAndTypes = directives.stream()
062                                .map(token -> Pair.createPair(token,
063                                                EDirectiveType
064                                                                .getDirectiveType(PreprocessorDirectiveUtils.getDirectiveCommand(token).orElse(null))))
065                                .collect(PairList.toPairList());
066
067                Stack<Pair<IToken, EDirectiveType>> stack = new Stack<>();
068                for (Pair<IToken, EDirectiveType> directiveAndType : directivesAndTypes) {
069                        handleDanglingDirectives(directiveAndType, stack);
070                }
071                handleUnclosedDirectives(stack);
072        }
073
074        /**
075         * Creates findings for all dangling BRANCHING or CLOSING directives which are
076         * not preceded by a corresponding OPENING or BRANCHING directive.
077         */
078        private void handleDanglingDirectives(Pair<IToken, EDirectiveType> directiveAndType,
079                        Stack<Pair<IToken, EDirectiveType>> stack) throws CheckException {
080                switch (directiveAndType.getSecond()) {
081                case CLOSING:
082                        while (!stack.isEmpty() && stack.peek().getSecond() != EDirectiveType.OPENING) {
083                                stack.pop();
084                        }
085                        if (stack.empty()) {
086                                createFinding(directiveAndType.getFirst());
087                        } else {
088                                stack.pop();
089                        }
090                        break;
091                case BRANCHING:
092                        if (stack.isEmpty() || !OPENING_BRANCHING.contains(stack.peek().getSecond())) {
093                                // add dummy opening
094                                stack.push(Pair.createPair(null, EDirectiveType.OPENING));
095                                createFinding(directiveAndType.getFirst());
096                        }
097                case OPENING:
098                        stack.push(directiveAndType);
099                        break;
100                default:
101                        break;
102                }
103
104        }
105
106        /**
107         * Creates findings for all tokens which remained in the stack, because they are
108         * unclosed OPENING or BRANCHING directives
109         */
110        private void handleUnclosedDirectives(Stack<Pair<IToken, EDirectiveType>> stack) throws CheckException {
111                while (!stack.isEmpty()) {
112                        Pair<IToken, EDirectiveType> entry = stack.pop();
113
114                        if (!stack.isEmpty() && stack.peek().getFirst() == null) {
115                                // Already created finding
116                                stack.pop();
117                                continue;
118                        }
119
120                        createFinding(entry.getFirst());
121                }
122
123        }
124
125        private void createFinding(IToken token) throws CheckException {
126                EDirectiveType type = EDirectiveType
127                                .getDirectiveType(PreprocessorDirectiveUtils.getDirectiveCommand(token).get());
128                switch (type) {
129                case OPENING:
130                        createFinding("There is no `#endif` found for `" + token.getText() + "`", token);
131                        break;
132                case BRANCHING:
133                case CLOSING:
134                        createFinding("There is no `#if`, `#ifdef` or `#ifndef` found for `" + token.getText() + "`", token);
135                        break;
136                default:
137                        break;
138                }
139        }
140
141        /** This enumeration describes the type of a preprocessor directive. */
142        private enum EDirectiveType {
143
144                /** EDirectiveType for opening directives */
145                OPENING("#if", "#ifdef", "#ifndef"),
146
147                /** EDirectiveType for branching directives */
148                BRANCHING("#else", "#elif"),
149
150                /** EDirectiveType for closing directives */
151                CLOSING("#endif"),
152
153                /** EDirectiveType for all other directives */
154                OTHER();
155
156                /** List of keywords, which corresponds to this type */
157                private final Set<String> keywords;
158
159                /** Constructor. Keywords must be lowercase. */
160                EDirectiveType(String... keywords) {
161                        this.keywords = CollectionUtils.asHashSet(keywords);
162                }
163
164                /**
165                 * Returns the directive type of a keyword. If the keyword is not found a OTHER
166                 * is returned.
167                 * 
168                 * @param keyword
169                 *            must be lowercase.
170                 */
171                static EDirectiveType getDirectiveType(String keyword) {
172                        for (EDirectiveType type : EDirectiveType.values()) {
173                                if (type.keywords.contains(keyword)) {
174                                        return type;
175                                }
176                        }
177                        return OTHER;
178                }
179        }
180}
