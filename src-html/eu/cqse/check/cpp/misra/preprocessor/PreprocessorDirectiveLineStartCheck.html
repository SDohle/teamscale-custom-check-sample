006package eu.cqse.check.cpp.misra.preprocessor;

008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009
010import java.util.EnumSet;
011import java.util.Set;
012
013import org.conqat.lib.commons.collections.CollectionUtils;
014
015import eu.cqse.check.cpp.misra.TokenFilterCheckBase;
016import eu.cqse.check.framework.core.ACheck;
017import eu.cqse.check.framework.core.CheckGroupNames;
018import eu.cqse.check.framework.core.ECheckParameter;
019import eu.cqse.check.framework.core.EFindingEnablement;
020import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
021import eu.cqse.check.framework.scanner.ELanguage;
022import eu.cqse.check.framework.scanner.ETokenType;
023import eu.cqse.check.framework.scanner.IToken;
024
025/**
026 * Checks if every line starting with '#' as token is a valid preprocessing
027 * directive.
028 */
029@ACheck(name = PreprocessorDirectiveLineStartCheck.DESCRIPTION, description = PreprocessorDirectiveLineStartCheck.DESCRIPTION + ".", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.RED, languages = {
030                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
031@ACheckRuleset(ruleId = "20.13", ruleset = MISRA)
032public class PreprocessorDirectiveLineStartCheck extends TokenFilterCheckBase {
033
034        /* package */ static final String DESCRIPTION = "Every line starting with '#' has to be a valid preprocessing directive";
035
036        /** List of valid preprocessor directives (lowercased). */
037        private static final Set<String> VALID_PREPROCESSOR_DIRECTIVES = CollectionUtils.asHashSet("#include", "#define",
038                        "#undef", "#if", "#ifdef", "#ifndef", "#endif", "#error", "#else", "#elif", "#pragma", "#import");
039
040        /** The two token types for preprocessor tokens. */
041        private static final EnumSet<ETokenType> PREPROCESSOR_TOKEN_TYPES = EnumSet.of(ETokenType.PREPROCESSOR_DIRECTIVE,
042                        ETokenType.PREPROCESSOR_INCLUDE);
043
044        /** Counter for the actual line to check only the first token of every line. */
045        private int line = -1;
046
047        /** {@inheritDoc} */
048        @Override
049        protected String getFindingMessage(IToken token) {
050                return DESCRIPTION + ".";
051        }
052
053        /** {@inheritDoc} */
054        @Override
055        protected boolean isValidToken(IToken token) {
056                if (token.getLineNumber() <= line) {
057                        return true;
058                }
059                line += 1;
060                return !PREPROCESSOR_TOKEN_TYPES.contains(token.getType()) || VALID_PREPROCESSOR_DIRECTIVES
061                                .contains(PreprocessorDirectiveUtils.getDirectiveCommand(token).orElse(null));
062        }
063
064}
