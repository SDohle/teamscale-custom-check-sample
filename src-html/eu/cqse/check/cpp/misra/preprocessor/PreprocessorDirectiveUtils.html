006package eu.cqse.check.cpp.misra.preprocessor;
007
008import java.util.Optional;
009import java.util.regex.Matcher;
010import java.util.regex.Pattern;
011
012import eu.cqse.check.framework.scanner.IToken;
013
014/**
015 * Implements utility functions for handling preprocessor directives
016 */
017public class PreprocessorDirectiveUtils {
018
019        private static final Pattern DIRECTIVE_PARSE_PATTERN = Pattern.compile("^\\s*#\\s*([a-zA-Z0-9]+)");
020
021        /**
022         * Extracts the lowercased directive command of a directive token (if any). For
023         * example "#define".
024         */
025        public static Optional<String> getDirectiveCommand(IToken token) {
026                Matcher matcher = DIRECTIVE_PARSE_PATTERN.matcher(token.getText());
027                if (!matcher.find()) {
028                        return Optional.empty();
029                }
030
031                return Optional.of('#' + matcher.group(1).toLowerCase());
032        }
033}
