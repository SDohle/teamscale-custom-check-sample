006package eu.cqse.check.cpp.misra.switches;

008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009
010import eu.cqse.check.base.EntityCheckBase;
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckException;
013import eu.cqse.check.framework.core.CheckGroupNames;
014import eu.cqse.check.framework.core.ECheckParameter;
015import eu.cqse.check.framework.core.EFindingEnablement;
016import eu.cqse.check.framework.core.phase.ECodeViewOption;
017import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
018import eu.cqse.check.framework.scanner.ELanguage;
019import eu.cqse.check.framework.shallowparser.SubTypeNames;
020import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
021import eu.cqse.check.framework.util.CppLanguageFeatureParser;
022
023/**
024 * Switch statements should have more than one path. Otherwise it is redundant.
025 */
026@ACheck(name = "Switch statement with single possible execution path", description = SwitchClauseNumberCheck.CHECK_DESCRIPTION, groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
027                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
028@ACheckRuleset(ruleId = "16.6", ruleset = MISRA)
029public class SwitchClauseNumberCheck extends EntityCheckBase {
030
031        /* package */ static final String CHECK_DESCRIPTION = "Switch statements that only have one execution path (e.g. a single `case` statement) are confusing, as they can be replaced with a simple `if`-statement instead.";
032
033        /** {@inheritDoc} */
034        @Override
035        protected String getXPathSelectionString() {
036                return "//*[subtype('" + SubTypeNames.SWITCH + "')]";
037        }
038
039        @Override
040        protected ECodeViewOption getCodeViewOption() {
041                return ECodeViewOption.FILTERED_PREPROCESSED;
042        }
043
044        /** {@inheritDoc} */
045        @Override
046        protected void processEntity(ShallowEntity entity) throws CheckException {
047                int validCases = 0;
048
049                boolean previousWasCase = false;
050                for (ShallowEntity shallowEntity : entity.getChildren()) {
051                        boolean isCaseOrDefault = CppLanguageFeatureParser.isCaseOrDefault(shallowEntity);
052                        if (!previousWasCase && isCaseOrDefault) {
053                                validCases++;
054                        }
055                        previousWasCase = isCaseOrDefault;
056                }
057
058                if (validCases < 2) {
059                        createFinding("`switch` has only one possible execution path", entity);
060                }
061
062        }
063}
