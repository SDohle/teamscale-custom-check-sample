006package eu.cqse.check.cpp.misra.switches;
007
008import static eu.cqse.check.framework.core.CheckGroupNames.READABILITY;
009import static eu.cqse.check.framework.core.ECheckParameter.ABSTRACT_SYNTAX_TREE;
010import static eu.cqse.check.framework.core.EFindingEnablement.RED;
011import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
012import static eu.cqse.check.framework.scanner.ELanguage.CPP;
013import static eu.cqse.check.framework.shallowparser.SubTypeNames.CASE;
014import static eu.cqse.check.framework.shallowparser.SubTypeNames.DEFAULT;
015import static eu.cqse.check.framework.shallowparser.SubTypeNames.SWITCH;
016import static eu.cqse.check.framework.shallowparser.framework.EShallowEntityType.META;
017
018import java.util.List;
019import java.util.stream.Collectors;
020
021import eu.cqse.check.base.EntityCheckBase;
022import eu.cqse.check.framework.core.ACheck;
023import eu.cqse.check.framework.core.CheckException;
024import eu.cqse.check.framework.core.phase.ECodeViewOption;
025import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
026import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
027
028/**
029 * Simply checks whether any label which is not first or last label is default.
030 */
031@ACheck(name = "Switch statement should have a default label as last switch label", description = "Switch statement should have a default label as last switch label.", groupName = READABILITY, defaultEnablement = RED, languages = {
032                CPP }, parameters = { ABSTRACT_SYNTAX_TREE })
033@ACheckRuleset(ruleId = "16.5", ruleset = MISRA)
034public class SwitchStatementDemandsDefaultLabelWellPlacedCheck extends EntityCheckBase {
035
036        /** {@inheritDoc} */
037        @Override
038        protected String getXPathSelectionString() {
039                return "//*[subtype('" + SWITCH + "')]";
040        }
041
042        @Override
043        protected ECodeViewOption getCodeViewOption() {
044                return ECodeViewOption.FILTERED_PREPROCESSED;
045        }
046
047        /** {@inheritDoc} */
048        @Override
049        protected void processEntity(ShallowEntity entity) throws CheckException {
050                List<ShallowEntity> labels = entity.getChildrenOfType(META).stream()
051                                .filter(SwitchStatementDemandsDefaultLabelWellPlacedCheck::isLabel).collect(Collectors.toList());
052
053                if (labels.isEmpty()) {
054                        return;
055                }
056                for (int i = 0; i < labels.size() - 1; i++) {
057                        ShallowEntity label = labels.get(i);
058                        if (isDefaultLabel(label)) {
059                                createFinding("`default` should be the last statement within `switch`", label.getStartLine());
060                        }
061                }
062        }
063
064        private static boolean isDefaultLabel(ShallowEntity entity) {
065                return entity.getType().equals(META) && DEFAULT.equals(entity.getSubtype());
066        }
067
068        private static boolean isLabel(ShallowEntity entity) {
069                String subtype = entity.getSubtype();
070                return entity.getType().equals(META) && (DEFAULT.equals(subtype) || CASE.equals(subtype));
071        }
072
073}
