006package eu.cqse.check.cpp.misra.switches;
007
008import static eu.cqse.check.framework.core.ruleset.ERuleset.MISRA;
009
010import eu.cqse.check.base.EntityCheckBase;
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckException;
013import eu.cqse.check.framework.core.CheckGroupNames;
014import eu.cqse.check.framework.core.ECheckParameter;
015import eu.cqse.check.framework.core.EFindingEnablement;
016import eu.cqse.check.framework.core.phase.ECodeViewOption;
017import eu.cqse.check.framework.core.ruleset.ACheckRuleset;
018import eu.cqse.check.framework.scanner.ELanguage;
019import eu.cqse.check.framework.shallowparser.SubTypeNames;
020import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
021
022/**
023 * Checks that switch statements are well formed.
024 * 
025 * @see WellFormedCheckVisitor information what well-formed means
026 */
027@ACheck(name = "Switch statements should not contain unexpected constructs", description = "Switch statements should not contain unusual constructs, such as `case` statements in sub-conditions or `default` clauses before `case` clauses.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
028                ELanguage.CPP }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
029@ACheckRuleset(ruleId = "16.1", ruleset = MISRA)
030public class SwitchStatementIsWellFormedCheck extends EntityCheckBase {
031
032        /** {@inheritDoc} */
033        @Override
034        protected String getXPathSelectionString() {
035                return "//*[subtype('" + SubTypeNames.SWITCH + "')]";
036        }
037
038        @Override
039        protected ECodeViewOption getCodeViewOption() {
040                return ECodeViewOption.FILTERED_PREPROCESSED;
041        }
042
043        /** {@inheritDoc} */
044        @Override
045        protected void processEntity(ShallowEntity entity) throws CheckException {
046                WellFormedCheckVisitor visitor = new WellFormedCheckVisitor();
047                entity.traverse(visitor);
048                if (!visitor.isInExitState()) {
049                        createFinding(visitor.getErrorMessage(), visitor.getCurrentEntity());
050                }
051
052        }
053
054}
