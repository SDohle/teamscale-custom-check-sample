001package eu.cqse.check.cs;

003import eu.cqse.check.base.EntityFindingCheckBase;
004import eu.cqse.check.framework.core.ACheck;
005import eu.cqse.check.framework.core.CheckGroupNames;
006import eu.cqse.check.framework.core.ECheckParameter;
007import eu.cqse.check.framework.scanner.ELanguage;
008import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
009
010/**
011 * {@ConQAT.Doc}
012 * 
013 * @author $Author: kinnen $
014 * @version $Rev: 53185 $
015 * @ConQAT.Rating GREEN Hash: 1E6C9577054AC2129A8244EC796782EC
016 */
017@ACheck(name = "Abstract types should not have constructors", description = "Constructors "
018                + "on abstract types can be called only by derived types. Because public constructors create "
019                + "instances of a type, and you cannot create instances of an abstract type, an abstract type "
020                + "that has a public constructor is incorrectly designed.", groupName = CheckGroupNames.BAD_PRACTICE, languages = {
021                                ELanguage.CS }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
022public class AbstractTypesShouldNotHaveConstructorsCheck extends EntityFindingCheckBase {
023
024        /** {@inheritDoc} */
025        @Override
026        protected String getXPathSelectionString() {
027                return "//TYPE[modifiers('public', 'abstract')]/METHOD[subtype('constructor') and modifiers('public')]";
028        }
029
030        /** {@inheritDoc} */
031        @Override
032        protected String getFindingMessage(ShallowEntity entity) {
033                return "An abstract type must not have a public constructor.";
034        }
035}
