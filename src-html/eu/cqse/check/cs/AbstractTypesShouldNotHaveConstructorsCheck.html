001package eu.cqse.check.cs;
002
003import eu.cqse.check.base.EntityFindingCheckBase;
004import eu.cqse.check.framework.core.ACheck;
005import eu.cqse.check.framework.core.CheckGroupNames;
006import eu.cqse.check.framework.core.ECheckParameter;
007import eu.cqse.check.framework.scanner.ELanguage;
008import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
009
010/**
011 * {@ConQAT.Doc}
012 */
013@ACheck(name = "Abstract types should not have constructors", description = "Constructors "
014                + "on abstract types can be called only by derived types. Because public constructors create "
015                + "instances of a type, and you cannot create instances of an abstract type. An abstract type "
016                + "that has a public constructor is incorrectly designed.", groupName = CheckGroupNames.BAD_PRACTICE, languages = {
017                                ELanguage.CS }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
018public class AbstractTypesShouldNotHaveConstructorsCheck extends EntityFindingCheckBase {
019
020        /** {@inheritDoc} */
021        @Override
022        protected String getXPathSelectionString() {
023                return "//TYPE[modifiers('public', 'abstract')]/METHOD[subtype('constructor') and modifiers('public')]";
024        }
025
026        /** {@inheritDoc} */
027        @Override
028        protected String getFindingMessage(ShallowEntity entity) {
029                return "An abstract type must not have a public constructor.";
030        }
031}
