006package eu.cqse.check.cs;
007
008import eu.cqse.check.base.EntityFindingCheckBase;
009import eu.cqse.check.framework.core.ACheck;
010import eu.cqse.check.framework.core.CheckException;
011import eu.cqse.check.framework.core.CheckGroupNames;
012import eu.cqse.check.framework.core.ECheckParameter;
013import eu.cqse.check.framework.scanner.ELanguage;
014import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
015
016/**
017 * {@ConQAT.Doc}
018 */
019@ACheck(name = "Avoid partial classes (C#)", description = "Partial classes allow the class definitions to be "
020                + "broken up across multiple files. This is considered as a bad practice, because you will not be able "
021                + "to see the bigger picture of the class anymore and may not realize the side-effects you are introducing "
022                + "to the other partial files. By only viewing and working on a part of the class, you will not see all "
023                + "the internal fields that need updating, thus leading to incompatible changes and higher possibility of breaking "
024                + "code or dependencies in the rest of the class.", groupName = CheckGroupNames.BAD_PRACTICE, languages = {
025                                ELanguage.CS }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
026public class CsAvoidPartialClassesCheck extends EntityFindingCheckBase {
027
028        /**
029         * .xaml.cs files are partial due to framework constraints, so this should not
030         * create a finding.
031         */
032        private static final String WHITELISTED_SUFFIX = ".xaml.cs";
033
034        /** {@inheritDoc} */
035        @Override
036        protected String getFindingMessage(ShallowEntity entity) {
037                return "Partial classes should not be used";
038        }
039
040        /** {@inheritDoc} */
041        @Override
042        protected void processEntity(ShallowEntity entity) throws CheckException {
043                if (!context.getUniformPath().toLowerCase().endsWith(WHITELISTED_SUFFIX)) {
044                        super.processEntity(entity);
045                }
046        }
047
048        /** {@inheritDoc} */
049        @Override
050        protected String getXPathSelectionString() {
051                return "//TYPE[modifiers('partial')]";
052        }
053
054}
