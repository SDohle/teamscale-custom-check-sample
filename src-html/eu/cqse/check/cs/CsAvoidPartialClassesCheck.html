006package eu.cqse.check.cs;
007
008import eu.cqse.check.base.EntityFindingCheckBase;
009import eu.cqse.check.framework.core.ACheck;
010import eu.cqse.check.framework.core.CheckGroupNames;
011import eu.cqse.check.framework.core.ECheckParameter;
012import eu.cqse.check.framework.scanner.ELanguage;
013import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
014
015/**
016 * {@ConQAT.Doc}
017 */
018@ACheck(name = "Avoid partial classes (C#)", description = "Partial classes allow the class definitions to be "
019                + "broken up across multiple files. This is considered as a bad practice, because you will not be able "
020                + "to see the bigger picture of the class anymore and may not realize the side-effects you are introducing "
021                + "to the other partial files. By only viewing and working on a part of the class, you will not see all "
022                + "the internal fields that need updating, thus leading to incompatible changes and higher possibility of breaking "
023                + "code or dependencies in the rest of the class.", groupName = CheckGroupNames.BAD_PRACTICE, languages = {
024                                ELanguage.CS }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
025public class CsAvoidPartialClassesCheck extends EntityFindingCheckBase {
026
027        /** {@inheritDoc} */
028        @Override
029        protected String getFindingMessage(ShallowEntity entity) {
030                return "Partial classes must not be used.";
031        }
032
033        /** {@inheritDoc} */
034        @Override
035        protected String getXPathSelectionString() {
036                return "//TYPE[modifiers('partial')]";
037        }
038
039}
