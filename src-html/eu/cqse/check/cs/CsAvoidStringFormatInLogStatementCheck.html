006package eu.cqse.check.cs;
007
008import org.conqat.lib.commons.collections.UnmodifiableList;
009
010import eu.cqse.check.base.AvoidStringFormatInLogStatementsBase;
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckException;
013import eu.cqse.check.framework.core.CheckGroupNames;
014import eu.cqse.check.framework.core.ECheckParameter;
015import eu.cqse.check.framework.core.EFindingEnablement;
016import eu.cqse.check.framework.scanner.ELanguage;
017import eu.cqse.check.framework.scanner.ETokenType;
018import eu.cqse.check.framework.scanner.IToken;
019import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
020
021/**
022 * Checks, whether String Interpolation is used in log statements in C#-Code.
023 */
024@ACheck(name = "Avoid String interpolation in log statements (C#)", description = "Log statements should not include interpolated Strings, because they can lead to unnecessary background operations even for disabled log levels.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.OFF, languages = {
025                ELanguage.CS }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
026public class CsAvoidStringFormatInLogStatementCheck extends AvoidStringFormatInLogStatementsBase {
027        /** The message of the finding. */
028        private static final String FINDING_MESSAGE = "Avoid using String interpolation in log statements";
029
030        /** The Sign for calling String Interpolation in Cs */
031        private static final String STRING_INTERPOLATION_SIGN = "$\"";
032
033        /** The letter 'dollar' used in a String */
034        private static final String DOLLAR_AS_PART_OF_TEXT = "\\$\"";
035
036        /** {@inheritDoc} */
037        @Override
038        protected void processEntity(ShallowEntity entity) throws CheckException {
039                UnmodifiableList<IToken> entityTokens = entity.includedTokens();
040                for (IToken currentToken : entityTokens) {
041
042                        if (currentToken.getType() == ETokenType.DOUBLE_ARROW) {
043                                // expression contains a lamda
044                                return;
045                        }
046                        // Theoretically, the $ can be used as a symbol in a String
047                        if ((currentToken.getText().contains(STRING_INTERPOLATION_SIGN)
048                                        && !currentToken.getText().contains(DOLLAR_AS_PART_OF_TEXT))
049                                        || currentToken.getType() == ETokenType.PLUS) {
050                                createFindingOnFirstLine(FINDING_MESSAGE, entity);
051
052                        }
053
054                }
055
056        }
057
058}
