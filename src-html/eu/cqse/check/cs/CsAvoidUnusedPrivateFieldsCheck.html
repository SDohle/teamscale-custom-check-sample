008package eu.cqse.check.cs;

010import eu.cqse.check.clike.CLikeAvoidUnusedPrivateFieldsCheckBase;
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckGroupNames;
013import eu.cqse.check.framework.core.ECheckParameter;
014import eu.cqse.check.framework.core.EFindingEnablement;
015import eu.cqse.check.framework.scanner.ELanguage;
016import eu.cqse.check.framework.shallowparser.SubTypeNames;
017import eu.cqse.check.framework.util.LanguageFeatureParser;
018
019/**
020 * {@ConQAT.Doc}
021 * 
022 * @author $Author: heinemann $
023 * @version $Rev: 55557 $
024 * @ConQAT.Rating GREEN Hash: 68D00C7E99D8653751F852DDB5F1FF78

026@ACheck(name = "Avoid unused private fields (C#)", description = "This rule is reported when a "
027                + "private field in your code exists but is not used by any code path.", groupName = CheckGroupNames.UNUSED_CODE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
028                                ELanguage.CS }, parameters = {
029                                                ECheckParameter.ABSTRACT_SYNTAX_TREE,
030                                                ECheckParameter.TYPE_RESOLUTION })
031public class CsAvoidUnusedPrivateFieldsCheck
032                extends CLikeAvoidUnusedPrivateFieldsCheckBase {
033
034        /** Constructor. */
035        public CsAvoidUnusedPrivateFieldsCheck() {
036                super(LanguageFeatureParser.CS);
037        }
038
039        /** {@inheritDoc} */
040        @Override
041        protected String getProcessedEntitiesXPath() {
042                return "descendant::METHOD | descendant::ATTRIBUTE | descendant::STATEMENT[not(subtype('"
043                                + SubTypeNames.USING + "'))] | descendant::META[subtype('"
044                                + SubTypeNames.CASE + "')]";
045        }
046}
