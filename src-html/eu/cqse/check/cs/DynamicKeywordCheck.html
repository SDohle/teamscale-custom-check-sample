006package eu.cqse.check.cs;
007
008import java.util.StringJoiner;
009
010import eu.cqse.check.base.EntityFindingCheckBase;
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckException;
013import eu.cqse.check.framework.core.CheckGroupNames;
014import eu.cqse.check.framework.core.ECheckParameter;
015import eu.cqse.check.framework.core.EFindingEnablement;
016import eu.cqse.check.framework.scanner.ELanguage;
017import eu.cqse.check.framework.shallowparser.SubTypeNames;
018import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
019import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
020import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
021
022/**
023 * {@ConQAT.Doc}
024 */
025@ACheck(name = DynamicKeywordCheck.NAME, description = DynamicKeywordCheck.DESCRIPTION, groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
026                ELanguage.CS }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
027public class DynamicKeywordCheck extends EntityFindingCheckBase {
028
029        /** The XPath expression to filter the `using` directives. */
030        private static final String USING_DIRECTIVES = "//META[subtype('" + SubTypeNames.USING + "')]";
031
032        /**
033         * When using the Interop with Office, the dynamic keyword can be useful. This
034         * constant is used to check if the Interop is used in the examined source code
035         * file.
036         */
037        private static final String MICROSOFT_OFFICE_INTEROP = "Microsoft.Office.Interop";
038
039        /** The name of the check. */
040        public static final String NAME = "Avoid using dynamic keyword";
041
042        /** The message of the check. */
043        public static final String MESSAGE = "Avoid `dynamic` on ";
044
045        /** The findings description. */
046        public static final String DESCRIPTION = "The `dynamic` keyword makes the compiler bypass some compile-time type checks.\n"
047                        + "Instead these checks will be executed at run-time.\n"
048                        + "Therefore the user instead of the programmer will receive resulting type-error messages.";
049
050        /**
051         * The XPath selection string selecting all entities that include the `dynamic`
052         * keyword used in {@link DynamicKeywordCheck#getXPathSelectionString()}
053         */
054        public static final String XPATH_SELECTION_STRING;
055        static {
056                StringJoiner joiner = new StringJoiner("[modifiers('dynamic')]|//", "//", "[modifiers('dynamic')]");
057                for (EShallowEntityType type : EShallowEntityType.values()) {
058                        joiner.add(type.toString());
059                }
060                XPATH_SELECTION_STRING = joiner.toString();
061        }
062
063        /** {@inheritDoc} */
064        @Override
065        public void execute() throws CheckException {
066                if (!ignoreFile()) {
067                        super.execute();
068                }
069        }
070
071        /**
072         * Checks the file for `using` directives that include
073         * {@link DynamicKeywordCheck#MICROSOFT_OFFICE_INTEROP}. <br>
074         * When this directive is included, it is one of the exceptions where the
075         * `dynamic` keyword can be useful and it should not be checked.
076         * 
077         * @return true if there is a `using` directive for
078         *         {@link DynamicKeywordCheck#MICROSOFT_OFFICE_INTEROP}, false otherwise
079         */
080        private boolean ignoreFile() throws CheckException {
081                for (ShallowEntity entity : filterEntities(select(USING_DIRECTIVES))) {
082                        if (TokenStreamTextUtils.concatTokenTexts(entity.includedTokens()).contains(MICROSOFT_OFFICE_INTEROP)) {
083                                return true;
084                        }
085                }
086                return false;
087        }
088
089        /** {@inheritDoc} */
090        @Override
091        protected String getFindingMessage(ShallowEntity entity) {
092                return MESSAGE + entity.getName();
093        }
094
095        /** {@inheritDoc} */
096        @Override
097        protected String getXPathSelectionString() {
098                return XPATH_SELECTION_STRING;
099        }
100}
