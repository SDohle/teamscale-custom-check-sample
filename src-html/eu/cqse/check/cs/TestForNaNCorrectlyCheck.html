001package eu.cqse.check.cs;
002
003import static org.conqat.lib.commons.string.StringUtils.EMPTY_STRING;
004
005import java.util.EnumSet;
006import java.util.List;
007import java.util.Set;
008
009import org.conqat.lib.commons.collections.CollectionUtils;
010import org.conqat.lib.commons.string.StringUtils;
011
012import eu.cqse.check.base.EntityTokenCheckBase;
013import eu.cqse.check.framework.core.ACheck;
014import eu.cqse.check.framework.core.CheckException;
015import eu.cqse.check.framework.core.CheckGroupNames;
016import eu.cqse.check.framework.core.ECheckParameter;
017import eu.cqse.check.framework.core.EFindingEnablement;
018import eu.cqse.check.framework.scanner.ELanguage;
019import eu.cqse.check.framework.scanner.ETokenType;
020import eu.cqse.check.framework.scanner.IToken;
021import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
022import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
023
024/**
025 * {@ConQAT.Doc}
026 */
027@ACheck(name = "Test for NaN correctly", description = TestForNaNCorrectlyCheck.DESCRIPTION, defaultEnablement = EFindingEnablement.RED, groupName = CheckGroupNames.CORRECTNESS, languages = {
028                ELanguage.CS }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
029public class TestForNaNCorrectlyCheck extends EntityTokenCheckBase {
030
031        /** The finding description */
032        protected static final String DESCRIPTION = "Use `IsNaN` to determine whether a value is not a number. "
033                        + "It is not possible to determine whether a value is not a number by comparing it to another value. "
034                        + "For example, `Double.NaN` is not even equal to itself.`";
035
036        /**
037         * The token offset of a NaN sequence as left operand of a binary operator.
038         */
039        private static final int LEFT_OPERAND_OFFSET = -3;
040
041        /**
042         * The token offset of a NaN sequence as right operand of a binary operator.
043         */
044        private static final int RIGHT_OPERAND_OFFSET = 1;
045
046        /** Token text sequences that represent the NaN value. */
047        private static final Set<String[]> NAN_SEQUENCES = CollectionUtils.asHashSet(new String[] { "Single", ".", "NaN" },
048                        new String[] { "Double", ".", "NaN" });
049
050        /** {@inheritDoc} */
051        @Override
052        protected String getXPathSelectionString() {
053                return "//STATEMENT | //ATTRIBUTE";
054        }
055
056        /** Tests if the given tokens contain a test against NaN. */
057        @Override
058        protected void processTokens(List<IToken> tokens) throws CheckException {
059                List<Integer> operatorIndices = TokenStreamUtils.findAll(tokens, EnumSet.of(ETokenType.EQEQ, ETokenType.NOTEQ));
060
061                for (Integer operatorIndex : operatorIndices) {
062                        int leftOperandIndex = operatorIndex + LEFT_OPERAND_OFFSET;
063                        if (leftOperandIndex < 0 || !checkIndexForNaN(tokens, leftOperandIndex)) {
064                                int rightOperandIndex = operatorIndex + RIGHT_OPERAND_OFFSET;
065                                if (rightOperandIndex < tokens.size()) {
066                                        checkIndexForNaN(tokens, rightOperandIndex);
067                                }
068                        }
069                }
070        }
071
072        /**
073         * Checks and reports a finding, if one of the NaN text sequences is found at
074         * the given index.
075         */
076        private boolean checkIndexForNaN(List<IToken> tokens, int index) throws CheckException {
077                for (String[] nanSequence : NAN_SEQUENCES) {
078                        if (TokenStreamTextUtils.hasSequence(tokens, index, nanSequence)) {
079                                createFinding(
080                                                "Value should not be tested against `" + StringUtils.concat(nanSequence, EMPTY_STRING) + "`",
081                                                tokens.get(index));
082                                return true;
083                        }
084                }
085                return false;
086        }
087}
