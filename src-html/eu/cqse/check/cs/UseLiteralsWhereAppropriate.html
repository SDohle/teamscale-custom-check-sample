017package eu.cqse.check.cs;
018
019import org.conqat.lib.commons.collections.UnmodifiableList;
020
021import eu.cqse.check.base.EntityCheckBase;
022import eu.cqse.check.framework.core.ACheck;
023import eu.cqse.check.framework.core.CheckException;
024import eu.cqse.check.framework.core.CheckGroupNames;
025import eu.cqse.check.framework.core.ECheckParameter;
026import eu.cqse.check.framework.core.EFindingEnablement;
027import eu.cqse.check.framework.scanner.ELanguage;
028import eu.cqse.check.framework.scanner.ETokenType;
029import eu.cqse.check.framework.scanner.ETokenType.ETokenClass;
030import eu.cqse.check.framework.scanner.IToken;
031import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
032import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
033
034/**
035 * {@ConQAT.Doc}
036 */
037@ACheck(name = "Use literals where appropriate", description = "A field is declared static "
038                + "and readonly and is initialized with a value that is computable at compile time. "
039                + "Change the declaration to a const field so that the value is computed at compile "
040                + "time instead of at runtime. ", groupName = CheckGroupNames.PERFORMANCE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
041                                ELanguage.CS }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
042public class UseLiteralsWhereAppropriate extends EntityCheckBase {
043
044        /**
045         * Returns an XPath selection string that selects attributes with 'static'
046         * and 'readonly' modifiers. This matches fields, but neither properties nor
047         * events.
048         */
049        @Override
050        protected String getXPathSelectionString() {
051                return "//ATTRIBUTE[subtype('attribute') and modifiers('static', 'readonly')]";
052        }
053
054        /**
055         * Given an entity that is a static readonly field declaration, creates a
056         * finding if the field is initialized with a constant.
057         */
058        @Override
059        protected void processEntity(ShallowEntity entity) throws CheckException {
060                UnmodifiableList<IToken> tokens = entity.ownStartTokens();
061
062                // Look for an EQ token
063                int eqIndex = TokenStreamUtils.firstTokenOfType(tokens, ETokenType.EQ);
064                if (eqIndex == TokenStreamUtils.NOT_FOUND) {
065                        return;
066                }
067
068                // Check what the two tokens after EQ are
069                if (eqIndex + 2 >= tokens.size()) {
070                        return;
071                }
072                IToken nextToken = tokens.get(eqIndex + 1);
073                ETokenType nextTokenType = nextToken.getType();
074                if (nextTokenType.getTokenClass() != ETokenClass.LITERAL) {
075                        return;
076                }
077                if (nextToken.getText().charAt(0) == '$') {
078                        // A string literal starting with '$' is used for string
079                        // interpolation. (This is not a constant.)
080                        return;
081                }
082
083                IToken nextNextToken = tokens.get(eqIndex + 2);
084                if (nextNextToken.getType() == ETokenType.SEMICOLON) {
085                        createFindingOnFirstLine("Use literals where appropriate.", entity);
086                }
087        }
088
089}
