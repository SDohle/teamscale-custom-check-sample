001package eu.cqse.check.cs;
002
003import java.util.List;
004
005import eu.cqse.check.base.EntityCheckBase;
006import eu.cqse.check.framework.core.ACheck;
007import eu.cqse.check.framework.core.CheckException;
008import eu.cqse.check.framework.core.CheckGroupNames;
009import eu.cqse.check.framework.core.ECheckParameter;
010import eu.cqse.check.framework.scanner.ELanguage;
011import eu.cqse.check.framework.scanner.ETokenType;
012import eu.cqse.check.framework.scanner.IToken;
013import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
014import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
015import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
016
017/**
018 * {@ConQAT.Doc}
019 * 
020 * @author $Author: kinnen $
021 * @version $Rev: 53185 $
022 * @ConQAT.Rating GREEN Hash: 74338D0C6FD93704A9901F84ECD93EEF
023 */
024@ACheck(name = "Use params for variable arguments", description = "A public or "
025                + "protected type contains a public or protected method that uses the VarArgs "
026                + "calling convention. ", groupName = CheckGroupNames.BAD_PRACTICE, languages = { ELanguage.CS }, parameters = {
027                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
028public class UseParamsForVariableArgumentsCheck extends EntityCheckBase {
029
030        /** The identifier that is used for variable arguments. */
031        private static final String ARGLIST_IDENTIFIER = "__arglist";
032
033        /** {@inheritDoc} */
034        @Override
035        protected String getXPathSelectionString() {
036                return "//TYPE[anymodifier('public', 'protected')]/METHOD[anymodifier('public', 'protected')]";
037        }
038
039        /** {@inheritDoc} */
040        @Override
041        protected void processEntity(ShallowEntity entity) throws CheckException {
042                List<IToken> parameterTokens = TokenStreamUtils.tokensBetweenWithNesting(entity.ownStartTokens(),
043                                ETokenType.LPAREN, ETokenType.RPAREN);
044                int argListIndex = TokenStreamTextUtils.findFirst(parameterTokens, ARGLIST_IDENTIFIER);
045                if (argListIndex != TokenStreamUtils.NOT_FOUND) {
046                        createFinding("Use 'params' instead of '" + ARGLIST_IDENTIFIER + "' for variable arguments.",
047                                        parameterTokens.get(argListIndex));
048                }
049        }
050}
