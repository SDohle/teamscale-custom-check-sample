017package eu.cqse.check.framework.checktest;
018
019import java.io.File;
020import java.io.IOException;
021import java.util.ArrayList;
022import java.util.HashMap;
023import java.util.List;
024import java.util.Map;
025
026import junit.framework.TestSuite;
027
028import org.conqat.lib.commons.filesystem.ClassPathUtils;
029import org.conqat.lib.commons.filesystem.FileExtensionFilter;
030import org.conqat.lib.commons.filesystem.FileSystemUtils;
031import org.conqat.lib.commons.test.CCSMTestCaseBase;
032import org.junit.Ignore;
033
034import eu.cqse.check.framework.core.CheckInfo;
035import eu.cqse.check.framework.core.registry.CheckRegistry;
036
037/**
038 * Base class for data-driven check tests.
039 * 
040 * @author $Author: hummelb $
041 * @version $Revision: 54345 $
042 * @ConQAT.Rating GREEN Hash: FA1ED425015CB995D0756390DDA2D621
043 */
044@Ignore
045public abstract class CheckTestBase extends CCSMTestCaseBase {
046
047        /** Extension of the expected result files. */
048        /* package */static final String RESULT_EXTENSION = "expected";
049
050        /** Creates smoke test suite. */
051        protected TestSuite createSuite() throws IOException {
052                List<File> files = findFiles();
053                Map<String, CheckInfo> checkInfoBySimpleClassName = buildCheckInfoMap();
054
055                TestSuite suite = new TestSuite("Check tests [" + files.size()
056                                + " test files]");
057
058                for (int i = 0; i < files.size(); ++i) {
059                        suite.addTest(new CheckTestlet(files.get(i),
060                                        checkInfoBySimpleClassName));
061                }
062                return suite;
063        }
064
065        /**
066         * Builds a map from simple class names to check infos. The simple class
067         * name must be unique. The rational here is avoid classes with same name in
068         * different packages (confusing) but also make the test-data more readable.
069         */
070        private Map<String, CheckInfo> buildCheckInfoMap() throws IOException {
071                // we expect the checks to reside in the same class folder as the tests
072                return buildCheckInfoMap(getClass());
073        }
074
075        /**
076         * Builds a map from simple class names to check infos. The simple class
077         * name must be unique. The rational here is avoid classes with same name in
078         * different packages (confusing) but also make the test-data more readable.
079         * The clazz will be used to resolve the classpath to search for checks.
080         */
081        public static Map<String, CheckInfo> buildCheckInfoMap(Class<?> clazz)
082                        throws IOException {
083                // we expect the checks to reside in the same class folder as the tests
084                String classPath = ClassPathUtils.createClassPath(null, clazz);
085                CheckRegistry.getInstance().registerCheckClassDirectory(
086                                new File(classPath));
087
088                Map<String, CheckInfo> checkInfoBySimpleClassName = new HashMap<>();
089                for (CheckInfo checkInfo : CheckRegistry.getInstance().getChecksInfos()) {
090                        if (checkInfoBySimpleClassName.put(checkInfo.getSimpleClassName(),
091                                        checkInfo) != null) {
092                                fail("Duplicate simple class name: "
093                                                + checkInfo.getSimpleClassName());
094                        }
095                }
096                return checkInfoBySimpleClassName;
097        }
098
099        /** Returns all files that have the {@link #RESULT_EXTENSION}. */
100        private List<File> findFiles() {
101                List<File> result = new ArrayList<>();
102                for (File file : FileSystemUtils.listFilesRecursively(useTestFile("."),
103                                new FileExtensionFilter(RESULT_EXTENSION))) {
104                        result.add(file);
105                }
106                return result;
107        }
108}
