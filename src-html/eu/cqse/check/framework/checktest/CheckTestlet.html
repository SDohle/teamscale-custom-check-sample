017package eu.cqse.check.framework.checktest;
018
019import java.io.File;
020import java.io.IOException;
021import java.util.Map;
022
023import org.conqat.lib.commons.filesystem.FileSystemUtils;
024import org.conqat.lib.commons.string.StringUtils;
025import org.conqat.lib.commons.test.TestletBase;
026import org.junit.Ignore;
027
028import eu.cqse.check.framework.core.CheckException;
029import eu.cqse.check.framework.core.CheckInfo;
030import eu.cqse.check.framework.core.CheckInstance;
031
032/**
033 * Testlet for a single test of a check implementation.
034 * 
035 * @author $Author: kinnen $
036 * @version $Rev: 54565 $
037 * @ConQAT.Rating GREEN Hash: C58AF1BC947311C9450B70331E0B2719
038 */
039@Ignore
040public class CheckTestlet extends TestletBase {
041
042        /** File used as reference. */
043        private final File referenceFile;
044
045        /** Maps simple class names to the corresponding check info. */
046        private final Map<String, CheckInfo> checkInfoBySimpleClassName;
047
048        /** Constructor. */
049        /* package */ CheckTestlet(File referenceFile,
050                        Map<String, CheckInfo> checkInfoBySimpleClassName) {
051                this.referenceFile = referenceFile;
052                this.checkInfoBySimpleClassName = checkInfoBySimpleClassName;
053        }
054
055        /** Parse whole file and check against reference. */
056        @Override
057        public void test() throws IOException, CheckException {
058                File codeFile = new File(StringUtils.stripSuffix(
059                                referenceFile.getPath(), "." + CheckTestBase.RESULT_EXTENSION));
060                assertTrue("Could not read corresponding code file " + codeFile,
061                                codeFile.canRead() && codeFile.isFile());
062
063                // simple check name is the name of the parent directory
064                String checkSimpleName = referenceFile.getParentFile().getName();
065
066                String expectedFindingsString = StringUtils
067                                .normalizeLineBreaks(FileSystemUtils.readFile(referenceFile));
068                assertFindingsString(checkSimpleName, codeFile, expectedFindingsString);
069        }
070
071        /**
072         * Asserts that the check executed on the given code file returns the
073         * findings as described by the findings string.
074         */
075        public void assertFindingsString(String checkSimpleName, File codeFile,
076                        String expectedFindingsString) throws CheckException, IOException {
077                assertFindingsString(checkSimpleName,
078                                FileSystemUtils.normalizeSeparators(codeFile.getAbsolutePath()),
079                                StringUtils.replaceLineBreaks(
080                                                FileSystemUtils.readFile(codeFile), "\n"),
081                                expectedFindingsString);
082        }
083
084        /**
085         * Asserts that the check executed on the given code file returns the
086         * findings as described by the findings string.
087         */
088        public static void assertFindingsString(String checkSimpleName,
089                        File codeFile, String expectedFindingsString,
090                        Map<String, CheckInfo> checks) throws CheckException, IOException {
091                assertFindingsString(checkSimpleName,
092                                FileSystemUtils.normalizeSeparators(codeFile.getAbsolutePath()),
093                                StringUtils.replaceLineBreaks(
094                                                FileSystemUtils.readFile(codeFile), "\n"),
095                                expectedFindingsString, checks);
096        }
097
098        /**
099         * Asserts that the check executed on the given code returns the findings as
100         * described by the findings string.
101         */
102        public void assertFindingsString(String checkSimpleName, String uniformPath,
103                        String code, String expectedFindingsString) throws CheckException {
104                assertFindingsString(checkSimpleName, uniformPath, code,
105                                expectedFindingsString, checkInfoBySimpleClassName);
106        }
107
108        /**
109         * Asserts that the check executed on the given code returns the findings as
110         * described by the findings string.
111         */
112        public static void assertFindingsString(String checkSimpleName,
113                        String uniformPath, String code, String expectedFindingsString,
114                        Map<String, CheckInfo> checks) throws CheckException {
115                CheckInfo checkInfo = checks.get(checkSimpleName);
116                assertNotNull("No check with simple class name " + checkSimpleName
117                                + " found!", checkInfo);
118
119                CheckInstance instance = new CheckInstance(checkInfo);
120                TestCheckContext context = new TestCheckContext(uniformPath, code,
121                                checkInfo);
122                instance.initialize();
123                instance.setContext(context);
124                instance.execute();
125
126                assertEquals(expectedFindingsString, context.getFindingsString());
127        }
128
129        /** Name of the test case is the name of the smoke test file. */
130        @Override
131        public String getName() {
132                return referenceFile.getPath();
133        }
134}
