001package eu.cqse.check.framework.core;
002
003import java.util.List;
004
005import org.conqat.lib.commons.assertion.CCSMAssert;
006import org.conqat.lib.commons.logging.ILogger;
007import org.jaxen.BaseXPath;
008import org.jaxen.JaxenException;
009
010import eu.cqse.check.framework.core.xpath.CheckXPathUtils;
011import eu.cqse.check.framework.scanner.IToken;
012import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
013
014/**
015 * Abstract base class for custom checks. A check is instantiated and
016 * initialized with a check context when its executing processor is set up. It
017 * is destroyed, when the executing processor has finished execution.
018 */
019public abstract class CheckImplementationBase {
020
021        /**
022         * The check context that is used to access custom check specific
023         * functionality.
024         */
025        protected ICheckContext context = null;
026
027        /** The logger that can be used within a custom check. */
028        private ILogger logger = null;
029
030        /**
031         * Sets the context. Must not be called if the context has already been set.
032         */
033        public void setContext(ICheckContext context) {
034                CCSMAssert.isTrue(this.context == null, "The context has already been set.");
035                this.context = context;
036        }
037
038        /**
039         * Sets the logger. This method must only be called one time after the check
040         * was instantiated.
041         */
042        private void setLogger(ILogger logger) {
043                this.logger = logger;
044        }
045
046        /** Returns the logger. */
047        public ILogger getLogger() {
048                return logger;
049        }
050
051        /**
052         * Initializes the custom check. This is called after the parameters have
053         * been set, but before the check is executed.
054         */
055        @SuppressWarnings("unused")
056        public void initialize() throws CheckException {
057                // empty dummy implementation
058        }
059
060        /** Executes the custom check with the given context. */
061        public abstract void execute() throws CheckException;
062
063        /**
064         * Selects all shallow entities from the abstract syntax tree, that match
065         * the given xpath-expression. This method can be used, if the executing
066         * check requests the {@link ECheckParameter#ABSTRACT_SYNTAX_TREE}
067         * parameter.
068         */
069        protected List<ShallowEntity> select(String xPathExpression) throws CheckException {
070                return select(context.getRootEntity(), xPathExpression);
071        }
072
073        /**
074         * Selects all shallow entities starting from the given entity, that match
075         * the given xpath-expression. This method is still selecting entities from
076         * the whole AST as long as the xPath expression does not limit the
077         * selection to a special axis. For more information have a look at
078         * http://www.w3schools.com/xsl/xpath_syntax.asp
079         */
080        @SuppressWarnings("unchecked")
081        protected List<ShallowEntity> select(ShallowEntity entity, String xPathExpression) throws CheckException {
082                BaseXPath xPath = CheckXPathUtils.getXPath(xPathExpression);
083                try {
084                        Object result = xPath.evaluate(entity);
085                        if (result instanceof List<?>) {
086                                return (List<ShallowEntity>) result;
087                        }
088                        throw new CheckException(
089                                        "Evaluating the xPath expression '" + xPathExpression + "' returned unexpected result: " + result);
090                } catch (JaxenException e) {
091                        throw new CheckException(e);
092                }
093        }
094
095        /**
096         * Creates a finding with the given message for the whole analyzed token
097         * element.
098         */
099        protected void createFinding(String message) throws CheckException {
100                context.createFindingForElement(message);
101        }
102
103        /** Creates a finding with the given message at the given token. */
104        protected void createFinding(String message, IToken token) throws CheckException {
105                createFinding(message, token, token);
106        }
107
108        /**
109         * Creates a finding with the given message beginning from startToken to
110         * endToken.
111         */
112        protected void createFinding(String message, IToken startToken, IToken endToken) throws CheckException {
113                context.createFindingForOffsets(message, startToken.getOffset(), endToken.getEndOffset());
114
115        }
116
117        /** Creates a finding with the given message for the given entity. */
118        protected void createFinding(String message, ShallowEntity entity) throws CheckException {
119                context.createFindingForOffsets(message, entity.getStartOffset(), entity.getEndOffset());
120        }
121
122        /**
123         * Creates a finding with the given message on the first line of the given
124         * entity.
125         */
126        protected void createFindingOnFirstLine(String message, ShallowEntity entity) throws CheckException {
127                createFinding(message, entity.getStartLine());
128        }
129
130        /** Creates a finding with the given message at the given line. */
131        protected void createFinding(String message, int line) throws CheckException {
132                context.createFindingForLines(message, line, line);
133        }
134
135        /**
136         * Creates a finding with the given message from the given start line
137         * (inclusive) to the given end line (inclusive).
138         */
139        protected void createFinding(String message, int startLine, int endLine) throws CheckException {
140                context.createFindingForLines(message, startLine, endLine);
141        }
142
143        /** Creates an instance of the given check class. */
144        public static CheckImplementationBase createInstance(Class<?> checkClass, ILogger logger) throws CheckException {
145                CheckImplementationBase implementation = null;
146                try {
147                        implementation = (CheckImplementationBase) checkClass.newInstance();
148                } catch (InstantiationException e) {
149                        throw new CheckException("Check class " + checkClass.getName() + " does not provide a default constructor.",
150                                        e);
151                } catch (IllegalAccessException e) {
152                        throw new CheckException("Check class " + checkClass.getName() + "'s default constructor is not visible.",
153                                        e);
154                } catch (ClassCastException e) {
155                        throw new CheckException(
156                                        "Check class " + checkClass.getName() + " does not implement ICheckImplementation.", e);
157                }
158                implementation.setLogger(logger);
159                return implementation;
160        }
161}
