001package eu.cqse.check.framework.core;
002
003import java.util.List;
004
005import org.conqat.lib.commons.assertion.CCSMAssert;
006import org.jaxen.BaseXPath;
007import org.jaxen.JaxenException;
008
009import eu.cqse.check.framework.core.xpath.CheckXPathUtils;
010import eu.cqse.check.framework.scanner.IToken;
011import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
012
013/**
014 * Abstract base class for custom checks. A check is instantiated and
015 * initialized with a check context when its executing processor is set up. It
016 * is destroyed, when the executing processor has finished execution.
017 * 
018 * @author $Author: hummelb $
019 * @version $Rev: 55561 $
020 * @ConQAT.Rating GREEN Hash: 3C32B17AF80D764EA1705ED77F68A769
021 */
022public abstract class CheckImplementationBase {
023
024        /**
025         * The check context that is used to access custom check specific
026         * functionality.
027         */
028        protected ICheckContext context = null;
029
030        /**
031         * Sets the context. Must not be called if the context has already been set.
032         */
033        public void setContext(ICheckContext context) {
034                CCSMAssert.isTrue(this.context == null,
035                                "The context has already been set.");
036                this.context = context;
037        }
038
039        /** Initializes the custom check. */
040        public void initialize() {
041                // empty dummy implementation
042        }
043
044        /** Executes the custom check with the given context. */
045        public abstract void execute() throws CheckException;
046
047        /**
048         * Selects all shallow entities from the abstract syntax tree, that match
049         * the given xpath-expression. This method can be used, if the executing
050         * check requests the {@link ECheckParameter#ABSTRACT_SYNTAX_TREE}
051         * parameter.
052         */
053        protected List<ShallowEntity> select(String xPathExpression)
054                        throws CheckException {
055                return select(context.getRootEntity(), xPathExpression);
056        }
057
058        /**
059         * Selects all shallow entities within the given entity, that match the
060         * given xpath-expression.
061         */
062        @SuppressWarnings("unchecked")
063        protected List<ShallowEntity> select(ShallowEntity entity,
064                        String xPathExpression) throws CheckException {
065                BaseXPath xPath = CheckXPathUtils.getXPath(xPathExpression);
066                try {
067                        Object result = xPath.evaluate(entity);
068                        if (result instanceof List<?>) {
069                                return (List<ShallowEntity>) result;
070                        }
071                        throw new CheckException(
072                                        "Evaluating the xPath expression '" + xPathExpression
073                                                        + "' returned unexpected result: " + result);
074                } catch (JaxenException e) {
075                        throw new CheckException(e);
076                }
077        }
078
079        /**
080         * Creates a finding with the given message for the whole analyzed token
081         * element.
082         */
083        protected void createFinding(String message) throws CheckException {
084                context.createFindingForElement(message);
085        }
086
087        /** Creates a finding with the given message at the given token. */
088        protected void createFinding(String message, IToken token)
089                        throws CheckException {
090                createFinding(message, token, token);
091        }
092
093        /**
094         * Creates a finding with the given message beginning from startToken to
095         * endToken.
096         */
097        protected void createFinding(String message, IToken startToken,
098                        IToken endToken) throws CheckException {
099                context.createFindingForOffsets(message, startToken.getOffset(),
100                                endToken.getEndOffset());
101
102        }
103
104        /** Creates a finding with the given message for the given entity. */
105        protected void createFinding(String message, ShallowEntity entity)
106                        throws CheckException {
107                context.createFindingForOffsets(message, entity.getStartOffset(),
108                                entity.getEndOffset());
109        }
110
111        /**
112         * Creates a finding with the given message on the first line of the given
113         * entity.
114         */
115        protected void createFindingOnFirstLine(String message,
116                        ShallowEntity entity) throws CheckException {
117                createFinding(message, entity.getStartLine());
118        }
119
120        /** Creates a finding with the given message at the given line. */
121        protected void createFinding(String message, int line)
122                        throws CheckException {
123                context.createFindingForLines(message, line, line);
124        }
125
126        /**
127         * Creates a finding with the given message from the given start line
128         * (inclusive) to the given end line (inclusive).
129         */
130        protected void createFinding(String message, int startLine, int endLine)
131                        throws CheckException {
132                context.createFindingForLines(message, startLine, endLine);
133        }
134
135        /** Creates an instance of the given check class. */
136        public static CheckImplementationBase createInstance(Class<?> checkClass)
137                        throws CheckException {
138                CheckImplementationBase implementation = null;
139                try {
140                        implementation = (CheckImplementationBase) checkClass.newInstance();
141                } catch (InstantiationException e) {
142                        throw new CheckException("Check class " + checkClass.getName()
143                                        + " does not provide a default constructor.", e);
144                } catch (IllegalAccessException e) {
145                        throw new CheckException("Check class " + checkClass.getName()
146                                        + "'s default constructor is not visible.", e);
147                } catch (ClassCastException e) {
148                        throw new CheckException("Check class " + checkClass.getName()
149                                        + " does not implement ICheckImplementation.", e);
150                }
151                return implementation;
152        }
153}
