001package eu.cqse.check.framework.core;
002
003/**
004 * An enumeration of all parameters that can be passed to a custom check.
005 */
006public enum ECheckParameter {
007
008        /**
009         * Parameter for a list of shallow entities that represent the abstract syntax
010         * tree of the token element that is currently analyzed.
011         */
012        ABSTRACT_SYNTAX_TREE,
013
014        /**
015         * Parameter for type resolution that can be applied to shallow entities.
016         */
017        TYPE_RESOLUTION,
018
019        /**
020         * Parameter for using the Clang parser infrastructure.
021         */
022        CLANG;
023
024}
