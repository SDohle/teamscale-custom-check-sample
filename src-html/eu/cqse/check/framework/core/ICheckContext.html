001package eu.cqse.check.framework.core;
002
003import java.util.List;
004
005import org.conqat.lib.commons.logging.ILogger;
006
007import eu.cqse.check.framework.scanner.ELanguage;
008import eu.cqse.check.framework.scanner.IToken;
009import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
010import eu.cqse.check.framework.typetracker.ITypeResolution;
011
012/**
013 * Context interface for custom checks, that provides custom check related
014 * functionality to a custom check.
015 *
016 * @author $Author: hummelb $
017 * @version $Rev: 55686 $
018 * @ConQAT.Rating GREEN Hash: 5385C0DADD8284F653640939E6970285
019 */
020public interface ICheckContext {
021
022        /** Returns the language of the analyzed token element. */
023        public ELanguage getLanguage();
024
025        /** Returns the uniform path of the analyzed token element. */
026        public String getUniformPath();
027
028        /** Returns the tokens of the analyzed token element. */
029        public List<IToken> getTokens() throws CheckException;
030
031        /**
032         * Returns the text content of the analyzed token element. The returned
033         * content is filtered, e.g. generated code is <b>not</b> included. For most
034         * checks this will be the preferred form of the content, as usually no
035         * findings shall be generated for filtered code.
036         */
037        public String getTextContent() throws CheckException;
038
039        /**
040         * Returns the <b>unfiltered</b> text context of the analyzed token element.
041         * The returned content may include generated code. In general, checks
042         * should use {@link #getTextContent()} and only very specific checks should
043         * need to access the raw content.
044         */
045        public String getUnfilteredTextContent() throws CheckException;
046
047        /**
048         * Returns the synthetic root entity of the abstract syntax tree. This
049         * method can be used, if the executing check requests the
050         * {@link ECheckParameter#ABSTRACT_SYNTAX_TREE} parameter.
051         */
052        public ShallowEntity getRootEntity() throws CheckException;
053
054        /**
055         * Returns the synthetic root entity of the abstract syntax tree based on
056         * the unfiltered content (see also {@link #getUnfilteredTextContent()}).
057         * This method can be used, if the executing check requests the
058         * {@link ECheckParameter#ABSTRACT_SYNTAX_TREE} parameter.
059         */
060        public ShallowEntity getUnfilteredRootEntity() throws CheckException;
061
062        /**
063         * Returns the abstract syntax tree of the analyzed token element as list of
064         * shallow entities. This method can be used, if the executing check
065         * requests the {@link ECheckParameter#ABSTRACT_SYNTAX_TREE} parameter.
066         */
067        public List<ShallowEntity> getAbstractSyntaxTree() throws CheckException;
068
069        /**
070         * Returns a type resolution, which contains information about known
071         * variables and their types. This method can be used, if the executing
072         * check request the {@link ECheckParameter#TYPE_RESOLUTION} parameter.
073         */
074        public ITypeResolution getTypeResolution() throws CheckException;
075
076        /**
077         * Returns a logger, which may be used to indicate problems in situations
078         * where the check may continue. In more severe cases, a
079         * {@link CheckException} should be thrown instead.
080         */
081        public ILogger getLogger();
082
083        /**
084         * Creates a finding with the given message for the whole analyzed token
085         * element.
086         */
087        public void createFindingForElement(String message) throws CheckException;
088
089        /**
090         * Creates a finding with the given message from the given start line
091         * (inclusive) to the given end line (inclusive).
092         */
093        public void createFindingForLines(String message, int startLine,
094                        int endLine) throws CheckException;
095
096        /**
097         * Creates a finding with the given message from the given start offset
098         * (inclusive) to the given end offset (inclusive).
099         */
100        public void createFindingForOffsets(String message, int startOffset,
101                        int endOffset) throws CheckException;
102
103        /**
104         * Returns the abstract syntax tree based on the unfiltered text content.
105         * (see also {@link #getUnfilteredTextContent()})
106         */
107        public List<ShallowEntity> getUnfilteredAbstractSyntaxTree()
108                        throws CheckException;
109
110        /**
111         * Returns the type resolution based on the unfiltered text content. (see
112         * also {@link #getUnfilteredTextContent()})
113         */
114        public ITypeResolution getUnfilteredTypeResolution() throws CheckException;
115}
