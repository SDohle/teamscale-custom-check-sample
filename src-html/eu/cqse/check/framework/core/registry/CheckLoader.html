001package eu.cqse.check.framework.core.registry;
002
003import java.lang.reflect.Field;
004import java.util.HashMap;
005import java.util.Map;
006
007import org.conqat.lib.commons.assertion.CCSMAssert;
008import org.conqat.lib.commons.reflect.ReflectionUtils;
009
010import eu.cqse.check.framework.core.ACheck;
011import eu.cqse.check.framework.core.CheckException;
012import eu.cqse.check.framework.core.CheckImplementationBase;
013import eu.cqse.check.framework.core.CheckInfo;
014import eu.cqse.check.framework.core.ECheckParameter;
015import eu.cqse.check.framework.core.option.ACheckOption;
016import eu.cqse.check.framework.core.option.CheckOption;
017import eu.cqse.check.framework.scanner.ELanguage;
018import eu.cqse.check.framework.shallowparser.ShallowParserFactory;
019import eu.cqse.check.framework.typetracker.TypeTrackerFactory;
020
021/**
022 * Class for loading custom checks.
023 * 
024 * @author $Author: heinemann $
025 * @version $Rev: 55558 $
026 * @ConQAT.Rating GREEN Hash: F730CF1CC119D0EE64EFEDF34B2F860A
027 */
028public class CheckLoader {
029
030        /** The class to load a check from. */
031        private final Class<?> checkClass;
032
033        /** Constructor. */
034        private CheckLoader(Class<?> checkClass) {
035                CCSMAssert.isNotNull(checkClass);
036                this.checkClass = checkClass;
037        }
038
039        /**
040         * Tries to load a check info from the check class that was set in the
041         * constructor. Returns null, if the given check class is not annotated with
042         * {@link ACheck}.
043         * 
044         * @throws CheckException
045         *             if loading fails
046         */
047        private CheckInfo load() throws CheckException {
048                ACheck checkAnnotation = checkClass.getAnnotation(ACheck.class);
049                if (checkAnnotation == null) {
050                        return null;
051                }
052
053                CheckImplementationBase dummyInstance = CheckImplementationBase
054                                .createInstance(checkClass);
055
056                if (!languagesSupportParameters(checkAnnotation.languages(),
057                                checkAnnotation.parameters())) {
058                        throw new CheckException("Unsupported parameters!");
059                }
060
061                return new CheckInfo(checkClass, checkAnnotation.name(),
062                                checkAnnotation.description(), checkAnnotation.groupName(),
063                                checkAnnotation.defaultEnablement(),
064                                checkAnnotation.languages(), checkAnnotation.parameters(),
065                                loadOptions(dummyInstance));
066        }
067
068        /** Returns whether the given parameters support the given languages. */
069        private static boolean languagesSupportParameters(ELanguage[] languages,
070                        ECheckParameter[] parameters) {
071                for (ELanguage language : languages) {
072                        for (ECheckParameter parameter : parameters) {
073                                if (!languageSupportParameter(language, parameter)) {
074                                        return false;
075                                }
076                        }
077                }
078                return true;
079        }
080
081        /** Returns whether the given parameter supports the given language. */
082        private static boolean languageSupportParameter(ELanguage language,
083                        ECheckParameter parameter) {
084                if (ECheckParameter.ABSTRACT_SYNTAX_TREE.equals(parameter)) {
085                        return ShallowParserFactory.supportsLanguage(language);
086                } else if (ECheckParameter.TYPE_RESOLUTION.equals(parameter)) {
087                        return TypeTrackerFactory.supportsLanguage(language);
088                }
089                return false;
090        }
091
092        /**
093         * Load options for a check. The given dummy instance is used to retrieve
094         * default values.
095         */
096        private Map<String, CheckOption<?>> loadOptions(
097                        CheckImplementationBase dummyInstance) throws CheckException {
098                Map<String, CheckOption<?>> options = new HashMap<>();
099                for (Field attribute : ReflectionUtils.getAllFields(checkClass)) {
100                        CheckOption<?> option = getOptionFromField(attribute, dummyInstance);
101
102                        if (option == null) {
103                                continue;
104                        }
105
106                        if (options.put(option.getName(), option) != null) {
107                                throw new CheckException("Check " + checkClass.getName()
108                                                + " has duplicated option: " + option.getName());
109                        }
110
111                }
112                return options;
113        }
114
115        /**
116         * Returns a CheckOption for the given field. If the given field is not
117         * annotated with ACheckOption null is returned.
118         */
119        @SuppressWarnings({ "rawtypes", "unchecked" })
120        private CheckOption<?> getOptionFromField(Field field,
121                        CheckImplementationBase dummyInstance) throws CheckException {
122                field.setAccessible(true);
123                ACheckOption annotation = field.getAnnotation(ACheckOption.class);
124                if (annotation == null) {
125                        return null;
126                }
127
128                try {
129                        Object defaultValue = field.get(dummyInstance);
130                        if (defaultValue == null) {
131                                throw new CheckException("Check " + checkClass.getName()
132                                                + " must have default values for options.");
133                        }
134
135                        Class<?> type = field.getType();
136                        if (type.isPrimitive()) {
137                                // If the type is primitive we wrap it into a the
138                                // corresponding wrapper class, as the whole custom check
139                                // option framework works with non-primitive types only
140                                // Writing values back to the attribute works due to
141                                // auto-boxing.
142                                type = ReflectionUtils.resolvePrimitiveClass(type);
143                        }
144
145                        return new CheckOption(annotation, field, defaultValue, type);
146                } catch (IllegalAccessException e) {
147                        CCSMAssert
148                                        .fail("Could not access default value: " + e.getMessage());
149                }
150                return null;
151        }
152
153        /**
154         * Tries to load a check info from the given check class. Returns null, if
155         * the given check class is not annotated with {@link ACheck}.
156         * 
157         * @throws CheckException
158         *             if loading fails
159         */
160        public static CheckInfo loadFromClass(Class<?> checkClass)
161                        throws CheckException {
162                return new CheckLoader(checkClass).load();
163        }
164}
