001package eu.cqse.check.framework.core.registry;
002
003import java.lang.reflect.Field;
004import java.util.HashMap;
005import java.util.Map;
006
007import org.conqat.lib.commons.assertion.CCSMAssert;
008import org.conqat.lib.commons.logging.ILogger;
009import org.conqat.lib.commons.logging.SimpleLogger;
010import org.conqat.lib.commons.reflect.ReflectionUtils;
011
012import eu.cqse.check.framework.core.ACheck;
013import eu.cqse.check.framework.core.CheckException;
014import eu.cqse.check.framework.core.CheckImplementationBase;
015import eu.cqse.check.framework.core.CheckInfo;
016import eu.cqse.check.framework.core.ECheckParameter;
017import eu.cqse.check.framework.core.option.ACheckOption;
018import eu.cqse.check.framework.core.option.CheckOption;
019import eu.cqse.check.framework.scanner.ELanguage;
020import eu.cqse.check.framework.shallowparser.ShallowParserFactory;
021import eu.cqse.check.framework.typetracker.TypeTrackerFactory;
022
023/**
024 * Class for loading custom checks.
025 */
026public class CheckLoader {
027
028        /**
029         * Dummy logger that is used while creating check implementation base
030         * instances.
031         */
032        private static final ILogger LOGGER = new SimpleLogger();
033
034        /** The class to load a check from. */
035        private final Class<?> checkClass;
036
037        /** Constructor. */
038        private CheckLoader(Class<?> checkClass) {
039                CCSMAssert.isNotNull(checkClass);
040                this.checkClass = checkClass;
041        }
042
043        /**
044         * Tries to load a check info from the check class that was set in the
045         * constructor. Returns null, if the given check class is not annotated with
046         * {@link ACheck}.
047         * 
048         * @throws CheckException
049         *             if loading fails
050         */
051        private CheckInfo load() throws CheckException {
052                ACheck checkAnnotation = checkClass.getAnnotation(ACheck.class);
053                if (checkAnnotation == null) {
054                        return null;
055                }
056
057                CheckImplementationBase dummyInstance = CheckImplementationBase.createInstance(checkClass, LOGGER);
058
059                if (!languagesSupportParameters(checkAnnotation.languages(), checkAnnotation.parameters())) {
060                        throw new CheckException("Unsupported parameters!");
061                }
062
063                return new CheckInfo(checkClass, checkAnnotation.name(), checkAnnotation.description(),
064                                checkAnnotation.groupName(), checkAnnotation.defaultEnablement(), checkAnnotation.languages(),
065                                checkAnnotation.parameters(), loadOptions(dummyInstance));
066        }
067
068        /** Returns whether the given parameters support the given languages. */
069        private static boolean languagesSupportParameters(ELanguage[] languages, ECheckParameter[] parameters) {
070                for (ELanguage language : languages) {
071                        for (ECheckParameter parameter : parameters) {
072                                if (!languageSupportParameter(language, parameter)) {
073                                        return false;
074                                }
075                        }
076                }
077                return true;
078        }
079
080        /** Returns whether the given parameter supports the given language. */
081        private static boolean languageSupportParameter(ELanguage language, ECheckParameter parameter) {
082                if (ECheckParameter.ABSTRACT_SYNTAX_TREE.equals(parameter)) {
083                        return ShallowParserFactory.supportsLanguage(language);
084                } else if (ECheckParameter.TYPE_RESOLUTION.equals(parameter)) {
085                        return TypeTrackerFactory.supportsLanguage(language);
086                }
087                return false;
088        }
089
090        /**
091         * Load options for a check. The given dummy instance is used to retrieve
092         * default values.
093         */
094        private Map<String, CheckOption<?>> loadOptions(CheckImplementationBase dummyInstance) throws CheckException {
095                Map<String, CheckOption<?>> options = new HashMap<>();
096                for (Field attribute : ReflectionUtils.getAllFields(checkClass)) {
097                        CheckOption<?> option = getOptionFromField(attribute, dummyInstance);
098
099                        if (option == null) {
100                                continue;
101                        }
102
103                        if (options.put(option.getName(), option) != null) {
104                                throw new CheckException(
105                                                "Check " + checkClass.getName() + " has duplicated option: " + option.getName());
106                        }
107
108                }
109                return options;
110        }
111
112        /**
113         * Returns a CheckOption for the given field. If the given field is not
114         * annotated with ACheckOption null is returned.
115         */
116        @SuppressWarnings({ "rawtypes", "unchecked" })
117        private CheckOption<?> getOptionFromField(Field field, CheckImplementationBase dummyInstance)
118                        throws CheckException {
119                field.setAccessible(true);
120                ACheckOption annotation = field.getAnnotation(ACheckOption.class);
121                if (annotation == null) {
122                        return null;
123                }
124
125                try {
126                        Object defaultValue = field.get(dummyInstance);
127                        if (defaultValue == null) {
128                                throw new CheckException("Check " + checkClass.getName() + " must have default values for options.");
129                        }
130
131                        Class<?> type = field.getType();
132                        if (type.isPrimitive()) {
133                                // If the type is primitive we wrap it into a the
134                                // corresponding wrapper class, as the whole custom check
135                                // option framework works with non-primitive types only
136                                // Writing values back to the attribute works due to
137                                // auto-boxing.
138                                type = ReflectionUtils.resolvePrimitiveClass(type);
139                        }
140
141                        return new CheckOption(annotation, field, defaultValue, type);
142                } catch (IllegalAccessException e) {
143                        CCSMAssert.fail("Could not access default value: " + e.getMessage());
144                }
145                return null;
146        }
147
148        /**
149         * Tries to load a check info from the given check class. Returns null, if
150         * the given check class is not annotated with {@link ACheck}.
151         * 
152         * @throws CheckException
153         *             if loading fails
154         */
155        public static CheckInfo loadFromClass(Class<?> checkClass) throws CheckException {
156                return new CheckLoader(checkClass).load();
157        }
158}
