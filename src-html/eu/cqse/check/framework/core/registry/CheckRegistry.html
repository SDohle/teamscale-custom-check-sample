035/**
036 * Registry singleton that manages custom checks.
037 */
038public class CheckRegistry {
039
040        /** Logger instance. */
041        private static final Logger LOGGER = LogManager.getLogger();
042
043        /** The singleton instance. */
044        private static CheckRegistry instance = null;
045
046        /** Mapping from check identifiers to the check information. */
047        private final Map<String, CheckInfo> checksInfo = new HashMap<>();
048
049        private final Map<String, Class<? extends IGlobalExtractionPhase<?, ?>>> checkPhases = new HashMap<>();
050
051        /**
052         * Hidden constructor to avoid multiple instantiation, should be accessed via
053         * {@link #getInstance()} in application code. Test for this package may access
054         * the constructor.
055         */
056        /* package */ CheckRegistry() {
057                // do nothing
058        }
059
060        /** Returns the singleton instance. */
061        public static synchronized CheckRegistry getInstance() {
062                if (instance == null) {
063                        instance = new CheckRegistry();
064                }
065                return instance;
066        }
067
068        /**
069         * Registers custom checks from the given package. Classes are discovered via
070         * the service loader mechanism. Therefore the checks must be annotated with
071         * {@link Check}.
072         *
073         * Uses the current thread's class loader for loading the checks. This means the
074         * classes directory needs to be already on the class path.
075         */
076        public void registerChecksByServiceLoader(String fullPackageName) {
077                Iterable<Class<?>> classNames = ClassIndexUtils.getAnnotated(Check.class);
078                for (Class<?> className : classNames) {
079                        try {
080                                if (className.getPackage().getName().startsWith(fullPackageName)) {
081                                        loadCheckFromClass(className);
082                                }
083                        } catch (NoClassDefFoundError | IOException e) {
084                                LOGGER.error("Failed to load check from class: " + className, e);
085                        }
086                }
087        }
088
089        /**
090         * Registers the given bundle and loads all classes from the given bundle that
091         * implement {@link CheckImplementationBase} and are annotated with
092         * {@link Check}.
093         *
094         * Uses the current thread's class loader for loading the checks. This means the
095         * classes directory needs already on the class path.
096         */
097        public void registerChecksFromClasspathDirectory(File classesDirectory) {
098                ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
099                try {
100                        registerChecksFromClasspathDirectory(classesDirectory, classLoader);
101                } catch (IOException e) {
102                        LOGGER.error("Failed to load checks directory: " + classesDirectory, e);
103                }
104        }
105
106        /**
107         * Uses the provided {@link ClassLoader} for loading the checks.
108         *
109         * @see #registerChecksFromClasspathDirectory(File)
110         */
111        private void registerChecksFromClasspathDirectory(File classesDirectory, ClassLoader classLoader)
112                        throws IOException {
113                List<String> classNames = ClassPathUtils.getClassNames(classesDirectory);
114                for (String className : classNames) {
115                        try {
116                                loadCheckFromClass(classLoader.loadClass(className));
117                        } catch (ClassNotFoundException | NoClassDefFoundError e) {
118                                LOGGER.error("Failed to load check from class: " + className, e);
119                        }
120                }
121        }
122
123        /**
124         * Searches the given directory for Jar files or classes directories containing
125         * custom checks.
126         *
127         * This supports registering Jars as well as compiled classes in a 'classes'
128         * directory (e.g. from an Eclipse project). Both are searched recursively, but
129         * whenever a folder contains a classes directory scanning for Jars and further
130         * classes directories is skipped. The restriction results from jar files being
131         * placed e.g by Gradle in the 'build' output of source projects, which would
132         * then be picked up and lead to problems. The same applies to the classes
133         * folder in the Gradle build output.
134         *
135         * The found jars and classes directories will be put on the classpath (using a
136         * separate classloader).
137         */
138        public void registerCheckDirectory(File customChecksDirectory) {
139                if (!customChecksDirectory.canRead()) {
140                        LOGGER.error("Cannot read from custom check directory: " + customChecksDirectory.getAbsolutePath());
141                        return;
142                }
143
144                try {
145                        Files.walkFileTree(customChecksDirectory.toPath(), new CustomCheckLoader());
146                } catch (IOException e) {
147                        LOGGER.error("Cannot traverse custom check directory: " + customChecksDirectory.getAbsolutePath());
148                }
149        }
150
151        /** Loads custom checks from the given Jar file. */
152        public void loadChecksFromJar(File jarFile) {
153
154                try {
155                        // Don't use try-with-resources here, because we deliberately want the
156                        // URLClassLoader to remain unclosed.
157                        // https://docs.oracle.com/javase/7/docs/api/java/net/URLClassLoader.html#close()
158                        URLClassLoader classLoader = createClassLoader(jarFile);
159
160                        for (String className : FileSystemUtils.listTopLevelClassesInJarFile(jarFile)) {
161                                loadCheckFromClass(classLoader.loadClass(className));
162                        }
163                } catch (IOException | ClassNotFoundException e) {
164                        LOGGER.error("Error loading custom checks from jar '" + jarFile + "': " + e.toString(), e);
165                }
166        }
167
168        /** Loads custom checks from the given classes directory. */
169        private void loadChecksFromClasses(File classesDir) {
170                try {
171                        registerChecksFromClasspathDirectory(classesDir, createClassLoader(classesDir));
172                } catch (IOException e) {
173                        LOGGER.error("Error loading custom checks classes dir '" + classesDir + "': " + e.toString(), e);
174                }
175        }
176
177        /** Creates a class loader for the jar file or directory path. */
178        private static URLClassLoader createClassLoader(File classpath) throws MalformedURLException {
179                ClassLoader contextClassLoader = Thread.currentThread().getContextClassLoader();
180                return new URLClassLoader(new URL[] { classpath.toURI().toURL() }, contextClassLoader);
181        }
182
183        /**
184         * Creates and registers a custom check instance from the given class. If the
185         * given class does not contain a custom check (class not annotated with
186         * {@link Check}), this method does nothing.
187         */
188        private void loadCheckFromClass(Class<?> clazz) throws IOException {
189                CheckInfo checkInfo;
190                try {
191                        checkInfo = CheckLoader.loadFromClass(clazz);
192                } catch (CheckException e) {
193                        LOGGER.error(e.getMessage(), e);
194                        return;
195                }
196
197                if (checkInfo == null) {
198                        return;
199                }
200
201                if (checksInfo.containsKey(checkInfo.getIdentifier())) {
202                        LOGGER.error("Check " + checkInfo.getGroupName() + ":" + checkInfo.getName() + " is already registered.");
203                        return;
204                }
205                for (Class<? extends IGlobalExtractionPhase<?, ?>> phase : checkInfo.getRequiredPhases()) {
206                        Class<? extends IGlobalExtractionPhase<?, ?>> existingPhase = checkPhases.get(phase.getName());
207                        if (existingPhase != null && existingPhase != phase) {
208                                throw new IOException(
209                                                "Check phase '" + phase.getName() + "' registered twice via different classloaders.");
210                        }
211                        checkPhases.put(phase.getName(), phase);
212                }
213                checksInfo.put(checkInfo.getIdentifier(), checkInfo);
214        }
215
216        /**
217         * Returns the check phase class with the given full qualified name or
218         * <code>null</code> if no such phase exists.
219         */
220        public Class<? extends IGlobalExtractionPhase<?, ?>> getCheckPhase(String phaseName) {
221                return checkPhases.get(phaseName);
222        }
223
224        /**
225         * Creates check instances for the given check identifiers.
226         *
227         * @throws CheckException
228         *             if there is no check for one of the given identifiers
229         */
230        public Collection<CheckInstance> instantiateChecks(List<String> identifiers, ILogger logger) throws CheckException {
231                List<CheckInstance> instances = new ArrayList<>();
232                for (String identifier : identifiers) {
233                        if (identifier.equals(
234                                        "Code Anomalies#:#Readability#:#Switch statement should have a default label as last switch label")) {
235                                // Old check that has been removed in TS-19527, but the migration was not
236                                // drop-in, so drop-in upgraded 5.2.0-5.2.5 instances might be affected.
237                                // This can removed in 5.3.0+
238                                continue;
239                        }
240                        CheckInfo checkInfo = checksInfo.get(identifier);
241                        if (checkInfo == null) {
242                                logger.error("No check with identifier '" + identifier + "' exists, skipping...");
243                                continue;
244                        }
245                        instances.add(new CheckInstance(checkInfo, logger));
246                }
247                return instances;
248        }
249
250        /** Returns a list of all registered checks. */
251        public Collection<CheckInfo> getChecksInfos() {
252                return CollectionUtils.asUnmodifiable(checksInfo.values());
253        }
254
255        /**
256         * File visitor that scans a file tree and loads custom checks from classes
257         * directories and jar files.
258         */
259        private final class CustomCheckLoader extends SimpleFileVisitor<Path> {
260                /** {@inheritDoc} */
261                @Override
262                public FileVisitResult preVisitDirectory(Path dir, BasicFileAttributes attrs) {
263                        Path classes = dir.resolve("classes/java/main");
264                        if (Files.isDirectory(classes)) {
265                                LOGGER.info("Registering custom checks from classes: " + classes);
266                                loadChecksFromClasses(classes.toFile());
267                                return FileVisitResult.SKIP_SUBTREE;
268                        }
269                        classes = dir.resolve("classes");
270                        if (Files.isDirectory(classes)
271                                        && !FileSystemUtils.normalizeSeparators(classes.toString()).contains("/test/")) {
272                                LOGGER.info("Registering custom checks from classes: " + classes);
273                                loadChecksFromClasses(classes.toFile());
274                                return FileVisitResult.SKIP_SUBTREE;
275                        }
276                        return FileVisitResult.CONTINUE;
277                }
278
279                /** {@inheritDoc} */
280                @Override
281                public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) {
282                        if (file.toString().endsWith(".jar")) {
283                                LOGGER.info("Registering custom checks from jar: " + file);
284                                loadChecksFromJar(file.toFile());
285                        }
286                        return FileVisitResult.CONTINUE;
287                }
288        }
