001package eu.cqse.check.framework.core.registry;
002
003import java.io.File;
004import java.io.IOException;
005import java.net.URL;
006import java.net.URLClassLoader;
007import java.util.ArrayList;
008import java.util.Collection;
009import java.util.HashMap;
010import java.util.List;
011import java.util.Map;
012
013import org.apache.log4j.Logger;
014import org.conqat.lib.commons.collections.CollectionUtils;
015import org.conqat.lib.commons.filesystem.ClassPathUtils;
016import org.conqat.lib.commons.filesystem.FileExtensionFilter;
017import org.conqat.lib.commons.filesystem.FileSystemUtils;
018
019import eu.cqse.check.framework.core.ACheck;
020import eu.cqse.check.framework.core.CheckException;
021import eu.cqse.check.framework.core.CheckImplementationBase;
022import eu.cqse.check.framework.core.CheckInfo;
023import eu.cqse.check.framework.core.CheckInstance;
024
025/**
026 * Registry singleton that manages custom checks.
027 *
028 * @author $Author: kinnen $
029 * @version $Rev: 55127 $
030 * @ConQAT.Rating YELLOW Hash: 906DDC4A55D2ABE11B406490C215553C
031 */
032public class CheckRegistry {
033
034        /** Logger instance. */
035        private static final Logger LOGGER = Logger.getLogger(CheckRegistry.class);
036
037        /** The singleton instance. */
038        private static CheckRegistry instance = null;
039
040        /** Mapping from check identifiers to the check information. */
041        private final Map<String, CheckInfo> checksInfo = new HashMap<>();
042
043        /** Private constructor to avoid multiple instantiation. */
044        private CheckRegistry() {
045                // do nothing
046        }
047
048        /** Returns the singleton instance. */
049        public static synchronized CheckRegistry getInstance() {
050                if (instance == null) {
051                        instance = new CheckRegistry();
052                }
053                return instance;
054        }
055
056        /**
057         * Registers the given bundle and loads all classes from the given bundle
058         * that implement {@link CheckImplementationBase} and are annotated with
059         * {@link ACheck}.
060         */
061        public void registerCheckClassDirectory(File classesDirectory) {
062                ClassLoader classLoader = Thread.currentThread()
063                                .getContextClassLoader();
064                List<String> classNames = ClassPathUtils
065                                .getClassNames(classesDirectory);
066                for (String className : classNames) {
067                        try {
068                                loadCheckFromClass(classLoader.loadClass(className));
069                        } catch (ClassNotFoundException e) {
070                                LOGGER.error("Failed to load check from class: " + className,
071                                                e);
072                        }
073                }
074        }
075
076        /**
077         * Creates and registers a custom check instance from the given class. If
078         * the given class does not contain a custom check (class not annotated with
079         * {@link ACheck}), this method does nothing.
080         */
081        private void loadCheckFromClass(Class<?> clazz) {
082                CheckInfo checkInfo = null;
083                try {
084                        checkInfo = CheckLoader.loadFromClass(clazz);
085                } catch (CheckException e) {
086                        LOGGER.error(e.getMessage(), e);
087                        return;
088                }
089
090                if (checkInfo == null) {
091                        return;
092                }
093
094                if (checksInfo.containsKey(checkInfo.getIdentifier())) {
095                        LOGGER.error("Check " + checkInfo.getGroupName() + ":"
096                                        + checkInfo.getName() + " is already registered.");
097                        return;
098                }
099                checksInfo.put(checkInfo.getIdentifier(), checkInfo);
100        }
101
102        /**
103         * Creates check instances for the given check identifiers.
104         *
105         * @throws CheckException
106         *             if there is no check for one of the given identifiers
107         */
108        public Collection<CheckInstance> instantiateChecks(List<String> identifiers)
109                        throws CheckException {
110                List<CheckInstance> instances = new ArrayList<>();
111                for (String identifier : identifiers) {
112                        CheckInfo checkInfo = checksInfo.get(identifier);
113                        if (checkInfo == null) {
114                                throw new CheckException("No check with identifier '"
115                                                + identifier + "' exists.");
116                        }
117                        instances.add(new CheckInstance(checkInfo));
118                }
119                return instances;
120        }
121
122        /** Returns a list of all registered checks. */
123        public Collection<CheckInfo> getChecksInfos() {
124                return CollectionUtils.asUnmodifiable(checksInfo.values());
125        }
126
127        /** Registers the given directory of custom checks bundled as JAR files. */
128        public void registerCheckDirectory(File customChecksDirectory) {
129                if (!customChecksDirectory.canRead()) {
130                        LOGGER.error("Cannot read from custom check directory: "
131                                        + customChecksDirectory.getAbsolutePath());
132                        return;
133                }
134                List<File> jarFiles = FileSystemUtils.listFilesRecursively(
135                                customChecksDirectory, new FileExtensionFilter("jar"));
136                for (File jarFile : jarFiles) {
137                        if (jarFile.isFile()) {
138                                loadChecksFromJAR(jarFile);
139                        }
140                }
141
142        }
143
144        /** Loads custom checks from the given JAR file. */
145        private void loadChecksFromJAR(File jarFile) {
146                ClassLoader contextClassLoader = Thread.currentThread()
147                                .getContextClassLoader();
148                try (URLClassLoader classLoader = new URLClassLoader(
149                                new URL[] { jarFile.toURI().toURL() }, contextClassLoader)) {
150                        List<String> classes = FileSystemUtils
151                                        .listTopLevelClassesInJarFile(jarFile);
152                        for (String clazz : classes) {
153                                loadCheckFromClass(classLoader.loadClass(clazz));
154                        }
155                } catch (IOException | ClassNotFoundException e) {
156                        LOGGER.error("Error loading custom checks from '" + jarFile + "': "
157                                        + e.toString(), e);
158                }
159        }
160}
