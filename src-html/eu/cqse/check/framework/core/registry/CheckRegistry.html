001package eu.cqse.check.framework.core.registry;
002
003import java.io.File;
004import java.io.IOException;
005import java.net.MalformedURLException;
006import java.net.URL;
007import java.net.URLClassLoader;
008import java.nio.file.FileVisitResult;
009import java.nio.file.Files;
010import java.nio.file.Path;
011import java.nio.file.SimpleFileVisitor;
012import java.nio.file.attribute.BasicFileAttributes;
013import java.util.ArrayList;
014import java.util.Collection;
015import java.util.HashMap;
016import java.util.List;
017import java.util.Map;
018
019import org.apache.logging.log4j.LogManager;
020import org.apache.logging.log4j.Logger;
021import org.conqat.lib.commons.collections.CollectionUtils;
022import org.conqat.lib.commons.filesystem.ClassPathUtils;
023import org.conqat.lib.commons.filesystem.FileSystemUtils;
024import org.conqat.lib.commons.logging.ILogger;
025
026import eu.cqse.check.framework.core.ACheck;
027import eu.cqse.check.framework.core.CheckException;
028import eu.cqse.check.framework.core.CheckImplementationBase;
029import eu.cqse.check.framework.core.CheckInfo;
030import eu.cqse.check.framework.core.CheckInstance;
031
032/**
033 * Registry singleton that manages custom checks.
034 */
035public class CheckRegistry {
036
037        /** Logger instance. */
038        private static final Logger LOGGER = LogManager.getLogger();
039
040        /** The singleton instance. */
041        private static CheckRegistry instance = null;
042
043        /** Mapping from check identifiers to the check information. */
044        private final Map<String, CheckInfo> checksInfo = new HashMap<>();
045
046        /**
047         * Hidden constructor to avoid multiple instantiation, should be accessed via
048         * {@link #getInstance()} in application code. Test for this package may access
049         * the constructor.
050         */
051        /* package */ CheckRegistry() {
052                // do nothing
053        }
054
055        /** Returns the singleton instance. */
056        public static synchronized CheckRegistry getInstance() {
057                if (instance == null) {
058                        instance = new CheckRegistry();
059                }
060                return instance;
061        }
062
063        /**
064         * Registers the given bundle and loads all classes from the given bundle that
065         * implement {@link CheckImplementationBase} and are annotated with
066         * {@link ACheck}.
067         * 
068         * Uses the current thread's class loader for loading the checks. This means the
069         * classes directory needs already on the class path.
070         */
071        public void registerChecksFromClasspathDirectory(File classesDirectory) {
072                ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
073                registerChecksFromClasspathDirectory(classesDirectory, classLoader);
074        }
075
076        /**
077         * Uses the provided {@link ClassLoader} for loading the checks.
078         * 
079         * @see #registerChecksFromClasspathDirectory(File)
080         */
081        private void registerChecksFromClasspathDirectory(File classesDirectory, ClassLoader classLoader) {
082                List<String> classNames = ClassPathUtils.getClassNames(classesDirectory);
083                for (String className : classNames) {
084                        try {
085                                loadCheckFromClass(classLoader.loadClass(className));
086                        } catch (ClassNotFoundException | NoClassDefFoundError e) {
087                                LOGGER.error("Failed to load check from class: " + className, e);
088                        }
089                }
090        }
091
092        /**
093         * Searches the given directory for Jar files or classes directories containing
094         * custom checks.
095         * 
096         * This supports registering Jars as well as compiled classes in a 'classes'
097         * directory (e.g. from an Eclipse project). Both are searched recursively, but
098         * whenever a folder contains a classes directory scanning for Jars and further
099         * classes directories is skipped. The restriction results from jar files being
100         * placed e.g by Gradle in the 'build' output of source projects, which would
101         * then be picked up and lead to problems. The same applies to the classes
102         * folder in the Gradle build output.
103         * 
104         * The found jars and classes directories will be put on the classpath (using a
105         * separate classloader).
106         */
107        public void registerCheckDirectory(File customChecksDirectory) {
108                if (!customChecksDirectory.canRead()) {
109                        LOGGER.error("Cannot read from custom check directory: " + customChecksDirectory.getAbsolutePath());
110                        return;
111                }
112
113                try {
114                        Files.walkFileTree(customChecksDirectory.toPath(), new CustomCheckLoader());
115                } catch (IOException e) {
116                        LOGGER.error("Cannot traverse custom check directory: " + customChecksDirectory.getAbsolutePath());
117                }
118        }
119
120        /** Loads custom checks from the given Jar file. */
121        public void loadChecksFromJar(File jarFile) {
122
123                try {
124                        // Don't use try-with-resources here, because we deliberately want the
125                        // URLClassLoader to remain unclosed.
126                        // https://docs.oracle.com/javase/7/docs/api/java/net/URLClassLoader.html#close()
127                        URLClassLoader classLoader = createClassLoader(jarFile);
128
129                        for (String className : FileSystemUtils.listTopLevelClassesInJarFile(jarFile)) {
130                                loadCheckFromClass(classLoader.loadClass(className));
131                        }
132                } catch (IOException | ClassNotFoundException e) {
133                        LOGGER.error("Error loading custom checks from jar '" + jarFile + "': " + e.toString(), e);
134                }
135        }
136
137        /** Loads custom checks from the given classes directory. */
138        private void loadChecksFromClasses(File classesDir) {
139                try {
140                        registerChecksFromClasspathDirectory(classesDir, createClassLoader(classesDir));
141                } catch (MalformedURLException e) {
142                        LOGGER.error("Error loading custom checks classes dir '" + classesDir + "': " + e.toString(), e);
143                }
144        }
145
146        /** Creates a class loader for the jar file or directory path. */
147        private static URLClassLoader createClassLoader(File classpath) throws MalformedURLException {
148                ClassLoader contextClassLoader = Thread.currentThread().getContextClassLoader();
149                return new URLClassLoader(new URL[] { classpath.toURI().toURL() }, contextClassLoader);
150        }
151
152        /**
153         * Creates and registers a custom check instance from the given class. If the
154         * given class does not contain a custom check (class not annotated with
155         * {@link ACheck}), this method does nothing.
156         */
157        private void loadCheckFromClass(Class<?> clazz) {
158                CheckInfo checkInfo = null;
159                try {
160                        checkInfo = CheckLoader.loadFromClass(clazz);
161                } catch (CheckException e) {
162                        LOGGER.error(e.getMessage(), e);
163                        return;
164                }
165
166                if (checkInfo == null) {
167                        return;
168                }
169
170                if (checksInfo.containsKey(checkInfo.getIdentifier())) {
171                        LOGGER.error("Check " + checkInfo.getGroupName() + ":" + checkInfo.getName() + " is already registered.");
172                        return;
173                }
174                checksInfo.put(checkInfo.getIdentifier(), checkInfo);
175        }
176
177        /**
178         * Creates check instances for the given check identifiers.
179         *
180         * @throws CheckException
181         *             if there is no check for one of the given identifiers
182         */
183        public Collection<CheckInstance> instantiateChecks(List<String> identifiers, ILogger logger) throws CheckException {
184                List<CheckInstance> instances = new ArrayList<>();
185                for (String identifier : identifiers) {
186                        CheckInfo checkInfo = checksInfo.get(identifier);
187                        if (checkInfo == null) {
188                                throw new CheckException("No check with identifier '" + identifier + "' exists.");
189                        }
190                        instances.add(new CheckInstance(checkInfo, logger));
191                }
192                return instances;
193        }
194
195        /** Returns a list of all registered checks. */
196        public Collection<CheckInfo> getChecksInfos() {
197                return CollectionUtils.asUnmodifiable(checksInfo.values());
198        }
199
200        /**
201         * File visitor that scans a file tree and loads custom checks from classes
202         * directories and jar files.
203         */
204        private final class CustomCheckLoader extends SimpleFileVisitor<Path> {
205                /** {@inheritDoc} */
206                @Override
207                public FileVisitResult preVisitDirectory(Path dir, BasicFileAttributes attrs) {
208                        Path classes = dir.resolve("classes/java/main");
209                        if (Files.isDirectory(classes)) {
210                                LOGGER.info("Registering custom checks from classes: " + classes);
211                                loadChecksFromClasses(classes.toFile());
212                                return FileVisitResult.SKIP_SUBTREE;
213                        }
214                        classes = dir.resolve("classes");
215                        if (Files.isDirectory(classes)) {
216                                LOGGER.info("Registering custom checks from classes: " + classes);
217                                loadChecksFromClasses(classes.toFile());
218                                return FileVisitResult.SKIP_SUBTREE;
219                        }
220                        return FileVisitResult.CONTINUE;
221                }
222
223                /** {@inheritDoc} */
224                @Override
225                public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) {
226                        if (file.toString().endsWith(".jar")) {
227                                LOGGER.info("Registering custom checks from jar: " + file);
228                                loadChecksFromJar(file.toFile());
229                        }
230                        return FileVisitResult.CONTINUE;
231                }
232        }
233}
