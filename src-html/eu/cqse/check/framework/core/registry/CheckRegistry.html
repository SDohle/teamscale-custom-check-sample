001package eu.cqse.check.framework.core.registry;
002
003import java.io.File;
004import java.io.IOException;
005import java.net.URL;
006import java.net.URLClassLoader;
007import java.util.ArrayList;
008import java.util.Collection;
009import java.util.HashMap;
010import java.util.List;
011import java.util.Map;
012
013import org.apache.log4j.Logger;
014import org.conqat.lib.commons.collections.CollectionUtils;
015import org.conqat.lib.commons.filesystem.ClassPathUtils;
016import org.conqat.lib.commons.filesystem.FileExtensionFilter;
017import org.conqat.lib.commons.filesystem.FileSystemUtils;
018import org.conqat.lib.commons.logging.ILogger;
019
020import eu.cqse.check.framework.core.ACheck;
021import eu.cqse.check.framework.core.CheckException;
022import eu.cqse.check.framework.core.CheckImplementationBase;
023import eu.cqse.check.framework.core.CheckInfo;
024import eu.cqse.check.framework.core.CheckInstance;
025
026/**
027 * Registry singleton that manages custom checks.
028 */
029public class CheckRegistry {
030
031        /** Logger instance. */
032        private static final Logger LOGGER = Logger.getLogger(CheckRegistry.class);
033
034        /** The singleton instance. */
035        private static CheckRegistry instance = null;
036
037        /** Mapping from check identifiers to the check information. */
038        private final Map<String, CheckInfo> checksInfo = new HashMap<>();
039
040        /** Private constructor to avoid multiple instantiation. */
041        private CheckRegistry() {
042                // do nothing
043        }
044
045        /** Returns the singleton instance. */
046        public static synchronized CheckRegistry getInstance() {
047                if (instance == null) {
048                        instance = new CheckRegistry();
049                }
050                return instance;
051        }
052
053        /**
054         * Registers the given bundle and loads all classes from the given bundle
055         * that implement {@link CheckImplementationBase} and are annotated with
056         * {@link ACheck}.
057         */
058        public void registerCheckClassDirectory(File classesDirectory) {
059                ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
060                List<String> classNames = ClassPathUtils.getClassNames(classesDirectory);
061                for (String className : classNames) {
062                        try {
063                                loadCheckFromClass(classLoader.loadClass(className));
064                        } catch (ClassNotFoundException e) {
065                                LOGGER.error("Failed to load check from class: " + className, e);
066                        }
067                }
068        }
069
070        /**
071         * Creates and registers a custom check instance from the given class. If
072         * the given class does not contain a custom check (class not annotated with
073         * {@link ACheck}), this method does nothing.
074         */
075        private void loadCheckFromClass(Class<?> clazz) {
076                CheckInfo checkInfo = null;
077                try {
078                        checkInfo = CheckLoader.loadFromClass(clazz);
079                } catch (CheckException e) {
080                        LOGGER.error(e.getMessage(), e);
081                        return;
082                }
083
084                if (checkInfo == null) {
085                        return;
086                }
087
088                if (checksInfo.containsKey(checkInfo.getIdentifier())) {
089                        LOGGER.error("Check " + checkInfo.getGroupName() + ":" + checkInfo.getName() + " is already registered.");
090                        return;
091                }
092                checksInfo.put(checkInfo.getIdentifier(), checkInfo);
093        }
094
095        /**
096         * Creates check instances for the given check identifiers.
097         *
098         * @throws CheckException
099         *             if there is no check for one of the given identifiers
100         */
101        public Collection<CheckInstance> instantiateChecks(List<String> identifiers, ILogger logger) throws CheckException {
102                List<CheckInstance> instances = new ArrayList<>();
103                for (String identifier : identifiers) {
104                        CheckInfo checkInfo = checksInfo.get(identifier);
105                        if (checkInfo == null) {
106                                throw new CheckException("No check with identifier '" + identifier + "' exists.");
107                        }
108                        instances.add(new CheckInstance(checkInfo, logger));
109                }
110                return instances;
111        }
112
113        /** Returns a list of all registered checks. */
114        public Collection<CheckInfo> getChecksInfos() {
115                return CollectionUtils.asUnmodifiable(checksInfo.values());
116        }
117
118        /** Registers the given directory of custom checks bundled as JAR files. */
119        public void registerCheckDirectory(File customChecksDirectory) {
120                if (!customChecksDirectory.canRead()) {
121                        LOGGER.error("Cannot read from custom check directory: " + customChecksDirectory.getAbsolutePath());
122                        return;
123                }
124                List<File> jarFiles = FileSystemUtils.listFilesRecursively(customChecksDirectory,
125                                new FileExtensionFilter("jar"));
126                for (File jarFile : jarFiles) {
127                        if (jarFile.isFile()) {
128                                loadChecksFromJAR(jarFile);
129                        }
130                }
131
132        }
133
134        /** Loads custom checks from the given JAR file. */
135        private void loadChecksFromJAR(File jarFile) {
136                ClassLoader contextClassLoader = Thread.currentThread().getContextClassLoader();
137                try (URLClassLoader classLoader = new URLClassLoader(new URL[] { jarFile.toURI().toURL() },
138                                contextClassLoader)) {
139                        List<String> classes = FileSystemUtils.listTopLevelClassesInJarFile(jarFile);
140                        for (String clazz : classes) {
141                                loadCheckFromClass(classLoader.loadClass(clazz));
142                        }
143                } catch (IOException | ClassNotFoundException e) {
144                        LOGGER.error("Error loading custom checks from '" + jarFile + "': " + e.toString(), e);
145                }
146        }
147}
