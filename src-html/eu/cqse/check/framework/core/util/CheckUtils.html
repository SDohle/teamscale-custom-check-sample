001package eu.cqse.check.framework.core.util;
002
003import java.util.EnumSet;
004
005import eu.cqse.check.framework.scanner.ETokenType;
006
007/**
008 * Utilities for the custom check framework.
009 * 
010 * @author $Author: heinemann $
011 * @version $Rev: 54015 $
012 * @ConQAT.Rating YELLOW Hash: 1EA5D1CEA82D4ECFF38841A5EDE71E7D
013 */
014public class CheckUtils {
015
016        /** Boolean operator types. */
017        public static final EnumSet<ETokenType> BOOLEAN_OPERATOR_TYPES = EnumSet.of(
018                        ETokenType.ANDAND, ETokenType.OROR, ETokenType.AND, ETokenType.OR,
019                        ETokenType.XOR);
020
021        /**
022         * String that is used to separate single parts of unique check identifiers.
023         */
024        private static final String SEPARATOR = "#:#";
025
026        /** Constructs a unique identifier out of the given group name and name. */
027        public static String buildIdentifier(String groupName, String name) {
028                StringBuilder builder = new StringBuilder(groupName);
029                builder.append(SEPARATOR);
030                builder.append(name);
031                return builder.toString();
032        }
033}
