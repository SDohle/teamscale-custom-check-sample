001package eu.cqse.check.framework.core.util;
002
003import java.util.EnumSet;
004
005import eu.cqse.check.framework.scanner.ETokenType;
006
007/**
008 * Utilities for the custom check framework.
009 */
010public class CheckUtils {
011
012        /** Boolean operator types. */
013        public static final EnumSet<ETokenType> BOOLEAN_OPERATOR_TYPES = EnumSet.of(ETokenType.ANDAND, ETokenType.OROR,
014                        ETokenType.AND, ETokenType.OR, ETokenType.XOR);
015
016        /**
017         * String that is used to separate single parts of unique check identifiers.
018         */
019        private static final String SEPARATOR = "#:#";
020
021        /** Constructs a unique identifier out of the given group name and name. */
022        public static String buildIdentifier(String groupName, String name) {
023                StringBuilder builder = new StringBuilder(groupName);
024                builder.append(SEPARATOR);
025                builder.append(name);
026                return builder.toString();
027        }
028}
