008package eu.cqse.check.framework.core.xpath;
009
010import java.util.List;
011
012import org.conqat.lib.commons.collections.CollectionUtils;
013
014import eu.cqse.check.framework.scanner.IToken;
015import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
016import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
017
018/**
019 * A shallow entity that contains all shallow entities of one file, that is used
020 * to represent the document root of xPath selections.
021 * 
022 * @author $Author: hummelb $
023 * @version $Rev: 52960 $
024 * @ConQAT.Rating GREEN Hash: EBDBA03F259C2E75D7911612914F02E6
025 */
026public class DocumentRootShallowEntity extends ShallowEntity {
027
028        /** Version for serialization. */
029        private static final long serialVersionUID = 1L;
030
031        /** Subtype name for the document root entity. */
032        public static final String DOCUMENT_ROOT_SUBTYPE_NAME = "document-root";
033
034        /** Constructor. */
035        public DocumentRootShallowEntity(String name) {
036                super(EShallowEntityType.MODULE, DOCUMENT_ROOT_SUBTYPE_NAME, name,
037                                CollectionUtils.<IToken> emptyList(), 0);
038        }
039
040        /** Sets the children for this document root. */
041        public void setChildren(List<ShallowEntity> children) {
042                for (ShallowEntity child : children) {
043                        addChild(child);
044                }
045        }
046}
