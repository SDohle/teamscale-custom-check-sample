191        /** This maps from extensions to languages. */
192        private static ListMap<String, ELanguage> extension2LanguageMap = new ListMap<String, ELanguage>();
193
194        /** The statement oracle for this language. */
195        private final IStatementOracle statementOracle;
196
197        /** Initialize {@link #extension2LanguageMap}. */
198        static {
199                for (ELanguage language : values()) {
200                        for (String extension : language.extensions) {
201                                extension2LanguageMap.add(extension.toLowerCase(), language);
202                        }
203                }
204        }
205
206        /** File extensions commonly used for this language. */
207        private final String[] extensions;
208
209        /**
210         * Pattern describing the parts of a comment line that should be trimmed to
211         * reveal the text.
212         */
213        private final Pattern commentLineTrimPattern;
214
215        /** Whether the language is case sensitive. */
216        private final boolean caseSensitive;
217
218        /** The readable name of the language, to be used, e.g., in a UI. */
219        private final String readableName;
220
221        /** Create language. */
222        private ELanguage(String readableName, Set<ETokenType> statementDelimiters, String commentLineTrimRegex,
223                        boolean caseSensitive, String... extensions) {
224                this(readableName, new StatementOracle(statementDelimiters), commentLineTrimRegex, caseSensitive, extensions);
225        }
226
227        /** Create language. */
228        private ELanguage(String readableName, IStatementOracle oracle, String commentLineTrimRegex, boolean caseSensitive,
229                        String... extensions) {
230                statementOracle = oracle;
231                this.commentLineTrimPattern = Pattern.compile(commentLineTrimRegex);
232                this.caseSensitive = caseSensitive;
233                this.extensions = extensions;
234                this.readableName = readableName;
235        }
236
237        /** Get the file extensions commonly used for this language. */
238        public String[] getFileExtensions() {
239                return CollectionUtils.copyArray(extensions);
240        }
241
242        /** Get statement oracle for this language. */
243        public IStatementOracle getStatementOracle() {
244                return statementOracle;
245        }
246
247        /**
248         * Gets the {@link ELanguage} value corresponding to the file extension of
249         * the path. Returns null if no extension was found. If there are multiple
250         * possible languages, the first one is returned.
251         */
252        public static ELanguage fromPath(String path) {
253                return fromFile(new File(path));
254        }
255
256        /**
257         * Gets the {@link ELanguage} value corresponding to the file extension of
258         * the file. Returns null if no extension was found. If there are multiple
259         * possible languages, the first one is returned.
260         */
261        public static ELanguage fromFile(File file) {
262                return fromFileExtension(FileSystemUtils.getFileExtension(file));
263        }
264
265        /**
266         * Gets the {@link ELanguage} value corresponding to the given file
267         * extension (without a dot). Returns null if no extension was found. If
268         * there are multiple possible languages, the first one is returned.
269         */
270        public static ELanguage fromFileExtension(String extension) {
271                if (extension == null) {
272                        return null;
273                }
274
275                List<ELanguage> result = extension2LanguageMap.getCollection(extension.toLowerCase());
276
277                if (result == null || result.isEmpty()) {
278                        return null;
279                }
280                return result.get(0);
281        }
282
283        /**
284         * Get the content of a comment, i.e. with the comment delimiters removed.
285         */
286        public String getCommentContent(String commentText) {
287                StringBuilder content = new StringBuilder();
288                for (String line : StringUtils.splitLinesAsList(commentText)) {
289                        if (content.length() > 0) {
290                                content.append(StringUtils.CR);
291                        }
292                        content.append(commentLineTrimPattern.matcher(line).replaceAll(StringUtils.EMPTY_STRING).trim());
293                }
294                return content.toString();
295        }
296
297        /** Return whether the language is case sensitive. */
298        public boolean isCaseSensitive() {
299                return caseSensitive;
300        }
301
302        /** Returns {@link #readableName}. */
303        public String getReadableName() {
304                return readableName;
305        }

307        /** List of languages that do not have methods. */
308        private static final EnumSet<ELanguage> LANGUAGES_WITHOUT_METHODS = EnumSet.of(LINE, NL_REQUIREMENTS, NL_TESTS,
309                        TEXT, XML, HANA_VIEW);
310
311        /** @return Whether the given language has the concept of methods. */
312        public static boolean languageHasMethods(ELanguage language) {
313                return !LANGUAGES_WITHOUT_METHODS.contains(language);
314        }
