017package eu.cqse.check.framework.scanner;
018
019import java.awt.Color;
020import java.awt.Font;
021import java.util.HashMap;
022
023import javax.swing.JEditorPane;
024import javax.swing.text.Element;
025import javax.swing.text.ViewFactory;
026
027import eu.cqse.check.framework.scanner.ETokenType.ETokenClass;
028
029/**
030 * View factory to be used with CCSM scanners. To use this implement an editor
031 * kit like the one below and set it on your text component via
032 * {@link JEditorPane#setEditorKit(javax.swing.text.EditorKit)}.
033 * 
034 * <pre>
035 * public class ExampleEditorKit extends StyledEditorKit {
036 * 
037 *      private final ViewFactory viewFactory;
038 * 
039 *      public ExampleEditorKit() {
040 *              ILenientScanner scanner = ScannerFactory.newLenientScanner(
041 *                              ELanguage.JAVA, &quot;&quot;, null);
042 *              HashMap&lt;ETokenClass, Color&gt; colorMap = new HashMap&lt;ETokenClass, Color&gt;();
043 *              colorMap.put(ETokenClass.COMMENT, new Color(51, 102, 102));
044 *              colorMap.put(ETokenClass.KEYWORD, new Color(102, 0, 102));
045 *              colorMap.put(ETokenClass.LITERAL, new Color(51, 0, 255));
046 *              colorMap.put(ETokenClass.ERROR, Color.red);
047 * 
048 *              HashMap&lt;ETokenClass, Integer&gt; styleMap = new HashMap&lt;ETokenClass, Integer&gt;();
049 *              styleMap.put(ETokenClass.COMMENT, Font.ITALIC);
050 *              styleMap.put(ETokenClass.KEYWORD, Font.BOLD);
051 * 
052 *              viewFactory = new SyntaxHighlightingViewFactory(scanner, colorMap,
053 *                              styleMap);
054 *      }
055 * 
056 *      &#064;Override
057 *      public ViewFactory getViewFactory() {
058 *              return viewFactory;
059 *      }
060 * }
061 * </pre>
062 * 
063 * @author $Author: hummelb $
064 * @version $Rev: 52656 $
065 * @ConQAT.Rating GREEN Hash: 7F1C1E535AB4C37607D92BF62533A56A
066 */
067public class SyntaxHighlightingViewFactory implements ViewFactory {
068
069        /** Scanner to be used. */
070        private final IScanner scanner;
071
072        /** Maps from token class to color. */
073        private final HashMap<ETokenClass, Color> colorMap;
074
075        /**
076         * Maps from token class to font style. Style integers are defined by class
077         * {@link Font}.
078         */
079        private final HashMap<ETokenClass, Integer> styleMap;
080
081        /**
082         * Create new view factory.
083         * 
084         * @param scanner
085         *            scanner to be used.
086         * @param colorMap
087         *            maps from token class to color.
088         * @param styleMap
089         *            maps from token class to font style. Style integers are
090         *            defined by class {@link Font}.
091         */
092        public SyntaxHighlightingViewFactory(IScanner scanner,
093                        HashMap<ETokenClass, Color> colorMap,
094                        HashMap<ETokenClass, Integer> styleMap) {
095                this.scanner = scanner;
096                this.colorMap = colorMap;
097                this.styleMap = styleMap;
098        }
099
100        /** Create view. */
101        @Override
102        public SyntaxHighlightingView create(Element element) {
103                return new SyntaxHighlightingView(element, scanner, colorMap, styleMap);
104        }
105}
