008package eu.cqse.check.framework.scanner;
009
010import org.conqat.lib.commons.equals.IEquator;
011
012/**
013 * A class for testing tokens for equality with respect to type and content.
014 * 
015 * @author $Author: hummelb $
016 * @version $Rev: 52656 $
017 * @ConQAT.Rating GREEN Hash: B4F248A7DC29385CFF5202295E53E5D6
018 */
019public class TokenEquator implements IEquator<IToken> {
020
021        /** Reusable instance. */
022        public static final TokenEquator INSTANCE = new TokenEquator();
023
024        /** {@inheritDoc} */
025        @Override
026        public boolean equals(IToken token1, IToken token2) {
027                return token1.getType() == token2.getType()
028                                && token1.getText().equals(token2.getText());
029        }
030}
