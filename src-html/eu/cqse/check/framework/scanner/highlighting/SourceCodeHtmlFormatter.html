052/**
053 * Helper class for formatting source code as HTML. This class is intended to be
054 * customized by subclassing, although it can also be used as is.
055 * 
056 * @author $Author: hummelb $
057 * @version $Rev: 52656 $
058 * @ConQAT.Rating GREEN Hash: 6BD65E29B4D1176A57F28BA6E6C8974A
059 */
060public class SourceCodeHtmlFormatter {
061
062        /**
063         * Cache that converts from style information as provided by
064         * {@link SourceCodeStyle} to CSS. We use a cache not for performance
065         * reasons, but to keep the number of created CSS classes low. Thus, we use
066         * a cache that never loses information.
067         */
068        @SuppressWarnings({ "unchecked", "rawtypes" })
069        private final ICache<ImmutablePair<Color, Integer>, CSSDeclarationBlock, NeverThrownRuntimeException> cssCache = new BasicCache<ImmutablePair<Color, Integer>, CSSDeclarationBlock, NeverThrownRuntimeException>(
070                        null, new CSSStyleFactory(),
071                        (ICacheBackend) ECachingStrategy.UNLIMITED.getBackend(0));
072
073        /** Base style for formatted code. */
074        private CSSDeclarationBlock baseStyle;
075
076        /** Style used for the line box of even lines. */
077        private CSSDeclarationBlock evenLineStyle;
078
079        /** Style used for the line box of odd lines. */
080        private CSSDeclarationBlock oddLineStyle;
081
082        /** Style used for the line number. */
083        private CSSDeclarationBlock lineNumberStyle;
084
085        /** Style used for disabled text. */
086        private CSSDeclarationBlock disabledStyle;
087
088        /** The writer for output. */
089        protected HTMLWriter writer;
090
091        /** The content. */
092        private String content;
093
094        /** Tokenized content. */
095        private List<IToken> tokens;
096
097        /** Disabled regions. */
098        private Queue<Region> disabledRegions;
099
100        /** The style used. */
101        private SourceCodeStyle style;
102
103        /** Global index into {@link #tokenIndex} during formatting. */
104        private int tokenIndex;
105
106        /** Global character position during formatting. */
107        private int characterPosition;
108
109        /**
110         * String to be used instead of a tab character. The default is the tab
111         * character (i.e. no replacement), but this can also be used to replace
112         * tabs with spaces.
113         */
114        private String tabReplacement = "\t";
115
116        /** Sets the string to be used instead of a tab character. */
117        public void setTabReplacement(String tabReplacement) {
118                this.tabReplacement = tabReplacement;
119        }
120
121        /** Formats the source code (given as token stream) using the writer. */
122        public void formatSourceCode(String content, ELanguage language,
123                        HTMLWriter writer) {
124                formatSourceCode(content, language, null, writer);
125        }
126
127        /**
128         * Formats the source code (given as token stream) using the writer.
129         * 
130         * @param disabledRegions
131         *            list of regions (begin and end offset) in the text that are
132         *            displayed as disabled. The regions may not overlap and must be
133         *            sorted by start position.
134         */
135        public void formatSourceCode(String content, ELanguage language,
136                        List<Region> disabledRegions, HTMLWriter writer) {
137                if (baseStyle == null) {
138                        init();
139                }
140
141                // we want to ensure single character line breaks to simplify offset
142                // calculations later on (we only pass line arrays)
143                this.content = StringUtils.replaceLineBreaks(content, "\n");
144                tokens = ScannerUtils.getTokens(content, language);
145                this.writer = writer;
146                style = SourceCodeStyle.get(language);
147
148                this.disabledRegions = new LinkedList<Region>();
149                if (disabledRegions != null) {
150                        this.disabledRegions.addAll(disabledRegions);
151                }
152
153                tokenIndex = 0;
154                characterPosition = 0;
155
156                format();
157        }
158
159        /** Performs the actual formatting. */
160        protected void format() {
161                writer.setSuppressLineBreaks(true);
162                String[] lines = StringUtils.splitLines(content);
163                for (int lineIndex = 0; lineIndex < lines.length; ++lineIndex) {
164                        formatLine(lines[lineIndex], lineIndex + 1);
165                }
166                writer.setSuppressLineBreaks(false);
167        }
168
169        /** Formats a single line including line numbers. */
170        protected void formatLine(String line, int lineNumber) {
171                if (lineNumber % 2 == 0) {
172                        writer.openElement(EHTMLElement.DIV, EHTMLAttribute.CLASS,
173                                        evenLineStyle);
174                } else {
175                        writer.openElement(EHTMLElement.DIV, EHTMLAttribute.CLASS,
176                                        oddLineStyle);
177                }
178
179                beforeFormatLine(lineNumber);
180
181                formatLineNumber(lineNumber);
182                insertLineNumberSpacer(lineNumber);
183                formatLineContent(line, lineNumber);
184
185                writer.closeElement(EHTMLElement.DIV);
186                writer.addNewLine();
187                characterPosition += 1;
188
189                clearPassedDisabledRegions();
190        }
191
192        /**
193         * Removes all disabled regions from the head of the queue that end before
194         * the current character offset. Returns <code>true</code> if at least one
195         * region has been removed, or <code>false</code> otherwise.
196         */
197        private boolean clearPassedDisabledRegions() {
198                boolean regionRemoved = false;
199                while (!disabledRegions.isEmpty()
200                                && characterPosition >= disabledRegions.peek().getEnd()) {
201                        disabledRegions.remove();
202                        regionRemoved = true;
203                }
204                return regionRemoved;
205        }
206
207        /** Allows deriving classes to add style attributes */
208        protected void beforeFormatLine(@SuppressWarnings("unused") int lineNumber) {
209                // empty default implementation
210        }
211
212        /**
213         * Template method for inserting HTML between the line number and the line
214         * content.
215         * 
216         * @param lineNumber
217         *            the number of this line, if required by subclasses.
218         */
219        protected void insertLineNumberSpacer(int lineNumber) {
220                // empty default implementation
221        }
222
223        /** Formats the line number. */
224        private void formatLineNumber(int lineNumber) {
225                writer.addClosedTextElement(EHTMLElement.SPAN,
226                                formatLineNumberString(lineNumber), EHTMLAttribute.CLASS,
227                                lineNumberStyleFor(lineNumber));
228        }
229
230        /**
231         * Allows deriving classes to adapt the line number style according to the
232         * line number
233         */
234        protected CSSDeclarationBlock lineNumberStyleFor(
235                        @SuppressWarnings("unused") int lineNumber) {
236                return lineNumberStyle;
237        }
238
239        /** Formats the given line number as a string. */
240        protected String formatLineNumberString(int lineNumber) {
241                return String.format("%5d: ", lineNumber);
242        }
243
244        /**
245         * Formats the content of the line.
246         * 
247         * @param lineNumber
248         *            the number of this line, if required by subclasses.
249         */
250        protected void formatLineContent(String line, int lineNumber) {
251                CSSDeclarationBlock currentCSS = null;
252                IToken currentToken = null;
253
254                boolean disabled = !disabledRegions.isEmpty()
255                                && disabledRegions.peek().containsPosition(characterPosition);
256                if (disabled) {
257                        currentCSS = disabledStyle;
258                        switchCSS(null, currentCSS);
259                }
260
261                for (int i = 0; i < line.length(); ++i) {
262
263                        if (disabled) {
264                                if (clearPassedDisabledRegions()) {
265                                        disabled = false;
266                                        // kill current token to restart search for token
267                                        currentToken = null;
268                                }
269                        } else if (!disabledRegions.isEmpty()
270                                        && disabledRegions.peek().containsPosition(
271                                                        characterPosition)) {
272                                disabled = true;
273                                switchCSS(currentCSS, disabledStyle);
274                                currentCSS = disabledStyle;
275                        } else if (currentToken == null
276                                        || characterPosition > currentToken.getEndOffset()) {
277                                CSSDeclarationBlock expectedCSS = null;
278                                while (tokenIndex < tokens.size()
279                                                && tokens.get(tokenIndex).getEndOffset() < characterPosition) {
280                                        tokenIndex += 1;
281                                }
282                                if (tokenIndex < tokens.size()
283                                                && characterPosition >= tokens.get(tokenIndex)
284                                                                .getOffset()) {
285                                        currentToken = tokens.get(tokenIndex);
286                                        expectedCSS = determineCSS(currentToken.getType());
287                                }
288                                switchCSS(currentCSS, expectedCSS);
289                                currentCSS = expectedCSS;
290                        }
291
292                        insertCharacter(line.charAt(i));
293                        characterPosition += 1;
294                }
295                switchCSS(currentCSS, null);
296        }
297
298        /**
299         * Inserts a character. This performs escaping directly, instead of calling
300         * {@link HTMLWriter#addText(String)}, as this is significantly faster for
301         * single characters (addText uses regular expressions for escaping).
302         */
303        private void insertCharacter(char c) {
304                switch (c) {
305                case '&':
306                        writer.addRawString("&amp;");
307                        break;
308                case '<':
309                        writer.addRawString("&lt;");
310                        break;
311                case '>':
312                        writer.addRawString("&gt;");
313                        break;
314                case '\"':
315                        writer.addRawString("&quot;");
316                        break;
317                case '\t':
318                        writer.addRawString(tabReplacement);
319                        break;
320                default:
321                        writer.addRawString(Character.toString(c));
322                }
323        }
324
325        /** Inserts code to switch between CSS classes (in required). */
326        protected void switchCSS(CSSDeclarationBlock current,
327                        CSSDeclarationBlock next) {
328
329                // Comparison with == is valid, as CSSDeclarationBlock has no proper
330                // equals() implementation and the CSSManagerBase works with an
331                // IdentityHashMap internally. Furthermore, the cache helps.
332                if (current == next) {
333                        return;
334                }
335
336                if (current != null) {
337                        writer.closeElement(EHTMLElement.SPAN);
338                }
339                if (next != null) {
340                        writer.openElement(EHTMLElement.SPAN, EHTMLAttribute.CLASS, next);
341                }
342        }
343
344        /** Returns the CSS class to be used for a token (or null). */
345        protected CSSDeclarationBlock determineCSS(ETokenType type) {
346                return cssCache.obtain(style.getStyle(type));
347        }
348
349        /**
350         * Lazy initialization. We do not initialize styles in the constructor, as
351         * this might lead to cases where this constructor calls overridden methods.
352         */
353        protected void init() {
354                baseStyle = createBaseStyle();
355                evenLineStyle = createEvenLineStyle(baseStyle);
356                oddLineStyle = createOddLineStyle(baseStyle);
357                lineNumberStyle = createLineNumberStyle(baseStyle);
358                disabledStyle = cssCache.obtain(getDisabledStyle());
359        }
360
361        /** Returns the style pair to be used for disabled code. */
362        protected ImmutablePair<Color, Integer> getDisabledStyle() {
363                return new Pair<Color, Integer>(new Color(0xaa, 0xaa, 0xaa), 0);
364        }
365
366        /** Creates the base style all other styles inherit from. */
367        protected CSSDeclarationBlock createBaseStyle() {
368                return new CSSDeclarationBlock(FONT_SIZE, "13px",
369                                ECSSProperty.WHITE_SPACE, "pre", FONT_FAMILY, "Monospace");
370        }
371
372        /** Creates the style used for the div element enclosing the line. */
373        protected CSSDeclarationBlock createLineStyle(CSSDeclarationBlock baseStyle) {
374                return new CSSDeclarationBlock(baseStyle, ECSSProperty.MARGIN_BOTTOM,
375                                "0");
376        }
377
378        /**
379         * Creates the style used for the div element enclosing even lines. Default
380         * implementation uses {@link #createLineStyle(CSSDeclarationBlock)}.
381         */
382        protected CSSDeclarationBlock createEvenLineStyle(
383                        CSSDeclarationBlock baseStyle) {
384                return createLineStyle(baseStyle);
385        }
386
387        /**
388         * Creates the style used for the div element enclosing odd lines. Default
389         * implementation uses {@link #createLineStyle(CSSDeclarationBlock)}.
390         */
391        protected CSSDeclarationBlock createOddLineStyle(
392                        CSSDeclarationBlock baseStyle) {
393                return createLineStyle(baseStyle);
394        }
395
396        /** Creates the style used for the line number. */
397        protected CSSDeclarationBlock createLineNumberStyle(
398                        CSSDeclarationBlock baseStyle) {
399                return new CSSDeclarationBlock(baseStyle, COLOR, "#aaaaaa");
400        }
401
402        /** Factory for converting AWT based styles to CSS. */
403        private final class CSSStyleFactory
404                        implements
405                        IParameterizedFactory<CSSDeclarationBlock, ImmutablePair<Color, Integer>, NeverThrownRuntimeException> {
406
407                /**
408                 * Content of style parameter is defined in {@link SourceCodeStyle}
409                 * class.
410                 */
411                @Override
412                public CSSDeclarationBlock create(ImmutablePair<Color, Integer> style) {
413                        CSSDeclarationBlock css = new CSSDeclarationBlock(baseStyle);
414                        css.setProperty(COLOR, ColorUtils.toHtmlString(style.getFirst()));
415                        if ((style.getSecond() & Font.BOLD) != 0) {
416                                css.setProperty(FONT_WEIGHT, "bold");
417                        }
418                        if ((style.getSecond() & Font.ITALIC) != 0) {
419                                css.setProperty(ECSSProperty.FONT_STYLE, "italic");
420                        }
421                        return css;
422                }
423        }
