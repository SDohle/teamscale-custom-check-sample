041/**
042 * Factory class for creation of shallow parsers and utility methods.
043 * 
044 * @author $Author: pagano $
045 * @version $Rev: 55066 $
046 * @ConQAT.Rating YELLOW Hash: 1D6F41D17B2FC4C6E89E02BB108A02BE
047 */
048public class ShallowParserFactory {
049
050        /** The map of parsers (initialized lazily). */
051        private static Map<ELanguage, IShallowParser> parsers;
052
053        static {
054                parsers = new EnumMap<ELanguage, IShallowParser>(ELanguage.class);
055                parsers.put(ELanguage.JAVA, new JavaShallowParser());
056                parsers.put(ELanguage.ADA, new AdaShallowParser());
057                parsers.put(ELanguage.CS, new CsShallowParser());
058                parsers.put(ELanguage.CPP, new CppShallowParser());
059                parsers.put(ELanguage.PLSQL, new PlsqlShallowParser());
060                parsers.put(ELanguage.ABAP, new AbapShallowParser());
061                parsers.put(ELanguage.JAVASCRIPT, new JavaScriptShallowParser());
062                parsers.put(ELanguage.RUBY, new RubyShallowParser());
063                parsers.put(ELanguage.PYTHON, new PythonShallowParser());
064                parsers.put(ELanguage.MAGIK, new MagikShallowParser());
065                parsers.put(ELanguage.DELPHI, new DelphiShallowParser());
066                parsers.put(ELanguage.MATLAB, new MatlabShallowParser());
067                parsers.put(ELanguage.FORTRAN, new FortranShallowParser());
068                parsers.put(ELanguage.IEC61131, new Iec61131ShallowParser());
069                parsers.put(ELanguage.XTEND, new XtendShallowParser());
070                parsers.put(ELanguage.OCAML, new OcamlShallowParser());
071                parsers.put(ELanguage.TSQL, new TsqlShallowParser());
072        }
073
074        /**
075         * Returns a new parser for the given language.
076         * <p>
077         * While we call this method "create" for consistency with other factories,
078         * the parsers are actually created only once and then returned over and
079         * over again. The reason is that parser creation may be expensive,
080         * especially when many very small code fragments are to be parsed. Reusing
081         * parsers is possible as the parsers no not hold state of a specific parse
082         * and even can be used concurrently in multiple threads.
083         * 
084         * @throws ShallowParserException
085         *             if the language is not (yet) supported by our framework.
086         */
087        public static IShallowParser createParser(ELanguage language)
088                        throws ShallowParserException {
089                IShallowParser parser = parsers.get(language);
090                if (parser == null) {
091                        throw new ShallowParserException("Shallow parsing for language "
092                                        + language + " not yet supported!");
093                }
094                return parser;
095        }
096
097        /**
098         * Returns whether the given language is supported by the parser factory.
099         */
100        public static boolean supportsLanguage(ELanguage language) {
101                return parsers.containsKey(language);
102        }
103}
