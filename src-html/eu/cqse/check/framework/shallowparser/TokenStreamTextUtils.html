001package eu.cqse.check.framework.shallowparser;
002
003import static eu.cqse.check.framework.shallowparser.TokenStreamUtils.NOT_FOUND;
004
005import java.util.ArrayList;
006import java.util.List;
007
008import eu.cqse.check.framework.scanner.ETokenType;
009import eu.cqse.check.framework.scanner.IToken;
010
011/**
012 * Utility methods for {@link IToken} lists that work on the tokens' text.
013 * 
014 * @author $Author: pagano $
015 * @version $Rev: 54869 $
016 * @ConQAT.Rating GREEN Hash: 1E121FB6DBF18A052428CBCB94BF2174
017 */
018public class TokenStreamTextUtils {
019
020        /**
021         * Returns the index of the first token whose token text equals the given
022         * text in the given token list. If the text is not found,
023         * {@link #NOT_FOUND} is returned.
024         */
025        public static int findFirst(List<IToken> tokens, String text) {
026                return findFirst(tokens, 0, tokens.size(), text);
027        }
028
029        /**
030         * Returns the first index of the given text not before the given start
031         * index and before the given end index. If the text is not found,
032         * {@link #NOT_FOUND} is returned.
033         */
034        public static int findFirst(List<IToken> tokens, int startOffset,
035                        int endOffset, String text) {
036                for (int i = startOffset; i < endOffset; i++) {
037                        if (tokens.get(i).getText().equals(text)) {
038                                return i;
039                        }
040                }
041                return NOT_FOUND;
042        }
043
044        /**
045         * Returns all indices of the given text within the given token list. If
046         * none is found, an empty list is returned.
047         */
048        public static List<Integer> findAll(List<IToken> tokens, String text) {
049                List<Integer> indices = new ArrayList<>();
050                for (int i = 0; i < tokens.size(); i++) {
051                        if (tokens.get(i).getText().equals(text)) {
052                                indices.add(i);
053                        }
054                }
055                return indices;
056        }
057
058        /**
059         * Returns whether the given token list contains a token with the given
060         * text.
061         */
062        public static boolean contains(List<IToken> tokens, String text) {
063                return findFirst(tokens, text) != NOT_FOUND;
064        }
065
066        /**
067         * Returns whether the given token list contains tokens with the given
068         * sequence of strings at the given start offset.
069         */
070        public static boolean hasSequence(List<IToken> tokens, int startOffset,
071                        String... sequence) {
072                if (startOffset < 0 || startOffset + sequence.length > tokens.size()) {
073                        return false;
074                }
075
076                for (int i = 0; i < sequence.length; i++) {
077                        if (!tokens.get(startOffset + i).getText().equals(sequence[i])) {
078                                return false;
079                        }
080                }
081                return true;
082        }
083
084        /**
085         * Returns the index of the first occurrence of the given sequence of
086         * strings in the given token list, beginning from the given start offset.
087         * The given start type is the type of the sequence's first token. If the
088         * sequence is not found, {@link #NOT_FOUND} is returned.
089         */
090        public static int findSequence(List<IToken> tokens, int startOffset,
091                        ETokenType startType, String... sequence) {
092                for (int i = startOffset; i < tokens.size() - sequence.length
093                                + 1; i++) {
094                        if (tokens.get(i).getType().equals(startType)) {
095                                if (hasSequence(tokens, i, sequence)) {
096                                        return i;
097                                }
098                        }
099                }
100                return NOT_FOUND;
101        }
102
103        /**
104         * Returns all indices of occurrences of the given sequence of token texts
105         * in the given token list, beginning from the given offset. The given start
106         * token type is the type of the first token of the sequence. If the
107         * sequence is not found, an empty list is returned.
108         */
109        public static List<Integer> findAllSequences(List<IToken> tokens,
110                        int startOffset, ETokenType startType, String... sequence) {
111                List<Integer> indices = new ArrayList<>();
112
113                while (startOffset < tokens.size() - sequence.length + 1) {
114                        startOffset = findSequence(tokens, startOffset, startType,
115                                        sequence);
116                        if (startOffset == NOT_FOUND) {
117                                break;
118                        }
119                        indices.add(startOffset);
120                        startOffset++;
121                }
122
123                return indices;
124        }
125
126        /** Concatenates the token's texts and returns them as string. */
127        public static String concatTokenTexts(List<IToken> tokens) {
128                return concatTokenTexts(tokens, "");
129        }
130
131        /** Concatenates the token's texts and returns them as string. */
132        public static String concatTokenTexts(List<IToken> tokens,
133                        String separator) {
134                StringBuilder builder = new StringBuilder();
135                for (int i = 0; i < tokens.size(); i++) {
136                        if (i > 0) {
137                                builder.append(separator);
138                        }
139                        builder.append(tokens.get(i).getText());
140                }
141                return builder.toString();
142        }
143
144        /** Concatenates the inner lists' token texts and returns them as list. */
145        public static List<String> concatAllTokenTexts(
146                        List<List<IToken>> tokenLists) {
147                List<String> strings = new ArrayList<String>();
148                for (List<IToken> tokenList : tokenLists) {
149                        strings.add(concatTokenTexts(tokenList));
150                }
151                return strings;
152
153        }
154
155        /**
156         * Converts the sublist of the given token list from the given start index
157         * (inclusive) to the given end index (exclusive) to a list of token texts.
158         */
159        public static List<String> getTokenTexts(List<IToken> tokens,
160                        int startIndex, int endIndex) {
161                List<String> resultStrings = new ArrayList<>();
162                for (int i = startIndex; i < endIndex; i++) {
163                        resultStrings.add(tokens.get(i).getText());
164                }
165                return resultStrings;
166        }
167
168        /**
169         * Converts the sublist of the given token list to a list of token texts.
170         */
171        public static List<String> getTokenTexts(List<IToken> tokens) {
172                return getTokenTexts(tokens, 0, tokens.size());
173        }
174}
