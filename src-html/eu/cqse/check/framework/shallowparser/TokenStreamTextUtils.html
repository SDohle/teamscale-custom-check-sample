001package eu.cqse.check.framework.shallowparser;
002
003import static eu.cqse.check.framework.shallowparser.TokenStreamUtils.NOT_FOUND;
004
005import java.util.ArrayList;
006import java.util.List;
007
008import org.conqat.lib.commons.assertion.CCSMAssert;
009import org.conqat.lib.commons.collections.CollectionUtils;
010import org.conqat.lib.commons.string.StringUtils;
011
012import eu.cqse.check.framework.scanner.ETokenType;
013import eu.cqse.check.framework.scanner.IToken;
014
015/**
016 * Utility methods for {@link IToken} lists that work on the tokens' text.
017 */
018public class TokenStreamTextUtils {
019
020        /**
021         * Returns the index of the first token whose token text equals the given
022         * text in the given token list. If the text is not found,
023         * {@link #NOT_FOUND} is returned.
024         */
025        public static int findFirst(List<IToken> tokens, String text) {
026                return findFirst(tokens, 0, tokens.size(), text);
027        }
028
029        /**
030         * Returns the first index of the given text not before the given start
031         * index and before the given end index. If the text is not found,
032         * {@link #NOT_FOUND} is returned.
033         */
034        public static int findFirst(List<IToken> tokens, int startOffset, int endOffset, String text) {
035                for (int i = startOffset; i < endOffset; i++) {
036                        if (tokens.get(i).getText().equals(text)) {
037                                return i;
038                        }
039                }
040                return NOT_FOUND;
041        }
042
043        /**
044         * Returns all indices of the given text within the given token list. If
045         * none is found, an empty list is returned.
046         */
047        public static List<Integer> findAll(List<IToken> tokens, String text) {
048                List<Integer> indices = new ArrayList<>();
049                for (int i = 0; i < tokens.size(); i++) {
050                        if (tokens.get(i).getText().equals(text)) {
051                                indices.add(i);
052                        }
053                }
054                return indices;
055        }
056
057        /**
058         * Returns whether the given token list contains a token with the given
059         * text.
060         */
061        public static boolean contains(List<IToken> tokens, String text) {
062                return findFirst(tokens, text) != NOT_FOUND;
063        }
064
065        /**
066         * Returns whether the given token list contains tokens with the given
067         * sequence of strings at the given start offset.
068         */
069        public static boolean hasSequence(List<IToken> tokens, int startOffset, String... sequence) {
070                CCSMAssert.isFalse(startOffset < 0, "startOffset must not be less than zero");
071                if (startOffset + sequence.length > tokens.size()) {
072                        return false;
073                }
074
075                for (int i = 0; i < sequence.length; i++) {
076                        if (!tokens.get(startOffset + i).getText().equals(sequence[i])) {
077                                return false;
078                        }
079                }
080                return true;
081        }
082
083        /**
084         * Returns the index of the first occurrence of the given sequence of
085         * strings in the given token list, beginning from the given start offset.
086         * The given start type is the type of the sequence's first token. If the
087         * sequence is not found, {@link #NOT_FOUND} is returned.
088         */
089        public static int findSequence(List<IToken> tokens, int startOffset, ETokenType startType, String... sequence) {
090                for (int i = startOffset; i < tokens.size() - sequence.length + 1; i++) {
091                        if (tokens.get(i).getType().equals(startType)) {
092                                if (hasSequence(tokens, i, sequence)) {
093                                        return i;
094                                }
095                        }
096                }
097                return NOT_FOUND;
098        }
099
100        /**
101         * Returns all indices of occurrences of the given sequence of token texts
102         * in the given token list, beginning from the given offset. The given start
103         * token type is the type of the first token of the sequence. If the
104         * sequence is not found, an empty list is returned.
105         */
106        public static List<Integer> findAllSequences(List<IToken> tokens, int startOffset, ETokenType startType,
107                        String... sequence) {
108                List<Integer> indices = new ArrayList<>();
109
110                while (startOffset < tokens.size() - sequence.length + 1) {
111                        startOffset = findSequence(tokens, startOffset, startType, sequence);
112                        if (startOffset == NOT_FOUND) {
113                                break;
114                        }
115                        indices.add(startOffset);
116                        startOffset++;
117                }
118
119                return indices;
120        }
121
122        /** Concatenates the token's texts and returns them as string. */
123        public static String concatTokenTexts(List<IToken> tokens) {
124                return concatTokenTexts(tokens, "");
125        }
126
127        /** Concatenates the token's texts and returns them as string. */
128        public static String concatTokenTexts(List<IToken> tokens, String separator) {
129                return StringUtils.concat(CollectionUtils.map(tokens, IToken::getText), separator);
130        }
131
132        /** Concatenates the inner lists' token texts and returns them as list. */
133        public static List<String> concatAllTokenTexts(List<List<IToken>> tokenLists) {
134                return CollectionUtils.map(tokenLists, TokenStreamTextUtils::concatTokenTexts);
135
136        }
137
138        /**
139         * Converts the sublist of the given token list from the given start index
140         * (inclusive) to the given end index (exclusive) to a list of token texts.
141         */
142        public static List<String> getTokenTexts(List<IToken> tokens, int startIndex, int endIndex) {
143                CCSMAssert.isTrue(startIndex >= 0, "startIndex must be greater or equal to zero");
144                CCSMAssert.isTrue(endIndex <= tokens.size(), "endIndex must be less or equal to tokens.size()");
145                return CollectionUtils.map(tokens.subList(startIndex, endIndex), IToken::getText);
146        }
147
148        /** Converts the given token list to a list of corresponding token texts. */
149        public static List<String> getTokenTexts(List<IToken> tokens) {
150                return getTokenTexts(tokens, 0, tokens.size());
151        }
152}
