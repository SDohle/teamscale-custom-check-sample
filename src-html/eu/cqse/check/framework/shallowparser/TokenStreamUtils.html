073         * {@link #firstTokenOfTypeSequence(List, int, int, ETokenType...)}
074         * 
075         * @return integer index (or {@link #NOT_FOUND}).
076         */
077        public static int firstTokenOfType(List<IToken> tokens, int startIndex, int endIndex, ETokenType... tokenTypes) {
078                CCSMAssert.isTrue(startIndex >= 0, "startIndex must be greater or equal to zero");
079                CCSMAssert.isTrue(endIndex <= tokens.size(), "endIndex must be less or equal to tokens.size()");
080
081                EnumSet<ETokenType> set = toEnumSet(tokenTypes);
082
083                for (int i = startIndex; i < endIndex; ++i) {
084                        if (set.contains(tokens.get(i).getType())) {
085                                return i;
086                        }
087                }
088                return NOT_FOUND;
089        }

253                for (int i = startIndex; i < endIndex; ++i) {

255                        if (types.isEmpty()) {

257                        }
258                }
259                return false;

262        /**
263         * Returns <code>true</code> if the list of tokens contains a consecutive
264         * subsequence of tokens that match the sequence of token types., false
265         * otherwise.
266         */
267        public static boolean containsSequence(List<IToken> tokens, int startIndex, int endIndex,
268                        ETokenType... tokenTypes) {
269                CCSMAssert.isTrue(startIndex >= 0, "startIndex must be greater or equal to zero");
270                CCSMAssert.isTrue(endIndex <= tokens.size(), "endIndex must be less or equal to tokens.size()");
271
272                OUTER: for (int i = startIndex; i <= endIndex - tokenTypes.length; ++i) {
273                        for (int j = 0; j < tokenTypes.length; ++j) {
274                                if (tokens.get(i + j).getType() != tokenTypes[j]) {
275                                        continue OUTER;
276                                }
277                        }
278                        return true;
279                }
280                return false;
281        }

293        /**
294         * Returns the sublist of tokens between the first occurrence of some given
295         * start token types and the first occurrence of any of the given end token
296         * types (after start). If no matches are found, an empty list is returned.
297         * The tokens for the start and end are not included in the returned sub
298         * list.
299         */
300        public static List<IToken> tokensBetween(List<IToken> tokens, EnumSet<ETokenType> startTypes,
301                        EnumSet<ETokenType> endTypes) {
302                int start = TokenStreamUtils.firstTokenOfType(tokens, startTypes.toArray(new ETokenType[0]));
303                if (start == NOT_FOUND) {
304                        return CollectionUtils.emptyList();
305                }
306                start += 1;
307
308                int end = TokenStreamUtils.firstTokenOfType(tokens, start, endTypes.toArray(new ETokenType[0]));
309                if (end == NOT_FOUND) {
310                        return CollectionUtils.emptyList();
311                }
312
313                return tokens.subList(start, end);
314        }

316        /**
317         * Returns the offset of the first token of closingType, thereby counting
318         * nesting occurrences of openingType and closingType. Returns
319         * {@link #NOT_FOUND} if none was found.
320         * 
321         * @param currentOffset
322         *            this is the offset to start the search and must be one token
323         *            <b>after</b> the opening token for which the closing token
324         *            shall be found.
325         */
326        public static int findMatchingClosingToken(List<IToken> tokens, int currentOffset, ETokenType openingType,
327                        ETokenType closingType) {
328                int nesting = 1;
329                for (; currentOffset < tokens.size(); ++currentOffset) {
330                        ETokenType tokenType = tokens.get(currentOffset).getType();
331                        if (tokenType == openingType) {
332                                nesting += 1;
333                        } else if (tokenType == closingType) {
334                                nesting -= 1;
335                                if (nesting == 0) {
336                                        return currentOffset;
337                                }
338                        }
339                }
340                return NOT_FOUND;
341        }
342
343        /**
344         * Returns the offset of the first token of openingType, thereby counting
345         * nesting occurrences of openingType and closingType. Returns
346         * {@link #NOT_FOUND} if none was found.
347         * 
348         * @param currentOffset
349         *            this is the offset to start the search and must be one token
350         *            <b>before</b> the closing token for which the opening token
351         *            shall be found.
352         */
353        public static int findMatchingOpeningToken(List<IToken> tokens, int currentOffset, ETokenType openingType,
354                        ETokenType closingType) {
355                int openBraces = 1;
356                for (; currentOffset >= 0; currentOffset--) {
357                        ETokenType currentTokenType = tokens.get(currentOffset).getType();
358                        if (currentTokenType.equals(openingType)) {
359                                openBraces--;
360                        } else if (currentTokenType.equals(closingType)) {
361                                openBraces++;
362                        }
363                        if (openBraces == 0) {
364                                return currentOffset;
365                        }
366                }
367                return NOT_FOUND;
368        }

406                }
407                return NOT_FOUND;

503        /**
504         * Splits the given token stream at the tokens of the given types, but at
505         * most as often as given in the limit parameter. The resulting list will
506         * have at most <code>limit</code> entries). The limit must be bigger than
507         * 0.
508         */
509        public static List<List<IToken>> split(List<IToken> tokens, EnumSet<ETokenType> splitTypes, int limit) {
510                CCSMAssert.isTrue(limit > 0, "The limit must be greater than zero");
511                List<List<IToken>> parts = new ArrayList<>();
512                int start = 0;
513
514                for (int i = 0; i < tokens.size(); i++) {
515                        if (parts.size() == limit - 1) {
516                                break;
517                        }
518                        if (splitTypes.contains(tokens.get(i).getType())) {
519                                List<IToken> part = tokens.subList(start, i);
520                                parts.add(part);
521                                start = i + 1;
522                        }
523                }
524
525                parts.add(tokens.subList(start, tokens.size()));
526                return parts;
527        }

529        /**
530         * Returns whether the given token list contains tokens with the given types
531         * in that order starting at the given start offset.
532         */
533        public static boolean hasTokenTypeSequence(List<IToken> tokens, int startOffset, ETokenType... sequence) {
534                if (startOffset + sequence.length > tokens.size() || startOffset < 0) {
535                        return false;
536                }
537
538                for (int i = 0; i < sequence.length; i++) {
539                        if (!tokens.get(i + startOffset).getType().equals(sequence[i])) {
540                                return false;
541                        }
542                }
543
544                return true;
545        }

611        /**
612         * Returns all tokens between the tokens of the opening and closing types
613         * regarding nesting beginning from the given start offset.
614         */
615        public static List<IToken> tokensBetweenWithNesting(List<IToken> tokens, int startOffset, ETokenType openingType,
616                        ETokenType closingType) {
617                int openingIndex = firstTokenOfType(tokens, startOffset, openingType);
618                if (openingIndex == NOT_FOUND) {
619                        return CollectionUtils.emptyList();
620                }
621
622                openingIndex += 1;
623                int closingIndex = findMatchingClosingToken(tokens, openingIndex, openingType, closingType);
624                if (closingIndex == NOT_FOUND) {
625                        return CollectionUtils.emptyList();
626                }
627
628                return tokens.subList(openingIndex, closingIndex);
629        }

631        /**
632         * Splits the given token list at the tokens of the given split type. If the
633         * split type is nested between the open and close tokens, it is ignored.
634         */
635        public static List<List<IToken>> splitWithNesting(List<IToken> tokens, ETokenType splitType, ETokenType openingType,
636                        ETokenType closingType) {
637                return splitWithNesting(tokens, splitType, Arrays.asList(openingType), Arrays.asList(closingType));
638        }

640        /**
641         * Splits the given token list at the tokens of the given split type. If the
642         * split type is nested between one of the open and close tokens, it is
643         * ignored. The open and close token type lists must have the same size.
644         * None of both must contain duplicates.
645         */
646        public static List<List<IToken>> splitWithNesting(List<IToken> tokens, ETokenType splitType,
647                        List<ETokenType> openingTypes, List<ETokenType> closingTypes) {
648                return splitWithNesting(tokens, EnumSet.of(splitType), openingTypes, closingTypes, Integer.MAX_VALUE);
649        }
650
651        /**
652         * Splits the given token list at the tokens of the given split types. If
653         * one of the split types is nested between one of the open and close
654         * tokens, it is ignored. The resulting list will have at most limit
655         * entries. Limit must be greater zero. The open and close token type lists
656         * must have the same size. None of both must contain duplicates.
657         */
658        public static List<List<IToken>> splitWithNesting(List<IToken> tokens, EnumSet<ETokenType> splitTypes,
659                        List<ETokenType> openingTypes, List<ETokenType> closingTypes, int limit) {
660                CCSMAssert.isTrue(limit > 0, "The limit must be greater than zero");
661
662                List<List<IToken>> parts = new ArrayList<>();
663                int start = 0;
664                for (NestingAwareTokenIterator iterator = new NestingAwareTokenIterator(tokens, 0, openingTypes,
665                                closingTypes); iterator.hasNext();) {
666                        IToken token = iterator.next();
667                        if (parts.size() == limit - 1) {
668                                break;
669                        }
670                        if (iterator.isTopLevel() && splitTypes.contains(token.getType())) {
671                                List<IToken> part = tokens.subList(start, iterator.getCurrentIndex());
672                                parts.add(part);
673                                start = iterator.getCurrentIndex() + 1;
674                        }
675                }
676
677                parts.add(tokens.subList(start, tokens.size()));
678                return parts;
679        }

698        /**
699         * Creates a new token list copying the given list and using the given
700         * reference token.
701         * 
702         * @param referenceToken
703         *            Reference token, whose offset, line number and origin id will
704         *            be copied to the new tokens.
705         * @param tokens
706         *            List of tokens to be duplicated.
707         */
708        public static List<IToken> copyTokens(IToken referenceToken, List<IToken> tokens) {
709                return CollectionUtils.map(tokens,
710                                token -> TokenStreamUtils.createToken(referenceToken, token.getText(), token.getType()));
711        }

713        /**
714         * Adds a new token to the given token list at the given position with the
715         * given type and text. The token that is at that position currently is
716         * taken as a reference. The new token will have the same offset, origin ID
717         * and line number as the reference token.
718         * 
719         * @param tokens
720         *            The token list to modify. Must not be unmodifiable or empty.
721         * @param position
722         *            The position at which to insert the new token. Also the
723         *            position of the reference token. If this is equal to the size
724         *            of the token list, the new token will be appended at the end
725         *            and the last token will be taken as the reference token.
726         * @param type
727         *            The token type of the new token.
728         * @param text
729         *            The text of the new token.
730         */
731        public static void addToken(List<IToken> tokens, int position, ETokenType type, String text) {
732                CCSMAssert.isFalse(tokens.isEmpty(),
733                                "Cannot create new tokens without at least one existing token as a reference");
734                int referencePosition = position;
735                if (position == tokens.size()) {
736                        referencePosition--;
737                }
738                IToken referenceToken = tokens.get(referencePosition);
739                IToken newToken = createToken(referenceToken, text, type);
740                tokens.add(position, newToken);
741        }

743        /**
744         * Removes the last token of the given token list if it matches the
745         * specified token type.
746         * 
747         * @param tokens
748         *            The token list to modify.
749         * @param type
750         *            Token type, which should be removed.
751         * @return The token list without the given token type at the end.
752         */
753        public static List<IToken> removeAtEnd(List<IToken> tokens, ETokenType type) {
754                if (endsWith(tokens, type)) {
755                        tokens = tokens.subList(0, tokens.size() - 1);
756                }
757                return tokens;
758        }

761         * Removes the first token of the given token list if it matches the
762         * specified token type.
763         * 
764         * @param tokens
765         *            The token list to modify.
766         * @param type

768         * @return The token list without the given token type at the beginning.

770        public static List<IToken> removeAtFront(List<IToken> tokens, ETokenType type) {
771                if (startsWith(tokens, type)) {
772                        tokens = CollectionUtils.getRest(tokens);
773                }
774                return tokens;

777        /**
778         * Searches for the first token with the given type and text (ignoring
779         * case).
780         * 
781         * @param tokens
782         *            The list of tokens to be searched.
783         * @param tokenText
784         *            The searched token should have this text (ignoring case).
785         * @param tokenTypes
786         *            The searched token should have one of these types.
787         * @return The first token that matches, null if none is found.
788         */
789        public static IToken getTokenByTypeAndText(List<IToken> tokens, String tokenText, Set<ETokenType> tokenTypes) {
790                for (IToken token : tokens) {
791                        if (tokenTypes.contains(token.getType()) && token.getText().equalsIgnoreCase(tokenText)) {
792                                return token;
793                        }
794                }
795                return null;
796        }
