017package eu.cqse.check.framework.shallowparser.framework;
018
019import java.util.ArrayList;
020import java.util.List;
021import java.util.Set;
022import java.util.Stack;
023
024import org.conqat.lib.commons.assertion.CCSMAssert;
025import org.conqat.lib.commons.collections.PairList;
026
027import eu.cqse.check.framework.scanner.IToken;
028
029/**
030 * This class holds the parser state and also implements parts of the parsing
031 * algorithm. The benefit of separating them into a separate class is that we
032 * can reuse the same parser multiple times without resets and also concurrently
033 * in separate threads.
034 *
035 * @see ShallowParserBase
036 */
037public class ParserState<STATE extends Enum<STATE>> {
038
039        /**
040         * Pairs of states (for which the recognizer is active) and the recognizer.
041         */
042        private final PairList<Set<STATE>, RecognizerBase<STATE>> recognizers;
043
044        /** Constructor. */
045        /* package */ ParserState(PairList<Set<STATE>, RecognizerBase<STATE>> recognizers) {
046                this.recognizers = recognizers;
047        }
048
049        /**
050         * Stack of entities created. If there are multiple top-level entities, this
051         * only contains the most recently created.
052         */
053        private final Stack<ShallowEntity> entities = new Stack<ShallowEntity>();
054
055        /** Stack of start positions for nested parses. */
056        private final Stack<Integer> currentMatchStart = new Stack<Integer>();
057
058        /** Stack of reference positions used for naming. */
059        private final Stack<Integer> currentReferencePosition = new Stack<Integer>();
060
061        /** The result list of all created top-level entities. */
062        final List<ShallowEntity> result = new ArrayList<ShallowEntity>();
063
064        /** Sets the given shallow entity as new target for adding children. */
065        public void setNode(ShallowEntity shallowEntity) {
066                if (entities.isEmpty()) {
067                        result.add(shallowEntity);
068                } else {
069                        entities.peek().addChild(shallowEntity);
070                }
071                entities.push(shallowEntity);
072        }
073
074        /** Marks the current node as completed. */
075        public void endNode(boolean expectsContinuation) {
076                CCSMAssert.isFalse(entities.isEmpty(), "Must have setNode() before each endNode()!");
077                entities.peek().setComplete(expectsContinuation);
078        }
079
080        /** Sets the name of the current node. */
081        public void setNodeName(String name) {
082                CCSMAssert.isFalse(entities.isEmpty(), "Must have setNode() before setNodeName()!");
083                entities.peek().setName(name);
084        }
085
086        /** Returns the name of the top-most ShallowEntity. */
087        public String getCurrentEntityName() {
088                CCSMAssert.isFalse(entities.isEmpty(), "May not access entity name when no nodes have been created before!");
089                return entities.peek().getName();
090        }
091
092        /** Returns the type of the top-most ShallowEntity. */
093        public EShallowEntityType getCurrentEntityType() {
094                CCSMAssert.isFalse(entities.isEmpty(), "May not access entity type when no nodes have been created before!");
095                return entities.peek().getType();
096        }
097
098        /**
099         * Attempts to parse exactly one entity in the given state. If the created
100         * entity is a continued node, this automatically continues parsing.
101         *
102         * @return Either the new parsing offset or {@link RecognizerBase#NO_MATCH}.
103         */
104        public int parse(STATE state, List<IToken> tokens, int startOffset) {
105                int entitiesSize = entities.size();
106
107                int resultOffset = RecognizerBase.NO_MATCH;
108                for (int i = 0; i < recognizers.size(); ++i) {
109                        if (recognizers.getFirst(i).contains(state)) {
110                                currentMatchStart.push(startOffset);
111                                currentReferencePosition.push(startOffset);
112
113                                int match = recognizers.getSecond(i).matches(this, tokens, startOffset);
114
115                                currentReferencePosition.pop();
116                                currentMatchStart.pop();
117
118                                if (match != RecognizerBase.NO_MATCH) {
119                                        resultOffset = match;
120                                        break;
121                                }
122                        }
123                }
124
125                // there was an entity created that needs completion
126                if (entities.size() > entitiesSize) {
127                        ShallowEntity entity = entities.pop();
128                        entity.setEndTokenIndex(resultOffset);
129
130                        // continue parsing if this is a continued entity (such as an
131                        // "if" followed by an "else").
132                        if (entity.isContinued() && resultOffset != RecognizerBase.NO_MATCH) {
133                                int nextOffset = parse(state, tokens, resultOffset);
134
135                                if (nextOffset != RecognizerBase.NO_MATCH) {
136                                        return nextOffset;
137                                        // otherwise fall through and use last parsing position
138                                }
139                        }
140                }
141
142                return resultOffset;
143        }
144
145        /** Returns the start position of the currently running match. */
146        public int getCurrentMatchStart() {
147                return currentMatchStart.peek();
148        }
149
150        /** Returns the reference position of the currently running match. */
151        public int getCurrentReferencePosition() {
152                return currentReferencePosition.peek();
153        }
154
155        /** Stores the given position as match start. */
156        public void markReferencePosition(int offset) {
157                currentReferencePosition.pop();
158                currentReferencePosition.push(offset);
159        }
160
161        /**
162         * Returns the size of the current entity stack. This can be used to
163         * determine the depth we are currently parsing at.
164         */
165        public int getEntityStackSize() {
166                return entities.size();
167        }
168}
