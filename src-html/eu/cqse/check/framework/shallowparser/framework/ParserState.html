017package eu.cqse.check.framework.shallowparser.framework;
018
019import java.util.ArrayList;
020import java.util.List;
021import java.util.Set;
022import java.util.Stack;
023
024import org.conqat.lib.commons.assertion.CCSMAssert;
025import org.conqat.lib.commons.collections.PairList;
026
027import eu.cqse.check.framework.scanner.IToken;
028
029/**
030 * This class holds the parser state and also implements parts of the parsing
031 * algorithm. The benefit of separating them into a separate class is that we
032 * can reuse the same parser multiple times without resets and also concurrently
033 * in separate threads.
034 *
035 * @see ShallowParserBase
036 */
037public class ParserState<STATE extends Enum<STATE>> {
038
039        /**
040         * Pairs of states (for which the recognizer is active) and the recognizer.
041         */
042        private final PairList<Set<STATE>, RecognizerBase<STATE>> recognizers;
043
044        /** Constructor. */
045        /* package */ ParserState(PairList<Set<STATE>, RecognizerBase<STATE>> recognizers) {
046                this.recognizers = recognizers;
047        }
048
049        /**
050         * Stack of entities created. If there are multiple top-level entities, this
051         * only contains the most recently created.
052         */
053        private final Stack<ShallowEntity> entities = new Stack<>();
054
055        /** Stack of start positions for nested parses. */
056        private final Stack<Integer> currentMatchStart = new Stack<>();
057
058        /** Stack of reference positions used for naming. */
059        private final Stack<Integer> currentReferencePosition = new Stack<>();
060
061        /** The result list of all created top-level entities. */
062        final List<ShallowEntity> result = new ArrayList<>();
063
064        /** Stores a forced next state (if any). */
065        private STATE forcedNextState = null;
066
067        /** Sets the given shallow entity as new target for adding children. */
068        public void setNode(ShallowEntity shallowEntity) {
069                if (entities.isEmpty()) {
070                        result.add(shallowEntity);
071                } else {
072                        entities.peek().addChild(shallowEntity);
073                }
074                entities.push(shallowEntity);
075
076                forcedNextState = null;
077        }
078
079        /** Marks the current node as completed. */
080        public void endNode(boolean expectsContinuation) {
081                CCSMAssert.isFalse(entities.isEmpty(), "Must have setNode() before each endNode()!");
082                entities.peek().setComplete(expectsContinuation);
083        }
084
085        /** Sets the name of the current node. */
086        public void setNodeName(String name) {
087                CCSMAssert.isFalse(entities.isEmpty(), "Must have setNode() before setNodeName()!");
088                entities.peek().setName(name);
089        }
090
091        /** Returns the name of the top-most ShallowEntity. */
092        public String getCurrentEntityName() {
093                CCSMAssert.isFalse(entities.isEmpty(), "May not access entity name when no nodes have been created before!");
094                return entities.peek().getName();
095        }
096
097        /** Returns the type of the top-most ShallowEntity. */
098        public EShallowEntityType getCurrentEntityType() {
099                CCSMAssert.isFalse(entities.isEmpty(), "May not access entity type when no nodes have been created before!");
100                return entities.peek().getType();
101        }
102
103        /**
104         * Attempts to parse exactly one entity in the given state. If the created
105         * entity is a continued node, this automatically continues parsing.
106         *
107         * @return Either the new parsing offset or {@link RecognizerBase#NO_MATCH}.
108         */
109        public int parse(STATE state, List<IToken> tokens, int startOffset) {
110                int entitiesSize = entities.size();
111
112                int resultOffset = RecognizerBase.NO_MATCH;
113                for (int i = 0; i < recognizers.size(); ++i) {
114                        if (recognizers.getFirst(i).contains(state)) {
115                                currentMatchStart.push(startOffset);
116                                currentReferencePosition.push(startOffset);
117
118                                int match = recognizers.getSecond(i).matches(this, tokens, startOffset);
119
120                                currentReferencePosition.pop();
121                                currentMatchStart.pop();
122
123                                if (match != RecognizerBase.NO_MATCH) {
124                                        resultOffset = match;
125                                        break;
126                                }
127                        }
128                }
129
130                // there was an entity created that needs completion
131                if (entities.size() > entitiesSize) {
132                        ShallowEntity entity = entities.pop();
133                        entity.setEndTokenIndex(resultOffset);
134
135                        // continue parsing if this is a continued entity (such as an
136                        // "if" followed by an "else").
137                        if (entity.isContinued() && resultOffset != RecognizerBase.NO_MATCH) {
138                                int nextOffset = parse(getContinuationState(state), tokens, resultOffset);
139
140                                if (nextOffset != RecognizerBase.NO_MATCH) {
141                                        return nextOffset;
142                                        // otherwise fall through and use last parsing position
143                                }
144                        }
145                }
146
147                return resultOffset;
148        }
149
150        /**
151         * Returns the state to use for node continuation, which might be
152         * {@link #forcedNextState} if set.
153         */
154        private STATE getContinuationState(STATE state) {
155                if (forcedNextState != null) {
156                        return forcedNextState;
157                }
158                return state;
159        }
160
161        /** Returns the start position of the currently running match. */
162        public int getCurrentMatchStart() {
163                return currentMatchStart.peek();
164        }
165
166        /** Returns the reference position of the currently running match. */
167        public int getCurrentReferencePosition() {
168                return currentReferencePosition.peek();
169        }
170
171        /** Stores the given position as match start. */
172        public void markReferencePosition(int offset) {
173                currentReferencePosition.pop();
174                currentReferencePosition.push(offset);
175        }
176
177        /**
178         * Returns the size of the current entity stack. This can be used to determine
179         * the depth we are currently parsing at.
180         */
181        public int getEntityStackSize() {
182                return entities.size();
183        }
184
185        /**
186         * Sets a "forced" state for the next parse in a "until" sequence. This is
187         * automatically reset in {@link #setNode(ShallowEntity)}. The current parsing
188         * alternative aborts if the next parse can't continue in the given state.
189         */
190        public void setForcedNextState(STATE forcedNextState) {
191                this.forcedNextState = forcedNextState;
192        }
193
194}
