017package eu.cqse.check.framework.shallowparser.framework;
018
019import java.lang.reflect.Array;
020import java.util.List;
021
022import org.conqat.lib.commons.region.Region;
023
024import eu.cqse.check.framework.scanner.ETokenType;
025import eu.cqse.check.framework.scanner.IToken;
026
027/**
028 * Utility methods for writing recognizers.
029 * 
030 * @author $Author: kinnen $
031 * @version $Rev: 53381 $
032 * @ConQAT.Rating GREEN Hash: 5035A330858C79BF4B8D9EE32C8FF96A
033 */
034public class RecognizerUtils {
035
036        /**
037         * Resolves a name based on the token stream and the parser state. The
038         * "name" can either be null, a constant name (String), an index into the
039         * token stream (int), a {@link Region} of the token stream, or an array
040         * (indicating multiple of those mentioned before.
041         * 
042         * @param extractStringContent
043         *            if this is true and the token is a string literal, the
044         *            surrounding quotes are removed.
045         */
046        public static <STATE extends Enum<STATE>> String resolveName(
047                        List<IToken> tokens, ParserState<STATE> parserState,
048                        int startOffset, Object name, boolean extractStringContent) {
049                if (name == null) {
050                        return null;
051                }
052
053                if (name instanceof String) {
054                        return (String) name;
055                }
056
057                if (name instanceof Number) {
058                        int index = ((Number) name).intValue();
059                        return normalizeText(
060                                        tokens.get(resolveIndex(parserState, startOffset, index)),
061                                        extractStringContent);
062                }
063
064                if (name instanceof Region) {
065                        return resolveRegionName(tokens, parserState, startOffset,
066                                        (Region) name, extractStringContent);
067                }
068
069                if (name.getClass().isArray()) {
070                        return resolveArrayName(tokens, parserState, startOffset, name,
071                                        extractStringContent);
072                }
073
074                throw new AssertionError("Unexpected type in resolving of name: "
075                                + name.getClass());
076        }
077
078        /** Handles name resolution for arrays. */
079        private static <STATE extends Enum<STATE>> String resolveArrayName(
080                        List<IToken> tokens, ParserState<STATE> parserState,
081                        int startOffset, Object name, boolean extractStringContent) {
082                StringBuilder sb = new StringBuilder();
083                for (int i = 0; i < Array.getLength(name); ++i) {
084                        if (i > 0) {
085                                sb.append(" ");
086                        }
087                        sb.append(resolveName(tokens, parserState, startOffset,
088                                        Array.get(name, i), extractStringContent));
089                }
090                return sb.toString();
091        }
092
093        /** Handles name resolution for regions. */
094        private static <STATE extends Enum<STATE>> String resolveRegionName(
095                        List<IToken> tokens, ParserState<STATE> parserState,
096                        int startOffset, Region region, boolean extractStringContent) {
097                int start = resolveIndex(parserState, startOffset, region.getStart());
098                int end = resolveIndex(parserState, startOffset, region.getEnd());
099                StringBuilder sb = new StringBuilder();
100                for (int i = start; i <= end; ++i) {
101                        if (i > start && !region.getOrigin().equals(Region.UNKNOWN_ORIGIN)) {
102                                sb.append(region.getOrigin());
103                        }
104                        sb.append(normalizeText(tokens.get(i), extractStringContent));
105                }
106                return sb.toString();
107        }
108
109        /** Resolves the index relative to the parser state. */
110        private static <STATE extends Enum<STATE>> int resolveIndex(
111                        ParserState<STATE> parserState, int startOffset, int index) {
112                if (index >= 0) {
113                        index = parserState.getCurrentReferencePosition() + index;
114                } else {
115                        // we need addition here, as index is negative
116                        index = startOffset + index;
117                }
118                return index;
119        }
120
121        /**
122         * Returns the token text. For case-insensitive languages, it is converted
123         * to lower case.
124         * 
125         * @param extractStringContent
126         *            if this is true and the token is a string literal, the
127         *            surrounding quotes are removed.
128         */
129        private static String normalizeText(IToken token,
130                        boolean extractStringContent) {
131                String text = token.getText();
132                if (extractStringContent
133                                && token.getType() == ETokenType.STRING_LITERAL) {
134                        return text.substring(1, text.length() - 1);
135                }
136
137                if (token.getLanguage().isCaseSensitive()) {
138                        return text;
139                }
140                return text.toLowerCase();
141        }
142}
