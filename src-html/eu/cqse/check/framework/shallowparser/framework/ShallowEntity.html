034/**
035 * An entity resulting from shallow parsing. For classes outside of this
036 * package, this class is immutable. An exception is made for addChild which can
037 * be overridden by subclasses.
038 * 
039 * Entities represent classes, methods, statements, etc.
040 * 
041 * Parser errors are usually marked by the inclusion of incomplete entities. For
042 * these entities, the end values (index, offset, etc.) may be wrong or even
043 * invalid.
044 * 
045 * @author $Author: heinemann $
046 * @version $Rev: 53883 $
047 * @ConQAT.Rating GREEN Hash: E0125068BA47B6E7C30BA86057FA2ECC
048 */
049public class ShallowEntity implements Serializable {
050
051        /** Version for serialization. */
052        private static final long serialVersionUID = 1L;
053
054        /** The type of this entity. */
055        private final EShallowEntityType type;
056
057        /**
058         * The subtype of this entity. Some well-known values are defined in
059         * {@link SubTypeNames}.
060         */
061        private final String subtype;
062
063        /** The name of the entity (may be null or empty string). */
064        private String name;
065
066        /** The parent entity (may be null for root entities). */
067        private ShallowEntity parent;
068
069        /** The list of children. */
070        private final List<ShallowEntity> children = new ArrayList<ShallowEntity>();
071
072        /** Whether this node has been completed, i.e. could be fully parsed. */
073        private boolean completed = false;
074
075        /**
076         * The underlying list of tokens (as seen by the shallow parser, i.e.
077         * without comments)
078         */
079        private final List<IToken> tokens;
080
081        /**
082         * The index of the first token in the token list as seen by the shallow
083         * parser
084         */
085        private int startTokenIndex;
086
087        /**
088         * The index of the end token (non-inclusive) in the token list as seen by
089         * the shallow parser
090         */
091        private int endTokenIndex = -1;
092
093        /**
094         * Marks this node as continued. See {@link #isContinued()} for an
095         * explanation.
096         */
097        private boolean continuedNode = false;

099        /** Constructor. */
100        public ShallowEntity(EShallowEntityType type, String subtype, String name,
101                        List<IToken> tokens, int startTokenIndex) {
102                this.type = type;
103                this.subtype = subtype;
104                this.name = name;
105                this.tokens = tokens;
106                this.startTokenIndex = startTokenIndex;
107        }

109        /**
110         * Returns the number of entity nodes including this entity, its children,
111         * grand children, etc.
112         */
113        public int getEntityCount() {
114                int result = 1;
115                for (ShallowEntity child : children) {
116                        result += child.getEntityCount();
117                }
118                return result;
119        }
120
121        /**
122         * Returns the number of entity nodes that are complete, including this
123         * entity, its children, grand children, etc.
124         */
125        public int getCompleteEntityCount() {
126                int result = 0;
127                if (isCompleted()) {
128                        result = 1;
129                }
130                for (ShallowEntity child : children) {
131                        result += child.getCompleteEntityCount();
132                }
133                return result;
134        }

136        /** Returns the type. */
137        public EShallowEntityType getType() {
138                return type;
139        }
140
141        /**
142         * Returns the subtype. Some well-known values are defined in
143         * {@link SubTypeNames}.
144         */
145        public String getSubtype() {
146                return subtype;
147        }
148
149        /**
150         * Returns the name of this entity. The result may be the empty String or
151         * <code>null</code>.
152         */
153        public String getName() {
154                return name;
155        }
156
157        /** Sets the name. */
158        /* package */void setName(String name) {
159                this.name = name;

162        /** Returns whether this shallow entity has children. */
163        public boolean hasChildren() {
164                return !children.isEmpty();
165        }
166
167        /**
168         * Returns the children. Returns an empty list, if there are no children.
169         */
170        public UnmodifiableList<ShallowEntity> getChildren() {
171                return CollectionUtils.asUnmodifiable(children);
172        }
173
174        /**
175         * Adds a child entity. As multiple top-level shallow entities can occur
176         * within one file, this method can be overridden by subclasses that
177         * simulate a root node of a shallow entity tree.
178         */
179        protected void addChild(ShallowEntity child) {
180                CCSMPre.isTrue(child.parent == null,
181                                "May not add entity to multiple parents!");
182                children.add(child);
183                child.parent = this;
184        }
185
186        /** Returns parent entity or null for the root. */
187        public ShallowEntity getParent() {
188                return parent;
189        }
190
191        /**
192         * Returns a new list containing all children of the given type in order.
193         */
194        public List<ShallowEntity> getChildrenOfType(EShallowEntityType type) {
195                List<ShallowEntity> result = new ArrayList<ShallowEntity>();
196                for (ShallowEntity child : children) {
197                        if (child.getType() == type) {
198                                result.add(child);
199                        }
200                }
201                return result;
202        }

204        /**
205         * Returns whether this node has been completed, i.e. could be fully parsed.
206         */
207        public boolean isCompleted() {
208                return completed;
209        }
210
211        /**
212         * Marks this node as completed and stores whether we expect the node to be
213         * continued. An example for a continued node is an "if" followed by an
214         * "else". After parsing the "if" and its body, the entity for the "if" is
215         * complete, but we expect the if statement to logically continue with the
216         * else.
217         */
218        /* package */void setComplete(boolean continuedNode) {
219                completed = true;
220                this.continuedNode = continuedNode;
221        }
222
223        /**
224         * Returns whether this node is continued, which means that the next sibling
225         * node logically is associated to this one. This feature is used, e.g., to
226         * connect the if-block and the corresponding else-block, which are parsed
227         * into separate nodes, but the first (if) node will be marked as continued.
228         * Note that even if this is true, a next sibling might not exist (typically
229         * the result of parsing errors).
230         */
231        public boolean isContinued() {
232                return continuedNode;
233        }

240        /**
241         * Returns the start token index. This is the index in a token list without
242         * comments.
243         */
244        public int getStartTokenIndex() {
245                return startTokenIndex;
246        }
247
248        /**
249         * Returns the start token index relative to the parent (i.e. this is valid
250         * for the tokens from {@link #includedTokens()} called for the parent).
251         */
252        public int getRelativeStartTokenIndex() {
253                if (parent == null) {
254                        return startTokenIndex;
255                }
256                return startTokenIndex - parent.startTokenIndex;
257        }
258
259        /**
260         * Returns the end token index relative to the parent (i.e. this is valid
261         * for the tokens from {@link #includedTokens()} called for the parent).
262         */
263        public int getRelativeEndTokenIndex() {
264                if (parent == null) {
265                        return endTokenIndex;
266                }
267                return endTokenIndex - parent.startTokenIndex;
268        }

275        /** Return start token */
276        private IToken getStartToken() {
277                return tokens.get(startTokenIndex);
278        }
279
280        /** Returns the (inclusive) offset of the start token */
281        public int getStartOffset() {
282                return getStartToken().getOffset();
283        }
284
285        /**
286         * Return (inclusive) offset of end token. This might be not the very last
287         * token, if parsing errors occurred.
288         */
289        public int getEndOffset() {
290                IToken endToken = getEndToken();
291                if (endToken == null) {
292                        return getStartToken().getEndOffset();
293                }
294                return endToken.getEndOffset();
295        }
296
297        /**
298         * Returns the (exclusive) end token index. This is the index in a list of
299         * tokens without comments. This may not be the correct index, if parsing
300         * errors occurred.
301         */
302        public int getEndTokenIndex() {
303                return endTokenIndex;
304        }
305
306        /**
307         * Returns the 1-based inclusive end line number. This might be not the very
308         * last line, if parsing errors occurred.
309         */
310        public int getEndLine() {
311                // we have to calculate +1 to convert to 1-based lines
312                IToken endToken = getEndToken();
313                if (endToken == null) {
314                        return getStartToken().getLineNumber() + 1;
315                }
316                return endToken.getLineNumber() + 1;
317        }
318
319        /** Return end token (or null if invalid). */
320        private IToken getEndToken() {
321                if (endTokenIndex <= 0) {
322                        return null;
323                }
324                return tokens.get(endTokenIndex - 1);
325        }
326
327        /**
328         * Returns a view of the included tokens without comments. This is typically
329         * non-empty, but may be empty in very special cases, such as parsing files
330         * with invalid syntax. For such files, this may even be empty although
331         * child token lists are non-empty.
332         */
333        public UnmodifiableList<IToken> includedTokens() {
334                return readOnlyTokenView(startTokenIndex, endTokenIndex);
335        }
336
337        /**
338         * Returns whether this is an empty entity, i.e. does not contain any
339         * tokens. An entity is empty only in very special cases, such as parsing
340         * files with invalid syntax.
341         */
342        public boolean isEmpty() {
343                return endTokenIndex <= startTokenIndex;
344        }
345
346        /**
347         * Returns a read-only view of the tokens from the (inclusive) start index
348         * to the (exclusive) end index.
349         */
350        private UnmodifiableList<IToken> readOnlyTokenView(int startIndex,
351                        int endIndex) {
352                if (endIndex <= startIndex) {
353                        return CollectionUtils.emptyList();
354                }
355                return CollectionUtils
356                                .asUnmodifiable(tokens.subList(startIndex, endIndex));
357        }
358
359        /**
360         * Returns a view of the tokens (without comments) from the beginning of the
361         * entity up to the first token included in the first child. For example for
362         * an if-block statement, this would include everything from the "if" to the
363         * first brace (inclusive). The first token of the first child statement
364         * would not be included.
365         */
366        public UnmodifiableList<IToken> ownStartTokens() {
367                if (children.isEmpty()) {
368                        return includedTokens();
369                }
370                return readOnlyTokenView(startTokenIndex,
371                                children.get(0).startTokenIndex);
372        }
373
374        /**
375         * Returns a view of the tokens (without comments) from the last token
376         * included in the last child up to the end of the entity. For example for a
377         * do-while statement, this would include everything from the last brace to
378         * the end of the "while". The last token of the last child statement would
379         * not be included.
380         */
381        public UnmodifiableList<IToken> ownEndTokens() {
382                if (children.isEmpty()) {
383                        return includedTokens();
384                }
385                return readOnlyTokenView(
386                                CollectionUtils.getLast(children).endTokenIndex, endTokenIndex);
387        }

403         * Returns a string representation of this node without including its

405         */
406        public String toLocalString() {
407                return toLocalString(getStartLine(), getEndLine());
408        }

410        /**
411         * Returns a string representation of this node without including its
412         * children using the given line representation.
413         */
414        private String toLocalString(int startLine, int endLine) {
415                String incomplete = StringUtils.EMPTY_STRING;
416                if (!isCompleted()) {
417                        incomplete = " [incomplete]";
418                }
419                return type + ": " + subtype + ": " + name + " (lines " + startLine
420                                + "-" + endLine + ")" + incomplete;
421        }
422
423        /** Returns an indented string representation. */
424        private void toString(StringBuilder sb, int indent) {
425                sb.append(StringUtils.fillString(2 * indent, ' '))
426                                .append(toLocalString()).append(StringUtils.CR);
427                for (ShallowEntity child : children) {
428                        child.toString(sb, indent + 1);
429                }
430        }

432        /**
433         * Traverses this entity depth-first. For details of visiting the entities,
434         * see {@link IShallowEntityVisitor}.
435         */
436        public void traverse(IShallowEntityVisitor visitor) {
437                if (visitor.visit(this)) {
438                        traverse(children, visitor);
439                }
440                visitor.endVisit(this);
441        }
442
443        /** Utility method for traversing multiple entities. */
444        public static void traverse(Collection<ShallowEntity> entities,
445                        IShallowEntityVisitor visitor) {
446                for (ShallowEntity entity : entities) {
447                        entity.traverse(visitor);
448                }
449        }

451        /**
452         * Removes all "contained" tokens (i.e. those for which
453         * {@link Predicate#test(Object)} returns true) from the underlying token
454         * stream and adjusts the entities accordingly.
455         */
456        public static void filterTokens(Collection<ShallowEntity> entities,
457                        Predicate<IToken> predicate) {
458                if (entities.isEmpty()) {
459                        return;
460                }
461
462                List<IToken> tokens = CollectionUtils.getAny(entities).tokens;
463                List<IToken> newTokens = new ArrayList<>();
464                final List<Integer> indexLookup = new ArrayList<>();
465
466                for (IToken token : tokens) {
467                        indexLookup.add(newTokens.size());
468                        if (!predicate.test(token)) {
469                                newTokens.add(token);
470                        }
471                }
472                indexLookup.add(newTokens.size());
473
474                tokens.clear();
475                tokens.addAll(newTokens);
476
477                traverse(entities, new ShallowEntityVisitorBase() {
478                        @Override
479                        public boolean visit(ShallowEntity entity) {
480                                entity.startTokenIndex = indexLookup
481                                                .get(entity.startTokenIndex);
482                                if (entity.endTokenIndex >= 0) {
483                                        entity.endTokenIndex = indexLookup
484                                                        .get(entity.endTokenIndex);
485                                }
486                                return true;
487                        }
488                });
489        }

491        /**
492         * Recursively rRemoves all completely empty shallow entities, i.e. those
493         * without own start and end tokens.
494         */
495        public static List<ShallowEntity> collapseEmptyEntities(
496                        Collection<ShallowEntity> entities) {
497                List<ShallowEntity> result = new ArrayList<>();
498                for (ShallowEntity entity : entities) {
499                        entity.collapseEmptyEntities();
500                        if (!entity.isCollapsible()) {
501                                result.add(entity);
502                        }
503                }
504                return result;
505        }
506
507        /**
508         * Removes all completely empty shallow entities, i.e. those without own
509         * start and end tokens, from the tree.
510         */
511        public void collapseEmptyEntities() {
512                for (ShallowEntity child : children) {
513                        child.collapseEmptyEntities();
514                }
515
516                List<ShallowEntity> newChildren = new ArrayList<>();
517                for (ShallowEntity child : children) {
518                        if (child.isCollapsible()) {
519                                newChildren.addAll(child.children);
520                        } else {
521                                newChildren.add(child);
522                        }
523                }
524
525                children.clear();
526                children.addAll(newChildren);
527        }
528
529        /** Returns whether this is collapsible, i.e. has no own tokens. */
530        private boolean isCollapsible() {
531                return ownStartTokens().isEmpty() && ownEndTokens().isEmpty();
532        }
