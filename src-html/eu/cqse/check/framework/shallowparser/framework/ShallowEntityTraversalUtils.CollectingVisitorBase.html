017package eu.cqse.check.framework.shallowparser.framework;
018
019import java.util.ArrayList;
020import java.util.Arrays;
021import java.util.Collection;
022import java.util.EnumSet;
023import java.util.List;
024import java.util.Set;
025import java.util.function.Predicate;
026
027/**
028 * Utility methods for working with shallow entities.
029 */
030public class ShallowEntityTraversalUtils {
031
032        /** Lists all entities. */
033        public static List<ShallowEntity> listAllEntities(Collection<ShallowEntity> entities) {
034                return listEntitiesOfTypes(entities, EnumSet.allOf(EShallowEntityType.class));
035        }
036
037        /** Lists all entities of the given type. */
038        public static List<ShallowEntity> listEntitiesOfType(Collection<ShallowEntity> entities, EShallowEntityType type) {
039                return listEntitiesOfTypes(entities, EnumSet.of(type));
040        }
041
042        /** Lists all entities of the given types. */
043        public static List<ShallowEntity> listEntitiesOfTypes(Collection<ShallowEntity> entities,
044                        final Set<EShallowEntityType> types) {
045                return new CollectingVisitorBase() {
046                        @Override
047                        protected boolean collect(ShallowEntity entity) {
048                                return types.contains(entity.getType());
049                        }
050                }.apply(entities);
051        }
052
053        /** Lists all entities that are selected by the given predicate. */
054        public static List<ShallowEntity> selectEntities(Collection<ShallowEntity> entities,
055                        final Predicate<ShallowEntity> predicate) {
056                return new CollectingVisitorBase() {
057                        @Override
058                        protected boolean collect(ShallowEntity entity) {
059                                return predicate.test(entity);
060                        }
061                }.apply(entities);
062        }
063
064        /**
065         * Returns the first incomplete entity found (or null). Unclosed entities
066         * correspond to parsing errors.
067         */
068        private static ShallowEntity findIncompleteEntity(ShallowEntity entity) {
069                if (!entity.isCompleted()) {
070                        return entity;
071                }
072                return findIncompleteEntity(entity.getChildren());
073        }
074
075        /**
076         * Returns the first incomplete entity found (or null). Unclosed entities
077         * correspond to parsing errors.
078         */
079        public static ShallowEntity findIncompleteEntity(List<ShallowEntity> entities) {
080                for (ShallowEntity entity : entities) {
081                        ShallowEntity incomplete = findIncompleteEntity(entity);
082                        if (incomplete != null) {
083                                return incomplete;
084                        }
085                }
086                return null;
087        }
088
089        /**
090         * Traverses the given collection of entities and returns a flat list of
091         * entities.
092         */
093        public static List<ShallowEntity> getAllEntities(Collection<ShallowEntity> entities) {
094                return new CollectingVisitorBase() {
095                        @Override
096                        protected boolean collect(ShallowEntity entity) {
097                                return true;
098                        }
099                }.apply(entities);
100        }
101
102        /**
103         * Lists all entities of the given type, but does not traverse into the
104         * entities, i.e. entities contained in the returned ones are not returned.
105         */
106        public static List<ShallowEntity> listEntitiesOfTypeNonRecursive(Collection<ShallowEntity> entities,
107                        final EShallowEntityType type) {
108                final List<ShallowEntity> methods = new ArrayList<>();
109                ShallowEntity.traverse(entities, new ShallowEntityVisitorBase() {
110                        /** {@inheritDoc} */
111                        @Override
112                        public boolean visit(ShallowEntity entity) {
113                                if (entity.getType() == type) {
114                                        methods.add(entity);
115                                        // do not recurse into methods
116                                        return false;
117                                }
118                                return true;
119                        }
120                });
121                return methods;
122        }
123
124        /**
125         * Returns all methods found in the given entities, but does not traverse
126         * into the entities, i.e. methods contained in other methods are not
127         * returned.
128         */
129        public static List<ShallowEntity> listMethodsNonRecursive(List<ShallowEntity> entities) {
130                return listEntitiesOfTypeNonRecursive(entities, EShallowEntityType.METHOD);
131        }
132
133        /** Empty default implementation of {@link IShallowEntityVisitor}. */
134        public static abstract class ShallowEntityVisitorBase implements IShallowEntityVisitor {
135                /** {@inheritDoc} */
136                @Override
137                public boolean visit(ShallowEntity entity) {
138                        return true;
139                }
140
141                /** {@inheritDoc} */
142                @Override
143                public void endVisit(ShallowEntity entity) {
144                        // nothing
145                }
146        }
147
148        /** Base class for visitors that collect shallow entities. */
149        public static abstract class CollectingVisitorBase extends ShallowEntityVisitorBase {
150
151                /** The collected entities. */
152                private final List<ShallowEntity> entities = new ArrayList<ShallowEntity>();
153
154                /** {@inheritDoc} */
155                @Override
156                public boolean visit(ShallowEntity entity) {
157                        if (collect(entity)) {
158                                entities.add(entity);
159                        }
160                        return true;
161                }
162
163                /**
164                 * Template method that returns true if the entity should be collected.
165                 */
166                protected abstract boolean collect(ShallowEntity entity);
167
168                /** Applies this collecting visitor and returns the collected result. */
169                public List<ShallowEntity> apply(Collection<ShallowEntity> entities) {
170                        ShallowEntity.traverse(entities, this);
171                        return this.entities;
172                }
173        }
174
175        /** Utility method for traversing multiple entities. */
176        public static void traverseWithKey(List<ShallowEntity> entities, IShallowEntityKeyVisitor visitor) {
177                traverseWithKey(entities, visitor, new int[0]);
178        }
179
180        /**
181         * Utility method for traversing multiple entities.
182         * 
183         * @param parentKey
184         *            Specifies the unique key of the entity lists parent.
185         */
186        private static void traverseWithKey(List<ShallowEntity> entities, IShallowEntityKeyVisitor visitor,
187                        int[] parentKey) {
188                int length = parentKey.length;
189                for (int i = 0; i < entities.size(); i++) {
190                        ShallowEntity entity = entities.get(i);
191                        int[] key = Arrays.copyOf(parentKey, length + 1);
192                        key[length] = i;
193                        if (visitor.visit(key, entity)) {
194                                traverseWithKey(entity.getChildren(), visitor, key);
195                        }
196                }
197        }
198
199        /** Empty default implementation of {@link IShallowEntityVisitor}. */
200        public interface IShallowEntityKeyVisitor {
201
202                /**
203                 * Denotes that visiting the entity begins.
204                 * 
205                 * @param key
206                 *            Key, which uniquely identifies the entity for the
207                 *            traversed list of {@link ShallowEntity}s.
208                 * 
209                 * @return true if the children of this entity are to be visited as
210                 *         well.
211                 */
212                public boolean visit(int[] key, ShallowEntity entity);
213        }
214
215        /**
216         * Returns a {@link ShallowEntity} from the given list of entities, which is
217         * identified by the given key.
218         * 
219         * @see ShallowEntityTraversalUtils#traverseWithKey(List,
220         *      IShallowEntityKeyVisitor)
221         */
222        public static ShallowEntity getEntityFromListWithKey(List<ShallowEntity> entities, int[] key) {
223                if (key.length == 0 || key[0] >= entities.size()) {
224                        return null;
225                }
226                ShallowEntity entity = entities.get(key[0]);
227                if (key.length == 1) {
228                        return entity;
229                }
230                return getEntityFromListWithKey(entity.getChildren(), Arrays.copyOfRange(key, 1, key.length));
231        }
232
233        /**
234         * Returns the next {@link ShallowEntity} after the given entity. Returns
235         * <code>null</code> if there is no subsequent entity.
236         */
237        public static ShallowEntity getSubsequentEntity(ShallowEntity entity) {
238                boolean found = false;
239                while (entity != null) {
240                        List<ShallowEntity> neighboringEntities = entity.getParent().getChildren();
241                        for (ShallowEntity neighboringEntity : neighboringEntities) {
242                                if (found) {
243                                        return neighboringEntity;
244                                } else if (neighboringEntity.equals(entity)) {
245                                        found = true;
246                                }
247                        }
248                        entity = entity.getParent();
249                        found = false;
250                }
251                return null;
252        }
