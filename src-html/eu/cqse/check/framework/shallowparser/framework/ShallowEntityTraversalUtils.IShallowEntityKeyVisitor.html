017package eu.cqse.check.framework.shallowparser.framework;
018
019import java.util.ArrayList;
020import java.util.Arrays;
021import java.util.Collection;
022import java.util.EnumSet;
023import java.util.List;
024import java.util.Set;
025import java.util.function.Predicate;
026
027/**
028 * Utility methods for working with shallow entities.
029 * 
030 * @author $Author: kinnen $
031 * @version $Rev: 54460 $
032 * @ConQAT.Rating GREEN Hash: F994E73684ED576DE11942A958CEEF3A
033 */
034public class ShallowEntityTraversalUtils {
035
036        /** Lists all entities. */
037        public static List<ShallowEntity> listAllEntities(
038                        Collection<ShallowEntity> entities) {
039                return listEntitiesOfTypes(entities,
040                                EnumSet.allOf(EShallowEntityType.class));
041        }
042
043        /** Lists all entities of the given type. */
044        public static List<ShallowEntity> listEntitiesOfType(
045                        Collection<ShallowEntity> entities, EShallowEntityType type) {
046                return listEntitiesOfTypes(entities, EnumSet.of(type));
047        }
048
049        /** Lists all entities of the given types. */
050        public static List<ShallowEntity> listEntitiesOfTypes(
051                        Collection<ShallowEntity> entities,
052                        final Set<EShallowEntityType> types) {
053                return new CollectingVisitorBase() {
054                        @Override
055                        protected boolean collect(ShallowEntity entity) {
056                                return types.contains(entity.getType());
057                        }
058                }.apply(entities);
059        }
060
061        /** Lists all entities that are selected by the given predicate. */
062        public static List<ShallowEntity> selectEntities(
063                        Collection<ShallowEntity> entities,
064                        final Predicate<ShallowEntity> predicate) {
065                return new CollectingVisitorBase() {
066                        @Override
067                        protected boolean collect(ShallowEntity entity) {
068                                return predicate.test(entity);
069                        }
070                }.apply(entities);
071        }
072
073        /**
074         * Returns the first incomplete entity found (or null). Unclosed entities
075         * correspond to parsing errors.
076         */
077        private static ShallowEntity findIncompleteEntity(ShallowEntity entity) {
078                if (!entity.isCompleted()) {
079                        return entity;
080                }
081                return findIncompleteEntity(entity.getChildren());
082        }
083
084        /**
085         * Returns the first incomplete entity found (or null). Unclosed entities
086         * correspond to parsing errors.
087         */
088        public static ShallowEntity findIncompleteEntity(
089                        List<ShallowEntity> entities) {
090                for (ShallowEntity entity : entities) {
091                        ShallowEntity incomplete = findIncompleteEntity(entity);
092                        if (incomplete != null) {
093                                return incomplete;
094                        }
095                }
096                return null;
097        }
098
099        /**
100         * Traverses the given collection of entities and returns a flat list of
101         * entities.
102         */
103        public static List<ShallowEntity> getAllEntities(
104                        Collection<ShallowEntity> entities) {
105                return new CollectingVisitorBase() {
106                        @Override
107                        protected boolean collect(ShallowEntity entity) {
108                                return true;
109                        }
110                }.apply(entities);
111        }
112
113        /**
114         * Lists all entities of the given type, but does not traverse into the
115         * entities, i.e. entities contained in the returned ones are not returned.
116         */
117        public static List<ShallowEntity> listEntitiesOfTypeNonRecursive(
118                        Collection<ShallowEntity> entities, final EShallowEntityType type) {
119                final List<ShallowEntity> methods = new ArrayList<>();
120                ShallowEntity.traverse(entities, new ShallowEntityVisitorBase() {
121                        /** {@inheritDoc} */
122                        @Override
123                        public boolean visit(ShallowEntity entity) {
124                                if (entity.getType() == type) {
125                                        methods.add(entity);
126                                        // do not recurse into methods
127                                        return false;
128                                }
129                                return true;
130                        }
131                });
132                return methods;
133        }
134
135        /**
136         * Returns all methods found in the given entities, but does not traverse
137         * into the entities, i.e. methods contained in other methods are not
138         * returned.
139         */
140        public static List<ShallowEntity> listMethodsNonRecursive(
141                        List<ShallowEntity> entities) {
142                return listEntitiesOfTypeNonRecursive(entities,
143                                EShallowEntityType.METHOD);
144        }
145
146        /** Empty default implementation of {@link IShallowEntityVisitor}. */
147        public static abstract class ShallowEntityVisitorBase
148                        implements IShallowEntityVisitor {
149                /** {@inheritDoc} */
150                @Override
151                public boolean visit(ShallowEntity entity) {
152                        return true;
153                }
154
155                /** {@inheritDoc} */
156                @Override
157                public void endVisit(ShallowEntity entity) {
158                        // nothing
159                }
160        }
161
162        /** Base class for visitors that collect shallow entities. */
163        public static abstract class CollectingVisitorBase
164                        extends ShallowEntityVisitorBase {
165
166                /** The collected entities. */
167                private final List<ShallowEntity> entities = new ArrayList<ShallowEntity>();
168
169                /** {@inheritDoc} */
170                @Override
171                public boolean visit(ShallowEntity entity) {
172                        if (collect(entity)) {
173                                entities.add(entity);
174                        }
175                        return true;
176                }
177
178                /**
179                 * Template method that returns true if the entity should be collected.
180                 */
181                protected abstract boolean collect(ShallowEntity entity);
182
183                /** Applies this collecting visitor and returns the collected result. */
184                public List<ShallowEntity> apply(Collection<ShallowEntity> entities) {
185                        ShallowEntity.traverse(entities, this);
186                        return this.entities;
187                }
188        }
189
190        /** Utility method for traversing multiple entities. */
191        public static void traverseWithKey(List<ShallowEntity> entities,
192                        IShallowEntityKeyVisitor visitor) {
193                traverseWithKey(entities, visitor, new int[0]);
194        }
195
196        /**
197         * Utility method for traversing multiple entities.
198         * 
199         * @param parentKey
200         *            Specifies the unique key of the entity lists parent.
201         */
202        private static void traverseWithKey(List<ShallowEntity> entities,
203                        IShallowEntityKeyVisitor visitor, int[] parentKey) {
204                int length = parentKey.length;
205                for (int i = 0; i < entities.size(); i++) {
206                        ShallowEntity entity = entities.get(i);
207                        int[] key = Arrays.copyOf(parentKey, length + 1);
208                        key[length] = i;
209                        if (visitor.visit(key, entity)) {
210                                traverseWithKey(entity.getChildren(), visitor, key);
211                        }
212                }
213        }
214
215        /** Empty default implementation of {@link IShallowEntityVisitor}. */
216        public interface IShallowEntityKeyVisitor {
217
218                /**
219                 * Denotes that visiting the entity begins.
220                 * 
221                 * @param key
222                 *            Key, which uniquely identifies the entity for the
223                 *            traversed list of {@link ShallowEntity}s.
224                 * 
225                 * @return true if the children of this entity are to be visited as
226                 *         well.
227                 */
228                public boolean visit(int[] key, ShallowEntity entity);
229        }
230
231        /**
232         * Returns a {@link ShallowEntity} from the given list of entities, which is
233         * identified by the given key.
234         * 
235         * @see ShallowEntityTraversalUtils#traverseWithKey(List,
236         *      IShallowEntityKeyVisitor)
237         */
238        public static ShallowEntity getEntityFromListWithKey(
239                        List<ShallowEntity> entities, int[] key) {
240                if (key.length == 0 || key[0] >= entities.size()) {
241                        return null;
242                }
243                ShallowEntity entity = entities.get(key[0]);
244                if (key.length == 1) {
245                        return entity;
246                }
247                return getEntityFromListWithKey(entity.getChildren(),
248                                Arrays.copyOfRange(key, 1, key.length));
249        }
