017package eu.cqse.check.framework.shallowparser.framework;
018
019import java.util.ArrayList;
020import java.util.Arrays;
021import java.util.EnumSet;
022import java.util.List;
023import java.util.Set;
024
025import org.conqat.lib.commons.assertion.CCSMAssert;
026import org.conqat.lib.commons.collections.PairList;
027
028import eu.cqse.check.framework.scanner.ETokenType;
029import eu.cqse.check.framework.scanner.ETokenType.ETokenClass;
030import eu.cqse.check.framework.scanner.IToken;
031import eu.cqse.check.framework.shallowparser.IShallowParser;
032
033/**
034 * Base class for shallow parsers that implements most of the generic parsing
035 * logic.
036 * 
037 * @param <STATE>
038 *            the enum used for describing parse states.
039 */
040public abstract class ShallowParserBase<STATE extends Enum<STATE>> implements IShallowParser {
041
042        /** The class of the enum used for describing the states. */
043        private final Class<STATE> stateClass;
044
045        /** The initial state corresponding to a top-level element. */
046        private final STATE initialState;
047
048        /**
049         * Pairs of states (for which the recognizer is active) and the recognizer.
050         */
051        private final PairList<Set<STATE>, RecognizerBase<STATE>> recognizers = new PairList<Set<STATE>, RecognizerBase<STATE>>();
052
053        /** Constructor. */
054        protected ShallowParserBase(Class<STATE> stateClass, STATE initialState) {
055                this.stateClass = stateClass;
056                this.initialState = initialState;
057        }
058
059        /** Creates a rule that is active in each of the given states. */
060        @SuppressWarnings("unchecked")
061        protected RecognizerBase<STATE> inState(STATE... states) {
062                RecognizerBase<STATE> recognizer = emptyRecognizer();
063
064                // we need this seemingly complicated construct as states may be empty
065                EnumSet<STATE> stateSet = EnumSet.noneOf(stateClass);
066                stateSet.addAll(Arrays.asList(states));
067                recognizers.add(stateSet, recognizer);
068
069                return recognizer;
070        }
071
072        /** Returns an empty recognizer that can be used for local sub-rules. */
073        protected RecognizerBase<STATE> emptyRecognizer() {
074                return new RecognizerBase<STATE>() {
075
076                        /** {@inheritDoc} */
077                        @Override
078                        protected String getRecognizerStringRepresentation() {
079                                return "empty";
080                        }
081                };
082        }
083
084        /** Creates a rule that is active in any state. */
085        protected RecognizerBase<STATE> inAnyState() {
086                return inState(stateClass.getEnumConstants());
087        }
088
089        /**
090         * Template method that can be used to ignore/filter certain tokens. Default
091         * implementation ignores comments.
092         * 
093         * @param previousToken
094         *            the last non-filtered token (can be used for context sensitive
095         *            filtering). This may be null is no previous token exists.
096         */
097        protected boolean isFilteredToken(IToken token, IToken previousToken) {
098                return token.getType().getTokenClass() == ETokenClass.COMMENT;
099        }
100
101        /** {@inheritDoc} */
102        @Override
103        public List<ShallowEntity> parseTopLevel(List<IToken> tokens) {
104                tokens = filterTokens(tokens);
105                return parse(initialState, tokens);
106        }
107
108        /**
109         * {@inheritDoc}
110         * <p>
111         * This works by attempting to parse from all possible states and returns
112         * the best result, where 'best' is measured in terms of completed entities.
113         */
114        @Override
115        public List<ShallowEntity> parseFragment(List<IToken> tokens) {
116                tokens = filterTokens(tokens);
117
118                List<ShallowEntity> best = new ArrayList<ShallowEntity>();
119                int bestCount = 0;
120                int bestStart = Integer.MAX_VALUE;
121
122                for (STATE state : stateClass.getEnumConstants()) {
123                        List<ShallowEntity> list = parse(state, tokens);
124                        int count = countCompleteEntities(list);
125                        int start = Integer.MAX_VALUE;
126                        if (!list.isEmpty()) {
127                                start = list.get(0).getStartTokenIndex();
128                        }
129
130                        // prefer parse results that produce more entities (or if same
131                        // number those that start earlier, i.e., skip less tokens)
132                        if (count > bestCount || (count == bestCount && start < bestStart)) {
133                                best = list;
134                                bestCount = count;
135                                bestStart = start;
136                        }
137                }
138                return best;
139        }
140
141        /** Recursively counts and returns the entities contained in the list. */
142        private static int countCompleteEntities(List<ShallowEntity> list) {
143                int result = 0;
144                for (ShallowEntity entity : list) {
145                        result += entity.getCompleteEntityCount();
146                }
147                return result;
148        }
149
150        /** Creates a filtered view of the tokens. */
151        protected List<IToken> filterTokens(List<IToken> tokens) {
152                List<IToken> result = new ArrayList<IToken>();
153                IToken previousToken = null;
154                for (IToken token : tokens) {
155                        if (!isFilteredToken(token, previousToken)) {
156                                result.add(token);
157                                previousToken = token;
158                        }
159                }
160                return result;
161        }
162
163        /**
164         * Performs a shallow parse for the given list of tokens and given initial
165         * state.
166         */
167        protected List<ShallowEntity> parse(STATE startState, List<IToken> tokens) {
168                ParserState<STATE> parserState = new ParserState<STATE>(recognizers);
169
170                int offset = 0;
171                while (offset < tokens.size()) {
172                        int match = parserState.parse(startState, tokens, offset);
173                        if (match == RecognizerBase.NO_MATCH) {
174                                offset += 1;
175                        } else {
176                                CCSMAssert.isTrue(match > offset, "The parser does not make any progress!");
177                                offset = match;
178                        }
179                }
180
181                return parserState.result;
182        }
183
184        /**
185         * Completes a recognizer with an end. If the next token is one of the
186         * provided continuation tokens, the entity creation finishes with a
187         * continued node.
188         * 
189         * @param continuationTokens
190         *            list of tokens that indicate a continued statement if
191         *            encountered at the end. May be null.
192         */
193        protected void endWithPossibleContinuation(RecognizerBase<STATE> recognizer,
194                        EnumSet<ETokenType> continuationTokens) {
195                if (continuationTokens != null && !continuationTokens.isEmpty()) {
196                        recognizer.sequenceBefore(continuationTokens).endNodeWithContinuation();
197                }
198                recognizer.endNode();
199        }
200}
