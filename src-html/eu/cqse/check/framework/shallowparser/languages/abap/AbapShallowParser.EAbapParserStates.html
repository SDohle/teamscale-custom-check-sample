037/**
038 * Shallow parser for ABAP. The following links are useful for writing the
039 * parser:
040 * <ul>
041 * <li><a href="http://help.sap.com/abapdocu_702/en/">ABAP Keyword
042 * Documentation</a></li>
043 * <li><a href=
044 * "http://help.sap.com/abapdocu_702/en/abenabap_statements_overview.htm" >ABAP
045 * Statements Overview</a></li>
046 * </ul>
047 */
048public class AbapShallowParser extends ShallowParserBase<AbapShallowParser.EAbapParserStates> {
049
050        /**
051         * Since there are no reserved words in ABAP, one can use statement names as
052         * identifiers. Therefore, whenever we expect an identifier, it's not safe
053         * to rely on the Scanner's categorization of identifiers, and we should
054         * allow keywords as well.
055         */
056        private static final EnumSet<ETokenClass> IDENTIFIER_LIKE = EnumSet.of(ETokenClass.IDENTIFIER, ETokenClass.KEYWORD);
057
058        /** Tokens that can introduce a simple statement. */
059        private static final EnumSet<ETokenType> SIMPLE_STATEMENT_START_TOKENS = EnumSet.of(ADD, ADD_CORRESPONDING, APPEND,
060                        ASSERT, ASSIGN, AUTHORITY_CHECK, BACK, BREAK_POINT, CALL, CHECK, CLEAR, CLOSE, COLLECT, COMMIT,
061                        COMMUNICATION, COMPUTE, CONCATENATE, CONDENSE, CONSTANTS, CONTEXTS, CONTINUE, CONTROLS, CONVERT, CREATE,
062                        DATA, DELETE, DEMAND, DESCRIBE, DETAIL, DIVIDE, DIVIDE_CORRESPONDING, EDITOR_CALL, ENHANCEMENT_POINT,
063                        EXISTS, EXIT, EXPORT, EXTRACT, FETCH, FIELDS, FIELD_GROUPS, FIELD_SYMBOLS, FIND, FORMAT, FREE, GENERATE,
064                        GET, HIDE, IDENTIFIER, IMPORT, INCLUDE, INFOTYPES, INPUT, INSERT, LEAVE, LOAD, LOCAL, LOG_POINT, MAXIMUM,
065                        MESSAGE, MINIMUM, MODIFY, MODULE, MOVE, MOVE_CORRESPONDING, MULTIPLY_CORRESPONDING, MULTIPLY, NAME,
066                        NEW_LINE, NEW_PAGE, NEW_SECTION, OPEN, OVERLAY, PACK, PACKAGE, PERFORM, POSITION, PRINT_CONTROL, PUT, RAISE,
067                        RANGES, READ, REFRESH, REJECT, REPLACE, RESERVE, RESUME, RETURN, ROLLBACK, SCROLL, SEARCH, SET, SHIFT, SKIP,
068                        SORT, SPLIT, STATICS, STOP, SUBMIT, SUBTRACT, SUBTRACT_CORRESPONDING, SUM, SUMMARY, SUMMING, SUPPLY,
069                        SUPPRESS, SYNTAX_CHECK, TRANSFER, TRANSLATE, TRUNCATE, TYPES, ULINE, UNASSIGN, UNPACK, UPDATE, WAIT, WINDOW,
070                        WRITE);

072        /**
073         * Set of keywords that start an event block (without keywords that require
074         * a preceding "at")
075         */
076        private static final EnumSet<ETokenType> EVENT_BLOCKS = EnumSet.of(INITIALIZATION, START_OF_SELECTION,
077                        END_OF_SELECTION, TOP_OF_PAGE, END_OF_PAGE, LOAD_OF_PROGRAM);
078
079        /**
080         * Set of keywords that end an event block (possibly indicating the start of
081         * the next one)
082         */
083        private static final EnumSet<ETokenType> EVENT_BLOCKS_END = EnumSet.of(AT, FORM, CLASS, INTERFACE);
084
085        // all tokens that start a new event block also end the current one.
086        static {
087                EVENT_BLOCKS_END.addAll(EVENT_BLOCKS);
088        }
089
090        /** The states used in this parser. */
091        public static enum EAbapParserStates {
092
093                /**
094                 * Top level state used for parsing constructs that are not nested in
095                 * other constructs.
096                 */
097                TOPLEVEL,
098
099                /**
100                 * A state to recognize declarations within classes. As many constructs
101                 * are allowed both top-level and in declarations, many rules are
102                 * registered for both.
103                 */
104                DECLARATIONS,
105
106                /**
107                 * A state to recognize statements, i.e. plain code in functions, etc.
108                 */
109                STATEMENTS
110        }
111
112        /** Constructor. */
113        public AbapShallowParser() {
114                super(EAbapParserStates.class, TOPLEVEL);
115
116                createMetaRules();
117                createTopLevelRules();
118                createTypeRules();
119                createAttributeRules();
120                createMethodRules();
121                createModuleRules();
122                createStatementRules();
123
124                inAnyState().sequence(DOT).createNode(EShallowEntityType.STATEMENT, SubTypeNames.EMPTY_STATEMENT).endNode();
125        }
126
127        /** Rules for parsing elements that are only expected top-level. */
128        private void createTopLevelRules() {
129
130                // since the report is not really a method, its statements are still
131                // parsed in the toplevel scope, not the statement scope
132                RecognizerBase<EAbapParserStates> reportRecognizer = inState(TOPLEVEL).sequence(EnumSet.of(REPORT, PROGRAM))
133                                .optional(COLON).sequence(IDENTIFIER_LIKE).createNode(EShallowEntityType.METHOD, 0, -1).skipTo(DOT)
134                                .parseUntilOrEof(TOPLEVEL);
135                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, reportRecognizer);
136
137                inState(TOPLEVEL).sequence(EnumSet.of(SELECTION_SCREEN)).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
138                                .endNode();
139
140                // top-of-page during line-selection is different from "top-of-page"
141                RecognizerBase<EAbapParserStates> topsRecognizer = inState(TOPLEVEL)
142                                .sequence(TOP_OF_PAGE, DURING, LINE_SELECTION)
143                                .createNode(EShallowEntityType.METHOD, SubTypeNames.TOP_OF_PAGE_DURING_LINE_SELECTION).skipTo(DOT)
144                                .parseUntilOrEof(STATEMENTS);
145                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, topsRecognizer);
146
147                // basic event blocks without "at"
148                RecognizerBase<EAbapParserStates> eventBlockRecognizer = inState(TOPLEVEL).sequence(EVENT_BLOCKS)
149                                .createNode(EShallowEntityType.METHOD, 0).skipTo(DOT).parseUntilOrEof(STATEMENTS);
150                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, eventBlockRecognizer);
151
152                createGetBlockRules();
153
154                createAtBlockRules();
155        }
156
157        /**
158         * Creates rules constructs starting with "GET". These can either be
159         * statements or event handlers.
160         */
161        private void createGetBlockRules() {
162                // "get reference of" and consorts
163                inState(TOPLEVEL, STATEMENTS).sequence(GET, ETokenClass.KEYWORD)
164                                .createNode(EShallowEntityType.STATEMENT, new int[] { 0, 1 }).skipTo(DOT).endNode();
165
166                // get ... late event handler
167                RecognizerBase<EAbapParserStates> getLateEventBlockRecognizer = inState(TOPLEVEL)
168                                .sequence(GET, IDENTIFIER, LATE).createNode(EShallowEntityType.METHOD, SubTypeNames.GET_LATE, 1)
169                                .skipTo(DOT).parseUntilOrEof(STATEMENTS);
170                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, getLateEventBlockRecognizer);
171
172                // get ... event handler
173                RecognizerBase<EAbapParserStates> getEventBlockRecognizer = inState(TOPLEVEL).sequence(GET, IDENTIFIER)
174                                .createNode(EShallowEntityType.METHOD, 0, 1).skipTo(DOT).parseUntilOrEof(STATEMENTS);
175                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, getEventBlockRecognizer);
176        }
177
178        /** Creates parsing rules for event handlers starting with "AT". */
179        private void createAtBlockRules() {
180                // at selection screen event handler (named)
181                RecognizerBase<EAbapParserStates> selectionScreenBlockRecognizer = inState(TOPLEVEL)
182                                .sequence(AT, SELECTION_SCREEN).skipTo(DOT).createNode(EShallowEntityType.METHOD,
183                                                SubTypeNames.AT_SELECTION_SCREEN, new Region(0, -2, StringUtils.SPACE))
184                                .parseUntilOrEof(STATEMENTS);
185                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, selectionScreenBlockRecognizer);
186
187                // at PF<..> event handler (named)
188                RecognizerBase<EAbapParserStates> atPfBlockRecognizer = inState(TOPLEVEL).sequence(AT, PF_EVENT)
189                                .createNode(EShallowEntityType.METHOD, SubTypeNames.PF_EVENT, new Region(1, 1)).skipTo(DOT)
190                                .parseUntilOrEof(STATEMENTS);
191                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, atPfBlockRecognizer);
192
193                // at ... event handler (anonymous)
194                RecognizerBase<EAbapParserStates> atEventBlockRecognizer = inState(TOPLEVEL)
195                                .sequence(AT, EnumSet.of(LINE_SELECTION, USER_COMMAND))
196                                .createNode(EShallowEntityType.METHOD, new int[] { 0, 1 }).skipTo(DOT).parseUntilOrEof(STATEMENTS);
197                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, atEventBlockRecognizer);
198        }
199
200        /**
201         * Ends the given recognizer if it either hits one of the given event block
202         * end tokens or a get event block or a module block.
203         */
204        private static void endMethodEntityOnEventBlock(EnumSet<ETokenType> eventBlocksEnd,
205                        RecognizerBase<EAbapParserStates> eventBlockRecognizer) {
206                eventBlockRecognizer.sequenceBefore(eventBlocksEnd).endNode();
207                eventBlockRecognizer.sequenceBefore(GET, IDENTIFIER).endNode();
208                eventBlockRecognizer.preCondition(new ModuleBlockStartRecognizer()).endNode();
209        }

211        /** Rules for parsing of meta elements. */
212        private void createMetaRules() {
213
214                inState(DECLARATIONS).sequence(EnumSet.of(PUBLIC, PROTECTED, PRIVATE), SECTION, DOT)
215                                .createNode(EShallowEntityType.META, SubTypeNames.VISIBILITY, 0).endNode();
216
217                inAnyState().sequence(EnumSet.of(TYPE_POOLS, TABLES)).createNode(EShallowEntityType.META, 0).skipTo(DOT)
218                                .endNode();
219
220                inAnyState().sequence(DEFINE).createNode(EShallowEntityType.META, SubTypeNames.MACRO)
221                                .skipTo(END_OF_DEFINITION, DOT).endNode();
222
223                inState(DECLARATIONS).sequence(EnumSet.of(INTERFACES, ALIASES)).createNode(EShallowEntityType.META, 0)
224                                .skipTo(DOT).endNode();
225        }

227        /** Rules for parsing classes, interfaces, types, and events. */
228        private void createTypeRules() {
229
230                createClassRules();
231
232                // interfaces
233                RecognizerBase<EAbapParserStates> interfaceAlternative = inState(TOPLEVEL, DECLARATIONS).sequence(INTERFACE,
234                                IDENTIFIER);
235                interfaceAlternative.sequence(EnumSet.of(LOAD, DEFERRED, LOCAL))
236                                .createNode(EShallowEntityType.TYPE, SubTypeNames.INTERFACE_PUBLICATION, 1).skipTo(DOT).endNode();
237                interfaceAlternative.createNode(EShallowEntityType.TYPE, SubTypeNames.INTERFACE_DEFINITION, 1).skipTo(DOT)
238                                .parseUntil(DECLARATIONS).sequence(ENDINTERFACE, DOT).endNode();
239
240        }
241
242        /** Rules for parsing ABAP OO classes. */
243        private void createClassRules() {
244                RecognizerBase<EAbapParserStates> classDefinitionAlternative = inState(TOPLEVEL, DECLARATIONS).sequence(CLASS,
245                                IDENTIFIER, DEFINITION);
246                classDefinitionAlternative.sequence(EnumSet.of(LOAD, DEFERRED, LOCAL))
247                                .createNode(EShallowEntityType.TYPE, SubTypeNames.CLASS_PUBLICATION, 1).skipTo(DOT).endNode();
248                classDefinitionAlternative.createNode(EShallowEntityType.TYPE, SubTypeNames.CLASS_DEFINITION, 1).skipTo(DOT)
249                                .parseUntil(DECLARATIONS).sequence(ENDCLASS, DOT).endNode();
250
251                inState(TOPLEVEL, DECLARATIONS).sequence(CLASS, IDENTIFIER, IMPLEMENTATION)
252                                .createNode(EShallowEntityType.TYPE, SubTypeNames.CLASS_IMPLEMENTATION, 1).skipTo(DOT)
253                                .parseUntil(DECLARATIONS).sequence(ENDCLASS, DOT).endNode();
254        }
255
256        /** Rules for parsing attributes. */
257        private void createAttributeRules() {
258                // Match record type declaration as single entity
259                inState(TOPLEVEL, DECLARATIONS).sequence(EnumSet.of(TYPES, CONSTANTS, DATA, STATICS, CLASS_DATA))
260                                .optional(COLON).sequence(BEGIN, OF, IDENTIFIER_LIKE).createNode(EShallowEntityType.ATTRIBUTE, 0, -1)
261                                .skipTo(END, OF).skipTo(DOT).endNode();
262
263                // types, events, class events
264                inState(TOPLEVEL, DECLARATIONS).sequence(EnumSet.of(EVENTS, CLASS_EVENTS))
265                                .createNode(EShallowEntityType.ATTRIBUTE, 0).skipTo(DOT).endNode();
266
267                inState(TOPLEVEL, DECLARATIONS)
268                                .sequence(EnumSet.of(CONSTANTS, NODES, STATICS, DATA, TYPES, FIELD_GROUPS, CLASS_DATA, RANGES,
269                                                SELECT_OPTIONS))
270                                .optional(COLON).sequence(IDENTIFIER_LIKE).createNode(EShallowEntityType.ATTRIBUTE, 0, -1).skipTo(DOT)
271                                .endNode();
272
273                inState(TOPLEVEL, DECLARATIONS).sequence(FIELD_SYMBOLS).optional(COLON).sequence(LT, IDENTIFIER_LIKE)
274                                .createNode(EShallowEntityType.ATTRIBUTE, 0, -1).skipTo(DOT).endNode();
275
276                inState(TOPLEVEL, DECLARATIONS)
277                                // The "parameter" keyword is the obsolete older form of
278                                // "parameters"
279                                .sequence(EnumSet.of(PARAMETER, PARAMETERS)).optional(COLON).sequence(IDENTIFIER_LIKE)
280                                .createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.PARAMETERS, -1).skipTo(DOT).endNode();
281        }

283        /**
284         * Rules for parsing methods, functions, and forms.
285         *
286         * We handle colons here to catch method declarations like <code>form:
287         * foo.</foo> as that's a declaration for form "foo" not ":foo".
288         */
289        private void createMethodRules() {
290                inState(DECLARATIONS).sequence(EnumSet.of(METHODS, CLASS_METHODS))
291                                .skipBefore(EnumSet.of(DOT, RETURNING, IMPORTING, EXPORTING, REDEFINITION, CHANGING, EXCEPTIONS))
292                                .createNode(EShallowEntityType.METHOD, SubTypeNames.METHOD_DECLARATION, new Region(1, -1)).skipTo(DOT)
293                                .endNode();
294
295                inState(DECLARATIONS).sequence(METHOD).optional(COLON).markStart().skipBefore(EnumSet.of(DOT, BY))
296                                .createNode(EShallowEntityType.METHOD, SubTypeNames.METHOD_IMPLEMENTATION, new Region(0, -1))
297                                .skipTo(DOT).parseUntil(STATEMENTS).sequence(ENDMETHOD, DOT).endNode();
298
299                inState(TOPLEVEL, DECLARATIONS).sequence(FUNCTION).optional(COLON).markStart()
300                                .skipBefore(EnumSet.of(DOT, IMPORTING, EXPORTING, TABLES, CHANGING, RAISING, EXCEPTIONS))
301                                .createNode(EShallowEntityType.METHOD, SubTypeNames.FUNCTION, new Region(0, -1)).skipTo(DOT)
302                                .parseUntil(STATEMENTS).sequence(ENDFUNCTION, DOT).endNode();
303
304                inState(TOPLEVEL).sequence(FORM).optional(COLON).sequence(IDENTIFIER_LIKE)
305                                .createNode(EShallowEntityType.METHOD, SubTypeNames.FORM, -1).skipTo(DOT).parseUntil(STATEMENTS)
306                                .sequence(ENDFORM, DOT).endNode();
307        }
308
309        /** Rules for parsing modules. */
310        private void createModuleRules() {
311                RecognizerBase<EAbapParserStates> moduleRecognizer = inState(TOPLEVEL)
312                                .preCondition(new ModuleBlockStartRecognizer()).sequence(MODULE).markStart()
313                                .skipBefore(EnumSet.of(INPUT, OUTPUT, DOT));
314                moduleRecognizer.sequence(INPUT, DOT).createNode(EShallowEntityType.METHOD, SubTypeNames.MODULE_INPUT, 0)
315                                .parseUntil(STATEMENTS).sequence(ENDMODULE, DOT).endNode();
316                moduleRecognizer.sequence(OUTPUT, DOT).createNode(EShallowEntityType.METHOD, SubTypeNames.MODULE_OUTPUT, 0)
317                                .parseUntil(STATEMENTS).sequence(ENDMODULE, DOT).endNode();
318                moduleRecognizer.sequence(DOT).createNode(EShallowEntityType.METHOD, SubTypeNames.MODULE_INPUT, 0)
319                                .parseUntil(STATEMENTS).sequence(ENDMODULE, DOT).endNode();
320        }

322        /** Rules for parsing statements. */
323        private void createStatementRules() {
324
325                // special rule that matches assignments to variables that have the same
326                // name as keywords.
327                inState(STATEMENTS).sequence(ETokenClass.KEYWORD, EQ).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
328                                .endNode();
329
330                createConditionalStatementRules();
331
332                // on change
333                RecognizerBase<EAbapParserStates> changeAlternative = inAnyState().sequence(ON, CHANGE, OF)
334                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.ON_CHANGE).skipTo(DOT).parseUntil(STATEMENTS)
335                                .sequenceBefore(EnumSet.of(ELSE, ENDON));
336                changeAlternative.sequence(ENDON, DOT).endNode();
337                changeAlternative.endNodeWithContinuation();
338
339                createLoopStatementRules();
340
341                // try/catch
342                RecognizerBase<EAbapParserStates> tryAlternative = inState(TOPLEVEL, STATEMENTS)
343                                .sequence(EnumSet.of(TRY, CATCH, CLEANUP)).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
344                                .parseUntil(STATEMENTS).sequenceBefore(EnumSet.of(ENDTRY, CATCH, ENDCATCH, CLEANUP));
345                tryAlternative.sequence(EnumSet.of(ENDTRY, ENDCATCH), DOT).endNode();
346                tryAlternative.endNodeWithContinuation();
347
348                createSelectRules();
349
350                // exec
351                inState(TOPLEVEL, STATEMENTS).sequence(EXEC, SQL)
352                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.NATIVE_SQL).skipTo(ENDEXEC, DOT).endNode();
353
354                // simple statements that start with a field symbol, e.g.
355                // "<fs>-foo = 12."
356                inState(TOPLEVEL, STATEMENTS).sequence(LT, SIMPLE_STATEMENT_START_TOKENS)
357                                .createNode(EShallowEntityType.STATEMENT, new Region(0, 2)).skipTo(DOT).endNode();
358
359                // Match record type declarations as a single entity, although they
360                // contain multiple dots.
361                inState(TOPLEVEL, STATEMENTS).sequence(EnumSet.of(TYPES, CONSTANTS, DATA, STATICS)).optional(COLON)
362                                .sequence(BEGIN, OF, IDENTIFIER_LIKE).createNode(EShallowEntityType.STATEMENT, 0, -1).skipTo(END, OF)
363                                .skipTo(DOT).endNode();
364
365                inState(TOPLEVEL, STATEMENTS).sequence(SIMPLE_STATEMENT_START_TOKENS)
366                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT).endNode();
367        }

369        /** Creates parsing rules for conditional statements. */
370        private void createConditionalStatementRules() {
371                // if/elseif
372                RecognizerBase<EAbapParserStates> ifAlternative = inState(TOPLEVEL, STATEMENTS).sequence(EnumSet.of(IF, ELSEIF))
373                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT).parseUntil(STATEMENTS)
374                                .sequenceBefore(EnumSet.of(ELSEIF, ELSE, ENDIF));
375                ifAlternative.sequence(ENDIF, DOT).endNode();
376                ifAlternative.endNodeWithContinuation();
377
378                // else
379                inState(TOPLEVEL, STATEMENTS).sequence(ELSE).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
380                                .parseUntil(STATEMENTS).sequence(EnumSet.of(ENDIF, ENDON), DOT).endNode();
381
382                // case/when
383                inState(TOPLEVEL, STATEMENTS).sequence(CASE).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
384                                .parseUntil(STATEMENTS).sequence(ENDCASE, DOT).endNode();
385                // we parse when as meta, so we add no additional nesting
386                inState(STATEMENTS).sequence(WHEN).createNode(EShallowEntityType.META, 0).skipTo(DOT).endNode();
387        }
388
389        /** Creates parsing rules for loops and loop-like constructs. */
390        private void createLoopStatementRules() {
391                // loops
392                inState(TOPLEVEL, STATEMENTS).sequence(LOOP).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
393                                .parseUntil(STATEMENTS).sequence(ENDLOOP, DOT).endNode();
394                inState(TOPLEVEL, STATEMENTS).sequence(DO).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
395                                .parseUntil(STATEMENTS).sequence(ENDDO, DOT).endNode();
396                inState(TOPLEVEL, STATEMENTS).sequence(WHILE).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
397                                .parseUntil(STATEMENTS).sequence(ENDWHILE, DOT).endNode();
398                inState(STATEMENTS).sequence(AT).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT).parseUntil(STATEMENTS)
399                                .sequence(ENDAT, DOT).endNode();
400
401                // loop likes
402                inAnyState().sequence(PROVIDE).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT).parseUntil(STATEMENTS)
403                                .sequence(ENDPROVIDE, DOT).endNode();
404                inAnyState().sequence(ENHANCEMENT).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
405                                .parseUntil(STATEMENTS).sequence(ENDENHANCEMENT, DOT).endNode();
406                inAnyState().sequence(ENHANCEMENT_SECTION).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
407                                .parseUntil(STATEMENTS).sequence(END_ENHANCEMENT_SECTION, DOT).endNode();
408        }

410        /**
411         * Creates the parsing rules for the select clause. This is tricky, because
412         * the rules whether a select block or a single statement select is
413         * expected, are not trivial.
414         */
415        private void createSelectRules() {
416                RecognizerBase<EAbapParserStates> selectAlternative = inState(TOPLEVEL, STATEMENTS).sequence(SELECT);
417                selectAlternative.sequence(LPAREN).createNode(EShallowEntityType.STATEMENT, "method call")
418                                .skipToWithNesting(RPAREN, LPAREN, RPAREN).skipTo(DOT).endNode();
419                selectAlternative.subRecognizer(new SingleSelectRecognizer(), 1, 1)
420                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.SINGLE_SELECT).endNode();
421                selectAlternative.createNode(EShallowEntityType.STATEMENT, SubTypeNames.SELECT_BLOCK).skipTo(DOT)
422                                .parseUntil(STATEMENTS).sequence(ENDSELECT, DOT).endNode();
423        }

425        /** {@inheritDoc} */
426        @Override
427        protected boolean isFilteredToken(IToken token, IToken previousToken) {
428                return super.isFilteredToken(token, previousToken) || token.getType() == PRAGMA_DIRECTIVE;
429        }
