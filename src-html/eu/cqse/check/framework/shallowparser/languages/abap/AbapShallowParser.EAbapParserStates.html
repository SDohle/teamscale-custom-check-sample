039/**
040 * Shallow parser for ABAP. The following links are useful for writing the
041 * parser:
042 * <ul>
043 * <li><a href="http://help.sap.com/abapdocu_702/en/">ABAP Keyword
044 * Documentation</a></li>
045 * <li><a href=
046 * "http://help.sap.com/abapdocu_702/en/abenabap_statements_overview.htm" >ABAP
047 * Statements Overview</a></li>
048 * </ul>
049 */
050public class AbapShallowParser extends ShallowParserBase<AbapShallowParser.EAbapParserStates> {
051
052        /**
053         * Since there are no reserved words in ABAP, one can use statement names as
054         * identifiers. Therefore, whenever we expect an identifier, it's not safe to
055         * rely on the Scanner's categorization of identifiers, and we should allow
056         * keywords and operators as well. Example for operator: {@link ETokenType#MOD}.
057         */
058        private static final EnumSet<ETokenClass> IDENTIFIER_LIKE = EnumSet.of(ETokenClass.IDENTIFIER, ETokenClass.KEYWORD,
059                        ETokenClass.OPERATOR);
060
061        /** Tokens that can introduce a simple statement. */
062        private static final EnumSet<ETokenType> SIMPLE_STATEMENT_START_TOKENS = EnumSet.of(ADD, ADD_CORRESPONDING, APPEND,
063                        ASSERT, ASSIGN, AUTHORITY_CHECK, BACK, BREAK_POINT, CALL, CASE, CHECK, CLEAR, CLOSE, COLLECT, COMMIT,
064                        COMMUNICATION, COMPUTE, CONCATENATE, CONDENSE, CONSTANTS, CONTEXTS, CONTINUE, CONTROLS, CONVERT, CREATE,
065                        DATA, DELETE, DEMAND, DESCRIBE, DETAIL, DIVIDE, DIVIDE_CORRESPONDING, DO, EDITOR_CALL, ENHANCEMENT_POINT,
066                        EXISTS, EXIT, EXPORT, EXTRACT, FETCH, FIELDS, FIELD_GROUPS, FIELD_SYMBOLS, FIND, FORMAT, FREE, GENERATE,
067                        GET, HIDE, IDENTIFIER, IF, IMPORT, INCLUDE, INFOTYPES, INPUT, INSERT, LEAVE, LOAD, LOCAL, LOG_POINT,
068                        MAXIMUM, MESSAGE, MINIMUM, MODIFY, MODULE, MOVE, MOVE_CORRESPONDING, MULTIPLY_CORRESPONDING, MULTIPLY, NAME,
069                        NEW_LINE, NEW_PAGE, NEW_SECTION, OPEN, OVERLAY, PACK, PACKAGE, PARAMETERS, PERFORM, POSITION, PRINT_CONTROL,
070                        PUT, RAISE, RANGES, READ, REFRESH, REJECT, REPLACE, RESERVE, RESUME, RETURN, ROLLBACK, SCROLL, SEARCH, SET,
071                        SHIFT, SKIP, SORT, SPLIT, STATICS, STOP, SUBMIT, SUBTRACT, SUBTRACT_CORRESPONDING, SUM, SUMMARY, SUMMING,
072                        SUPPLY, SUPPRESS, SYNTAX_CHECK, TRANSFER, TRANSLATE, TRUNCATE, TYPES, ULINE, UNASSIGN, UNPACK, UPDATE,
073                        VERSION, WAIT, WINDOW, WRITE, NEW);

075        /**
076         * Tokens that identify when to filter out an INCLUDE statement. We only want to
077         * find statements of type <code>INCLUDE foobar</code> where foobar is another
078         * ABAP source object, but not <code>INCLUDE TYPE foobar</code> or
079         * <code>INCLUDE STRUCTURE foobar</code>. Since one may name their included
080         * source object like an ABAP keyword, we need to have a blacklist approach
081         * here.
082         */
083        public static final Set<ETokenType> INCLUDE_FILTER_TOKEN_TYPES = EnumSet.of(ETokenType.STRUCTURE, ETokenType.TYPE);
084
085        /**
086         * Set of keywords that start an event block (without keywords that require a
087         * preceding "at")
088         */
089        private static final EnumSet<ETokenType> EVENT_BLOCKS = EnumSet.of(INITIALIZATION, START_OF_SELECTION,
090                        END_OF_SELECTION, TOP_OF_PAGE, END_OF_PAGE, LOAD_OF_PROGRAM);
091
092        /**
093         * Set of keywords that end an event block (possibly indicating the start of the
094         * next one)
095         */
096        private static final EnumSet<ETokenType> EVENT_BLOCKS_END = EnumSet.of(AT, FORM, CLASS, INTERFACE);
097
098        // all tokens that start a new event block also end the current one.
099        static {
100                EVENT_BLOCKS_END.addAll(EVENT_BLOCKS);
101        }
102
103        /** The states used in this parser. */
104        public static enum EAbapParserStates {
105
106                /**
107                 * Top level state used for parsing constructs that are not nested in other
108                 * constructs.
109                 */
110                TOPLEVEL,
111
112                /**
113                 * Type state used for parsing constructs that are direct children of a REPORT
114                 * or PROGRAM.
115                 */
116                IN_TYPE,
117
118                /**
119                 * A state to recognize declarations within classes. As many constructs are
120                 * allowed both top-level and in declarations, many rules are registered for
121                 * both.
122                 */
123                DECLARATIONS,
124
125                /**
126                 * A state to recognize statements, i.e. plain code in functions, etc.
127                 */
128                STATEMENTS
129        }
130
131        /** Constructor. */
132        public AbapShallowParser() {
133                super(EAbapParserStates.class, TOPLEVEL);
134
135                createMetaRules();
136                createTopLevelRules();
137                createTypeRules();
138                createAttributeRules();
139                createMethodRules();
140                createModuleRules();
141                createStatementRules();
142
143                inAnyState().sequence(DOT).createNode(EShallowEntityType.STATEMENT, SubTypeNames.EMPTY_STATEMENT).endNode();
144        }

146        /** Rules for parsing elements that are only expected top-level. */
147        private void createTopLevelRules() {
148
149                // since the report is not really a method, its statements are still
150                // parsed in the toplevel scope, not the statement scope
151                inState(TOPLEVEL).sequence(EnumSet.of(REPORT, PROGRAM)).optional(COLON).sequence(IDENTIFIER_LIKE)
152                                .createNode(EShallowEntityType.TYPE, 0, -1).skipTo(DOT).parseUntilOrEof(IN_TYPE);
153
154                inState(TOPLEVEL, STATEMENTS, IN_TYPE).sequence(EnumSet.of(SELECTION_SCREEN))
155                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT).endNode();
156
157                // top-of-page during line-selection is different from "top-of-page"
158                RecognizerBase<EAbapParserStates> topsRecognizer = inState(TOPLEVEL, IN_TYPE)
159                                .sequence(TOP_OF_PAGE, DURING, LINE_SELECTION)
160                                .createNode(EShallowEntityType.METHOD, SubTypeNames.TOP_OF_PAGE_DURING_LINE_SELECTION).skipTo(DOT)
161                                .parseUntilOrEof(STATEMENTS);
162                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, topsRecognizer);
163
164                // basic event blocks without "at"
165                RecognizerBase<EAbapParserStates> eventBlockRecognizer = inState(TOPLEVEL, IN_TYPE).sequence(EVENT_BLOCKS)
166                                .createNode(EShallowEntityType.METHOD, 0, -1).skipTo(DOT).parseUntilOrEof(STATEMENTS);
167                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, eventBlockRecognizer);
168
169                createGetBlockRules();
170
171                createAtBlockRules();
172        }
173
174        /**
175         * Creates rules constructs starting with "GET". These can either be statements
176         * or event handlers.
177         */
178        private void createGetBlockRules() {
179                // "get reference of" and consorts
180                inState(TOPLEVEL, STATEMENTS, IN_TYPE).sequence(GET, ETokenClass.KEYWORD)
181                                .createNode(EShallowEntityType.STATEMENT, new int[] { 0, 1 }).skipTo(DOT).endNode();
182
183                // get ... late event handler
184                RecognizerBase<EAbapParserStates> getLateEventBlockRecognizer = inState(TOPLEVEL, IN_TYPE)
185                                .sequence(GET, IDENTIFIER, LATE).createNode(EShallowEntityType.METHOD, SubTypeNames.GET_LATE, 1)
186                                .skipTo(DOT).parseUntilOrEof(STATEMENTS);
187                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, getLateEventBlockRecognizer);
188
189                // get ... event handler
190                RecognizerBase<EAbapParserStates> getEventBlockRecognizer = inState(TOPLEVEL, IN_TYPE).sequence(GET, IDENTIFIER)
191                                .createNode(EShallowEntityType.METHOD, 0, 1).skipTo(DOT).parseUntilOrEof(STATEMENTS);
192                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, getEventBlockRecognizer);
193        }
194
195        /** Creates parsing rules for event handlers starting with "AT". */
196        private void createAtBlockRules() {
197                // at selection screen event handler (named)
198                RecognizerBase<EAbapParserStates> selectionScreenBlockRecognizer = inState(TOPLEVEL, IN_TYPE)
199                                .sequence(AT, SELECTION_SCREEN).skipTo(DOT).createNode(EShallowEntityType.METHOD,
200                                                SubTypeNames.AT_SELECTION_SCREEN, new Region(0, -2, StringUtils.SPACE))
201                                .parseUntilOrEof(STATEMENTS);
202                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, selectionScreenBlockRecognizer);
203
204                // at PF<..> event handler (named)
205                RecognizerBase<EAbapParserStates> atPfBlockRecognizer = inState(TOPLEVEL, IN_TYPE).sequence(AT, PF_EVENT)
206                                .createNode(EShallowEntityType.METHOD, SubTypeNames.PF_EVENT, new Region(1, 1)).skipTo(DOT)
207                                .parseUntilOrEof(STATEMENTS);
208                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, atPfBlockRecognizer);
209
210                // at ... event handler (anonymous)
211                RecognizerBase<EAbapParserStates> atEventBlockRecognizer = inState(TOPLEVEL, IN_TYPE)
212                                .sequence(AT, EnumSet.of(LINE_SELECTION, USER_COMMAND))
213                                .createNode(EShallowEntityType.METHOD, new int[] { 0, 1 }).skipTo(DOT).parseUntilOrEof(STATEMENTS);
214                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, atEventBlockRecognizer);
215        }
216
217        /**
218         * Ends the given recognizer if it either hits one of the given event block end
219         * tokens or a get event block or a module block.
220         */
221        private static void endMethodEntityOnEventBlock(EnumSet<ETokenType> eventBlocksEnd,
222                        RecognizerBase<EAbapParserStates> eventBlockRecognizer) {
223                eventBlockRecognizer.sequenceBefore(eventBlocksEnd).endNode();
224                eventBlockRecognizer.sequenceBefore(GET, IDENTIFIER).endNode();
225                eventBlockRecognizer.preCondition(new ModuleBlockStartRecognizer()).endNode();
226        }

228        /** Rules for parsing of meta elements. */
229        private void createMetaRules() {
230
231                inState(DECLARATIONS).sequence(EnumSet.of(PUBLIC, PROTECTED, PRIVATE), SECTION, DOT)
232                                .createNode(EShallowEntityType.META, SubTypeNames.VISIBILITY, 0).endNode();
233
234                // TypePools/Tables with colon
235                inAnyState().sequence(EnumSet.of(TYPE_POOLS, TABLES)).sequence(COLON).createNode(EShallowEntityType.META, 0, 2)
236                                .skipTo(DOT).endNode();
237                // TypePools/Tables without colon (need two rules to get the the name
238                // right)
239                inAnyState().sequence(EnumSet.of(TYPE_POOLS, TABLES)).createNode(EShallowEntityType.META, 0, 1).skipTo(DOT)
240                                .endNode();
241
242                inAnyState().sequence(DEFINE).createNode(EShallowEntityType.META, SubTypeNames.MACRO)
243                                .skipTo(END_OF_DEFINITION, DOT).endNode();
244
245                inState(DECLARATIONS).sequence(EnumSet.of(INTERFACES, ALIASES)).createNode(EShallowEntityType.META, 0)
246                                .skipTo(DOT).endNode();
247
248                inState(IN_TYPE).sequence(INCLUDE).notPreCondition(emptyRecognizer().sequence(INCLUDE_FILTER_TOKEN_TYPES))
249                                .createNode(EShallowEntityType.META, 0).skipTo(DOT).endNode();
250        }

252        /** Rules for parsing classes, interfaces, types, and events. */
253        private void createTypeRules() {
254
255                createClassRules();
256
257                // interfaces
258                RecognizerBase<EAbapParserStates> interfaceAlternative = inState(TOPLEVEL, DECLARATIONS).sequence(INTERFACE,
259                                IDENTIFIER);
260                interfaceAlternative.sequence(EnumSet.of(LOAD, DEFERRED, LOCAL))
261                                .createNode(EShallowEntityType.TYPE, SubTypeNames.INTERFACE_PUBLICATION, 1).skipTo(DOT).endNode();
262                interfaceAlternative.createNode(EShallowEntityType.TYPE, SubTypeNames.INTERFACE_DEFINITION, 1).skipTo(DOT)
263                                .parseUntil(DECLARATIONS).sequence(ENDINTERFACE, DOT).endNode();
264
265        }
266
267        /** Rules for parsing ABAP OO classes. */
268        private void createClassRules() {
269                RecognizerBase<EAbapParserStates> classDefinitionAlternative = inState(TOPLEVEL, DECLARATIONS, IN_TYPE)
270                                .sequence(CLASS, IDENTIFIER, DEFINITION);
271                classDefinitionAlternative.sequence(EnumSet.of(LOAD, DEFERRED, LOCAL))
272                                .createNode(EShallowEntityType.TYPE, SubTypeNames.CLASS_PUBLICATION, 1).skipTo(DOT).endNode();
273                classDefinitionAlternative.createNode(EShallowEntityType.TYPE, SubTypeNames.CLASS_DEFINITION, 1).skipTo(DOT)
274                                .parseUntil(DECLARATIONS).sequence(ENDCLASS, DOT).endNode();
275
276                inState(TOPLEVEL, DECLARATIONS, IN_TYPE).sequence(CLASS, IDENTIFIER, IMPLEMENTATION)
277                                .createNode(EShallowEntityType.TYPE, SubTypeNames.CLASS_IMPLEMENTATION, 1).skipTo(DOT)
278                                .parseUntil(DECLARATIONS).sequence(ENDCLASS, DOT).endNode();
279        }
280
281        /** Rules for parsing attributes. */
282        private void createAttributeRules() {
283                // Match BEGIN OF COMMON PART declarations as single entity
284                inState(TOPLEVEL, DECLARATIONS, IN_TYPE).sequence(DATA, BEGIN, OF, COMMON, PART).optional(IDENTIFIER_LIKE)
285                                .createNode(EShallowEntityType.ATTRIBUTE, 0, -1).skipTo(END, OF, COMMON, PART).skipTo(DOT).endNode();
286
287                // Match record type declaration as single entity
288                inState(TOPLEVEL, DECLARATIONS, IN_TYPE).sequence(EnumSet.of(TYPES, CONSTANTS, DATA, STATICS, CLASS_DATA))
289                                .optional(COLON).sequence(BEGIN, OF, IDENTIFIER_LIKE).createNode(EShallowEntityType.ATTRIBUTE, 0, -1)
290                                .skipTo(END, OF).skipTo(DOT).endNode();
291
292                // types, events, class events
293                inState(TOPLEVEL, DECLARATIONS, IN_TYPE).sequence(EnumSet.of(EVENTS, CLASS_EVENTS))
294                                .createNode(EShallowEntityType.ATTRIBUTE, 0).skipTo(DOT).endNode();
295
296                inState(TOPLEVEL, DECLARATIONS, IN_TYPE)
297                                .sequence(EnumSet.of(CONSTANTS, NODES, STATICS, DATA, TYPES, FIELD_GROUPS, CLASS_DATA, RANGES,
298                                                SELECT_OPTIONS, INFOTYPES))
299                                .optional(COLON).sequence(IDENTIFIER_LIKE).createNode(EShallowEntityType.ATTRIBUTE, 0, -1).skipTo(DOT)
300                                .endNode();
301
302                inState(TOPLEVEL, DECLARATIONS, IN_TYPE).sequence(FIELD_SYMBOLS).optional(COLON).sequence(LT, IDENTIFIER_LIKE)
303                                .createNode(EShallowEntityType.ATTRIBUTE, 0, -1).skipTo(DOT).endNode();
304
305                inState(TOPLEVEL, DECLARATIONS, IN_TYPE)
306                                // The "parameter" keyword is the obsolete older form of
307                                // "parameters"
308                                .sequence(EnumSet.of(PARAMETER, PARAMETERS)).optional(COLON).sequence(IDENTIFIER_LIKE)
309                                .createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.PARAMETERS, -1).skipTo(DOT).endNode();
310        }

312        /**
313         * Rules for parsing methods, functions, and forms.
314         *
315         * We handle colons here to catch method declarations like <code>form:
316         * foo.</foo> as that's a declaration for form "foo" not ":foo".
317         */
318        private void createMethodRules() {
319                inState(DECLARATIONS).sequence(EnumSet.of(METHODS, CLASS_METHODS))
320                                .skipBefore(
321                                                EnumSet.of(ABSTRACT, DOT, RETURNING, IMPORTING, EXPORTING, REDEFINITION, CHANGING, EXCEPTIONS))
322                                .createNode(EShallowEntityType.METHOD, SubTypeNames.METHOD_DECLARATION, new Region(1, -1)).skipTo(DOT)
323                                .endNode();
324
325                inState(DECLARATIONS).sequence(METHOD).optional(COLON).markStart().skipBefore(EnumSet.of(DOT, BY))
326                                .createNode(EShallowEntityType.METHOD, SubTypeNames.METHOD_IMPLEMENTATION, new Region(0, -1))
327                                .skipTo(DOT).parseUntil(STATEMENTS).sequence(ENDMETHOD, DOT).endNode();
328                inState(TOPLEVEL, STATEMENTS).sequence(METHOD).optional(COLON).markStart().skipBefore(EnumSet.of(DOT, BY))
329                                .createNode(EShallowEntityType.METHOD, SubTypeNames.METHOD_IMPLEMENTATION, new Region(0, -1))
330                                .skipTo(DOT).parseUntil(STATEMENTS).sequence(ENDMETHOD, DOT).endNode();
331                inState(TOPLEVEL, IN_TYPE).sequence(FORM).optional(COLON).sequence(IDENTIFIER_LIKE)
332                                .createNode(EShallowEntityType.METHOD, SubTypeNames.FORM, -1).skipTo(DOT).parseUntil(STATEMENTS)
333                                .sequence(ENDFORM, DOT).endNode();
334
335                inState(TOPLEVEL, DECLARATIONS).sequence(FUNCTION).optional(COLON).markStart()
336                                .skipBefore(EnumSet.of(DOT, IMPORTING, EXPORTING, TABLES, CHANGING, RAISING, EXCEPTIONS))
337                                .createNode(EShallowEntityType.METHOD, SubTypeNames.FUNCTION, new Region(0, -1)).skipTo(DOT)
338                                .parseUntil(STATEMENTS).sequence(ENDFUNCTION, DOT).endNode();
339
340                RecognizerBase<EAbapParserStates> toplevelStartOfSelection = inState(IN_TYPE)
341                                .sequenceBefore(EnumSet.complementOf(EVENT_BLOCKS_END))
342                                .sequenceBefore(EnumSet.complementOf(EnumSet.of(MODULE)))
343                                .createNode(EShallowEntityType.METHOD, SubTypeNames.START_OF_SELECTION, SubTypeNames.START_OF_SELECTION)
344                                .parseUntilOrEof(STATEMENTS);
345                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, toplevelStartOfSelection);
346
347                RecognizerBase<EAbapParserStates> toplevelStartOfSelectionWithModule = inState(IN_TYPE).sequenceBefore(MODULE)
348                                .notPreCondition(new ModuleBlockStartRecognizer())
349                                .createNode(EShallowEntityType.METHOD, SubTypeNames.START_OF_SELECTION, SubTypeNames.START_OF_SELECTION)
350                                .parseUntilOrEof(STATEMENTS);
351                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, toplevelStartOfSelectionWithModule);
352
353        }

355        /** Rules for parsing modules. */
356        private void createModuleRules() {
357                RecognizerBase<EAbapParserStates> moduleRecognizer = inState(TOPLEVEL, IN_TYPE)
358                                .preCondition(new ModuleBlockStartRecognizer()).sequence(MODULE).markStart()
359                                .skipBefore(EnumSet.of(INPUT, OUTPUT, DOT));
360                moduleRecognizer.sequence(INPUT, DOT).createNode(EShallowEntityType.METHOD, SubTypeNames.MODULE_INPUT, 0)
361                                .parseUntil(STATEMENTS).sequence(ENDMODULE, DOT).endNode();
362                moduleRecognizer.sequence(OUTPUT, DOT).createNode(EShallowEntityType.METHOD, SubTypeNames.MODULE_OUTPUT, 0)
363                                .parseUntil(STATEMENTS).sequence(ENDMODULE, DOT).endNode();
364                moduleRecognizer.sequence(DOT).createNode(EShallowEntityType.METHOD, SubTypeNames.MODULE_INPUT, 0)
365                                .parseUntil(STATEMENTS).sequence(ENDMODULE, DOT).endNode();
366        }

368        /** Rules for parsing statements. */
369        private void createStatementRules() {
370
371                // special rule that matches assignments to variables that have the same
372                // name as keywords.
373                inState(STATEMENTS).sequence(ETokenClass.KEYWORD, EQ).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
374                                .endNode();
375
376                createConditionalStatementRules();
377
378                // on change
379                RecognizerBase<EAbapParserStates> changeAlternative = inAnyState().sequence(ON, CHANGE, OF)
380                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.ON_CHANGE).skipTo(DOT).parseUntil(STATEMENTS)
381                                .sequenceBefore(EnumSet.of(ELSE, ENDON));
382                changeAlternative.sequence(ENDON, DOT).endNode();
383                changeAlternative.endNodeWithContinuation();
384                inState(TOPLEVEL, STATEMENTS).sequence(IDENTIFIER_LIKE, ETokenType.LPAREN)
385                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT).endNode();
386
387                createLoopStatementRules();
388
389                // try/catch
390                RecognizerBase<EAbapParserStates> tryAlternative = inState(TOPLEVEL, STATEMENTS)
391                                .sequence(EnumSet.of(TRY, CATCH, CLEANUP)).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
392                                .parseUntil(STATEMENTS).sequenceBefore(EnumSet.of(ENDTRY, CATCH, ENDCATCH, CLEANUP));
393                tryAlternative.sequence(EnumSet.of(ENDTRY, ENDCATCH), DOT).endNode();
394                tryAlternative.endNodeWithContinuation();
395
396                createSelectRules();
397
398                // exec
399                inState(TOPLEVEL, STATEMENTS).sequence(EXEC, SQL)
400                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.NATIVE_SQL).skipTo(ENDEXEC, DOT).endNode();
401
402                // simple statements that start with a field symbol, e.g.
403                // "<fs>-foo = 12."
404                inState(TOPLEVEL, STATEMENTS).sequence(LT).createNode(EShallowEntityType.STATEMENT, new Region(0, 2))
405                                .skipTo(DOT).endNode();
406
407                // Match record type declarations as a single entity, although they
408                // contain multiple dots.
409                inState(TOPLEVEL, STATEMENTS).sequence(EnumSet.of(TYPES, CONSTANTS, DATA, STATICS)).optional(COLON)
410                                .sequence(BEGIN, OF, IDENTIFIER_LIKE).createNode(EShallowEntityType.STATEMENT, 0, -1).skipTo(END, OF)
411                                .skipTo(DOT).endNode();
412
413                inState(TOPLEVEL, STATEMENTS).sequence(SIMPLE_STATEMENT_START_TOKENS)
414                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT).endNode();
415        }
416
417        /** Creates parsing rules for conditional statements. */
418        private void createConditionalStatementRules() {
419                // if/elseif
420                RecognizerBase<EAbapParserStates> ifAlternative = inState(TOPLEVEL, STATEMENTS).sequence(EnumSet.of(IF, ELSEIF))
421                                .sequenceBefore(EnumSet.complementOf(EnumSet.of(ARROW))).createNode(EShallowEntityType.STATEMENT, 0)
422                                .skipTo(DOT).parseUntil(STATEMENTS).sequenceBefore(EnumSet.of(ELSEIF, ELSE, ENDIF));
423                ifAlternative.sequence(ENDIF, DOT).endNode();
424                ifAlternative.endNodeWithContinuation();
425
426                // else
427                inState(TOPLEVEL, STATEMENTS).sequence(ELSE).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
428                                .parseUntil(STATEMENTS).sequence(EnumSet.of(ENDIF, ENDON), DOT).endNode();
429
430                // case/when
431                inState(TOPLEVEL, STATEMENTS).sequence(CASE).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
432                                .parseUntil(STATEMENTS).sequence(ENDCASE, DOT).endNode();
433                // we parse when as meta, so we add no additional nesting
434                inState(STATEMENTS).sequence(WHEN).createNode(EShallowEntityType.META, 0).skipTo(DOT).endNode();
435        }

437        /** Creates parsing rules for loops and loop-like constructs. */
438        private void createLoopStatementRules() {
439                // loops
440                inState(TOPLEVEL, STATEMENTS).sequence(LOOP).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
441                                .parseUntil(STATEMENTS).sequence(ENDLOOP, DOT).endNode();
442                inState(TOPLEVEL, STATEMENTS).sequence(DO).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
443                                .parseUntil(STATEMENTS).sequence(ENDDO, DOT).endNode();
444                inState(TOPLEVEL, STATEMENTS).sequence(WHILE).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
445                                .parseUntil(STATEMENTS).sequence(ENDWHILE, DOT).endNode();
446                inState(STATEMENTS).sequence(AT).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT).parseUntil(STATEMENTS)
447                                .sequence(ENDAT, DOT).endNode();
448
449                // loop likes
450                inAnyState().sequence(PROVIDE).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT).parseUntil(STATEMENTS)
451                                .sequence(ENDPROVIDE, DOT).endNode();
452                inAnyState().sequence(ENHANCEMENT).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
453                                .parseUntil(STATEMENTS).sequence(ENDENHANCEMENT, DOT).endNode();
454                inAnyState().sequence(ENHANCEMENT_SECTION).createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
455                                .parseUntil(STATEMENTS).sequence(END_ENHANCEMENT_SECTION, DOT).endNode();
456        }

458        /**
459         * Creates the parsing rules for the select clause. This is tricky, because the
460         * rules whether a select block or a single statement select is expected, are
461         * not trivial.
462         */
463        private void createSelectRules() {
464                RecognizerBase<EAbapParserStates> selectAlternative = inState(TOPLEVEL, STATEMENTS).sequence(SELECT);
465                selectAlternative.sequence(LPAREN).createNode(EShallowEntityType.STATEMENT, "method call")
466                                .skipToWithNesting(RPAREN, LPAREN, RPAREN).skipTo(DOT).endNode();
467                selectAlternative.subRecognizer(new SingleSelectRecognizer(), 1, 1)
468                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.SINGLE_SELECT).endNode();
469                selectAlternative.createNode(EShallowEntityType.STATEMENT, SubTypeNames.SELECT_BLOCK).skipTo(DOT)
470                                .parseUntil(STATEMENTS).sequence(ENDSELECT, DOT).endNode();
471        }
