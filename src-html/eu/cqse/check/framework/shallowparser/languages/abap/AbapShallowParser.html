055        /**
056         * Since there are no reserved words in ABAP, one can use statement names as
057         * identifiers. Therefore, whenever we expect an identifier, it's not safe
058         * to rely on the Scanner's categorization of identifiers, and we should
059         * allow keywords as well.
060         */
061        private static final EnumSet<ETokenClass> IDENTIFIER_LIKE = EnumSet
062                        .of(ETokenClass.IDENTIFIER, ETokenClass.KEYWORD);
063

087        /**
088         * Set of keywords that start an event block (without keywords that require
089         * a preceding "at")
090         */
091        private static final EnumSet<ETokenType> EVENT_BLOCKS = EnumSet.of(
092                        INITIALIZATION, START_OF_SELECTION, END_OF_SELECTION, TOP_OF_PAGE,
093                        END_OF_PAGE, LOAD_OF_PROGRAM);
094
095        /**
096         * Set of keywords that end an event block (possibly indicating the start of
097         * the next one)
098         */
099        private static final EnumSet<ETokenType> EVENT_BLOCKS_END = EnumSet.of(AT,
100                        FORM, CLASS, INTERFACE);
101
102        // all tokens that start a new event block also end the current one.
103        static {
104                EVENT_BLOCKS_END.addAll(EVENT_BLOCKS);
105        }
106
107        /** The states used in this parser. */
108        public static enum EAbapParserStates {
109
110                /**
111                 * Top level state used for parsing constructs that are not nested in
112                 * other constructs.
113                 */
114                TOPLEVEL,
115
116                /**
117                 * A state to recognize declarations within classes. As many constructs
118                 * are allowed both top-level and in declarations, many rules are
119                 * registered for both.
120                 */
121                DECLARATIONS,
122
123                /**
124                 * A state to recognize statements, i.e. plain code in functions, etc.
125                 */
126                STATEMENTS
127        }
128
129        /** Constructor. */
130        public AbapShallowParser() {
131                super(EAbapParserStates.class, TOPLEVEL);
132
133                createMetaRules();
134                createTopLevelRules();
135                createTypeRules();
136                createAttributeRules();
137                createMethodRules();
138                createModuleRules();
139                createStatementRules();
140
141                inAnyState().sequence(DOT).createNode(EShallowEntityType.STATEMENT,
142                                SubTypeNames.EMPTY_STATEMENT).endNode();
143        }

145        /** Rules for parsing elements that are only expected top-level. */
146        private void createTopLevelRules() {
147
148                // since the report is not really a method, its statements are still
149                // parsed in the toplevel scope, not the statement scope
150                RecognizerBase<EAbapParserStates> reportRecognizer = inState(TOPLEVEL)
151                                .sequence(EnumSet.of(REPORT, PROGRAM)).optional(COLON)
152                                .sequence(IDENTIFIER_LIKE)
153                                .createNode(EShallowEntityType.METHOD, 0, -1).skipTo(DOT)
154                                .parseUntilOrEof(TOPLEVEL);
155                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, reportRecognizer);
156
157                inState(TOPLEVEL).sequence(EnumSet.of(SELECTION_SCREEN))
158                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
159                                .endNode();
160
161                // top-of-page during line-selection is different from "top-of-page"
162                RecognizerBase<EAbapParserStates> topsRecognizer = inState(TOPLEVEL)
163                                .sequence(TOP_OF_PAGE, DURING, LINE_SELECTION)
164                                .createNode(EShallowEntityType.METHOD,
165                                                SubTypeNames.TOP_OF_PAGE_DURING_LINE_SELECTION)
166                                .skipTo(DOT).parseUntilOrEof(STATEMENTS);
167                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, topsRecognizer);
168
169                // basic event blocks without "at"
170                RecognizerBase<EAbapParserStates> eventBlockRecognizer = inState(
171                                TOPLEVEL).sequence(EVENT_BLOCKS)
172                                                .createNode(EShallowEntityType.METHOD, 0).skipTo(DOT)
173                                                .parseUntilOrEof(STATEMENTS);
174                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, eventBlockRecognizer);
175
176                createGetBlockRules();
177
178                createAtBlockRules();
179        }
180
181        /**
182         * Creates rules constructs starting with "GET". These can either be
183         * statements or event handlers.
184         */
185        private void createGetBlockRules() {
186                // "get reference of" and consorts
187                inState(TOPLEVEL, STATEMENTS).sequence(GET, ETokenClass.KEYWORD)
188                                .createNode(EShallowEntityType.STATEMENT, new int[] { 0, 1 })
189                                .skipTo(DOT).endNode();
190
191                // get ... late event handler
192                RecognizerBase<EAbapParserStates> getLateEventBlockRecognizer = inState(
193                                TOPLEVEL)
194                                                .sequence(GET, IDENTIFIER, LATE)
195                                                .createNode(EShallowEntityType.METHOD,
196                                                                SubTypeNames.GET_LATE, 1)
197                                                .skipTo(DOT).parseUntilOrEof(STATEMENTS);
198                endMethodEntityOnEventBlock(EVENT_BLOCKS_END,
199                                getLateEventBlockRecognizer);
200
201                // get ... event handler
202                RecognizerBase<EAbapParserStates> getEventBlockRecognizer = inState(
203                                TOPLEVEL).sequence(GET, IDENTIFIER)
204                                                .createNode(EShallowEntityType.METHOD, 0, 1).skipTo(DOT)
205                                                .parseUntilOrEof(STATEMENTS);
206                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, getEventBlockRecognizer);
207        }
208
209        /** Creates parsing rules for event handlers starting with "AT". */
210        private void createAtBlockRules() {
211                // at selection screen event handler (named)
212                RecognizerBase<EAbapParserStates> selectionScreenBlockRecognizer = inState(
213                                TOPLEVEL).sequence(AT, SELECTION_SCREEN)
214                                                .skipTo(DOT)
215                                                .createNode(EShallowEntityType.METHOD,
216                                                                SubTypeNames.AT_SELECTION_SCREEN,
217                                                                new Region(0, -2, StringUtils.SPACE))
218                                                .parseUntilOrEof(STATEMENTS);
219                endMethodEntityOnEventBlock(EVENT_BLOCKS_END,
220                                selectionScreenBlockRecognizer);
221
222                // at PF<..> event handler (named)
223                RecognizerBase<EAbapParserStates> atPfBlockRecognizer = inState(
224                                TOPLEVEL)
225                                                .sequence(AT, PF_EVENT)
226                                                .createNode(EShallowEntityType.METHOD,
227                                                                SubTypeNames.PF_EVENT, new Region(1, 1))
228                                                .skipTo(DOT).parseUntilOrEof(STATEMENTS);
229                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, atPfBlockRecognizer);
230
231                // at ... event handler (anonymous)
232                RecognizerBase<EAbapParserStates> atEventBlockRecognizer = inState(
233                                TOPLEVEL)
234                                                .sequence(AT, EnumSet.of(LINE_SELECTION, USER_COMMAND))
235                                                .createNode(EShallowEntityType.METHOD,
236                                                                new int[] { 0, 1 })
237                                                .skipTo(DOT).parseUntilOrEof(STATEMENTS);
238                endMethodEntityOnEventBlock(EVENT_BLOCKS_END, atEventBlockRecognizer);
239        }

241        /**
242         * Ends the given recognizer if it either hits one of the given event block
243         * end tokens or a get event block or a module block.
244         */
245        private static void endMethodEntityOnEventBlock(
246                        EnumSet<ETokenType> eventBlocksEnd,
247                        RecognizerBase<EAbapParserStates> eventBlockRecognizer) {
248                eventBlockRecognizer.sequenceBefore(eventBlocksEnd).endNode();
249                eventBlockRecognizer.sequenceBefore(GET, IDENTIFIER).endNode();
250                eventBlockRecognizer.preCondition(new ModuleBlockStartRecognizer())
251                                .endNode();
252        }

254        /** Rules for parsing of meta elements. */
255        private void createMetaRules() {
256
257                inState(DECLARATIONS)
258                                .sequence(EnumSet.of(PUBLIC, PROTECTED, PRIVATE), SECTION, DOT)
259                                .createNode(EShallowEntityType.META, SubTypeNames.VISIBILITY, 0)
260                                .endNode();
261
262                inAnyState().sequence(EnumSet.of(TYPE_POOLS, TABLES))
263                                .createNode(EShallowEntityType.META, 0).skipTo(DOT).endNode();
264
265                inAnyState().sequence(DEFINE)
266                                .createNode(EShallowEntityType.META, SubTypeNames.MACRO)
267                                .skipTo(END_OF_DEFINITION, DOT).endNode();
268
269                inState(DECLARATIONS).sequence(EnumSet.of(INTERFACES, ALIASES))
270                                .createNode(EShallowEntityType.META, 0).skipTo(DOT).endNode();
271        }

273        /** Rules for parsing classes, interfaces, types, and events. */
274        private void createTypeRules() {
275
276                createClassRules();
277
278                // interfaces
279                RecognizerBase<EAbapParserStates> interfaceAlternative = inState(
280                                TOPLEVEL, DECLARATIONS).sequence(INTERFACE, IDENTIFIER);
281                interfaceAlternative.sequence(EnumSet.of(LOAD, DEFERRED, LOCAL))
282                                .createNode(EShallowEntityType.TYPE,
283                                                SubTypeNames.INTERFACE_PUBLICATION, 1)
284                                .skipTo(DOT).endNode();
285                interfaceAlternative
286                                .createNode(EShallowEntityType.TYPE,
287                                                SubTypeNames.INTERFACE_DEFINITION, 1)
288                                .skipTo(DOT).parseUntil(DECLARATIONS)
289                                .sequence(ENDINTERFACE, DOT).endNode();
290
291        }

293        /** Rules for parsing ABAP OO classes. */
294        private void createClassRules() {
295                RecognizerBase<EAbapParserStates> classDefinitionAlternative = inState(
296                                TOPLEVEL, DECLARATIONS).sequence(CLASS, IDENTIFIER, DEFINITION);
297                classDefinitionAlternative.sequence(EnumSet.of(LOAD, DEFERRED, LOCAL))
298                                .createNode(EShallowEntityType.TYPE,
299                                                SubTypeNames.CLASS_PUBLICATION, 1)
300                                .skipTo(DOT).endNode();
301                classDefinitionAlternative
302                                .createNode(EShallowEntityType.TYPE,
303                                                SubTypeNames.CLASS_DEFINITION, 1)
304                                .skipTo(DOT).parseUntil(DECLARATIONS).sequence(ENDCLASS, DOT)
305                                .endNode();
306
307                inState(TOPLEVEL, DECLARATIONS)
308                                .sequence(CLASS, IDENTIFIER, IMPLEMENTATION)
309                                .createNode(EShallowEntityType.TYPE,
310                                                SubTypeNames.CLASS_IMPLEMENTATION, 1)
311                                .skipTo(DOT).parseUntil(DECLARATIONS).sequence(ENDCLASS, DOT)
312                                .endNode();
313        }

315        /** Rules for parsing attributes. */
316        private void createAttributeRules() {
317                // Match record type declaration as single entity
318                inState(TOPLEVEL, DECLARATIONS)
319                                .sequence(
320                                                EnumSet.of(TYPES, CONSTANTS, DATA, STATICS, CLASS_DATA))
321                                .optional(COLON).sequence(BEGIN, OF, IDENTIFIER_LIKE)
322                                .createNode(EShallowEntityType.ATTRIBUTE, 0, -1).skipTo(END, OF)
323                                .skipTo(DOT).endNode();
324
325                // types, events, class events
326                inState(TOPLEVEL, DECLARATIONS)
327                                .sequence(EnumSet.of(EVENTS, CLASS_EVENTS))
328                                .createNode(EShallowEntityType.ATTRIBUTE, 0).skipTo(DOT)
329                                .endNode();
330
331                inState(TOPLEVEL, DECLARATIONS)
332                                .sequence(EnumSet.of(CONSTANTS, NODES, STATICS, DATA, TYPES,
333                                                FIELD_GROUPS, CLASS_DATA, RANGES, SELECT_OPTIONS))
334                                .optional(COLON).sequence(IDENTIFIER_LIKE)
335                                .createNode(EShallowEntityType.ATTRIBUTE, 0, -1).skipTo(DOT)
336                                .endNode();
337
338                inState(TOPLEVEL, DECLARATIONS).sequence(FIELD_SYMBOLS).optional(COLON)
339                                .sequence(LT, IDENTIFIER_LIKE)
340                                .createNode(EShallowEntityType.ATTRIBUTE, 0, -1).skipTo(DOT)
341                                .endNode();
342
343                inState(TOPLEVEL, DECLARATIONS)
344                                // The "parameter" keyword is the obsolete older form of
345                                // "parameters"
346                                .sequence(EnumSet.of(PARAMETER, PARAMETERS)).optional(COLON)
347                                .sequence(IDENTIFIER_LIKE)
348                                .createNode(EShallowEntityType.ATTRIBUTE,
349                                                SubTypeNames.PARAMETERS, -1)
350                                .skipTo(DOT).endNode();
351        }

353        /**
354         * Rules for parsing methods, functions, and forms.
355         * 
356         * We handle colons here to catch method declarations like <code>form:
357         * foo.</foo> as that's a declaration for form "foo" not ":foo".
358         */
359        private void createMethodRules() {
360                inState(DECLARATIONS).sequence(EnumSet.of(METHODS, CLASS_METHODS))
361                                .skipBefore(EnumSet.of(DOT, RETURNING, IMPORTING, EXPORTING,
362                                                REDEFINITION, CHANGING, EXCEPTIONS))
363                                .createNode(EShallowEntityType.METHOD,
364                                                SubTypeNames.METHOD_DECLARATION, new Region(1, -1))
365                                .skipTo(DOT).endNode();
366
367                inState(DECLARATIONS).sequence(METHOD).optional(COLON).markStart()
368                                .skipBefore(EnumSet.of(DOT, BY))
369                                .createNode(EShallowEntityType.METHOD,
370                                                SubTypeNames.METHOD_IMPLEMENTATION, new Region(0, -1))
371                                .skipTo(DOT).parseUntil(STATEMENTS).sequence(ENDMETHOD, DOT)
372                                .endNode();
373
374                inState(TOPLEVEL, DECLARATIONS).sequence(FUNCTION).optional(COLON)
375                                .markStart().skipTo(DOT)
376                                .createNode(EShallowEntityType.METHOD, SubTypeNames.FUNCTION,
377                                                new Region(0, -2))
378                                .parseUntil(STATEMENTS).sequence(ENDFUNCTION, DOT).endNode();
379
380                inState(TOPLEVEL).sequence(FORM).optional(COLON)
381                                .sequence(IDENTIFIER_LIKE)
382                                .createNode(EShallowEntityType.METHOD, SubTypeNames.FORM, -1)
383                                .skipTo(DOT).parseUntil(STATEMENTS).sequence(ENDFORM, DOT)
384                                .endNode();
385        }

387        /** Rules for parsing modules. */
388        private void createModuleRules() {
389                RecognizerBase<EAbapParserStates> moduleRecognizer = inState(TOPLEVEL)
390                                .preCondition(new ModuleBlockStartRecognizer()).sequence(MODULE)
391                                .markStart().skipBefore(EnumSet.of(INPUT, OUTPUT, DOT));
392                moduleRecognizer.sequence(INPUT, DOT)
393                                .createNode(EShallowEntityType.METHOD,
394                                                SubTypeNames.MODULE_INPUT, 0)
395                                .parseUntil(STATEMENTS).sequence(ENDMODULE, DOT).endNode();
396                moduleRecognizer.sequence(OUTPUT, DOT)
397                                .createNode(EShallowEntityType.METHOD,
398                                                SubTypeNames.MODULE_OUTPUT, 0)
399                                .parseUntil(STATEMENTS).sequence(ENDMODULE, DOT).endNode();
400                moduleRecognizer.sequence(DOT)
401                                .createNode(EShallowEntityType.METHOD,
402                                                SubTypeNames.MODULE_INPUT, 0)
403                                .parseUntil(STATEMENTS).sequence(ENDMODULE, DOT).endNode();
404        }

406        /** Rules for parsing statements. */
407        private void createStatementRules() {
408
409                // special rule that matches assignments to variables that have the same
410                // name as keywords.
411                inState(STATEMENTS).sequence(ETokenClass.KEYWORD, EQ)
412                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
413                                .endNode();
414
415                createConditionalStatementRules();
416
417                // on change
418                RecognizerBase<EAbapParserStates> changeAlternative = inAnyState()
419                                .sequence(ON, CHANGE, OF)
420                                .createNode(EShallowEntityType.STATEMENT,
421                                                SubTypeNames.ON_CHANGE)
422                                .skipTo(DOT).parseUntil(STATEMENTS)
423                                .sequenceBefore(EnumSet.of(ELSE, ENDON));
424                changeAlternative.sequence(ENDON, DOT).endNode();
425                changeAlternative.endNodeWithContinuation();
426
427                createLoopStatementRules();
428
429                // try/catch
430                RecognizerBase<EAbapParserStates> tryAlternative = inState(TOPLEVEL,
431                                STATEMENTS).sequence(EnumSet.of(TRY, CATCH, CLEANUP))
432                                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
433                                                .parseUntil(STATEMENTS).sequenceBefore(
434                                                                EnumSet.of(ENDTRY, CATCH, ENDCATCH, CLEANUP));
435                tryAlternative.sequence(EnumSet.of(ENDTRY, ENDCATCH), DOT).endNode();
436                tryAlternative.endNodeWithContinuation();
437
438                createSelectRules();
439
440                // exec
441                inState(TOPLEVEL, STATEMENTS).sequence(EXEC, SQL)
442                                .createNode(EShallowEntityType.STATEMENT,
443                                                SubTypeNames.NATIVE_SQL)
444                                .skipTo(ENDEXEC, DOT).endNode();
445
446                // simple statements that start with a field symbol, e.g.
447                // "<fs>-foo = 12."
448                inState(TOPLEVEL, STATEMENTS)
449                                .sequence(LT, SIMPLE_STATEMENT_START_TOKENS)
450                                .createNode(EShallowEntityType.STATEMENT, new Region(0, 2))
451                                .skipTo(DOT).endNode();
452
453                // Match record type declarations as a single entity, although they
454                // contain multiple dots.
455                inState(TOPLEVEL, STATEMENTS)
456                                .sequence(EnumSet.of(TYPES, CONSTANTS, DATA, STATICS))
457                                .optional(COLON).sequence(BEGIN, OF, IDENTIFIER_LIKE)
458                                .createNode(EShallowEntityType.STATEMENT, 0, -1).skipTo(END, OF)
459                                .skipTo(DOT).endNode();
460
461                inState(TOPLEVEL, STATEMENTS).sequence(SIMPLE_STATEMENT_START_TOKENS)
462                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
463                                .endNode();
464        }

466        /** Creates parsing rules for conditional statements. */
467        private void createConditionalStatementRules() {
468                // if/elseif
469                RecognizerBase<EAbapParserStates> ifAlternative = inState(TOPLEVEL,
470                                STATEMENTS).sequence(EnumSet.of(IF, ELSEIF))
471                                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
472                                                .parseUntil(STATEMENTS)
473                                                .sequenceBefore(EnumSet.of(ELSEIF, ELSE, ENDIF));
474                ifAlternative.sequence(ENDIF, DOT).endNode();
475                ifAlternative.endNodeWithContinuation();
476
477                // else
478                inState(TOPLEVEL, STATEMENTS).sequence(ELSE)
479                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
480                                .parseUntil(STATEMENTS).sequence(EnumSet.of(ENDIF, ENDON), DOT)
481                                .endNode();
482
483                // case/when
484                inState(TOPLEVEL, STATEMENTS).sequence(CASE)
485                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
486                                .parseUntil(STATEMENTS).sequence(ENDCASE, DOT).endNode();
487                // we parse when as meta, so we add no additional nesting
488                inState(STATEMENTS).sequence(WHEN)
489                                .createNode(EShallowEntityType.META, 0).skipTo(DOT).endNode();
490        }

492        /** Creates parsing rules for loops and loop-like constructs. */
493        private void createLoopStatementRules() {
494                // loops
495                inState(TOPLEVEL, STATEMENTS).sequence(LOOP)
496                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
497                                .parseUntil(STATEMENTS).sequence(ENDLOOP, DOT).endNode();
498                inState(TOPLEVEL, STATEMENTS).sequence(DO)
499                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
500                                .parseUntil(STATEMENTS).sequence(ENDDO, DOT).endNode();
501                inState(TOPLEVEL, STATEMENTS).sequence(WHILE)
502                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
503                                .parseUntil(STATEMENTS).sequence(ENDWHILE, DOT).endNode();
504                inState(STATEMENTS).sequence(AT)
505                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
506                                .parseUntil(STATEMENTS).sequence(ENDAT, DOT).endNode();
507
508                // loop likes
509                inAnyState().sequence(PROVIDE)
510                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
511                                .parseUntil(STATEMENTS).sequence(ENDPROVIDE, DOT).endNode();
512                inAnyState().sequence(ENHANCEMENT)
513                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
514                                .parseUntil(STATEMENTS).sequence(ENDENHANCEMENT, DOT).endNode();
515                inAnyState().sequence(ENHANCEMENT_SECTION)
516                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DOT)
517                                .parseUntil(STATEMENTS).sequence(END_ENHANCEMENT_SECTION, DOT)
518                                .endNode();
519        }

521        /**
522         * Creates the parsing rules for the select clause. This is tricky, because
523         * the rules whether a select block or a single statement select is
524         * expected, are not trivial.
525         */
526        private void createSelectRules() {
527                RecognizerBase<EAbapParserStates> selectAlternative = inState(TOPLEVEL,
528                                STATEMENTS).sequence(SELECT);
529                selectAlternative.sequence(LPAREN)
530                                .createNode(EShallowEntityType.STATEMENT, "method call")
531                                .skipToWithNesting(RPAREN, LPAREN, RPAREN).skipTo(DOT)
532                                .endNode();
533                selectAlternative.subRecognizer(new SingleSelectRecognizer(), 1, 1)
534                                .createNode(EShallowEntityType.STATEMENT,
535                                                SubTypeNames.SINGLE_SELECT)
536                                .endNode();
537                selectAlternative
538                                .createNode(EShallowEntityType.STATEMENT,
539                                                SubTypeNames.SELECT_BLOCK)
540                                .skipTo(DOT).parseUntil(STATEMENTS).sequence(ENDSELECT, DOT)
541                                .endNode();
542        }

544        /** {@inheritDoc} */
545        @Override
546        protected boolean isFilteredToken(IToken token, IToken previousToken) {
547                return super.isFilteredToken(token, previousToken)
548                                || token.getType() == PRAGMA_DIRECTIVE;
549        }
