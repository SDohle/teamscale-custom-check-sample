085/**
086 * Shallow parser for Ada.
087 * <p>
088 * A good introduction to Ada can be found here:
089 * http://en.wikibooks.org/wiki/Ada_Programming
090 * <p>
091 * A reference is available here:
092 * http://www.adaic.org/resources/add_content/standards/05rm/html/RM-TOC.html
093 * <p>
094 * What this parser does and does not:
095 * <ul>
096 * <li>The "with" and "use" statements as well as pragmas are parsed as meta
097 * data.</li>
098 * <li>Packages, types, functions, procedures, entries are parsed as expected.</li>
099 * <li>It recognizes the nesting of statements (e.g. in loops), but does not
100 * parse into the statements. For example, it recognizes an if-statement and
101 * provides the list of sub-statements, but does not provide direct access to
102 * the if-condition.</li>
103 * <li>The parser does not support non-ASCII characters in identifiers, although
104 * Ada95 supports unicode here. Actually, this is not an issue of the parser but
105 * the underlying scanner.</li>
106 * </ul>
107 * <p>
108 * Implementation hints:
109 * <ul>
110 * <li>Several rules are registered for any state although they would be
111 * expected to occur only in state DECLARATIONS. This is to allow the parser to
112 * recover from state STATEMENTS, if an arbitrary chunk of Ada code is to be
113 * parsed.</li>
114 * </ul>
115 * 
116 * @author $Author: hummelb $
117 * @version $Rev: 52657 $
118 * @ConQAT.Rating GREEN Hash: F3CF98F7EC4E3226092E2ED0806831BB
119 */
120public class AdaShallowParser extends
121                ShallowParserBase<AdaShallowParser.EAdaParserStates> {
122
123        /** The states used in this parser. */
124        public static enum EAdaParserStates {
125
126                /** A state to recognize declarations. */
127                DECLARATIONS,
128
129                /** A state to recognize statements. */
130                STATEMENTS
131        }
132
133        /** Constructor. */
134        public AdaShallowParser() {
135                super(EAdaParserStates.class, DECLARATIONS);
136
137                createMetaRules();
138                createSpecificationRules();
139                createBodyAndTypeRules();
140                createMethodAndAttributeRules();
141                createTypeRules();
142                createStatementRules();
143        }
144
145        /** Create rules for parsing meta elements. */
146        private void createMetaRules() {
147                // parse generic prefix as meta
148                // we skip from "with" to ";" to not stumble over subprogram parameters
149                inAnyState()
150                                .sequence(GENERIC)
151                                .createNode(EShallowEntityType.META, 0)
152                                .skipBeforeWithNesting(
153                                                EnumSet.of(PACKAGE, PROCEDURE, FUNCTION, ENTRY), WITH,
154                                                SEMICOLON).endNodeWithContinuation();
155
156                // parse use and with as meta
157                inAnyState().sequence(EnumSet.of(WITH, USE, PRAGMA))
158                                .createNode(EShallowEntityType.META, 0).skipTo(SEMICOLON)
159                                .endNode();
160
161                // parse pragma as meta
162                inAnyState().sequence(PREPROCESSOR_DIRECTIVE)
163                                .createNode(EShallowEntityType.META, "pragma").endNode();
164
165                // deal with dangling end by inserting broken node
166                inAnyState().sequence(END)
167                                .createNode(EShallowEntityType.META, "dangling end")
168                                .skipTo(SEMICOLON); // endNode() omitted!
169        }
170
171        /** Creates parsing rules for package and task specifications. */
172        private void createSpecificationRules() {
173                // package specification
174                RecognizerBase<EAdaParserStates> packageSpecAlternative = inAnyState()
175                                .sequence(PACKAGE, IDENTIFIER).skipBefore(
176                                                EnumSet.of(SEMICOLON, IS, RENAMES));
177                packageSpecAlternative
178                                .sequence(SEMICOLON)
179                                .createNode(EShallowEntityType.MODULE, "package specification",
180                                                -2).endNode();
181                packageSpecAlternative.sequence(RENAMES)
182                                .createNode(EShallowEntityType.MODULE, "package renaming", -2)
183                                .skipTo(SEMICOLON).endNode();
184                packageSpecAlternative
185                                .sequence(IS, NEW)
186                                .createNode(EShallowEntityType.TYPE,
187                                                "generic package instantiation", -3).skipTo(SEMICOLON)
188                                .endNode();
189                packageSpecAlternative
190                                .sequence(IS)
191                                .createNode(EShallowEntityType.MODULE, "package specification",
192                                                -2).parseUntil(DECLARATIONS).sequence(END)
193                                .skipTo(SEMICOLON).endNode();
194
195                // task specification
196                RecognizerBase<EAdaParserStates> taskSpecAlternative = inAnyState()
197                                .sequence(EnumSet.of(TASK, PROTECTED), IDENTIFIER)
198                                .createNode(EShallowEntityType.MODULE,
199                                                new Object[] { 0, "specification" }, -1)
200                                .skipBefore(EnumSet.of(SEMICOLON, IS));
201                taskSpecAlternative.sequence(SEMICOLON).endNode();
202                taskSpecAlternative.sequence(IS).parseUntil(DECLARATIONS).sequence(END)
203                                .skipTo(SEMICOLON).endNode();
204        }
205
206        /**
207         * Creates parsing rules for package/task/protected body and task/protected
208         * type.
209         */
210        private void createBodyAndTypeRules() {
211                // package body, task body, protected body
212                RecognizerBase<EAdaParserStates> packageBodyAlternative1 = inAnyState()
213                                .sequence(EnumSet.of(PACKAGE, TASK, PROTECTED), BODY,
214                                                IDENTIFIER).skipTo(IS)
215                                .createNode(EShallowEntityType.MODULE, new int[] { 0, 1 }, -2);
216                packageBodyAlternative1.sequence(SEPARATE, SEMICOLON).endNode();
217                RecognizerBase<EAdaParserStates> packageBodyAlternative2 = packageBodyAlternative1
218                                .parseUntil(DECLARATIONS);
219                packageBodyAlternative2.sequence(END).skipTo(SEMICOLON).endNode();
220                completeBlock(packageBodyAlternative2.sequence(BEGIN));
221
222                // task types and protected types
223                RecognizerBase<EAdaParserStates> taskTypeAlternative = inAnyState()
224                                .sequence(EnumSet.of(TASK, PROTECTED), TYPE, IDENTIFIER)
225                                .createNode(EShallowEntityType.MODULE, new int[] { 0, 1 })
226                                .skipBefore(EnumSet.of(SEMICOLON, IS));
227                taskTypeAlternative.sequence(SEMICOLON).endNode();
228                taskTypeAlternative.sequence(IS).parseUntil(DECLARATIONS).sequence(END)
229                                .skipTo(SEMICOLON).endNode();
230
231                // new...with is skipped
232                inState(DECLARATIONS).sequence(NEW).skipTo(WITH);
233        }
234
235        /**
236         * Creates rules for parsing methods (functions, etc.) and attributes (and
237         * local variables).
238         */
239        private void createMethodAndAttributeRules() {
240                // functions, procedures (including operator overloading), entries
241                RecognizerBase<EAdaParserStates> functionAlternative = inAnyState()
242                                .sequence(EnumSet.of(PROCEDURE, FUNCTION, ENTRY),
243                                                EnumSet.of(IDENTIFIER, STRING_LITERAL))
244                                .createNode(EShallowEntityType.METHOD, 0, 1)
245                                .skipBefore(EnumSet.of(SEMICOLON, IS));
246                functionAlternative.sequence(SEMICOLON).endNode();
247                functionAlternative.sequence(IS, EnumSet.of(SEPARATE, ABSTRACT, NEW))
248                                .skipTo(SEMICOLON).endNode();
249                completeBlock(functionAlternative.sequence(IS).parseUntil(DECLARATIONS)
250                                .sequence(BEGIN));
251
252                // variables and constants
253                inState(DECLARATIONS).sequence(IDENTIFIER, COLON)
254                                .createNode(EShallowEntityType.ATTRIBUTE, "variable", 0)
255                                .skipTo(SEMICOLON).endNode();
256        }
257
258        /** Creates rules for parsing types and similar constructs. */
259        private void createTypeRules() {
260                // types/subtypes
261                RecognizerBase<EAdaParserStates> typeAlternative = inAnyState()
262                                .sequence(EnumSet.of(TYPE, SUBTYPE), IDENTIFIER)
263                                .createNode(EShallowEntityType.TYPE, 0, 1)
264                                .skipBefore(EnumSet.of(SEMICOLON, IS));
265                typeAlternative.sequence(SEMICOLON).endNode();
266                typeAlternative.sequence(IS, NULL, RECORD, SEMICOLON).endNode();
267
268                RecognizerBase<EAdaParserStates> typeAlternative2 = typeAlternative
269                                .sequence(IS).skipBefore(EnumSet.of(SEMICOLON, RECORD, NULL));
270                typeAlternative2.sequence(SEMICOLON).endNode();
271                typeAlternative2.sequence(NULL).skipTo(SEMICOLON).endNode();
272                typeAlternative2.sequence(RECORD).skipTo(END, ENDRECORD, SEMICOLON)
273                                .endNode();
274
275                // representation clauses
276                RecognizerBase<EAdaParserStates> overlayAlternative = inState(
277                                DECLARATIONS).sequence(FOR)
278                                .createNode(EShallowEntityType.TYPE, "representation clause")
279                                .skipBefore(EnumSet.of(SEMICOLON, RECORD));
280                overlayAlternative.sequence(SEMICOLON).endNode();
281                overlayAlternative.sequence(RECORD).skipTo(END, ENDRECORD, SEMICOLON)
282                                .endNode();
283        }

285        /** Creates the rules needed for parsing statements. */
286        private void createStatementRules() {
287                // if/elseif
288                RecognizerBase<EAdaParserStates> ifAlternative = inState(STATEMENTS)
289                                .sequence(EnumSet.of(IF, ELSEIF))
290                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(THEN)
291                                .parseUntil(STATEMENTS)
292                                .sequenceBefore(EnumSet.of(ELSEIF, ELSE, END));
293                ifAlternative.sequence(END).skipTo(SEMICOLON).endNode();
294                ifAlternative.endNodeWithContinuation();
295
296                // else (both for if and select)
297                inState(STATEMENTS).sequence(ELSE)
298                                .createNode(EShallowEntityType.STATEMENT, 0)
299                                .parseUntil(STATEMENTS).sequence(END).skipTo(SEMICOLON)
300                                .endNode();
301
302                // case
303                inState(STATEMENTS).sequence(CASE)
304                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(IS)
305                                .parseUntil(STATEMENTS).sequence(END).skipTo(SEMICOLON)
306                                .endNode();
307
308                // when (in case, select, and exception handlers)
309                inState(STATEMENTS).sequence(WHEN)
310                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(EQ, GT)
311                                .endNode();
312
313                // loop/block labels as meta
314                inState(STATEMENTS).sequence(IDENTIFIER, COLON)
315                                .sequenceBefore(EnumSet.of(WHILE, FOR, LOOP, BEGIN))
316                                .createNode(EShallowEntityType.META, "Loop name").endNode();
317
318                // loops
319                inState(STATEMENTS).sequence(LOOP)
320                                .createNode(EShallowEntityType.STATEMENT, 0)
321                                .parseUntil(STATEMENTS).sequence(END).skipTo(SEMICOLON)
322                                .endNode();
323                inState(STATEMENTS).sequence(EnumSet.of(WHILE, FOR))
324                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(LOOP)
325                                .parseUntil(STATEMENTS).sequence(END).skipTo(SEMICOLON)
326                                .endNode();
327
328                // blocks
329                completeBlock(inState(STATEMENTS).sequence(IDENTIFIER, COLON, DECLARE)
330                                .createNode(EShallowEntityType.STATEMENT, "block")
331                                .parseUntil(DECLARATIONS).sequence(BEGIN));
332                completeBlock(inState(STATEMENTS).sequence(DECLARE)
333                                .createNode(EShallowEntityType.STATEMENT, "block")
334                                .parseUntil(DECLARATIONS).sequence(BEGIN));
335                completeBlock(inState(STATEMENTS).sequence(BEGIN).createNode(
336                                EShallowEntityType.STATEMENT, "block"));
337
338                // accept/do
339                inState(STATEMENTS).sequence(ACCEPT)
340                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DO)
341                                .parseUntil(STATEMENTS).sequence(END).skipTo(SEMICOLON)
342                                .endNode();
343
344                // select/or
345                RecognizerBase<EAdaParserStates> selectAlternative = inState(STATEMENTS)
346                                .sequence(EnumSet.of(SELECT, OR, ELSE))
347                                .createNode(EShallowEntityType.STATEMENT, 0)
348                                .parseUntil(STATEMENTS);
349                selectAlternative.sequenceBefore(EnumSet.of(OR, ELSE))
350                                .endNodeWithContinuation();
351                selectAlternative.sequence(END).skipTo(SEMICOLON).endNode();
352
353                // goto labels as meta
354                inState(STATEMENTS).sequence(LEFT_LABEL_BRACKET)
355                                .skipTo(RIGHT_LABEL_BRACKET)
356                                .createNode(EShallowEntityType.META, "goto label").endNode();
357
358                // basic statement
359                inState(STATEMENTS)
360                                .sequence(
361                                                EnumSet.of(IDENTIFIER, NULL, RETURN, GOTO, EXIT, ABORT,
362                                                                DELAY, RAISE, TERMINATE))
363                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(SEMICOLON)
364                                .endNode();
365        }
366
367        /** Completes the rule describing blocks. */
368        private void completeBlock(RecognizerBase<EAdaParserStates> initialSequence) {
369                RecognizerBase<EAdaParserStates> alt = initialSequence
370                                .parseUntil(STATEMENTS);
371                alt.sequence(EXCEPTION).parseUntil(STATEMENTS).sequence(END)
372                                .skipTo(SEMICOLON).endNode();
373                alt.sequence(END).skipTo(SEMICOLON).endNode();
374        }

376        /**
377         * {@inheritDoc}
378         * <p>
379         * Maps "and then" and "or else" to simple "and" and "or", as the additional
380         * "then" and "else" keywords may shallow parsing much harder.
381         */
382        @Override
383        protected boolean isFilteredToken(IToken token, IToken previousToken) {
384                ETokenType previousType = null;
385                if (previousToken != null) {
386                        previousType = previousToken.getType();
387                }
388                ETokenType type = token.getType();
389
390                if (previousType == AND && type == THEN) {
391                        return true;
392                }
393                if (previousType == OR && type == ELSE) {
394                        return true;
395                }
396                return super.isFilteredToken(token, previousToken);
397        }
