017package eu.cqse.check.framework.shallowparser.languages.base;
018
019import static eu.cqse.check.framework.scanner.ETokenType.CASE;
020import static eu.cqse.check.framework.scanner.ETokenType.CATCH;
021import static eu.cqse.check.framework.scanner.ETokenType.COLON;
022import static eu.cqse.check.framework.scanner.ETokenType.COMMA;
023import static eu.cqse.check.framework.scanner.ETokenType.DEFAULT;
024import static eu.cqse.check.framework.scanner.ETokenType.DO;
025import static eu.cqse.check.framework.scanner.ETokenType.ELSE;
026import static eu.cqse.check.framework.scanner.ETokenType.EQ;
027import static eu.cqse.check.framework.scanner.ETokenType.FINALLY;
028import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
029import static eu.cqse.check.framework.scanner.ETokenType.IF;
030import static eu.cqse.check.framework.scanner.ETokenType.LBRACE;
031import static eu.cqse.check.framework.scanner.ETokenType.LBRACK;
032import static eu.cqse.check.framework.scanner.ETokenType.LITERALS;
033import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
034import static eu.cqse.check.framework.scanner.ETokenType.RBRACE;
035import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
036import static eu.cqse.check.framework.scanner.ETokenType.SEMICOLON;
037import static eu.cqse.check.framework.scanner.ETokenType.TRY;
038import static eu.cqse.check.framework.scanner.ETokenType.WHILE;
039import static eu.cqse.check.framework.shallowparser.languages.base.EGenericParserStates.IN_EXPRESSION;
040import static eu.cqse.check.framework.shallowparser.languages.base.EGenericParserStates.IN_METHOD;
041import static eu.cqse.check.framework.shallowparser.languages.base.EGenericParserStates.IN_TYPE;
042
043import java.util.EnumSet;
044import java.util.HashSet;
045
046import eu.cqse.check.framework.scanner.ETokenType;
047import eu.cqse.check.framework.shallowparser.SubTypeNames;
048import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
049import eu.cqse.check.framework.shallowparser.framework.RecognizerBase;
050import eu.cqse.check.framework.shallowparser.framework.ShallowParserBase;
051
052/**
053 * Base class for C-style languages (C++, Java, C#).
054 */
055public abstract class CStyleShallowParserBase extends ShallowParserBase<EGenericParserStates> {
056
057        /** Constructor. */
058        protected CStyleShallowParserBase() {
059                super(EGenericParserStates.class, EGenericParserStates.TOP_LEVEL);
060                createMetaRules();
061                createTypeRules();
062                createClassElementsRules();
063                createStatementRules();
064                createSubExpressionRules();
065        }
066
067        /** Creates rules for meta elements. */
068        protected void createMetaRules() {
069                // deal with dangling closing braces by inserting broken node
070                inAnyState().sequence(RBRACE).createNode(EShallowEntityType.META, "dangling closing brace"); // endNode()
071                                                                                                                                                                                                                // omitted
072        }
073
074        /** Parser rules for module/namespace and type creation. */
075        protected void createTypeRules() {
076                // types; we have to ensure when skipping to the LBRACE, that there is
077                // no earlier SEMICOLON
078                inAnyState().repeated(getTypeModifier()).markStart().sequence(getTypeKeywords(), getValidIdentifiers())
079                                .skipBefore(EnumSet.of(SEMICOLON, LBRACE)).sequence(LBRACE).createNode(EShallowEntityType.TYPE, 0, 1)
080                                .parseUntil(IN_TYPE).sequence(RBRACE).endNode();
081        }
082
083        /**
084         * Returns the valid type modifiers for the language. Default implementation
085         * returns empty set. Override to use correct modifiers.
086         */
087        protected EnumSet<ETokenType> getTypeModifier() {
088                return EnumSet.noneOf(ETokenType.class);
089        }
090
091        /** Returns the set of keywords that start a type. */
092        protected abstract EnumSet<ETokenType> getTypeKeywords();
093
094        /** Parser rules for both attributes and methods. */
095        protected abstract void createClassElementsRules();
096
097        /** Creates parser rules for statements. */
098        protected void createStatementRules() {
099                createEmptyStatementRule();
100                createLabelRule();
101                createElseIfRule();
102                createBasicBlockRules();
103                createCaseRule();
104                createDoWhileRule();
105                createGenericBlockRule();
106                createSimpleStatementRule();
107        }
108
109        /** The empty statement. */
110        private void createEmptyStatementRule() {
111                inState(IN_METHOD).sequence(SEMICOLON).createNode(EShallowEntityType.STATEMENT, SubTypeNames.EMPTY_STATEMENT)
112                                .endNode();
113        }
114
115        /** Matches labels. */
116        private void createLabelRule() {
117                // filter out labels as meta as they do not increase statement count
118                inState(IN_METHOD).sequence(getValidIdentifiers(), COLON)
119                                .createNode(EShallowEntityType.META, SubTypeNames.LABEL, 0).endNode();
120        }
121
122        /** Special rule for else-if. */
123        private void createElseIfRule() {
124                RecognizerBase<EGenericParserStates> elseIfAlternative = inState(IN_METHOD).sequence(ELSE, IF)
125                                .createNode(EShallowEntityType.STATEMENT, new int[] { 0, 1 })
126                                .skipNested(LPAREN, RPAREN, getSubExpressionRecognizer());
127                endWithPossibleContinuation(elseIfAlternative.sequence(LBRACE).parseUntil(IN_METHOD).sequence(RBRACE),
128                                EnumSet.of(ELSE));
129                endWithPossibleContinuation(elseIfAlternative.parseOnce(IN_METHOD), EnumSet.of(ELSE));
130        }
131
132        /**
133         * Block constructs, such as if/else, while/for/switch, try/catch/finally,
134         * synchronized (only in some languages).
135         */
136        private void createBasicBlockRules() {
137                createBlockRuleWithContinuation(getSimpleBlockKeywordsWithParentheses(), null, true, false);
138                createBlockRuleWithContinuation(getSimpleBlockKeywordsWithoutParentheses(), null, false, false);
139                createBlockRuleWithContinuation(EnumSet.of(IF), EnumSet.of(ELSE), true, false);
140                createBlockRuleWithContinuation(EnumSet.of(TRY, CATCH), EnumSet.of(CATCH, FINALLY), true, true);
141        }
142
143        /**
144         * Case statement is parsed as meta, as it is hardly a statement on its own.
145         */
146        protected void createCaseRule() {
147                HashSet<ETokenType> literalsAndIdentifiers = new HashSet<ETokenType>(LITERALS);
148                literalsAndIdentifiers.addAll(getValidIdentifiers());
149                inState(IN_METHOD).sequence(CASE, literalsAndIdentifiers, COLON).createNode(EShallowEntityType.META, 0, 1)
150                                .endNode();
151                inState(IN_METHOD).sequence(CASE, LPAREN).skipToWithNesting(RPAREN, LPAREN, RPAREN).sequence(COLON)
152                                .createNode(EShallowEntityType.META, 0).endNode();
153
154                inState(IN_METHOD).sequence(DEFAULT, COLON).createNode(EShallowEntityType.META, 0).endNode();
155        }
156
157        /** Creates the do-while rule. */
158        private void createDoWhileRule() {
159                RecognizerBase<EGenericParserStates> doWhileAlternative = inState(IN_METHOD).sequence(DO)
160                                .createNode(EShallowEntityType.STATEMENT, 0);
161                doWhileAlternative.sequence(LBRACE).parseUntil(IN_METHOD).sequence(RBRACE, WHILE)
162                                .skipNested(LPAREN, RPAREN, getSubExpressionRecognizer()).optional(ETokenType.SEMICOLON).endNode();
163                doWhileAlternative.parseOnce(IN_METHOD).sequence(WHILE).skipNested(LPAREN, RPAREN, getSubExpressionRecognizer())
164                                .optional(ETokenType.SEMICOLON).endNode();
165        }
166
167        /** Generic block. */
168        private void createGenericBlockRule() {
169                inState(IN_METHOD).sequence(LBRACE).createNode(EShallowEntityType.STATEMENT, SubTypeNames.ANONYMOUS_BLOCK)
170                                .parseUntil(IN_METHOD).sequence(RBRACE).endNode();
171        }
172
173        /** Simple statement. */
174        protected void createSimpleStatementRule() {
175                // heuristic for detecting local variables
176                completeSimpleStatement(typePatternInState(IN_METHOD).markStart().sequenceBefore(getValidIdentifiers(),
177                                EnumSet.of(COMMA, EQ, SEMICOLON, LBRACK)), SubTypeNames.LOCAL_VARIABLE);
178
179                completeSimpleStatement(inState(IN_METHOD).sequence(getStatementStartTokens()));
180                completeSimpleStatement(typePatternInState(IN_METHOD));
181                completeSimpleStatement(inState(IN_METHOD).sequence(LITERALS));
182        }
183
184        /**
185         * Creates rules for dealing with constructs in subexpressions, such as
186         * anonymous classes, lambdas, etc.
187         */
188        protected void createSubExpressionRules() {
189                // default implementation does nothing
190        }
191
192        /**
193         * Returns a recognizer used for detecting sub expressions (anonymous
194         * classes, lambdas, etc.) within expressions. This may return null (which
195         * is done by the default implementation).
196         */
197        protected RecognizerBase<EGenericParserStates> getSubExpressionRecognizer() {
198                return null;
199        }
200
201        /**
202         * Returns the set of all valid identifiers, i.e. token types that can be
203         * used to name elements in the language.
204         */
205        protected EnumSet<ETokenType> getValidIdentifiers() {
206                return EnumSet.of(IDENTIFIER);
207        }
208
209        /**
210         * Returns the set of all keywords that start a simple block with optional
211         * parentheses (see implementers for examples).
212         */
213        protected abstract EnumSet<ETokenType> getSimpleBlockKeywordsWithParentheses();
214
215        /**
216         * Returns the set of all keywords that start a simple block but are never
217         * followed by parentheses (see implementers for examples).
218         */
219        protected abstract EnumSet<ETokenType> getSimpleBlockKeywordsWithoutParentheses();
220
221        /**
222         * Returns a set of all tokens that can start a statement, besides a type
223         * (see {@link #typePatternInState(EGenericParserStates...)} and a literal.
224         */
225        protected abstract EnumSet<ETokenType> getStatementStartTokens();
226
227        /** Creates a recognizer that matches all valid types. */
228        protected abstract RecognizerBase<EGenericParserStates> typePattern(
229                        RecognizerBase<EGenericParserStates> currentState);
230
231        /**
232         * Creates a recognizer that matches all valid types, starting from the
233         * given state.
234         */
235        protected RecognizerBase<EGenericParserStates> typePatternInState(EGenericParserStates... states) {
236                return typePattern(inState(states));
237        }
238
239        /**
240         * Creates a rule for recognizing a statement starting with a single
241         * keyword, optionally followed by an expression in parentheses, and
242         * followed by a block or a single statement.
243         * 
244         * @param continuationTokens
245         *            list of tokens that indicate a continued statement if
246         *            encountered after the block. May be null.
247         */
248        protected void createBlockRuleWithContinuation(EnumSet<ETokenType> startTokens,
249                        EnumSet<ETokenType> continuationTokens, boolean canBeFollowedByParentheses, boolean alwaysBraces) {
250                RecognizerBase<EGenericParserStates> alternative = inState(IN_METHOD).sequence(startTokens)
251                                .createNode(EShallowEntityType.STATEMENT, 0);
252                if (canBeFollowedByParentheses) {
253                        alternative = alternative.skipNested(LPAREN, RPAREN, getSubExpressionRecognizer());
254                }
255
256                if (alwaysBraces) {
257                        alternative = alternative.skipBefore(LBRACE);
258                }
259
260                endWithPossibleContinuation(alternative.sequence(LBRACE).parseUntil(IN_METHOD).sequence(RBRACE),
261                                continuationTokens);
262                endWithPossibleContinuation(alternative.parseOnce(IN_METHOD), continuationTokens);
263        }
264
265        /** Completes a recognizer for a simple statement. */
266        protected void completeSimpleStatement(RecognizerBase<EGenericParserStates> baseRecognizer) {
267                completeSimpleStatement(baseRecognizer, SubTypeNames.SIMPLE_STATEMENT);
268        }
269
270        /** Completes a recognizer for a simple statement. */
271        protected void completeSimpleStatement(RecognizerBase<EGenericParserStates> baseRecognizer, String subtype) {
272                RecognizerBase<EGenericParserStates> alternative = baseRecognizer
273                                .createNode(EShallowEntityType.STATEMENT, subtype, 0).skipBeforeWithNesting(
274                                                EnumSet.of(SEMICOLON, RBRACE), LBRACE, RBRACE, LPAREN, RPAREN, getSubExpressionRecognizer());
275
276                alternative.sequence(SEMICOLON).endNode();
277
278                // this (empty) alternative captures the case where a statement is not
279                // closed by a semicolon, so we deliberately leave it open. While in
280                // most languages this is an error (and then this rule helps us to
281                // continue parsing), in C++ you can construct valid statements without
282                // semicolon using macros (although it is discouraged).
283                alternative.sequence();
284        }
285
286        /** Creates rules for parsing lambdas with arrows like in Java or C#. */
287        protected void createLambdaWithArrowRules(ETokenType arrowType) {
288                // lambda expressions
289                completeLambda(inState(IN_EXPRESSION).sequence(getValidIdentifiers()), arrowType);
290                completeLambda(inState(IN_EXPRESSION).sequence(LPAREN).skipTo(RPAREN), arrowType);
291
292                // additional rule for parsing lambda expressions (without braces). see
293                // completeLambda() for details
294                // the node start is moved one token to the right so the shallow
295                // entities produced by this rule don't include the arrow (instead it
296                // will be included in the parent entity)
297                inState(IN_EXPRESSION).sequence(arrowType)
298                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.LAMBDA_EXPRESSION, null, 1)
299                                .skipBeforeWithNesting(EnumSet.of(RPAREN, SEMICOLON, RBRACE, COMMA), LPAREN, RPAREN, LBRACE, RBRACE)
300                                .endNode();
301        }
302
303        /** Completes a rule for parsing lambda expressions. */
304        private static void completeLambda(RecognizerBase<EGenericParserStates> ruleStart, ETokenType arrowType) {
305                RecognizerBase<EGenericParserStates> lambdaAlternative = ruleStart.createNode(EShallowEntityType.METHOD,
306                                SubTypeNames.LAMBDA);
307                lambdaAlternative.sequence(arrowType, LBRACE).parseUntil(IN_METHOD).sequence(RBRACE).endNode();
308
309                // we start parsing before the arrow as this allows our special
310                // statement rule to capture this case. This is required, as this kind
311                // of expression is not terminated by a semicolon.
312                lambdaAlternative.sequenceBefore(arrowType).parseOnce(IN_EXPRESSION).endNode();
313        }
