017package eu.cqse.check.framework.shallowparser.languages.base;
018
019import static eu.cqse.check.framework.scanner.ETokenType.LBRACE;
020import static eu.cqse.check.framework.scanner.ETokenType.LBRACK;
021import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
022import static eu.cqse.check.framework.scanner.ETokenType.RBRACE;
023import static eu.cqse.check.framework.scanner.ETokenType.RBRACK;
024import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
025import static eu.cqse.check.framework.scanner.ETokenType.SEMICOLON;
026
027import java.util.EnumMap;
028import java.util.List;
029import java.util.Map;
030import java.util.Stack;
031
032import eu.cqse.check.framework.shallowparser.SubTypeNames;
033import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
034import eu.cqse.check.framework.shallowparser.framework.ParserState;
035import eu.cqse.check.framework.shallowparser.framework.RecognizerBase;
036import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
037import eu.cqse.check.framework.scanner.ETokenType;
038import eu.cqse.check.framework.scanner.IToken;
039
040/**
041 * Base class for recognizing statements in a language whose statements are
042 * primarily line-based. It also assumes that a semicolon may always be used to
043 * terminate a statement.
044 * 
045 * @author $Author: kinnen $
046 * @version $Rev: 53317 $
047 * @ConQAT.Rating GREEN Hash: CA64EC05F6533BF76D74C0CDB6D2A090
048 */
049public abstract class LineBasedStatementRecognizerBase<STATE extends Enum<STATE>>
050                extends RecognizerBase<STATE> {
051
052        /** Matched tokens for nesting in complex statements. */
053        protected final static Map<ETokenType, ETokenType> NESTING_MATCH = new EnumMap<ETokenType, ETokenType>(
054                        ETokenType.class);
055
056        static {
057                NESTING_MATCH.put(LPAREN, RPAREN);
058                NESTING_MATCH.put(LBRACK, RBRACK);
059                NESTING_MATCH.put(LBRACE, RBRACE);
060        }
061
062        /** {@inheritDoc} */
063        @Override
064        protected int matchesLocally(ParserState<STATE> parserState,
065                        List<IToken> tokens, int startOffset) {
066                IToken lastToken = null;
067                Stack<ETokenType> expectedClosing = new Stack<ETokenType>();
068
069                // create a node here, so we can append function nodes
070                parserState.setNode(new ShallowEntity(getEntityType(),
071                                getEntitySubtypeName(), tokens.get(startOffset).getText(),
072                                tokens, startOffset));
073
074                while (true) {
075                        if (startOffset >= tokens.size()) {
076                                return startOffset;
077                        }
078
079                        IToken token = tokens.get(startOffset);
080                        ETokenType tokenType = token.getType();
081
082                        if (!expectedClosing.isEmpty()
083                                        && tokenType == expectedClosing.peek()) {
084                                expectedClosing.pop();
085                        } else if (expectedClosing.isEmpty() && tokenType == SEMICOLON) {
086                                return startOffset + 1;
087                        } else if (expectedClosing.isEmpty()
088                                        && startsNewStatement(token, lastToken)) {
089                                return startOffset;
090                        } else if (tokenStartsSubParse(tokenType, tokens, startOffset)) {
091                                int next = parserState.parse(getSubParseState(), tokens,
092                                                startOffset);
093                                if (next == NO_MATCH) {
094                                        return NO_MATCH;
095                                }
096                                startOffset = next;
097                                lastToken = tokens.get(startOffset - 1);
098                                continue;
099                        } else if (NESTING_MATCH.containsKey(tokenType)) {
100                                expectedClosing.push(NESTING_MATCH.get(tokenType));
101                        }
102
103                        lastToken = token;
104                        startOffset += 1;
105                }
106        }
107
108        /**
109         * @return The entity type that is to be created when matching.
110         */
111        protected EShallowEntityType getEntityType() {
112                return EShallowEntityType.STATEMENT;
113        }
114
115        /**
116         * @return The entity subtype name to be used when creating a node.
117         */
118        protected String getEntitySubtypeName() {
119                return SubTypeNames.SIMPLE_STATEMENT;
120        }
121
122        /** Returns the state to be used for a sub parse. */
123        protected abstract STATE getSubParseState();
124
125        /**
126         * Returns true if the token signals to start a sub parse (e.g. embedded
127         * classes, functions, etc.).
128         * 
129         * @param tokenType
130         *            the type of the current token.
131         * @param tokens
132         *            the entire token stream.
133         * @param offset
134         *            the offset of the current token in the token stream.
135         */
136        protected abstract boolean tokenStartsSubParse(ETokenType tokenType,
137                        List<IToken> tokens, int offset);
138
139        /** Returns true if the given token starts a new statement. */
140        protected abstract boolean startsNewStatement(IToken token, IToken lastToken);
141}
