088         * Create rules for matching divisions. Though there are 4 divisions, we
089         * take only the ID division as being in top level and everything afterwards

110        /**
111         * Create rule to identify a program, method, factory, object or class.
112         * Factories and objects do not need to have names like programs, methods
113         * and classes in Enterprise COBOL for z/OS
114         */
115        private void createRuleForProgramType(ETokenType startType, EShallowEntityType entityType, ETokenType endType,
116                        boolean canIncludeEndIdentifier) {
117                RecognizerBase<ECobolParserState> recognizer = inAnyState().sequence(startType, DOT);
118                if (canIncludeEndIdentifier) {
119                        recognizer = recognizer.sequence(EnumSet.of(IDENTIFIER, STRING_LITERAL));
120                        if (startType == PROGRAM_ID) {
121                                recognizer = recognizer.optional(IS).optional(EnumSet.of(RECURSIVE, COMMON, INITIAL)).skipTo(DOT);
122                        } else if (startType == CLASS_ID) {
123                                recognizer = recognizer.optional(INHERITS, IDENTIFIER).sequence(DOT);
124                        } else {
125                                recognizer = recognizer.sequence(DOT);
126                        }
127
128                        // To have the test{@link
129                        // CobolShallowParserTest#testMinimalProgramStructure} not return
130                        // an incompletely parsed node program node (since there is no end
131                        // node), we use parseUntilOrEof()
132                        recognizer = recognizer.createNode(entityType, 2).parseUntilOrEof(TOP_LEVEL);
133                } else {
134                        recognizer = recognizer.createNode(entityType, 0).parseUntilOrEof(TOP_LEVEL);
135                }
136
137                recognizer.sequence(END, endType, DOT).endNode();
138                if (canIncludeEndIdentifier) {
139                        recognizer.sequence(END, endType, IDENTIFIER, DOT).endNode();
140                }
141        }

985        /**
986         * Completes the rule provided in a recognizer depending on the statement
987         * end type. For SIMPLE verbs, matches endToken. For IMPERATIVE verbs,
988         * parses once IN_STATEMENT while for SCOPE verbs, checks if outside of
989         * scope and parses like IMPERATIVE verb otherwise parses until endToken.
990         */
991        private static void completeRule(RecognizerBase<ECobolParserState> recognizer, ETokenType endToken,
992                        EStatementEndType statementEndType) {
993                switch (statementEndType) {
994                case SIMPLE:
995                        if (endToken != null) {
996                                recognizer.sequence(endToken).endNode();
997                        }
998
999                        // Ensures that statements are properly closed with a DOT (if
1000                        // present). This fixes the dot-close-all scope bug as well as
1001                        // causes the code to be correctly displayed in the code perspective
1002                        // (especially in pretty-print mode).
1003                        recognizer.notPreCondition(new OutOfScopeRecognizer()).sequenceBefore(DOT).endNode();
1004                        recognizer.preCondition(new OutOfScopeRecognizer()).sequence(DOT).endNode();
1005                        recognizer.endNode();
1006                        break;
1007                case IMPERATIVE:
1008                        recognizer.parseOnce(IN_STATEMENT).endNode();
1009                        break;
1010                case SCOPE:
1011                default:
1012                        // Extra check we are not outside a scope. Fall-through is intended
1013                        recognizer.preCondition(new OutOfScopeRecognizer()).parseOnce(IN_STATEMENT).endNode();
1014                        RecognizerBase<ECobolParserState> finalRecognizer = recognizer.notPreCondition(new OutOfScopeRecognizer())
1015                                        .parseUntil(IN_STATEMENT);
1016                        if (endToken != null) {
1017                                finalRecognizer.sequence(endToken).endNode();
1018                        }
1019
1020                        // Not 'eating' the dot to fix the DOT-close all scope bug
1021                        finalRecognizer.sequenceBefore(DOT).endNode();
1022                }
1023        }

1826         * Create rules for the various perform statements
1827         */
1828        private void createPerformRules() {
1829                RecognizerBase<ECobolParserState> performRecognizer = inState(TOP_LEVEL, IN_STATEMENT).sequence(PERFORM);

1839        /**
1840         * Complete a rule for the perform verb.
1841         */
1842        private void completePerformRule(RecognizerBase<ECobolParserState> recognizer, EStatementEndType statementEndType) {
1843                EnumSet<ETokenType> skippedTokens = EnumSet.of(WITH, TEST, BEFORE, AFTER);
1844                RecognizerBase<ECobolParserState> timesRecognizer = recognizer.sequence(EnumSet.of(IDENTIFIER, INTEGER_LITERAL),
1845                                TIMES);
1846                RecognizerBase<ECobolParserState> untilRecognizer = recognizer.skipAny(skippedTokens).sequence(UNTIL)
1847                                .subRecognizer(getConditionSubRecognizer());
1848                RecognizerBase<ECobolParserState> varyingUntilRecognizer = recognizer.skipAny(skippedTokens).sequence(VARYING)
1849                                .skipTo(UNTIL).subRecognizer(getConditionSubRecognizer());
1850                RecognizerBase<ECobolParserState> repeatedAfterUntilRecognizer = emptyRecognizer().sequence(AFTER).skipTo(UNTIL)
1851                                .subRecognizer(getConditionSubRecognizer());
1852                RecognizerBase<ECobolParserState> varyingUntilAfterRecognizer = varyingUntilRecognizer
1853                                .repeatedSubRecognizer(repeatedAfterUntilRecognizer);
1854
1855                completeRule(varyingUntilAfterRecognizer, END_PERFORM, statementEndType);
1856                completeRule(varyingUntilRecognizer, END_PERFORM, statementEndType);
1857                completeRule(untilRecognizer, END_PERFORM, statementEndType);
1858                completeRule(timesRecognizer, END_PERFORM, statementEndType);
1859                if (statementEndType == SIMPLE) {
1860                        recognizer.notPreCondition(new OutOfScopeRecognizer()).sequenceBefore(END_PERFORM).endNode();
1861                }
1862                completeRule(recognizer, END_PERFORM, statementEndType);
1863        }

1865        /**
1866         * Get the recognizer to match arithmetic expressions. A comma is not an
1867         * arithmetic operator but its included because as a Cobol separator, it
1868         * helps to match repeated identifiers or expressions
1869         */
1870        private RecognizerBase<ECobolParserState> getArithmeticExpressionRecognizer(boolean includeComma) {
1871                EnumSet<ETokenType> operatorTokens = EnumSet.of(PLUS, MINUS, DIV, MULT, B_AND, B_OR, B_NOT, B_XOR);
1872                EnumSet<ETokenType> operandTokens = EnumSet.copyOf(IDENTIFIER_LITERALS_AND_ZERO);
1873                operandTokens.addAll(EnumSet.of(NULL, NULLS, SELF));
1874                operandTokens.remove(STRING_LITERAL);
1875                if (includeComma) {
1876                        operandTokens.add(COMMA);
1877                }
1878                RecognizerBase<ECobolParserState> operatorBeforeExpressionRecognizer = emptyRecognizer()
1879                                .sequence(operatorTokens);
1880                operatorBeforeExpressionRecognizer.sequence(operandTokens);
1881                operatorBeforeExpressionRecognizer.skipNested(LPAREN, RPAREN);
1882                RecognizerBase<ECobolParserState> repeatedExpresionRecognizer = emptyRecognizer();
1883                repeatedExpresionRecognizer.subRecognizer(operatorBeforeExpressionRecognizer);
1884                repeatedExpresionRecognizer.skipNested(LPAREN, RPAREN);
1885
1886                RecognizerBase<ECobolParserState> result = emptyRecognizer();
1887                result.sequence(operandTokens).repeatedSubRecognizer(repeatedExpresionRecognizer);
1888                result.skipNested(LPAREN, RPAREN).repeatedSubRecognizer(repeatedExpresionRecognizer);
1889                return result;
1890        }

1892        /**
1893         * Returns a basic condition sub recognizer.
1894         */
1895        private RecognizerBase<ECobolParserState> getConditionSubRecognizer() {
1896                EnumSet<ETokenType> leftOperandTokens = EnumSet.copyOf(IDENTIFIER_LITERALS_AND_ZERO);
1897                leftOperandTokens.addAll(EnumSet.of(NULL, NULLS, SELF));
1898                EnumSet<ETokenType> rightOperandTokens = EnumSet.copyOf(leftOperandTokens);
1899                rightOperandTokens.addAll(EnumSet.of(NUMERIC, ALPHABETIC, ALPHABETIC_UPPER, ALPHABETIC_LOWER, DBCS, KANJI));
1900
1901                RecognizerBase<ECobolParserState> singleOperandRecognizer = emptyRecognizer();
1902                singleOperandRecognizer.sequence(leftOperandTokens);
1903
1904                RecognizerBase<ECobolParserState> result = emptyRecognizer();
1905                result.optionalSubRecognizer(getArithmeticExpressionRecognizer(false))
1906                                .optionalSubRecognizer(singleOperandRecognizer);
1907                return result;
1908        }

1930        /**
1931         * Create the IF-statement and $IF-meta rules.
1932         */
1933        private void createIfRule(ETokenType ifType, EnumSet<ETokenType> elseOtherwise, ETokenType endIfType,
1934                        EShallowEntityType entityType) {
1935                RecognizerBase<ECobolParserState> ifRecognizer = inState(TOP_LEVEL, IN_STATEMENT).sequence(ifType)
1936                                .createNode(entityType, 0).parseUntil(IN_STATEMENT);
1937                ifRecognizer.sequenceBefore(elseOtherwise).endNodeWithContinuation();
1938
1939                // Precondition ensures that we are parsing an if-block without an else.
1940                ifRecognizer.preCondition(new NoElseRecognizer(ifType, endIfType)).sequence(endIfType).endNode();
1941                ifRecognizer.sequenceBefore(DOT).endNode();
1942        }
