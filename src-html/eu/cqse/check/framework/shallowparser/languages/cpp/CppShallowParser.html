098/**
099 * Shallow parser for C/C++.
100 * <p>
101 * What this parser does and does not:
102 * <ul>
103 * <li>The parser recognizes types (classes, enums, interfaces), methods and
104 * attributes, and individual statements.</li>
105 * <li>It recognizes the nesting of statements (e.g. in loops), but does not
106 * parse into the statements. For example, it recognizes an if-statement and
107 * provides the list of sub-statements, but does not provide direct access to
108 * the if-condition.</li>
109 * <li>All preprocessor statements are parsed as meta information.</li>
110 * <li>Template declarations are parsed as preceding meta information.</li>
111 * <li>Forward declarations are handled as meta information.</li>
112 * <li>We heuristically filter code generating macros, such as
113 * "CREATE_STUFF(MyClass)".</li>
114 * </ul>
115 *
116 * @author $Author: kinnen $
117 * @version $Rev: 53037 $
118 * @ConQAT.Rating GREEN Hash: 5018A2C55DE7FF72F1B12FE7B1CE8271
119 */
120public class CppShallowParser extends CStyleShallowParserBase {
121
122        /** Keywords used for primitive types. */
123        /* package */static final EnumSet<ETokenType> PRIMITIVE_TYPES = EnumSet.of(
124                        VOID, BYTE, SHORT, INT, LONG, FLOAT, DOUBLE, SIGNED, UNSIGNED,
125                        CHAR, BOOL);
126
127        /**
128         * Set of common "keywords" that are not actually part of the language but
129         * rather used by certain compilers and implicitly defined using macros. The
130         * solution used here is to filter them out.
131         */
132        private static final Set<String> PSEUDO_KEYWORDS = new HashSet<String>(
133                        Arrays.asList(
134                                        // typically found in Windows compilers
135                                        "__fastcall", "__export", "__forceinline", "_cdecl",
136                                        "_stdcall", "__stdcall", "WINAPI", "APIENTRY", "CALLBACK",
137                                        // used by the common Qt library
138                                        "Q_EXPORT",
139                                        // keywords found in ISA dialog manager
140                                        "DML_c", "DM_CALLBACK", "__1", "__2", "__3", "__4", "__5",
141                                        "__6", "__7", "DM_ENTRY", "DML_pascal", "DML_default",
142                                        // project specific keywords
143                                        "IGVPWORD_API"));

157        /** Creates namespace specific rules. */
158        private void createNamespaceRules() {
159                // using
160                inAnyState().sequence(USING).createNode(EShallowEntityType.META, 0)
161                                .skipTo(SEMICOLON).endNode();
162
163                // namespace
164                RecognizerBase<EGenericParserStates> namespaceAlternative = inAnyState()
165                                .sequence(NAMESPACE).skipBefore(EnumSet.of(SEMICOLON, LBRACE));
166                namespaceAlternative.sequence(LBRACE)
167                                .createNode(EShallowEntityType.MODULE, 0, new Region(1, -2))
168                                .parseUntil(TOP_LEVEL).sequence(RBRACE).endNode();
169                namespaceAlternative.sequence(SEMICOLON)
170                                .createNode(EShallowEntityType.META, 0, new Region(1, -2))
171                                .endNode();
172        }
173
174        /** {@inheritDoc} */
175        @Override
176        protected void createTypeRules() {
177                // typedef
178                RecognizerBase<EGenericParserStates> typeInTypedefAlternative = inAnyState()
179                                .sequence(TYPEDEF).optional(CONST);
180                typeInTypedefAlternative
181                                .sequenceBefore(getTypeKeywords(), IDENTIFIER, LBRACE)
182                                .createNode(EShallowEntityType.TYPE, 0).parseOnce(TOP_LEVEL)
183                                .skipTo(IDENTIFIER, SEMICOLON).endNodeWithName(-2);
184                typeInTypedefAlternative.sequenceBefore(getTypeKeywords(), LBRACE)
185                                .createNode(EShallowEntityType.TYPE, 0).parseOnce(TOP_LEVEL)
186                                .skipTo(IDENTIFIER, SEMICOLON).endNodeWithName(-2);
187
188                RecognizerBase<EGenericParserStates> simpleTypedefAlternative = inAnyState()
189                                .sequence(TYPEDEF).createNode(EShallowEntityType.TYPE, 0)
190                                .skipBefore(IDENTIFIER, EnumSet.of(SEMICOLON, RPAREN, LBRACK));
191                simpleTypedefAlternative.markStart()
192                                .sequenceBefore(IDENTIFIER, EnumSet.of(SEMICOLON, LBRACK))
193                                .skipToWithNesting(SEMICOLON, LBRACK, RBRACK)
194                                .endNodeWithName(0);
195                simpleTypedefAlternative.markStart().sequence(IDENTIFIER, RPAREN)
196                                .skipTo(SEMICOLON).endNodeWithName(0);
197
198                // enum (both anonymous and named)
199                RecognizerBase<EGenericParserStates> enumAlternative = inAnyState()
200                                .sequence(ENUM);
201                enumAlternative.sequence(LBRACE).createNode(EShallowEntityType.TYPE, 0)
202                                .skipTo(RBRACE).optional(SEMICOLON).endNode();
203                enumAlternative.sequence(IDENTIFIER, EnumSet.of(LBRACE, COLON))
204                                .createNode(EShallowEntityType.TYPE, 0, 1).skipTo(RBRACE)
205                                .optional(SEMICOLON).endNode();
206
207                // types; we have to ensure when skipping to the LBRACE, that there is
208                // no earlier SEMICOLON or EQ, as in these cases it is a forward
209                // declaration or a variable.
210                inAnyState().sequence(getTypeKeywords(), getValidIdentifiers())
211                                .skipBefore(EnumSet.of(SEMICOLON, LBRACE, EQ)).sequence(LBRACE)
212                                .createNode(EShallowEntityType.TYPE, 0, 1).parseUntil(IN_TYPE)
213                                .sequence(RBRACE).optional(SEMICOLON).endNode();
214
215                // anonymous types
216                inAnyState().sequence(getTypeKeywords(), LBRACE)
217                                .createNode(EShallowEntityType.TYPE, 0, "<anonymous>")
218                                .parseUntil(IN_TYPE).sequence(RBRACE).optional(SEMICOLON)
219                                .endNode();
220        }

234        /**
235         * Creates a new recognizer that can match a scope prefix for a method-like
236         * construct. This includes sequences of identifiers with double colon,
237         * possibly intermixed with template arguments.
238         */
239        /* package */RecognizerBase<EGenericParserStates> createScopeRecognizer() {
240                // remember the start of the recognizer chain (we can not used the
241                // result of the method chain, as this would be the last recognizer!)
242                RecognizerBase<EGenericParserStates> result = emptyRecognizer();
243                result.sequence(IDENTIFIER).skipNested(LT, GT).sequence(SCOPE);
244                return result;
245        }

247        /** {@inheritDoc} */
248        @Override
249        protected void createCaseRule() {
250                super.createCaseRule();
251
252                // C/C++ also allows parentheses here and type casts (hence two sets of
253                // nested parentheses).
254                inState(IN_METHOD).markStart().sequence(CASE).skipTo(COLON)
255                                .createNode(EShallowEntityType.META, 0).endNode();
256        }
257
258        /** {@inheritDoc} */
259        @Override
260        protected void createSimpleStatementRule() {
261
262                EnumSet<ETokenType> separators = EnumSet.of(LBRACE, RBRACE);
263                separators.addAll(ETokenType.KEYWORDS);
264
265                inState(IN_METHOD).sequence(new UppercaseIdentifierMatcher())
266                                .skipNested(LPAREN, RPAREN).optional(PREPROCESSOR_DIRECTIVE)
267                                .sequenceBefore(separators)
268                                .createNode(EShallowEntityType.STATEMENT, "Expanded macro", 0)
269                                .endNode();
270
271                super.createSimpleStatementRule();
272        }

294        /** {@inheritDoc} */
295        @Override
296        protected RecognizerBase<EGenericParserStates> typePattern(
297                        RecognizerBase<EGenericParserStates> currentState) {
298
299                EnumSet<ETokenType> extendedTypeKeywords = EnumSet
300                                .copyOf(getTypeKeywords());
301                extendedTypeKeywords.add(TYPENAME);
302
303                EnumSet<ETokenType> typeOrIdentifier = EnumSet.of(IDENTIFIER);
304                typeOrIdentifier.addAll(PRIMITIVE_TYPES);
305
306                return currentState
307                                .repeated(EnumSet.of(CONST, STATIC, VIRTUAL, EXTERN, NEAR, FAR))
308                                .optional(extendedTypeKeywords)
309                                .subRecognizer(createScopeRecognizer(), 0, Integer.MAX_VALUE)
310                                .sequence(typeOrIdentifier)
311                                .skipNested(LT, GT)
312                                .skipAny(EnumSet.of(MULT, AND, CONST))
313                                .skipNested(LBRACK, RBRACK)
314                                .skipAny(
315                                                EnumSet.of(MULT, AND, CONST, LBRACK, RBRACK, NEAR, FAR));
316        }

318        /** {@inheritDoc} */
319        @Override
320        protected boolean isFilteredToken(IToken token, IToken previousToken) {
321                if (token.getType() == IDENTIFIER
322                                && PSEUDO_KEYWORDS.contains(token.getText())) {
323                        return true;
324                }
325
326                return super.isFilteredToken(token, previousToken);
327        }
328
329        /** {@inheritDoc} */
330        @Override
331        protected List<IToken> filterTokens(List<IToken> tokens) {
332                return filterGCCAttributes(super.filterTokens(tokens));
333        }
334
335        /**
336         * Filters GCC attributes. See e.g.
337         * http://gcc.gnu.org/onlinedocs/gcc/Type-Attributes.html
338         */
339        private List<IToken> filterGCCAttributes(List<IToken> tokens) {
340                List<IToken> result = new ArrayList<>();
341                boolean inAttribute = false;
342                int openBraces = 0;
343                for (int i = 0; i < tokens.size(); i++) {
344                        IToken token = tokens.get(i);
345                        if (token.getText().equals("__attribute__")) {
346                                inAttribute = true;
347                        } else if (inAttribute && token.getType() == ETokenType.LPAREN) {
348                                openBraces++;
349                        } else if (inAttribute && token.getType() == ETokenType.RPAREN) {
350                                openBraces--;
351                                if (openBraces == 0) {
352                                        inAttribute = false;
353                                }
354                        } else if (!inAttribute) {
355                                result.add(token);
356                        }
357                }
358                return result;
359        }
