108/**
109 * Shallow parser for C/C++14.
110 * <p>
111 * What this parser does and does not:
112 * <ul>
113 * <li>The parser recognizes types (classes, enums, interfaces), methods and
114 * attributes, individual statements and lambdas.</li>
115 * <li>It recognizes the nesting of statements (e.g. in loops), but does not
116 * parse into the statements. For example, it recognizes an if-statement and
117 * provides the list of sub-statements, but does not provide direct access to
118 * the if-condition.</li>
119 * <li>All preprocessor statements are parsed as meta information.</li>
120 * <li>Template declarations are parsed as preceding meta information.</li>
121 * <li>Forward declarations are handled as meta information.</li>
122 * <li>We heuristically filter code generating macros, such as
123 * "CREATE_STUFF(MyClass)".</li>
124 * </ul>
125 */
126public class CppShallowParser extends CStyleShallowParserBase {

129        protected static final EnumSet<ETokenType> PRIMITIVE_TYPES = EnumSet.of(VOID, BYTE, SHORT, INT, LONG, FLOAT, DOUBLE,
130                        SIGNED, UNSIGNED, CHAR, BOOL);

136        private static final EnumSet<ETokenType> LAMBDA_MODIFIERS = EnumSet.of(CONSTEXPR, MUTABLE);

142        private static final EnumSet<ETokenType> TYPE_OR_IDENTIFIER = EnumSet.copyOf(BASE_TYPE_OR_IDENTIFIER);

154        /**
155         * Set of common "keywords" that are not actually part of the language but
156         * rather used by certain compilers and implicitly defined using macros. The
157         * solution used here is to filter them out.
158         */
159        private static final Set<String> PSEUDO_KEYWORDS = new HashSet<String>(Arrays.asList(
160                        // typically found in Windows compilers
161                        "__fastcall", "__export", "__forceinline", "_cdecl", "_stdcall", "__stdcall", "WINAPI", "APIENTRY",
162                        "CALLBACK",
163                        // used by the common Qt library
164                        "Q_EXPORT",
165                        // keywords found in ISA dialog manager
166                        "DML_c", "DM_CALLBACK", "__1", "__2", "__3", "__4", "__5", "__6", "__7", "DM_ENTRY", "DML_pascal",
167                        "DML_default",
168                        // project specific keywords
169                        "IGVPWORD_API"));

189        /** Creates namespace specific rules. */
190        private void createNamespaceRules() {
191                // using
192                inAnyState().sequence(USING).createNode(EShallowEntityType.META, 0).skipTo(SEMICOLON).endNode();
193
194                // namespace
195                RecognizerBase<EGenericParserStates> namespaceAlternative = inAnyState().sequence(NAMESPACE)
196                                .skipBefore(EnumSet.of(SEMICOLON, LBRACE));
197                namespaceAlternative.sequence(LBRACE).createNode(EShallowEntityType.MODULE, 0, new Region(1, -2))
198                                .parseUntil(TOP_LEVEL).sequence(RBRACE).endNode();
199                namespaceAlternative.sequence(SEMICOLON).createNode(EShallowEntityType.META, 0, new Region(1, -2)).endNode();
200        }

202        /** {@inheritDoc} */
203        @Override
204        protected void createTypeRules() {
205                // typedef
206                RecognizerBase<EGenericParserStates> typeInTypedefAlternative = inAnyState().sequence(TYPEDEF).optional(CONST);
207                typeInTypedefAlternative.sequenceBefore(getTypeKeywords(), IDENTIFIER, LBRACE)
208                                .createNode(EShallowEntityType.TYPE, 0).parseOnce(TOP_LEVEL).skipTo(IDENTIFIER, SEMICOLON)
209                                .endNodeWithName(-2);
210                typeInTypedefAlternative.sequenceBefore(getTypeKeywords(), LBRACE).createNode(EShallowEntityType.TYPE, 0)
211                                .parseOnce(TOP_LEVEL).skipTo(IDENTIFIER, SEMICOLON).endNodeWithName(-2);
212
213                RecognizerBase<EGenericParserStates> simpleTypedefAlternative = inAnyState().sequence(TYPEDEF)
214                                .createNode(EShallowEntityType.TYPE, 0).skipBefore(IDENTIFIER, EnumSet.of(SEMICOLON, RPAREN, LBRACK));
215                simpleTypedefAlternative.markStart().sequenceBefore(IDENTIFIER, EnumSet.of(SEMICOLON, LBRACK))
216                                .skipToWithNesting(SEMICOLON, LBRACK, RBRACK).endNodeWithName(0);
217                simpleTypedefAlternative.markStart().sequence(IDENTIFIER, RPAREN).skipTo(SEMICOLON).endNodeWithName(0);
218
219                createEnumRules();
220
221                // types; we have to ensure when skipping to the LBRACE, that there is
222                // no earlier SEMICOLON or EQ, as in these cases it is a forward
223                // declaration or a variable.
224                inAnyState().sequence(getTypeKeywords(), getValidIdentifiers()).skipBefore(EnumSet.of(SEMICOLON, LBRACE, EQ))
225                                .sequence(LBRACE).createNode(EShallowEntityType.TYPE, 0, 1).parseUntil(IN_TYPE).sequence(RBRACE)
226                                .optional(SEMICOLON).endNode();
227
228                // anonymous types
229                inAnyState().sequence(getTypeKeywords(), LBRACE).createNode(EShallowEntityType.TYPE, 0, "<anonymous>")
230                                .parseUntil(IN_TYPE).sequence(RBRACE).optional(SEMICOLON).endNode();
231        }
232
233        /** Creates rules for parsing enums and enum classes. */
234        private void createEnumRules() {
235                // enum (both anonymous and named)
236                finishEnumDeclaration(inAnyState().sequence(ENUM), SubTypeNames.ENUM, 1);
237                finishEnumDeclaration(inAnyState().sequence(ENUM, CLASS), SubTypeNames.ENUM_CLASS, 2);
238
239                RecognizerBase<EGenericParserStates> enumLiteralAlternative = inState(IN_TYPE).sequence(IDENTIFIER);
240                finishEnumLiteral(enumLiteralAlternative.sequenceBefore(EnumSet.of(COMMA, RBRACE)));
241                finishEnumLiteral(enumLiteralAlternative.sequence(EQ).skipBefore(EnumSet.of(COMMA, RBRACE)));
242        }
243
244        /** Finishes the rule for enum or enum class declarations. */
245        private static void finishEnumDeclaration(RecognizerBase<EGenericParserStates> enumAlternative, String subType,
246                        Object name) {
247                enumAlternative.sequence(LBRACE).createNode(EShallowEntityType.TYPE, subType).parseUntil(IN_TYPE)
248                                .sequence(RBRACE).optional(SEMICOLON).endNode();
249                enumAlternative.sequence(IDENTIFIER).skipTo(LBRACE).createNode(EShallowEntityType.TYPE, subType, name)
250                                .parseUntil(IN_TYPE).sequence(RBRACE).optional(SEMICOLON).endNode();
251        }
252
253        /** Finishes the rule for a enum literal. */
254        private static void finishEnumLiteral(RecognizerBase<EGenericParserStates> enumLiteralAlternative) {
255                enumLiteralAlternative.createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.ENUM_LITERAL, 0).endNode();
256        }

279        /**
280         * Creates a new recognizer that can match a scope prefix for a method-like
281         * construct. This includes sequences of identifiers with double colon,
282         * possibly intermixed with template arguments.
283         */
284        /* package */RecognizerBase<EGenericParserStates> createScopeRecognizer() {
285                // remember the start of the recognizer chain (we can not used the
286                // result of the method chain, as this would be the last recognizer)
287                RecognizerBase<EGenericParserStates> result = emptyRecognizer();
288                result.sequence(IDENTIFIER).subRecognizer(new CppSkipTemplateSpecificationRecognizer(), 0, 1).sequence(SCOPE);
289                return result;
290        }

292        /** {@inheritDoc} */
293        @Override
294        protected void createCaseRule() {
295                super.createCaseRule();
296
297                // C/C++ also allows parentheses here and type casts (hence two sets of
298                // nested parentheses).
299                inState(IN_METHOD).markStart().sequence(CASE).skipTo(COLON).createNode(EShallowEntityType.META, 0).endNode();
300        }

302        /** {@inheritDoc} */
303        @Override
304        protected void createSimpleStatementRule() {
305
306                EnumSet<ETokenType> separators = EnumSet.of(LBRACE, RBRACE);
307                separators.addAll(ETokenType.KEYWORDS);
308
309                inState(IN_METHOD).sequence(new UppercaseIdentifierMatcher()).skipNested(LPAREN, RPAREN)
310                                .optional(PREPROCESSOR_DIRECTIVE).sequenceBefore(separators)
311                                .createNode(EShallowEntityType.STATEMENT, "Expanded macro", 0).endNode();
312
313                super.createSimpleStatementRule();
314        }

335        /** {@inheritDoc} */
336        @Override
337        protected RecognizerBase<EGenericParserStates> typePattern(RecognizerBase<EGenericParserStates> currentState) {
338
339                EnumSet<ETokenType> extendedTypeKeywords = EnumSet.copyOf(getTypeKeywords());
340                extendedTypeKeywords.add(TYPENAME);
341
342                return currentState.repeated(getMethodAndTypeDeclarationModifiers()).skipNested(LPAREN, RPAREN)
343                                .optional(extendedTypeKeywords).skipNested(LPAREN, RPAREN)
344                                .subRecognizer(createScopeRecognizer(), 0, Integer.MAX_VALUE).optional(EnumSet.of(SIGNED, UNSIGNED))
345                                .sequence(getTypeOrIdentifier()).skipNested(LPAREN, RPAREN)
346                                .subRecognizer(new CppSkipTemplateSpecificationRecognizer(), 0, 1)
347                                .skipAny(EnumSet.of(MULT, AND, ANDAND, CONST, CONSTEXPR)).skipNested(LBRACK, RBRACK)
348                                .skipAny(EnumSet.of(MULT, AND, ANDAND, CONST, CONSTEXPR, LBRACK, RBRACK, NEAR, FAR));
349        }

351        /** {@inheritDoc} */
352        @Override
353        protected void createSubExpressionRules() {
354                RecognizerBase<EGenericParserStates> lambdaAlternative = inState(IN_EXPRESSION).skipNested(LBRACK, RBRACK)
355                                .skipNested(LPAREN, RPAREN).repeated(LAMBDA_MODIFIERS).optional(EnumSet.of(NOEXCEPT, THROW))
356                                .skipNested(LPAREN, RPAREN);
357
358                finishLambdaRule(lambdaAlternative.sequence(POINTERTO).repeated(getTypeKeywords())
359                                .optional(EnumSet.of(SIGNED, UNSIGNED)).optional(getTypeOrIdentifier()).skipNested(LPAREN, RPAREN));
360                finishLambdaRule(lambdaAlternative);
361
362        }

376        /** {@inheritDoc} */
377        @Override
378        protected boolean isFilteredToken(IToken token, IToken previousToken) {
379                if (token.getType() == IDENTIFIER && PSEUDO_KEYWORDS.contains(token.getText())) {
380                        return true;
381                }
382
383                return super.isFilteredToken(token, previousToken);
384        }

392        /**
393         * Filters GCC attributes. See e.g.
394         * http://gcc.gnu.org/onlinedocs/gcc/Type-Attributes.html
395         */
396        private static List<IToken> filterGCCAttributes(List<IToken> tokens) {
397                List<IToken> result = new ArrayList<>();
398                boolean inAttribute = false;
399                int openBraces = 0;
400                for (int i = 0; i < tokens.size(); i++) {
401                        IToken token = tokens.get(i);
402                        if (token.getText().equals("__attribute__")) {
403                                inAttribute = true;
404                        } else if (inAttribute && token.getType() == LPAREN) {
405                                openBraces++;
406                        } else if (inAttribute && token.getType() == RPAREN) {
407                                openBraces--;
408                                if (openBraces == 0) {
409                                        inAttribute = false;
410                                }
411                        } else if (!inAttribute) {
412                                result.add(token);
413                        }
414                }
415                return result;
416        }
417
418        /**
419         * Filters C++11 attributes. See e.g.
420         * http://en.cppreference.com/w/cpp/language/attributes
421         */
422        private static List<IToken> filterCpp11Attributes(List<IToken> tokens) {
423                List<IToken> result = new ArrayList<>();
424                boolean inAttribute = false;
425                for (int i = 0; i < tokens.size(); i++) {
426                        ETokenType searchType = LBRACK;
427                        if (inAttribute) {
428                                searchType = RBRACK;
429                        }
430
431                        if (TokenStreamUtils.tokenTypesAt(tokens, i, searchType, searchType)) {
432                                inAttribute = !inAttribute;
433                                i++;
434                        } else if (!inAttribute) {
435                                result.add(tokens.get(i));
436                        }
437                }
438                return result;
439        }
