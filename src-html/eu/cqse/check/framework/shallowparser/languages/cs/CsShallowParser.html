038/**
039 * Shallow parser for C#
040 * <p>
041 * What this parser does and does not:
042 * <ul>
043 * <li>The parser recognizes types (classes, enums, interfaces), methods and
044 * attributes, and individual statements.</li>
045 * <li>It recognizes the nesting of statements (e.g. in loops), but does not
046 * parse into the statements. For example, it recognizes an if-statement and
047 * provides the list of sub-statements, but does not provide direct access to
048 * the if-condition.</li>
049 * <li>Using statements and annotations are parsed as meta information.</li>
050 * </ul>
051 *
052 * @author $Author: kinnen $
053 * @version $Rev: 54082 $
054 * @ConQAT.Rating GREEN Hash: 65EBC383EB1163BAE88C930FD8748286
055 */
056public class CsShallowParser extends CStyleShallowParserBase {
057
058        /**
059         * A set of all token types that can be used as valid identifiers. See
060         * http://msdn.microsoft.com/en-us/library/x53a06bb.aspx for the full list.
061         */
062        public static final EnumSet<ETokenType> VALID_IDENTIFIERS = EnumSet.of(
063                        IDENTIFIER, ADD, ALIAS, ASCENDING, ASYNC, AWAIT, DESCENDING,
064                        DYNAMIC, FROM, GET, GLOBAL, GROUP, INTO, JOIN, LET, ORDERBY,
065                        PARTIAL, REMOVE, SELECT, SET, VALUE, VAR, WHERE, YIELD);
066
067        /** All primitive types. */
068        private static final EnumSet<ETokenType> PRIMITIVE_TYPES = EnumSet.of(VOID,
069                        BYTE, SHORT, INT, LONG, FLOAT, DOUBLE, CHAR, BOOL, STRING, OBJECT,
070                        DECIMAL, SBYTE, USHORT, UINT, ULONG);
071
072        /** {@inheritDoc} */
073        @Override
074        protected void createMetaRules() {
075                // using
076                inState(TOP_LEVEL, IN_MODULE).markStart().sequence(USING)
077                                .optional(STATIC).markStart().skipTo(SEMICOLON)
078                                .createNode(EShallowEntityType.META, SubTypeNames.USING,
079                                                new Region(0, -2))
080                                .endNode();
081
082                // annotations
083                inState(IN_TYPE, IN_MODULE, TOP_LEVEL).sequence(LBRACK)
084                                .createNode(EShallowEntityType.META, "annotation")
085                                .skipToWithNesting(RBRACK, LBRACK, RBRACK).endNode();
086
087                // preprocessor stuff
088                inAnyState().sequence(PREPROCESSOR_DIRECTIVE)
089                                .createNode(EShallowEntityType.META, 0).endNode();
090
091                super.createMetaRules();
092        }

094        /** {@inheritDoc} */
095        @Override
096        protected void createTypeRules() {
097                // namespace
098                inState(TOP_LEVEL, IN_MODULE).sequence(NAMESPACE, getValidIdentifiers())
099                                .skipTo(LBRACE)
100                                .createNode(EShallowEntityType.MODULE, 0, new Region(1, -2))
101                                .parseUntil(IN_MODULE).sequence(RBRACE).endNode();
102
103                super.createTypeRules();
104        }

119        /** {@inheritDoc} */
120        @Override
121        protected void createClassElementsRules() {
122                // simple enum literals
123                inState(IN_TYPE).sequence(IDENTIFIER)
124                                .sequenceBefore(EnumSet.of(COMMA, EQ, RBRACE))
125                                .createNode(EShallowEntityType.ATTRIBUTE, "enum literal", 0)
126                                .skipBefore(EnumSet.of(COMMA, RBRACE)).optional(COMMA)
127                                .endNode();
128
129                // delegates
130                typePattern(inState(TOP_LEVEL, IN_MODULE, IN_TYPE).sequence(DELEGATE))
131                                .sequence(getValidIdentifiers(), LPAREN)
132                                .createNode(EShallowEntityType.METHOD, 0, -2).skipTo(RPAREN)
133                                .skipTo(SEMICOLON).endNode();
134
135                createMethodRules();
136                createPropertiesRules();
137                createEventsRules();
138
139                // attributes, e.g., fields (must be after method, as this would also
140                // match methods)
141                typePatternInState(IN_TYPE).sequence(getValidIdentifiers())
142                                .createNode(EShallowEntityType.ATTRIBUTE, "attribute", -1)
143                                .skipToWithNesting(SEMICOLON, LBRACE, RBRACE,
144                                                getSubExpressionRecognizer())
145                                .endNode();
146
147                // static initializer, get/set for properties, add/remove in events
148                inState(IN_TYPE).sequence(EnumSet.of(GET, SET), SEMICOLON)
149                                .createNode(EShallowEntityType.METHOD,
150                                                new Object[] { "empty", 0 })
151                                .endNode();
152                inState(IN_TYPE)
153                                .sequence(EnumSet.of(STATIC, GET, SET, ADD, REMOVE), LBRACE)
154                                .createNode(EShallowEntityType.METHOD, 0).parseUntil(IN_METHOD)
155                                .sequence(RBRACE).endNode();
156        }

158        /** Creates the rules for parsing events. */
159        private void createEventsRules() {
160                RecognizerBase<EGenericParserStates> eventRecognizer = inState(
161                                TOP_LEVEL, IN_MODULE, IN_TYPE).repeated(
162                                                EnumSet.of(PRIVATE, PROTECTED, PUBLIC, INTERNAL))
163                                                .markStart().sequence(EVENT)
164                                                .sequence(getValidIdentifiers()).skipNested(LT, GT)
165                                                .subRecognizer(
166                                                                createExplicitInterfaceQualifierRecognizer(), 0,
167                                                                Integer.MAX_VALUE)
168                                                .sequence(getValidIdentifiers())
169                                                .createNode(EShallowEntityType.ATTRIBUTE, 0, -1);
170                eventRecognizer.sequence(LBRACE).parseUntil(IN_TYPE).sequence(RBRACE)
171                                .endNode();
172                eventRecognizer.skipTo(SEMICOLON).endNode();
173        }
174
175        /** Creates the rules for parsing properties. */
176        private void createPropertiesRules() {
177                RecognizerBase<EGenericParserStates> alternatives = typePatternInState(
178                                IN_TYPE).subRecognizer(
179                                                createExplicitInterfaceQualifierRecognizer(), 0,
180                                                Integer.MAX_VALUE).sequence(getValidIdentifiers());
181
182                alternatives.sequence(DOUBLE_ARROW)
183                                .createNode(EShallowEntityType.ATTRIBUTE, "property", -2)
184                                .parseOnce(IN_METHOD).endNode();
185
186                RecognizerBase<EGenericParserStates> braceAlternatives = alternatives
187                                .sequence(LBRACE)
188                                .createNode(EShallowEntityType.ATTRIBUTE, "property", -2)
189                                .parseUntil(IN_TYPE).sequence(RBRACE);
190                braceAlternatives.sequence(EQ).parseOnce(IN_METHOD).endNode();
191                braceAlternatives.endNode();
192        }

194        /**
195         * Creates the rules for all method-like constructs inside types.
196         */
197        private void createMethodRules() {
198                // indexers
199                completeMethod("indexer", EShallowEntityType.ATTRIBUTE, IN_TYPE,
200                                typePatternInState(IN_TYPE)
201                                                .subRecognizer(
202                                                                createExplicitInterfaceQualifierRecognizer(), 0,
203                                                                Integer.MAX_VALUE)
204                                                .markStart().sequence(THIS, LBRACK).skipTo(RBRACK));
205
206                // operator overloading
207                createOperatorOverloadingRules();
208
209                // methods
210                completeMethod("method", EShallowEntityType.METHOD, IN_METHOD,
211                                typePatternInState(IN_TYPE)
212                                                .subRecognizer(
213                                                                createExplicitInterfaceQualifierRecognizer(), 0,
214                                                                Integer.MAX_VALUE)
215                                                .markStart().sequence(getValidIdentifiers())
216                                                .skipNested(LT, GT).sequence(LPAREN).skipTo(RPAREN));
217
218                // constructor
219                finishConstructorLike(inState(IN_TYPE)
220                                .repeated(EnumSet.of(PRIVATE, PROTECTED, PUBLIC, INTERNAL))
221                                .markStart(), "constructor");
222
223                // static constructor
224                finishConstructorLike(inState(IN_TYPE).sequence(STATIC).markStart(),
225                                SubTypeNames.STATIC_CONSTRUCTOR);
226
227                // destructor
228                finishConstructorLike(inState(IN_TYPE)
229                                .repeated(EnumSet.of(PRIVATE, PROTECTED, PUBLIC, INTERNAL))
230                                .sequence(COMP).markStart(), "destructor");
231        }
232
233        /** Rules for operator overloading. */
234        private void createOperatorOverloadingRules() {
235                EnumSet<ETokenType> primitiveOrIdentifier = EnumSet
236                                .copyOf(PRIMITIVE_TYPES);
237                primitiveOrIdentifier.add(IDENTIFIER);
238
239                completeMethod("operator", EShallowEntityType.METHOD, IN_METHOD,
240                                inState(IN_TYPE)
241                                                .repeated(EnumSet.of(PRIVATE, PROTECTED, PUBLIC,
242                                                                INTERNAL, STATIC))
243                                                .sequence(IMPLICIT, OPERATOR).markStart()
244                                                .sequence(primitiveOrIdentifier, LPAREN)
245                                                .skipTo(RPAREN));
246                completeMethod("operator", EShallowEntityType.METHOD, IN_METHOD,
247                                typePatternInState(IN_TYPE).sequence(OPERATOR).markStart()
248                                                .sequence(EnumSet.of(ETokenClass.OPERATOR,
249                                                                ETokenClass.KEYWORD), LPAREN)
250                                                .skipTo(RPAREN));
251        }

253        /**
254         * Finishes a recognizer that begins parsing a constructor like method.
255         * Those are constructors, static constructors and destructors. The
256         * corresponding subtype must be passed.
257         */
258        private void finishConstructorLike(
259                        RecognizerBase<EGenericParserStates> recognizer, String subtype) {
260                recognizer.sequence(getValidIdentifiers(), LPAREN).skipTo(RPAREN)
261                                .skipToWithNesting(LBRACE, LPAREN, RPAREN)
262                                .createNode(EShallowEntityType.METHOD, subtype, 0)
263                                .parseUntil(IN_METHOD).sequence(RBRACE).endNode();
264        }

266        /**
267         * Creates a new recognizer that can match an explicit interface qualifier
268         * prefix for a method-like construct. This includes sequences of
269         * identifiers with dots, possibly intermixed with template arguments.
270         */
271        private RecognizerBase<EGenericParserStates> createExplicitInterfaceQualifierRecognizer() {
272                // remember the start of the recognizer chain (we can not use the
273                // result of the method chain, as this would be the last recognizer)
274                RecognizerBase<EGenericParserStates> result = emptyRecognizer();
275                result.sequence(getValidIdentifiers()).skipNested(LT, GT).sequence(DOT);
276                return result;
277        }

279        /**
280         * Completes a method-like construct. This begins with searching for the
281         * first semicolon or brace, i.e., the parameter list should already be
282         * skipped. This ends either in a complete method with a body, or with a
283         * semicolon and thus is just an abstract method.
284         */
285        private static void completeMethod(String name, EShallowEntityType nodeType,
286                        EGenericParserStates subParseState,
287                        RecognizerBase<EGenericParserStates> start) {
288                RecognizerBase<EGenericParserStates> alternative = start
289                                .skipBefore(EnumSet.of(LBRACE, SEMICOLON, DOUBLE_ARROW));
290
291                // for lambdas, we always parse IN_METHOD and ignore the subParseState
292                RecognizerBase<EGenericParserStates> lambdaAlternative = alternative
293                                .sequence(DOUBLE_ARROW).createNode(nodeType, name, 0);
294                lambdaAlternative.sequence(LBRACE).parseUntil(IN_METHOD)
295                                .sequence(RBRACE).endNode();
296                lambdaAlternative.parseOnce(IN_METHOD).endNode();
297
298                alternative.sequence(LBRACE).createNode(nodeType, name, 0)
299                                .parseUntil(subParseState).sequence(RBRACE).endNode();
300                alternative.sequence(SEMICOLON)
301                                .createNode(nodeType, "abstract " + name, 0).endNode();
302        }

304        /** {@inheritDoc} */
305        @Override
306        protected void createCaseRule() {
307                super.createCaseRule();
308
309                // C# also allows any kind of constant expression as a case label, e.g:
310                // Foo.BAR + Foo.GOO << 12
311                inState(IN_METHOD).markStart().sequence(CASE).skipTo(COLON)
312                                .createNode(EShallowEntityType.META, 0).endNode();
313        }

347        /** {@inheritDoc} */
348        @Override
349        protected RecognizerBase<EGenericParserStates> typePattern(
350                        RecognizerBase<EGenericParserStates> currentState) {
351                EnumSet<ETokenType> modifierKeywords = EnumSet.of(PRIVATE, PROTECTED,
352                                PUBLIC, INTERNAL, VIRTUAL, ABSTRACT, ASYNC, CONST, EXTERN,
353                                OVERRIDE, READONLY, UNSAFE, VOLATILE, NEW, STATIC);
354                EnumSet<ETokenType> typeStart = EnumSet.copyOf(PRIMITIVE_TYPES);
355                typeStart.addAll(getValidIdentifiers());
356
357                // we include "?" in the skipping section to deal with nullable types
358                // (e.g. Foo? foo;)
359                // the repeated (DOT, typeStart) is used for full qualified type names
360                return currentState.repeated(modifierKeywords).sequence(typeStart)
361                                .repeated(DOT, typeStart).optional(QUESTION).skipNested(LT, GT)
362                                .subRecognizer(new ArrayBracketsRecognizer(), 0,
363                                                Integer.MAX_VALUE);
364        }

366        /** {@inheritDoc} */
367        @Override
368        protected void createSubExpressionRules() {
369                // anonymous delegate methods
370                inState(IN_EXPRESSION).sequence(DELEGATE, LPAREN)
371                                .createNode(EShallowEntityType.METHOD, "anonymous method")
372                                .skipToWithNesting(RPAREN, LPAREN, RPAREN).sequence(LBRACE)
373                                .parseUntil(IN_METHOD).sequence(RBRACE).endNode();
374
375                // lambda expressions
376                completeLambda(inState(IN_EXPRESSION).sequence(getValidIdentifiers()));
377                completeLambda(inState(IN_EXPRESSION).sequence(LPAREN).skipTo(RPAREN));
378
379                // additional rule for parsing lambda expressions (without braces). see
380                // completeLambda() for details
381                // the node start is moved one token to the right so the shallow
382                // entities produced by this rule don't include the double arrow
383                // (instead it will be included in the parent entity)
384                inState(IN_EXPRESSION).sequence(DOUBLE_ARROW)
385                                .createNode(EShallowEntityType.STATEMENT,
386                                                SubTypeNames.LAMBDA_EXPRESSION, null, 1)
387                                .skipBeforeWithNesting(
388                                                EnumSet.of(RPAREN, SEMICOLON, RBRACE, COMMA), LPAREN,
389                                                RPAREN, LBRACE, RBRACE)
390                                .endNode();
391        }
392
393        /** Completes a rule for parsing lambda expressions. */
394        private static void completeLambda(
395                        RecognizerBase<EGenericParserStates> ruleStart) {
396                RecognizerBase<EGenericParserStates> lambdaAlternative = ruleStart
397                                .createNode(EShallowEntityType.METHOD, "lambda");
398                lambdaAlternative.sequence(DOUBLE_ARROW, LBRACE).parseUntil(IN_METHOD)
399                                .sequence(RBRACE).endNode();
400
401                // we start parsing before the double arrow (=>) as this allows our
402                // special statement rule to capture this case. This is required, as
403                // this kind of expression is not terminated by a semicolon.
404                lambdaAlternative.sequenceBefore(DOUBLE_ARROW).parseOnce(IN_EXPRESSION)
405                                .endNode();
406        }

411                return new CsDelegateAndLambdaRecognizer();
