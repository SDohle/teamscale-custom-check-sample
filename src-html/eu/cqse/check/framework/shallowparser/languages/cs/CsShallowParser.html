039/**
040 * Shallow parser for C#
041 * <p>
042 * What this parser does and does not:
043 * <ul>
044 * <li>The parser recognizes types (classes, enums, interfaces), methods and
045 * attributes, and individual statements.</li>
046 * <li>It recognizes the nesting of statements (e.g. in loops), but does not
047 * parse into the statements. For example, it recognizes an if-statement and
048 * provides the list of sub-statements, but does not provide direct access to
049 * the if-condition.</li>
050 * <li>Using statements and annotations are parsed as meta information.</li>
051 * </ul>
052 */
053public class CsShallowParser extends CStyleShallowParserBase {
054
055        /**
056         * A set of all token types that can be used as valid identifiers. See
057         * http://msdn.microsoft.com/en-us/library/x53a06bb.aspx for the full list.
058         */
059        public static final EnumSet<ETokenType> VALID_IDENTIFIERS = EnumSet.of(IDENTIFIER, ADD, ALIAS, ASCENDING, ASYNC,
060                        AWAIT, DESCENDING, DYNAMIC, FROM, GET, GLOBAL, GROUP, INTO, JOIN, LET, ORDERBY, PARTIAL, REMOVE, SELECT,
061                        SET, VALUE, VAR, WHERE, YIELD);
062
063        /** All primitive types. */
064        private static final EnumSet<ETokenType> PRIMITIVE_TYPES = EnumSet.of(VOID, BYTE, SHORT, INT, LONG, FLOAT, DOUBLE,
065                        CHAR, BOOL, STRING, OBJECT, DECIMAL, SBYTE, USHORT, UINT, ULONG);
066
067        /** {@inheritDoc} */
068        @Override
069        protected void createMetaRules() {
070                // using
071                inState(TOP_LEVEL, IN_MODULE).markStart().sequence(USING).optional(STATIC).markStart().skipTo(SEMICOLON)
072                                .createNode(EShallowEntityType.META, SubTypeNames.USING, new Region(0, -2)).endNode();
073
074                // annotations
075                inState(IN_TYPE, IN_MODULE, TOP_LEVEL).sequence(LBRACK).createNode(EShallowEntityType.META, "annotation")
076                                .skipToWithNesting(RBRACK, LBRACK, RBRACK).endNode();
077
078                // preprocessor stuff
079                inAnyState().sequence(PREPROCESSOR_DIRECTIVE).createNode(EShallowEntityType.META, 0).endNode();
080
081                super.createMetaRules();
082        }

084        /** {@inheritDoc} */
085        @Override
086        protected void createTypeRules() {
087                // namespace
088                inState(TOP_LEVEL, IN_MODULE).sequence(NAMESPACE, getValidIdentifiers()).skipTo(LBRACE)
089                                .createNode(EShallowEntityType.MODULE, 0, new Region(1, -2)).parseUntil(IN_MODULE).sequence(RBRACE)
090                                .endNode();
091
092                super.createTypeRules();
093        }

107        /** {@inheritDoc} */
108        @Override
109        protected void createClassElementsRules() {
110                // simple enum literals
111                inState(IN_TYPE).sequence(IDENTIFIER).sequenceBefore(EnumSet.of(COMMA, EQ, RBRACE))
112                                .createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.ENUM_LITERAL, 0)
113                                .skipBefore(EnumSet.of(COMMA, RBRACE)).optional(COMMA).endNode();
114
115                // delegates
116                typePattern(inState(TOP_LEVEL, IN_MODULE, IN_TYPE).sequence(DELEGATE)).sequence(getValidIdentifiers(), LPAREN)
117                                .createNode(EShallowEntityType.METHOD, 0, -2).skipTo(RPAREN).skipTo(SEMICOLON).endNode();
118
119                createMethodRules();
120                createPropertiesRules();
121                createEventsRules();
122
123                // attributes, e.g., fields (must be after method, as this would also
124                // match methods)
125                typePatternInState(IN_TYPE).sequence(getValidIdentifiers())
126                                .createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.ATTRIBUTE, -1)
127                                .skipToWithNesting(SEMICOLON, LBRACE, RBRACE, getSubExpressionRecognizer()).endNode();
128
129                // static initializer, get/set for properties, add/remove in events
130                inState(IN_TYPE).sequence(EnumSet.of(GET, SET), SEMICOLON).createNode(EShallowEntityType.METHOD,
131                                new Object[] { "empty", 0 }, new PropertyAccessNameResolver<EGenericParserStates>()).endNode();
132                inState(IN_TYPE).sequence(EnumSet.of(GET, SET), LBRACE)
133                                .createNode(EShallowEntityType.METHOD, 0, new PropertyAccessNameResolver<EGenericParserStates>())
134                                .parseUntil(IN_METHOD).sequence(RBRACE).endNode();
135                inState(IN_TYPE).sequence(EnumSet.of(STATIC, ADD, REMOVE), LBRACE).createNode(EShallowEntityType.METHOD, 0)
136                                .parseUntil(IN_METHOD).sequence(RBRACE).endNode();
137        }

139        /** Creates the rules for parsing events. */
140        private void createEventsRules() {
141                RecognizerBase<EGenericParserStates> eventRecognizer = inState(TOP_LEVEL, IN_MODULE, IN_TYPE)
142                                .repeated(EnumSet.of(PRIVATE, PROTECTED, PUBLIC, INTERNAL)).markStart().sequence(EVENT)
143                                .sequence(getValidIdentifiers()).skipNested(LT, GT)
144                                .subRecognizer(createExplicitInterfaceQualifierRecognizer(), 0, Integer.MAX_VALUE)
145                                .sequence(getValidIdentifiers()).createNode(EShallowEntityType.ATTRIBUTE, 0, -1);
146                eventRecognizer.sequence(LBRACE).parseUntil(IN_TYPE).sequence(RBRACE).endNode();
147                eventRecognizer.skipTo(SEMICOLON).endNode();
148        }
149
150        /** Creates the rules for parsing properties. */
151        private void createPropertiesRules() {
152                RecognizerBase<EGenericParserStates> alternatives = typePatternInState(IN_TYPE)
153                                .subRecognizer(createExplicitInterfaceQualifierRecognizer(), 0, Integer.MAX_VALUE)
154                                .sequence(getValidIdentifiers());
155
156                alternatives.sequence(DOUBLE_ARROW).createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.PROPERTY, -2)
157                                .parseOnce(IN_METHOD).endNode();
158
159                RecognizerBase<EGenericParserStates> braceAlternatives = alternatives.sequence(LBRACE)
160                                .createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.PROPERTY, -2).parseUntil(IN_TYPE)
161                                .sequence(RBRACE);
162                braceAlternatives.sequence(EQ).parseOnce(IN_METHOD).endNode();
163                braceAlternatives.endNode();
164        }
165
166        /**
167         * Creates the rules for all method-like constructs inside types.
168         */
169        private void createMethodRules() {
170                // indexers
171                completeMethod("indexer", EShallowEntityType.ATTRIBUTE, IN_TYPE,
172                                typePatternInState(IN_TYPE)
173                                                .subRecognizer(createExplicitInterfaceQualifierRecognizer(), 0, Integer.MAX_VALUE).markStart()
174                                                .sequence(THIS, LBRACK).skipTo(RBRACK));
175
176                // operator overloading
177                createOperatorOverloadingRules();
178
179                // methods
180                completeMethod("method", EShallowEntityType.METHOD, IN_METHOD,
181                                typePatternInState(IN_TYPE)
182                                                .subRecognizer(createExplicitInterfaceQualifierRecognizer(), 0, Integer.MAX_VALUE).markStart()
183                                                .sequence(getValidIdentifiers()).skipNested(LT, GT).sequence(LPAREN).skipTo(RPAREN));
184
185                // constructor
186                finishConstructorLike(inState(IN_TYPE).repeated(EnumSet.of(PRIVATE, PROTECTED, PUBLIC, INTERNAL)).markStart(),
187                                SubTypeNames.CONSTRUCTOR);
188
189                // static constructor
190                finishConstructorLike(inState(IN_TYPE).sequence(STATIC).markStart(), SubTypeNames.STATIC_CONSTRUCTOR);
191
192                // destructor
193                finishConstructorLike(
194                                inState(IN_TYPE).repeated(EnumSet.of(PRIVATE, PROTECTED, PUBLIC, INTERNAL)).sequence(COMP).markStart(),
195                                SubTypeNames.DESTRUCTOR);
196        }
197
198        /** Rules for operator overloading. */
199        private void createOperatorOverloadingRules() {
200                EnumSet<ETokenType> primitiveOrIdentifier = EnumSet.copyOf(PRIMITIVE_TYPES);
201                primitiveOrIdentifier.add(IDENTIFIER);
202
203                completeMethod(SubTypeNames.OPERATOR, EShallowEntityType.METHOD, IN_METHOD,
204                                inState(IN_TYPE).repeated(EnumSet.of(PRIVATE, PROTECTED, PUBLIC, INTERNAL, STATIC))
205                                                .sequence(IMPLICIT, OPERATOR).markStart().sequence(primitiveOrIdentifier, LPAREN)
206                                                .skipTo(RPAREN));
207                completeMethod(SubTypeNames.OPERATOR, EShallowEntityType.METHOD, IN_METHOD,
208                                typePatternInState(IN_TYPE).sequence(OPERATOR).markStart()
209                                                .sequence(EnumSet.of(ETokenClass.OPERATOR, ETokenClass.KEYWORD), LPAREN).skipTo(RPAREN));
210        }

212        /**
213         * Finishes a recognizer that begins parsing a constructor like method.
214         * Those are constructors, static constructors and destructors. The
215         * corresponding subtype must be passed.
216         */
217        private void finishConstructorLike(RecognizerBase<EGenericParserStates> recognizer, String subtype) {
218                recognizer.sequence(getValidIdentifiers(), LPAREN).skipTo(RPAREN).skipToWithNesting(LBRACE, LPAREN, RPAREN)
219                                .createNode(EShallowEntityType.METHOD, subtype, 0).parseUntil(IN_METHOD).sequence(RBRACE).endNode();
220        }
221
222        /**
223         * Creates a new recognizer that can match an explicit interface qualifier
224         * prefix for a method-like construct. This includes sequences of
225         * identifiers with dots, possibly intermixed with template arguments.
226         */
227        private RecognizerBase<EGenericParserStates> createExplicitInterfaceQualifierRecognizer() {
228                // remember the start of the recognizer chain (we can not use the
229                // result of the method chain, as this would be the last recognizer)
230                RecognizerBase<EGenericParserStates> result = emptyRecognizer();
231                result.sequence(getValidIdentifiers()).skipNested(LT, GT).sequence(DOT);
232                return result;
233        }
234
235        /**
236         * Completes a method-like construct. This begins with searching for the
237         * first semicolon or brace, i.e., the parameter list should already be
238         * skipped. This ends either in a complete method with a body, or with a
239         * semicolon and thus is just an abstract method.
240         */
241        private static void completeMethod(String name, EShallowEntityType nodeType, EGenericParserStates subParseState,
242                        RecognizerBase<EGenericParserStates> start) {
243                RecognizerBase<EGenericParserStates> alternative = start
244                                .skipBefore(EnumSet.of(LBRACE, SEMICOLON, DOUBLE_ARROW));
245
246                // for lambdas, we always parse IN_METHOD and ignore the subParseState
247                RecognizerBase<EGenericParserStates> lambdaAlternative = alternative.sequence(DOUBLE_ARROW).createNode(nodeType,
248                                name, 0);
249                lambdaAlternative.sequence(LBRACE).parseUntil(IN_METHOD).sequence(RBRACE).endNode();
250                lambdaAlternative.parseOnce(IN_METHOD).endNode();
251
252                alternative.sequence(LBRACE).createNode(nodeType, name, 0).parseUntil(subParseState).sequence(RBRACE).endNode();
253                alternative.sequence(SEMICOLON).createNode(nodeType, "abstract " + name, 0).endNode();
254        }

256        /** {@inheritDoc} */
257        @Override
258        protected void createCaseRule() {
259                super.createCaseRule();
260
261                // C# also allows any kind of constant expression as a case label, e.g:
262                // Foo.BAR + Foo.GOO << 12
263                inState(IN_METHOD).markStart().sequence(CASE).skipTo(COLON).createNode(EShallowEntityType.META, 0).endNode();
264        }

299        /** {@inheritDoc} */
300        @Override
301        protected RecognizerBase<EGenericParserStates> typePattern(RecognizerBase<EGenericParserStates> currentState) {
302                EnumSet<ETokenType> modifierKeywords = EnumSet.of(PRIVATE, PROTECTED, PUBLIC, INTERNAL, VIRTUAL, ABSTRACT,
303                                ASYNC, CONST, EXTERN, OVERRIDE, READONLY, UNSAFE, VOLATILE, NEW, STATIC, SEALED);
304                EnumSet<ETokenType> typeStart = EnumSet.copyOf(PRIMITIVE_TYPES);
305                typeStart.addAll(getValidIdentifiers());
306
307                // we include "?" in the skipping section to deal with nullable types
308                // (e.g. Foo? foo;)
309                // the repeated (DOT, typeStart) is used for full qualified type names
310                return currentState.repeated(modifierKeywords).sequence(typeStart).repeated(DOT, typeStart).optional(QUESTION)
311                                .skipNested(LT, GT).subRecognizer(new ArrayBracketsRecognizer(), 0, Integer.MAX_VALUE);
312        }

314        /** {@inheritDoc} */
315        @Override
316        protected void createSubExpressionRules() {
317                // anonymous delegate methods
318                inState(IN_EXPRESSION).sequence(DELEGATE, LPAREN)
319                                .createNode(EShallowEntityType.METHOD, SubTypeNames.ANONYMOUS_METHOD)
320                                .skipToWithNesting(RPAREN, LPAREN, RPAREN).sequence(LBRACE).parseUntil(IN_METHOD).sequence(RBRACE)
321                                .endNode();
322
323                createLambdaWithArrowRules(DOUBLE_ARROW);
324        }
325
326        /** {@inheritDoc} */
327        @Override
328        protected RecognizerBase<EGenericParserStates> getSubExpressionRecognizer() {
329                return new CsDelegateAndLambdaRecognizer();
330        }
