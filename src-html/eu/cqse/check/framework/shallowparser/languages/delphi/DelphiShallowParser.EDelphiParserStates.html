129/**
130 * A shallow parser for Delphi.
131 * 
132 * This parser currently recognizes types (classes, records, enumerations,
133 * ranges, sets), procedures, functions and global directives like
134 * initialization and finalization sections. Within methods, control structures
135 * and simple statements can be parsed. The parser does not parse into the
136 * statements. Nested methods are supported.
137 * 
138 * @author $Author: hummelb $
139 * @version $Rev: 53669 $
140 * @ConQAT.Rating GREEN Hash: E437DA99738951AAED40E71762336564
141 */
142public class DelphiShallowParser
143                extends ShallowParserBase<EDelphiParserStates> {
144
145        /** All possible states of the DelphiShallowParser. */
146        public static enum EDelphiParserStates {
147
148                /** Top-level state. */
149                TOP_LEVEL,
150
151                /** Inside a interface state. */
152                IN_INTERFACE,
153
154                /**
155                 * Inside a implementation section (and initialization and
156                 * finalization).
157                 */
158                IN_IMPLEMENTATION,
159
160                /** Inside a type-section. */
161                IN_TYPE,
162
163                /** Inside a type declaration, for example a class or a record. */
164                IN_TYPE_DECL,
165
166                /** Inside a case statement within a type declaration. */
167                IN_TYPE_CASE,
168
169                /** Inside the label of a case statement within a type declaration. */
170                IN_TYPE_CASE_LABEL,
171
172                /** Inside an enum declaration. */
173                IN_ENUM,
174
175                /** Inside a variable-section. */
176                IN_VAR,
177
178                /** Inside a const-section. */
179                IN_CONST,
180
181                /** Inside a case-of statement. */
182                IN_CASE,
183
184                /** Inside a method's head/variable declarations. */
185                IN_METHOD_HEAD,
186
187                /** Inside a method's implementation. */
188                IN_METHOD
189        }

191        /** A set of all token types, that indicate a new top-level section. */
192        private static final EnumSet<ETokenType> TOP_LEVEL_KEYWORDS = EnumSet
193                        .of(IMPLEMENTATION, INTERFACE, BEGIN, FINALIZATION, INITIALIZATION);

196        private static final EDelphiParserStates[] SECTION_STATES = { IN_INTERFACE,
197                        IN_IMPLEMENTATION };

204                        .of(PROGRAM, UNIT, INTERFACE, IMPLEMENTATION, TYPE, VAR, CONST,
205                                        PROCEDURE, FUNCTION, BEGIN, END, INITIALIZATION,
206                                        FINALIZATION);

209         * A set of all token types that mark the beginning of a

258        /** Constructor. */
259        public DelphiShallowParser() {
260                super(EDelphiParserStates.class, TOP_LEVEL);
261                createTopLevelRules();
262                createTypeRules();
263                createEnumRules();
264                createTypeDeclRules();
265                createMethodSectionRules();
266                createVarRules();
267                createConstRules();
268                createMethodRules();
269                createClassAttributeRules();

272        /** Create rules for all top-level structures. */
273        private void createTopLevelRules() {
274                // program or unit declaration
275                inState(TOP_LEVEL).sequence(EnumSet.of(PROGRAM, UNIT)).skipTo(SEMICOLON)
276                                .createNode(EShallowEntityType.META, 0, 1).endNode();
277
278                // uses declaration
279                inAnyState().sequence(USES).skipTo(SEMICOLON)
280                                .createNode(EShallowEntityType.META, 0).endNode();
281
282                // interface section
283                inState(TOP_LEVEL).sequence(INTERFACE)
284                                .createNode(EShallowEntityType.META, 0)
285                                .parseUntilOrEof(IN_INTERFACE)
286                                .sequenceBefore(TOP_LEVEL_KEYWORDS).endNode();
287
288                // implementation section
289                inState(TOP_LEVEL).sequence(IMPLEMENTATION)
290                                .createNode(EShallowEntityType.META, 0)
291                                .parseUntilOrEof(IN_IMPLEMENTATION)
292                                .sequenceBefore(TOP_LEVEL_KEYWORDS).endNode();
293
294                // list of type declarations
295                inState(IN_INTERFACE).sequence(TYPE).parseUntilOrEof(IN_TYPE)
296                                .sequenceBefore(TOP_LEVEL_OR_SECTION_KEYWORDS);
297
298                // variable declaration
299                inState(SECTION_STATES).sequence(VAR).parseUntilOrEof(IN_VAR)
300                                .sequenceBefore(TOP_LEVEL_OR_SECTION_KEYWORDS);
301
302                // const declaration
303                inState(SECTION_STATES).sequence(CONST).parseUntilOrEof(IN_CONST)
304                                .sequenceBefore(TOP_LEVEL_OR_SECTION_KEYWORDS);
305
306                // declaration of methods
307                inState(IN_INTERFACE).sequence(METHOD_KEYWORDS, IDENTIFIER)
308                                .repeated(DOT, IDENTIFIER)
309                                .createNode(EShallowEntityType.METHOD, 0, new Region(1, -1))
310                                .skipNested(LPAREN, RPAREN).skipTo(SEMICOLON).endNode();
311
312                // method implementation (nested functions are
313                // processed in IN_METHOD state)
314                RecognizerBase<EDelphiParserStates> methodRecognizer = inState(
315                                IN_IMPLEMENTATION, IN_METHOD)
316                                                .sequence(METHOD_KEYWORDS, IDENTIFIER)
317                                                .repeated(DOT, IDENTIFIER)
318                                                .createNode(EShallowEntityType.METHOD, 0,
319                                                                new Region(1, -1))
320                                                .skipNested(LPAREN, RPAREN).skipTo(SEMICOLON)
321                                                .parseUntilOrEof(IN_METHOD);
322                methodRecognizer.sequence(END).optional(SEMICOLON).endNode();
323
324                // global begin
325                inState(TOP_LEVEL, IN_IMPLEMENTATION).sequence(BEGIN)
326                                .createNode(EShallowEntityType.METHOD,
327                                                SubTypeNames.GLOBAL_BEGIN)
328                                .parseUntil(IN_METHOD).sequence(END).optional(SEMICOLON)
329                                .endNode();
330
331                // initialization/finalization
332                inState(TOP_LEVEL).sequence(EnumSet.of(INITIALIZATION, FINALIZATION))
333                                .createNode(EShallowEntityType.METHOD, 0).parseUntil(IN_METHOD)
334                                .sequenceBefore(TOP_LEVEL_OR_SECTION_KEYWORDS).endNode();
335        }

337        /**
338         * Create rules for parsing types like classes, records, enumerations,
339         * function pointers, ranges, sets and meta-classes.
340         */
341        private void createTypeRules() {
342                RecognizerBase<EDelphiParserStates> typeAlternative = inState(IN_TYPE)
343                                .sequence(IDENTIFIER).skipTo(EQ);
344
345                // existing type alias
346                typeAlternative.optional(TYPE).sequence(IDENTIFIER, SEMICOLON)
347                                .createNode(EShallowEntityType.TYPE, SubTypeNames.ALIAS, 0)
348                                .endNode();
349
350                // type pointer
351                typeAlternative.sequence(XOR, IDENTIFIER, SEMICOLON)
352                                .createNode(EShallowEntityType.TYPE, SubTypeNames.TYPE_POINTER,
353                                                0)
354                                .endNode();
355
356                // array or file of existing types
357                typeAlternative.sequence(EnumSet.of(ARRAY, FILE))
358                                .skipToWithNesting(SEMICOLON, LPAREN, RPAREN)
359                                .createNode(EShallowEntityType.TYPE, 2, 0).endNode();
360
361                // function pointer declaration
362                typeAlternative.sequence(EnumSet.of(PROCEDURE, FUNCTION))
363                                .createNode(EShallowEntityType.TYPE,
364                                                SubTypeNames.FUNCTION_POINTER, 0)
365                                .skipNested(LPAREN, RPAREN).skipTo(SEMICOLON).endNode();
366
367                // anonymous method declaration
368                typeAlternative.sequence(REFERENCE, TO, EnumSet.of(PROCEDURE, FUNCTION))
369                                .createNode(EShallowEntityType.TYPE,
370                                                SubTypeNames.ANONYMOUS_METHOD, 0)
371                                .skipNested(LPAREN, RPAREN).skipTo(SEMICOLON).endNode();
372
373                // enum declaration
374                typeAlternative.sequence(LPAREN)
375                                .createNode(EShallowEntityType.TYPE, SubTypeNames.ENUM, 0)
376                                .parseUntil(IN_ENUM).sequence(RPAREN, SEMICOLON).endNode();
377
378                // range declaration
379                EnumSet<ETokenType> rangeTypes = EnumSet.of(INTEGER_LITERAL,
380                                STRING_LITERAL);
381                typeAlternative.sequence(rangeTypes, DOUBLE_DOT, rangeTypes, SEMICOLON)
382                                .createNode(EShallowEntityType.TYPE, SubTypeNames.RANGE, 0)
383                                .endNode();
384
385                // set declaration
386                typeAlternative.sequence(SET, OF)
387                                .createNode(EShallowEntityType.TYPE, SubTypeNames.SET, 0)
388                                .skipTo(SEMICOLON).endNode();
389
390                // meta-class type
391                typeAlternative.sequence(CLASS, OF, CLASS, TYPE, SEMICOLON)
392                                .createNode(EShallowEntityType.TYPE, SubTypeNames.META_CLASS, 0)
393                                .endNode();
394
395                // class/record/interface/dispinterface declaration with possible
396                // inheritance
397                RecognizerBase<EDelphiParserStates> declAlternative = typeAlternative
398                                .optional(PACKED)
399                                .sequence(EnumSet.of(CLASS, INTERFACE, RECORD, DISPINTERFACE))
400                                .createNode(EShallowEntityType.TYPE, -1, 0)
401                                .skipNested(LPAREN, RPAREN);
402                // helper declaration
403                endTypeDeclaration(declAlternative.sequence(IDENTIFIER)
404                                .skipNested(LPAREN, RPAREN).sequence(FOR, IDENTIFIER));
405                endTypeDeclaration(declAlternative);
406        }

408        /**
409         * End a type declaration recognizer. Either a semicolon or a type body is
410         * expected.
411         */
412        private static void endTypeDeclaration(
413                        RecognizerBase<EDelphiParserStates> recognizer) {
414                recognizer.sequence(SEMICOLON).endNode();
415                recognizer.parseUntil(IN_TYPE_DECL).sequence(END).optional(SEMICOLON)
416                                .endNode();
417        }

419        /** Create rules for parsing enumerations. */
420        private void createEnumRules() {
421                // beginning enum-literal
422                inState(IN_ENUM).sequence(IDENTIFIER).optional(EQ, INTEGER_LITERAL)
423                                .createNode(EShallowEntityType.ATTRIBUTE,
424                                                SubTypeNames.ENUM_LITERAL, 0)
425                                .endNode();
426
427                // enum-literal separated with comma
428                inState(IN_ENUM).sequence(COMMA, IDENTIFIER)
429                                .optional(EQ, INTEGER_LITERAL)
430                                .createNode(EShallowEntityType.ATTRIBUTE,
431                                                SubTypeNames.ENUM_LITERAL, 1)
432                                .endNode();
433        }

435        /** Create rules for parsing classes and records. */
436        private void createTypeDeclRules() {
437                // access modifiers
438                inState(IN_TYPE_DECL).optional(STRICT).sequence(ACCESS_MODIFIERS)
439                                .createNode(EShallowEntityType.META, new Region(0, -1))
440                                .endNode();
441
442                // function/procedure declaration
443                RecognizerBase<EDelphiParserStates> methodBase = inState(IN_TYPE_DECL)
444                                .optional(CLASS).sequence(METHOD_KEYWORDS, IDENTIFIER)
445                                .createNode(EShallowEntityType.METHOD, -2, -1)
446                                .skipNested(LPAREN, RPAREN).skipTo(SEMICOLON);
447                appendMethodModifierRules(methodBase);
448
449                // property declaration
450                inState(IN_TYPE_DECL).sequence(PROPERTY).skipTo(SEMICOLON)
451                                .createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.PROPERTY,
452                                                1)
453                                .endNode();
454
455                // attribute declaration
456                inState(IN_TYPE_DECL).sequence(IDENTIFIER).skipTo(SEMICOLON)
457                                .createNode(EShallowEntityType.ATTRIBUTE,
458                                                SubTypeNames.ATTRIBUTE, 0)
459                                .endNode();
460
461                // case statement
462                inState(IN_TYPE_DECL).sequence(CASE).skipTo(OF)
463                                .createNode(EShallowEntityType.META, SubTypeNames.CASE)
464                                .parseUntil(IN_TYPE_CASE).sequenceBefore(TYPE_MEMBER_TOKENS)
465                                .endNode();
466
467                createTypeCaseRules();
468        }
469
470        /** Create rules for parsing case statements within type declarations. */
471        private void createTypeCaseRules() {
472                inState(IN_TYPE_CASE).sequence(CASE_LITERALS).skipTo(COLON)
473                                .sequence(LPAREN)
474                                .createNode(EShallowEntityType.META, SubTypeNames.CASE_LABEL, 0)
475                                .parseUntil(IN_TYPE_CASE_LABEL).sequence(RPAREN, SEMICOLON)
476                                .endNode();
477
478                inState(IN_TYPE_CASE_LABEL).optional(SEMICOLON).markStart()
479                                .sequence(IDENTIFIER, COLON)
480                                .createNode(EShallowEntityType.ATTRIBUTE,
481                                                SubTypeNames.ATTRIBUTE, 0)
482                                .skipBefore(EnumSet.of(SEMICOLON, RPAREN)).endNode();
483        }

485        /**
486         * Create rules for parsing sections within a method. Possible sections are
487         * variable, constant, type declarations or the method's body.
488         */
489        private void createMethodSectionRules() {
490                // variable declaration within method
491                inState(IN_METHOD).sequence(VAR).parseUntil(IN_VAR)
492                                .sequenceBefore(METHOD_SECTION_TOKENS);
493
494                // const declaration within method
495                inState(IN_METHOD).sequence(CONST).parseUntil(IN_CONST)
496                                .sequenceBefore(METHOD_SECTION_TOKENS);
497
498                // type declaration within method
499                inState(IN_METHOD).sequence(TYPE)
500                                .parseUntil(EDelphiParserStates.IN_TYPE)
501                                .sequenceBefore(METHOD_SECTION_TOKENS);
502
503                // begin of method body
504                inState(IN_METHOD).sequence(BEGIN).parseUntil(IN_METHOD)
505                                .sequenceBefore(END).optional(SEMICOLON);
506        }

543        /**
544         * Append rules for parsing method/function/procedure modifiers to the given
545         * recognizer.
546         */
547        private static void appendMethodModifierRules(
548                        RecognizerBase<EDelphiParserStates> recognizer) {
549                RecognizerBase<EDelphiParserStates> modifierAlternative = recognizer
550                                .repeated(METHOD_MODIFIERS, SEMICOLON);
551                // parse external directive
552                modifierAlternative.sequence(EXTERNAL).skipTo(SEMICOLON).endNode();
553                // the last modifier was the end
554                modifierAlternative.endNode();
555        }

557        /** Create rules for parsing if/else structures. */
558        private void createIfElseRules() {
559                RecognizerBase<EDelphiParserStates> ifAlternative = inState(IN_METHOD)
560                                .sequence(IF).createNode(EShallowEntityType.STATEMENT, 0)
561                                .skipTo(THEN);
562                endWithPossibleContinuation(ifAlternative.sequence(BEGIN)
563                                .parseUntil(IN_METHOD).sequence(END).optional(SEMICOLON),
564                                EnumSet.of(ELSE));
565                endWithPossibleContinuation(ifAlternative.parseOnce(IN_METHOD),
566                                EnumSet.of(ELSE));
567
568                RecognizerBase<EDelphiParserStates> elseIfAlternative = inState(
569                                IN_METHOD).sequence(ELSE, IF)
570                                                .createNode(EShallowEntityType.STATEMENT,
571                                                                new int[] { 0, 1 })
572                                                .skipTo(THEN);
573                endWithPossibleContinuation(elseIfAlternative.sequence(BEGIN)
574                                .parseUntil(IN_METHOD).sequence(END).optional(SEMICOLON),
575                                EnumSet.of(ELSE));
576                endWithPossibleContinuation(elseIfAlternative.parseOnce(IN_METHOD),
577                                EnumSet.of(ELSE));
578
579                RecognizerBase<EDelphiParserStates> elseAlternative = inState(IN_METHOD)
580                                .sequence(ELSE).createNode(EShallowEntityType.STATEMENT, 0);
581                elseAlternative.sequence(BEGIN).parseUntil(IN_METHOD).sequence(END)
582                                .optional(SEMICOLON).endNode();
583                elseAlternative.parseOnce(IN_METHOD).endNode();
584        }

586        /** Create rules for parsing try, except and finally structures. */
587        private void createTryExceptFinallyRules() {
588                RecognizerBase<EDelphiParserStates> tryAlternative = inState(IN_METHOD)
589                                .sequence(TRY).createNode(EShallowEntityType.STATEMENT, 0)
590                                .parseUntil(IN_METHOD);
591                endWithContinuationOrEnd(tryAlternative, EnumSet.of(EXCEPT, FINALLY));
592
593                RecognizerBase<EDelphiParserStates> exceptAlternative = inState(
594                                IN_METHOD).sequence(EXCEPT)
595                                                .createNode(EShallowEntityType.STATEMENT, 0)
596                                                .parseUntil(IN_METHOD);
597                endWithContinuationOrEnd(exceptAlternative,
598                                EnumSet.of(EXCEPT, FINALLY));
599
600                inState(IN_METHOD).sequence(FINALLY)
601                                .createNode(EShallowEntityType.STATEMENT, 0)
602                                .parseUntil(IN_METHOD).sequence(END).optional(SEMICOLON)
603                                .endNode();
604
605                // on clause
606                RecognizerBase<EDelphiParserStates> onAlternative = inState(IN_METHOD)
607                                .sequence(ON, IDENTIFIER, COLON, IDENTIFIER, DO)
608                                .createNode(EShallowEntityType.META, 0, 1);
609                endWithPossibleContinuation(
610                                onAlternative.sequence(BEGIN).parseUntil(IN_METHOD)
611                                                .sequence(END).optional(SEMICOLON),
612                                EnumSet.of(ON, ELSE));
613                endWithPossibleContinuation(onAlternative.parseOnce(IN_METHOD),
614                                EnumSet.of(ON, ELSE));
615        }

617        /**
618         * End the given recognizer's current node with continuation, if one of the
619         * given continuation tokens is recognized and sequence before it. Otherwise
620         * expect an end and a semicolon token and end the current node.
621         */
622        private static void endWithContinuationOrEnd(
623                        RecognizerBase<EDelphiParserStates> recognizer,
624                        EnumSet<ETokenType> continuationTokens) {
625                recognizer.sequenceBefore(continuationTokens).endNodeWithContinuation();
626                recognizer.sequence(END).optional(SEMICOLON).endNode();
627        }

629        /** Create rules for parsing loops. */
630        private void createLoopRules() {
631                // repeat until loop
632                inState(IN_METHOD).sequence(REPEAT)
633                                .createNode(EShallowEntityType.STATEMENT, 0)
634                                .parseUntil(IN_METHOD).sequence(UNTIL).skipTo(SEMICOLON)
635                                .endNode();
636
637                // while/for loop
638                RecognizerBase<EDelphiParserStates> whileForAlternative = inState(
639                                IN_METHOD).sequence(EnumSet.of(WHILE, FOR))
640                                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DO);
641                whileForAlternative.sequence(BEGIN).parseUntil(IN_METHOD).sequence(END)
642                                .optional(SEMICOLON).endNode();
643                whileForAlternative.parseOnce(IN_METHOD).endNode();
644        }

646        /** Create rules for parsing case structures. */
647        private void createCaseRules() {
648                inState(IN_METHOD).sequence(CASE).skipTo(OF)
649                                .createNode(EShallowEntityType.STATEMENT, 0).parseUntil(IN_CASE)
650                                .sequence(END).optional(SEMICOLON).endNode();
651
652                appendCaseLabelBody(inState(IN_CASE).sequence(CASE_LITERALS)
653                                .skipTo(COLON)
654                                .createNode(EShallowEntityType.META, SubTypeNames.CASE, 0));
655
656                RecognizerBase<EDelphiParserStates> elseRecognizer = inState(IN_CASE)
657                                .sequence(ELSE).createNode(EShallowEntityType.META, 0);
658                elseRecognizer.parseUntil(IN_METHOD).sequenceBefore(END).endNode();
659        }
