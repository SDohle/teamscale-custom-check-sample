113/**
114 * A shallow parser for Fortran.
115 * 
116 * This parser supports the free form of Fortran 2008.
117 * 
118 * @author $Author: hummelb $
119 * @version $Rev: 52657 $
120 * @ConQAT.Rating GREEN Hash: B96028DDFA354B5E339BE8C0159D1736
121 */
122public class FortranShallowParser extends
123                ShallowParserBase<EFortranParserStates> {
124
125        /** All possible states of a FortranShallowParser. */
126        public static enum EFortranParserStates {
127                /** Top-level state. */
128                TOP_LEVEL,
129
130                /** Inside a module. */
131                IN_MODULE,
132
133                /** Inside a type declaration. */
134                IN_TYPE,
135
136                /** Inside a enum declaration. */
137                IN_ENUM,
138
139                /** Inside a method's body. */
140                IN_METHOD,
141
142                /** Inside a select statement. */
143                IN_SELECT,
144        }
145
146        /** All tokens that indicate a simple statement. */
147        private static final EnumSet<ETokenType> STATEMENT_TOKENS = EnumSet.of(
148                        ALLOCATE, ASYNCHRONOUS, CALL, CLOSE, COMMON, CONTINUE, CYCLE, DATA,
149                        DEALLOCATE, ENDFILE, ENTRY, EQUIVALENCE, EXIT, EXTERNAL, GOTO,
150                        IDENTIFIER, INTEGER_LITERAL, INQUIRE, LOCK, NAMELIST, NULLIFY,
151                        OPEN, PAUSE, PARAMETER, PRINT, RETURN, REWIND, REWRITE, STOP,
152                        UNLOCK, WAIT, WRITE, VOLATILE);
153
154        /** All parser states in which statements can occur. */
155        private static final EFortranParserStates[] STATEMENT_STATES = { IN_MODULE,
156                        IN_METHOD };
157
158        /** All modifiers that can be used in front of a function declaration. */
159        private static final EnumSet<ETokenType> FUNCTION_MODIFIERS = EnumSet.of(
160                        MODULE, ELEMENTAL, PURE, RECURSIVE);
161
162        /** All modifiers that can be used in front of a procedure declaration. */
163        private static final EnumSet<ETokenType> PROCEDURE_MODIFIERS = EnumSet
164                        .of(MODULE);
165
166        /** All token types that can be used as a type. */
167        private static final EnumSet<ETokenType> TYPE_TOKENS = EnumSet.of(
168                        IDENTIFIER, TYPE, CLASS);
169
170        /** Constructor. */
171        public FortranShallowParser() {
172                super(EFortranParserStates.class, TOP_LEVEL);
173                createTopLevelRules();
174                createInModuleRules();
175                createInTypeRules();
176                createInEnumRule();
177                createMethodRules();
178                createStatementRules();
179                createAnyStateRules();
180        }
181
182        /** Create rules for parsing top-level constructs like modules or program. */
183        private void createTopLevelRules() {
184                // module
185                inState(TOP_LEVEL).sequence(MODULE, IDENTIFIER)
186                                .createNode(EShallowEntityType.MODULE, 0, 1).sequence(EOL)
187                                .parseUntil(IN_MODULE).sequence(END).optional(MODULE)
188                                .optional(IDENTIFIER).endNode();
189
190                // program
191                inState(TOP_LEVEL).sequence(PROGRAM, IDENTIFIER)
192                                .createNode(EShallowEntityType.METHOD, 0, -1).sequence(EOL)
193                                .parseUntil(IN_MODULE).sequence(END).optional(PROGRAM)
194                                .optional(IDENTIFIER).endNode();
195
196                // submodule
197                inState(TOP_LEVEL).sequence(SUBMODULE)
198                                .skipToWithNesting(IDENTIFIER, LPAREN, RPAREN)
199                                .createNode(EShallowEntityType.MODULE, 0, -1).sequence(EOL)
200                                .parseUntil(IN_MODULE).sequence(END).optional(SUBMODULE)
201                                .optional(IDENTIFIER).endNode();
202        }
203
204        /** Create rules for parsing some meta information. */
205        private void createAnyStateRules() {
206                inAnyState().sequence(IMPLICIT).createNode(EShallowEntityType.META, 0)
207                                .skipTo(EOL).endNode();
208
209                inAnyState().sequence(CONTAINS).createNode(EShallowEntityType.META, 0)
210                                .endNode();
211
212                inState(IN_MODULE, IN_METHOD).sequence(USE)
213                                .createNode(EShallowEntityType.META, 0).skipTo(EOL).endNode();
214        }
215
216        /** Create rules for parsing the body of a module. */
217        private void createInModuleRules() {
218                // visibility
219                inState(IN_MODULE).sequence(EnumSet.of(PUBLIC, PRIVATE))
220                                .createNode(EShallowEntityType.META, 0).skipTo(EOL).endNode();
221
222                // interface with name
223                inState(IN_MODULE).sequence(INTERFACE, IDENTIFIER)
224                                .createNode(EShallowEntityType.TYPE, 0, 1)
225                                .parseUntil(IN_MODULE).sequence(END).optional(INTERFACE)
226                                .sequence(IDENTIFIER).endNode();
227
228                // interface with no name
229                inState(IN_MODULE).sequence(INTERFACE)
230                                .createNode(EShallowEntityType.TYPE, 0).parseUntil(IN_MODULE)
231                                .sequence(END).optional(INTERFACE).endNode();
232
233                // type declaration
234                RecognizerBase<EFortranParserStates> typeAlternative = inState(
235                                IN_MODULE).sequence(TYPE);
236                typeAlternative.optional(DOUBLE_COLON).sequence(IDENTIFIER, EOL)
237                                .createNode(EShallowEntityType.TYPE, 0, -2).parseUntil(IN_TYPE)
238                                .sequence(END).optional(TYPE).optional(IDENTIFIER).endNode();
239                typeAlternative.sequence(COMMA)
240                                .skipBefore(EnumSet.of(DOUBLE_COLON, EOL))
241                                .sequence(DOUBLE_COLON, IDENTIFIER, EOL)
242                                .createNode(EShallowEntityType.TYPE, 0, -2).parseUntil(IN_TYPE)
243                                .sequence(END).optional(TYPE).optional(IDENTIFIER).endNode();
244
245                // enum declaration
246                inState(IN_MODULE).sequence(ENUM)
247                                .skipBefore(EnumSet.of(DOUBLE_COLON, EOL))
248                                .sequence(DOUBLE_COLON, IDENTIFIER, EOL)
249                                .createNode(EShallowEntityType.TYPE, 0, -2).parseUntil(IN_ENUM)
250                                .sequence(END).optional(ENUM).optional(IDENTIFIER).endNode();
251        }
252
253        /** Create rules for parsing type bodies. */
254        private void createInTypeRules() {
255                // sequence
256                inState(IN_TYPE).sequence(SEQUENCE, EOL)
257                                .createNode(EShallowEntityType.META, 0).endNode();
258
259                // attribute
260                inState(IN_TYPE)
261                                .sequence(TYPE_TOKENS)
262                                .skipBefore(EnumSet.of(DOUBLE_COLON, EOL))
263                                .sequence(DOUBLE_COLON, IDENTIFIER)
264                                .createNode(EShallowEntityType.ATTRIBUTE,
265                                                SubTypeNames.ATTRIBUTE, -1).skipTo(EOL).endNode();
266
267                // procedure
268                inState(IN_TYPE).sequence(PROCEDURE, DOUBLE_COLON, IDENTIFIER)
269                                .skipTo(EOL).createNode(EShallowEntityType.METHOD, 0, 2)
270                                .endNode();
271        }
272
273        /** Create rule for parsing an enum body. */
274        private void createInEnumRule() {
275                inState(IN_ENUM).sequence(ENUMERATOR).optional(DOUBLE_COLON)
276                                .sequence(IDENTIFIER)
277                                .createNode(EShallowEntityType.ATTRIBUTE, 0, -1).skipTo(EOL)
278                                .endNode();
279        }
280
281        /** Create rules for parsing method declarations. */
282        private void createMethodRules() {
283                // subroutine
284                inState(TOP_LEVEL, IN_MODULE, IN_METHOD)
285                                .sequence(SUBROUTINE, IDENTIFIER)
286                                .createNode(EShallowEntityType.METHOD, 0, 1).skipTo(EOL)
287                                .parseUntil(IN_METHOD).sequence(END).optional(SUBROUTINE)
288                                .optional(IDENTIFIER).endNode();
289
290                // function
291                inState(TOP_LEVEL, IN_MODULE, IN_METHOD)
292                                .optional(FUNCTION_MODIFIERS, TYPE_TOKENS)
293                                .skipBefore(EnumSet.of(FUNCTION, EOL))
294                                .sequence(FUNCTION, IDENTIFIER)
295                                .createNode(EShallowEntityType.METHOD, -2, -1).skipTo(EOL)
296                                .parseUntil(IN_METHOD).sequence(END).optional(FUNCTION)
297                                .optional(IDENTIFIER).endNode();
298
299                // variable declaration
300                inState(IN_MODULE, IN_METHOD)
301                                .sequence(TYPE_TOKENS)
302                                .skipBefore(EnumSet.of(DOUBLE_COLON, EOL))
303                                .sequence(DOUBLE_COLON, IDENTIFIER)
304                                .createNode(EShallowEntityType.STATEMENT,
305                                                SubTypeNames.LOCAL_VARIABLE, -1).skipTo(EOL).endNode();
306
307                // procedure
308                inState(IN_MODULE, IN_TYPE).repeated(PROCEDURE_MODIFIERS)
309                                .sequence(PROCEDURE, IDENTIFIER)
310                                .createNode(EShallowEntityType.METHOD, -2, -1).skipTo(EOL)
311                                .endNode();
312        }

314        /**
315         * Create rules for parsing statements. All constructs are taken from
316         * http://de.wikibooks.org/wiki/Fortran:_Fortran_95:
317         * _Verzweigungen_und_Schleifen.
318         */
319        private void createStatementRules() {
320                createArithmeticIfRule();
321
322                createSingleLineConstructs(IF, WHERE, FORALL);
323
324                createLoopRules();
325                createSelectRules();
326
327                createElseIfRule();
328                createContinuedConstructRules(IF, ELSEIF, ELSE, THEN);
329                createContinuedConstructRules(WHERE, ELSEWHERE, ELSEWHERE, null);
330
331                createConcurrentRules();
332
333                createSimpleStatementRules();
334        }
335
336        /**
337         * Create rule for parsing the arithmetic if construct (e.g. 'if
338         * (&lt;arithmetic expression&gt;) 1, 2, 3')
339         */
340        private void createArithmeticIfRule() {
341                inState(STATEMENT_STATES)
342                                .sequence(IF)
343                                .skipNested(LPAREN, RPAREN)
344                                .sequence(INTEGER_LITERAL, COMMA, INTEGER_LITERAL, COMMA,
345                                                INTEGER_LITERAL, EOL)
346                                .createNode(EShallowEntityType.STATEMENT, 0).endNode();
347        }
348
349        /**
350         * Create rules for parsing constructs with a condition in parenthesis and
351         * one statement in the same line for all given token types. A single line
352         * construct is e.g. 'if (&lt;condition&gt;) &lt;statement&gt;'
353         */
354        private void createSingleLineConstructs(ETokenType... constructTokens) {
355                for (ETokenType constructToken : constructTokens) {
356                        inState(STATEMENT_STATES).sequence(constructToken, LPAREN)
357                                        .skipToWithNesting(RPAREN, LPAREN, RPAREN)
358                                        .sequenceBefore(STATEMENT_TOKENS)
359                                        .createNode(EShallowEntityType.STATEMENT, 0)
360                                        .parseOnce(IN_METHOD).sequence(EOL).endNode();
361                }
362        }
363
364        /** Create rules for parsing block and critical sections. */
365        private void createConcurrentRules() {
366                inState(STATEMENT_STATES).sequence(BLOCK)
367                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(EOL)
368                                .parseUntil(IN_METHOD).sequence(END).optional(BLOCK).endNode();
369
370                inState(STATEMENT_STATES).sequence(CRITICAL)
371                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(EOL)
372                                .parseUntil(IN_METHOD).sequence(END).optional(CRITICAL)
373                                .endNode();
374        }

376        /** Create rules for parsing a select statement. */
377        private void createSelectRules() {
378                finishConstructWithOptionalName(
379                                inState(STATEMENT_STATES)
380                                                .optional(IDENTIFIER, COLON)
381                                                .markStart()
382                                                .sequence(SELECT)
383                                                .optional(EnumSet.of(CASE, TYPE))
384                                                .sequence(LPAREN)
385                                                .skipToWithNesting(RPAREN, LPAREN, RPAREN)
386                                                .sequence(EOL)
387                                                .createNode(EShallowEntityType.STATEMENT,
388                                                                SubTypeNames.SELECT).parseUntil(IN_SELECT),
389                                SELECT);
390
391                RecognizerBase<EFortranParserStates> labelAlternative = inState(
392                                IN_SELECT).sequence(EnumSet.of(CASE, TYPE, CLASS)).optional(IS)
393                                .skipNested(LPAREN, RPAREN).optional(IDENTIFIER).sequence(EOL)
394                                .createNode(EShallowEntityType.META, 0).parseUntil(IN_METHOD);
395                labelAlternative.sequenceBefore(END).endNode();
396                labelAlternative.sequenceBefore(EnumSet.of(CASE, TYPE, CLASS))
397                                .endNodeWithContinuation();
398
399                inState(IN_SELECT).sequence(EnumSet.of(CASE, CLASS), DEFAULT)
400                                .optional(IDENTIFIER).sequence(EOL)
401                                .createNode(EShallowEntityType.META, 0).parseUntil(IN_METHOD)
402                                .sequenceBefore(END).endNode();
403        }
404
405        /** Create rules for parsing do and forall loops. */
406        private void createLoopRules() {
407                // do/do concurrent/do while loop
408                finishConstructWithOptionalName(
409                                inState(STATEMENT_STATES)
410                                                .optional(IDENTIFIER, COLON)
411                                                .markStart()
412                                                .sequence(DO)
413                                                .optional(EnumSet.of(CONCURRENT, WHILE))
414                                                .createNode(EShallowEntityType.STATEMENT,
415                                                                new Region(0, -1)).skipTo(EOL)
416                                                .parseUntil(IN_METHOD), DO);
417
418                // forall loop
419                finishConstructWithOptionalName(
420                                inState(STATEMENT_STATES).optional(IDENTIFIER, COLON)
421                                                .markStart().sequence(FORALL, LPAREN)
422                                                .skipToWithNesting(RPAREN, LPAREN, RPAREN)
423                                                .createNode(EShallowEntityType.STATEMENT, 0)
424                                                .sequence(EOL).parseUntil(IN_METHOD), FORALL);
425        }

427        /**
428         * Create a special rule for handling 'else if' that consists of two
429         * separate tokens.
430         */
431        private void createElseIfRule() {
432                RecognizerBase<EFortranParserStates> recognizer = inState(
433                                STATEMENT_STATES)
434                                .sequence(ELSE, IF, LPAREN)
435                                .skipToWithNesting(RPAREN, LPAREN, RPAREN)
436                                .sequence(THEN)
437                                .optional(IDENTIFIER)
438                                .sequence(EOL)
439                                .createNode(EShallowEntityType.STATEMENT,
440                                                SubTypeNames.ELSE_IF_NOSPACE).parseUntil(IN_METHOD);
441                finishConstructWithOptionalName(recognizer, IF);
442                recognizer.sequenceBefore(EnumSet.of(ELSE, ELSEIF))
443                                .endNodeWithContinuation();
444        }
445
446        /**
447         * Create rules for parsing constructs than can be continued. The given
448         * begin token is the token that begins the construct for e.g. 'if'. The
449         * continuation token can continue a construct, e.g. 'else if'. The final
450         * token is the last unconditional continuation token, e.g. 'else'. The
451         * complete construct is ended with 'end &lt;beginToken&gt;'. The given
452         * suffix token must be recognized after the parenthesis of the construct
453         * e.g. 'then' in an 'if' construct. The suffix token can be null, if there
454         * is none. The construct can be prefixed by a construct name, e.g. 'name:if
455         * (...)'
456         */
457        private void createContinuedConstructRules(ETokenType beginToken,
458                        ETokenType continuationToken, ETokenType finalToken,
459                        ETokenType suffixToken) {
460                createContinuedConstructRuleWithName(beginToken, beginToken,
461                                EnumSet.of(continuationToken, finalToken), suffixToken, true);
462                createContinuedConstructRuleWithName(beginToken, continuationToken,
463                                EnumSet.of(continuationToken, finalToken), suffixToken, false);
464                createEndContinuedConstructRule(beginToken, finalToken);
465        }
466
467        /**
468         * Create a rule for parsing a statement of a construct that can be
469         * continued. The construct token is the token of the construct type, e.g.
470         * 'if'. The statement token is the beginning token of the construct
471         * statement, e.g. 'if' or 'else if'. The given continuation tokens are
472         * those, that can continue the current construct statement. The given
473         * suffix token must be recognized after the parenthesis of the construct
474         * e.g. 'then' in an 'if' construct. The suffix token can be null, if there
475         * is none. If nameInFront is true, an optional name label can prefix the
476         * construct. If it is false, the construct can be followed by a name.
477         */
478        private void createContinuedConstructRuleWithName(
479                        ETokenType constructToken, ETokenType partToken,
480                        EnumSet<ETokenType> continuationTokens, ETokenType suffixToken,
481                        boolean nameInFront) {
482                RecognizerBase<EFortranParserStates> recognizer = inState(STATEMENT_STATES);
483                if (nameInFront) {
484                        recognizer = recognizer.optional(IDENTIFIER, COLON);
485                }
486                recognizer = recognizer.markStart().sequence(partToken, LPAREN)
487                                .skipToWithNesting(RPAREN, LPAREN, RPAREN);
488                if (suffixToken != null) {
489                        recognizer = recognizer.sequence(suffixToken);
490                }
491                if (!nameInFront) {
492                        recognizer = recognizer.optional(IDENTIFIER);
493                }
494                recognizer = recognizer.sequence(EOL)
495                                .createNode(EShallowEntityType.STATEMENT, 0)
496                                .parseUntil(IN_METHOD);
497
498                finishConstructWithOptionalName(recognizer, constructToken);
499                recognizer.sequenceBefore(continuationTokens).endNodeWithContinuation();
500        }

502        /**
503         * Create a rule for parsing the end of a continued construct. The construct
504         * token is the token of the construct type, e.g. 'if'. The final token is
505         * the beginning token of the statement, e.g. 'else'. An optional identifier
506         * can follow for named constructs.
507         */
508        private void createEndContinuedConstructRule(ETokenType constructToken,
509                        ETokenType finalToken) {
510                RecognizerBase<EFortranParserStates> finalAlternative = inState(
511                                STATEMENT_STATES).sequence(finalToken).optional(IDENTIFIER)
512                                .sequence(EOL).createNode(EShallowEntityType.STATEMENT, 0)
513                                .parseUntil(IN_METHOD);
514
515                finishConstructWithOptionalName(finalAlternative, constructToken);
516        }
517
518        /**
519         * Finishes the given recognizer with 'end' and the optional given token
520         * type. An optional identifier can follow for named constructs.
521         */
522        private void finishConstructWithOptionalName(
523                        RecognizerBase<EFortranParserStates> recognizer,
524                        ETokenType constructType) {
525                recognizer.sequence(END).optional(constructType).optional(IDENTIFIER)
526                                .sequence(EOL).endNode();
527        }
528
529        /** Creates rules for parsing simple statements. */
530        private void createSimpleStatementRules() {
531                inState(STATEMENT_STATES)
532                                .sequence(STATEMENT_TOKENS)
533                                .createNode(EShallowEntityType.STATEMENT,
534                                                SubTypeNames.SIMPLE_STATEMENT, 0)
535                                .skipBefore(EnumSet.of(EOL, SEMICOLON)).optional(SEMICOLON)
536                                .endNode();
537        }
