092/**
093 * Shallow parser for Gosu.
094 * <p>
095 * Started this parser from the JavaScriptShallowParser since Gosu seems to be a
096 * simple Java variant without semicolons. Gosu grammar:
097 * https://gosu-lang.github.io/grammar.html
098 */
099public class GosuShallowParser extends ShallowParserBase<EGosuParserStates> {
100
101        /** The states used in this parser. */
102        public static enum EGosuParserStates {
103
104                /** Top-level parsing state. */
105                TOP_LEVEL,
106
107                /** Parsing within a type, such as a TypeScript class. */
108                IN_TYPE,
109
110                /** Parsing in an expression */
111                IN_INTERFACE,
112
113                /** Parsing within a method. */
114                IN_METHOD,
115
116                /** Parses a lambda (arrow notation) construct. */
117                IN_LAMBDA,
118
119                /** Parsing in an expression */
120                IN_EXPRESSION;
121        }
122
123        /** Valid modifiers in Gosu */
124        static final EnumSet<ETokenType> MODIFIERS = EnumSet.of(PRIVATE, INTERNAL, PROTECTED, PUBLIC, STATIC,
125                        ABSTRACT, OVERRIDE, FINAL, TRANSIENT, OVERRIDE);
126
127        /** All tokens that mark the termination of an interface member entity. */
128        private static final EnumSet<ETokenType> MEMBER_TERMINATOR_TOKENS = EnumSet.of(
129                        // the interface-closing bracket
130                        RBRACE,
131                        // the keywords starting new interface members
132                        FUNCTION, PROPERTY, VAR, CLASS, INTERFACE, ENUM, CONSTRUCT,
133                        // the modifiers
134                        AT, PRIVATE, INTERNAL, PROTECTED, PUBLIC, STATIC, ABSTRACT, OVERRIDE, FINAL, TRANSIENT, OVERRIDE);
135
136        /**
137         * All opening parenthesis types that are allowed e.g., in type definitions
138         */
139        private static final List<ETokenType> ALL_OPEN_PAREN = Arrays.asList(LPAREN, LBRACE, LT, LBRACK);
140        /**
141         * All closing parenthesis types that are allowed e.g., in type definitions
142         */
143        private static final List<ETokenType> ALL_CLOSE_PAREN = Arrays.asList(RPAREN, RBRACE, GT, RBRACK);
144
145        /** All token types that can be used as literal or identifier in Gosu */
146        private static final Set<ETokenType> LITERALS_AND_IDENTIFIERS = CollectionUtils.unionSet(LITERALS, IDENTIFIERS);
147
148        /** Constructor. */
149        public GosuShallowParser() {
150                super(EGosuParserStates.class, TOP_LEVEL);
151                // top-level statements (using, ...)
152                createMetaRules();
153                // types (classes, interfaces, ...)
154                createModuleRules();
155                // entities in types
156                createTypeMemberWithBodyRules();
157                createAbstractTypeMemberRules();
158                createGlobalVariableRules();
159                // expressions
160                createSubExpressionRules();
161                // must be last as it contains the simple statement matcher
162                createStatementRules();
163        }
164
165        /** Creates parsing rules for meta elements. */
166        private void createMetaRules() {
167                inState(TOP_LEVEL).sequence(PACKAGE, IDENTIFIER).repeated(DOT, IDENTIFIER)
168                                .createNode(EShallowEntityType.META, SubTypeNames.PACKAGE, new Region(1, -1)).optional(SEMICOLON)
169                                .endNode();
170                inState(TOP_LEVEL).sequence(USES, IDENTIFIER).repeated(DOT, IDENTIFIER)
171                                .createNode(EShallowEntityType.META, SubTypeNames.USES, new Region(1, -1)).optional(SEMICOLON)
172                                .endNode();
173        }
174
175        /**
176         * Creates rules for parsing namespace/module constructs.
177         */
178        private void createModuleRules() {
179                // class and enum can occur top-level or in a type
180                inStateSkipModifiers(TOP_LEVEL, IN_TYPE).markStart().sequence(EnumSet.of(CLASS, ENUM), IDENTIFIER)
181                                .skipTo(LBRACE).createNode(EShallowEntityType.TYPE, 0, 1).parseUntil(IN_TYPE).sequence(RBRACE)
182                                .endNode();
183                // interface has members without body. They need special handling since
184                // it is hard to match when they terminate (no braces)
185                inStateSkipModifiers(TOP_LEVEL, IN_TYPE, IN_INTERFACE).markStart().sequence(INTERFACE, IDENTIFIER)
186                                .skipTo(LBRACE).createNode(EShallowEntityType.TYPE, 0, 1).parseUntil(EGosuParserStates.IN_INTERFACE)
187                                .sequence(RBRACE).endNode();
188                // enhancement (can't occur in a type)
189                inStateSkipModifiers(TOP_LEVEL).markStart().sequence(ENHANCEMENT, IDENTIFIER).skipTo(LBRACE)
190                                .createNode(EShallowEntityType.TYPE, 0, 1).parseUntil(IN_TYPE).sequence(RBRACE).endNode();
191        }
192
193        /**
194         * Creates a RecognizerBase that starts in the given ParserStates and skips
195         * {@link #MODIFIERS}.
196         */
197        private RecognizerBase<EGosuParserStates> inStateSkipModifiers(EGosuParserStates... states) {
198                return inState(states).repeated(MODIFIERS);
199        }
200
201        /**
202         * Rules for global variables (in classes, interfaces, ...)
203         */
204        private void createGlobalVariableRules() {
205                Set<ETokenType> typeMemberTerminatorsAndEq = CollectionUtils.unionSet(MEMBER_TERMINATOR_TOKENS,
206                                Collections.singleton(EQ));
207                RecognizerBase<EGosuParserStates> varStart = inStateSkipModifiers(IN_TYPE, IN_INTERFACE).markStart()
208                                .sequence(VAR, IDENTIFIER)
209                                .skipBeforeWithNesting(typeMemberTerminatorsAndEq, ALL_OPEN_PAREN, ALL_CLOSE_PAREN);
210                /*
211                 * This got a little complicated. After the EQ sign, we might want a
212                 * subparse, but it could also be a simple literal ("foo"). Therefore,
213                 * we use an optional sub parser and skip to the next member terminator
214                 * afterwards
215                 */
216                // var declaration with EQ and sub expression
217                varStart.sequence(EQ).createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.GLOBAL_VARIABLE, 1)
218                                .optionalSubRecognizer(emptyRecognizer().parseOnce(IN_EXPRESSION))
219                                .skipBeforeWithNesting(MEMBER_TERMINATOR_TOKENS, ALL_OPEN_PAREN, ALL_CLOSE_PAREN).endNode();
220
221                // var without EQ
222                varStart.sequenceBefore(MEMBER_TERMINATOR_TOKENS)
223                                .createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.GLOBAL_VARIABLE, 1).endNode();
224        }
225
226        /**
227         * Members of types that have no bodies (abstract and in-interface)
228         */
229        private void createAbstractTypeMemberRules() {
230                // function, property, ... without block (in interface or in abstract
231                // class)
232                interfaceMemberSkipBeforeNextMember(FUNCTION, IDENTIFIER)
233                                .createNode(EShallowEntityType.METHOD, SubTypeNames.FUNCTION, 1).endNode();
234
235                interfaceMemberSkipBeforeNextMember(CONSTRUCT)
236                                .createNode(EShallowEntityType.METHOD, SubTypeNames.CONSTRUCTOR, 0).endNode();
237
238                interfaceMemberSkipBeforeNextMember(PROPERTY, EnumSet.of(GET, SET), IDENTIFIER)
239                                .createNode(EShallowEntityType.METHOD, SubTypeNames.PROPERTY, 1).endNode();
240        }
241
242        /**
243         * Creates a RecognizerBase that matches an interface or type member that
244         * has the given match terms after the modifiers and before the first
245         * LPAREN. The RecognizerBase to before the next Member or to the end of the
246         * interface/type.
247         */
248        private RecognizerBase<EGosuParserStates> interfaceMemberSkipBeforeNextMember(Object... memberMatchTerms) {
249                return inStateSkipModifiers(IN_INTERFACE, IN_TYPE).markStart().sequence(memberMatchTerms).skipTo(LPAREN)
250                                .skipToWithNesting(RPAREN, LPAREN, RPAREN)
251                                .skipBeforeWithNesting(MEMBER_TERMINATOR_TOKENS, ALL_OPEN_PAREN, ALL_CLOSE_PAREN);
252        }
253
254        /**
255         * Non-abstract members of Types
256         */
257        private void createTypeMemberWithBodyRules() {
258                // function
259                skipToTypeMemberBody(inStateSkipModifiers(IN_TYPE).markStart().sequence(FUNCTION, IDENTIFIER).skipTo(LPAREN)
260                                .skipToWithNesting(RPAREN, LPAREN, RPAREN))
261                                                .createNode(EShallowEntityType.METHOD, SubTypeNames.FUNCTION, 1).parseUntil(IN_METHOD)
262                                                .sequence(RBRACE).endNode();
263                // construct
264                skipToTypeMemberBody(inStateSkipModifiers(IN_TYPE).markStart().sequence(CONSTRUCT).skipTo(LPAREN)
265                                .skipToWithNesting(RPAREN, LPAREN, RPAREN))
266                                                .createNode(EShallowEntityType.METHOD, SubTypeNames.CONSTRUCTOR, 0).parseUntil(IN_METHOD)
267                                                .sequence(RBRACE).endNode();
268                // property
269                skipToTypeMemberBody(
270                                inStateSkipModifiers(IN_TYPE).markStart().sequence(PROPERTY).sequence(EnumSet.of(GET, SET), IDENTIFIER))
271                                                .createNode(EShallowEntityType.METHOD, SubTypeNames.PROPERTY, 1).parseUntil(IN_METHOD)
272                                                .sequence(RBRACE).endNode();
273        }
274
275        /**
276         * Skips anything before the LBRACE starting the body. If there is a
277         * {@link #MEMBER_TERMINATOR_TOKENS} before, this does not match.
278         */
279        private static RecognizerBase<EGosuParserStates> skipToTypeMemberBody(
280                        RecognizerBase<EGosuParserStates> recognizerBase) {
281                Set<ETokenType> typeMemberTerminatorsAndLbrace = CollectionUtils.unionSet(MEMBER_TERMINATOR_TOKENS,
282                                Collections.singleton(LBRACE));
283                return recognizerBase.skipBefore(typeMemberTerminatorsAndLbrace).sequence(LBRACE);
284        }
285
286        /** Creates parsing rules for statements. */
287        private void createStatementRules() {
288
289                // empty statement
290                inAnyState().sequence(SEMICOLON).createNode(EShallowEntityType.STATEMENT, SubTypeNames.EMPTY_STATEMENT)
291                                .endNode();
292
293                createElseIfRule();
294                createSimpleBlockRules();
295                createSwitchCaseRules();
296
297                // var statement
298                RecognizerBase<EGosuParserStates> varStatementRecognizer = emptyRecognizer();
299                varStatementRecognizer.repeated(MODIFIERS).markStart().sequence(VAR, IDENTIFIER);
300                inState(IN_METHOD, TOP_LEVEL, IN_TYPE).preCondition(varStatementRecognizer)
301                                .subRecognizer(
302                                                new GosuSimpleStatementRecognizer(EShallowEntityType.STATEMENT, SubTypeNames.LOCAL_VARIABLE), 1,
303                                                1)
304                                .endNode();
305
306                // simple statement
307                inState(IN_METHOD, TOP_LEVEL, IN_TYPE)
308                                .sequenceBefore(EnumSet.of(ETokenClass.OPERATOR, ETokenClass.LITERAL, ETokenClass.IDENTIFIER,
309                                                ETokenClass.KEYWORD))
310                                .subRecognizer(
311                                                new GosuSimpleStatementRecognizer(EShallowEntityType.STATEMENT, SubTypeNames.SIMPLE_STATEMENT),
312                                                1, 1)
313                                .endNode();
314
315                // in a lambda, there could be a statement such as this:
316                // " ... -> (a == b) and not(c)
317                // therefore, we have to start parsing simple statements on LPAREN, too
318                inState(IN_METHOD).sequenceBefore(LPAREN)
319                                .subRecognizer(
320                                                new GosuSimpleStatementRecognizer(EShallowEntityType.STATEMENT, SubTypeNames.SIMPLE_STATEMENT),
321                                                1, 1)
322                                .endNode();
323        }
324
325        /** Creates the rule for "else if" construct. */
326        private void createElseIfRule() {
327                RecognizerBase<EGosuParserStates> elseIfAlternative = inState(IN_METHOD, TOP_LEVEL).sequence(ELSE, IF)
328                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.ELSE_IF)
329                                .skipNested(LPAREN, RPAREN, getSubExpressionRecognizer());
330                endWithPossibleContinuation(elseIfAlternative.sequence(LBRACE).parseUntil(IN_METHOD).sequence(RBRACE),
331                                EnumSet.of(ELSE));
332                endWithPossibleContinuation(elseIfAlternative.parseOnce(IN_METHOD), EnumSet.of(ELSE));
333        }
334
335        /** Creates simple block rules. */
336        private void createSimpleBlockRules() {
337                createBlockRuleWithContinuation(EnumSet.of(WHILE, FOR, SWITCH, WITH), null, true);
338                createBlockRuleWithContinuation(EnumSet.of(ELSE, FINALLY), null, false);
339                createBlockRuleWithContinuation(EnumSet.of(IF), EnumSet.of(ELSE), true);
340                createBlockRuleWithContinuation(EnumSet.of(TRY, CATCH), EnumSet.of(CATCH, FINALLY), true);
341        }
342
343        /** Creates rules for switch/case. */
344        private void createSwitchCaseRules() {
345                inAnyState().sequence(CASE, LITERALS_AND_IDENTIFIERS, COLON)
346                                .createNode(EShallowEntityType.META, new int[] { 0, 1 }).endNode();
347                inAnyState().sequence(CASE).skipToWithNesting(COLON, LPAREN, RPAREN).createNode(EShallowEntityType.META, 0)
348                                .endNode();
349
350                inAnyState().sequence(DEFAULT, COLON).createNode(EShallowEntityType.META, 0).endNode();
351        }
352
353        /**
354         * Creates a rule for recognizing a statement starting with a single
355         * keyword, optionally followed by an expression in parentheses, and
356         * followed by a block or a single statement.
357         *
358         * @param continuationTokens
359         *            list of tokens that indicate a continued statement if
360         *            encountered after the block. May be null.
361         */
362        private void createBlockRuleWithContinuation(EnumSet<ETokenType> startTokens,
363                        EnumSet<ETokenType> continuationTokens, boolean canBeFollowedByParentheses) {
364                RecognizerBase<EGosuParserStates> alternative = inAnyState().sequence(startTokens)
365                                .createNode(EShallowEntityType.STATEMENT, 0);
366                if (canBeFollowedByParentheses) {
367                        alternative = alternative.skipNested(LPAREN, RPAREN, getSubExpressionRecognizer());
368                }
369
370                endWithPossibleContinuation(alternative.sequence(LBRACE).parseUntil(IN_METHOD).sequence(RBRACE),
371                                continuationTokens);
372                endWithPossibleContinuation(alternative.parseOnce(IN_METHOD), continuationTokens);
373        }
374
375        /**
376         * Create rules for shallow entities that can occur in expressions e.g.,
377         * lambdas.
378         */
379        protected void createSubExpressionRules() {
380                // lambda expressions
381                RecognizerBase<EGosuParserStates> lambdaStart = inState(IN_EXPRESSION).sequence(BACKSLASH).skipTo(ARROW)
382                                .createNode(EShallowEntityType.METHOD, SubTypeNames.LAMBDA);
383                lambdaStart.sequence(LBRACE).parseUntil(IN_METHOD).sequence(RBRACE).endNode();
384                lambdaStart.parseOnce(IN_METHOD).endNode();
385        }
386
387        /** Returns a sub expression recognizer */
388        protected RecognizerBase<EGosuParserStates> getSubExpressionRecognizer() {
389                RecognizerBase<EGosuParserStates> subExpressionRule = emptyRecognizer();
390                // the BACKSLASH should trigger the lambda parsing rules
391                subExpressionRule.sequenceBefore(BACKSLASH).parseOnce(EGosuParserStates.IN_EXPRESSION);
392                return subExpressionRule;
393        }
