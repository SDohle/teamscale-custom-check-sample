095/**
096 * Parser for the language IEC 61131-3 Structured Text.
097 *
098 * This parser does not (yet) support PROGRAM and CONFIGURATION (see
099 * https://en.wikipedia.org/wiki/Structured_text), as we have no example code
100 * for this.
101 *
102 * @author $Author: hummelb $
103 * @version $Rev: 52657 $
104 * @ConQAT.Rating GREEN Hash: F4EC08559C41F1CFEC6B61E0C9D9C1D4
105 */
106public class Iec61131ShallowParser extends
107                ShallowParserBase<EIec61131ParserStates> {
108
109        /** The states used in this parser. */
110        public static enum EIec61131ParserStates {
111
112                /** Top-level state. */
113                TOP_LEVEL,
114
115                /** In variable declarations. */
116                IN_VARS,
117
118                /** In methods, algorithms, etc. */
119                IN_METHOD,
120
121                /** In case statement. */
122                IN_CASE,
123
124                /** System var decl of SV Files. */
125                IN_SYSTEM_VAR_DECL
126        }
127
128        /** A set of all token types that mark the beginning of a simple statement. */
129        private static final EnumSet<ETokenType> SIMPLE_STATEMENT_KEYWORDS = EnumSet
130                        .of(IDENTIFIER, RETURN, EXIT);
131
132        /** A set of all token types that can be used as case literals. */
133        private static final EnumSet<ETokenType> CASE_LITERALS = EnumSet.of(
134                        IDENTIFIER, INTEGER_LITERAL, FLOATING_POINT_LITERAL);
135
136        /** Constructor. */
137        public Iec61131ShallowParser() {
138                super(EIec61131ParserStates.class, EIec61131ParserStates.TOP_LEVEL);
139
140                createTopLevelRules();
141                createInVariableRules();
142                createInMethodRules();
143                createInSystemVarDeclRules();
144        }
145
146        /** Creates the rules for parsing top-level elements. */
147        private void createTopLevelRules() {
148                inState(EIec61131ParserStates.TOP_LEVEL).sequence(IMPORT)
149                                .createNode(EShallowEntityType.META, 0).skipTo(END_IMPORT)
150                                .endNode();
151
152                inState(EIec61131ParserStates.TOP_LEVEL)
153                                .sequence(ALGORITHM_BLOCK, IDENTIFIER)
154                                .optional(WITH, INTERFACE, IDENTIFIER)
155                                .createNode(EShallowEntityType.MODULE, 0, 1)
156                                .parseUntil(EIec61131ParserStates.TOP_LEVEL)
157                                .sequence(END_ALGORITHM_BLOCK).optional(SEMICOLON).endNode();
158
159                inState(EIec61131ParserStates.TOP_LEVEL)
160                                .sequence(
161                                                EnumSet.of(VAR, SYSTEM_VAR, SYSTEM_VAR_IN, VAR_EXTERNAL))
162                                .createNode(EShallowEntityType.META, 0)
163                                .parseUntil(EIec61131ParserStates.IN_VARS).sequence(END_VAR)
164                                .optional(SEMICOLON).endNode();
165
166                inState(EIec61131ParserStates.TOP_LEVEL).sequence(SYSTEM_OBJECT)
167                                .createNode(EShallowEntityType.META, 0)
168                                .parseUntil(EIec61131ParserStates.IN_VARS).sequence(END_OBJECT)
169                                .endNode();
170
171                inState(EIec61131ParserStates.TOP_LEVEL)
172                                .sequence(ALGORITHM, IDENTIFIER)
173                                .createNode(EShallowEntityType.METHOD, 0, 1)
174                                .parseUntil(EIec61131ParserStates.IN_METHOD)
175                                .sequence(END_ALGORITHM).optional(SEMICOLON).endNode();
176
177                inState(EIec61131ParserStates.TOP_LEVEL)
178                                .sequence(FUNCTION_BLOCK, IDENTIFIER)
179                                .createNode(EShallowEntityType.METHOD, 0, 1)
180                                .parseUntil(EIec61131ParserStates.IN_METHOD)
181                                .sequence(END_FUNCTION_BLOCK).optional(SEMICOLON).endNode();
182
183                inState(EIec61131ParserStates.TOP_LEVEL)
184                                .sequence(PROCESS_ALGORITHM, IDENTIFIER, ON, IDENTIFIER)
185                                .createNode(EShallowEntityType.METHOD, 0, 1)
186                                .optional(AUTOSTART)
187                                .parseUntil(EIec61131ParserStates.IN_METHOD)
188                                .sequence(END_ALGORITHM).optional(SEMICOLON).endNode();
189
190                inState(EIec61131ParserStates.TOP_LEVEL)
191                                .sequence(EVENT_ALGORITHM, IDENTIFIER)
192                                .createNode(EShallowEntityType.METHOD, 0, 1)
193                                .skipTo(WITH, IDENTIFIER)
194                                .parseUntil(EIec61131ParserStates.IN_METHOD)
195                                .sequence(END_ALGORITHM).optional(SEMICOLON).endNode();
196
197                inState(EIec61131ParserStates.TOP_LEVEL)
198                                .sequence(EnumSet.of(EVENT_ALGORITHM, POSTUPDATE_ALGORITHM),
199                                                IDENTIFIER).createNode(EShallowEntityType.METHOD, 0, 1)
200                                .skipTo(WITH, IDENTIFIER).optional(XOR)
201                                .repeated(COMMA, IDENTIFIER)
202                                .parseUntil(EIec61131ParserStates.IN_METHOD)
203                                .sequence(END_ALGORITHM).optional(SEMICOLON).endNode();
204
205                inState(EIec61131ParserStates.TOP_LEVEL)
206                                .sequence(EnumSet.of(FUNCTION, PLAUSIBILITY_FUNCTION),
207                                                IDENTIFIER).createNode(EShallowEntityType.METHOD, 0, 1)
208                                .skipTo(COLON).sequence(IDENTIFIER)
209                                .parseUntil(EIec61131ParserStates.IN_METHOD)
210                                .sequence(END_FUNCTION).optional(SEMICOLON).endNode();
211
212                inState(EIec61131ParserStates.TOP_LEVEL).sequence(SYSTEM_VAR_DECL)
213                                .createNode(EShallowEntityType.TYPE, 0)
214                                .parseUntil(EIec61131ParserStates.IN_SYSTEM_VAR_DECL)
215                                .sequence(END).endNode();
216        }
217
218        /** Creates the rules for parsing within variable blocks. */
219        private void createInVariableRules() {
220                inState(EIec61131ParserStates.IN_VARS)
221                                .sequence(IDENTIFIER, COLON)
222                                .createNode(EShallowEntityType.ATTRIBUTE,
223                                                SubTypeNames.VARIABLE, 0).skipTo(SEMICOLON).endNode();
224        }
225
226        /** Creates the rules for parsing within methods (statements). */
227        private void createInMethodRules() {
228                inState(EIec61131ParserStates.IN_METHOD)
229                                .sequence(EnumSet.of(VAR, VAR_INPUT, VAR_TEMP, VAR_IN_OUT))
230                                .createNode(EShallowEntityType.META, 0).optional(CONSTANT)
231                                .parseUntil(EIec61131ParserStates.IN_VARS).sequence(END_VAR)
232                                .endNode();
233
234                createCaseRule();
235                createLoopRules();
236                createIfElseRules();
237
238                inState(EIec61131ParserStates.IN_METHOD)
239                                .sequence(
240                                                EnumSet.of(GO_ON_TRANSITION, TRANSITION,
241                                                                EXIT_TRANSITION, GO_ON_EXIT_TRANSITION))
242                                .createNode(EShallowEntityType.STATEMENT, 0)
243                                .skipTo(END_TRANSITION).optional(SEMICOLON).endNode();
244
245                inState(EIec61131ParserStates.IN_METHOD)
246                                .sequence(ACTION, IDENTIFIER, COLON)
247                                .createNode(EShallowEntityType.STATEMENT, 0, 1)
248                                .parseUntil(EIec61131ParserStates.IN_METHOD)
249                                .sequence(END_ACTION).optional(SEMICOLON).endNode();
250
251                inState(EIec61131ParserStates.IN_METHOD)
252                                .sequence(EnumSet.of(STEP, INITIAL_STEP), IDENTIFIER, COLON)
253                                .createNode(EShallowEntityType.STATEMENT, 0, 1)
254                                .parseUntil(EIec61131ParserStates.IN_METHOD).sequence(END_STEP)
255                                .optional(SEMICOLON).endNode();
256
257                createSimpleStatementRules();
258        }
259
260        /** Creates the rules for parsing the case statement. */
261        private void createCaseRule() {
262                inStatementStates().sequence(CASE)
263                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(OF)
264                                .parseUntil(EIec61131ParserStates.IN_CASE).sequence(END_CASE)
265                                .optional(SEMICOLON).endNode();
266
267                inState(EIec61131ParserStates.IN_CASE).repeated(CASE_LITERALS, COMMA)
268                                .sequence(CASE_LITERALS, COLON)
269                                .createNode(EShallowEntityType.META, SubTypeNames.CASE_LABEL)
270                                .endNode();
271                inState(EIec61131ParserStates.IN_CASE).sequence(ELSE)
272                                .createNode(EShallowEntityType.META, SubTypeNames.CASE_ELSE)
273                                .endNode();
274        }
275
276        /** Returns the states that expect "plain" statements. */
277        private RecognizerBase<EIec61131ParserStates> inStatementStates() {
278                return inState(EIec61131ParserStates.IN_METHOD,
279                                EIec61131ParserStates.IN_CASE);
280        }
281
282        /** Creates the rules for parsing loops. */
283        private void createLoopRules() {
284                inStatementStates().sequence(FOR)
285                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DO)
286                                .parseUntil(EIec61131ParserStates.IN_METHOD).sequence(END_FOR)
287                                .optional(SEMICOLON).endNode();
288
289                inStatementStates().sequence(WHILE)
290                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(DO)
291                                .parseUntil(EIec61131ParserStates.IN_METHOD)
292                                .sequence(END_WHILE).optional(SEMICOLON).endNode();
293
294                inStatementStates().sequence(REPEAT)
295                                .createNode(EShallowEntityType.STATEMENT, 0)
296                                .parseUntil(EIec61131ParserStates.IN_METHOD).sequence(UNTIL)
297                                .skipTo(SEMICOLON).endNode();
298        }
299
300        /** Create rules for parsing if/else structures. */
301        private void createIfElseRules() {
302                RecognizerBase<EIec61131ParserStates> ifAlternative = inStatementStates()
303                                .sequence(EnumSet.of(IF, ELSIF))
304                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(THEN)
305                                .parseUntil(EIec61131ParserStates.IN_METHOD);
306                ifAlternative.sequenceBefore(EnumSet.of(ELSE, ELSIF))
307                                .endNodeWithContinuation();
308                ifAlternative.sequence(END_IF).optional(SEMICOLON).endNode();
309
310                // else is used in "if" (only use IN_METHOD here, as IN_CASE case is
311                // handled elsewhere)
312                RecognizerBase<EIec61131ParserStates> elseAlternative = inState(
313                                EIec61131ParserStates.IN_METHOD).sequence(ELSE)
314                                .createNode(EShallowEntityType.STATEMENT, 0)
315                                .parseUntil(EIec61131ParserStates.IN_METHOD);
316                elseAlternative.sequenceBefore(END_CASE).endNode();
317                elseAlternative.sequence(END_IF, SEMICOLON).endNode();
318        }
319
320        /** Create rules for parsing simple statements within methods. */
321        private void createSimpleStatementRules() {
322                // simple statement
323                inStatementStates()
324                                .sequence(SIMPLE_STATEMENT_KEYWORDS)
325                                .createNode(EShallowEntityType.STATEMENT,
326                                                SubTypeNames.SIMPLE_STATEMENT, 0).skipTo(SEMICOLON)
327                                .endNode();
328
329                // empty statement
330                inStatementStates()
331                                .sequence(SEMICOLON)
332                                .createNode(EShallowEntityType.STATEMENT,
333                                                SubTypeNames.EMPTY_STATEMENT).endNode();
334        }
335
336        /** Creates the rules for the system var decl section of SV files. */
337        private void createInSystemVarDeclRules() {
338                // we parse these parts as attributes, as this makes it less likely to
339                // interface with structuring metrics.
340                inState(EIec61131ParserStates.IN_SYSTEM_VAR_DECL)
341                                .sequence(IDENTIFIER)
342                                .createNode(EShallowEntityType.ATTRIBUTE,
343                                                SubTypeNames.IEC_SV_VARIABLE, 0)
344                                .skipBefore(EnumSet.of(SEMICOLON, ELEMENT)).optional(SEMICOLON)
345                                .endNode();
346                inState(EIec61131ParserStates.IN_SYSTEM_VAR_DECL)
347                                .sequence(ELEMENT)
348                                .createNode(EShallowEntityType.ATTRIBUTE,
349                                                SubTypeNames.IEC_SV_ELEMENT)
350                                .skipBefore(EnumSet.of(SEMICOLON, ELEMENT)).optional(SEMICOLON)
351                                .endNode();
352        }
