090/**
091 * Shallow parser for Java.
092 * <p>
093 * What this parser does and does not:
094 * <ul>
095 * <li>The parser recognizes types (classes, enums, interfaces), methods and
096 * attributes, and individual statements.</li>
097 * <li>It recognizes the nesting of statements (e.g. in loops), but does not
098 * parse into the statements. For example, it recognizes an if-statement and
099 * provides the list of sub-statements, but does not provide direct access to
100 * the if-condition.</li>
101 * <li>Import and package statements are parsed as meta information.</li>
102 * <li>Annotations are recognized as meta information, but only annotations at
103 * types and methods. Annotations at parameters are not parsed, as the parser
104 * does not parse into the parameter list of methods.</li>
105 * <li>The parser does not recognize anonymous classes. These are treated as a
106 * single long statement or attribute. Inner classes, however, are parsed
107 * correctly.</li>
108 * <li>The parser can deal with multiple classes in a single file.</li>
109 * </ul>
110 *
111 * @author $Author: hummelb $
112 * @version $Rev: 52657 $
113 * @ConQAT.Rating GREEN Hash: 31717C70FEE7B90D70F6CC889C78A775
114 */
115public class JavaShallowParser extends CStyleShallowParserBase {

117        /** {@inheritDoc} */
118        @Override
119        protected void createMetaRules() {
120                // imports and package
121                inAnyState().markStart().sequence(EnumSet.of(IMPORT, PACKAGE))
122                                .skipTo(SEMICOLON)
123                                .createNode(EShallowEntityType.META, 0, new Region(1, -2))
124                                .endNode();
125
126                // annotations; the spec allows both whitespace and comments between the
127                // '@' and the identifier, but as we filter comments before-hand, this
128                // is not an issue.
129                inAnyState()
130                                .sequence(AT_OPERATOR, IDENTIFIER)
131                                .repeated(DOT, IDENTIFIER)
132                                .createNode(EShallowEntityType.META, "annotation",
133                                                new Region(1, -1)).skipNested(LPAREN, RPAREN).endNode();
134
135                super.createMetaRules();
136        }

142        }

150        /** {@inheritDoc} */
151        @Override
152        protected void createClassElementsRules() {
153                createMethodRule();
154
155                // simple enum literals (without parentheses)
156                RecognizerBase<EGenericParserStates> enumLiteral = inState(IN_TYPE)
157                                .sequence(IDENTIFIER)
158                                .sequenceBefore(EnumSet.of(SEMICOLON, COMMA, LBRACE))
159                                .createNode(EShallowEntityType.ATTRIBUTE, "enum literal", 0);
160                enumLiteral.sequence(EnumSet.of(SEMICOLON, COMMA)).endNode();
161                enumLiteral.sequence(LBRACE).parseUntil(IN_TYPE)
162                                .sequence(RBRACE, EnumSet.of(SEMICOLON, COMMA)).endNode();
163
164                createConstructorRule();
165
166                // attributes (must be after method, as this would also match methods)
167                typePatternInState(IN_TYPE)
168                                .sequence(IDENTIFIER)
169                                .createNode(EShallowEntityType.ATTRIBUTE, "attribute", -1)
170                                .skipToWithNesting(SEMICOLON, LBRACE, RBRACE,
171                                                getSubExpressionRecognizer()).endNode();
172
173                // static initializer
174                inState(IN_TYPE)
175                                .sequence(STATIC, LBRACE)
176                                .createNode(EShallowEntityType.METHOD, "static initializer",
177                                                "<sinit>").parseUntil(IN_METHOD).sequence(RBRACE)
178                                .endNode();
179
180                // non-static initializer
181                inState(IN_TYPE)
182                                .sequence(LBRACE)
183                                .createNode(EShallowEntityType.METHOD,
184                                                "non-static initializer", "<init>")
185                                .parseUntil(IN_METHOD).sequence(RBRACE).endNode();
186        }
187
188        /** Recognizes methods. */
189        private void createMethodRule() {
190                RecognizerBase<EGenericParserStates> methodAlternative = typePatternInState(
191                                IN_TYPE).markStart().sequence(IDENTIFIER, LPAREN)
192                                .skipToWithNesting(RPAREN, LPAREN, RPAREN)
193                                .skipBefore(EnumSet.of(LBRACE, SEMICOLON, DEFAULT));
194                methodAlternative.sequence(LBRACE)
195                                .createNode(EShallowEntityType.METHOD, "method", 0)
196                                .parseUntil(IN_METHOD).sequence(RBRACE).endNode();
197                // annotations
198                methodAlternative.sequence(DEFAULT)
199                                .createNode(EShallowEntityType.METHOD, "abstract method", 0)
200                                .skipToWithNesting(SEMICOLON, LBRACE, RBRACE).endNode();
201                methodAlternative.sequence(SEMICOLON)
202                                .createNode(EShallowEntityType.METHOD, "abstract method", 0)
203                                .endNode();
204        }
205
206        /** Recognizes constructors or enum literals with parentheses. */
207        private void createConstructorRule() {
208                RecognizerBase<EGenericParserStates> constructorOrEnum = inState(
209                                IN_TYPE).optional(EnumSet.of(PUBLIC, PRIVATE, PROTECTED))
210                                .markStart().sequence(IDENTIFIER, LPAREN)
211                                .skipToWithNesting(RPAREN, LPAREN, RPAREN);
212                constructorOrEnum.sequence(EnumSet.of(SEMICOLON, COMMA))
213                                .createNode(EShallowEntityType.ATTRIBUTE, "enum literal", 0)
214                                .endNode();
215                RecognizerBase<EGenericParserStates> constructorOrEnum2 = constructorOrEnum
216                                .skipTo(LBRACE);
217                // at this point this still could be a constructor or an enum literal.
218                // Heuristic used: If we encounter the '@' sign (from an annotation) or
219                // a method modifier, we assume enum literal.
220                constructorOrEnum2
221                                .sequenceBefore(
222                                                EnumSet.of(AT_OPERATOR, PUBLIC, PRIVATE, PROTECTED))
223                                .createNode(EShallowEntityType.ATTRIBUTE, "enum literal", 0)
224                                .parseUntil(IN_TYPE).sequence(RBRACE)
225                                .sequence(EnumSet.of(SEMICOLON, COMMA)).endNode();
226                constructorOrEnum2
227                                .createNode(EShallowEntityType.METHOD, "constructor", 0)
228                                .parseUntil(IN_METHOD).sequence(RBRACE).endNode();
229        }

252        protected void createCaseRule() {
253                super.createCaseRule();
254
255                // Java also allows fully/partially qualified constants
256                inState(IN_METHOD).markStart().sequence(CASE).repeated(IDENTIFIER, DOT)

262        /** {@inheritDoc} */
263        @Override
264        protected RecognizerBase<EGenericParserStates> typePattern(
265                        RecognizerBase<EGenericParserStates> currentState) {
266                EnumSet<ETokenType> modifierKeywords = EnumSet.of(STATIC, FINAL,
267                                PRIVATE, PROTECTED, PUBLIC, ABSTRACT, NATIVE, SYNCHRONIZED,
268                                TRANSIENT, VOLATILE);
269                EnumSet<ETokenType> typeNames = EnumSet.of(IDENTIFIER, VOID, BYTE,
270                                SHORT, INT, LONG, FLOAT, DOUBLE, CHAR, BOOLEAN);
271                return currentState.repeated(modifierKeywords).skipNested(LT, GT)
272                                .repeated(IDENTIFIER, DOT).optional(THIS, DOT)
273                                .sequence(typeNames).skipNested(LT, GT)
274                                .skipAny(EnumSet.of(LBRACK, RBRACK));
275        }
276
277        /** {@inheritDoc} */
278        @Override
279        protected void createSubExpressionRules() {
280                inState(IN_EXPRESSION)
281                                .sequence(NEW)
282                                .markStart()
283                                .repeated(IDENTIFIER, DOT)
284                                .sequence(IDENTIFIER)
285                                .createNode(EShallowEntityType.TYPE,
286                                                SubTypeNames.ANONYMOUS_CLASS, new Region(0, -1))
287                                .skipNested(LT, GT)
288                                .skipNested(LPAREN, RPAREN, getSubExpressionRecognizer())
289                                .sequence(LBRACE).parseUntil(IN_TYPE).sequence(RBRACE)
290                                .endNode();
291        }
292
293        /** {@inheritDoc} */
294        @Override
295        protected RecognizerBase<EGenericParserStates> getSubExpressionRecognizer() {
296                return new JavaAnonymousClassRecognizer();
297        }
