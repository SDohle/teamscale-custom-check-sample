092/**
093 * Shallow parser for Java.
094 * <p>
095 * What this parser does and does not:
096 * <ul>
097 * <li>The parser recognizes types (classes, enums, interfaces), methods and
098 * attributes, and individual statements.</li>
099 * <li>It recognizes the nesting of statements (e.g. in loops), but does not
100 * parse into the statements. For example, it recognizes an if-statement and
101 * provides the list of sub-statements, but does not provide direct access to
102 * the if-condition.</li>
103 * <li>Import and package statements are parsed as meta information.</li>
104 * <li>Annotations are recognized as meta information, but only annotations at
105 * types and methods. Annotations at parameters are not parsed, as the parser
106 * does not parse into the parameter list of methods.</li>
107 * <li>The parser does not recognize anonymous classes. These are treated as a
108 * single long statement or attribute. Inner classes, however, are parsed
109 * correctly.</li>
110 * <li>The parser can deal with multiple classes in a single file.</li>
111 * </ul>
112 */
113public class JavaShallowParser extends CStyleShallowParserBase {

115        /** {@inheritDoc} */
116        @Override
117        protected void createMetaRules() {
118                // imports and package
119                inAnyState().sequence(IMPORT, STATIC).skipTo(SEMICOLON)
120                                .createNode(EShallowEntityType.META, SubTypeNames.STATIC_IMPORT, new Region(2, -2)).endNode();
121                inAnyState().sequence(EnumSet.of(IMPORT, PACKAGE)).skipTo(SEMICOLON)
122                                .createNode(EShallowEntityType.META, 0, new Region(1, -2)).endNode();
123
124                // annotations; the spec allows both whitespace and comments between the
125                // '@' and the identifier, but as we filter comments before-hand, this
126                // is not an issue.
127                inAnyState().sequence(AT_OPERATOR, IDENTIFIER).repeated(DOT, IDENTIFIER)
128                                .createNode(EShallowEntityType.META, "annotation", new Region(1, -1)).skipNested(LPAREN, RPAREN)
129                                .endNode();
130
131                super.createMetaRules();
132        }

138        }

146        /** {@inheritDoc} */
147        @Override
148        protected void createClassElementsRules() {
149                createMethodRule();
150
151                // simple enum literals (without parentheses)
152                RecognizerBase<EGenericParserStates> enumLiteral = inState(IN_TYPE).sequence(IDENTIFIER)
153                                .sequenceBefore(EnumSet.of(SEMICOLON, COMMA, LBRACE))
154                                .createNode(EShallowEntityType.ATTRIBUTE, "enum literal", 0);
155                enumLiteral.sequence(EnumSet.of(SEMICOLON, COMMA)).endNode();
156                enumLiteral.sequence(LBRACE).parseUntil(IN_TYPE).sequence(RBRACE, EnumSet.of(SEMICOLON, COMMA)).endNode();
157
158                createConstructorRule();
159
160                // attributes (must be after method, as this would also match methods)
161                typePatternInState(IN_TYPE).sequence(IDENTIFIER).createNode(EShallowEntityType.ATTRIBUTE, "attribute", -1)
162                                .skipToWithNesting(SEMICOLON, LBRACE, RBRACE, getSubExpressionRecognizer()).endNode();
163
164                // static initializer
165                inState(IN_TYPE).sequence(STATIC, LBRACE).createNode(EShallowEntityType.METHOD, "static initializer", "<sinit>")
166                                .parseUntil(IN_METHOD).sequence(RBRACE).endNode();
167
168                // non-static initializer
169                inState(IN_TYPE).sequence(LBRACE).createNode(EShallowEntityType.METHOD, "non-static initializer", "<init>")
170                                .parseUntil(IN_METHOD).sequence(RBRACE).endNode();
171        }
172
173        /** Recognizes methods. */
174        private void createMethodRule() {
175                RecognizerBase<EGenericParserStates> methodAlternative = typePatternInState(IN_TYPE).markStart()
176                                .sequence(IDENTIFIER, LPAREN).skipToWithNesting(RPAREN, LPAREN, RPAREN)
177                                .skipBefore(EnumSet.of(LBRACE, SEMICOLON, DEFAULT));
178                methodAlternative.sequence(LBRACE).createNode(EShallowEntityType.METHOD, "method", 0).parseUntil(IN_METHOD)
179                                .sequence(RBRACE).endNode();
180                // annotations
181                methodAlternative.sequence(DEFAULT).createNode(EShallowEntityType.METHOD, "abstract method", 0)
182                                .skipToWithNesting(SEMICOLON, LBRACE, RBRACE).endNode();
183                methodAlternative.sequence(SEMICOLON).createNode(EShallowEntityType.METHOD, "abstract method", 0).endNode();
184        }
185
186        /** Recognizes constructors or enum literals with parentheses. */
187        private void createConstructorRule() {
188                RecognizerBase<EGenericParserStates> constructorOrEnum = inState(IN_TYPE)
189                                .optional(EnumSet.of(PUBLIC, PRIVATE, PROTECTED)).markStart().sequence(IDENTIFIER, LPAREN)
190                                .skipToWithNesting(RPAREN, LPAREN, RPAREN);
191                constructorOrEnum.sequence(EnumSet.of(SEMICOLON, COMMA))
192                                .createNode(EShallowEntityType.ATTRIBUTE, "enum literal", 0).endNode();
193                RecognizerBase<EGenericParserStates> constructorOrEnum2 = constructorOrEnum.skipTo(LBRACE);
194                // at this point this still could be a constructor or an enum literal.
195                // Heuristic used: If we encounter the '@' sign (from an annotation) or
196                // a method modifier, we assume enum literal.
197                constructorOrEnum2.sequenceBefore(EnumSet.of(AT_OPERATOR, PUBLIC, PRIVATE, PROTECTED))
198                                .createNode(EShallowEntityType.ATTRIBUTE, "enum literal", 0).parseUntil(IN_TYPE).sequence(RBRACE)
199                                .sequence(EnumSet.of(SEMICOLON, COMMA)).endNode();
200                constructorOrEnum2.createNode(EShallowEntityType.METHOD, "constructor", 0).parseUntil(IN_METHOD)
201                                .sequence(RBRACE).endNode();
202        }

225        protected void createCaseRule() {
226                super.createCaseRule();

229                inState(IN_METHOD).markStart().sequence(CASE).repeated(IDENTIFIER, DOT).sequence(IDENTIFIER, COLON)
230                                .createNode(EShallowEntityType.META, 0, new Region(1, -2)).endNode();

233        /** {@inheritDoc} */
234        @Override
235        protected RecognizerBase<EGenericParserStates> typePattern(RecognizerBase<EGenericParserStates> currentState) {
236                EnumSet<ETokenType> modifierKeywords = EnumSet.of(STATIC, FINAL, PRIVATE, PROTECTED, PUBLIC, ABSTRACT, NATIVE,
237                                SYNCHRONIZED, TRANSIENT, VOLATILE);
238                EnumSet<ETokenType> typeNames = EnumSet.of(IDENTIFIER, VOID, BYTE, SHORT, INT, LONG, FLOAT, DOUBLE, CHAR,
239                                BOOLEAN);
240                return currentState.repeated(modifierKeywords).skipNested(LT, GT).repeated(IDENTIFIER, DOT).optional(THIS, DOT)
241                                .sequence(typeNames).skipNested(LT, GT).skipAny(EnumSet.of(LBRACK, RBRACK));
242        }
243
244        /** {@inheritDoc} */
245        @Override
246        protected void createSubExpressionRules() {
247                inState(IN_EXPRESSION).sequence(NEW).markStart().repeated(IDENTIFIER, DOT).sequence(IDENTIFIER)
248                                .createNode(EShallowEntityType.TYPE, SubTypeNames.ANONYMOUS_CLASS, new Region(0, -1)).skipNested(LT, GT)
249                                .skipNested(LPAREN, RPAREN, getSubExpressionRecognizer()).sequence(LBRACE).parseUntil(IN_TYPE)
250                                .sequence(RBRACE).endNode();
251
252                createLambdaWithArrowRules(ARROW);
253        }
254
255        /** {@inheritDoc} */
256        @Override
257        protected RecognizerBase<EGenericParserStates> getSubExpressionRecognizer() {
258                return new JavaAnonymousClassAndLambdaRecognizer();
259        }
