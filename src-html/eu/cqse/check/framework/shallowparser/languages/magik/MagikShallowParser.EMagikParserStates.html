080/**
081 * Shallow parser for the language
082 * <a href="http://en.wikipedia.org/wiki/Magik_%28programming_language%29"
083 * >Magik</a>.
084 */
085public class MagikShallowParser extends ShallowParserBase<EMagikParserStates> {
086
087        /** The states used in this parser. */
088        public static enum EMagikParserStates {
089
090                /** Single state, as any construct can occur at any place. */
091                ANY
092        }
093
094        /** Constructor. */
095        public MagikShallowParser() {
096                super(EMagikParserStates.class, EMagikParserStates.ANY);
097
098                createMetaRules();
099                createMethodRules();
100                createProcedureAndHandlerRules();
101                createStatementRules();
102        }
103
104        /** Creates parsing rules for meta elements. */
105        private void createMetaRules() {
106                inState(ANY).sequence(PACKAGE, IDENTIFIER).createNode(EShallowEntityType.META, 0, 1).endNode();
107
108                inState(ANY).sequence(PRAGMA).createNode(EShallowEntityType.META, 0).skipNested(LPAREN, RPAREN).endNode();
109        }
110
111        /** Create rules for parsing methods. */
112        private void createMethodRules() {
113                RecognizerBase<EMagikParserStates> methodAlternative = inState(ANY)
114                                .repeated(EnumSet.of(PRIVATE, ITERATE, ABSTRACT)).markStart()
115                                .sequence(METHOD, IDENTIFIER, DOT, IDENTIFIER);
116
117                // assigner (<<)
118                methodAlternative.sequence(ASSIGN, IDENTIFIER)
119                                .createNode(EShallowEntityType.METHOD, "method", new Region(1, -2)).parseUntil(ANY).sequence(ENDMETHOD)
120                                .endNode();
121                // method with (); we include the "(" in the name of the method to avoid
122                // name clashes
123                methodAlternative.sequence(LPAREN).createNode(EShallowEntityType.METHOD, "method", new Region(1, -1))
124                                .skipTo(RPAREN).parseUntil(ANY).sequence(ENDMETHOD).endNode();
125                // method without ()
126                methodAlternative.createNode(EShallowEntityType.METHOD, "method", new Region(1, -1)).parseUntil(ANY)
127                                .sequence(ENDMETHOD).endNode();
128
129                inState(ANY).repeated(EnumSet.of(PRIVATE, ITERATE, ABSTRACT)).markStart().sequence(METHOD, IDENTIFIER, LBRACK)
130                                .createNode(EShallowEntityType.METHOD, "method", new Region(1, 3)).skipTo(RBRACK).parseUntil(ANY)
131                                .sequence(ENDMETHOD).endNode();
132        }
133
134        /** Creates rules for parsing procedures and handlers. */
135        private void createProcedureAndHandlerRules() {
136                // procedures
137                completeProcedure(inState(ANY).sequence(PROC).createNode(EShallowEntityType.METHOD, "anonymous procedure"));
138                completeProcedure(inState(ANY).repeated(EnumSet.of(GLOBAL, DYNAMIC, LOCAL)).markStart()
139                                .sequence(IDENTIFIER, ASSIGN, PROC).createNode(EShallowEntityType.METHOD, "procedure", 0));
140
141                // handlers
142                completeProcedure(inState(ANY).sequence(HANDLE).markStart().skipTo(WITH).sequence(PROC)
143                                .createNode(EShallowEntityType.METHOD, "handler", 0));
144
145        }
146
147        /** Completes a created procedure node. */
148        private void completeProcedure(RecognizerBase<EMagikParserStates> procedureStart) {
149                procedureStart.optional(AT_OPERATOR, IDENTIFIER).skipNested(LPAREN, RPAREN).parseUntil(ANY).sequence(END_PROC)
150                                .endNode();
151        }
152
153        /** Create rules for statements. */
154        private void createStatementRules() {
155                createIfElseRules();
156                createLoopRules();
157                createBlockRules();
158
159                // simple statement must be the very last
160                inState(ANY).subRecognizer(new MagikSimpleStatementRecognizer(), 1, 1).endNode();
161        }
162
163        /**
164         * Creats the rules for parsing blocks and block-like constructs (try, lock,
165         * etc.).
166         */
167        private void createBlockRules() {
168                // general blocks
169                inState(ANY).sequence(BLOCK).createNode(EShallowEntityType.STATEMENT, 0).parseUntil(ANY).sequence(END_BLOCK)
170                                .endNode();
171
172                // exception handling (try/when)
173                inState(ANY).sequence(TRY).createNode(EShallowEntityType.STATEMENT, 0).parseUntil(ANY).sequenceBefore(WHEN)
174                                .endNodeWithContinuation();
175                RecognizerBase<EMagikParserStates> whenAlternative = inState(ANY).sequence(WHEN, IDENTIFIER)
176                                .createNode(EShallowEntityType.STATEMENT, 0).parseUntil(ANY).sequenceBefore(EnumSet.of(WHEN, ENDTRY));
177                whenAlternative.sequence(ENDTRY).endNode();
178                whenAlternative.endNodeWithContinuation();
179
180                // catch blocks
181                inState(ANY).sequence(CATCH).optional(COLON).optional(EnumSet.of(IDENTIFIER, SELF))
182                                .createNode(EShallowEntityType.STATEMENT, 0).parseUntil(ANY).sequence(ENDCATCH).endNode();
183
184                // protect
185                inState(ANY).sequence(PROTECT).optional(LOCKING, EnumSet.of(IDENTIFIER, SELF))
186                                .createNode(EShallowEntityType.STATEMENT, 0).parseUntil(ANY).sequenceBefore(PROTECTED)
187                                .endNodeWithContinuation();
188                inState(ANY).sequence(PROTECTED).createNode(EShallowEntityType.STATEMENT, 0).parseUntil(ANY)
189                                .sequence(END_PROTECT).endNode();
190
191                // lock
192                inState(ANY).sequence(LOCK).optional(COLON).optional(EnumSet.of(IDENTIFIER, SELF))
193                                .createNode(EShallowEntityType.STATEMENT, 0).parseUntil(ANY).sequence(END_LOCK).endNode();
194        }
195
196        /** Creates the rules for parsing loops. */
197        private void createLoopRules() {
198                completeLoop(inState(ANY).sequence(LOOP));
199                completeLoop(inState(ANY).sequence(EnumSet.of(FOR, OVER)).skipTo(LOOP));
200                inState(ANY).sequence(FINALLY).createNode(EShallowEntityType.STATEMENT, 0).parseUntil(ANY).sequence(ENDLOOP)
201                                .endNode();
202        }
203
204        /** Creates the rules for parsing if/else statements. */
205        private void createIfElseRules() {
206                RecognizerBase<EMagikParserStates> ifAlternative = inState(ANY).sequence(EnumSet.of(IF, ELIF))
207                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(THEN).parseUntil(ANY)
208                                .sequenceBefore(EnumSet.of(ELIF, ELSE, END_IF));
209                ifAlternative.sequence(END_IF).endNode();
210                ifAlternative.endNodeWithContinuation();
211
212                inState(ANY).sequence(ELSE).createNode(EShallowEntityType.STATEMENT, 0).parseUntil(ANY).sequence(END_IF)
213                                .endNode();
214        }
215
216        /** Completes a loop construct, handling the finally case. */
217        private void completeLoop(RecognizerBase<EMagikParserStates> loopStart) {
218                RecognizerBase<EMagikParserStates> loopEndAlternative = loopStart.createNode(EShallowEntityType.STATEMENT, 0)
219                                .optional(AT_OPERATOR, IDENTIFIER).parseUntil(ANY).sequenceBefore(EnumSet.of(ENDLOOP, FINALLY));
220                loopEndAlternative.sequence(ENDLOOP).endNode();
221                loopEndAlternative.endNodeWithContinuation();
222        }
223
224        /** {@inheritDoc} */
225        @Override
226        protected boolean isFilteredToken(IToken token, IToken previousToken) {
227                return super.isFilteredToken(token, previousToken) || token.getType() == TERMINATE || token.getType() == EOL;
228        }
