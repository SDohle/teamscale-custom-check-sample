059/**
060 * Recognizer for simple statements in Magik. This handles the statements ending
061 * at the end of line and allows parsing into sub expressions, such as local
062 * procedures within statements.
063 * 
064 * @author $Author: hummelb $
065 * @version $Rev: 52657 $
066 * @ConQAT.Rating GREEN Hash: 6C6D875E5041F4B16DD027734623A94D
067 */
068public class MagikSimpleStatementRecognizer extends
069                LineBasedStatementRecognizerBase<EMagikParserStates> {
070
071        /**
072         * Token types that end a statement when at the end of line even though they
073         * are keywords.
074         */
075        private static final Set<ETokenType> VALID_STATEMENT_END_TOKEN_TYPES = EnumSet
076                        .of(LEAVE, RETURN, THROW, CONTINUE, SELF, CLONE, UNSET, END_BLOCK,
077                                        ENDCATCH, END_IF, END_LOCK, ENDLOOP, ENDMETHOD, END_PROC,
078                                        END_PROTECT, ENDTRY);
079
080        /** Token types that end a statement even in the same line. */
081        private static final Set<ETokenType> SAME_LINE_STATEMENT_DELIMITER_TOKEN_TYPES = EnumSet
082                        .of(ELIF, ELSE, END_BLOCK, ENDCATCH, END_IF, END_LOCK, ENDLOOP,
083                                        ENDMETHOD, END_PROC, END_PROTECT, ENDTRY);
084
085        /** Token types that start a sub parse. */
086        private static final Set<ETokenType> SUB_PARSE_TOKEN_TYPES = EnumSet.of(IF,
087                        FOR, OVER, LOOP, BLOCK, PROC, TRY, CATCH, PROTECT, LOCK);
088
089        /** {@inheritDoc} */
090        @Override
091        protected EMagikParserStates getSubParseState() {
092                return ANY;
093        }
094
095        /** {@inheritDoc} */
096        @Override
097        protected boolean tokenStartsSubParse(ETokenType tokenType,
098                        List<IToken> tokens, int offset) {
099                return SUB_PARSE_TOKEN_TYPES.contains(tokenType);
100        }
101
102        /** {@inheritDoc} */
103        @Override
104        protected boolean startsNewStatement(IToken token, IToken lastToken) {
105                if (lastToken == null) {
106                        return false;
107                }
108
109                // same line => usually no new statement
110                if (lastToken.getLineNumber() == token.getLineNumber()) {
111                        return SAME_LINE_STATEMENT_DELIMITER_TOKEN_TYPES.contains(token
112                                        .getType());
113                }
114
115                // if the previous line ended in an operator or a keyword, we can not
116                // end here
117                ETokenType lastTokenType = lastToken.getType();
118                ETokenClass lastTokenClass = lastTokenType.getTokenClass();
119                if (lastTokenClass == ETokenClass.OPERATOR
120                                || lastTokenClass == ETokenClass.KEYWORD
121                                || lastTokenType == ETokenType.DOT) {
122                        return VALID_STATEMENT_END_TOKEN_TYPES.contains(lastTokenType);
123                }
124
125                return true;
126        }
127}
