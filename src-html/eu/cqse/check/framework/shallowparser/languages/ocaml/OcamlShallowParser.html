093/** A shallow parser for OCaml. */
094public class OcamlShallowParser
095                extends ShallowParserBase<EGenericParserStates> {
096
097        /** Constructor. */
098        public OcamlShallowParser() {
099                super(EGenericParserStates.class, TOP_LEVEL);
100                createDefinitionRules();
101                createTypeRules();
102                createClassElementsRules();
103                createNestingExpressionRules();
104                createNonNestingExpressionRules();
105                createSimplifiedRecExpressionRules();
106                // Create an incomplete node if something is still unrecognized.
107                inAnyState().markStart().sequence(EnumSet.allOf(ETokenType.class))
108                                .createNode(EShallowEntityType.META, 0, null);
109        }
110
111        /** Definition rules. */
112        private void createDefinitionRules() {
113                inState(TOP_LEVEL).sequence(OPEN)
114                                .createNode(EShallowEntityType.META, "open", null)
115                                .sequence(IDENTIFIER).repeated(DOT, IDENTIFIER)
116                                .endNode();
117                inState(TOP_LEVEL).sequence(MODULE).markStart().sequence(IDENTIFIER)
118                                .skipTo(EQ).sequence(STRUCT)
119                                .createNode(EShallowEntityType.META, "module-def", 0)
120                                .skipTo(END).endNode();
121                inState(TOP_LEVEL).sequence(MODULE).markStart().sequence(IDENTIFIER)
122                                .skipTo(EQ).sequence(IDENTIFIER)
123                                .createNode(EShallowEntityType.META, "module-def", 0)
124                                .repeated(DOT, IDENTIFIER).endNode();
125                inState(TOP_LEVEL).sequence(EXCEPTION, IDENTIFIER);
126                inState(TOP_LEVEL).sequence(LET).optional(REC).markStart()
127                                .sequence(IDENTIFIER)
128                                .createNode(EShallowEntityType.METHOD, "let-decl", 0).skipTo(EQ)
129                                .parseOnce(IN_EXPRESSION).endNode();
130        }
131
132        /** Class definition rule */
133        private void createTypeRules() {
134                inState(TOP_LEVEL).sequence(CLASS).markStart()
135                                .createNode(EShallowEntityType.TYPE, SubTypeNames.CLASS, 0)
136                                .sequence(IDENTIFIER).skipTo(EQ)
137                                .sequence(OBJECT).optional(LPAREN, IDENTIFIER, RPAREN)
138                                .parseUntil(IN_TYPE).sequence(END)
139                                .endNode();
140        }
141
142        /** Create rules for class elements, i.e., methods. */
143        private void createClassElementsRules() {
144                inState(IN_TYPE).sequence(METHOD).optional(PRIVATE).markStart()
145                                .sequence(IDENTIFIER) // method name
146                                .createNode(EShallowEntityType.METHOD, SubTypeNames.METHOD, 0)
147                                .skipTo(EQ)
148                                .parseOnce(IN_EXPRESSION) // method body
149                                .endNode();
150                inState(IN_TYPE).sequence(INHERIT, IDENTIFIER).repeated(DOT,
151                                IDENTIFIER);
152        }
153
154        /** Rules to handle nesting expressions, i.e., if, while, for, match, .. */
155        private void createNestingExpressionRules() {
156
157                inState(IN_EXPRESSION).sequence(IF) // 4a: if with else branch
158                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.IF)
159                                .parseOnce(IN_EXPRESSION).sequence(THEN)
160                                .parseOnce(IN_EXPRESSION).sequence(ELSE)
161                                .parseOnce(IN_EXPRESSION).endNode();
162
163                inState(IN_EXPRESSION).sequence(IF) // 4b: if without else branch
164                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.IF)
165                                .parseOnce(IN_EXPRESSION).sequence(THEN)
166                                .parseOnce(IN_EXPRESSION).endNode();
167
168                inState(IN_EXPRESSION).sequence(WHILE) // 4c
169                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.WHILE)
170                                .parseOnce(IN_EXPRESSION).sequence(DO).parseOnce(IN_EXPRESSION)
171                                .sequence(DONE).endNode();
172
173                inState(IN_EXPRESSION).sequence(FOR, IDENTIFIER, EQ) // 4d
174                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.FOR)
175                                .parseOnce(IN_EXPRESSION).sequence(TO).parseOnce(IN_EXPRESSION)
176                                .sequence(DO).parseOnce(IN_EXPRESSION).sequence(DONE).endNode();
177
178                // rule 4e: match expression
179                RecognizerBase<EGenericParserStates> recognizedMatch = inState(IN_EXPRESSION)
180                                .sequence(MATCH)
181                                .createNode(EShallowEntityType.STATEMENT, "match-expr")
182                                .parseOnce(IN_EXPRESSION).sequence(WITH);
183                patternMatchingRecognizer(recognizedMatch).endNode();
184
185                RecognizerBase<EGenericParserStates> recognizedFunction = // 4f
186                inState(IN_EXPRESSION).sequence(FUNCTION)
187                                .createNode(EShallowEntityType.STATEMENT, "function-expr");
188                patternMatchingRecognizer(recognizedFunction).endNode();
189
190                inState(IN_EXPRESSION).sequence(FUN) // 4g
191                                .createNode(EShallowEntityType.STATEMENT, "fun-expr")
192                                .skipTo(MINUSGREATER).parseOnce(IN_EXPRESSION).endNode();
193
194                patternMatchingRecognizer(inState(IN_EXPRESSION).sequence(TRY) // 4h
195                                .parseOnce(IN_EXPRESSION).sequence(WITH)).endNode();
196
197                inState(IN_EXPRESSION).sequence(OBJECT) // rule 4i
198                                .createNode(EShallowEntityType.STATEMENT, "object-expr")
199                                .parseUntil(IN_TYPE).sequence(END).endNode();
200        }
201
202        /** Rules to handle non-nesting expressions */
203        private void createNonNestingExpressionRules() {
204
205                inState(IN_EXPRESSION).sequence(BEGIN).parseOnce(IN_EXPRESSION) // 4v
206                                .sequence(END);
207
208                inState(IN_EXPRESSION).sequence(LBRACK) // rule 4n: list expressions
209                                .createNode(EShallowEntityType.STATEMENT, "list-expr")
210                                .skipToWithNesting(RBRACK, LBRACK, RBRACK).endNode();
211                
212                inState(IN_EXPRESSION).sequence(LET).optional(REC).markStart() // 4q
213                                .sequence(IDENTIFIER)
214                                .createNode(EShallowEntityType.STATEMENT, "let-expr", 0)
215                                .skipTo(EQ).parseOnce(IN_EXPRESSION).sequence(IN)
216                                .parseOnce(IN_EXPRESSION).endNode();
217
218                inState(IN_EXPRESSION).sequence(NEW).markStart()
219                                .sequence(IDENTIFIER).repeated(DOT, IDENTIFIER)
220                                .createNode(EShallowEntityType.STATEMENT, "new-expr", 0)
221                                .endNode();
222
223        }
224
225        /**
226         * Simplified expression rules. (TODO: general rules would be
227         * left-recursive.)
228         */
229        private void createSimplifiedRecExpressionRules() {
230
231                inState(IN_EXPRESSION).sequence(IDENTIFIER) // 4s
232                                .sequence(SHARP, IDENTIFIER)
233                                .createNode(EShallowEntityType.STATEMENT, "method-call-expr", 0)
234                                .sequence(IDENTIFIER).endNode();
235
236                inState(IN_EXPRESSION).sequence(getUnaryConstructors()) // rule 4u
237                                .parseOnce(IN_EXPRESSION);
238
239                inState(IN_EXPRESSION).sequence(IDENTIFIER) // rule 4j
240                                .repeated(DOT, IDENTIFIER)
241                                .createNode(EShallowEntityType.STATEMENT, "id-expr",
242                                                new Region(0, -1))
243                                .endNode();
244
245                inState(IN_EXPRESSION).sequence(getConstantTokens()) // rule 4k
246                                .createNode(EShallowEntityType.STATEMENT, "const-expr", 0)
247                                .endNode();
248
249                inState(IN_EXPRESSION).markStart().sequence(LPAREN, RPAREN) // rule 4k
250                                .createNode(EShallowEntityType.STATEMENT, "const-expr",
251                                                new Region(0, -1))
252                                .endNode();
253
254                inState(IN_EXPRESSION).sequence(LPAREN).parseOnce(IN_EXPRESSION) // 4l
255                                .sequence(RPAREN);
256
257                inState(IN_EXPRESSION).sequence(LPAREN).parseOnce(IN_EXPRESSION) // 4t
258                                .sequence(COLONGREATER, IDENTIFIER).repeated(DOT, IDENTIFIER)
259                                .sequence(RPAREN);
260
261        }
262
263        /** Recognizes a pattern-matching construct */
264        private static RecognizerBase<EGenericParserStates> patternMatchingRecognizer(
265                        RecognizerBase<EGenericParserStates> currentState) {
266                return currentState.skipTo(MINUSGREATER).parseOnce(IN_EXPRESSION);
267        }
268
269        /**
270         * Returns a set of tokens representing constant values, except () and [].
271         */
272        private static EnumSet<ETokenType> getConstantTokens() {
273                return EnumSet.of(INTEGER_LITERAL, FLOATING_POINT_LITERAL, IDENTIFIER,
274                                FALSE, TRUE);
275        }
276
277        /** Returns a set of unary constructors */
278        private static EnumSet<ETokenType> getUnaryConstructors() {
279                return EnumSet.of(REF, EXCLAMATION, MINUS);
280        }
281
282        /** Redundant method, collecting all token names. Remove it. */
283        private void bull() {
284                inAnyState().skipBefore(AND, CLASS, COLONGREATER, DO, DONE, END,
285                                EXCEPTION, ELSE, EOF, EXCLAMATION, EXTERNAL, FOR, FUN, FUNCTION,
286                                IN, INCLUDE, INHERIT, LBRACK, LPAREN, LET, MATCH, MINUS,
287                                MINUSGREATER, MODULE, NEW, OBJECT, OPEN, OR, PRIVATE,
288                                RBRACK, REC, REF, RPAREN, STRUCT, TO, TYPE, UNDERSCORE,
289                                WITH, WHILE);
290        }
