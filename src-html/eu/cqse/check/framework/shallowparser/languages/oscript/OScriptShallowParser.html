115 * Shallow Parser for the OpenText OScript language
116 */
117public class OScriptShallowParser extends ShallowParserBase<EOScriptParserState> {

119        /** Available types for variables */
120        private static final EnumSet<ETokenType> VARIABLE_TYPES = EnumSet.of(BOOLEAN, INTEGER, REAL, STRING, OBJECT, ASSOC,
121                        LIST, DATE, DYNAMIC, IDENTIFIER);
122
123        /** Tokens indicating end of stataments */
124        private static final EnumSet<ETokenType> STATEMENT_END_TOKENS = EnumSet.of(EOL, SEMICOLON);
125
126        /** Set of all OScript built-in keyword identifiers */
127        private static final EnumSet<ETokenType> OSCRIPT_IDENTIFIERS = EnumSet.of(IDENTIFIER, ADDFEATURE, AND, ASSOC,
128                        BOOLEAN, BREAK, BREAKIF, BY, CASE, CONTINUE, CONTINUEIF, DATE, DEFAULT, DEFINE, DO, DOWNTO, DYNAMIC, ELSE,
129                        ELSEIF, END, ENDIF, ENDSCRIPT, FINAL, FOR, FUNCTION, GE, GOTO, GT, IF, IFDEF, IFNDEF, IN, INHERITS, INTEGER,
130                        INTERFACE, LE, LIST, LT, NAME, NE, NODEBUG, NONE, NOT, OBJECT, OR, OVERRIDE, PACKAGE, PARENT, PRIVATE,
131                        PUBLIC, REAL, REPEAT, RETURN, SCRIPT, SET, STRING, SUPER, SWITCH, THEN, THIS, TO, UNDEFINED, UNTIL, USING,
132                        WHILE);
133
134        /** Possible modifier for function or attributes */
135        private static final EnumSet<ETokenType> MODIFIER = EnumSet.of(OVERRIDE, PUBLIC, PRIVATE);
136
137        /** Constructor. */
138        public OScriptShallowParser() {
139                super(EOScriptParserState.class, EOScriptParserState.TOP_LEVEL);
140                createTopLevelRules();
141                createVariableRules();
142                createFunctionRules();
143                createScriptRules();
144                createRulesForStatements();
145        }
146
147        /**
148         * Rules for every variable declaration and usage.
149         */
150        private void createVariableRules() {
151                // Global variables
152                inState(TOP_LEVEL).subRecognizer(getTypeRecognizer()).subRecognizer(getVariableRecognizer())
153                                .createNode(ATTRIBUTE, GLOBAL_VARIABLE, 0).endNode();
154
155                // Attributes
156                inState(IN_OBJECT, IN_SCRIPT).optional(MODIFIER).optionalSubRecognizer(getTypeRecognizer())
157                                .subRecognizer(getVariableRecognizer()).createNode(ATTRIBUTE, SubTypeNames.ATTRIBUTE, 0).endNode();
158
159                // Local variables
160                inState(IN_FUNCTION).subRecognizer(getTypeRecognizer()).subRecognizer(getVariableRecognizer())
161                                .createNode(STATEMENT, LOCAL_VARIABLE, 0).endNode();
162                inState(IN_FUNCTION).subRecognizer(getVariableNameRecognizer()).subRecognizer(getDeclarationRecognizer())
163                                .createNode(STATEMENT, LOCAL_VARIABLE, 0).endNode();
164
165                // Assignments
166                inAnyState().subRecognizer(getVariableNameRecognizer()).subRecognizer(getAssignmentRecognizer())
167                                .createNode(STATEMENT, ASSIGNMENT, 0).endNode();
168        }
169
170        /**
171         * Recognizer for the usage of a variable. It checks if it is a valid variable
172         * name and end.
173         */
174        private RecognizerBase<EOScriptParserState> getVariableRecognizer() {
175                RecognizerBase<EOScriptParserState> recognizer = emptyRecognizer();
176                recognizer.subRecognizer(getVariableNameRecognizer()).subRecognizer(getStatementEndRecognizer());
177                return recognizer;
178        }
179
180        /**
181         * Recognizer for the name or way a variable is adressed. This also includes
182         * stuff like: test.test().test
183         */
184        private RecognizerBase<EOScriptParserState> getVariableNameRecognizer() {
185                RecognizerBase<EOScriptParserState> recognizer = emptyRecognizer();
186                RecognizerBase<EOScriptParserState> start = recognizer.optional(EnumSet.of(DOLLAR, DOT)).markStart()
187                                .optionalSubRecognizer(getMethodCallRecognizer(false)).optional(DOT).repeated(IDENTIFIER, DOT);
188
189                // Munich RE's scripts has some weird object property assignment
190                // <code>a.(b)=c<code>
191                start.repeated(IDENTIFIER, DOT).optional(IDENTIFIER, DOT).sequence(LPAREN).skipToWithNesting(RPAREN, LPAREN,
192                                RPAREN);
193                start.sequence(IDENTIFIER);
194                return recognizer;
195        }
196
197        /** Recognizer for a variable declaration */
198        private RecognizerBase<EOScriptParserState> getDeclarationRecognizer() {
199                RecognizerBase<EOScriptParserState> recognizer = emptyRecognizer();
200                recognizer.sequence(COMMA);
201                recognizer.sequence(STATEMENT_END_TOKENS);
202                return recognizer;
203        }
204
205        /** Recognizer for an assignment starting with an '='. */
206        private RecognizerBase<EOScriptParserState> getAssignmentRecognizer() {
207                RecognizerBase<EOScriptParserState> recognizer = emptyRecognizer();
208                RecognizerBase<EOScriptParserState> assignmentRecognizer = recognizer.sequence(EQUAL);
209
210                assignmentRecognizer.subRecognizer(getMethodCallRecognizer(false)).repeated(DOT, IDENTIFIER);
211                assignmentRecognizer.sequence(LBRACE).skipToWithNesting(RBRACE, LBRACE, RBRACE);
212                assignmentRecognizer.sequence(LBRACK).skipToWithNesting(RBRACK, LBRACK, RBRACK);
213                assignmentRecognizer.skipTo(STATEMENT_END_TOKENS);
214
215                return recognizer;
216        }
217
218        /**
219         * Returns a recognizer which identifies if it is a valid statement end. This
220         * recognizer starts after the variable identifier.
221         * 
222         * String a -->| = 3 (so here it would recognize every valid part which can come
223         * after "String a"
224         */
225        private RecognizerBase<EOScriptParserState> getStatementEndRecognizer() {
226                RecognizerBase<EOScriptParserState> recognizer = emptyRecognizer();
227                recognizer.subRecognizer(getDeclarationRecognizer());
228                recognizer.subRecognizer(getAssignmentRecognizer());
229                return recognizer;
230        }
231
232        /**
233         * A recognizer for the declaration of a variable with possible modifier and
234         * type. Stops before the identifier.
235         */
236        private RecognizerBase<EOScriptParserState> getTypeRecognizer() {
237                RecognizerBase<EOScriptParserState> recognizer = emptyRecognizer();
238                recognizer.optional(MODIFIER).sequence(VARIABLE_TYPES).sequenceBefore(IDENTIFIER);
239                return recognizer;
240        }
241
242        /**
243         * Recognizer for a method call. This covers also multiple chained method calls
244         * and calls over multiple lines.
245         *
246         * For example: test.test().test(\n, a,\n b)
247         */
248        private RecognizerBase<EOScriptParserState> getMethodCallRecognizer(boolean markStart) {
249                RecognizerBase<EOScriptParserState> recognizer = emptyRecognizer();
250                if (markStart) {
251                        recognizer.optional(EnumSet.of(DOLLAR, DOT)).markStart().subRecognizer(getSimpleMethodCallRecognizer(), 1,
252                                        Integer.MAX_VALUE);
253                } else {
254                        recognizer.optional(EnumSet.of(DOLLAR, DOT)).subRecognizer(getSimpleMethodCallRecognizer(), 1,
255                                        Integer.MAX_VALUE);
256                }
257                return recognizer;
258        }
259
260        /**
261         * Recognizer for a simple method call like .test(), $test() or test()
262         */
263        private RecognizerBase<EOScriptParserState> getSimpleMethodCallRecognizer() {
264                RecognizerBase<EOScriptParserState> recognizer = emptyRecognizer();
265                recognizer.optional(DOT).repeated(IDENTIFIER, DOT).sequence(IDENTIFIER, LPAREN).skipToWithNesting(RPAREN,
266                                LPAREN, RPAREN);
267                return recognizer;
268        }

270        /**
271         * Rules for top-level declarations: package and object statements. These are
272         * according to the new source format.
273         */
274        private void createTopLevelRules() {
275                // Package
276                inState(TOP_LEVEL).sequence(PACKAGE).skipTo(STATEMENT_END_TOKENS)
277                                .createNode(MODULE, SubTypeNames.PACKAGE, new Region(1, -2)).endNode();
278
279                // Object
280                inState(TOP_LEVEL).sequence(PUBLIC, OBJECT, IDENTIFIER).markStart().createNode(TYPE, SubTypeNames.OBJECT, -1)
281                                .optional(INHERITS, IDENTIFIER).repeated(DOUBLE_COLON, IDENTIFIER).parseUntil(IN_OBJECT).sequence(END)
282                                .endNode();
283
284                inState(TOP_LEVEL).sequence(EnumSet.of(NAME, PARENT, ADDFEATURE)).skipTo(STATEMENT_END_TOKENS)
285                                .createNode(META, 0).endNode();
286
287                // Preprocessor directives
288                inAnyState().sequence(PREPROCESSOR_DIRECTIVE).createNode(EShallowEntityType.META, 0).endNode();
289        }
290
291        /** Rule for a script */
292        private void createScriptRules() {
293                inState(TOP_LEVEL, IN_OBJECT).optional(EnumSet.of(OVERRIDE, PUBLIC, PRIVATE))
294                                .sequence(SCRIPT, OSCRIPT_IDENTIFIERS).createNode(TYPE, SubTypeNames.SCRIPT, -1).parseUntil(IN_SCRIPT)
295                                .sequence(ENDSCRIPT).endNode();
296        }
297
298        /** Rule for a function */
299        private void createFunctionRules() {
300                inState(TOP_LEVEL, IN_OBJECT, IN_SCRIPT).optional(MODIFIER)
301                                .sequence(FUNCTION,
302                                                EnumSet.of(BOOLEAN, INTEGER, REAL, STRING, OBJECT, ASSOC, LIST, DATE, DYNAMIC, VOID),
303                                                OSCRIPT_IDENTIFIERS, LPAREN)
304                                .createNode(METHOD, SubTypeNames.FUNCTION, 2).skipToWithNesting(RPAREN, LPAREN, RPAREN)
305                                .parseUntil(IN_FUNCTION).sequence(END).endNode();
306        }
307
308        /** Rules for function and script statements */
309        private void createRulesForStatements() {
310                createRulesForConditionStatements();
311                createRulesForLoopStatements();
312
313                // Labels and goto jumps
314                inState(TOP_LEVEL, IN_FUNCTION, IN_SCRIPT).sequence(IDENTIFIER, COLON).createNode(STATEMENT, SubTypeNames.LABEL)
315                                .optional(STATEMENT_END_TOKENS).endNode();
316                inState(TOP_LEVEL, IN_FUNCTION, IN_SCRIPT).sequence(GOTO, IDENTIFIER).createNode(STATEMENT, -1)
317                                .optional(STATEMENT_END_TOKENS).endNode();
318
319                // Simple keywords
320                inState(TOP_LEVEL, IN_FUNCTION, IN_SCRIPT).sequence(EnumSet.of(RETURN, BREAK, CONTINUE, BREAKIF, CONTINUEIF))
321                                .createNode(STATEMENT, 0).skipNested(LPAREN, RPAREN).skipTo(STATEMENT_END_TOKENS).endNode();
322
323                // Label
324                inAnyState().sequence(IDENTIFIER, COLON).createNode(STATEMENT, SubTypeNames.LABEL, 0)
325                                .skipTo(STATEMENT_END_TOKENS).endNode();
326
327                // Method calls
328                inAnyState().subRecognizer(getMethodCallRecognizer(true)).createNode(STATEMENT, SIMPLE_STATEMENT, 0).endNode();
329        }
330
331        /** Create rules for condition statements */
332        private void createRulesForConditionStatements() {
333                // If Statement
334                createIfRules(IF);
335                createIfRules(ELSE);
336                createIfRules(ELSEIF);
337
338                // Switch
339                inState(TOP_LEVEL, IN_FUNCTION, IN_SCRIPT).sequence(SWITCH).createNode(STATEMENT, 0)
340                                .skipToWithNesting(STATEMENT_END_TOKENS, LPAREN, RPAREN).parseUntil(IN_FUNCTION).sequence(END)
341                                .endNode();
342                inState(TOP_LEVEL, IN_FUNCTION, IN_SCRIPT).sequence(CASE)
343                                .skipToWithNesting(STATEMENT_END_TOKENS, LPAREN, RPAREN)
344                                .createNode(EShallowEntityType.META, new int[] { 0, 1 }).parseUntil(IN_FUNCTION).sequence(END)
345                                .endNode();
346                inState(TOP_LEVEL, IN_FUNCTION, IN_SCRIPT).sequence(DEFAULT, STATEMENT_END_TOKENS)
347                                .createNode(EShallowEntityType.META, 0).parseUntil(IN_FUNCTION).sequence(END).endNode();
348        }
349
350        /** Create rules for If/else-if conditions */
351        private void createIfRules(ETokenType conditionToken) {
352                RecognizerBase<EOScriptParserState> ifRecognizer = inState(TOP_LEVEL, IN_FUNCTION, IN_SCRIPT)
353                                .sequence(conditionToken).skipToWithNesting(STATEMENT_END_TOKENS, LPAREN, RPAREN)
354                                .createNode(STATEMENT, 0).parseUntil(IN_FUNCTION);
355                if (conditionToken == ELSE) {
356                        ifRecognizer.sequence(END).endNode();
357                        return;
358                }
359
360                ifRecognizer.sequenceBefore(EnumSet.of(ELSE, ELSEIF)).endNodeWithContinuation();
361                ifRecognizer.sequence(END).endNode();
362        }
363
364        /** Create rules for loop statements */
365        private void createRulesForLoopStatements() {
366                // While
367                inState(TOP_LEVEL, IN_FUNCTION, IN_SCRIPT).sequence(WHILE).createNode(STATEMENT, 0).skipNested(LPAREN, RPAREN)
368                                .parseUntil(IN_FUNCTION).sequence(END).endNode();
369
370                // Repeat
371                inState(TOP_LEVEL, IN_FUNCTION, IN_SCRIPT).sequence(REPEAT).createNode(STATEMENT, 0).parseUntil(IN_FUNCTION)
372                                .sequence(WHILE).skipNested(LPAREN, RPAREN).endNode();
373
374                // For...each, C-style and structured for loops
375                RecognizerBase<EOScriptParserState> forRecognizer = inState(TOP_LEVEL, IN_FUNCTION, IN_SCRIPT).sequence(FOR)
376                                .markStart();
377                forRecognizer.sequence(IDENTIFIER).createNode(STATEMENT, -2).skipNested(LPAREN, RPAREN)
378                                .skipTo(STATEMENT_END_TOKENS).parseUntil(IN_FUNCTION).sequence(END).endNode();
379                forRecognizer.skipNested(LPAREN, RPAREN).createNode(STATEMENT, -3).parseUntil(IN_FUNCTION).sequence(END)
380                                .endNode();
381        }
