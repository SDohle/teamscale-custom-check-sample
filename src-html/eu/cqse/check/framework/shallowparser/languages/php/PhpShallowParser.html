103/**
104 * PHP shallow parser.
105 *
106 * Some special cases for this parser:
107 * <ul>
108 * <li>Beware, that in most cases field definitions / variable declarations are
109 * recognized as simple statements.</li>
110 * <li>As of PHP 5.4.0, writing CONTINUE without a semicolon raises a compile
111 * error. However this parser accepts CONTINUE from the prior versions as well.
112 * </li>
113 * </ul>
114 */
115public class PhpShallowParser extends ShallowParserBase<EGenericParserStates> {
116
117        /** All token types that may stand at the beginning of new statement. */
118        private static final EnumSet<ETokenType> STATEMENT_START_TOKENS = EnumSet.of(BACKSLASH, BREAK, CLONE, ECHO, GOTO,
119                        IDENTIFIER, MINUSMINUS, NAMESPACE, NEW, PLUSPLUS, PRINT, RETURN, STATIC, THIS, THROW, YIELD, AT);
120        /** A set of access modifiers. */
121        private static final EnumSet<ETokenType> ACCESS_MODIFIERS = EnumSet.of(PRIVATE, PUBLIC, PROTECTED, STATIC, GLOBAL,
122                        CONST, ABSTRACT, FINAL, VAR);
123
124        /** Valid identifier token types. */
125        private static final EnumSet<ETokenType> VALID_IDENTIFIERS = EnumSet.of(IDENTIFIER, CLONE);

143                createEmptyStatementRule();
144        }

146        /**
147         * Creates the rule for anonymous inner classes and anonymous functions.
148         * They are typically found within other statements.
149         */
150        private void createInExpressionRules() {
151
152                // Anonymous inner classes
153                RecognizerBase<EGenericParserStates> innerClassBase = inState(IN_EXPRESSION).sequence(NEW, CLASS)
154                                .createNode(EShallowEntityType.TYPE, SubTypeNames.ANONYMOUS_CLASS, 1).skipBefore(LBRACE);
155                endTypeRule(innerClassBase);
156
157                // Anonymous functions
158                RecognizerBase<EGenericParserStates> closureBase = inState(IN_EXPRESSION).sequence(FUNCTION)
159                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.ANONYMOUS_FUNCTION)
160                                .skipNested(LPAREN, RPAREN, createSubexpressionRecognizer());
161
162                RecognizerBase<EGenericParserStates> closureBaseWithUse = closureBase.sequence(USE).skipNested(LPAREN, RPAREN,
163                                createSubexpressionRecognizer());
164
165                endMethodRule(closureBaseWithUse);
166                endMethodRule(closureBase);
167        }

169        /**
170         * Create a rule that matches try catch finally statements.
171         */
172        private void createTryCatchRule() {
173                RecognizerBase<EGenericParserStates> tryBase = inState(TOP_LEVEL, IN_METHOD).sequence(TRY)
174                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.TRY);
175
176                RecognizerBase<EGenericParserStates> catchBase = inState(TOP_LEVEL, IN_METHOD).sequence(CATCH)
177                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.CATCH).skipNested(LPAREN, RPAREN);
178
179                RecognizerBase<EGenericParserStates> finallyBase = inState(TOP_LEVEL, IN_METHOD).sequence(FINALLY)
180                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.FINALLY);
181
182                continueTryCatchRule(tryBase);
183                continueTryCatchRule(catchBase);
184                continueTryCatchRule(finallyBase);
185
186        }
187
188        /**
189         * Continue the try catch finally rule.
190         */
191        private static void continueTryCatchRule(RecognizerBase<EGenericParserStates> base) {
192
193                RecognizerBase<EGenericParserStates> multilineBase = base.sequence(LBRACE).parseUntil(IN_METHOD)
194                                .sequence(RBRACE);
195                RecognizerBase<EGenericParserStates> singlelineBase = base.parseOnce(IN_METHOD);
196
197                endTryCatchRule(multilineBase);
198                endTryCatchRule(singlelineBase);
199        }
200
201        /**
202         * End the try catch finally rule.
203         */
204        private static void endTryCatchRule(RecognizerBase<EGenericParserStates> base) {
205                base.sequenceBefore(EnumSet.of(CATCH, FINALLY)).endNodeWithContinuation();
206                base.endNode();
207        }

213                createConstructorRule(CONSTRUCTOR, SubTypeNames.CONSTRUCTOR);
214                createConstructorRule(DESTRUCTOR, SubTypeNames.DESTRUCTOR);

217        /**
218         * Create a rule that matches constructors or destructors.
219         */
220        private void createConstructorRule(ETokenType tokenType, String subtypeName) {
221                RecognizerBase<EGenericParserStates> base = inState(IN_TYPE).repeated(ACCESS_MODIFIERS).sequence(FUNCTION)
222                                .sequence(tokenType).createNode(EShallowEntityType.METHOD, subtypeName).skipNested(LPAREN, RPAREN);
223                base.sequence(LBRACE).parseUntil(IN_METHOD).sequence(RBRACE).endNode();
224                base.sequence(EnumSet.of(SEMICOLON, TEMPLATE_CODE_END)).endNode();
225        }

227        /**
228         * Create a rule that matches modules (e.g. namespaces).
229         */
230        private void createNamespaceRule() {
231
232                RecognizerBase<EGenericParserStates> base = inState(TOP_LEVEL).sequence(NAMESPACE)
233                                // We have to make sure to not match the sequence NAMESPACE -
234                                // BACKSLASH here. These can be used to call functions etc.
235                                .sequenceBefore(EnumSet.complementOf(EnumSet.of(BACKSLASH)))
236                                .skipBefore(EnumSet.of(SEMICOLON, TEMPLATE_CODE_END, LBRACE))
237                                .createNode(EShallowEntityType.MODULE, SubTypeNames.NAMESPACE, new Region(1, -1));
238
239                base.sequence(EnumSet.of(SEMICOLON, TEMPLATE_CODE_END)).endNode();
240                base.sequence(LBRACE).parseUntil(TOP_LEVEL).sequence(RBRACE).endNode();
241        }

247                inAnyState().sequence(SEMICOLON).createNode(EShallowEntityType.STATEMENT, SubTypeNames.EMPTY_STATEMENT)
248                                .endNode();

261        /**
262         * Create rules that match labels, use and declare statements.
263         */
264        private void createMetaRules() {
265                // Declare
266                RecognizerBase<EGenericParserStates> baseDeclare = inState(TOP_LEVEL, IN_METHOD).sequence(DECLARE)
267                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.DECLARATION).skipNested(LPAREN, RPAREN);
268                baseDeclare.sequence(EnumSet.of(SEMICOLON, TEMPLATE_CODE_END)).endNode();
269                endControlFlowStatement(baseDeclare, ENDDECLARE);
270
271                // Use
272                RecognizerBase<EGenericParserStates> baseUse = inAnyState().sequence(USE)
273                                .createNode(EShallowEntityType.META, SubTypeNames.USE, new Region(1, -1))
274                                .skipBefore(EnumSet.of(TEMPLATE_CODE_END, SEMICOLON, LBRACE));
275
276                baseUse.sequence(EnumSet.of(SEMICOLON, TEMPLATE_CODE_END)).endNode();
277                baseUse.sequence(LBRACE).parseUntil(IN_METHOD).sequence(RBRACE).endNode();
278
279                createRequireRule(REQUIRE, SubTypeNames.REQUIRE);
280                createRequireRule(REQUIRE_ONCE, SubTypeNames.REQUIRE_ONCE);
281                createRequireRule(INCLUDE, SubTypeNames.INCLUDE);
282                createRequireRule(INCLUDE_ONCE, SubTypeNames.INCLUDE_ONCE);
283
284                // Label
285                inAnyState().sequence(VALID_IDENTIFIERS, COLON).createNode(EShallowEntityType.META, SubTypeNames.LABEL, 0)
286                                .endNode();
287        }
288
289        /** Creates a rule for include/require */
290        private void createRequireRule(ETokenType requireToken, String subtype) {
291                inAnyState().sequence(requireToken).createNode(EShallowEntityType.META, subtype, new Region(1, -1))
292                                .skipTo(EnumSet.of(TEMPLATE_CODE_END, SEMICOLON)).endNode();
293        }

295        /**
296         * Create a rule that matches switch case.
297         */
298        private void createSwitchCaseRule() {
299                inState(IN_METHOD).sequence(EnumSet.of(CASE, DEFAULT)).createNode(EShallowEntityType.META, 0)
300                                .optional(VALID_IDENTIFIERS).repeated(COLON, COLON, VALID_IDENTIFIERS)
301                                // "case 1;" is allowed.
302                                .skipTo(EnumSet.of(COLON, SEMICOLON, TEMPLATE_CODE_END)).endNode();
303
304                RecognizerBase<EGenericParserStates> base = inState(TOP_LEVEL, IN_METHOD).sequence(SWITCH)
305                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.SWITCH)
306                                .skipNested(LPAREN, RPAREN, createSubexpressionRecognizer());
307                endControlFlowStatement(base, ENDSWITCH);
308        }

310        /**
311         * Create a rule that matches type declarations.
312         */
313        private void createTypeRule() {
314                createTypeRule(CLASS, SubTypeNames.CLASS);
315                createTypeRule(TRAIT, SubTypeNames.TRAIT);
316                createInterfaceRule();
317        }

319        /**
320         * Create a rule that matches interface declarations.
321         */
322        private void createInterfaceRule() {
323                RecognizerBase<EGenericParserStates> interfaceRule = beginTypeRule(INTERFACE, SubTypeNames.INTERFACE);
324
325                // This rule is necessary for types extending qualified identifiers.
326                // E.g., "interface Foo extends Namespace1\Namespace2\A"
327                endInterfaceRule(interfaceRule.sequence(EXTENDS).skipBefore(LBRACE));
328                // This rule captures interfaces without body (even without open/closing
329                // braces)
330                endInterfaceRule(interfaceRule.sequence(EXTENDS, VALID_IDENTIFIERS).repeated(COMMA, VALID_IDENTIFIERS));
331                endInterfaceRule(interfaceRule);
332        }
333
334        /** End a interface declaration rule. */
335        private static void endInterfaceRule(RecognizerBase<EGenericParserStates> interfaceRule) {
336                endTypeRule(interfaceRule.sequenceBefore(LBRACE));
337                interfaceRule.endNode();
338        }

341         * Create a rule that matches a specific type declaration.

343        private void createTypeRule(ETokenType type, String subtypeName) {
344                endTypeRule(beginTypeRule(type, subtypeName).skipBefore(LBRACE));

348        private RecognizerBase<EGenericParserStates> beginTypeRule(ETokenType type, String subtypeName) {
349                return inState(TOP_LEVEL, IN_METHOD).repeated(ACCESS_MODIFIERS).sequence(type).markStart()
350                                .sequence(VALID_IDENTIFIERS).createNode(EShallowEntityType.TYPE, subtypeName, 0);
351        }

354         * End type rule creation with a parse within IN_TYPE surrounded by braces.

356        private static void endTypeRule(RecognizerBase<EGenericParserStates> base) {
357                base.sequence(LBRACE).parseUntil(IN_TYPE).sequence(RBRACE).endNode();
358        }

360        /**
361         * Create a rule that matches variables and attributes.
362         */
363        private void createVariableRule() {
364                inState(TOP_LEVEL, IN_METHOD).repeated(ACCESS_MODIFIERS).markStart().sequence(VALID_IDENTIFIERS)
365                                .sequence(EnumSet.of(SEMICOLON, TEMPLATE_CODE_END))
366                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.LOCAL_VARIABLE, 0).endNode();
367
368                inState(TOP_LEVEL, IN_METHOD).sequence(ACCESS_MODIFIERS).repeated(ACCESS_MODIFIERS).markStart()
369                                .sequence(VALID_IDENTIFIERS).createNode(EShallowEntityType.STATEMENT, SubTypeNames.LOCAL_VARIABLE, 0)
370                                .skipToWithNesting(EnumSet.of(SEMICOLON, TEMPLATE_CODE_END), LPAREN, RPAREN,
371                                                createSubexpressionRecognizer())
372                                .endNode();
373
374                inState(IN_TYPE).repeated(ACCESS_MODIFIERS).markStart().sequence(VALID_IDENTIFIERS).sequenceBefore(EQ)
375                                .createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.ATTRIBUTE, 0)
376                                .skipToWithNesting(EnumSet.of(SEMICOLON, TEMPLATE_CODE_END), LPAREN, RPAREN,
377                                                createSubexpressionRecognizer())
378                                .endNode();
379
380                inState(IN_TYPE).repeated(ACCESS_MODIFIERS).markStart()
381                                .sequence(VALID_IDENTIFIERS, EnumSet.of(SEMICOLON, TEMPLATE_CODE_END))
382                                .createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.ATTRIBUTE, 0).endNode();
383        }

385        /**
386         * Create rules that match all kinds of loop statements.
387         */
388        private void createLoopRules() {
389                createLoopRule(FOREACH, ENDFOREACH);
390                createLoopRule(FOR, ENDFOR);
391                createLoopRule(WHILE, ENDWHILE);
392
393                // Do-While
394                RecognizerBase<EGenericParserStates> base = inState(IN_METHOD, TOP_LEVEL).sequence(ETokenType.DO)
395                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.DO);
396                base.sequence(LBRACE).parseUntil(IN_METHOD).sequence(RBRACE, WHILE)
397                                .skipNested(LPAREN, RPAREN, createSubexpressionRecognizer()).optional(SEMICOLON).endNode();
398                base.parseOnce(IN_METHOD).sequence(WHILE).skipNested(LPAREN, RPAREN, createSubexpressionRecognizer())
399                                .optional(SEMICOLON).endNode();
400        }

402        /**
403         * Create a rule that matches the normal version as well as the alternative
404         * versions of loop statements.
405         */
406        private void createLoopRule(ETokenType startToken, ETokenType endToken) {
407                RecognizerBase<EGenericParserStates> base = inState(IN_METHOD, TOP_LEVEL).sequence(startToken)
408                                .skipNested(LPAREN, RPAREN, createSubexpressionRecognizer())
409                                .createNode(EShallowEntityType.STATEMENT, 0);
410                endControlFlowStatement(base, endToken);
411        }

413        /**
414         * Create a rule that matches if statements.
415         */
416        private void createIfElseRule() {
417                RecognizerBase<EGenericParserStates> base = inState(IN_METHOD, TOP_LEVEL).sequence(EnumSet.of(IF, ELSEIF))
418                                .createNode(EShallowEntityType.STATEMENT, 0)
419                                .skipNested(LPAREN, RPAREN, createSubexpressionRecognizer());
420                continueIfElseRule(base, true);
421
422                RecognizerBase<EGenericParserStates> elseBase = inState(IN_METHOD, TOP_LEVEL).sequence(ELSE)
423                                .createNode(EShallowEntityType.STATEMENT, 0);
424                continueIfElseRule(elseBase, false);
425        }
426
427        /**
428         * Continues an if else rule possibly with continuation.
429         */
430        private static void continueIfElseRule(RecognizerBase<EGenericParserStates> base, boolean allowContinuation) {
431                RecognizerBase<EGenericParserStates> multiLineBase = base.sequence(LBRACE).parseUntil(IN_METHOD)
432                                .sequence(RBRACE);
433
434                RecognizerBase<EGenericParserStates> multiLineBaseAlternative = base.sequence(COLON).parseUntil(IN_METHOD)
435                                .sequenceBefore(EnumSet.of(ELSE, ELSEIF, ENDIF));
436
437                RecognizerBase<EGenericParserStates> singleLineBase = base.parseOnce(IN_METHOD);
438
439                endIfElseRule(singleLineBase, false, allowContinuation);
440                endIfElseRule(multiLineBase, false, allowContinuation);
441                endIfElseRule(multiLineBaseAlternative, true, allowContinuation);
442        }
443
444        /**
445         * Ends an if else rule possibly with continuation.
446         */
447        private static void endIfElseRule(RecognizerBase<EGenericParserStates> base, boolean expectsEndif,
448                        boolean allowContinuation) {
449                if (allowContinuation) {
450                        base.sequenceBefore(EnumSet.of(ELSE, ELSEIF)).endNodeWithContinuation();
451                }
452
453                if (expectsEndif) {
454                        base.optional(ENDIF, SEMICOLON).endNode();
455                        base.optional(ENDIF).endNode();
456                } else {
457                        base.endNode();
458                }
459        }

461        /**
462         * Create a rule that matches method declarations. In PHP there is the
463         * concept of conditional functions, therefore we recognize functions within
464         * IN_METHOD, too.
465         */
466        private void createMethodRule() {
467                RecognizerBase<EGenericParserStates> base = inState(TOP_LEVEL, IN_TYPE, IN_METHOD).repeated(ACCESS_MODIFIERS)
468                                .sequence(FUNCTION).optional(AND).sequence(VALID_IDENTIFIERS)
469                                .createNode(EShallowEntityType.METHOD, SubTypeNames.METHOD, -1)
470                                // we need this skip because of strict typing
471                                .skipBeforeWithNesting(EnumSet.of(LBRACE, SEMICOLON, TEMPLATE_CODE_END), LPAREN, RPAREN);
472
473                endMethodRule(base);
474                // abstract method in e.g. interfaces.
475                base.sequence(EnumSet.of(SEMICOLON, TEMPLATE_CODE_END)).endNode();
476        }

479         * Ends a standard method rule with a parse within IN_METHOD surrounded by
480         * braces.
481         */
482        private static void endMethodRule(RecognizerBase<EGenericParserStates> base) {
483                base.sequence(LBRACE).parseUntil(IN_METHOD).sequence(RBRACE).endNode();
484        }

486        /**
487         * Create a rule that matches single statements.
488         */
489        private void createSingleStatementRule() {
490                inState(TOP_LEVEL, IN_METHOD).sequenceBefore(STATEMENT_START_TOKENS)
491                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.SIMPLE_STATEMENT, 0)
492                                .skipToWithNesting(EnumSet.of(SEMICOLON, TEMPLATE_CODE_END), LPAREN, RPAREN,
493                                                createSubexpressionRecognizer())
494                                .endNode();
495
496                // Statements that are enclosed in parenthesis.
497                inState(TOP_LEVEL, IN_METHOD).sequenceBefore(LPAREN)
498                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.SIMPLE_STATEMENT, 0)
499                                .skipToWithNesting(EnumSet.of(SEMICOLON, TEMPLATE_CODE_END), LPAREN, RPAREN,
500                                                createSubexpressionRecognizer())
501                                .endNode();
502
503                // Continue is a special case, as one could omit the
504                // semicolon prior to PHP 5.4.0.
505                RecognizerBase<EGenericParserStates> continueBase = inState(TOP_LEVEL, IN_METHOD).sequence(CONTINUE)
506                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.SIMPLE_STATEMENT, 0);
507
508                EnumSet<ETokenType> continueTypes = EnumSet.copyOf(VALID_IDENTIFIERS);
509                continueTypes.add(INTEGER_LITERAL);
510                continueBase.sequence(continueTypes, SEMICOLON).endNode();
511                continueBase.optional(SEMICOLON).endNode();
512        }
513
514        /**
515         * Ends a typical control flow statement with.
516         */
517        private static void endControlFlowStatement(RecognizerBase<EGenericParserStates> base,
518                        ETokenType alternativeEndToken) {
519                base.sequence(COLON).parseUntil(IN_METHOD).sequence(alternativeEndToken).optional(SEMICOLON).endNode();
520                base.sequence(LBRACE).parseUntil(IN_METHOD).sequence(RBRACE).optional(SEMICOLON).endNode();
521                base.sequence(SEMICOLON).endNode();
522                base.parseOnce(IN_METHOD).optional(SEMICOLON).endNode();
523        }

529                return new PhpSubExpressionRecognizer();
