045        /** The states used in this parser. */
046        public static enum EPlsqlParserStates {
047
048                /** A state to recognize declarations. */
049                DECLARATIONS,
050
051                /** A state to recognize statements. */
052                STATEMENTS
053        }

055        /**
056         * In PL/SQL all keywords may also be used as identifiers. There are also
057         * reserved words, which are not allowed as identifiers.
058         */

097        /** Constructor. */
098        public PlsqlShallowParser() {
099                super(EPlsqlParserStates.class, DECLARATIONS);
100
101                createMetaRules();
102                createPackageAndTypeRules();
103                createMethodAndAttributeRules();
104                createStatementRules();
105        }

107        /** Create rules for parsing meta elements. */
108        private void createMetaRules() {
109
110                // SHOW ERROR is tricky, as it allows for multiple abbreviations
111                inState(DECLARATIONS)
112                                .sequence(new IdentifierPrefixMatcher("sho"),
113                                                new IdentifierPrefixMatcher("err")).optional(SEMICOLON)
114                                .createNode(EShallowEntityType.META, "show errors").endNode();
115
116                // exit
117                inState(DECLARATIONS).sequence(EXIT).optional(INTEGER_LITERAL)
118                                .optional(SEMICOLON)
119                                .createNode(EShallowEntityType.META, "exit").endNode();
120
121                // pragma
122                inAnyState().sequence(PRAGMA)
123                                .createNode(EShallowEntityType.META, "pragma")
124                                .skipTo(SEMICOLON).endNode();
125
126                // exception section
127                inState(STATEMENTS).sequence(EXCEPTION)
128                                .createNode(EShallowEntityType.META, "exception section")
129                                .endNode();
130
131                // single and double 'at' sign execution; see
132                // http://docs.oracle.com/cd/B19306_01/server.102/b14357/ch12003.htm#BACIEHDJ
133                inState(DECLARATIONS).sequence(AT).optional(AT)
134                                .createNode(EShallowEntityType.META, "run script")
135                                .repeated(EnumSet.of(IDENTIFIER, DOT, MINUS, MOD)).endNode();
136
137                // SQL statements
138                inAnyState()
139                                .sequence(
140                                                EnumSet.of(AGGREGATE, ALTER, COMMIT, DELETE, GRANT,
141                                                                INSERT, LOCK, ROLLBACK, SAVEPOINT, SELECT,
142                                                                DROP, MERGE, UPDATE))
143                                .createNode(EShallowEntityType.STATEMENT, "SQL", 0)
144                                .skipTo(SEMICOLON).endNode();
145                inAnyState().sequence(SET, TRANSACTION)
146                                .createNode(EShallowEntityType.STATEMENT, "SQL", 0)
147                                .skipTo(SEMICOLON).endNode();
148                inAnyState().sequence(CREATE).optional(OR, REPLACE).optional(PUBLIC)
149                                .sequence(SYNONYM)
150                                .createNode(EShallowEntityType.STATEMENT, "SQL", 0)
151                                .skipTo(SEMICOLON).endNode();
152
153                // set
154                EnumSet<ETokenType> setIdentifiers = EnumSet.of(IDENTIFIER, DEFINE);
155                inState(DECLARATIONS).sequence(SET, setIdentifiers, PLSQL_IDENTIFIERS)
156                                .createNode(EShallowEntityType.META, "set")
157                                .repeated(setIdentifiers, PLSQL_IDENTIFIERS)
158                                .optional(SEMICOLON).endNode();
159
160                // deal with dangling end by inserting broken node
161                inAnyState().sequence(END)
162                                .createNode(EShallowEntityType.META, "dangling end")
163                                .skipTo(SEMICOLON); // endNode() omitted!
164        }

166        /** Creates parsing rules for packages and types. */
167        private void createPackageAndTypeRules() {
168
169                // packages
170                RecognizerBase<EPlsqlParserStates> optionalBeginAlternative = createOrReplace()
171                                .sequence(PACKAGE)
172                                .optional(BODY)
173                                .markStart()
174                                .repeated(PLSQL_IDENTIFIERS, DOT)
175                                .sequence(PLSQL_IDENTIFIERS)
176                                .createNode(EShallowEntityType.MODULE, "package",
177                                                new Region(0, -1)).skipTo(EnumSet.of(IS, AS))
178                                .parseUntil(DECLARATIONS);
179                optionalBeginAlternative.sequence(BEGIN).parseUntil(STATEMENTS)
180                                .sequence(END).skipTo(SEMICOLON).endNode();
181                optionalBeginAlternative.sequence(END).skipTo(SEMICOLON).endNode();
182
183                // type body
184                createOrReplace()
185                                .sequence(TYPE, BODY)
186                                .markStart()
187                                .repeated(PLSQL_IDENTIFIERS, DOT)
188                                .sequence(PLSQL_IDENTIFIERS)
189                                .createNode(EShallowEntityType.MODULE, "type body",
190                                                new Region(0, -1)).skipTo(EnumSet.of(IS, AS))
191                                .parseUntil(DECLARATIONS).sequence(END).skipTo(SEMICOLON)
192                                .endNode();
193
194                // type
195                RecognizerBase<EPlsqlParserStates> typeMatcher = createOrReplace()
196                                .markStart().sequence(TYPE).repeated(PLSQL_IDENTIFIERS, DOT)
197                                .sequence(PLSQL_IDENTIFIERS)
198                                .createNode(EShallowEntityType.TYPE, 0, new Region(1, -1))
199                                .skipBefore(EnumSet.of(IS, AS, UNDER, SEMICOLON, DIV));
200                typeMatcher.sequence(EnumSet.of(SEMICOLON, DIV)).endNode();
201                RecognizerBase<EPlsqlParserStates> typeMatcher2 = typeMatcher
202                                .skipBefore(EnumSet.of(IS, AS, UNDER)).optional(
203                                                EnumSet.of(IS, AS));
204                typeMatcher2.sequence(EnumSet.of(OBJECT, UNDER)).skipTo(LPAREN)
205                                .parseUntil(DECLARATIONS)
206                                // closing paren is swallowed by decl rules
207                                .repeated(EnumSet.of(NOT, FINAL, INSTANTIABLE))
208                                .sequence(EnumSet.of(SEMICOLON, DIV)).endNode();
209                typeMatcher2.skipTo(EnumSet.of(SEMICOLON, DIV)).endNode();
210
211                // top-level code block
212                inState(DECLARATIONS).sequence(BEGIN)
213                                .createNode(EShallowEntityType.METHOD, "top-level code")
214                                .parseUntil(STATEMENTS).sequence(END).skipTo(SEMICOLON)
215                                .endNode();
216        }

218        /** Create parser rules for functions, procedures, constructors, etc. */
219        private void createMethodAndAttributeRules() {
220
221                // function/procedure
222                RecognizerBase<EPlsqlParserStates> methodStart = createOrReplace()
223                                .repeated(
224                                                EnumSet.of(MAP, NOT, OVERRIDING, ORDER, FINAL,
225                                                                INSTANTIABLE, MEMBER, STATIC, CONSTRUCTOR))
226                                .markStart().sequence(EnumSet.of(PROCEDURE, FUNCTION))
227                                .repeated(PLSQL_IDENTIFIERS, DOT).sequence(PLSQL_IDENTIFIERS)
228                                .createNode(EShallowEntityType.METHOD, 0, new Region(1, -1))
229                                .skipNested(LPAREN, RPAREN).optional(RETURN, SELF, AS, RESULT)
230                                .skipBefore(EnumSet.of(SEMICOLON, IS, AS, RPAREN, COMMA));
231                methodStart.sequence(EnumSet.of(SEMICOLON, RPAREN, COMMA)).endNode();
232                RecognizerBase<EPlsqlParserStates> methodStart2 = methodStart
233                                .sequence(EnumSet.of(IS, AS));
234                methodStart2.sequence(EnumSet.of(LANGUAGE, EXTERNAL)).skipTo(SEMICOLON)
235                                .endNode();
236                methodStart2
237                                .parseUntil(DECLARATIONS)
238                                .sequence(BEGIN)
239                                .parseUntil(STATEMENTS)
240                                .sequence(END)
241                                .skipToWithNesting(EnumSet.of(SEMICOLON, RPAREN, COMMA),
242                                                LPAREN, RPAREN).endNode();
243
244                // exception declaration
245                inState(DECLARATIONS)
246                                .sequence(PLSQL_IDENTIFIERS, EXCEPTION, SEMICOLON)
247                                .createNode(EShallowEntityType.META, "exception declaration", 0)
248                                .endNode();
249
250                // trigger
251                RecognizerBase<EPlsqlParserStates> triggerMatch = createOrReplace()
252                                .markStart().sequence(TRIGGER).repeated(PLSQL_IDENTIFIERS, DOT)
253                                .sequence(PLSQL_IDENTIFIERS)
254                                .createNode(EShallowEntityType.METHOD, 0, new Region(1, -1))
255                                .skipBefore(EnumSet.of(SEMICOLON, DECLARE, BEGIN));
256                triggerMatch.sequence(SEMICOLON).endNode();
257                triggerMatch.sequence(BEGIN).parseUntil(STATEMENTS).sequence(END)
258                                .skipTo(SEMICOLON).endNode();
259                triggerMatch.sequence(DECLARE).parseUntil(DECLARATIONS).sequence(BEGIN)
260                                .parseUntil(STATEMENTS).sequence(END).skipTo(SEMICOLON)
261                                .endNode();
262
263                // cursor declaration
264                inState(DECLARATIONS).sequence(CURSOR).skipTo(SEMICOLON)
265                                .createNode(EShallowEntityType.ATTRIBUTE, "cursor", 1)
266                                .endNode();
267
268                // variables and constants
269                inState(DECLARATIONS)
270                                .sequence(PLSQL_IDENTIFIERS)
271                                .createNode(EShallowEntityType.ATTRIBUTE, "variable", 0)
272                                .skipToWithNesting(EnumSet.of(SEMICOLON, RPAREN, COMMA),
273                                                LPAREN, RPAREN).endNode();
274        }

276        /** Matches the optional CREATE OR REPLACE clause. */
277        private RecognizerBase<EPlsqlParserStates> createOrReplace() {
278                return inState(DECLARATIONS).optional(CREATE).optional(OR, REPLACE);
279        }

281        /** Creates parser rules for statements. */
282        private void createStatementRules() {
283                // if/elseif
284                RecognizerBase<EPlsqlParserStates> ifAlternative = inState(STATEMENTS)
285                                .sequence(EnumSet.of(IF, ELSIF))
286                                .createNode(EShallowEntityType.STATEMENT, 0)
287                                .skipToWithNesting(THEN, CASE, END).parseUntil(STATEMENTS)
288                                .sequenceBefore(EnumSet.of(ELSIF, ELSE, END));
289                ifAlternative.sequence(END, IF, SEMICOLON).endNode();
290                ifAlternative.endNodeWithContinuation();
291
292                // else (both for if and case)
293                RecognizerBase<EPlsqlParserStates> elseMatcher = inState(STATEMENTS)
294                                .sequence(ELSE).createNode(EShallowEntityType.STATEMENT, 0)
295                                .parseUntil(STATEMENTS);
296                elseMatcher.sequence(END, IF).skipTo(SEMICOLON).endNode();
297                elseMatcher.sequenceBefore(END, CASE).endNode();
298
299                // loops
300                inState(STATEMENTS).sequence(LOOP)
301                                .createNode(EShallowEntityType.STATEMENT, 0)
302                                .parseUntil(STATEMENTS).sequence(END, LOOP).skipTo(SEMICOLON)
303                                .endNode();
304                inState(STATEMENTS).sequence(EnumSet.of(WHILE, FOR))
305                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(LOOP)
306                                .parseUntil(STATEMENTS).sequence(END, LOOP).skipTo(SEMICOLON)
307                                .endNode();
308
309                // blocks
310                inState(STATEMENTS).sequence(DECLARE)
311                                .createNode(EShallowEntityType.STATEMENT, "block")
312                                .parseUntil(DECLARATIONS).sequence(BEGIN)
313                                .parseUntil(STATEMENTS).sequence(END).skipTo(SEMICOLON)
314                                .endNode();
315                inState(STATEMENTS).sequence(BEGIN)
316                                .createNode(EShallowEntityType.STATEMENT, "block")
317                                .parseUntil(STATEMENTS).sequence(END).skipTo(SEMICOLON)
318                                .endNode();
319
320                // case
321                inState(STATEMENTS).sequence(CASE)
322                                .createNode(EShallowEntityType.STATEMENT, 0)
323                                .skipBefore(EnumSet.of(WHEN, ELSE)).parseUntil(STATEMENTS)
324                                .sequence(END, CASE).skipTo(SEMICOLON).endNode();
325
326                // when (in exceptions or case)
327                inState(STATEMENTS).sequence(WHEN).skipTo(THEN)
328                                .createNode(EShallowEntityType.META, "when", 1).endNode();
329
330                // labels
331                inState(STATEMENTS)
332                                .sequence(LEFT_LABEL_BRACKET, PLSQL_IDENTIFIERS,
333                                                RIGHT_LABEL_BRACKET)
334                                .createNode(EShallowEntityType.META, SubTypeNames.LABEL, 1)
335                                .endNode();
336
337                // basic statement
338                EnumSet<ETokenType> basicStatementStarts = EnumSet
339                                .copyOf(PLSQL_IDENTIFIERS);
340                basicStatementStarts.addAll(Arrays.asList(SELF, RETURN, GOTO, FETCH,
341                                NULL_LITERAL));
342                inState(STATEMENTS).sequence(basicStatementStarts)
343                                .createNode(EShallowEntityType.STATEMENT, 0).skipTo(SEMICOLON)
344                                .endNode();
345        }

347        /** Matcher for identifiers by a prefix. */
348        private static class IdentifierPrefixMatcher implements ITokenMatcher {
349
350                /** The prefix. */
351                private final String prefix;
352
353                /** Constructor. */
354                public IdentifierPrefixMatcher(String prefix) {
355                        this.prefix = prefix.toLowerCase();
356                }
357
358                /** {@inheritDoc} */
359                @Override
360                public boolean matches(IToken token) {
361                        return token.getType() == IDENTIFIER
362                                        && token.getText().toLowerCase().startsWith(prefix);
363                }
364        }
