017package eu.cqse.check.framework.shallowparser.languages.python;

019import static eu.cqse.check.framework.shallowparser.languages.python.PythonShallowParser.EPythonParserStates.ANY;
020import static eu.cqse.check.framework.shallowparser.languages.python.PythonShallowParser.EPythonParserStates.IN_CLASS;
021import static eu.cqse.check.framework.scanner.ETokenType.AT;
022import static eu.cqse.check.framework.scanner.ETokenType.CLASS;
023import static eu.cqse.check.framework.scanner.ETokenType.COLON;
024import static eu.cqse.check.framework.scanner.ETokenType.DEDENT;
025import static eu.cqse.check.framework.scanner.ETokenType.DEF;
026import static eu.cqse.check.framework.scanner.ETokenType.DOT;
027import static eu.cqse.check.framework.scanner.ETokenType.ELIF;
028import static eu.cqse.check.framework.scanner.ETokenType.ELSE;
029import static eu.cqse.check.framework.scanner.ETokenType.EOL;
030import static eu.cqse.check.framework.scanner.ETokenType.EXCEPT;
031import static eu.cqse.check.framework.scanner.ETokenType.FINALLY;
032import static eu.cqse.check.framework.scanner.ETokenType.FOR;
033import static eu.cqse.check.framework.scanner.ETokenType.FROM;
034import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
035import static eu.cqse.check.framework.scanner.ETokenType.IF;
036import static eu.cqse.check.framework.scanner.ETokenType.IMPORT;
037import static eu.cqse.check.framework.scanner.ETokenType.INDENT;
038import static eu.cqse.check.framework.scanner.ETokenType.LBRACK;
039import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
040import static eu.cqse.check.framework.scanner.ETokenType.RBRACK;
041import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
042import static eu.cqse.check.framework.scanner.ETokenType.SEMICOLON;
043import static eu.cqse.check.framework.scanner.ETokenType.TRY;
044import static eu.cqse.check.framework.scanner.ETokenType.WHILE;
045import static eu.cqse.check.framework.scanner.ETokenType.WITH;
046
047import java.util.EnumSet;
048
049import eu.cqse.check.framework.shallowparser.SubTypeNames;
050import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
051import eu.cqse.check.framework.shallowparser.framework.RecognizerBase;
052import eu.cqse.check.framework.shallowparser.framework.ShallowParserBase;
053import eu.cqse.check.framework.shallowparser.languages.python.PythonShallowParser.EPythonParserStates;
054import org.conqat.lib.commons.region.Region;
055import eu.cqse.check.framework.scanner.IToken;
056
057/**
058 * Shallow parser for Python.
059 *
060 * @author $Author: hummelb $
061 * @version $Rev: 52657 $
062 * @ConQAT.Rating GREEN Hash: 21D8CE326C9F28F3B89B6B6B68151786
063 */
064public class PythonShallowParser extends ShallowParserBase<EPythonParserStates> {
065
066        /** The states used in this parser. */
067        public static enum EPythonParserStates {
068
069                /**
070                 * Any state apart from in-class states. Typically, any construct can
071                 * occur at any place.
072                 */
073                ANY,
074                
075                /** State applying within a class scope. */
076                IN_CLASS
077        }
078
079        /** Constructor. */
080        public PythonShallowParser() {
081                super(EPythonParserStates.class, EPythonParserStates.ANY);
082
083                createErrorRules();
084                createImportRules();
085                createAnnotationRules();
086                createClassRules();
087                createFunctionRules();
088                createStatementRules();
089        }
090
091        /** Create rules for handling error handling. */
092        private void createErrorRules() {
093                // unmatched indent/dedent: no endNode to keep the node incomplete
094                inAnyState().sequence(INDENT).createNode(EShallowEntityType.META,
095                                "Unmatched indent");
096                inAnyState().sequence(DEDENT).createNode(EShallowEntityType.META,
097                                "Unmatched dedent");
098        }
099
100        /** Creates parsing rules for imports. */
101        private void createImportRules() {
102                inAnyState().sequence(EnumSet.of(IMPORT, FROM))
103                                .createNode(EShallowEntityType.META, 0).skipTo(EOL).endNode();
104        }
105
106        /** Creates parsing rules for annotations. */
107        private void createAnnotationRules() {
108                inAnyState()
109                                .sequence(AT, IDENTIFIER)
110                                .repeated(DOT, IDENTIFIER)
111                                .createNode(EShallowEntityType.META, "annotation",
112                                                new Region(1, -1)).skipNested(LPAREN, RPAREN).endNode();
113        }
114
115        /** Creates parsing rules for classes. */
116        private void createClassRules() {
117                RecognizerBase<EPythonParserStates> classAlternative = inAnyState()
118                                .sequence(CLASS, IDENTIFIER).skipNested(LPAREN, RPAREN)
119                                .sequence(COLON)
120                                .createNode(EShallowEntityType.TYPE, "class", 1);
121                addBlockClosingAlternatives(classAlternative, IN_CLASS);
122        }
123
124        /** Creates parsing rules for functions. */
125        private void createFunctionRules() {
126                RecognizerBase<EPythonParserStates> functionAlternative = inAnyState()
127                                .sequence(DEF, IDENTIFIER).skipNested(LPAREN, RPAREN)
128                                .sequence(COLON)
129                                .createNode(EShallowEntityType.METHOD, "method", 1);
130                addBlockClosingAlternatives(functionAlternative, ANY);
131        }
132
133        /** Creates parsing rules for statements. */
134        private void createStatementRules() {
135                // empty statement
136                inAnyState()
137                                .sequence(SEMICOLON)
138                                .createNode(EShallowEntityType.STATEMENT,
139                                                SubTypeNames.EMPTY_STATEMENT).endNode();
140
141                // block statements
142                RecognizerBase<EPythonParserStates> ifAlternative = inAnyState()
143                                .sequence(
144                                                EnumSet.of(IF, ELIF, ELSE, TRY, EXCEPT, FINALLY, WHILE,
145                                                                FOR, WITH))
146                                .createNode(EShallowEntityType.STATEMENT, 0)
147                                .skipToWithNesting(COLON, LBRACK, RBRACK);
148                addBlockClosingAlternatives(ifAlternative, ANY);
149
150                // remove any isolated EOLs
151                inAnyState().sequence(EOL);
152
153                // in class attributes
154                inState(IN_CLASS).sequenceBefore(IDENTIFIER)
155                                .subRecognizer(new PythonAttributeRecognizer(), 1, 1).endNode();
156
157                // simple statement
158                inAnyState()
159                                .sequenceBefore(EnumSet.complementOf(EnumSet.of(DEF, CLASS)))
160                                .subRecognizer(new PythonSimpleStatementRecognizer(), 1, 1)
161                                .endNode();
162        }
163
164        /**
165         * Adds two different rules for closing a block:
166         * <ul>
167         * <li>Closing a block by finding a dedent</li>
168         * <li>Single line that ends with EOL, typically this means multiple
169         * statements on one line</li>
170         * </ul>
171         * 
172         * @param matchingAlternative
173         *            The block recognizer to be closed.
174         * @param innerBlockState
175         *            The {@link EPythonParserStates} used within the block this
176         *            method is closing.
177         */
178        private void addBlockClosingAlternatives(
179                        RecognizerBase<EPythonParserStates> matchingAlternative,
180                        EPythonParserStates innerBlockState) {
181                matchingAlternative.sequence(EOL, INDENT).parseUntil(innerBlockState)
182                                .sequence(DEDENT).endNode();
183                matchingAlternative.parseUntil(innerBlockState).sequence(EOL).endNode();
184        }
185
186        /** {@inheritDoc} */
187        @Override
188        protected boolean isFilteredToken(IToken token, IToken previousToken) {
189                if (super.isFilteredToken(token, previousToken)) {
190                        return true;
191                }
192                // Don't allow double EOLs
193                return previousToken != null && previousToken.getType() == EOL
194                                && token.getType() == EOL;
195        }
