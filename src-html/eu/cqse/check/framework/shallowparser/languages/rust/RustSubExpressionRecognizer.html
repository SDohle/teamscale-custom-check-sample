017package eu.cqse.check.framework.shallowparser.languages.rust;
018
019import static eu.cqse.check.framework.scanner.ETokenType.GT;
020import static eu.cqse.check.framework.scanner.ETokenType.IF;
021import static eu.cqse.check.framework.scanner.ETokenType.LBRACE;
022import static eu.cqse.check.framework.scanner.ETokenType.MATCH;
023import static eu.cqse.check.framework.scanner.ETokenType.OR;
024import static eu.cqse.check.framework.scanner.ETokenType.OROR;
025import static eu.cqse.check.framework.scanner.ETokenType.RBRACE;
026import static eu.cqse.check.framework.scanner.ETokenType.RBRACK;
027import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
028import static eu.cqse.check.framework.scanner.ETokenType.ETokenClass.DELIMITER;
029import static eu.cqse.check.framework.scanner.ETokenType.ETokenClass.OPERATOR;
030import static eu.cqse.check.framework.shallowparser.languages.rust.RustShallowParser.ERustParserStates.IN_EXPRESSION;
031
032import java.util.EnumSet;
033import java.util.List;
034
035import eu.cqse.check.framework.scanner.ETokenType;
036import eu.cqse.check.framework.scanner.ETokenType.ETokenClass;
037import eu.cqse.check.framework.scanner.IToken;
038import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
039import eu.cqse.check.framework.shallowparser.framework.ParserState;
040import eu.cqse.check.framework.shallowparser.framework.RecognizerBase;
041import eu.cqse.check.framework.shallowparser.languages.rust.RustShallowParser.ERustParserStates;
042
043/**
044 * A sub-expression recognizer that finds if, match and anonymous block and
045 * lambda expressions in Rust.
046 */
047public class RustSubExpressionRecognizer extends RecognizerBase<ERustParserStates> {
048
049        /**
050         * A set of token types that start a sub structure expression like if and
051         * match.
052         */
053        private static final EnumSet<ETokenType> SUB_STRUCTURE_START_TOKENS = EnumSet.of(IF, MATCH);
054
055        /**
056         * A set of token classes that indicate a sub expression block if they are
057         * followed by a opening brace.
058         */
059        private static final EnumSet<ETokenClass> SUB_BLOCK_INDICATOR_CLASSES = EnumSet.of(OPERATOR, DELIMITER);
060
061        /**
062         * A set of token types that indicate that a following brace is no sub
063         * expression block.
064         */
065        private static final EnumSet<ETokenType> SUB_BLOCK_EXCLUDED_TOKENS = EnumSet.of(RPAREN, RBRACK, RBRACE, GT);
066
067        /**
068         * A set of token classes that indicate a lambda expression block if they
069         * are followed by a pipe.
070         */
071        private static final EnumSet<ETokenClass> LAMBDA_INDICATOR_CLASSES = EnumSet.of(OPERATOR, DELIMITER);
072
073        /**
074         * A set of token types that indicate that a following pipe is no lambda.
075         */
076        private static final EnumSet<ETokenType> LAMBDA_EXCLUDED_TOKENS = EnumSet.of(RPAREN, RBRACK, RBRACE);
077
078        /** A set of token types that start a lambda expression. */
079        private static final EnumSet<ETokenType> LAMBDA_START_TOKENS = EnumSet.of(OR, OROR);
080
081        /** {@inheritDoc} */
082        @Override
083        protected int matchesLocally(ParserState<ERustParserStates> parserState, List<IToken> tokens, int startOffset) {
084                if (startOffset < tokens.size()) {
085                        if (isSubStructureStart(tokens, startOffset)) {
086                                return parserState.parse(IN_EXPRESSION, tokens, startOffset);
087                        } else if (isSubBlockStart(tokens, startOffset)) {
088                                return parserState.parse(IN_EXPRESSION, tokens, startOffset + 1);
089                        } else if (isLambdaStart(tokens, startOffset)) {
090                                return parserState.parse(IN_EXPRESSION, tokens, startOffset);
091                        }
092                }
093                return NO_MATCH;
094        }
095
096        /**
097         * Returns whether a sub structure starts at the given offset in the given
098         * token list.
099         */
100        private static boolean isSubStructureStart(List<IToken> tokens, int startOffset) {
101                return SUB_STRUCTURE_START_TOKENS.contains(tokens.get(startOffset).getType());
102        }
103
104        /**
105         * Returns whether a sub expression block starts at the given offset in the
106         * given token list. The heuristic is to look at the token in front of the
107         * opening brace and check whether it is an operator or delimiter.
108         */
109        private static boolean isSubBlockStart(List<IToken> tokens, int startOffset) {
110                ETokenType firstType = tokens.get(startOffset).getType();
111                return !SUB_BLOCK_EXCLUDED_TOKENS.contains(firstType)
112                                && SUB_BLOCK_INDICATOR_CLASSES.contains(firstType.getTokenClass())
113                                && TokenStreamUtils.tokenTypesAt(tokens, startOffset + 1, LBRACE);
114        }
115
116        /**
117         * Returns whether a lambda starts at the given offset in the given token
118         * list. The heuristic is to look for | or || tokens that are preceeded by
119         * either an operator or delimiter and thus are no bitwise operators. Some
120         * tokens like closing parenthesis, brackets and braces still indicate that
121         * the following pipe does not start a lambda.
122         */
123        private static boolean isLambdaStart(List<IToken> tokens, int startOffset) {
124                if (!isLambdaIndicated(tokens, startOffset)) {
125                        return false;
126                }
127
128                return startOffset >= 0 && LAMBDA_START_TOKENS.contains(tokens.get(startOffset).getType());
129        }
130
131        /**
132         * Returns whether the token that preceeds the token at startOffset
133         * indicates a lambda expression.
134         */
135        private static boolean isLambdaIndicated(List<IToken> tokens, int startOffset) {
136                if (startOffset <= 0) {
137                        return false;
138                }
139                ETokenType previousType = tokens.get(startOffset - 1).getType();
140                return LAMBDA_INDICATOR_CLASSES.contains(previousType.getTokenClass())
141                                && !LAMBDA_EXCLUDED_TOKENS.contains(previousType);
142        }
143
144}
