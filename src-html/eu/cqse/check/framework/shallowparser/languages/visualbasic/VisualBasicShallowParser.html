039/**
040 * VisualBasic.NET shallow parser.
041 * 
042 * The shallow parser follows the
043 * <a href="https://msdn.microsoft.com/en-us/library/aa711636(v=vs.71).aspx">VB.
044 * NET</a> specification at
045 * 
046 * 
047 * Notes:
048 * <ul>
049 * <li>At the current state the parser should support multiple visual basic
050 * dialects. There is only one exception: The parser can not deal with single
051 * line if statements in other dialects. Expect incomplete entities in those
052 * cases.</li>
053 * <li>Some unreserved keywords are treated like reserved keywords, as they are
054 * only valid in a certain context (e.g. Linq expressions) and we currently do
055 * not have such functionality (
056 * <a href="https://msdn.microsoft.com/en-us/library/dd409611.aspx">VB.NET
057 * Keywords</a>).</li>
058 * <li>Getter and Setter of properties are reported as e.g. "getter for <name of
059 * property>".</li>
060 * <li>Attribute lists that give additional compiler information to all kinds of
061 * types, methods etc. are reported as meta annotations.</li>
062 * <li>Preprocessor directives are reported as meta entity.</li>
063 * <li>Do-Until loops are reported as do-while loops. Note that do loops in
064 * VB.NET may behave like while loops, depending on the position of the
065 * condition.</li>
066 * <li>The Declare structure is reported as method.</li>
067 * <li>The form meta data is not parsed into.</li>
068 * <li>Lambdas are not distinguished from other variables.</li>
069 * </ul>
070 */
071public class VisualBasicShallowParser extends ShallowParserBase<EVisualBasicParserState> {
072
073        /**
074         * All token types that open a new scope.
075         */
076        private static final List<ETokenType> OPENING_TYPES = Arrays.asList(LPAREN, LBRACE, LBRACK);
077
078        /**
079         * All token types that close a scope.
080         */
081        private static final List<ETokenType> CLOSING_TYPES = Arrays.asList(RPAREN, RBRACE, RBRACK);
082
083        /**
084         * All types that do not indicate a new line. This excludes COLON as it
085         * allows to continue with a new statement in the very same line.
086         */
087        private static final EnumSet<ETokenType> NO_NEW_LINE = EnumSet.complementOf(EnumSet.of(EOL, COLON));
088
089        /**
090         * All token types that may modify type, attribute or method declarations.
091         * Does not include 'MUSTOVERRIDE' as this is an indicator for a different
092         * rule.
093         */
094        private static final EnumSet<ETokenType> DECLARATION_MODIFIER = EnumSet.of(PRIVATE, PUBLIC, PROTECTED, MUSTINHERIT,
095                        NOTINHERITABLE, OVERLOADS, SHARED, OVERRIDES, SHADOWS, NOTOVERRIDABLE, FRIEND, READONLY, ASYNC, WITHEVENTS,
096                        PARTIAL, WIDENING, NARROWING, CONST, STATIC, DEFAULT, DIM);
097
098        /**
099         * All token types a statement can start with.
100         */
101        private static final EnumSet<ETokenType> STATEMENT_START = EnumSet.of(IDENTIFIER, DIM, RETURN, ME, MYBASE, THROW,
102                        AWAIT, RAISEEVENT, REDIM, ADDHANDLER, REMOVEHANDLER, EXIT_DO, EXIT_FOR, EXIT_SUB, EXIT_WHILE, RESUME,
103                        ONERROR, STOP, GOTO, YIELD, ERROR, CALL, CONTINUE, END, ERASE, CTYPE);
104
105        /**
106         * Types that continue a statement only if EOL is a preceding token or EOL
107         * is a following token.
108         * https://msdn.microsoft.com/en-us/library/865x40k4.aspx
109         */
110        private static final EnumSet<ETokenType> STATEMENT_CONTINUATION_BOTH_ONLY = EnumSet.of(COMMA, EQ, ANDEQ, PLUSEQ,
111                        MINUSEQ, MULTEQ, DIVEQ, INTDIVEQ, POWEREQ, LSHIFTEQ, RSHIFTEQ, PLUS, MINUS, DIV, INTDIV, MULT, MOD, NOT, GT,
112                        LTEQ, GTEQ, POWER, LSHIFT, RSHIFT, ANDALSO, OR, ORELSE, LIKE, XOR, IS, IS_NOT, AGGREGATE, GROUPBY,
113                        GROUPJOIN, JOIN, LET, ORDERBY, SELECT, SKIP, SKIPWHILE, TAKE, TAKEWHILE, WHERE, IN, INTO, ON, ASCENDING,
114                        DESCENDING, FROM, EQUALS);
115
116        /** Types that continue a statement only if EOL is a preceding token. */
117        private static final EnumSet<ETokenType> STATEMENT_CONTINUATION_PRECEDING_ONLY = EnumSet.of(DISTINCT);
118
119        /** Types that continue a statement only if EOL is a following token. */
120        private static final EnumSet<ETokenType> STATEMENT_CONTINUATION_FOLLOWING_ONLY = EnumSet.of(DOT, LT);
121
122        /**
123         * Types that interrupt a statement. This is typically EOL. Furthermore it
124         * contains COLON which directly ends the statement and allows the next
125         * statement to start in the same line. Because of single line if statements
126         * we have to add ELSE here.
127         */
128        private static final EnumSet<ETokenType> STATEMENT_INTERRUPT_TYPES = EnumSet.of(EOL, COLON, ELSE, END_IF, END_WHILE,
129                        END_SUB, END_FUNCTION, END_NAMESPACE, END_USING, END_WITH, END_TRY, END_EVENT, END_SYNCLOCK, END_SELECT,
130                        END_ENUM, END_CLASS, END_TYPE, END_GET, END_SET, END_OPERATOR, END_INTERFACE, END_STRUCTURE);
131
132        /**
133         * Types that continue a statement if EOL is a preceding token or EOL is a
134         * following token. https://msdn.microsoft.com/en-us/library/865x40k4.aspx
135         */
136        private static final EnumSet<ETokenType> STATEMENT_CONTINUATION_BOTH = EnumSet
137                        .copyOf(STATEMENT_CONTINUATION_BOTH_ONLY);
138
139        /** Types that continue a statement if EOL is a preceding token. */
140        private static final EnumSet<ETokenType> STATEMENT_CONTINUATION_PRECEDING = EnumSet
141                        .copyOf(STATEMENT_CONTINUATION_PRECEDING_ONLY);
142
143        /** Types that continue a statement if EOL is a following token. */
144        private static final EnumSet<ETokenType> STATEMENT_CONTINUATION_FOLLOWING = EnumSet
145                        .copyOf(STATEMENT_CONTINUATION_FOLLOWING_ONLY);
146        /**
147         * Types that indicate a situation where implicit line conversion might be
148         * applicable or should not be applied.
149         */
150        private static final EnumSet<ETokenType> STATEMENT_CONTINUATION_CANDIDATES = EnumSet
151                        .copyOf(STATEMENT_INTERRUPT_TYPES);
152
153        static {
154                STATEMENT_CONTINUATION_CANDIDATES.addAll(STATEMENT_CONTINUATION_FOLLOWING_ONLY);
155                STATEMENT_CONTINUATION_CANDIDATES.addAll(STATEMENT_CONTINUATION_PRECEDING_ONLY);
156                STATEMENT_CONTINUATION_CANDIDATES.addAll(STATEMENT_CONTINUATION_BOTH_ONLY);
157
158                STATEMENT_CONTINUATION_PRECEDING.addAll(STATEMENT_CONTINUATION_BOTH_ONLY);
159                STATEMENT_CONTINUATION_FOLLOWING.addAll(STATEMENT_CONTINUATION_BOTH_ONLY);
160
161                STATEMENT_CONTINUATION_BOTH.addAll(STATEMENT_CONTINUATION_FOLLOWING_ONLY);
162                STATEMENT_CONTINUATION_BOTH.addAll(STATEMENT_CONTINUATION_PRECEDING_ONLY);
163        }

186        /**
187         * Hide line separators EOL and COLON by parsing them without creating
188         * nodes.
189         */
190        private void hideLineSeparators() {
191
192                // Does not include IN_SINGLE_LINE_IF, because we require EOL and COLON
193                // as indicators.
194                inState(TOP_LEVEL, IN_TYPE, IN_ENUM, IN_EVENT_DECLARATION, IN_INTERFACE, IN_METHOD)
195                                .sequence(EnumSet.of(EOL, COLON));
196
197        }

199        /**
200         * Create rules to match event declarations.
201         */
202        private void createEventRules() {
203                skipToEndOfStatement(inState(IN_TYPE).repeated(DECLARATION_MODIFIER).sequence(EVENT).markStart()
204                                .createNode(EShallowEntityType.ATTRIBUTE, "event", 0)).endNode();
205
206                RecognizerBase<EVisualBasicParserState> base = skipToEndOfStatement(
207                                inState(IN_TYPE).repeated(DECLARATION_MODIFIER).sequence(IDENTIFIER, EVENT).markStart()
208                                                .createNode(EShallowEntityType.ATTRIBUTE, "event", 0));
209                base.parseUntil(IN_EVENT_DECLARATION).sequence(END_EVENT).endNode();
210
211                // Parse the handler definitions inside of the custom event
212                EVisualBasicParserState[] states = { IN_EVENT_DECLARATION };
213                createComplexStatementRule(states, ADDHANDLER, END, ADDHANDLER);
214                createComplexStatementRule(states, REMOVEHANDLER, END, REMOVEHANDLER);
215                createComplexStatementRule(states, RAISEEVENT, END, RAISEEVENT);
216        }

218        /**
219         * Create rules to match visual basics equivalent to switch/case:
220         * select/case.
221         */
222        private void createSelectCaseRules() {
223                skipToEndOfStatement(inState(IN_METHOD, IN_SINGLE_LINE_IF).sequence(SELECTCASE)
224                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.SELECT)).parseUntil(IN_METHOD)
225                                                .sequence(ETokenType.END_SELECT).endNode();
226
227                skipToEndOfStatement(inState(IN_METHOD, IN_SINGLE_LINE_IF).sequence(CASE).createNode(EShallowEntityType.META,
228                                SubTypeNames.CASE)).optional(ELSE).endNode();
229        }

231        /**
232         * Create rules that match all kinds of loops or the goto statement.
233         */
234        private void createLoopRules() {
235                // DO-WHILE
236                RecognizerBase<EVisualBasicParserState> baseRecognizerDoWhile = inState(IN_METHOD, IN_SINGLE_LINE_IF)
237                                .sequence(DO).optional(EnumSet.of(WHILE, UNTIL));
238                baseRecognizerDoWhile = skipToEndOfStatement(
239                                baseRecognizerDoWhile.createNode(EShallowEntityType.STATEMENT, SubTypeNames.DO)).parseUntil(IN_METHOD)
240                                                .sequence(LOOP);
241                skipToEndOfStatement(baseRecognizerDoWhile).endNode();
242
243                // WHILE
244                skipToEndOfStatement(inState(IN_METHOD, IN_SINGLE_LINE_IF).sequence(WHILE)
245                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.WHILE)).parseUntil(IN_METHOD).sequence(END_WHILE)
246                                                .endNode();
247
248                // FOR
249                RecognizerBase<EVisualBasicParserState> baseRecognizerFor = skipToEndOfStatement(
250                                inState(IN_METHOD, IN_SINGLE_LINE_IF).sequence(FOR).createNode(EShallowEntityType.STATEMENT,
251                                                SubTypeNames.FOR)).parseUntil(IN_METHOD).sequence(NEXT);
252                skipToEndOfStatement(baseRecognizerFor).endNode();
253
254        }

256        /**
257         * Create a rule for complex blocks that contain nodes that have to end with
258         * continuation. These are e.g. try/catch/finally or if/elseif/else
259         * constructs.
260         */
261        private void createBlockRuleWithContinuation(EnumSet<ETokenType> possibleStartTypes,
262                        EnumSet<ETokenType> continuationIndicators, ETokenType terminationType) {
263
264                RecognizerBase<EVisualBasicParserState> base = inState(IN_METHOD, IN_SINGLE_LINE_IF)
265                                .sequence(possibleStartTypes).createNode(EShallowEntityType.STATEMENT, -1);
266                base = skipToEndOfStatement(base).parseUntil(IN_METHOD);
267
268                base.sequenceBefore(continuationIndicators).endNodeWithContinuation();
269                base.sequence(terminationType).endNode();
270        }
271
272        /**
273         * Create rules that match try catch finally constructs.
274         */
275        private void createTryCatchFinallyRules() {
276                createBlockRuleWithContinuation(EnumSet.of(TRY, CATCH, FINALLY), EnumSet.of(CATCH, FINALLY), END_TRY);
277        }

280         * Create rules for more complex statements like WITH, USING and SYNCLOCK.

282        private void createComplexStatementRules() {

300         * Create the rules for enum declaration and definition.
301         */
302        private void createEnumRules() {
303                // Enum declaration
304                RecognizerBase<EVisualBasicParserState> base = inState(TOP_LEVEL, IN_TYPE).repeated(DECLARATION_MODIFIER)

306                skipToEndOfStatement(base).parseUntil(IN_ENUM).sequence(END_ENUM).endNode();

314        /**
315         * Create rules that match properties. A matched property is reported as the
316         * shallow entity 'ATTRIBUTE' with potential getter and setter methods.
317         */
318        private void createPropertyRules() {
319                RecognizerBase<EVisualBasicParserState> base = inState(IN_TYPE).repeated(DECLARATION_MODIFIER)
320                                .sequence(PROPERTY, IDENTIFIER).createNode(EShallowEntityType.ATTRIBUTE, SubTypeNames.PROPERTY, -1);
321
322                base = skipToEndOfStatement(base);
323
324                // The property is treated as a type if a getter or setter is found
325                // next.
326                base.sequence(EOL).sequenceBefore(EnumSet.of(GET, SET)).parseUntil(IN_TYPE).sequence(END_PROPERTY).endNode();
327
328                // Otherwise the property could have an automatically created getter and
329                // setter and therefore will end without a body.
330                base.optional(END_PROPERTY).endNode();
331        }

333        /**
334         * Create rules that cover conditionals (If, ElseIf).
335         */
336        private void createConditionalsRules() {
337                // For single line if statements we cannot use the same method as for
338                // try-catch structures, so we check this case with a precondition.
339                // Refer to Single-Line Syntax for further details:
340                // https://msdn.microsoft.com/en-us/library/752y8abs.aspx
341
342                // Recognize single line if statements
343                RecognizerBase<EVisualBasicParserState> singleLineIfSelector = emptyRecognizer();
344                singleLineIfSelector.sequence(IF).skipTo(THEN).repeated(COLON).sequence(NO_NEW_LINE);
345
346                RecognizerBase<EVisualBasicParserState> base = inState(IN_METHOD, IN_SINGLE_LINE_IF)
347                                .preCondition(singleLineIfSelector).sequence(IF).createNode(EShallowEntityType.STATEMENT, -1)
348                                .skipTo(THEN).parseUntil(IN_SINGLE_LINE_IF).sequenceBefore(EnumSet.of(EOL, ELSE));
349
350                base.sequenceBefore(ELSE).endNodeWithContinuation();
351                base.sequence(EOL).endNode();
352
353                // Recognize single line else statements
354                RecognizerBase<EVisualBasicParserState> elseRecognizer = emptyRecognizer();
355                elseRecognizer.sequence(ELSE).repeated(COLON).sequenceBefore(NO_NEW_LINE);
356
357                inState(IN_METHOD, IN_SINGLE_LINE_IF).preCondition(elseRecognizer).sequence(ELSE)
358                                .createNode(EShallowEntityType.STATEMENT, -1).parseUntil(IN_SINGLE_LINE_IF).sequence(EOL)
359                                .optional(END_IF).endNode();
360
361                // Multi line If Statements
362                createBlockRuleWithContinuation(EnumSet.of(IF, ELSEIF, ELSE), EnumSet.of(ELSEIF, ELSE), END_IF);
363
364        }

376         * Create a rule that matches simple statements.
377         */
378        private void createSimpleStatementRule() {
379                // Local variables either begin with at least one modifier
380                RecognizerBase<EVisualBasicParserState> localVariable = inState(IN_METHOD, IN_SINGLE_LINE_IF)

385                // Allow a statement to start with DOT, too, because of its use in a
386                // WITH statement.
387                RecognizerBase<EVisualBasicParserState> simpleStatement = inState(TOP_LEVEL, IN_TYPE, IN_METHOD,

389                                                .createNode(EShallowEntityType.STATEMENT, SubTypeNames.SIMPLE_STATEMENT, -1);
390                skipToEndOfStatement(simpleStatement).endNode();

393        /**
394         * Create rules for methods.
395         */
396        private void createMethodRules() {
397                createMethodRule(FUNCTION, END_FUNCTION, IDENTIFIER, SubTypeNames.METHOD);
398                createMethodRule(SUB, END_SUB, IDENTIFIER, SubTypeNames.METHOD);
399                createMethodRule(SUB, END_SUB, NEW, SubTypeNames.CONSTRUCTOR);

401                // Special methods: GET and SET in properties.
402                createGetSetRule(GET, END_GET);
403                createGetSetRule(SET, END_SET);
404
405                // Operator overloading.
406                createOperatorRule();
407
408                // Declare
409                RecognizerBase<EVisualBasicParserState> baseRecognizerDeclare = inState(TOP_LEVEL, IN_TYPE)
410                                .repeated(DECLARATION_MODIFIER).sequence(ETokenType.DECLARE)
411                                // IDENTIFIER needed to match "PtrSafe"
412                                .optional(EnumSet.of(ANSI, AUTO, UNICODE, IDENTIFIER)).sequence(EnumSet.of(SUB, FUNCTION)).markStart()
413                                .createNode(EShallowEntityType.METHOD, SubTypeNames.DECLARATION, 0);
414                skipToEndOfStatement(baseRecognizerDeclare).endNode();
415        }

440        /**
441         * Create the rule to match a method that is identified by startTokenType
442         * and parses the contents of the method until endTokenType is reached.
443         */
444        private void createMethodRule(ETokenType startTokenType, ETokenType endTokenType, Object identifierTokenType,
445                        String subtypeName) {
446
447                // Classes may define methods with 'MUSTOVERRIDE' that do not have a
448                // method body.
449                RecognizerBase<EVisualBasicParserState> abstractTypeBase = inState(TOP_LEVEL, IN_TYPE)
450                                .repeated(DECLARATION_MODIFIER).sequence(MUSTOVERRIDE).repeated(DECLARATION_MODIFIER)
451                                .sequence(startTokenType).sequence(identifierTokenType)
452                                .createNode(EShallowEntityType.METHOD, subtypeName, -1);
453                skipToEndOfStatement(abstractTypeBase).endNode();
454
455                // Typical method declarations.
456                RecognizerBase<EVisualBasicParserState> typeBase = inState(TOP_LEVEL, IN_TYPE);
457                typeBase = appendMethodNode(typeBase, startTokenType, identifierTokenType, subtypeName);
458                skipToEndOfStatement(typeBase).parseUntil(IN_METHOD).sequence(endTokenType).endNode();
459
460                // For interfaces the methods do not have a body.
461                RecognizerBase<EVisualBasicParserState> interfaceBase = inState(IN_INTERFACE);
462                interfaceBase = appendMethodNode(interfaceBase, startTokenType, identifierTokenType, subtypeName);
463                skipToEndOfStatement(interfaceBase).endNode();
464
465        }

467        /**
468         * Create rule to match a method that is identified by startTokenType and
469         * parses the contents of the method until endTokenType is reached.
470         */
471        private static RecognizerBase<EVisualBasicParserState> appendMethodNode(
472                        RecognizerBase<EVisualBasicParserState> base, ETokenType startTokenType, Object identifierTokenType,
473                        String subtypeName) {
474                return base.repeated(DECLARATION_MODIFIER).sequence(startTokenType).sequence(identifierTokenType)
475                                .createNode(EShallowEntityType.METHOD, subtypeName, -1);
476        }

498        /**
499         * Create rules for classes etc.
500         */
501        private void createTypeLikeRules() {
502                createTypeLikeRule(EShallowEntityType.TYPE, IN_TYPE, CLASS, END_CLASS, SubTypeNames.CLASS);
503                createTypeLikeRule(EShallowEntityType.MODULE, IN_TYPE, MODULE, END_MODULE, SubTypeNames.MODULE);
504                createTypeLikeRule(EShallowEntityType.TYPE, IN_INTERFACE, INTERFACE, END_INTERFACE, SubTypeNames.INTERFACE);
505                createTypeLikeRule(EShallowEntityType.TYPE, IN_TYPE, STRUCTURE, END_STRUCTURE, SubTypeNames.STRUCTURE);
506                createTypeLikeRule(EShallowEntityType.TYPE, IN_TYPE, TYPE, END_TYPE, SubTypeNames.TYPE_ALIAS);
507        }
508
509        /**
510         * Appends a recognizer to the sequence, that repeatedly skips INHERITS and
511         * IMPLEMENTS code parts. This is necessary because they may continue a
512         * statement in the next line.
513         */
514        private RecognizerBase<EVisualBasicParserState> appendSkipToEndOfTypeRecognizer(
515                        RecognizerBase<EVisualBasicParserState> base) {
516
517                RecognizerBase<EVisualBasicParserState> extendedTypeRecognizer = emptyRecognizer();
518                skipToEndOfStatement(extendedTypeRecognizer.repeated(EOL).sequence(EnumSet.of(INHERITS, IMPLEMENTS)));
519
520                return base.repeatedSubRecognizer(extendedTypeRecognizer);
521        }
522
523        /**
524         * Create a rule for the given type. Modules are treated here as well, as
525         * they are syntactically similar to types.
526         */
527        private void createTypeLikeRule(EShallowEntityType shallowEntityType, EVisualBasicParserState subParseState,
528                        ETokenType startTokenType, ETokenType endTokenType, String subtypeName) {
529
530                RecognizerBase<EVisualBasicParserState> base = inState(TOP_LEVEL, IN_TYPE).repeated(DECLARATION_MODIFIER)
531                                .sequence(startTokenType, IDENTIFIER).createNode(shallowEntityType, subtypeName, -1);
532
533                appendSkipToEndOfTypeRecognizer(skipToEndOfStatement(base)).parseUntil(subParseState).sequence(endTokenType)
534                                .endNode();
535        }
536
537        /**
538         * Create the rules that recognize meta statements such as imports.
539         */
540        private void createMetaRules() {
541
542                // Attribute lists
543                inAnyState().sequence(LT).createNode(EShallowEntityType.META, SubTypeNames.ANNOTATION, 1)
544                                .skipToWithNesting(GT, OPENING_TYPES, CLOSING_TYPES).endNode();
545
546                // Matches attribute and version meta data in e.g. *.bas files
547                skipToEndOfStatement(inState(TOP_LEVEL).sequence(IDENTIFIER).createNode(EShallowEntityType.META, -1)).endNode();
548
549                // Imports
550                RecognizerBase<EVisualBasicParserState> baseRecognizerImports = inState(TOP_LEVEL).sequence(IMPORTS)
551                                .createNode(EShallowEntityType.META, SubTypeNames.IMPORT);
552                skipToEndOfStatement(baseRecognizerImports).endNode();
553
554                // Preprocessor directives
555                inAnyState().sequence(HASH).createNode(EShallowEntityType.META, "preprocessor directive").skipTo(EOL).endNode();
556
557                // Goto label
558                inState(IN_METHOD, IN_SINGLE_LINE_IF).sequence(IDENTIFIER, COLON)
559                                .createNode(EShallowEntityType.META, SubTypeNames.LABEL, -2).endNode();
560
561                // Options
562                inState(TOP_LEVEL).sequence(OPTION).createNode(EShallowEntityType.META, 0, 1).skipForward(1)
563                                // Binary, Text and Off are not scanned as separate keywords.
564                                // They are unreserved keywords:
565                                // https://msdn.microsoft.com/en-us/library/dd409611.aspx
566                                .optional(EnumSet.of(ON, IDENTIFIER))
567                                // People might write all options in one line.
568                                .optional(COLON).endNode();
569
570                // Form meta data encapsulated by BEGIN and END
571                inState(TOP_LEVEL).sequence(BEGIN).createNode(EShallowEntityType.META, SubTypeNames.FORM)
572                                .skipToWithNesting(END, BEGIN, END).endNode();
573        }
574
575        /**
576         * Skip to the end of a statement and return the modified recognizer.
577         */
578        private RecognizerBase<EVisualBasicParserState> skipToEndOfStatement(RecognizerBase<EVisualBasicParserState> base) {
579                // Statements in VB.NET end with the first EOL that is encountered
580                // except the compiler can perform Implicit Line Conversion:
581                // https://msdn.microsoft.com/en-us/library/865x40k4.aspx
582
583                // Save the reference to the skipBeforeWithNesting recognizer, as
584                // this gives the same result as using the reference to the empty
585                // recognizer.
586                RecognizerBase<EVisualBasicParserState> statementContinuationRecognizer = emptyRecognizer()
587                                .skipBeforeWithNesting(STATEMENT_CONTINUATION_CANDIDATES, OPENING_TYPES, CLOSING_TYPES);
588                statementContinuationRecognizer.sequence(STATEMENT_CONTINUATION_FOLLOWING, EOL);
589                statementContinuationRecognizer.sequence(STATEMENT_CONTINUATION_BOTH);
590                statementContinuationRecognizer.sequence(EOL).sequenceBefore(STATEMENT_CONTINUATION_PRECEDING);
591
592                return base.repeatedSubRecognizer(statementContinuationRecognizer)
593                                .skipBeforeWithNesting(STATEMENT_INTERRUPT_TYPES, OPENING_TYPES, CLOSING_TYPES);
594        }
