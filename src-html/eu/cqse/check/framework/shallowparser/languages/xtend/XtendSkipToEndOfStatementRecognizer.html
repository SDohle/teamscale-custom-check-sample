017package eu.cqse.check.framework.shallowparser.languages.xtend;
018
019import static eu.cqse.check.framework.scanner.ETokenType.IF;
020import static eu.cqse.check.framework.scanner.ETokenType.LBRACE;
021import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
022import static eu.cqse.check.framework.scanner.ETokenType.RBRACE;
023import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
024import static eu.cqse.check.framework.scanner.ETokenType.SEMICOLON;
025import static eu.cqse.check.framework.scanner.ETokenType.SWITCH;
026import static eu.cqse.check.framework.scanner.ETokenType.TRY;
027
028import java.util.EnumMap;
029import java.util.EnumSet;
030import java.util.List;
031import java.util.Map;
032import java.util.Stack;
033
034import eu.cqse.check.framework.scanner.ETokenType;
035import eu.cqse.check.framework.scanner.ETokenType.ETokenClass;
036import eu.cqse.check.framework.scanner.IToken;
037import eu.cqse.check.framework.shallowparser.framework.ParserState;
038import eu.cqse.check.framework.shallowparser.framework.RecognizerBase;
039import eu.cqse.check.framework.shallowparser.languages.xtend.XtendShallowParser.EXtendShallowParserState;
040
041/**
042 * We have to use this recognizer to match single statements in xtend, because
043 * there is no indicator like semicolon or end of line, that marks the end of a
044 * statement. Additionally xtend allows multiple statements in one single line.
045 * <br>
046 * 
047 * The recognizer recognizes specific constructs within a simple statement,
048 * namely: try, switch and if statements.<br>
049 * 
050 * @author $Author: baumeister $
051 * @version $Rev: $
052 * @ConQAT.Rating GREEN Hash: 9ADD58A74AA74955897F127CA425FADC
053 */
054public class XtendSkipToEndOfStatementRecognizer
055                extends RecognizerBase<EXtendShallowParserState> {
056
057        /** Contains the token classes LITERAL and IDENTIFIER. */
058        private final static EnumSet<ETokenClass> LITERAL_OR_IDENTIFIER = EnumSet
059                        .of(ETokenClass.LITERAL, ETokenClass.IDENTIFIER);
060
061        /** Contains the token types VAL and VAR. */
062        private final static EnumSet<ETokenType> VAL_OR_VAR = EnumSet
063                        .of(ETokenType.VAL, ETokenType.VAR);
064
065        /** Contains closing parenthesis token types. */
066        private static final EnumSet<ETokenType> CLOSING_PARENTHESIS = EnumSet
067                        .of(ETokenType.RPAREN, ETokenType.RBRACK, ETokenType.RBRACE);
068
069        /** Contains token types THROW and NEW. */
070        private static final EnumSet<ETokenType> THROW_OR_NEW = EnumSet
071                        .of(ETokenType.THROW, ETokenType.NEW);
072
073        /** Contains the token classes IDENTIFIER and KEYWORD. */
074        private final static EnumSet<ETokenClass> IDENTIFIER_OR_KEYWORD = EnumSet
075                        .of(ETokenClass.IDENTIFIER, ETokenClass.KEYWORD);
076
077        /**
078         * Contains token types that are allowed to be used in front of a keyword.
079         * Those are EQ, AS and DOT.
080         */
081        private static final EnumSet<ETokenType> ALLOWED_IN_FRONT_OF_KEYWORD = EnumSet
082                        .of(ETokenType.EQ, ETokenType.AS_OPERATOR, ETokenType.DOT);
083
084        /** Matched tokens for nesting in complex Xtend statements. */
085        private final static Map<ETokenType, ETokenType> XTEND_NESTING_MATCH = new EnumMap<ETokenType, ETokenType>(
086                        ETokenType.class);
087
088        static {
089                XTEND_NESTING_MATCH.put(LPAREN, RPAREN);
090                XTEND_NESTING_MATCH.put(LBRACE, RBRACE);
091        }
092
093        /** {@inheritDoc} */
094        @Override
095        protected int matchesLocally(ParserState<EXtendShallowParserState> parserState,
096                        List<IToken> tokens, int offset) {
097                if (offset <= 0) {
098                        return NO_MATCH;
099                }
100
101                IToken lastToken = tokens.get(offset - 1);
102                Stack<ETokenType> expectedClosing = new Stack<ETokenType>();
103
104                while (true) {
105                        if (offset >= tokens.size()) {
106                                return offset;
107                        }
108
109                        IToken token = tokens.get(offset);
110                        ETokenType tokenType = token.getType();
111
112                        if (!expectedClosing.isEmpty()
113                                        && tokenType == expectedClosing.peek()) {
114                                expectedClosing.pop();
115                        } else if (expectedClosing.isEmpty() && tokenType == SEMICOLON) {
116                                return offset + 1;
117                        } else if (expectedClosing.isEmpty()
118                                        && startsNewStatement(token, lastToken)) {
119                                return offset;
120                        } else if (XTEND_NESTING_MATCH.containsKey(tokenType)) {
121                                expectedClosing.push(XTEND_NESTING_MATCH.get(tokenType));
122                        } else if (tokenType == ETokenType.LBRACK
123                                        && lastToken.getType() == ETokenType.HASH_OPERATOR) {
124                                expectedClosing.push(ETokenType.RBRACK);
125                        } else {
126                                int next = startSubParse(parserState, tokens, offset,
127                                                tokenType);
128                                if (next == NO_MATCH) {
129                                        return NO_MATCH;
130                                }
131                                if (next != offset) {
132                                        offset = next;
133                                        lastToken = tokens.get(offset - 1);
134                                        continue;
135                                }
136                        }
137
138                        lastToken = token;
139                        offset += 1;
140                }
141        }
142
143        /**
144         * Checks if a new statement is about to start. This means, that
145         * <code>lastToken</code> is part of the statement and <code>token</code> is
146         * not. <br>
147         * Defaults to false.
148         */
149        private static boolean startsNewStatement(IToken token, IToken lastToken) {
150                ETokenType tokenType = token.getType();
151                ETokenType lastTokenType = lastToken.getType();
152
153                ETokenClass tokenClass = tokenType.getTokenClass();
154                ETokenClass lastTokenClass = lastTokenType.getTokenClass();
155
156                if (VAL_OR_VAR.contains(tokenType)) {
157                        return true;
158                }
159
160                if (VAL_OR_VAR.contains(lastTokenType)) {
161                        return false;
162                }
163
164                if (LITERAL_OR_IDENTIFIER.contains(tokenClass)
165                                && LITERAL_OR_IDENTIFIER.contains(lastTokenClass)) {
166                        return true;
167                }
168
169                if (CLOSING_PARENTHESIS.contains(lastTokenType)
170                                && LITERAL_OR_IDENTIFIER.contains(tokenClass)) {
171                        return true;
172                }
173
174                if (CLOSING_PARENTHESIS.contains(tokenType)) {
175                        return true;
176                }
177
178                if (LITERAL_OR_IDENTIFIER.contains(lastTokenClass)
179                                && tokenType == ETokenType.LBRACE) {
180                        return true;
181                }
182
183                // Treat return explicitly, to allow cases like 'return null'
184                if (lastTokenType == ETokenType.RETURN
185                                && (LITERAL_OR_IDENTIFIER.contains(tokenClass)
186                                                || tokenClass == ETokenClass.KEYWORD)) {
187                        return false;
188                }
189
190                if (THROW_OR_NEW.contains(lastTokenType)
191                                && (IDENTIFIER_OR_KEYWORD.contains(tokenClass))) {
192                        return false;
193                }
194
195                if (lastTokenClass == ETokenClass.KEYWORD
196                                && LITERAL_OR_IDENTIFIER.contains(tokenClass)) {
197                        return true;
198                }
199
200                // RETURN, THROW, and NEW have already been handled.
201                return !ALLOWED_IN_FRONT_OF_KEYWORD.contains(lastTokenType)
202                                && lastTokenClass != ETokenClass.OPERATOR
203                                && tokenClass == ETokenClass.KEYWORD;
204        }
205
206        /**
207         * Starts a subparser, if tokenType equals TRY, SWITCH or IF. Also starts to
208         * subparse in case of a lambda expression.
209         * 
210         * @return new startOffset
211         */
212        private static int startSubParse(ParserState<EXtendShallowParserState> parserState,
213                        List<IToken> tokens, int startOffset, ETokenType tokenType) {
214                if (tokenType == TRY || tokenType == SWITCH || tokenType == IF) {
215                        return parserState.parse(EXtendShallowParserState.IN_METHOD, tokens,
216                                        startOffset);
217                } else if (tokenType == ETokenType.LBRACK
218                                && tokens.get(startOffset - 1)
219                                                .getType() != ETokenType.HASH_OPERATOR
220                                && tokens.get(startOffset + 1).getType() != ETokenType.RBRACK) {
221                        return parserState.parse(EXtendShallowParserState.IN_LAMBDA, tokens,
222                                        startOffset);
223                }
224                return startOffset;
225        }
226}
