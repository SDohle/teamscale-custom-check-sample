021import static eu.cqse.check.framework.shallowparser.util.EntitySelectionExpressionParsingException.EParsingExceptionMessage.MISPLACED_CLOSING_PARENTHESIS;
022import static eu.cqse.check.framework.shallowparser.util.EntitySelectionExpressionParsingException.EParsingExceptionMessage.MISSING_CLOSING_PARENTHESIS;
023import static eu.cqse.check.framework.shallowparser.util.EntitySelectionExpressionParsingException.EParsingExceptionMessage.PARAMETER_MISSING;
024import static eu.cqse.check.framework.shallowparser.util.EntitySelectionExpressionParsingException.EParsingExceptionMessage.PREDICATE_CONSTRUCTION_FAILED;
025import static eu.cqse.check.framework.shallowparser.util.EntitySelectionExpressionParsingException.EParsingExceptionMessage.PREDICATE_NOT_FOUND;

027import static eu.cqse.check.framework.shallowparser.util.EntitySelectionExpressionParsingException.EParsingExceptionMessage.UNSUPPORTED_PARAMETER;

030import java.lang.reflect.Method;

032import java.util.HashMap;

034import java.util.function.Predicate;

036import org.conqat.lib.commons.assertion.CCSMAssert;
037import org.conqat.lib.commons.string.StringUtils;

039import eu.cqse.check.framework.shallowparser.ShallowParserException;

041import eu.cqse.check.framework.shallowparser.util.EntitySelectionExpressionParsingException.EParsingExceptionMessage;

043/**
044 * Parses expressions for selecting {@link ShallowEntity}s from a parse tree.
045 * The expressions may be formed by '&', '|', '!', parentheses, and all
046 * predicates defined in {@link EntitySelectionPredicates} (i.e. you can use all
047 * public methods defined there without respect of case). Predicates may contain
048 * additional underscores and dashes to improve readability, so 'simpleGetter',
049 * 'simplegetter', 'simple_getter' and 'simple-getter' would all be treated the
050 * same.
051 * 
052 * An example for an expression that matches all public attributes and methods,
053 * but no simple setters/getter and no methods annotated with '@Override' would
054 * be
055 * 
056 * <pre>
057 *     public & (attribute | method) & !(simple-getter | simple-setter | annotated(override))
058 * </pre>
059 * 
060 * @author $Author: kinnen $
061 * @version $Rev: 54460 $
062 * @ConQAT.Rating GREEN Hash: 8F3B8D841A8CFBE0AE3FCED751912E8B
063 */
064public class EntitySelectionExpressionParser {
065
066        /** The available factory methods by normalized name. */
067        private static Map<String, Method> factoryMethods;
068
069        /** The expression to parse. */
070        private final String expression;
071
072        /** The current position into {@link #expression} (during parsing). */
073        private int position = 0;
074
075        /** Constructor. */
076        private EntitySelectionExpressionParser(String expression) {
077                if (factoryMethods == null) {
078                        factoryMethods = loadFactoryMethods();
079                }
080
081                this.expression = expression;
082        }
083
084        /**
085         * Returns a map of factory methods extracted from
086         * {@link EntitySelectionPredicates}.
087         */
088        private static Map<String, Method> loadFactoryMethods() {
089                Map<String, Method> methods = new HashMap<String, Method>();
090                for (Method method : EntitySelectionPredicates.class
091                                .getDeclaredMethods()) {
092                        if (!Modifier.isPublic(method.getModifiers())
093                                        || !Modifier.isStatic(method.getModifiers())) {
094                                continue;
095                        }
096
097                        assertParameterTypes(method);
098
099                        methods.put(normalizeName(method.getName()), method);
100                }
101                return methods;
102        }
103
104        /**
105         * Raises an assertion error if the method has more then one parameter or
106         * its single parameter is not a string.
107         */
108        private static void assertParameterTypes(Method method) {
109                Class<?>[] parameterTypes = method.getParameterTypes();
110                boolean noParameters = parameterTypes.length == 0;
111                boolean oneStringParameter = parameterTypes.length == 1
112                                && parameterTypes[0] == String.class;
113                CCSMAssert.isTrue(noParameters || oneStringParameter,
114                                "Factory methods in "
115                                                + EntitySelectionPredicates.class.getSimpleName()
116                                                + " must have no parameters or one string parameter.");
117        }
118
119        /** Parses the given expression and returns a corresponding predicate. */
120        public static Predicate<ShallowEntity> parse(String expression)
121                        throws ShallowParserException {
122                return new EntitySelectionExpressionParser(expression).parse();
123        }
124
125        /** Parses the {@link #expression}. */
126        private Predicate<ShallowEntity> parse() throws ShallowParserException {
127                Predicate<ShallowEntity> predicate = parse(false, true);
128                if (position < expression.length()) {
129                        error(MISPLACED_CLOSING_PARENTHESIS);
130                }
131                return predicate;
132        }
133
134        /**
135         * Parses the {@link #expression}.
136         * 
137         * @param expectClosingParenthesis
138         *            if this is true, the parse was started from an opening
139         *            parenthesis and a closing parenthesis would be expected at the
140         *            end of the local expression.
141         * @param mayParseBinary
142         *            if this is true, parse not only a single term but potentially
143         *            continue with parsing binary operators. This is used to
144         *            implement a very simple operator precedence for '!'.
145         */
146        private Predicate<ShallowEntity> parse(boolean expectClosingParenthesis,
147                        boolean mayParseBinary) throws ShallowParserException {
148
149                Predicate<ShallowEntity> result = null;
150                while (position < expression.length()) {
151                        char next = expression.charAt(position++);
152
153                        if (Character.isWhitespace(next)) {
154                                continue;
155                        }
156
157                        if (isIdentifierCharacter(next)) {
158                                assertNoResult(result);
159                                position -= 1;
160                                result = parsePrimitiveExpression();
161                                if (!mayParseBinary) {
162                                        return result;
163                                }
164                                continue;
165                        }
166
167                        switch (next) {
168                        case '(':
169                                assertNoResult(result);
170                                result = parse(true, true);
171                                break;
172
173                        case ')':
174                                if (!expectClosingParenthesis) {
175                                        position -= 1; // leave for one of the outer calls
176                                }
177                                return assertResult(result);
178
179                        case '&':
180                                if (!mayParseBinary) {
181                                        return assertResult(result);
182                                }
183                                result = assertResult(result).and(parse(false, true));
184                                break;
185
186                        case '|':
187                                if (!mayParseBinary) {
188                                        return assertResult(result);
189                                }
190                                result = assertResult(result).or(parse(false, true));
191                                break;
192
193                        case '!':
194                                assertNoResult(result);
195                                result = parse(false, false).negate();
196                                break;
197
198                        default:
199                                error(UNEXPECTED_CHARACTER);
200                        }
201                }
202
203                if (expectClosingParenthesis) {
204                        error(MISSING_CLOSING_PARENTHESIS);
205                }
206                return assertResult(result);
207        }
208
209        /** Parses a primitive expression. */
210        private Predicate<ShallowEntity> parsePrimitiveExpression()
211                        throws ShallowParserException {
212                StringBuilder nameBuilder = new StringBuilder();
213                while (position < expression.length()
214                                && isIdentifierCharacter(expression.charAt(position))) {
215                        nameBuilder.append(expression.charAt(position++));
216                }
217
218                return createPredicate(normalizeName(nameBuilder.toString()),
219                                extractParameter());
220        }
221
222        /**
223         * Extracts a predicate parameter or returns null if no parameter is
224         * provided.
225         */
226        private String extractParameter()
227                        throws EntitySelectionExpressionParsingException {
228                // skip whitespace
229                while (position < expression.length()
230                                && Character.isWhitespace(expression.charAt(position))) {
231                        position += 1;
232                }
233
234                if (position >= expression.length()
235                                || expression.charAt(position) != '(') {
236                        return null;
237                }
238
239                StringBuilder parameterBuilder = new StringBuilder();
240                position += 1;
241                while (position < expression.length()
242                                && expression.charAt(position) != ')') {
243                        char next = expression.charAt(position);
244                        parameterBuilder.append(next);
245                        position += 1;
246                }
247
248                if (position >= expression.length()) {
249                        error(MISSING_CLOSING_PARENTHESIS);
250                }
251                position += 1;
252
253                if (parameterBuilder.length() == 0) {
254                        return null;
255                }
256
257                return parameterBuilder.toString().replaceAll("['\"]",
258                                StringUtils.EMPTY_STRING);
259        }
260
261        /**
262         * Normalizes the predicate name. This removes the prefix "select", makes
263         * the text lowercase, and discards all underscores and dashes.
264         */
265        private static String normalizeName(String name) {
266                name = name.toLowerCase();
267                name = name.replaceAll("[-_]", StringUtils.EMPTY_STRING);
268                name = StringUtils.stripPrefix(name, "select");
269                return name;
270        }
271
272        /**
273         * Returns whether the given character is expected to occur in identifiers,
274         * i.e. is alphanumeric, an underscore, or a dash.
275         */
276        private static boolean isIdentifierCharacter(char character) {
277                return Character.isJavaIdentifierPart(character) || character == '-';
278        }
279
280        /**
281         * Creates a predicate of given (normalized) name. This never returns null.
282         * 
283         * @param parameter
284         *            the parameter to the predicate. May be null to indicate a
285         *            parameterless predicate.
286         */
287        @SuppressWarnings("unchecked")
288        private Predicate<ShallowEntity> createPredicate(String name,
289                        String parameter) throws ShallowParserException {
290                Method method = factoryMethods.get(name);
291
292                try {
293                        if (method == null) {
294                                error(PREDICATE_NOT_FOUND);
295                        } else if (method.getParameterTypes().length == 0) {
296                                if (parameter != null) {
297                                        error(UNSUPPORTED_PARAMETER);
298                                }
299                                return (Predicate<ShallowEntity>) method.invoke(null);
300                        } else {
301                                if (parameter == null) {
302                                        error(PARAMETER_MISSING);
303                                }
304                                return (Predicate<ShallowEntity>) method.invoke(null,
305                                                parameter);
306                        }
307                } catch (IllegalAccessException e) {
308                        error(PREDICATE_CONSTRUCTION_FAILED, e);
309                } catch (InvocationTargetException e) {
310                        if (e.getCause() instanceof ShallowParserException) {
311                                throw (ShallowParserException) e.getCause();
312                        }
313                        error(PREDICATE_CONSTRUCTION_FAILED, e.getCause());
314                }
315                throw new AssertionError("This line should not be reachable!");
316        }
317
318        /**
319         * Asserts that the result is still null, i.e. there has been no previous
320         * expression at this level. The error message hence reports a missing
321         * binary operator.
322         */
323        private void assertNoResult(Predicate<ShallowEntity> result)
324                        throws EntitySelectionExpressionParsingException {
325                if (result != null) {
326                        error(EXPECTED_BINARY_OPERATOR);
327                }
328        }
329
330        /**
331         * Asserts that the result is not null, i.e. there has been a previous
332         * expression at this level. The error message hence reports a missing
333         * previous expression. Returns the parameter for convenience.
334         */
335        private Predicate<ShallowEntity> assertResult(
336                        Predicate<ShallowEntity> result)
337                                        throws EntitySelectionExpressionParsingException {
338                if (result == null) {
339                        error(EXPECTED_EXPRESSION);
340                }
341                return result;
342        }
343
344        /**
345         * Throws an exception with given message and details on the current parsing
346         * position.
347         */
348        private void error(EParsingExceptionMessage messageIdentifier)
349                        throws EntitySelectionExpressionParsingException {
350                throw new EntitySelectionExpressionParsingException(messageIdentifier,
351                                expression, position);
352        }
353
354        /**
355         * Throws an exception with given message and details on the current parsing
356         * position.
357         */
358        private void error(EParsingExceptionMessage messageIdentifier,
359                        Throwable cause) throws EntitySelectionExpressionParsingException {
360                throw new EntitySelectionExpressionParsingException(messageIdentifier,
361                                expression, position, cause);
362        }
