036/**
037 * Defines valid primitive expressions that can be used in the
038 * {@link EntitySelectionExpressionParser}. Each expression is defined by a
039 * public static method of same name. Methods found here must be parameterless
040 * or accept a single string parameter and must return an {@link Predicate} over
041 * {@link ShallowEntity}. To avoid clashes with Java keywords, an optional
042 * prefix "select" may be used.
043 * 
044 * Note that the comments for the public methods describe not exactly the
045 * function, but rather the returned predicate.
046 * 
047 * @author $Author: hummelb $
048 * @version $Rev: 55575 $
049 * @ConQAT.Rating GREEN Hash: F8ADD30114F3079B9E197B7EDBBEA5F2
050 */
051public class EntitySelectionPredicates {
052
053        /** The visibility modifiers in C++. */
054        private static final EnumSet<ETokenType> CPP_VISIBILITY_MODIFIERS = EnumSet
055                        .of(ETokenType.PUBLIC, ETokenType.PRIVATE, ETokenType.PROTECTED);
056
057        /** Selects all modules/namespaces. */
058        public static Predicate<ShallowEntity> module() {
059                return typePredicate(EShallowEntityType.MODULE);
060        }
061
062        /** Selects all types/classes. */
063        public static Predicate<ShallowEntity> type() {
064                return typePredicate(EShallowEntityType.TYPE)
065                                .and(entity -> !entity.getSubtype().equals("typedef"));
066        }
067
068        /** Selects all methods/functions. */
069        public static Predicate<ShallowEntity> method() {
070                return entity -> entity.getType() == EShallowEntityType.METHOD
071                                && (entity.getParent() == null || entity.getParent()
072                                                .getType() != EShallowEntityType.ATTRIBUTE);
073        }
074
075        /** Selects all shallow entities whose subtype is a declaration. */
076        public static Predicate<ShallowEntity> declaration() {
077                return new Predicate<ShallowEntity>() {
078                        @Override
079                        public boolean test(ShallowEntity entity) {
080                                if (entity.getSubtype().contains(SubTypeNames.DECLARATION)) {
081                                        return true;
082                                }
083
084                                // also consider inline methods in classes
085                                if (ELanguage.CPP.equals(getLanguage(entity))) {
086                                        ShallowEntity parent = entity.getParent();
087                                        return parent != null
088                                                        && parent.getType() == EShallowEntityType.TYPE;
089                                }
090                                return false;
091                        }
092                };
093        }
094
095        /** Selects all shallow entities that are internal (C# keyword). */
096        public static Predicate<ShallowEntity> internal() {
097                return new Predicate<ShallowEntity>() {
098                        @Override
099                        public boolean test(ShallowEntity entity) {
100                                if (TokenStreamUtils.containsAny(entity.ownStartTokens(),
101                                                ETokenType.INTERNAL)) {
102                                        return true;
103                                }
104                                ShallowEntity parent = entity.getParent();
105                                return parent != null && TokenStreamUtils.containsAny(
106                                                parent.ownStartTokens(), ETokenType.INTERNAL);
107                        }
108                };
109        }
110
111        /** Selects all attributes. */
112        public static Predicate<ShallowEntity> attribute() {
113                return typePredicate(EShallowEntityType.ATTRIBUTE);
114        }
115
116        /**
117         * Selects all properties, defined attributes that have only method
118         * children.
119         */
120        public static Predicate<ShallowEntity> property() {
121                return new Predicate<ShallowEntity>() {
122                        @Override
123                        public boolean test(ShallowEntity entity) {
124                                if (entity.getType() != EShallowEntityType.ATTRIBUTE) {
125                                        return false;
126                                }
127
128                                for (ShallowEntity child : entity.getChildren()) {
129                                        if (child.getType() != EShallowEntityType.METHOD) {
130                                                return false;
131                                        }
132                                }
133
134                                return !entity.getChildren().isEmpty();
135                        }
136                };
137        }
138
139        /** Selects all statements. */
140        public static Predicate<ShallowEntity> statement() {
141                return typePredicate(EShallowEntityType.STATEMENT);
142        }
143
144        /** Selects all meta information (defines, annotations, etc.). */
145        public static Predicate<ShallowEntity> meta() {
146                return typePredicate(EShallowEntityType.META);
147        }
148
149        /** Creates a predicate for checking the type of an entity. */
150        private static Predicate<ShallowEntity> typePredicate(
151                        final EShallowEntityType type) {
152                return entity -> entity.getType() == type;
153        }
154
155        /** Matches all public entitys. */
156        public static Predicate<ShallowEntity> selectPublic() {
157                return modifierPredicate(ETokenType.PUBLIC);
158        }
159
160        /** Matches all protected entitys. */
161        public static Predicate<ShallowEntity> selectProtected() {
162                return modifierPredicate(ETokenType.PROTECTED);
163        }
164
165        /** Matches all entitys marked with override. */
166        public static Predicate<ShallowEntity> selectOverride() {
167                return modifierPredicate(ETokenType.OVERRIDE);
168        }
169
170        /** Matches all private entitys. */
171        public static Predicate<ShallowEntity> selectPrivate() {
172                return modifierPredicate(ETokenType.PRIVATE);
173        }
174
175        /** Matches all entitys with default visibility. */
176        public static Predicate<ShallowEntity> selectDefault() {
177                return new Predicate<ShallowEntity>() {
178                        @Override
179                        public boolean test(ShallowEntity element) {
180                                List<IToken> tokens = element.ownStartTokens();
181                                return !TokenStreamUtils.containsAny(tokens, 0, tokens.size(),
182                                                ETokenType.PUBLIC, ETokenType.PROTECTED,
183                                                ETokenType.PRIVATE);
184                        }
185                };
186        }
187
188        /** Matches all final entitys. */
189        public static Predicate<ShallowEntity> selectFinal() {
190                return modifierPredicate(ETokenType.FINAL);
191        }
192
193        /** Matches all static entitys. */
194        public static Predicate<ShallowEntity> selectStatic() {
195                return modifierPredicate(ETokenType.STATIC);
196        }
197
198        /** Returns a predicate that checks for existence of the given modifier. */
199        private static Predicate<ShallowEntity> modifierPredicate(
200                        final ETokenType modifier) {
201                return new Predicate<ShallowEntity>() {
202                        @Override
203                        public boolean test(ShallowEntity element) {
204                                if (TokenStreamUtils.find(element.ownStartTokens(),
205                                                modifier) != TokenStreamUtils.NOT_FOUND) {
206                                        return true;
207                                }
208
209                                if (ELanguage.CPP == getLanguage(element)
210                                                && CPP_VISIBILITY_MODIFIERS.contains(modifier)) {
211                                        return hasCppVisibilityModifer(element, modifier);
212                                }
213                                return false;
214                        }
215                };
216        }
217
218        /**
219         * Determines the language of the given entity or <code>null</code> if it
220         * cannot be determined
221         */
222        private static ELanguage getLanguage(ShallowEntity entity) {
223                UnmodifiableList<IToken> tokens = entity.includedTokens();
224                if (tokens.isEmpty()) {
225                        return null;
226                }
227                return CollectionUtils.getAny(tokens).getLanguage();
228        }
229
230        /**
231         * Checks whether the given shallow entity is affected by the specified C++
232         * visibility modifier by backwards searching the sibling shallow entities.
233         * The given modifier must be one of public, protected, or private.
234         */
235        private static boolean hasCppVisibilityModifer(ShallowEntity entity,
236                        ETokenType modifier) {
237                ShallowEntity parent = entity.getParent();
238                if (parent == null) {
239                        return false;
240                }
241
242                UnmodifiableList<ShallowEntity> siblings = parent.getChildren();
243                for (int i = siblings.indexOf(entity) - 1; i >= 0; i--) {
244                        ShallowEntity sibling = siblings.get(i);
245
246                        if (isCppVisibilityModifier(sibling)) {
247                                return TokenStreamUtils.containsAny(sibling.ownStartTokens(),
248                                                modifier);
249                        }
250                }
251
252                // no explicit modifier found -> assume default
253                if (parent.getType() == EShallowEntityType.TYPE) {
254                        if (SubTypeNames.STRUCT.equals(parent.getSubtype())) {
255                                return ETokenType.PUBLIC.equals(modifier);
256                        }
257                        return ETokenType.PRIVATE.equals(modifier);
258                }
259
260                return false;
261        }
262
263        /** Returns whether the given entity is a C++ visibility modifier. */
264        private static boolean isCppVisibilityModifier(ShallowEntity entity) {
265                return entity.getType() == EShallowEntityType.META && TokenStreamUtils
266                                .containsAny(entity.ownStartTokens(), CPP_VISIBILITY_MODIFIERS);
267        }
268
269        /** Matches all primitive entities (i.e. those without children). */
270        public static Predicate<ShallowEntity> primitive() {
271                return entity -> entity.getChildren().isEmpty();
272        }
273
274        /** Matches entities by their subtype. */
275        public static Predicate<ShallowEntity> subtype(final String subtype) {
276                return entity -> subtype.equals(entity.getSubtype());
277        }
278
279        /** Matches entities by their name. */
280        public static Predicate<ShallowEntity> name(final String name) {
281                return entity -> name.equals(entity.getName());
282        }
283
284        /**
285         * Matches all entities that are annotated with an annotation of given name
286         * (excluding the '@' sign).
287         */
288        public static Predicate<ShallowEntity> annotated(String annotationName) {
289                final String normalizedAnnotationName = StringUtils
290                                .stripPrefix(annotationName, "@");
291                return new Predicate<ShallowEntity>() {
292                        @Override
293                        public boolean test(ShallowEntity entity) {
294                                // annotations must be part of same parent
295                                ShallowEntity parent = entity.getParent();
296                                if (parent == null) {
297                                        return false;
298                                }
299
300                                List<ShallowEntity> children = parent.getChildren();
301                                int index = children.indexOf(entity);
302                                index -= 1;
303                                while (index >= 0 && children.get(index)
304                                                .getType() == EShallowEntityType.META) {
305                                        ShallowEntity metaEntity = children.get(index);
306                                        if ("annotation".equals(metaEntity.getSubtype())
307                                                        && normalizedAnnotationName
308                                                                        .equalsIgnoreCase(metaEntity.getName())) {
309                                                return true;
310                                        }
311                                        index -= 1;
312                                }
313                                return false;
314                        }
315                };
316        }

318        /**
319         * Matches simple getters, i.e. methods starting with "get" or "is" and with
320         * at most one contained statement.
321         */
322        public static Predicate<ShallowEntity> simpleGetter() {
323                return simpleMethod("get").or(simpleMethod("is"));
324        }
325
326        /**
327         * Matches simple setters, i.e. methods starting with "set" and with at most
328         * one contained statement.
329         */
330        public static Predicate<ShallowEntity> simpleSetter() {
331                return simpleMethod("set");
332        }
333
334        /**
335         * Matches simple methods with the given prefix. A method is simple, if it
336         * contains as most one statement.
337         */
338        public static Predicate<ShallowEntity> simpleMethod(
339                        final String namePrefix) {
340                return new Predicate<ShallowEntity>() {
341                        @Override
342                        public boolean test(ShallowEntity entity) {
343                                boolean isSimpleMethod = entity
344                                                .getType() == EShallowEntityType.METHOD
345                                                && getStatementCount(entity) <= 1;
346                                if (!isSimpleMethod) {
347                                        return false;
348                                }
349
350                                if (entity.getName() == null) {
351                                        return StringUtils.isEmpty(namePrefix);
352                                }
353                                return entity.getName().startsWith(namePrefix);
354                        }
355
356                        /**
357                         * Returns the number of statements contained as children of the
358                         * entity.
359                         */
360                        private int getStatementCount(ShallowEntity entity) {
361                                return ShallowEntityTraversalUtils
362                                                .listEntitiesOfType(entity.getChildren(),
363                                                                EShallowEntityType.STATEMENT)
364                                                .size();
365                        }
366                };
367        }
