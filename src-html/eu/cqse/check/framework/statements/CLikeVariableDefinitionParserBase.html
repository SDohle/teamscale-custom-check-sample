017package eu.cqse.check.framework.statements;
018
019import static eu.cqse.check.framework.scanner.ETokenType.COMMA;
020import static eu.cqse.check.framework.scanner.ETokenType.EQ;
021import static eu.cqse.check.framework.scanner.ETokenType.GT;
022import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
023import static eu.cqse.check.framework.scanner.ETokenType.LBRACE;
024import static eu.cqse.check.framework.scanner.ETokenType.LBRACK;
025import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
026import static eu.cqse.check.framework.scanner.ETokenType.LT;
027import static eu.cqse.check.framework.scanner.ETokenType.RBRACE;
028import static eu.cqse.check.framework.scanner.ETokenType.RBRACK;
029import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
030import static eu.cqse.check.framework.scanner.ETokenType.SEMICOLON;
031import static eu.cqse.check.framework.shallowparser.TokenStreamUtils.NOT_FOUND;
032
033import java.util.ArrayList;
034import java.util.Arrays;
035import java.util.EnumSet;
036import java.util.List;
037
038import org.conqat.lib.commons.collections.CollectionUtils;
039
040import eu.cqse.check.framework.scanner.ETokenType;
041import eu.cqse.check.framework.scanner.IToken;
042import eu.cqse.check.framework.shallowparser.SubTypeNames;
043import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
044import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
045import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
046import eu.cqse.check.framework.statements.VariableDefinition.EVariableType;
047
048/**
049 * Base class for variable definition parsers of C-like languages. Handles
050 * parsing of method parameters, fields, local variables, for and foreach
051 * statements.
052 * 
053 * @author $Author: streitel $
054 * @version $Rev: 53491 $
055 * @ConQAT.Rating YELLOW Hash: AAB2DC878D6A4F911FA71FB4F9EC59B4
056 */
057public abstract class CLikeVariableDefinitionParserBase implements
058                IVariableDefinitionParser {
059
060        /**
061         * The token types that open a new nesting level. The order of the entries
062         * corresponds to {@link #OPENING_BRACES}.
063         */
064        protected static final List<ETokenType> CLOSING_BRACES = Arrays.asList(
065                        RBRACE, RBRACK, RPAREN, GT);
066
067        /**
068         * The token types that close a nesting level. The order of the entries
069         * corresponds to {@link #CLOSING_BRACES}.
070         */
071        protected static final List<ETokenType> OPENING_BRACES = Arrays.asList(
072                        LBRACE, LBRACK, LPAREN, LT);
073
074        /**
075         * The separator in a foreach statement between the variable declaration and
076         * the collection expression.
077         */
078        private final ETokenType forEachSparator;
079
080        /** Constructor. */
081        public CLikeVariableDefinitionParserBase(ETokenType forEachSparator) {
082                this.forEachSparator = forEachSparator;
083        }
084
085        /** {@inheritDoc} */
086        @Override
087        public List<VariableDefinition> parseVariableDefinitions(
088                        ShallowEntity entity) {
089                switch (entity.getType()) {
090                case METHOD:
091                        return parseMethodParameters(entity);
092                case ATTRIBUTE:
093                        if (entity.getParent().getType() == EShallowEntityType.TYPE) {
094                                return parseFields(entity);
095                        }
096                        return CollectionUtils.emptyList();
097                case STATEMENT:
098                        switch (entity.getSubtype()) {
099                        case SubTypeNames.LOCAL_VARIABLE:
100                                return parseLocalVariable(entity);
101                        case SubTypeNames.FOR:
102                        case SubTypeNames.FOREACH:
103                                return parseForStatement(entity);
104                        case SubTypeNames.CATCH:
105                                return parseDeclarationBlock(entity, SEMICOLON);
106                        }
107                }
108
109                return CollectionUtils.emptyList();
110        }
111
112        /**
113         * Parses a block statement (i.e. one that can contain other statements)
114         * that declares one or more variables separated by the given separator
115         * token, e.g. <code>catch</code> or try-with-resources.
116         */
117        protected List<VariableDefinition> parseDeclarationBlock(
118                        ShallowEntity entity, ETokenType separatorToken) {
119                List<IToken> tokens = getTokensBetweenOutermostParentheses(entity);
120                return parseDefinitionFragment(tokens, separatorToken,
121                                EVariableType.LOCAL_VARIABLE);
122        }
123
124        /** Parses for and foreach statements. */
125        private List<VariableDefinition> parseForStatement(ShallowEntity entity) {
126                List<IToken> tokens = getTokensBetweenOutermostParentheses(entity);
127
128                ETokenType splitTokenType = SEMICOLON;
129                if (TokenStreamUtils.containsAny(tokens, forEachSparator)) {
130                        splitTokenType = forEachSparator;
131                }
132
133                List<List<IToken>> split = TokenStreamUtils.splitWithNesting(tokens,
134                                splitTokenType, OPENING_BRACES, CLOSING_BRACES);
135                List<IToken> declarationTokens = split.get(0);
136                return parseDefinitionFragment(declarationTokens, COMMA,
137                                EVariableType.LOCAL_VARIABLE);
138        }
139
140        /**
141         * Returns all tokens between the outermost pair of parentheses in the given
142         * entities start tokens.
143         */
144        protected List<IToken> getTokensBetweenOutermostParentheses(
145                        ShallowEntity entity) {
146                return TokenStreamUtils.tokensBetweenWithNesting(
147                                entity.includedTokens(), LPAREN, RPAREN);
148        }
149
150        /**
151         * Parses an excerpt of the token stream that represents a declaration of
152         * one or more variables or parameters. Each declaration is separated from
153         * the next one by a separator token, usually comma or semicolon.
154         */
155        private List<VariableDefinition> parseDefinitionFragment(
156                        List<IToken> tokens, ETokenType splitTokenType,
157                        EVariableType variableType) {
158                List<List<IToken>> split = TokenStreamUtils.splitWithNesting(tokens,
159                                splitTokenType, OPENING_BRACES, CLOSING_BRACES);
160                List<VariableDefinition> definitions = new ArrayList<>();
161                for (List<IToken> variableDeclarationTokens : split) {
162                        VariableDefinition definition = parseVariableOrField(
163                                        variableDeclarationTokens, variableType);
164                        if (definition != null) {
165                                definitions.add(definition);
166                        }
167                }
168                return definitions;
169        }
170
171        /** Parses a simple statement that declares one or more local variables. */
172        private List<VariableDefinition> parseLocalVariable(ShallowEntity entity) {
173                return parseDefinitionFragment(entity.includedTokens(), COMMA,
174                                EVariableType.LOCAL_VARIABLE);
175        }
176
177        /**
178         * Parses an entity that represents a field declaration.
179         */
180        private List<VariableDefinition> parseFields(ShallowEntity entity) {
181                return parseDefinitionFragment(entity.ownStartTokens(), COMMA,
182                                EVariableType.FIELD);
183        }
184
185        /** Parses a variable or field declaration. */
186        private VariableDefinition parseVariableOrField(List<IToken> tokens,
187                        EVariableType variableType) {
188                int equalsIndex = TokenStreamUtils.find(tokens, EQ);
189                if (equalsIndex == NOT_FOUND) {
190                        // declaration without assignment, thus the assigned variable is the
191                        // last identifier in the statement
192                        int lastIdentifierIndex = TokenStreamUtils.findLast(tokens,
193                                        getIdentifierTypes());
194                        if (lastIdentifierIndex == NOT_FOUND) {
195                                // assume the entity is broken due to code that does not compile
196                                return null;
197                        }
198
199                        return new VariableDefinition(tokens.get(lastIdentifierIndex),
200                                        tokens, variableType);
201                }
202
203                return new VariableDefinition(tokens.get(equalsIndex - 1), tokens,
204                                variableType);
205        }
206
207        /**
208         * Returns the token types that may be used for identifiers. May be
209         * overwritten by subclasses. The default just returns
210         * {@link ETokenType#IDENTIFIER}.
211         */
212        protected EnumSet<ETokenType> getIdentifierTypes() {
213                return EnumSet.of(IDENTIFIER);
214        }
215
216        /** Parses all parameters of the given method entity. */
217        private List<VariableDefinition> parseMethodParameters(ShallowEntity entity) {
218                List<IToken> tokens = getTokensBetweenOutermostParentheses(entity);
219                return parseDefinitionFragment(tokens, COMMA,
220                                EVariableType.METHOD_PARAMETER);
221        }
222
223}
