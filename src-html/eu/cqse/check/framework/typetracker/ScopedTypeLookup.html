017package eu.cqse.check.framework.typetracker;
018
019import java.util.HashMap;
020import java.util.HashSet;
021import java.util.Map;
022import java.util.Set;
023
024import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
025
026/**
027 * A scoped type lookup is an associative data structure that maps variable
028 * names to their type information. A scoped type lookup can create child
029 * lookups, that can shadow variables from the parent. Further imported
030 * namespaces are stored.
031 */
032public class ScopedTypeLookup {
033
034        /** The default package separator. */
035        private static final String DEFAULT_PACKAGE_SEPARATOR = ".";
036
037        /** Set of imported namespaces. */
038        private Set<String> importedNamespaces;
039
040        /**
041         * An internal lookup, that maps variable names to their type information.
042         */
043        private Map<String, TypedVariable> variableTypeInfos;
044
045        /** The parent lookup or null if this is the root lookup. */
046        private ScopedTypeLookup parent;
047
048        /** The package separator that is used by this lookup. */
049        private String packageSeparator;
050
051        /** Creates a new lookup with no parent. */
052        public ScopedTypeLookup() {
053                this(DEFAULT_PACKAGE_SEPARATOR);
054        }
055
056        /** Creates a new lookup with no parent and the given package separator. */
057        public ScopedTypeLookup(String packageSeparator) {
058                this(null, packageSeparator);
059        }
060
061        /**
062         * Creates a new lookup with the given parent. This constructor is only used
063         * internally. Child-creation is done with {@link #createChildScope()}.
064         */
065        private ScopedTypeLookup(ScopedTypeLookup parent) {
066                this(parent, parent.getPackageSeparator());
067        }
068
069        /**
070         * Creates a new lookup with the given parent and package separator. This
071         * constructor is only used internally. Child-creation is done with
072         * {@link #createChildScope()}.
073         */
074        private ScopedTypeLookup(ScopedTypeLookup parent, String packageSeparator) {
075                this.parent = parent;
076                this.packageSeparator = packageSeparator;
077                this.importedNamespaces = new HashSet<String>();
078                this.variableTypeInfos = new HashMap<>();
079        }
080
081        /** Returns the parent lookup or null if this is the root lookup. */
082        public ScopedTypeLookup getParent() {
083                return parent;
084        }
085
086        /** Returns the package separator. */
087        public String getPackageSeparator() {
088                return packageSeparator;
089        }
090
091        /** Creates a new lookup, with this lookup as parent. */
092        public ScopedTypeLookup createChildScope() {
093                return new ScopedTypeLookup(this);
094        }
095
096        /** Returns a set of all existing keys. */
097        public Set<String> getAllVariables() {
098                Set<String> variableSet = new HashSet<>(variableTypeInfos.keySet());
099                if (parent != null) {
100                        variableSet.addAll(parent.getAllVariables());
101                }
102                return variableSet;
103        }
104
105        /**
106         * Associates the given variable with the given type information in this
107         * lookup. Returns the old type information or null if none was set before.
108         */
109        public TypedVariable putTypeInfo(String variableName, TypedVariable typeInfo) {
110                return variableTypeInfos.put(variableName, typeInfo);
111        }
112
113        /**
114         * Returns the type information for the given key. If the key is not found
115         * in this scoped lookup, the parent lookups will be searched recursively.
116         * If the key does not exist, null is returned.
117         */
118        public TypedVariable getTypeInfo(String variableName) {
119                TypedVariable value = variableTypeInfos.get(variableName);
120                if (value == null && parent != null) {
121                        value = parent.getTypeInfo(variableName);
122                }
123                return value;
124        }
125
126        /** Adds an imported namespace to the scoped lookup. */
127        public boolean addImportedNamespace(String namespace) {
128                return importedNamespaces.add(namespace);
129        }
130
131        /** Returns a set of all imported namespaces. */
132        public Set<String> getImportedNamespaces() {
133                Set<String> namespaces = new HashSet<String>(importedNamespaces);
134                if (parent != null) {
135                        namespaces.addAll(parent.getImportedNamespaces());
136                }
137                return namespaces;
138        }
139
140        /**
141         * Returns whether this lookup contains information about the given
142         * variable.
143         */
144        public boolean containsVariable(String variableName) {
145                return getTypeInfo(variableName) != null;
146        }
147
148        /**
149         * Returns whether this lookup contains information about a variable with
150         * the given variable that was defined in the given entity.
151         */
152        public boolean containsVariableWithEntity(String variableName, ShallowEntity definitionEntity) {
153                TypedVariable typeInfo = variableTypeInfos.get(variableName);
154                if (typeInfo != null && typeInfo.getDeclaringEntity().equals(definitionEntity)) {
155                        return true;
156                }
157                if (parent != null) {
158                        return parent.containsVariableWithEntity(variableName, definitionEntity);
159                }
160                return false;
161        }
162
163        /**
164         * Returns whether the variable with the given name is shadowing the
165         * variable that was defined by the given entity.
166         */
167        public boolean isShadowingVariable(String variableName, ShallowEntity variableEntity) {
168                if (parent == null) {
169                        return false;
170                }
171                if (variableTypeInfos.containsKey(variableName)) {
172                        return parent.containsVariableWithEntity(variableName, variableEntity);
173                }
174                return parent.isShadowingVariable(variableName, variableEntity);
175        }
176
177        /**
178         * Returns whether the given variable has the type with the given full
179         * qualified type name.
180         */
181        public boolean hasFullQualifiedTypeName(String variableName, String fullQualifiedTypeName) {
182                TypedVariable info = getTypeInfo(variableName);
183                if (info == null) {
184                        throw new IllegalArgumentException("Type information for the given variableName does not exist.");
185                }
186
187                String typeName = info.getTypeName();
188                if (typeName.equals(fullQualifiedTypeName)) {
189                        return true;
190                }
191
192                Set<String> namespaces = getImportedNamespaces();
193                for (String namespace : namespaces) {
194                        String potentialFullQualifiedTypeName = namespace + packageSeparator + typeName;
195                        if (potentialFullQualifiedTypeName.equals(fullQualifiedTypeName)) {
196                                return true;
197                        }
198                }
199
200                return false;
201        }
202}
