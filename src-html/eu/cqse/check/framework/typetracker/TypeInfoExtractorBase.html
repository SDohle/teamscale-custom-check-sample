017package eu.cqse.check.framework.typetracker;
018
019import java.util.List;
020
021import org.conqat.lib.commons.assertion.CCSMAssert;
022import org.conqat.lib.commons.collections.CollectionUtils;
023import org.conqat.lib.commons.logging.ILogger;
024
025import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
026
027/**
028 * An abstract base class for type info extractors. This base class implements
029 * the {@link ITypeInfoExtractor#extract(ShallowEntity)}-method and passes the
030 * given entity to a method corresponding to the entity's type.
031 * 
032 * @author $Author: pagano $
033 * @version $Rev: 54869 $
034 * @ConQAT.Rating GREEN Hash: F432ACA3DEA49B406A88A7E986DF013B
035 */
036public abstract class TypeInfoExtractorBase implements ITypeInfoExtractor {
037
038        /** logger instance for printing warnings. */
039        protected ILogger logger;
040
041        /** {@inheritDoc} */
042        @Override
043        public List<TypedVariable> extract(ShallowEntity entity) {
044                switch (entity.getType()) {
045                case MODULE:
046                        return extractFromModule(entity);
047                case META:
048                        return extractFromMeta(entity);
049                case TYPE:
050                        return extractFromType(entity);
051                case ATTRIBUTE:
052                        return extractFromAttribute(entity);
053                case METHOD:
054                        return extractFromMethod(entity);
055                case STATEMENT:
056                        return extractFromStatement(entity);
057                default:
058                        CCSMAssert.fail("No valid entity type.");
059                        return null;
060                }
061        }
062
063        /** Extracts type information from module entities. */
064        public List<TypedVariable> extractFromModule(
065                        @SuppressWarnings("unused") ShallowEntity moduleEntity) {
066                return CollectionUtils.emptyList();
067        }
068
069        /** Extracts type information from meta entities. */
070        public List<TypedVariable> extractFromMeta(
071                        @SuppressWarnings("unused") ShallowEntity metaEntity) {
072                return CollectionUtils.emptyList();
073        }
074
075        /** Extracts type information from type entities. */
076        public List<TypedVariable> extractFromType(
077                        @SuppressWarnings("unused") ShallowEntity typeEntity) {
078                return CollectionUtils.emptyList();
079        }
080
081        /** Extracts type information from attribute entities. */
082        public List<TypedVariable> extractFromAttribute(
083                        @SuppressWarnings("unused") ShallowEntity attributeEntity) {
084                return CollectionUtils.emptyList();
085        }
086
087        /** Extracts type information from method entities. */
088        public List<TypedVariable> extractFromMethod(
089                        @SuppressWarnings("unused") ShallowEntity methodEntity) {
090                return CollectionUtils.emptyList();
091        }
092
093        /** Extracts type information from statement entities. */
094        public List<TypedVariable> extractFromStatement(
095                        @SuppressWarnings("unused") ShallowEntity statementEntity) {
096                return CollectionUtils.emptyList();
097        }
098
099        /** {@inheritDoc} */
100        @Override
101        public void setLogger(ILogger logger) {
102                this.logger = logger;
103        }
104}
