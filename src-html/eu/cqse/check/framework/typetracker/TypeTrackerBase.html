017package eu.cqse.check.framework.typetracker;
018
019import java.util.EnumSet;
020import java.util.HashMap;
021import java.util.List;
022import java.util.Map;
023import java.util.Set;
024
025import org.conqat.lib.commons.assertion.CCSMAssert;
026import org.conqat.lib.commons.logging.ILogger;
027
028import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
029import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
030
031/**
032 * Abstract base class for type trackers.
033 * 
034 * @author $Author: goeb $
035 * @version $Rev: 55881 $
036 * @ConQAT.Rating GREEN Hash: 388CF33BEEE7A0CEE93957F5A6044DE3
037 */
038public abstract class TypeTrackerBase<T extends ITypeInfoExtractor>
039                implements ITypeTracker {
040
041        /**
042         * A set of all entity types that provide an ordered introduction of type
043         * information.
044         */
045        private static final EnumSet<EShallowEntityType> ORDERED_ENTITY_TYPES = EnumSet
046                        .of(EShallowEntityType.METHOD, EShallowEntityType.STATEMENT);
047
048        /** Mapping from shallow entities to type lookup maps. */
049        private final Map<ShallowEntity, ScopedTypeLookup> typeLookups = new HashMap<>();
050
051        /** The internally used type info extractor. */
052        protected final T typeInfoExtractor;
053
054        /** Constructor. */
055        protected TypeTrackerBase(T typeInfoExtractor) {
056                CCSMAssert.isNotNull(typeInfoExtractor);
057                this.typeInfoExtractor = typeInfoExtractor;
058        }
059
060        /** {@inheritDoc} */
061        @Override
062        public void setLogger(ILogger logger) {
063                typeInfoExtractor.setLogger(logger);
064        }
065
066        /** {@inheritDoc} */
067        @Override
068        public ITypeResolution createTypeResolution(
069                        List<ShallowEntity> rootEntities) {
070                ScopedTypeLookup variableLookup = new ScopedTypeLookup();
071                return createTypeResolution(rootEntities, variableLookup);
072        }
073
074        /** {@inheritDoc} */
075        @Override
076        public ITypeResolution createTypeResolution(List<ShallowEntity> entities,
077                        ScopedTypeLookup variableLookup) {
078                addDefaultImports(variableLookup);
079                createTypeInformation(entities, variableLookup, false);
080                return new PrecomputedTypeResolution(
081                                new HashMap<ShallowEntity, ScopedTypeLookup>(typeLookups));
082        }
083
084        /**
085         * Recursive method that is called to traverse all given entities. The
086         * parent tracked entity can be null, if the given entities are top-level
087         * entities. The given variable lookup contains type information for all
088         * known variables. Ordered specifies whether an ordered scope should be
089         * used.
090         */
091        private void createTypeInformation(List<ShallowEntity> entities,
092                        ScopedTypeLookup variableLookup, boolean ordered) {
093                for (ShallowEntity entity : entities) {
094                        processImport(entity, variableLookup);
095
096                        List<TypedVariable> typeInfo = typeInfoExtractor.extract(entity);
097                        boolean newTypeInfo = !typeInfo.isEmpty();
098
099                        if (ordered && newTypeInfo) {
100                                variableLookup = variableLookup.createChildScope();
101                        }
102
103                        ScopedTypeLookup tmpVariableLookup = variableLookup;
104                        if (newTypeInfo) {
105                                if (needsSeparateScope(entity)) {
106                                        tmpVariableLookup = tmpVariableLookup.createChildScope();
107                                }
108                                for (TypedVariable info : typeInfo) {
109                                        tmpVariableLookup.putTypeInfo(info.getVariableName(), info);
110                                }
111                        }
112
113                        typeLookups.put(entity, tmpVariableLookup);
114
115                        List<ShallowEntity> children = entity.getChildren();
116                        if (!children.isEmpty()) {
117                                createTypeInformation(children,
118                                                tmpVariableLookup.createChildScope(),
119                                                needsOrderedScope(entity));
120                        }
121                }
122        }
123
124        /** Adds default namespaces to the initial variable lookup. */
125        protected void addDefaultImports(
126                        @SuppressWarnings("unused") ScopedTypeLookup variableLookup) {
127                // do nothing by default
128        }
129
130        /**
131         * Checks whether the given entity is an import of a namespace and adds the
132         * namespace to the variable lookup.
133         */
134        protected void processImport(
135                        @SuppressWarnings("unused") ShallowEntity entity,
136                        @SuppressWarnings("unused") ScopedTypeLookup variableLookup) {
137                // do nothing by default
138        }
139
140        /**
141         * Returns whether the given entity needs an ordered scope. This is used e.
142         * g. for methods or statements, where variables are only accessible if they
143         * have been introduced in an entity earlier in the code. Other entities
144         * like types don't need an ordered scope, because introduced attributes are
145         * visible everywhere within the class.
146         */
147        // TODO (AG) It is unclear what "previous" means here. Earlier in the code?
148        // Or upwards in the entity hierarchy? I think the latter, since I can
149        // easily reference methods that are defined below.
150        // (F3): I think it means earlier in the code. So in methods and statements
151        // variables have to be declared before usage
152        // TODO (AG) In ABAP you can even access local variables before you define
153        // them. You should never do that, but you can. So maybe we need this to be
154        // language-dependent similar to needsSeparateScope?
155        // (F3): OMG. Didn't know that. I have overridden it in ABAPTypeTracker.
156        protected boolean needsOrderedScope(ShallowEntity entity) {
157                return ORDERED_ENTITY_TYPES.contains(entity.getType());
158        }
159
160        /**
161         * Returns whether this entity needs a separate variable scope. This is the
162         * case for methods and some statements that are determined by
163         * {@link #getSeparateScopeSubtypes()}, as otherwise those would introduce
164         * type information into the wrong scope.
165         */
166        private boolean needsSeparateScope(ShallowEntity entity) {
167                if (entity.getType().equals(EShallowEntityType.METHOD)
168                                || getSeparateScopeSubtypes().contains(entity.getSubtype())) {
169                        return true;
170                }
171                return false;
172        }
173
174        /** Returns all statement sub-types that need an separate scope. */
175        protected abstract Set<String> getSeparateScopeSubtypes();
176
177}
