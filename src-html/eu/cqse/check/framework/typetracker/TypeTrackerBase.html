017package eu.cqse.check.framework.typetracker;
018
019import java.util.EnumSet;
020import java.util.HashMap;
021import java.util.List;
022import java.util.Map;
023import java.util.Set;
024
025import org.conqat.lib.commons.logging.ILogger;
026
027import eu.cqse.check.framework.scanner.ELanguage;
028import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
029import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
030
031/**
032 * Abstract base class for type trackers.
033 * <p>
034 * While this base class itself does not keep any state and is thread-safe,
035 * implementing classes may introduce such behavior.
036 *
037 * @see ITypeTracker
038 */
039public abstract class TypeTrackerBase implements ITypeTracker {
040
041        /**
042         * A set of all entity types that provide an ordered introduction of type
043         * information, i.e. entities inside which variables are only visible after
044         * they have been declared. This is typically the case inside methods and
045         * also inside control-flow blocks.
046         *
047         * @see #needsOrderedScope(ShallowEntity)
048         */
049        private static final EnumSet<EShallowEntityType> ORDERED_ENTITY_TYPES = EnumSet.of(EShallowEntityType.METHOD,
050                        EShallowEntityType.STATEMENT);
051
052        /** Language of the type tracker. */
053        private final ELanguage language;
054
055        /** Constructor. */
056        public TypeTrackerBase(ELanguage language) {
057                this.language = language;
058        }
059
060        /** {@inheritDoc} */
061        @Override
062        public ITypeResolution createTypeResolution(List<ShallowEntity> rootEntities, ILogger logger) {
063                ScopedTypeLookup variableLookup = new ScopedTypeLookup();
064                return createTypeResolution(rootEntities, logger, variableLookup);
065        }
066
067        /** {@inheritDoc} */
068        @Override
069        public ITypeResolution createTypeResolution(List<ShallowEntity> rootEntities, ILogger logger,
070                        ScopedTypeLookup variableLookup) {
071
072                ITypeInfoExtractor typeInfoExtractor = TypeInfoExtractorFactory.getTypeInfoExtractor(language, rootEntities,
073                                logger);
074
075                addDefaultImports(variableLookup);
076
077                return new PrecomputedTypeResolution(
078                                createTypeInformation(rootEntities, variableLookup, false, typeInfoExtractor));
079        }
080
081        /**
082         * Recursively creates type information for the given root entities.
083         *
084         * @param rootEntities
085         *            The root entities to create type information for.
086         * @param variableLookup
087         *            Contains type information for all known variables.
088         * @param ordered
089         *            Specifies whether an ordered scope should be used.
090         * @param typeInfoExtractor
091         *            The extractor to create type information.
092         */
093        private Map<ShallowEntity, ScopedTypeLookup> createTypeInformation(List<ShallowEntity> rootEntities,
094                        ScopedTypeLookup variableLookup, boolean ordered, ITypeInfoExtractor typeInfoExtractor) {
095                Map<ShallowEntity, ScopedTypeLookup> typeLookups = new HashMap<>();
096                createTypeInformation(rootEntities, variableLookup, ordered, typeLookups, typeInfoExtractor);
097                return typeLookups;
098        }
099
100        /**
101         * Recursive method that is called to traverse all given entities. Scopes
102         * are created as needed. Child entities always have a new child scope.
103         * Sibling entities by default share the same scope (e.g. members of a
104         * class) unless {@link #needsSeparateScope(ShallowEntity)} is {@code true}
105         * or the siblings are ordered. In these cases, a new scope is created
106         * whenever a new variable is declared.
107         *
108         * @param entities
109         *            The entities to extract information from. May be a file's root
110         *            entities or the children of any entity down the AST.
111         * @param variableLookup
112         *            Contains type information for all known variables.
113         * @param ordered
114         *            Specifies whether an ordered scope should be used.
115         * @param typeLookups
116         *            Mapping from shallow entities to type lookup maps.
117         * @param typeInfoExtractor
118         *            The extractor to create type information.
119         */
120        private void createTypeInformation(List<ShallowEntity> entities, ScopedTypeLookup variableLookup, boolean ordered,
121                        Map<ShallowEntity, ScopedTypeLookup> typeLookups, ITypeInfoExtractor typeInfoExtractor) {
122                for (ShallowEntity entity : entities) {
123                        processImport(entity, variableLookup);
124
125                        List<TypedVariable> typeInfo = typeInfoExtractor.extract(entity);
126                        boolean newTypeInfo = !typeInfo.isEmpty();
127
128                        if (ordered && newTypeInfo) {
129                                variableLookup = variableLookup.createChildScope();
130                        }
131
132                        ScopedTypeLookup currentLookup = variableLookup;
133                        if (newTypeInfo) {
134                                if (needsSeparateScope(entity)) {
135                                        currentLookup = currentLookup.createChildScope();
136                                }
137                                for (TypedVariable info : typeInfo) {
138                                        currentLookup.putTypeInfo(info.getVariableName(), info);
139                                }
140                        }
141                        typeLookups.put(entity, currentLookup);
142
143                        List<ShallowEntity> children = entity.getChildren();
144                        if (!children.isEmpty()) {
145                                createTypeInformation(children, currentLookup.createChildScope(), needsOrderedScope(entity),
146                                                typeLookups, typeInfoExtractor);
147                        }
148                }
149        }
150
151        /**
152         * Adds default namespaces to the initial variable lookup.
153         *
154         * This is intended to be overwritten by subclasses. The default
155         * implementation does nothing.
156         */
157        protected void addDefaultImports(@SuppressWarnings("unused") ScopedTypeLookup variableLookup) {
158                // do nothing by default
159        }
160
161        /**
162         * Checks whether the given entity is an import of a namespace and adds the
163         * namespace to the variable lookup.
164         *
165         * This is intended to be overwritten by subclasses. The default
166         * implementation does nothing.
167         */
168        protected void processImport(@SuppressWarnings("unused") ShallowEntity entity,
169                        @SuppressWarnings("unused") ScopedTypeLookup variableLookup) {
170                // do nothing by default
171        }
172
173        /**
174         * Returns whether the given entity needs an ordered scope. This is used
175         * e.g. for methods or statements, where variables are only accessible if
176         * they have been introduced in an entity earlier in the code. Other
177         * entities like types don't need an ordered scope, because introduced
178         * attributes are visible everywhere within the class.
179         *
180         * The default implementation returns {@code true} if the given entity's
181         * type is contained in {@link #ORDERED_ENTITY_TYPES}. Subclasses may
182         * override this in order to provide language-specific behavior.
183         */
184        protected boolean needsOrderedScope(ShallowEntity entity) {
185                return ORDERED_ENTITY_TYPES.contains(entity.getType());
186        }
187
188        /**
189         * Returns whether this entity needs a separate variable scope. This is the
190         * case for methods and some statements that are determined by
191         * {@link #getSeparateScopeSubtypes()}, as otherwise those would introduce
192         * type information into the wrong scope.
193         */
194        private boolean needsSeparateScope(ShallowEntity entity) {
195                if (entity.getType().equals(EShallowEntityType.METHOD)
196                                || getSeparateScopeSubtypes().contains(entity.getSubtype())) {
197                        return true;
198                }
199                return false;
200        }
201
202        /** Returns all statement sub-types that need a separate scope. */
203        protected abstract Set<String> getSeparateScopeSubtypes();
204
205}
