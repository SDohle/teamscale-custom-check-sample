047/**
048 * Extracts type information from ABAP shallow entities that are either
049 * statements, methods or attributes. Other shallow entity types don't introduce
050 * any type information in ABAP.
051 *
052 * Since ABAP is case-insensitive, all variables and types are lower-cased.
053 *
054 * The {@link AbapTypeInfoExtractor} requires a reference to the rootEntities,
055 * because it needs to have access to the whole file in order to extract all
056 * type infos. Methods in ABAP are the reason for this, because their parameters
057 * are not declared in the same shallow entity as the method implementation. If
058 * no rootEntities are given the corresponding methods return an empty list of
059 * types when called.
060 *
061 * Please note that the extractor does not handle colon-statements (e.g.
062 * <code>data: foo type i, bar type i.</code>) at all. These need to be "rolled
063 * out" into separate statements on the token level before parsing and
064 * subsequently feeding the obtained shallow entities to the extractor.
065 */
066public class AbapTypeInfoExtractor extends TypeInfoExtractorBase {
067
068        /**
069         * Marker type for a dynamic type that we can't resolve statically or based on a
070         * single file.
071         */
072        public static final String DYNAMIC_TYPE_MARKER = "dynamicType";
073
074        /**
075         * Return value used as type for a structured datatype, aka record datatype.
076         */
077        public static final String STRUCTURED_DATATYPE = "Structured Datatype";
078
079        /**
080         * Names of sub types in which variables can be declared.
081         */
082        public static final Set<String> VARIABLE_DECLARATION_SUBTYPES = CollectionUtils.asHashSet(SubTypeNames.CLASS_DATA,
083                        SubTypeNames.CONSTANTS, SubTypeNames.DATA, SubTypeNames.FIELD_SYMBOLS, SubTypeNames.PARAMETERS,
084                        SubTypeNames.STATICS);
085
086        /**
087         * Part of a declaration for a local table with the "new" ABAP data declaration
088         * syntax. The full statement would look like this:
089         * <code>SELECT foo FROM bar INTO TABLE @DATA(lt_localTable)</code>
090         */
091        private static final ETokenType[] LOCAL_TABLE_DECLARATION = new ETokenType[] { ETokenType.INTO, ETokenType.TABLE,
092                        ETokenType.IDENTIFIER, ETokenType.LPAREN, ETokenType.IDENTIFIER, ETokenType.RPAREN };
093
094        /**
095         * Holds all shallow entities of the current file. May be null e.g. when called
096         * from the refactoring detection.
097         */
098        private List<ShallowEntity> rootEntities;
099
100        /**
101         * @param rootEntities
102         *            All shallow entities of the current file(see
103         *            {@link #rootEntities})
104         */
105        public AbapTypeInfoExtractor(List<ShallowEntity> rootEntities) {
106                this.rootEntities = rootEntities;
107        }
108
109        /**
110         * {@inheritDoc}
111         *
112         * Extracts type information from declaration entities of type statement, e.g.
113         * like DATA or FIELD-SYMBOL blocks.
114         */
115        @Override
116        public List<TypedVariable> extractFromStatement(ShallowEntity statementEntity) {
117                if (VARIABLE_DECLARATION_SUBTYPES.contains(statementEntity.getSubtype())) {
118                        return extractFromVariableDeclarationEntity(statementEntity);
119                }
120                if (SubTypeNames.SINGLE_SELECT.equals(statementEntity.getSubtype())
121                                && TokenStreamUtils.containsSequence(statementEntity.ownStartTokens(), LOCAL_TABLE_DECLARATION)) {
122                        return extractFromLocalTableDeclaration(statementEntity);
123
124                }
125                return super.extractFromStatement(statementEntity);
126        }
127
128        /**
129         * Extracts local variable definitions from SELECT statemetns with the "new"
130         * ABAP data declaration syntax. The full statement would look like this:
131         * <code>SELECT foo FROM bar INTO TABLE @DATA(lt_localTable)</code>
132         */
133        private static List<TypedVariable> extractFromLocalTableDeclaration(ShallowEntity statementEntity) {
134                List<TypedVariable> extractedVariables = new ArrayList<>();
135                List<IToken> tokens = statementEntity.ownStartTokens();
136                List<Integer> startIndices = TokenStreamUtils.allStartingIndicesOfTypeSequence(tokens, 0, tokens.size(),
137                                LOCAL_TABLE_DECLARATION);
138                for (Integer startIndex : startIndices) {
139                        if ("@DATA".equalsIgnoreCase(tokens.get(startIndex + 2).getText())) {
140                                String tableName = AbapLanguageFeatureParser.normalizeVariable(tokens.get(startIndex + 4).getText());
141                                extractedVariables
142                                                .add(new TypedVariable(tableName, "TABLE", CollectionUtils.emptyList(), statementEntity));
143                        }
144                }
145                return extractedVariables;
146        }
147
148        /**
149         * {@inheritDoc}
150         *
151         * This applies to all of {@link #VARIABLE_DECLARATION_SUBTYPES} as well as
152         * SELECT-OPTIONS, which are always parsed as attribute.
153         */
154        @Override
155        public List<TypedVariable> extractFromAttribute(ShallowEntity attributeEntity) {
156                if (VARIABLE_DECLARATION_SUBTYPES.contains(attributeEntity.getSubtype())) {
157                        return extractFromVariableDeclarationEntity(attributeEntity);
158                }
159                if (attributeEntity.getSubtype().equals(SubTypeNames.SELECT_OPTIONS)) {
160                        return extractFromSelectOptionsEntity(attributeEntity);
161                }
162
163                return super.extractFromAttribute(attributeEntity);
164        }
165
166        /** {@inheritDoc} */
167        @Override
168        public List<TypedVariable> extractFromMethod(ShallowEntity entity) {
169                switch (entity.getSubtype()) {
170                case SubTypeNames.FORM:
171                        // Forms and functions work the same way.
172                case SubTypeNames.FUNCTION:
173                        return extractFromForm(entity, entity.ownStartTokens());
174                case SubTypeNames.METHOD_DECLARATION:
175                        return extractFromForm(entity, entity.includedTokens());
176                case SubTypeNames.METHOD_IMPLEMENTATION:
177                        return extractFromMethodImplementation(entity);
178                default:
179                        return super.extractFromMethod(entity);
180                }
181        }
182
183        /**
184         * Extracts type information from a METHOD subroutine.
185         *
186         * @return A list of {@link TypedVariable}s belonging to the parameters of the
187         *         method. If no rootEnities are set for the method an empty list is
188         *         returned.
189         */
190        private List<TypedVariable> extractFromMethodImplementation(ShallowEntity methodEntity) {
191                if (rootEntities == null) {
192                        return CollectionUtils.emptyList();
193                }
194
195                List<IToken> declaration = LanguageFeatureParser.ABAP.getDeclarationTokensForMethod(rootEntities, methodEntity);
196
197                // If no declaration has been found for various reasons return no types.
198                // This may happen when called with a non complete list of
199                // #rootEntities. Maybe even for a complete list when the method
200                // implements an interface. The TypeInfoExtractor also is called from
201                // the MethodRefactoringDetector base with a non complete list of root
202                // entities.
203                if (declaration == null) {
204                        return CollectionUtils.emptyList();
205                }
206
207                return extractFromForm(methodEntity, declaration);
208        }
209
210        /** Extracts type information from a FORM subroutine. */
211        private static List<TypedVariable> extractFromForm(ShallowEntity entity, List<IToken> formTokens) {
212                return LanguageFeatureParser.ABAP.getTypeInfoForMethodParameters(entity,
213                                formTokens.subList(2, formTokens.size()));
214        }
215
216        /** Extracts type information from a variable declaration entity. */
217        private static List<TypedVariable> extractFromVariableDeclarationEntity(ShallowEntity entity) {
218                List<IToken> tokens = entity.ownStartTokens();
219                if (tokens.size() <= 1) {
220                        return Collections.emptyList();
221                }
222
223                String variableName;
224                String typeName;
225                if (TokenStreamUtils.hasTokenTypeSequence(tokens, 1, BEGIN, OF)) {
226                        variableName = tokens.get(3).getText();
227                        typeName = STRUCTURED_DATATYPE;
228                } else if (TokenStreamUtils.hasTokenTypeSequence(tokens, 0, DATA, LPAREN, IDENTIFIER, RPAREN, EQ)) {
229                        // we can't determine the type based on file-local information in this case
230                        variableName = tokens.get(2).getText();
231                        typeName = DYNAMIC_TYPE_MARKER;
232                } else if (TokenStreamUtils.hasTokenTypeSequence(tokens, 1, LT, IDENTIFIER, GT)) {
233                        // Handle field symbol
234                        variableName = TokenStreamTextUtils.concatTokenTexts(tokens.subList(1, 4));
235                        int startOffsetOfType = 4;
236                        typeName = LanguageFeatureParser.ABAP.getNextTypeName(tokens, startOffsetOfType).getFirst();
237                } else {
238                        variableName = tokens.get(1).getText();
239                        int startOffsetOfType = 2;
240                        typeName = LanguageFeatureParser.ABAP.getNextTypeName(tokens, startOffsetOfType).getFirst();
241                }
242                return Collections.singletonList(new TypedVariable(AbapLanguageFeatureParser.normalizeVariable(variableName),
243                                typeName, CollectionUtils.emptyList(), entity));
244        }
245
246        /**
247         * Extracts names from an entity of type RANGES, FIELD-GROUPS or TYPES in order
248         * to have a common abstraction for naming checks. This is similar to
249         * {@link #extractFromVariableDeclarationEntity(ShallowEntity)}, but does not
250         * attempt to read type information. Ranges, field-groups and types are not
251         * actually variables and therefore should generally not appear in the type
252         * lookup along actual variables.
253         */
254        public static List<TypedVariable> extractFromVariableDeclarationLikeEntity(ShallowEntity entity) {
255                List<IToken> tokens = entity.ownStartTokens();
256                if (tokens.size() <= 1) {
257                        return Collections.emptyList();
258                }
259
260                if (TokenStreamUtils.hasTokenTypeSequence(tokens, 1, BEGIN, OF)) {
261                        String variableName = tokens.get(3).getText();
262                        return Collections
263                                        .singletonList(new TypedVariable(AbapLanguageFeatureParser.normalizeVariable(variableName),
264                                                        STRUCTURED_DATATYPE, CollectionUtils.emptyList(), entity));
265                }
266
267                String variableName = tokens.get(1).getText();
268                return Collections.singletonList(new TypedVariable(AbapLanguageFeatureParser.normalizeVariable(variableName),
269                                null, CollectionUtils.emptyList(), entity));
270        }
271
272        /** Extracts type information from select-options tokens. */
273        private static List<TypedVariable> extractFromSelectOptionsEntity(ShallowEntity entity) {
274                List<IToken> tokens = entity.ownStartTokens();
275                String variableName = AbapLanguageFeatureParser.normalizeVariable(tokens.get(1).getText());
276                return Collections.singletonList(
277                                new TypedVariable(variableName, SubTypeNames.SELECT_OPTIONS, CollectionUtils.emptyList(), entity));
278        }
