017package eu.cqse.check.framework.typetracker.abap;
018
019import static eu.cqse.check.framework.scanner.ETokenType.BEGIN;
020import static eu.cqse.check.framework.scanner.ETokenType.END;
021import static eu.cqse.check.framework.scanner.ETokenType.GT;
022import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
023import static eu.cqse.check.framework.scanner.ETokenType.LT;
024import static eu.cqse.check.framework.scanner.ETokenType.OF;
025
026import java.util.ArrayList;
027import java.util.HashSet;
028import java.util.List;
029import java.util.Set;
030
031import org.conqat.lib.commons.collections.CollectionUtils;
032import org.conqat.lib.commons.collections.Pair;
033
034import eu.cqse.check.framework.scanner.IToken;
035import eu.cqse.check.framework.shallowparser.SubTypeNames;
036import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
037import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
038import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
039import eu.cqse.check.framework.typetracker.TypeInfoExtractorBase;
040import eu.cqse.check.framework.typetracker.TypedVariable;
041import eu.cqse.check.framework.util.LanguageFeatureParser;

043/**
044 * Extracts type information from ABAP shallow entities that are either
045 * statements, methods or attributes. Other shallow entity types don't introduce
046 * any type information in ABAP.
047 * 
048 * The {@link AbapTypeInfoExtractor} implements the
049 * {@link IFullScopeTypeInfoExtractor}, because it needs to have access to the
050 * whole file in order to extract all type infos. Methods in ABAP are the reason
051 * for this, because their parameters are not declared in the same shallow
052 * entity as the method implementation. If no rootEntities are given the
053 * corresponding methods return an empty list of types when called.
054 * 
055 * @author $Author: roehm $
056 * @version $Rev: 55961 $
057 * @ConQAT.Rating YELLOW Hash: 6BA45052FC16FAE6AEC3C3C5CA896DE4
058 */
059public class AbapTypeInfoExtractor extends TypeInfoExtractorBase
060                implements IFullScopeTypeInfoExtractor {
061
062        /**
063         * Return value used as type for a structured datatype, aka record datatype.
064         */
065        public static final String STRUCTURED_DATATYPE = "Structured Datatype";
066
067        /**
068         * Names of sub types in which variables can be declared.
069         */
070        private static final Set<String> VARIABLE_DECLARATION_SUBTYPES = CollectionUtils.asHashSet(
071                        SubTypeNames.CLASS_DATA,SubTypeNames.CONSTANTS, SubTypeNames.DATA,
072                        SubTypeNames.FIELD_SYMBOLS, SubTypeNames.PARAMETERS, SubTypeNames.STATICS);
073
074        /**
075         * Holds all shallow entities of the current file. May be null e.g. when
076         * called from the refactoring detection.
077         */
078        private List<ShallowEntity> rootEntities;
079
080        /** {@inheritDoc} */
081        @Override
082        public void setRootEntities(List<ShallowEntity> rootEntities) {
083                this.rootEntities = rootEntities;
084        }
085
086        /**
087         * {@inheritDoc}
088         * 
089         * Extracts type information from declaration entities of type statement,
090         * e.g. like DATA or FIELD-SYMBOL blocks.
091         */
092        @Override
093        public List<TypedVariable> extractFromStatement(
094                        ShallowEntity statementEntity) {
095                if (VARIABLE_DECLARATION_SUBTYPES
096                                .contains(statementEntity.getSubtype())) {
097                        return extractFromVariableDeclarationEntity(statementEntity);
098                }
099
100                return super.extractFromStatement(statementEntity);
101        }
102
103        /** {@inheritDoc} */
104        @Override
105        public List<TypedVariable> extractFromAttribute(
106                        ShallowEntity attributeEntity) {
107                if (VARIABLE_DECLARATION_SUBTYPES
108                                .contains(attributeEntity.getSubtype())) {
109                        return extractFromVariableDeclarationEntity(attributeEntity);
110                }
111
112                return super.extractFromAttribute(attributeEntity);
113        }
114
115        /** {@inheritDoc} */
116        @Override
117        public List<TypedVariable> extractFromMethod(ShallowEntity entity) {
118                switch (entity.getSubtype()) {
119                case SubTypeNames.FORM:
120                        return extractFromForm(entity, entity.ownStartTokens());
121                case SubTypeNames.METHOD_DECLARATION:
122                        // METHOD_DECLARATION is handled the same way as
123                        // METHOD_IMPLEMENTATION
124                case SubTypeNames.METHOD_IMPLEMENTATION:
125                        return extractFromMethodImplementation(entity);
126                default:
127                        return super.extractFromMethod(entity);
128                }
129        }
130
131        /**
132         * Extracts type information from a METHOD subroutine.
133         * 
134         * @return A list of {@link TypedVariable}s belonging to the parameters of
135         *         the method. If no rootEnities are set for the method an empty
136         *         list is returned.
137         */
138        private List<TypedVariable> extractFromMethodImplementation(
139                        ShallowEntity methodEntity) {
140                if (rootEntities == null) {
141                        return CollectionUtils.emptyList();
142                }
143
144                List<IToken> declaration = LanguageFeatureParser.ABAP
145                                .getDeclarationForMethod(rootEntities, methodEntity);
146
147                /*
148                 * If no declaration has been found for various reasons return no types.
149                 * This may happen when called with a non complete list of
150                 * #rootEntities. Maybe even for a complete list when the method
151                 * implements an interface. The TypeInfoExtractor also is called from
152                 * the MethodRefactoringDetector base with a non complete list of root
153                 * entities.
154                 */
155                if (declaration == null) {
156                        return CollectionUtils.emptyList();
157                }
158
159                return extractFromForm(methodEntity, declaration);
160        }
161
162        /** Extracts type information from a FORM subroutine. */
163        private List<TypedVariable> extractFromForm(ShallowEntity entity,
164                        List<IToken> formTokens) {
165                return LanguageFeatureParser.ABAP.getTypeInfoForMethodParameters(entity,
166                                formTokens.subList(2, formTokens.size()), logger);
167        }
168
169        /** Extracts type information from variable tokens. */
170        private List<TypedVariable> extractFromVariableDeclarationEntity(
171                        ShallowEntity entity) {
172                List<TypedVariable> typeInfo = new ArrayList<>();
173                List<List<IToken>> tokensWithoutBeginEndBlocks = getTokensWithReducedBeginEndBlocks(
174                                entity);
175
176                for (List<IToken> tokens : tokensWithoutBeginEndBlocks) {
177                        if (TokenStreamUtils.hasTokenTypeSequence(tokens, 1, BEGIN, OF)) {
178                                String variableName = tokens.get(3).getText();
179                                typeInfo.add(
180                                                new TypedVariable(variableName, STRUCTURED_DATATYPE,
181                                                                CollectionUtils.emptyList(), entity));
182                        } else {
183                                int startOffsetOfType;
184                                String variableName;
185                                // Handle field symbol
186                                if (TokenStreamUtils.hasTokenTypeSequence(tokens, 1, LT,
187                                                IDENTIFIER, GT)) {
188                                        variableName = TokenStreamTextUtils
189                                                        .concatTokenTexts(tokens.subList(1, 4));
190                                        startOffsetOfType = 4;
191                                } else {
192                                        variableName = tokens.get(1).getText();
193                                        startOffsetOfType = 2;
194                                }
195                                Pair<String, Integer> type = LanguageFeatureParser.ABAP
196                                                .getNextTypeName(tokens, startOffsetOfType, logger);
197                                typeInfo.add(new TypedVariable(variableName, type.getFirst(),
198                                                CollectionUtils.emptyList(), entity));
199                        }
200                }
201
202                return typeInfo;
203        }
204
205        /**
206         * Retrieves all tokens of a shallow entity and performs two operations:
207         * <ul>
208         * <li>Rolls out colons and thereby splits token list
209         * <li>Removes END of tokens and inner tokens of a BEGIN OF ... END OF block
210         * </ul>
211         * From a BEGIN OF ... END OF block, only the first tokens representing
212         * "BEGIN OF name" remains.
213         * 
214         * @param entity
215         *            A shallow entity declaring a variable.
216         * @return A list of token lists resulting from the two operations.
217         */
218        private static List<List<IToken>> getTokensWithReducedBeginEndBlocks(
219                        ShallowEntity entity) {
220                List<IToken> entityTokens = entity.ownStartTokens();
221                List<List<IToken>> splitTokens = LanguageFeatureParser.ABAP
222                                .rolloutColonStatement(entityTokens);
223                List<List<IToken>> resultList = new ArrayList<List<IToken>>();
224
225                boolean inBeginEndBlock = false;
226                for (List<IToken> tokens : splitTokens) {
227                        if (TokenStreamUtils.hasTokenTypeSequence(tokens, 1, BEGIN, OF)) {
228                                inBeginEndBlock = true;
229                                resultList.add(tokens);
230                        } else if (TokenStreamUtils.hasTokenTypeSequence(tokens, 1, END,
231                                        OF)) {
232                                inBeginEndBlock = false;
233                                continue;
234                        } else if (inBeginEndBlock) {
235                                continue;
236                        } else {
237                                resultList.add(tokens);
238                        }
239                }
240
241                return resultList;
242        }
