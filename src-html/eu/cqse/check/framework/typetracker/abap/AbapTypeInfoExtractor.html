017package eu.cqse.check.framework.typetracker.abap;
018
019import static eu.cqse.check.framework.scanner.ETokenType.BEGIN;
020import static eu.cqse.check.framework.scanner.ETokenType.GT;
021import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
022import static eu.cqse.check.framework.scanner.ETokenType.LT;
023import static eu.cqse.check.framework.scanner.ETokenType.OF;
024
025import java.util.Collections;
026import java.util.List;
027import java.util.Set;
028
029import org.conqat.lib.commons.collections.CollectionUtils;
030import org.conqat.lib.commons.collections.Pair;
031import org.conqat.lib.commons.logging.ILogger;
032
033import eu.cqse.check.framework.scanner.IToken;
034import eu.cqse.check.framework.shallowparser.SubTypeNames;
035import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
036import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
037import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
038import eu.cqse.check.framework.typetracker.TypeInfoExtractorBase;
039import eu.cqse.check.framework.typetracker.TypedVariable;
040import eu.cqse.check.framework.util.LanguageFeatureParser;
041import eu.cqse.check.framework.util.abap.AbapLanguageFeatureParser;

043/**
044 * Extracts type information from ABAP shallow entities that are either
045 * statements, methods or attributes. Other shallow entity types don't introduce
046 * any type information in ABAP.
047 *
048 * Since ABAP is case-insensitive, all variables and types are lower-cased.
049 *
050 * The {@link AbapTypeInfoExtractor} requires a reference to the rootEntities,
051 * because it needs to have access to the whole file in order to extract all
052 * type infos. Methods in ABAP are the reason for this, because their parameters
053 * are not declared in the same shallow entity as the method implementation. If
054 * no rootEntities are given the corresponding methods return an empty list of
055 * types when called.
056 *
057 * Please note that the extractor does not handle colon-statements (e.g.
058 * <code>data: foo type i, bar type i.</code>) at all. These need to be "rolled
059 * out" into separate statements on the token level before parsing and
060 * subsequently feeding the obtained shallow entities to the extractor.
061 */
062public class AbapTypeInfoExtractor extends TypeInfoExtractorBase {
063
064        /**
065         * Return value used as type for a structured datatype, aka record datatype.
066         */
067        public static final String STRUCTURED_DATATYPE = "Structured Datatype";
068
069        /**
070         * Names of sub types in which variables can be declared.
071         */
072        private static final Set<String> VARIABLE_DECLARATION_SUBTYPES = CollectionUtils.asHashSet(SubTypeNames.CLASS_DATA,
073                        SubTypeNames.CONSTANTS, SubTypeNames.DATA, SubTypeNames.FIELD_SYMBOLS, SubTypeNames.PARAMETERS,
074                        SubTypeNames.STATICS);
075
076        /**
077         * Holds all shallow entities of the current file. May be null e.g. when
078         * called from the refactoring detection.
079         */
080        private List<ShallowEntity> rootEntities;
081
082        /** Logger instance for printing warnings. */
083        private ILogger logger;
084
085        /**
086         * @param rootEntities
087         *            All shallow entities of the current file(see
088         *            {@link #rootEntities})
089         * @param logger
090         *            Logger object that is used to communicate warnings.
091         */
092        public AbapTypeInfoExtractor(List<ShallowEntity> rootEntities, ILogger logger) {
093                this.rootEntities = rootEntities;
094                this.logger = logger;
095        }
096
097        /**
098         * {@inheritDoc}
099         *
100         * Extracts type information from declaration entities of type statement,
101         * e.g. like DATA or FIELD-SYMBOL blocks.
102         */
103        @Override
104        public List<TypedVariable> extractFromStatement(ShallowEntity statementEntity) {
105                if (VARIABLE_DECLARATION_SUBTYPES.contains(statementEntity.getSubtype())) {
106                        return Collections.singletonList(extractFromVariableDeclarationEntity(statementEntity));
107                }
108
109                return super.extractFromStatement(statementEntity);
110        }
111
112        /**
113         * {@inheritDoc}
114         *
115         * This applies to all of {@link #VARIABLE_DECLARATION_SUBTYPES} as well as
116         * SELECT-OPTIONS, which are always parsed as attribute.
117         */
118        @Override
119        public List<TypedVariable> extractFromAttribute(ShallowEntity attributeEntity) {
120                if (VARIABLE_DECLARATION_SUBTYPES.contains(attributeEntity.getSubtype())) {
121                        return Collections.singletonList(extractFromVariableDeclarationEntity(attributeEntity));
122                }
123                if (attributeEntity.getSubtype().equals(SubTypeNames.SELECT_OPTIONS)) {
124                        return extractFromSelectOptionsEntity(attributeEntity);
125                }
126
127                return super.extractFromAttribute(attributeEntity);
128        }
129
130        /** {@inheritDoc} */
131        @Override
132        public List<TypedVariable> extractFromMethod(ShallowEntity entity) {
133                switch (entity.getSubtype()) {
134                case SubTypeNames.FORM:
135                        // Forms and functions work the same way.
136                case SubTypeNames.FUNCTION:
137                        return extractFromForm(entity, entity.ownStartTokens());
138                case SubTypeNames.METHOD_DECLARATION:
139                        // METHOD_DECLARATION is handled the same way as
140                        // METHOD_IMPLEMENTATION
141                case SubTypeNames.METHOD_IMPLEMENTATION:
142                        return extractFromMethodImplementation(entity);
143                default:
144                        return super.extractFromMethod(entity);
145                }
146        }
147
148        /**
149         * Extracts type information from a METHOD subroutine.
150         *
151         * @return A list of {@link TypedVariable}s belonging to the parameters of
152         *         the method. If no rootEnities are set for the method an empty
153         *         list is returned.
154         */
155        private List<TypedVariable> extractFromMethodImplementation(ShallowEntity methodEntity) {
156                if (rootEntities == null) {
157                        return CollectionUtils.emptyList();
158                }
159
160                List<IToken> declaration = LanguageFeatureParser.ABAP.getDeclarationTokensForMethod(rootEntities, methodEntity);
161
162                // If no declaration has been found for various reasons return no types.
163                // This may happen when called with a non complete list of
164                // #rootEntities. Maybe even for a complete list when the method
165                // implements an interface. The TypeInfoExtractor also is called from
166                // the MethodRefactoringDetector base with a non complete list of root
167                // entities.
168                if (declaration == null) {
169                        return CollectionUtils.emptyList();
170                }
171
172                return extractFromForm(methodEntity, declaration);
173        }
174
175        /** Extracts type information from a FORM subroutine. */
176        private List<TypedVariable> extractFromForm(ShallowEntity entity, List<IToken> formTokens) {
177                return LanguageFeatureParser.ABAP.getTypeInfoForMethodParameters(entity,
178                                formTokens.subList(2, formTokens.size()), logger);
179        }
180
181        /** Extracts type information from a variable declaration entity. */
182        private TypedVariable extractFromVariableDeclarationEntity(ShallowEntity entity) {
183                List<IToken> tokens = entity.ownStartTokens();
184
185                if (TokenStreamUtils.hasTokenTypeSequence(tokens, 1, BEGIN, OF)) {
186                        String variableName = AbapLanguageFeatureParser.normalizeVariable(tokens.get(3).getText());
187                        return new TypedVariable(variableName, STRUCTURED_DATATYPE, CollectionUtils.emptyList(), entity);
188                }
189
190                int startOffsetOfType;
191                String variableName;
192                // Handle field symbol
193                if (TokenStreamUtils.hasTokenTypeSequence(tokens, 1, LT, IDENTIFIER, GT)) {
194                        variableName = TokenStreamTextUtils.concatTokenTexts(tokens.subList(1, 4));
195                        startOffsetOfType = 4;
196                } else {
197                        variableName = tokens.get(1).getText();
198                        startOffsetOfType = 2;
199                }
200                Pair<String, Integer> type = LanguageFeatureParser.ABAP.getNextTypeName(tokens, startOffsetOfType, logger);
201                return new TypedVariable(AbapLanguageFeatureParser.normalizeVariable(variableName), type.getFirst(),
202                                CollectionUtils.emptyList(), entity);
203        }
204
205        /**
206         * Extracts names from an entity of type RANGES, FIELD-GROUPS or TYPES in
207         * order to have a common abstraction for naming checks. This is similar to
208         * {@link #extractFromVariableDeclarationEntity(ShallowEntity)}, but does
209         * not attempt to read type information. Ranges, field-groups and types are
210         * not actually variables and therefore should generally not appear in the
211         * type lookup along actual variables.
212         */
213        public static TypedVariable extractFromVariableDeclarationLikeEntity(ShallowEntity entity) {
214                List<IToken> tokens = entity.ownStartTokens();
215
216                if (TokenStreamUtils.hasTokenTypeSequence(tokens, 1, BEGIN, OF)) {
217                        String variableName = tokens.get(3).getText();
218                        return new TypedVariable(AbapLanguageFeatureParser.normalizeVariable(variableName), STRUCTURED_DATATYPE,
219                                        CollectionUtils.emptyList(), entity);
220                }
221
222                String variableName = tokens.get(1).getText();
223                return new TypedVariable(AbapLanguageFeatureParser.normalizeVariable(variableName), null,
224                                CollectionUtils.emptyList(),
225                                entity);
226        }
227
228        /** Extracts type information from select-options tokens. */
229        private static List<TypedVariable> extractFromSelectOptionsEntity(ShallowEntity entity) {
230                List<IToken> tokens = entity.ownStartTokens();
231                String variableName = AbapLanguageFeatureParser.normalizeVariable(tokens.get(1).getText());
232                return Collections.singletonList(
233                                new TypedVariable(variableName, SubTypeNames.SELECT_OPTIONS, CollectionUtils.emptyList(), entity));
234        }
235}
