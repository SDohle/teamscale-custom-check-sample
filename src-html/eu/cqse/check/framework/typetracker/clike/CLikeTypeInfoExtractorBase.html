017package eu.cqse.check.framework.typetracker.clike;
018
019import static eu.cqse.check.framework.scanner.ETokenType.COLON;
020import static eu.cqse.check.framework.scanner.ETokenType.LBRACE;
021import static eu.cqse.check.framework.scanner.ETokenType.RBRACE;
022
023import java.util.ArrayList;
024import java.util.Arrays;
025import java.util.List;
026
027import org.conqat.lib.commons.assertion.CCSMAssert;
028import org.conqat.lib.commons.collections.CollectionUtils;
029import org.conqat.lib.commons.collections.Pair;
030
031import eu.cqse.check.framework.scanner.ETokenType;
032import eu.cqse.check.framework.scanner.IToken;
033import eu.cqse.check.framework.shallowparser.SubTypeNames;
034import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
035import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
036import eu.cqse.check.framework.typetracker.TypeInfoExtractorBase;
037import eu.cqse.check.framework.typetracker.TypedVariable;
038import eu.cqse.check.framework.util.CLikeLanguageFeatureParserBase;
039
040/**
041 * The clike type info extractor is an abstract base class for type info
042 * extractors for clike languages. Type information in clike languages is only
043 * introduces by shallow entities that are either statements, methods or
044 * attributes. Other shallow entity types don't introduce any type information.
045 */
046public abstract class CLikeTypeInfoExtractorBase extends TypeInfoExtractorBase {
047
048        /** The feature parser. */
049        protected final CLikeLanguageFeatureParserBase featureParser;
050
051        /** Constructor. */
052        public CLikeTypeInfoExtractorBase(CLikeLanguageFeatureParserBase languageUtils) {
053                CCSMAssert.isNotNull(languageUtils);
054                this.featureParser = languageUtils;
055        }
056
057        /** {@inheritDoc} */
058        @Override
059        public List<TypedVariable> extractFromAttribute(ShallowEntity attributeEntity) {
060                String subtype = attributeEntity.getSubtype();
061                if (subtype.equals(SubTypeNames.ATTRIBUTE)) {
062                        return extractFromVariableTokens(attributeEntity, attributeEntity.ownStartTokens());
063                }
064                return super.extractFromAttribute(attributeEntity);
065        }
066
067        /** {@inheritDoc} */
068        @Override
069        public List<TypedVariable> extractFromMethod(ShallowEntity methodEntity) {
070                List<List<IToken>> splitParameterTokens = featureParser.getSplitParameterTokens(methodEntity);
071
072                List<TypedVariable> typeInfo = new ArrayList<>();
073                for (List<IToken> parameter : splitParameterTokens) {
074                        Pair<List<IToken>, String> modifiersAndType = featureParser.getModifiersAndTypeFromTokens(parameter);
075                        String typeName = modifiersAndType.getSecond();
076                        List<IToken> modifiers = modifiersAndType.getFirst();
077
078                        IToken variableName = featureParser.getVariableNameFromTokens(parameter);
079                        if (variableName != null) {
080                                typeInfo.add(new TypedVariable(variableName.getText(), typeName, modifiers, methodEntity));
081                        }
082                }
083
084                return typeInfo;
085        }
086
087        /**
088         * Extracts type information from the start tokens of for, foreach and using
089         * statements. The given end token type is used to determine the end of the
090         * variable declaration within the loop. For loops use SEMICOLON, while
091         * foreach loops use IN and using statements use RPAREN.
092         */
093        protected List<TypedVariable> extractFromForLikeTokens(ShallowEntity entity, ETokenType endToken) {
094                List<IToken> variableTokens = featureParser.getVariableTokensFromForLikeTokens(entity.ownStartTokens(),
095                                endToken);
096                return extractFromVariableTokens(entity, variableTokens);
097        }
098
099        /** Extracts type information from variable tokens. */
100        protected List<TypedVariable> extractFromVariableTokens(ShallowEntity entity, List<IToken> variableTokens) {
101                List<List<IToken>> splitTokens = featureParser.splitVariableTokens(variableTokens);
102                CCSMAssert.isFalse(splitTokens.isEmpty(), "splitTokens must have at least one element");
103
104                List<IToken> firstTokens = splitTokens.get(0);
105                if (!featureParser.isVariableDeclaration(firstTokens)) {
106                        return CollectionUtils.emptyList();
107                }
108
109                Pair<List<IToken>, String> modifiersAndType = featureParser.getModifiersAndTypeFromTokens(firstTokens);
110                String typeName = modifiersAndType.getSecond();
111                List<IToken> modifiers = modifiersAndType.getFirst();
112
113                List<TypedVariable> typeInfo = new ArrayList<>();
114                for (List<IToken> tokens : splitTokens) {
115                        IToken variableName = featureParser.getVariableNameFromTokens(tokens);
116                        if (variableName != null) {
117                                typeInfo.add(new TypedVariable(variableName.getText(), typeName, modifiers, entity));
118                        }
119                }
120
121                return typeInfo;
122        }
123
124        /** Extracts type information from the given catch statement. */
125        protected List<TypedVariable> extractFromCatchTokens(ShallowEntity entity) {
126                List<IToken> exceptionTokens = TokenStreamUtils.tokensBetween(entity.ownStartTokens(), ETokenType.LPAREN,
127                                ETokenType.RPAREN);
128                List<TypedVariable> typeInfo = new ArrayList<>();
129                if (exceptionTokens.size() <= 1) {
130                        return typeInfo;
131                }
132
133                Pair<List<IToken>, String> modifiersAndType = featureParser.getModifiersAndTypeFromTokens(exceptionTokens);
134                String typeName = modifiersAndType.getSecond();
135                List<IToken> modifiers = modifiersAndType.getFirst();
136
137                IToken variableName = featureParser.getVariableNameFromTokens(exceptionTokens);
138                if (variableName != null) {
139                        typeInfo.add(new TypedVariable(variableName.getText(), typeName, modifiers, entity));
140                }
141
142                return typeInfo;
143        }
144
145        /**
146         * Extracts type information from a for or for each loop. The loop type is
147         * distinguished by looking for a colon.
148         */
149        protected List<TypedVariable> extractFromForConstructWithPossibleColon(ShallowEntity statementEntity) {
150                ETokenType endToken = ETokenType.SEMICOLON;
151                if (hasTopLevelColon(statementEntity)) {
152                        endToken = ETokenType.COLON;
153                }
154                return extractFromForLikeTokens(statementEntity, endToken);
155        }
156
157        /**
158         * Returns whether the given entity has a top-level colon that is not nested
159         * in between braces.
160         */
161        private static boolean hasTopLevelColon(ShallowEntity statementEntity) {
162                return TokenStreamUtils.findFirstTopLevel(statementEntity.ownStartTokens(), COLON, Arrays.asList(LBRACE),
163                                Arrays.asList(RBRACE)) != TokenStreamUtils.NOT_FOUND;
164        }
165}
