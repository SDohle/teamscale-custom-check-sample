035/**
036 * Allows resolving the type of variables and constructor calls in an
037 * import-sensitive manner. Methods can be assumed to return values of some
038 * type. Can also be used to check whether some type is imported.
039 */
040public class JavaImportSensitiveTypeResolver {
041
042        /**
043         * Holds all imported types as a map from short type name to the fully qualified
044         * type name.
045         */
046        private final Map<String, String> typeImports;
047
048        /** All package names that are imported using the wildcard ('*'). */
049        private final Set<String> wildcardImports = new HashSet<>();
050
051        public JavaImportSensitiveTypeResolver(ShallowEntity root) throws CheckException {
052                typeImports = new HashMap<>();
053                List<ShallowEntity> importEntities = CheckImplementationBase.select(root,
054                                "/META[subtype('" + SubTypeNames.IMPORT + "')]");
055                for (ShallowEntity entity : importEntities) {
056                        if (TokenStreamUtils.firstTokenOfType(entity.ownStartTokens(), STATIC) != TokenStreamUtils.NOT_FOUND) {
057                                continue; // ignore static imports
058                        }
059
060                        String importIdentifier = entity.getName();
061                        Pair<String, String> pPackageImported = StringUtils.splitAtLast(importIdentifier, '.');
062                        if (pPackageImported.getSecond().equals("*")) {
063                                wildcardImports.add(pPackageImported.getFirst());
064                        } else {
065                                typeImports.put(pPackageImported.getSecond(), importIdentifier);
066                        }
067                }
068        }
069
070        /**
071         * @see #isOfType(List, Set, ScopedTypeLookup, Set, Set) assume about no methods
072         *      that they return this type.
073         */
074        public boolean isOfType(List<IToken> tokens, Set<String> fullyQualifiedTypeNames, ScopedTypeLookup lookup) {
075                return isOfType(tokens, fullyQualifiedTypeNames, lookup, Collections.emptySet());
076        }
077
078        /**
079         * @see #isOfType(List, Set, ScopedTypeLookup, Set, Set) assume about no static
080         *      methods that they return this type.
081         */
082        public boolean isOfType(List<IToken> tokens, Set<String> fullyQualifiedTypeNames, ScopedTypeLookup lookup,
083                        Set<String> methods) {
084                return isOfType(tokens, fullyQualifiedTypeNames, lookup, methods, Collections.emptySet());
085        }
086
087        /**
088         * Checks whether the <b>last expression</b> in {@code tokens} is of one of the
089         * specified <b>fully qualified</b> types. Assume this is true if the expression
090         * is a call to a method in {@code methods}, or in {@code staticMethods}, which
091         * should be as qualified as they're expected to appear in code.</br>
092         * Limitations:
093         * <ul>
094         * <li>We don't handle nested classes as types</li>
095         * <li>We only automatically check constructor calls and the local typing
096         * context</li>
097         * <li>We don't handle superfluous enclosing parenthesis, e.g. (foo).method</li>
098         * </ul>
099         */
100        public boolean isOfType(List<IToken> tokens, Set<String> fullyQualifiedTypeNames, ScopedTypeLookup lookup,
101                        Set<String> methods, Set<String> staticMethods) {
102                if (tokens.isEmpty()) {
103                        return false;
104                }
105
106                IToken last = Iterables.getLast(tokens);
107                switch (last.getType()) {
108                case IDENTIFIER:
109                        TypedVariable typedVariable = lookup.getTypeInfo(last.getText());
110                        if (typedVariable == null) {
111                                return false; // don't know the type
112                        }
113                        return typeIsOf(typedVariable.getTypeName(), fullyQualifiedTypeNames);
114                case RPAREN:
115                        return callReturnsType(tokens, fullyQualifiedTypeNames, methods, staticMethods);
116                default:
117                        return false;
118                }
119        }
120
121        private boolean callReturnsType(List<IToken> tokens, Set<String> fullyQualifiedTypeNames, Set<String> methods,
122                        Set<String> staticMethods) {
123                int indexMatchingLparen = TokenStreamUtils.findMatchingOpeningToken(tokens, tokens.size() - 2,
124                                ETokenType.LPAREN, ETokenType.RPAREN);
125                CCSMAssert.isFalse(indexMatchingLparen == TokenStreamUtils.NOT_FOUND,
126                                "tokens must include the opening parenthesis");
127                if (indexMatchingLparen == 0) {// skip completely parenthesized expression
128                        return false;
129                }
130                // First detect a qualified sequence, then try to treat it as a
131                // constructor or static method call
132                int qSequenceEndNonInclusive = indexMatchingLparen;
133                // skip generics
134                if (indexMatchingLparen - 2 >= 0 && tokens.get(indexMatchingLparen - 1).getType() == ETokenType.GT) {
135                        qSequenceEndNonInclusive = TokenStreamUtils.findMatchingOpeningToken(tokens, indexMatchingLparen - 2,
136                                        ETokenType.LT, ETokenType.GT);
137                        CCSMAssert.isFalse(qSequenceEndNonInclusive == TokenStreamUtils.NOT_FOUND,
138                                        "tokens must include opening generics delimiter");
139                }
140                int qSequenceStart = TokenStreamUtils.firstTokenOfAlternatingTypes(tokens, qSequenceEndNonInclusive - 1,
141                                ETokenType.IDENTIFIER, ETokenType.DOT);
142                if (qSequenceStart != TokenStreamUtils.NOT_FOUND) {
143                        String qSequence = TokenStreamTextUtils
144                                        .concatTokenTexts(tokens.subList(qSequenceStart, qSequenceEndNonInclusive));
145                        if (qSequenceStart - 1 >= 0 && tokens.get(qSequenceStart - 1).getType() == NEW) {
146                                return typeIsOf(qSequence, fullyQualifiedTypeNames);
147                        }
148                        if (staticMethods.contains(qSequence)) {
149                                return true;
150                        }
151                }
152                // check non-static methods
153                if (tokens.get(indexMatchingLparen - 1).getType() == ETokenType.IDENTIFIER) {
154                        return methods.contains(tokens.get(indexMatchingLparen - 1).getText());
155                }
156                return false;
157        }
158
159        /**
160         * Checks if a simple {@code type} is any of {@code fullyQualifiedNames}
161         */
162        public boolean typeIsOf(String type, Set<String> fullyQualifiedTypeNames) {
163                String fullTypeName;
164                if (type.contains(".")) {
165                        // assume fully qualified
166                        fullTypeName = type;
167                } else {
168                        fullTypeName = typeImports.get(type);
169                }
170                if (fullTypeName != null) {
171                        return fullyQualifiedTypeNames.contains(fullTypeName);
172                }
173                // check wildcard imports
174                return fullyQualifiedTypeNames.stream().filter(fqType -> fqType.endsWith(type))// matching simple type name
175                                .anyMatch(this::isWildcardImported);
176        }
177
178        /**
179         * Checks whether type given by its fully qualified name is imported. Also
180         * considers wildcard imports.
181         */
182        public boolean isImported(String fullyQualifiedName) {
183                String simpleName = StringUtils.getLastPart(fullyQualifiedName, '.');
184                if (typeImports.containsKey(simpleName)) {
185                        return typeImports.get(simpleName).equals(fullyQualifiedName);
186                }
187                return isWildcardImported(fullyQualifiedName);
188        }
189
190        private boolean isWildcardImported(String fullyQualifiedName) {
191                return wildcardImports.contains(StringUtils.removeLastPart(fullyQualifiedName, '.'));
192        }
