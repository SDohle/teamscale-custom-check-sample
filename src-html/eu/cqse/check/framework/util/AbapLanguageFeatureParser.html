070        /**
071         * EnumSet containing all token types which can appear as section headers in
072         * a parameter declaration.
073         */
074        public static final EnumSet<ETokenType> PARAMETER_SECTION_TOKENS = EnumSet
075                        .of(USING, CHANGING, IMPORTING, EXPORTING, RETURNING, TABLES,
076                                        TESTING, RAISING);

078        /** {@inheritDoc} */
079        @Override
080        public ELanguage getLanguage() {
081                return ELanguage.ABAP;

084        /**
085         * Gets the type name from the given list of tokens.
086         * 
087         * @param tokens
088         *            List of tokens. For example v1 TYPE t value(v2) LIKE c v TYPE
089         *            t v3.
090         * 
091         * @return Returns a tuple of the type name and the corresponding index of
092         *         the last type token. The index is ensured to be bigger or equal
093         *         to startIndex-1. If no further type is found, ("", startIndex-1)
094         *         is returned.
095         */
096        public Pair<String, Integer> getNextTypeName(List<IToken> tokens,
097                        int startIndex, ILogger logger) {
098                int typeKeywordIndex = TokenStreamUtils.findAny(tokens, startIndex,
099                                TYPE, LIKE, STRUCTURE);
100                if (typeKeywordIndex == TokenStreamUtils.NOT_FOUND) {
101                        logger.debug("TYPE, LIKE or STRUCTURE expected but found "
102                                        + tokens.toString());
103                        // No type is given
104                        return new Pair<>(StringUtils.EMPTY_STRING, startIndex - 1);
105                }
106
107                // Find the beginning of the next type declaration and try to step
108                // backwards in order to find the end of the current variable type. This
109                // is done because you can put so many different expressions behind the
110                // type keyword, that it is nearly impossible to get an exhaustive list.
111                int endIndex = TokenStreamUtils.findAny(tokens, typeKeywordIndex + 1,
112                                TYPE, LIKE, STRUCTURE);
113                if (endIndex == TokenStreamUtils.NOT_FOUND) {
114                        endIndex = TokenStreamUtils.find(tokens, typeKeywordIndex, DOT);
115                        if (endIndex == TokenStreamUtils.NOT_FOUND) {
116                                endIndex = tokens.size();
117                        }
118                } else if (endIndex >= 3 && tokens.get(endIndex - 3).getType() == LPAREN
119                                && tokens.get(endIndex - 1).getType() == RPAREN) {
120                        // If the next parameter is defined using a value(var2) or similar
121                        // notation we have to skip those tokens in order to get to the
122                        // type we are looking for.
123                        endIndex -= 4;
124                } else {
125                        // Normally the TYPE keyword is preceded by a parameter name, which
126                        // we have to skip to reach the end of the previous parameter type.
127                        endIndex--;
128                }
129
130                /*
131                 * Type declarations may be ambiguous. Look at the following: I TYPE
132                 * STRUCTURE TYPE STRUCTURE. This can either mean we have 5 variables
133                 * (I, Type, STRUCTURE, ...) or (I of type STRUCTURE, TYPE and
134                 * STRUCTURE) or (I, TYPE and STRUCTURE of type STRUCTURE) and so on.
135                 * And we did not yet manage to find a reliable pattern on how the
136                 * compiler parses those declarations. Especially because leaving out
137                 * the type of a parameter is not officially documented in any ABAP
138                 * book, but it is used throughout the whole MR project we used to test
139                 * it. And the discussion is still open on how to deal with it.
140                 * Currently we agreed on ignoring cases that we are not able to parse
141                 * correctly and providing a workaround for cases where the above method
142                 * fails. The workaround is what follows.
143                 */
144                if (typeKeywordIndex < endIndex) {
145                        return new Pair<>(TokenStreamTextUtils.concatTokenTexts(
146                                        tokens.subList(typeKeywordIndex + 1, endIndex),
147                                        StringUtils.SPACE), endIndex - 1);
148                }
149                return new Pair<>(StringUtils.EMPTY_STRING,
150                                Math.max(startIndex, endIndex));
151        }

153        /**
154         * Gets type information from a list of tokens holding a method's parameter
155         * declaration part, which may either use the
156         * <code>FOR EVENT ... OF ... IMPORTING</code> or the normal sectioned
157         * parameter style.
158         */
159        public List<TypedVariable> getTypeInfoForMethodParameters(
160                        ShallowEntity entity, List<IToken> tokens, ILogger logger) {
161                if (TokenStreamUtils.startsWith(tokens, FOR, EVENT)) {
162                        return processMethodEventHandler(entity, tokens);
163                }
164                return processParameterList(entity, tokens, logger);
165        }
166
167        /**
168         * Gets type information from a list of tokens holding a method's parameter
169         * declaration part, which may contain USING, CHANGING, IMPORTING, EXPORTING
170         * and RETURNING sections.
171         * 
172         * For example <code>METHODS method IMPORTING v TYPE t value(v2) LIKE c v3
173         * CHANGING v4 TYPE t EXPORTING v5 TYPE t</code>
174         */
175        private List<TypedVariable> processParameterList(ShallowEntity entity,
176                        List<IToken> tokens, ILogger logger) {
177                List<TypedVariable> typeInfo = new ArrayList<>();
178                List<Integer> positionsOfSectionTokens = TokenStreamUtils
179                                .findAll(tokens, PARAMETER_SECTION_TOKENS);
180
181                // Tokens before first parameter section token are irrelevant because
182                // they cannot declare parameters
183                for (int i = 0; i < positionsOfSectionTokens.size(); i++) {
184                        int positionOfCurrentSectionToken = positionsOfSectionTokens.get(i);
185                        String sectionName = tokens.get(positionOfCurrentSectionToken)
186                                        .getText();
187
188                        int endOfSection;
189                        if (i < positionsOfSectionTokens.size() - 1) {
190                                endOfSection = positionsOfSectionTokens.get(i + 1);
191                        } else {
192                                endOfSection = tokens.size();
193                        }
194
195                        List<IToken> section = tokens
196                                        .subList(positionOfCurrentSectionToken + 1, endOfSection);
197                        processHeaderSection(entity, logger, typeInfo, section,
198                                        sectionName);
199                }
200
201                return typeInfo;
202        }

204        /**
205         * Processes a list of parameters that have been defined within the same
206         * parameter section. The detected parameters are added to the given list of
207         * typeInfos.
208         */
209        private void processHeaderSection(ShallowEntity containingEntity,
210                        ILogger logger, List<TypedVariable> typeInfo, List<IToken> section,
211                        String sectionName) {
212                List<String> modifiers = new ArrayList<String>();
213                modifiers.add(sectionName);
214
215                int dotIndex = TokenStreamUtils.find(section, DOT);
216                if (dotIndex != TokenStreamUtils.NOT_FOUND) {
217                        section = section.subList(0, dotIndex);
218                }
219                int currentIndex = 0;
220                while (currentIndex < section.size()) {
221                        String variableName;
222                        // If the method parameter is marked as pass-by-value with
223                        // <code>value(varName) TYPE t</code>, skip the parenthesized part
224                        if (TokenStreamUtils.tokenTypesAt(section, currentIndex + 1, LPAREN,
225                                        IDENTIFIER, RPAREN)) {
226                                variableName = section.get(currentIndex + 2).getText();
227                                currentIndex += 4;
228                        } else {
229                                variableName = section.get(currentIndex).getText();
230                                currentIndex++;
231                        }
232
233                        Pair<String, Integer> type = getNextTypeName(section, currentIndex,
234                                        logger);
235                        typeInfo.add(new TypedVariable(variableName, type.getFirst(),
236                                        modifiers, containingEntity));
237                        currentIndex = type.getSecond() + 1;
238                }
239        }

241        /**
242         * Gets type information from a list of tokens holding a method's parameter
243         * declaration part in the form of
244         * <code>([CLASS-]METHODS handler )FOR EVENT evt OF class|intf IMPORTING e1 e2 ...</code>
245         * The parameters that are returned are e1, e2, ...
246         */
247        private static List<TypedVariable> processMethodEventHandler(
248                        ShallowEntity entity, List<IToken> tokens) throws AssertionError {
249                int importingIndex = TokenStreamUtils.find(tokens, IMPORTING);
250                if (importingIndex == TokenStreamUtils.NOT_FOUND) {
251                        // Only happens in non standard conform code.
252                        return CollectionUtils.emptyList();
253                }
254
255                List<IToken> parameterTokens = tokens.subList(importingIndex + 1,
256                                tokens.size());
257                return CollectionUtils.filterAndMap(parameterTokens,
258                                token -> token.getType() != DOT,
259                                token -> new TypedVariable(token.getText(),
260                                                StringUtils.EMPTY_STRING, CollectionUtils.emptyList(),
261                                                entity));
262        }

264        /**
265         * Rolls out the colon notation in ABAP. e.g.
266         * <code>DATA: t TYPE i, a TYPE c.</code> becomes
267         * 
268         * <pre>
269         * DATA t TYPE i.
270         * DATA a TYPE c.
271         * </pre>
272         * 
273         * If no colon notation is found the given list of tokens is returned
274         * (wrapped in another list).
275         */
276        public List<List<IToken>> rolloutColonStatement(List<IToken> tokens) {
277                int colonIndex = TokenStreamUtils.find(tokens, ETokenType.COLON);
278                // If this statement does not contain a row of statements we don't have
279                // to do anything
280                if (colonIndex == TokenStreamUtils.NOT_FOUND) {
281                        return Arrays.asList(tokens);
282                }
283
284                List<IToken> startTokens = tokens.subList(0, colonIndex);
285                IToken dot = CollectionUtils.getLast(tokens);
286                List<List<IToken>> splitTokens = TokenStreamUtils.split(
287                                tokens.subList(colonIndex + 1, tokens.size() - 1), COMMA);
288
289                return CollectionUtils.map(splitTokens, fragment -> {
290                        List<IToken> newTokens = new ArrayList<>(startTokens);
291                        newTokens.addAll(fragment);
292                        newTokens.add(dot);
293                        return newTokens;
294                });
295        }

297        /**
298         * Returns the method declaration which belongs to the given method
299         * implementation entity. May return <code>null</code> if the declaration
300         * was not found, because the rootEntities given are incomplete.
301         */
302        public List<IToken> getDeclarationForMethod(
303                        List<ShallowEntity> rootEntities, ShallowEntity methodEntity) {
304                ShallowEntity classDeclaration = getClassDeclaration(rootEntities,
305                                methodEntity.getParent());
306                if (classDeclaration == null) {
307                        return null;
308                }
309                List<List<IToken>> methodDeclarations = getMethodDeclarations(
310                                classDeclaration);
311                return methodDeclarations.stream()
312                                .filter(declaration -> isMethodDeclarationFor(
313                                                methodEntity.getName(), declaration))
314                                .findFirst().orElse(null);
315        }
316
317        /**
318         * Gets the corresponding class declaration for the given class
319         * implementation. This may return <code>null</code> if the rootEntities are
320         * incomplete.
321         */
322        private static ShallowEntity getClassDeclaration(
323                        List<ShallowEntity> rootEntities,
324                        ShallowEntity classImplementation) {
325                List<ShallowEntity> typeEntities = ShallowEntityTraversalUtils
326                                .listEntitiesOfType(rootEntities, EShallowEntityType.TYPE);
327                for (ShallowEntity typeEntity : typeEntities) {
328                        if (typeEntity.getSubtype().equals(SubTypeNames.CLASS_DEFINITION)
329                                        && classImplementation.getName()
330                                                        .equals(typeEntity.getName())) {
331                                return typeEntity;
332                        }
333                }
334                return null;
335        }
336
337        /**
338         * Returns all method declarations which are defined inside the given class.
339         * Colon notations are already rolled out and removed in the returned list.
340         */
341        private List<List<IToken>> getMethodDeclarations(
342                        ShallowEntity classDeclaration) {
343                return classDeclaration.getChildrenOfType(EShallowEntityType.METHOD)
344                                .stream()
345                                .filter(method -> method.getSubtype()
346                                                .equals(SubTypeNames.METHOD_DECLARATION))
347                                .map(method -> rolloutColonStatement(method.includedTokens()))
348                                .flatMap(List::stream).collect(toList());
349        }

351        /**
352         * Returns whether the method name of the a given method declaration matches
353         * the given methodName.
354         * 
355         * The declaration looks like
356         * <code>[CLASS-]METHODS name IMPORTING ...</code>
357         */
358        private static boolean isMethodDeclarationFor(String methodName,
359                        List<IToken> declaration) {
360                CCSMPre.isTrue(declaration.size() >= 2,
361                                "Method declaration is expected to always have 2 or more tokens.");
362                return declaration.get(1).getText().equals(methodName);
363        }
