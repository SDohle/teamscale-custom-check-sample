092        /** Constructor. */
093        protected CLikeLanguageFeatureParserBase(
094                        EnumSet<ETokenType> validIdentifiers,
095                        EnumSet<ETokenType> typeTokens) {
096                this.validIdentifiers = validIdentifiers;
097                this.typeTokens = typeTokens;
098                this.noVariableSuccessorTypes = EnumSet.copyOf(validIdentifiers);
099                this.noVariableSuccessorTypes.addAll(BASIC_NO_VARIABLE_SUCCESSOR_TYPES);
100                this.useExtractor = getVariableUseExtractor();
101        }

103        /** Returns whether the given tokens are a variable declaration. */
104        public boolean isVariableDeclaration(List<IToken> variableTokens) {
105                int equalIndex = TokenStreamUtils.find(variableTokens, EQ);
106                // If there is an equals sign we assume the variable name to be in front
107                // it. Otherwise the last valid identifier will be used.
108                int nameIndex = equalIndex - 1;
109
110                if (equalIndex == TokenStreamUtils.NOT_FOUND) {
111                        nameIndex = TokenStreamUtils.findLast(variableTokens, 0,
112                                        variableTokens.size(), validIdentifiers);
113                }
114
115                return nameIndex > 0;
116
117                // TODO (LH) Can't we write this method like this:
118                // if (equalIndex == TokenStreamUtils.NOT_FOUND) {
119                // return TokenStreamUtils.findLast(variableTokens, 0,
120                // variableTokens.size(), validIdentifiers) > 0;
121                // }
122                // return equalIndex - 1 > 0;
123
124        }

126        /**
127         * Returns whether the given type name matches the given full qualified type
128         * name considering the given imported namespaces. If the given type name is
129         * null, false is returned.
130         */
131        public boolean matchesFullQualifiedTypeName(String typeName,
132                        List<String> importedNamespaces, String fullQualifiedTypeName) {
133                if (typeName == null) {
134                        return false;
135                } else if (typeName.equals(fullQualifiedTypeName)) {
136                        return true;
137                }
138
139                for (String namespace : importedNamespaces) {
140                        String potentialFullQualifiedTypeName = namespace + "." + typeName;
141                        if (potentialFullQualifiedTypeName.equals(fullQualifiedTypeName)) {
142                                return true;
143                        }
144                }
145
146                return false;
147        }

149        /** Extracts all tokens that are part of a condition from a statement. */
150        public List<IToken> extractConditionTokens(ShallowEntity statement) {
151                String subtype = statement.getSubtype();
152                if (subtype.equals(SubTypeNames.FOR)) {
153                        return TokenStreamUtils.tokensBetween(statement.ownStartTokens(),
154                                        ETokenType.SEMICOLON, ETokenType.SEMICOLON);
155                } else if (subtype.equals(SubTypeNames.DO)) {
156                        List<IToken> tokens = statement.includedTokens();
157                        int whileOffset = TokenStreamUtils.findLast(tokens,
158                                        ETokenType.WHILE);
159                        return TokenStreamUtils.tokensBetweenWithNesting(tokens,
160                                        whileOffset, ETokenType.LPAREN, ETokenType.RPAREN);
161                } else {
162                        return TokenStreamUtils.tokensBetweenWithNesting(
163                                        statement.ownStartTokens(), ETokenType.LPAREN,
164                                        ETokenType.RPAREN);
165                }
166        }

168        /** Returns whether the given entity is static. */
169        public boolean isStatic(ShallowEntity entity) {
170                return TokenStreamUtils.find(entity.ownStartTokens(),
171                                ETokenType.STATIC) != TokenStreamUtils.NOT_FOUND;
172        }

174        /**
175         * Returns whether the given entity only has static members. If there are no
176         * attribute and methods members, <code>false</code> is returned.
177         */
178        public boolean hasOnlyStaticMembers(ShallowEntity clazz) {
179                CCSMAssert.isTrue(clazz.getType() == EShallowEntityType.TYPE,
180                                "clazz.getType() must be \"TYPE\"");
181                CCSMAssert.isTrue(clazz.getSubtype().equals("class"),
182                                "clazz.getSubType() must be \"class\"");
183
184                List<ShallowEntity> children = clazz.getChildren();
185                int childCount = 0;
186                for (ShallowEntity child : children) {
187                        if (child.getType() == EShallowEntityType.ATTRIBUTE
188                                        || child.getType() == EShallowEntityType.METHOD
189                                        || child.getType() == EShallowEntityType.TYPE) {
190                                childCount++;
191                                if (!isStatic(child)) {
192                                        return false;
193                                }
194                        }
195                }
196
197                return childCount > 0;
198        }

200        /** Returns a list of all generic names for the given type. */
201        public List<String> getGenericNames(ShallowEntity type) {
202                CCSMAssert.isTrue(type.getType() == EShallowEntityType.TYPE,
203                                "type.getType() must be \"TYPE\"");
204                List<IToken> typeTokens = type.ownStartTokens();
205                // find the type's name
206                int nameIndex = TokenStreamUtils.find(typeTokens,
207                                ETokenType.IDENTIFIER);
208                if (nameIndex == TokenStreamUtils.NOT_FOUND
209                                || nameIndex >= typeTokens.size() - 1) {
210                        return CollectionUtils.emptyList();
211                }
212
213                // check if the name is followed by a '<'
214                if (typeTokens.get(nameIndex + 1).getType() != ETokenType.LT) {
215                        return CollectionUtils.emptyList();
216                }
217
218                List<IToken> genericTokens = TokenStreamUtils.tokensBetweenWithNesting(
219                                typeTokens, ETokenType.LT, ETokenType.GT);
220                List<List<IToken>> splitTokens = TokenStreamUtils.split(genericTokens,
221                                ETokenType.COMMA);
222
223                return TokenStreamTextUtils.concatAllTokenTexts(splitTokens);
224        }

226        /** Returns whether the given method has a void return type. */
227        public boolean hasVoidReturnType(ShallowEntity method) {
228                CCSMAssert.isTrue(method.getType() == EShallowEntityType.METHOD,
229                                "method.getType() must be \"METHOD\"");
230                List<IToken> methodTokens = method.ownStartTokens();
231                // if the return type is void, it is two tokens before the opening
232                // parenthesis, as it cannot be generic
233                int returnTypeIndex = TokenStreamUtils.find(methodTokens,
234                                ETokenType.LPAREN) - 2;
235                if (returnTypeIndex < 0) {
236                        return false;
237                }
238
239                return methodTokens.get(returnTypeIndex).getType().equals(VOID);
240        }

242        /** Returns a list of parameter token lists for the given method. */
243        public List<List<IToken>> getParameterTokens(ShallowEntity method) {
244                CCSMAssert.isTrue(method.getType() == EShallowEntityType.METHOD,
245                                "method.getType() must be \"METHOD\"");
246                List<IToken> parameterTokens = TokenStreamUtils
247                                .tokensBetweenWithNesting(method.ownStartTokens(),
248                                                ETokenType.LPAREN, ETokenType.RPAREN);
249                if (parameterTokens.isEmpty()) {
250                        return CollectionUtils.emptyList();
251                }
252                return splitVariableTokens(parameterTokens);
253        }

255        /**
256         * Extracts a list of modifiers and a type name from the given token list.
257         * The given tokens must have the pattern of a variable declaration. If no
258         * type can be found, an empty list and null is returned. The generic part
259         * of types is omitted.
260         */
261        public Pair<List<String>, String> getModifiersAndTypeFromTokens(
262                        List<IToken> tokens) {
263                int endIndex = TokenStreamUtils.find(tokens, EQ);
264                if (endIndex == TokenStreamUtils.NOT_FOUND) {
265                        // if there is no assignment, we search for the variable name
266                        endIndex = TokenStreamUtils.findLast(tokens, validIdentifiers);
267                } else {
268                        // otherwise we set the end index to the variable name which is in
269                        // front of the equals sign
270                        endIndex -= 1;
271                }
272
273                // skip the generic part of the type
274                if (endIndex > 1 && tokens.get(endIndex - 1).getType().equals(GT)) {
275                        endIndex = TokenStreamUtils.findMatchingOpeningToken(tokens,
276                                        endIndex - 2, LT, GT);
277                }
278
279                int typeIndex = TokenStreamUtils.findLast(tokens, 0, endIndex,
280                                typeTokens);
281                if (typeIndex == TokenStreamUtils.NOT_FOUND) {
282                        return new Pair<List<String>, String>(
283                                        CollectionUtils.<String> emptyList(), null);
284                }
285
286                typeIndex = getTypeNameStartIndex(tokens, typeIndex);
287                List<String> typeNameTokens = TokenStreamTextUtils.getTokenTexts(tokens,
288                                typeIndex, endIndex);
289                String typeName = StringUtils.concat(typeNameTokens,
290                                StringUtils.EMPTY_STRING);
291                List<String> modifiers = TokenStreamTextUtils.getTokenTexts(tokens, 0,
292                                typeIndex);
293                return new Pair<List<String>, String>(modifiers, typeName);
294        }

296        /**
297         * Return the start index of the type name that ends at the given index
298         * within the given token list. This method is used to find type names that
299         * consist of multiple tokens, e. g. names that are prefixed with a
300         * namespace.
301         */
302        protected int getTypeNameStartIndex(List<IToken> tokens,
303                        int lastTypeTokenIndex) {
304                int i = lastTypeTokenIndex;
305                while (i > 1) {
306                        ETokenType first = tokens.get(i - 1).getType();
307                        ETokenType second = tokens.get(i - 2).getType();
308
309                        if (first != DOT || !validIdentifiers.contains(second)) {
310                                break;
311                        }
312
313                        i -= 2;
314                }
315                return i;
316        }

318        /**
319         * Splits the given variable declaration tokens at commas regarding nesting
320         * between parenthesis, brackets, braces and angle brackets.
321         */
322        public List<List<IToken>> splitVariableTokens(List<IToken> variableTokens) {
323                List<List<IToken>> splitTokens = TokenStreamUtils.splitWithNesting(
324                                variableTokens, COMMA,
325                                Arrays.asList(LPAREN, LBRACK, LBRACE, LT),
326                                Arrays.asList(RPAREN, RBRACK, RBRACE, GT));
327                return splitTokens;
328        }

330        /** Returns all variable name tokens from the given variable tokens. */
331        public List<IToken> getVariableNamesFromTokens(
332                        List<IToken> variableTokens) {
333                List<List<IToken>> splitTokens = splitVariableTokens(variableTokens);
334
335                List<IToken> variableNames = new ArrayList<>();
336                for (List<IToken> tokens : splitTokens) {
337                        IToken name = getVariableNameFromTokens(tokens);
338                        if (name != null) {
339                                variableNames.add(name);
340                        }
341                }
342                return variableNames;
343        }

345        /**
346         * Returns the variable name token from the given token list or
347         * <code>null</code> if none is found. The list of tokens must have the
348         * pattern of a variable declaration.
349         */
350        public IToken getVariableNameFromTokens(List<IToken> tokens) {
351                int equalIndex = TokenStreamUtils.find(tokens, EQ);
352                // If there is an equals sign we assume the variable name to be in front
353                // it. Otherwise the last valid identifier will be used.
354                int nameIndex = equalIndex - 1;
355                if (equalIndex == TokenStreamUtils.NOT_FOUND) {
356                        nameIndex = TokenStreamUtils.findLast(tokens, 0, tokens.size(),
357                                        validIdentifiers);
358                }
359
360                if (nameIndex < 0) {
361                        return null;
362                }
363                return tokens.get(nameIndex);
364        }

366        /**
367         * Returns the variable name of an exception within a catch statement or
368         * null if the exception is not named.
369         */
370        public IToken getVariableNameFromCatchTokens(List<IToken> catchTokens) {
371                List<IToken> exceptionTokens = TokenStreamUtils.tokensBetween(
372                                catchTokens, ETokenType.LPAREN, ETokenType.RPAREN);
373
374                int doubleIdentifierIndex = TokenStreamUtils.findTokenTypeSequence(
375                                exceptionTokens, 0, ETokenType.IDENTIFIER,
376                                ETokenType.IDENTIFIER);
377                if (doubleIdentifierIndex == TokenStreamUtils.NOT_FOUND) {
378                        return null;
379                }
380                return exceptionTokens.get(doubleIdentifierIndex + 1);
381        }

383        /**
384         * Returns the type name of an exception within a catch statement or null if
385         * none is found.
386         */
387        public String getTypeNameFromCatchTokens(List<IToken> catchTokens) {
388                List<IToken> exceptionTokens = TokenStreamUtils.tokensBetween(
389                                catchTokens, ETokenType.LPAREN, ETokenType.RPAREN);
390                if (exceptionTokens.isEmpty()) {
391                        return null;
392                }
393
394                if (exceptionTokens.size() == 1) {
395                        return exceptionTokens.get(0).getText();
396                }
397
398                int typeEndIndex = TokenStreamUtils.findTokenTypeSequence(
399                                exceptionTokens, 0, ETokenType.IDENTIFIER,
400                                ETokenType.IDENTIFIER);
401                if (typeEndIndex == TokenStreamUtils.NOT_FOUND) {
402                        typeEndIndex = exceptionTokens.size() - 1;
403                }
404
405                int typeStartIndex = getTypeNameStartIndex(exceptionTokens,
406                                typeEndIndex);
407                return TokenStreamTextUtils.concatTokenTexts(
408                                exceptionTokens.subList(typeStartIndex, typeEndIndex + 1));
409        }

411        /**
412         * Returns all variable declaration tokens within a for-like statement.
413         * For-like statements are for loops and similar constructs like using and
414         * catch statements in Cs. The given end token type specifies the token
415         * where the variable declaration tokens end.
416         */
417        public List<IToken> getVariableTokensFromForLikeTokens(
418                        List<IToken> forLikeTokens, ETokenType endToken) {
419                int leftParenIndex = TokenStreamUtils.find(forLikeTokens, LPAREN);
420                if (leftParenIndex == TokenStreamUtils.NOT_FOUND) {
421                        return CollectionUtils.emptyList();
422                }
423
424                int endIndex;
425                if (endToken == RPAREN) {
426                        endIndex = TokenStreamUtils.findMatchingClosingToken(forLikeTokens,
427                                        leftParenIndex + 1, LPAREN, RPAREN);
428                } else {
429                        endIndex = TokenStreamUtils.find(forLikeTokens, endToken);
430                }
431
432                if (endIndex == TokenStreamUtils.NOT_FOUND) {
433                        return CollectionUtils.emptyList();
434                }
435
436                List<IToken> variableTokens = forLikeTokens.subList(leftParenIndex + 1,
437                                endIndex);
438                return variableTokens;
439        }

441        /**
442         * Returns all token indices from the given tokens that are uses of the
443         * variable with the given name. The heuristic is to look for all
444         * occurrences of the variable name in the given tokens. An occurrence is
445         * considered as variable usage, if the following token is not in
446         * {@link #noVariableSuccessorTypes}. Those token types are set from the
447         * subclass for a concrete language. Internally a
448         * {@link CLikeVariableUseExtractor} is used.
449         * 
450         * @param isField
451         *            if the variable is a field
452         * @param isShadowed
453         *            if the variable is a field and shadowed by another local
454         *            variable
455         */
456        public List<Integer> getVariableUsesFromTokens(List<IToken> tokens,
457                        String variableName, boolean isField, boolean isShadowed) {
458                return useExtractor.extractVariableUses(tokens, variableName, isField,
459                                isShadowed);
460        }

505        /**
506         * Returns a list of names that are imported by the given "import" entities.
507         * If the import is an aliasing import (in the case of C#) it is ignored.
508         */
509        public List<String> getImportedNames(List<ShallowEntity> importEntities) {
510                List<String> importedNames = new ArrayList<>();
511                for (ShallowEntity usingEntity : importEntities) {
512                        String importName = getImportName(usingEntity);
513                        if (importName != null) {
514                                importedNames.add(importName);
515                        }
516                }
517                return importedNames;
518        }
