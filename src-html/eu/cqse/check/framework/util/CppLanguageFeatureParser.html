057/**
058 * Language feature parser for C/C++.
059 */
060public class CppLanguageFeatureParser extends CLikeLanguageFeatureParserBase {
061
062        /** All token types that can be used to specify a type. */
063        public static final EnumSet<ETokenType> ADDITIONAL_TYPE_TOKENS = EnumSet.of(IDENTIFIER, AUTO);
064
065        /** All token types that can be used to specify a type. */
066        public static final EnumSet<ETokenType> PRIMITIVE_TYPE_TOKENS = EnumSet.of(SIGNED, UNSIGNED, BOOL, CHAR, SHORT, INT,
067                        LONG, FLOAT, DOUBLE, VOID);
068
069        /** All token types that can be used as identifier. */
070        public static final EnumSet<ETokenType> VALID_IDENTIFIERS = EnumSet.of(IDENTIFIER, ETokenType.OPERATOR);
071
072        /**
073         * All tokens that should prevent a LT to be interpreted as template (if
074         * seen before).
075         */
076        private static final Set<ETokenType> NON_GENERIC_PREFIX_TOKENS = EnumSet.of(ETokenType.LT, ETokenType.OPERATOR);
077
078        /** Constructor. */
079        public CppLanguageFeatureParser() {
080                super(ELanguage.CPP, VALID_IDENTIFIERS, PRIMITIVE_TYPE_TOKENS, ADDITIONAL_TYPE_TOKENS, SCOPE, "::");
081        }
082
083        /** {@inheritDoc} */
084        @Override
085        public boolean isImport(ShallowEntity entity) {
086                if (!(entity.getType().equals(EShallowEntityType.META) && entity.getSubtype().equals(SubTypeNames.USING))) {
087                        return false;
088                }
089                List<IToken> tokens = entity.ownStartTokens();
090                return tokens.size() >= 4 && tokens.get(1).getType().equals(NAMESPACE);
091        }
092
093        /** {@inheritDoc} */
094        @Override
095        public String getImportName(ShallowEntity entity) {
096                CCSMAssert.isTrue(isImport(entity), "entity.getType() must be equal to EShallowEntityType.META and "
097                                + "entity.getSubtype() must be equal to SubTypeNames.USING");
098                List<IToken> tokens = entity.ownStartTokens();
099                return TokenStreamTextUtils.concatTokenTexts(tokens.subList(2, tokens.size() - 1));
100        }
101
102        /** {@inheritDoc} */
103        @Override
104        public boolean hasVariableLengthArgumentList(ShallowEntity method) {
105                return TokenStreamUtils.containsAll(method.ownStartTokens(), ELLIPSIS);
106        }
107
108        /** Returns whether the given shallow entity is virtual. */
109        public boolean isVirtual(ShallowEntity entity) {
110                return TokenStreamUtils.tokenStreamContains(entity.ownStartTokens(), VIRTUAL);
111        }
112
113        /** Returns whether the given entity is static and const. */
114        public boolean isStaticConst(ShallowEntity entity) {
115                return TokenStreamUtils.containsAll(entity.ownStartTokens(), STATIC, CONST);
116        }
117
118        /** Returns whether the given entity is a class or a struct. */
119        public boolean isClassOrStruct(ShallowEntity entity) {
120                return entity.getType() == EShallowEntityType.TYPE
121                                && (SubTypeNames.CLASS.equals(entity.getSubtype()) || SubTypeNames.STRUCT.equals(entity.getSubtype()));
122        }
123
124        /** Returns whether the given entity is a top-level class or struct. */
125        public boolean isTopLevelClassOrStruct(ShallowEntity entity) {
126                ShallowEntity parent = entity.getParent();
127                boolean isTopLevel = parent == null || parent.getType().equals(EShallowEntityType.MODULE);
128                return isTopLevel && isClassOrStruct(entity);
129        }
130
131        /** {@inheritDoc} */
132        @Override
133        protected CLikeVariableUseExtractor getVariableUseExtractor() {
134                return new CLikeVariableUseExtractor(EnumSet.of(DOT, POINTERTO, SCOPE), noVariableSuccessorTypes);
135        }
136
137        /** {@inheritDoc} */
138        @Override
139        protected int reverseSkipToType(List<IToken> tokens) {
140                int endIndex = super.reverseSkipToType(tokens);
141                if (endIndex < 0) {
142                        return endIndex;
143                }
144
145                return reverseSkipScope(tokens, endIndex);
146        }
147
148        /**
149         * Skips a scope specification in front of an attribute/method. E. g. the
150         * scope specification in "int Foo::test;" is "Foo::". Returns the index
151         * pointing to the first token of the scope specification. This method is
152         * aware of template specifications within the scope specification.
153         */
154        private int reverseSkipScope(List<IToken> tokens, int endIndex) {
155                while (endIndex > 1 && tokens.get(endIndex - 1).getType() == SCOPE) {
156                        if (tokens.get(endIndex - 2).getType() == GT) {
157                                endIndex = skipGenericReverse(tokens, endIndex - 2);
158                        } else {
159                                endIndex -= 1;
160                        }
161
162                        if (endIndex <= 0 || !validIdentifiers.contains(tokens.get(endIndex - 1).getType())) {
163                                return -1;
164                        }
165                        endIndex = endIndex - 1;
166                }
167                return endIndex;
168        }
169
170        /**
171         * {@inheritDoc}
172         * 
173         * In C++ constructs like "std::vector&lt;SomeType&lt;(1&gt;2)&gt;&gt;" are
174         * allowed, thus we cannot simply skip to the matching closing token.
175         */
176        @Override
177        public int skipGeneric(List<IToken> tokens, int startIndex) {
178                int angleDepth = 1;
179                int parenDepth = 0;
180
181                for (int i = startIndex + 1; i < tokens.size(); i++) {
182                        switch (tokens.get(i).getType()) {
183                        case LPAREN:
184                                parenDepth++;
185                                break;
186                        case RPAREN:
187                                parenDepth--;
188                                break;
189                        case LT:
190                                if (parenDepth <= 0) {
191                                        angleDepth++;
192                                }
193                                break;
194                        case GT:
195                                if (parenDepth <= 0) {
196                                        angleDepth--;
197                                        if (angleDepth == 0) {
198                                                return i;
199                                        }
200                                }
201                                break;
202                        }
203                }
204
205                return -1;
206        }
207
208        /**
209         * {@inheritDoc}
210         * 
211         * In C++ constructs like "std::vector&lt;SomeType&lt;(1&gt;2)&gt;&gt;" are
212         * allowed, thus we cannot simply skip to the matching closing token.
213         */
214        @Override
215        public int skipGenericReverse(List<IToken> tokens, int startIndex) {
216                int angleDepth = 1;
217                int parenDepth = 0;
218
219                for (int i = startIndex - 1; i >= 0; i--) {
220                        switch (tokens.get(i).getType()) {
221                        case LPAREN:
222                                parenDepth--;
223                                break;
224                        case RPAREN:
225                                parenDepth++;
226                                break;
227                        case LT:
228                                if (parenDepth <= 0) {
229                                        angleDepth--;
230                                        if (angleDepth == 0) {
231                                                return i;
232                                        }
233                                }
234                                break;
235                        case GT:
236                                if (parenDepth <= 0) {
237                                        angleDepth++;
238                                }
239                                break;
240                        default:
241                                break;
242                        }
243                }
244                return -1;
245        }
246
247        /** {@inheritDoc} */
248        @Override
249        protected int getMethodOpeningParenthesisIndex(List<IToken> methodTokens) {
250                for (int i = 0; i < methodTokens.size(); i++) {
251                        switch (methodTokens.get(i).getType()) {
252                        case LT:
253                                if (i == 0 || !NON_GENERIC_PREFIX_TOKENS.contains(methodTokens.get(i - 1).getType())) {
254                                        i = skipGeneric(methodTokens, i);
255                                        if (i < 0) {
256                                                return -1;
257                                        }
258                                }
259                                break;
260                        case LPAREN:
261                                return i;
262                        }
263                }
264                return -1;
265        }
