079        /** All token types that can be used to specify a type. */

083        /** All token types that can be used as identifier. */
084        private static final EnumSet<ETokenType> VALID_IDENTIFIERS = EnumSet.of(IDENTIFIER, ETokenType.OPERATOR);

086        /**
087         * Regular expression describing a valid C/C++ identifier. It can contain
088         * alphabets, digits and underscores, but cannot start with a digit.
089         */
090        public static final Pattern VALID_IDENTIFIER_PATTERN = Pattern.compile("[\\p{Alpha}_][\\w]*");

092        /**
093         * All tokens that should prevent a LT to be interpreted as template (if seen
094         * before).
095         */
096        private static final Set<ETokenType> NON_GENERIC_PREFIX_TOKENS = EnumSet.of(ETokenType.LT, ETokenType.OPERATOR);

098        /** The names of unconditional exit statements for a switch. */
099        private static final Set<String> UNCONDITIONAL_SWITCH_EXIT_STATEMENT_NAMES = CollectionUtils.asHashSet("break",
100                        "return", "goto", "continue", "throw",
101                        // In theory, exit() can refer to a function (and not terminate the program).
102                        // In practice, exit() seems to be a standard termination function.
103                        "exit");

105        /**
106         * Maximal number of tokens that we allow an alias to grow to, to protect
107         * against too large replacements.
108         */
109        private static final int ALIAS_EXPANSION_SIZE_LIMIT = 1000;

111        /**
112         * Names of functions in the C++ standard lib that cause program termination.
113         * The functions are called via fully qualified name (e.g., std::abort(...)).
114         */
115        public static final HashSet<String> STD_LIB_PROGRAM_TERMINATION_FUNCTIONS = CollectionUtils.asHashSet("abort",
116                        "terminate", "exit", "quick_exit", "_Exit");

124        /** {@inheritDoc} */
125        @Override
126        public boolean isImport(ShallowEntity entity) {
127                if (!(entity.getType().equals(EShallowEntityType.META) && entity.getSubtype().equals(SubTypeNames.USING))) {
128                        return false;
129                }
130                List<IToken> tokens = entity.ownStartTokens();
131                return tokens.size() >= 4 && tokens.get(1).getType().equals(NAMESPACE);
132        }
133
134        /** {@inheritDoc} */
135        @Override
136        public String getImportName(ShallowEntity entity) {
137                CCSMAssert.isTrue(isImport(entity), "entity.getType() must be equal to EShallowEntityType.META and "
138                                + "entity.getSubtype() must be equal to SubTypeNames.USING");
139                List<IToken> tokens = entity.ownStartTokens();
140                return TokenStreamTextUtils.concatTokenTexts(tokens.subList(2, tokens.size() - 1));
141        }

143        /** Returns whether the given shallow entity is virtual. */
144        public static boolean isVirtual(ShallowEntity entity) {
145                return TokenStreamUtils.contains(entity.ownStartTokens(), VIRTUAL);
146        }

148        /** Returns whether the given entity is a constant */
149        public static boolean isConstant(ShallowEntity entity) {
150                List<IToken> tokens = entity.ownStartTokens();
151
152                // C++: const keyword, but no '*', as this indicates (variable) pointer
153                return TokenStreamUtils.containsAny(tokens, ETokenType.CONST, ETokenType.CONSTEXPR)
154                                && !TokenStreamUtils.containsAny(tokens, ETokenType.MULT);
155        }

157        /**
158         * A C++ method outside of a class/struct must be scoped with ::. This method
159         * can be used to distinguish between a C and a C++ method.
160         */
161        public static boolean isCppMethod(ShallowEntity entity) {
162                CCSMAssert.isTrue(entity.getType() == EShallowEntityType.METHOD, "Entity is no method");
163                if (TokenStreamUtils.contains(entity.ownStartTokens(), ETokenType.SCOPE)) {
164                        return true;
165                }
166                return isInClassOrStruct(entity);
167        }
168
169        /**
170         * Heuristic method to distinguish C from C++, first by file ending and then by
171         * characteristic keywords.
172         */
173        public static boolean isCpp(String uniformPath, List<IToken> tokens) {
174                if (uniformPath.endsWith(".cpp") || uniformPath.endsWith(".hpp")) {
175                        return true;
176                }
177                return TokenStreamUtils.containsAny(tokens, SCOPE, CLASS);
178        }

180        /**
181         * Checks if the parent of the entity is a class or a struct.
182         */
183        private static boolean isInClassOrStruct(ShallowEntity entity) {
184                ShallowEntity parent = entity.getParent();
185                if (parent == null) {
186                        return false;
187                }
188                return parent.getSubtype().equals(SubTypeNames.CLASS) || parent.getSubtype().equals(SubTypeNames.STRUCT);
189        }
190
191        /** Returns whether the given entity is a top-level class. */
192        public static boolean isTopLevelClass(ShallowEntity entity) {
193                ShallowEntity parent = entity.getParent();
194                boolean isTopLevel = parent == null || parent.getType().equals(EShallowEntityType.MODULE);
195                boolean isClass = entity.getType() == EShallowEntityType.TYPE
196                                && (SubTypeNames.CLASS.equals(entity.getSubtype()));
197                return isTopLevel && isClass;
198        }

200        @Override
201        protected int reverseSkipToType(List<IToken> tokens) {
202                int endIndex = super.reverseSkipToType(tokens);
203                if (endIndex < 0) {
204                        return endIndex;
205                }
206
207                return reverseSkipScope(tokens, endIndex);
208        }
209
210        /**
211         * Skips a scope specification in front of an attribute/method. E. g. the scope
212         * specification in "int Foo::test;" is "Foo::". Returns the index pointing to
213         * the first token of the scope specification. This method is aware of template
214         * specifications within the scope specification.
215         */
216        private int reverseSkipScope(List<IToken> tokens, int endIndex) {
217                while (endIndex > 1 && tokens.get(endIndex - 1).getType() == SCOPE) {
218                        if (tokens.get(endIndex - 2).getType() == GT) {
219                                endIndex = skipGenericReverse(tokens, endIndex - 2);
220                        } else {
221                                endIndex -= 1;
222                        }
223
224                        if (endIndex <= 0 || !validIdentifiers.contains(tokens.get(endIndex - 1).getType())) {
225                                return -1;
226                        }
227                        endIndex = endIndex - 1;
228                }
229                return endIndex;
230        }

232        /**
233         * {@inheritDoc}
234         * 
235         * In C++ constructs like "std::vector&lt;SomeType&lt;(1&gt;2)&gt;&gt;" are
236         * allowed, thus we cannot simply skip to the matching closing token.
237         */
238        @Override
239        public int skipGeneric(List<IToken> tokens, int startIndex) {
240                int angleDepth = 1;
241                int parenDepth = 0;
242
243                for (int i = startIndex + 1; i < tokens.size(); i++) {
244                        switch (tokens.get(i).getType()) {
245                        case LPAREN:
246                                parenDepth++;
247                                break;
248                        case RPAREN:
249                                parenDepth--;
250                                break;
251                        case LT:
252                                if (parenDepth <= 0) {
253                                        angleDepth++;
254                                }
255                                break;
256                        case GT:
257                                if (parenDepth <= 0) {
258                                        angleDepth--;
259                                        if (angleDepth == 0) {
260                                                return i;
261                                        }
262                                }
263                                break;
264                        default:
265                                break;
266                        }
267                }
268
269                return -1;
270        }
271
272        /**
273         * {@inheritDoc}
274         * 
275         * In C++ constructs like "std::vector&lt;SomeType&lt;(1&gt;2)&gt;&gt;" are
276         * allowed, thus we cannot simply skip to the matching closing token.
277         */
278        @Override
279        public int skipGenericReverse(List<IToken> tokens, int startIndex) {
280                int angleDepth = 1;
281                int parenDepth = 0;
282
283                for (int i = startIndex - 1; i >= 0; i--) {
284                        switch (tokens.get(i).getType()) {
285                        case LPAREN:
286                                parenDepth--;
287                                break;
288                        case RPAREN:
289                                parenDepth++;
290                                break;
291                        case LT:
292                                if (parenDepth <= 0) {
293                                        angleDepth--;
294                                        if (angleDepth == 0) {
295                                                return i;
296                                        }
297                                }
298                                break;
299                        case GT:
300                                if (parenDepth <= 0) {
301                                        angleDepth++;
302                                }
303                                break;
304                        default:
305                                break;
306                        }
307                }
308                return -1;
309        }

311        @Override
312        protected int getMethodOpeningParenthesisIndex(List<IToken> methodTokens) {
313                for (int i = 0; i < methodTokens.size(); i++) {
314                        switch (methodTokens.get(i).getType()) {
315                        case LT:
316                                if (i == 0 || !NON_GENERIC_PREFIX_TOKENS.contains(methodTokens.get(i - 1).getType())) {
317                                        i = skipGeneric(methodTokens, i);
318                                        if (i < 0) {
319                                                return -1;
320                                        }
321                                }
322                                break;
323                        case LPAREN:
324                                return i;
325                        default:
326                                break;
327                        }
328                }
329                return -1;
330        }

338        /**
339         * Entity has to be META and its SubTypName <i>case</i> or <i>default</i> to
340         * return true
341         */
342        public static boolean isCaseOrDefault(ShallowEntity entity) {
343                return entity.getType().equals(META)
344                                && (entity.getSubtype().equals(CASE) || entity.getSubtype().equals(DEFAULT));
345        }

347        /**
348         * Returns whether the given entity is a valid exit statement for a case
349         * statement (break, goto, return, continue, throw, std::abort, std::terminate,
350         * std::exit).
351         */
352        public static boolean isCaseOrDefaultExitStatement(ShallowEntity entity) {
353                if (entity.getType().equals(STATEMENT) && entity.getSubtype().equals(SIMPLE_STATEMENT)) {
354                        if (UNCONDITIONAL_SWITCH_EXIT_STATEMENT_NAMES.contains(entity.getName())) {
355                                return true;
356                        }
357                        return TokenStreamUtils.hasTokenTypeSequence(entity.ownStartTokens(), 0, IDENTIFIER, SCOPE, IDENTIFIER)
358                                        && "std".equals(entity.ownStartTokens().get(0).getText())
359                                        && STD_LIB_PROGRAM_TERMINATION_FUNCTIONS.contains(entity.ownStartTokens().get(2).getText());
360
361                }
362                return false;
363        }

365        /**
366         * Returns whether the given entity is a m unconditional exit statement for a
367         * case statement. This checks
368         * {@link #isCaseOrDefaultExitStatement(ShallowEntity)}, but also recurses in
369         * case of blocks and conditionals.
370         * 
371         * @param lastEntity
372         *            the last entity of the block that should be exited
373         *            unconditionally.
374         */
375        public static boolean isUnconditionalSwitchExit(ShallowEntity lastEntity) {
376                if (lastEntity.getSubtype().equals(SubTypeNames.ANONYMOUS_BLOCK)) {
377                        return isUnconditionalSwitchExit(lastEntity.getChildren());
378                }
379
380                if (lastEntity.getSubtype().equals(SubTypeNames.ELSE)) {
381                        List<ShallowEntity> siblings = lastEntity.getParent().getChildren();
382                        int index = siblings.indexOf(lastEntity);
383
384                        // check the "else"
385                        if (!isUnconditionalSwitchExit(lastEntity.getChildren())) {
386                                return false;
387                        }
388                        index -= 1;
389
390                        // check any else-if
391                        while (index > 0 && SubTypeNames.ELSE_IF.equals(siblings.get(index).getSubtype())) {
392                                if (!isUnconditionalSwitchExit(siblings.get(index).getChildren())) {
393                                        return false;
394                                }
395                                index -= 1;
396                        }
397
398                        // check the "if"
399                        return isUnconditionalSwitchExit(siblings.get(index).getChildren());
400                }
401
402                return isCaseOrDefaultExitStatement(lastEntity);
403        }
404
405        private static boolean isUnconditionalSwitchExit(List<ShallowEntity> entities) {
406                return !entities.isEmpty() && isUnconditionalSwitchExit(CollectionUtils.getLast(entities));
407        }

409        /**
410         * Extracts a mapping of types regarding the typedef instructions found in the
411         * given tokens. The returned aliases will already be fully expanded, i.e. not
412         * recursive expansion is needed anymore.
413         * 
414         * @param tokens
415         *            the tokens that may contain typedefs
416         * @return a mapping between the aliases (identifiers) and the tokens that get
417         *         replaced by the aliases
418         */
419        public static Map<String, List<IToken>> getAliasMapping(List<IToken> tokens) {
420                Map<String, List<IToken>> aliases = new HashMap<>();
421
422                for (int i = 0; i < tokens.size(); i++) {
423                        IToken current = tokens.get(i);
424                        ETokenType currentType = current.getType();
425                        if (currentType == ETokenType.TYPEDEF || currentType == ETokenType.USING) {
426                                int semicolonIndex = TokenStreamUtils.findFirstTopLevel(tokens, i + 1, EnumSet.of(ETokenType.SEMICOLON),
427                                                Arrays.asList(ETokenType.LBRACE), Arrays.asList(ETokenType.RBRACE));
428                                if (semicolonIndex == TokenStreamUtils.NOT_FOUND) {
429                                        continue;
430                                }
431
432                                List<IToken> tokenBuffer = tokens.subList(i + 1, semicolonIndex);
433                                i = semicolonIndex;
434
435                                if (currentType == ETokenType.TYPEDEF && isTypedefUsedForAliasResolution(tokenBuffer)) {
436                                        IToken lastToken = tokenBuffer.get(tokenBuffer.size() - 1);
437                                        insertAlias(lastToken.getText(), tokenBuffer.subList(0, tokenBuffer.size() - 1), aliases);
438                                } else if (currentType == ETokenType.USING
439                                                && TokenStreamUtils.startsWith(tokenBuffer, ETokenType.IDENTIFIER, ETokenType.EQ)) {
440                                        insertAlias(tokenBuffer.get(0).getText(), tokenBuffer.subList(2, tokenBuffer.size()), aliases);
441                                }
442                        }
443                }
444                return aliases;
445        }
446
447        /**
448         * Returns whether the given tokens from a typeef should be used for alias
449         * resolution. We explicitly skip typedefs with function types (as there is no
450         * clear replacement) and containing inline union/struct types. This is
451         * identified by checking for braces and parentheses.
452         */
453        private static boolean isTypedefUsedForAliasResolution(List<IToken> tokens) {
454                return !TokenStreamUtils.containsAny(tokens, ETokenType.LBRACE, ETokenType.LPAREN);
455        }

457        /**
458         * Inserts a new alias, first resolving the tokens of the alias. This respects
459         * the expansion limit set in {@link #ALIAS_EXPANSION_SIZE_LIMIT}.
460         */
461        private static void insertAlias(String name, List<IToken> tokens, Map<String, List<IToken>> aliases) {
462                List<IToken> resolved = resolveAliases(tokens, aliases);
463                if (resolved.size() > ALIAS_EXPANSION_SIZE_LIMIT) {
464                        LOGGER.warn("Skipping alias for " + name + " as resolved replacement would exceed "
465                                        + ALIAS_EXPANSION_SIZE_LIMIT + " tokens in file " + CollectionUtils.getAny(tokens).getOriginId());
466                } else {
467                        aliases.put(name, resolved);
468                }
469        }

471        /**
472         * Replaces the aliases introduced by typedefs and usings with their respective
473         * original type.
474         * 
475         * @param tokens
476         *            The tokens in which the aliases are resolved
477         * 
478         * @param aliasMapping
479         *            The resolved alias returned by @{@link #getAliasMapping(List)}.
480         */
481        public static List<IToken> resolveAliases(List<IToken> tokens, Map<String, List<IToken>> aliasMapping) {
482                List<IToken> resolved = new ArrayList<>();
483                for (IToken token : tokens) {
484                        if (token.getType() == IDENTIFIER && aliasMapping.containsKey(token.getText())) {
485                                resolved.addAll(aliasMapping.get(token.getText()));
486                        } else {
487                                resolved.add(token);
488                        }
489                }
490                return resolved;
491        }
