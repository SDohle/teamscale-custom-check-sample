033/** Language feature parser for Objective-C. */
034public class ObjectiveCLanguageFeatureParser extends CLikeLanguageFeatureParserBase {
035
036        /** All token types that can be used to specify a type. */
037        private static final EnumSet<ETokenType> ADDITIONAL_TYPE_TOKENS = EnumSet.of(IDENTIFIER, ID, AUTO);
038
039        /** All token types that can be used to specify a type. */
040        private static final EnumSet<ETokenType> PRIMITIVE_TYPE_TOKENS = EnumSet.of(SIGNED, UNSIGNED, BOOL, CHAR, SHORT,
041                        INT, LONG, FLOAT, DOUBLE, VOID);
042
043        /** All token types that can be used as identifier. */
044        private static final EnumSet<ETokenType> VALID_IDENTIFIERS = EnumSet.of(IDENTIFIER, ETokenType.OPERATOR);
045
046        /** Names of classes that count as generic exceptions. */
047        private static final Set<String> GENERIC_EXCEPTION_NAMES = CollectionUtils.asHashSet("NSException", "id");
048
049        ObjectiveCLanguageFeatureParser() {
050                super(ELanguage.OBJECTIVE_C, VALID_IDENTIFIERS, PRIMITIVE_TYPE_TOKENS, ADDITIONAL_TYPE_TOKENS, SCOPE, ".",
051                                new CLikeVariableUseExtractor(EnumSet.of(DOT), VALID_IDENTIFIERS));
052        }
053
054        @Override
055        public boolean isImport(ShallowEntity entity) {
056                return entity.getType() == EShallowEntityType.META && entity.getSubtype().equals(SubTypeNames.AT_IMPORT);
057        }
058
059        @Override
060        public String getImportName(ShallowEntity entity) {
061                return TokenStreamTextUtils.concatTokenTexts(
062                                TokenStreamUtils.tokensBetween(entity.ownStartTokens(), ETokenType.IMPORT, ETokenType.SEMICOLON));
063        }
064
065        /**
066         * Determines whether the given class name represents a generic exception.
067         */
068        public boolean isGenericExceptionClass(String className) {
069                return GENERIC_EXCEPTION_NAMES.contains(className);
070        }
071}
