017package eu.cqse.check.framework.util;
018
019import java.util.EnumSet;
020
021import org.conqat.lib.commons.string.StringUtils;
022
023import eu.cqse.check.framework.scanner.ETokenType;
024import eu.cqse.check.framework.scanner.IToken;
025
026/**
027 * Utility methods for {@link IToken}s.
028 * 
029 * @author $Author: streitel $
030 * @version $Rev: 55859 $
031 * @ConQAT.Rating GREEN Hash: A64B93183ED6AFAB83A22B13E486A78B
032 */
033public class TokenUtils {
034
035        /**
036         * Set of token types that always have to be treated case-sensitive, e.g.
037         * String literals.
038         */
039        public static final EnumSet<ETokenType> ALWAYS_CASE_SENSITIVE_TOKEN_TYPES = EnumSet
040                        .of(ETokenType.STRING_LITERAL, ETokenType.CHARACTER_LITERAL,
041                                        ETokenType.UNTERMINATED_STRING_LITERAL,
042                                        ETokenType.UNTERMINATED_CHARACTER_LITERAL);
043
044        /**
045         * Estimates the zero-based end line of the specified token by looking where
046         * the provided next token begins. Note that this cannot check, if the
047         * provided next token is really the next token, nor if there are blank
048         * lines between the two tokens.
049         * <p>
050         * This method is more efficient than
051         * {@link #calculateEndLineByCountingLines(IToken)}, but not as precise.
052         * 
053         * @param token
054         *            the token whose end line should be calculated.
055         * @param nextToken
056         *            the lookahead token.
057         * @return the end line of the specified first token.
058         */
059        public static int estimateEndLineByLookahead(IToken token,
060                        IToken nextToken) {
061                int tokenEndLine = nextToken.getLineNumber() - 1;
062                return Math.max(tokenEndLine, token.getLineNumber());
063        }
064
065        /**
066         * Calculates the zero-based end line of the specified token by counting the
067         * lines of the text contained within.
068         * <p>
069         * This method is less efficient than
070         * {@link #estimateEndLineByLookahead(IToken, IToken)}, but always yields
071         * the correct result.
072         * 
073         * @param token
074         *            the token whose end line should be calculated.
075         * @return the end line of the specified token.
076         */
077        public static int calculateEndLineByCountingLines(IToken token) {
078                int tokenEndLine = token.getLineNumber()
079                                + StringUtils.countLines(token.getText()) - 1;
080                return Math.max(tokenEndLine, token.getLineNumber());
081        }
082}
