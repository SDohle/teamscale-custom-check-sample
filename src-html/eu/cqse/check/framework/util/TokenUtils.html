017package eu.cqse.check.framework.util;
018
019import java.util.EnumSet;
020import java.util.Objects;
021
022import org.conqat.lib.commons.string.StringUtils;
023
024import eu.cqse.check.framework.scanner.ETokenType;
025import eu.cqse.check.framework.scanner.IToken;
026
027/**
028 * Utility methods for {@link IToken}s.
029 */
030public class TokenUtils {
031
032        /**
033         * Set of token types that always have to be treated case-sensitive, e.g.
034         * String literals.
035         */
036        public static final EnumSet<ETokenType> ALWAYS_CASE_SENSITIVE_TOKEN_TYPES = EnumSet.of(ETokenType.STRING_LITERAL,
037                        ETokenType.CHARACTER_LITERAL, ETokenType.UNTERMINATED_STRING_LITERAL,
038                        ETokenType.UNTERMINATED_CHARACTER_LITERAL);
039
040        /**
041         * Estimates the zero-based end line of the specified token by looking where
042         * the provided next token begins. Note that this cannot check, if the
043         * provided next token is really the next token, nor if there are blank
044         * lines between the two tokens.
045         * <p>
046         * This method is more efficient than
047         * {@link #calculateEndLineByCountingLines(IToken)}, but not as precise.
048         * 
049         * @param token
050         *            the token whose end line should be calculated.
051         * @param nextToken
052         *            the lookahead token.
053         * @return the end line of the specified first token.
054         */
055        public static int estimateEndLineByLookahead(IToken token, IToken nextToken) {
056                int tokenEndLine = nextToken.getLineNumber() - 1;
057                return Math.max(tokenEndLine, token.getLineNumber());
058        }
059
060        /**
061         * Calculates the zero-based end line of the specified token by counting the
062         * lines of the text contained within.
063         * <p>
064         * This method is less efficient than
065         * {@link #estimateEndLineByLookahead(IToken, IToken)}, but always yields
066         * the correct result.
067         * 
068         * @param token
069         *            the token whose end line should be calculated.
070         * @return the end line of the specified token.
071         */
072        public static int calculateEndLineByCountingLines(IToken token) {
073                int tokenEndLine = token.getLineNumber() + StringUtils.countLines(token.getText()) - 1;
074                return Math.max(tokenEndLine, token.getLineNumber());
075        }
076
077        /**
078         * Checks if the given tokens are of equal type and have the equal text. The
079         * method is <code>null</code>-safe.
080         * 
081         * @return <code>true</code> if both tokens are <code>null</code> or if both
082         *         tokens are of the same {@link ETokenType} and have the same text,
083         *         <code>false</code> otherwise.
084         */
085        public static boolean isEqualTypeAndText(IToken t1, IToken t2) {
086                if (t1 == t2) {
087                        return true;
088                }
089                if (t1 == null || t2 == null) {
090                        return false;
091                }
092                return t1.getType() == t2.getType() && Objects.equals(t1.getText(), t2.getText());
093        }
094}
