061/**
062 * Language feature parser for ABAP.
063 *
064 * Since ABAP is case-insensitive, all variables and types are lower-cased.
065 */
066public class AbapLanguageFeatureParser implements ILanguageFeatureParser {

068        /**
069         * EnumSet containing all token types which can appear as section headers in
070         * a parameter declaration.
071         */
072        public static final EnumSet<ETokenType> PARAMETER_SECTION_TOKENS = EnumSet.of(USING, CHANGING, IMPORTING, EXPORTING,
073                        RETURNING, TABLES, TESTING, RAISING);
074
075        /** {@inheritDoc} */
076        @Override
077        public ELanguage getLanguage() {
078                return ELanguage.ABAP;
079        }

081        /**
082         * Gets the type name from the given list of tokens.
083         * 
084         * @param tokens
085         *            List of tokens. For example v1 TYPE t value(v2) LIKE c v TYPE
086         *            t v3.
087         * 
088         * @return Returns a tuple of the type name and the corresponding index of
089         *         the last type token. The index is ensured to be bigger or equal
090         *         to startIndex-1. If no further type is found, ("", startIndex-1)
091         *         is returned.
092         */
093        public Pair<String, Integer> getNextTypeName(List<IToken> tokens, int startIndex, ILogger logger) {
094                int typeKeywordIndex = TokenStreamUtils.firstTokenOfType(tokens, startIndex, TYPE, LIKE, STRUCTURE);
095                if (typeKeywordIndex == TokenStreamUtils.NOT_FOUND) {
096                        logger.debug("TYPE, LIKE or STRUCTURE expected but found " + tokens.toString());
097                        // No type is given
098                        return new Pair<>(StringUtils.EMPTY_STRING, startIndex - 1);
099                }
100
101                // Find the beginning of the next type declaration and try to step
102                // backwards in order to find the end of the current variable type. This
103                // is done because you can put so many different expressions behind the
104                // type keyword, that it is nearly impossible to get an exhaustive list.
105                int endIndex = TokenStreamUtils.firstTokenOfType(tokens, typeKeywordIndex + 1, TYPE, LIKE, STRUCTURE);
106                if (endIndex == TokenStreamUtils.NOT_FOUND) {
107                        endIndex = TokenStreamUtils.firstTokenOfType(tokens, typeKeywordIndex, DOT);
108                        if (endIndex == TokenStreamUtils.NOT_FOUND) {
109                                endIndex = tokens.size();
110                        }
111                } else if (endIndex >= 3 && tokens.get(endIndex - 3).getType() == LPAREN
112                                && tokens.get(endIndex - 1).getType() == RPAREN) {
113                        // If the next parameter is defined using a value(var2) or similar
114                        // notation we have to skip those tokens in order to get to the
115                        // type we are looking for.
116                        endIndex -= 4;
117                } else {
118                        // Normally the TYPE keyword is preceded by a parameter name, which
119                        // we have to skip to reach the end of the previous parameter type.
120                        endIndex--;
121                }
122
123                /*
124                 * Type declarations may be ambiguous. Look at the following: I TYPE
125                 * STRUCTURE TYPE STRUCTURE. This can either mean we have 5 variables
126                 * (I, Type, STRUCTURE, ...) or (I of type STRUCTURE, TYPE and
127                 * STRUCTURE) or (I, TYPE and STRUCTURE of type STRUCTURE) and so on.
128                 * And we did not yet manage to find a reliable pattern on how the
129                 * compiler parses those declarations. Especially because leaving out
130                 * the type of a parameter is not officially documented in any ABAP
131                 * book, but it is used throughout the whole MR project we used to test
132                 * it. And the discussion is still open on how to deal with it.
133                 * Currently we agreed on ignoring cases that we are not able to parse
134                 * correctly and providing a workaround for cases where the above method
135                 * fails. The workaround is what follows.
136                 */
137                if (typeKeywordIndex < endIndex) {
138                        return new Pair<>(TokenStreamTextUtils
139                                        .concatTokenTexts(tokens.subList(typeKeywordIndex + 1, endIndex), StringUtils.SPACE).toLowerCase(),
140                                        endIndex - 1);
141                }
142                return new Pair<>(StringUtils.EMPTY_STRING, Math.max(startIndex, endIndex));
143        }

145        /**
146         * Gets type information from a list of tokens holding a method's parameter
147         * declaration part, which may either use the
148         * <code>FOR EVENT ... OF ... IMPORTING</code> or the normal sectioned
149         * parameter style.
150         */
151        public List<TypedVariable> getTypeInfoForMethodParameters(ShallowEntity entity, List<IToken> tokens,
152                        ILogger logger) {
153                if (TokenStreamUtils.startsWith(tokens, FOR, EVENT)) {
154                        return processMethodEventHandler(entity, tokens);
155                }
156                return processParameterList(entity, tokens, logger);
157        }
158
159        /**
160         * Gets type information from a list of tokens holding a method's parameter
161         * declaration part, which may contain USING, CHANGING, IMPORTING, EXPORTING
162         * and RETURNING sections.
163         * 
164         * For example <code>METHODS method IMPORTING v TYPE t value(v2) LIKE c v3
165         * CHANGING v4 TYPE t EXPORTING v5 TYPE t</code>
166         */
167        private List<TypedVariable> processParameterList(ShallowEntity entity, List<IToken> tokens, ILogger logger) {
168                List<TypedVariable> typeInfo = new ArrayList<>();
169                List<Integer> positionsOfSectionTokens = TokenStreamUtils.findAll(tokens, PARAMETER_SECTION_TOKENS);
170
171                // Tokens before first parameter section token are irrelevant because
172                // they cannot declare parameters
173                for (int i = 0; i < positionsOfSectionTokens.size(); i++) {
174                        int positionOfCurrentSectionToken = positionsOfSectionTokens.get(i);
175                        IToken sectionName = tokens.get(positionOfCurrentSectionToken);
176
177                        int endOfSection;
178                        if (i < positionsOfSectionTokens.size() - 1) {
179                                endOfSection = positionsOfSectionTokens.get(i + 1);
180                        } else {
181                                endOfSection = tokens.size();
182                        }
183
184                        List<IToken> section = tokens.subList(positionOfCurrentSectionToken + 1, endOfSection);
185                        processHeaderSection(entity, logger, typeInfo, section, sectionName);
186                }
187
188                return typeInfo;
189        }
190
191        /**
192         * Processes a list of parameters that have been defined within the same
193         * parameter section. The detected parameters are added to the given list of
194         * typeInfos.
195         */
196        private void processHeaderSection(ShallowEntity containingEntity, ILogger logger, List<TypedVariable> typeInfo,
197                        List<IToken> section, IToken sectionName) {
198                List<IToken> modifiers = new ArrayList<>();
199                modifiers.add(sectionName);
200
201                int dotIndex = TokenStreamUtils.firstTokenOfType(section, DOT);
202                if (dotIndex != TokenStreamUtils.NOT_FOUND) {
203                        section = section.subList(0, dotIndex);
204                }
205                int currentIndex = 0;
206                while (currentIndex < section.size()) {
207                        String variableName;
208                        // If the method parameter is marked as pass-by-value with
209                        // <code>value(varName) TYPE t</code>, skip the parenthesized part
210                        if (TokenStreamUtils.tokenTypesAt(section, currentIndex + 1, LPAREN, IDENTIFIER, RPAREN)) {
211                                variableName = section.get(currentIndex + 2).getText();
212                                currentIndex += 4;
213                        } else {
214                                variableName = section.get(currentIndex).getText();
215                                currentIndex++;
216                        }
217
218                        Pair<String, Integer> type = getNextTypeName(section, currentIndex, logger);
219                        typeInfo.add(new TypedVariable(AbapCheckUtils.normalizeVariableName(variableName),
220                                        type.getFirst().toLowerCase(), modifiers, containingEntity));
221                        currentIndex = type.getSecond() + 1;
222                }
223        }

225        /**
226         * Gets type information from a list of tokens holding a method's parameter
227         * declaration part in the form of
228         * <code>([CLASS-]METHODS handler )FOR EVENT evt OF class|intf IMPORTING e1 e2 ...</code>
229         * The parameters that are returned are e1, e2, ...
230         */
231        private static List<TypedVariable> processMethodEventHandler(ShallowEntity entity, List<IToken> tokens)
232                        throws AssertionError {
233                int importingIndex = TokenStreamUtils.firstTokenOfType(tokens, IMPORTING);
234                if (importingIndex == TokenStreamUtils.NOT_FOUND) {
235                        // Only happens in non standard conform code.
236                        return CollectionUtils.emptyList();
237                }
238
239                List<IToken> parameterTokens = tokens.subList(importingIndex + 1, tokens.size());
240                return CollectionUtils.filterAndMap(parameterTokens, token -> token.getType() != DOT,
241                                token -> new TypedVariable(AbapCheckUtils.normalizeVariableName(token.getText()),
242                                                StringUtils.EMPTY_STRING, CollectionUtils.emptyList(), entity));
243        }

245        /**
246         * Returns the method declaration which belongs to the given method
247         * implementation entity. May return <code>null</code> if the declaration
248         * was not found, because the rootEntities given are incomplete.
249         */
250        public List<IToken> getDeclarationForMethod(List<ShallowEntity> rootEntities, ShallowEntity methodEntity) {
251                ShallowEntity classDeclaration = getClassDeclaration(rootEntities, methodEntity.getParent());
252                if (classDeclaration == null) {
253                        return null;
254                }
255                List<List<IToken>> methodDeclarations = getMethodDeclarations(classDeclaration);
256                return methodDeclarations.stream()
257                                .filter(declaration -> isMethodDeclarationFor(methodEntity.getName(), declaration)).findFirst()
258                                .orElse(null);
259        }
260
261        /**
262         * Gets the corresponding class declaration for the given class
263         * implementation. This may return <code>null</code> if the rootEntities are
264         * incomplete.
265         */
266        private static ShallowEntity getClassDeclaration(List<ShallowEntity> rootEntities,
267                        ShallowEntity classImplementation) {
268                List<ShallowEntity> typeEntities = ShallowEntityTraversalUtils.listEntitiesOfType(rootEntities,
269                                EShallowEntityType.TYPE);
270                for (ShallowEntity typeEntity : typeEntities) {
271                        if (typeEntity.getSubtype().equals(SubTypeNames.CLASS_DEFINITION)
272                                        && classImplementation.getName().equals(typeEntity.getName())) {
273                                return typeEntity;
274                        }
275                }
276                return null;
277        }
278
279        /**
280         * Returns all method declarations which are defined inside the given class.
281         * Colon notations are already rolled out and removed in the returned list.
282         */
283        private static List<List<IToken>> getMethodDeclarations(ShallowEntity classDeclaration) {
284                return CollectionUtils.filterAndMap(classDeclaration.getChildrenOfType(EShallowEntityType.METHOD),
285                                method -> method.getSubtype().equals(SubTypeNames.METHOD_DECLARATION), ShallowEntity::includedTokens);
286        }
287
288        /**
289         * Returns whether the method name of the a given method declaration matches
290         * the given methodName.
291         * 
292         * The declaration looks like
293         * <code>[CLASS-]METHODS name IMPORTING ...</code>
294         */
295        private static boolean isMethodDeclarationFor(String methodName, List<IToken> declaration) {
296                CCSMAssert.isTrue(declaration.size() >= 2, "Method declaration is expected to always have 2 or more tokens.");
297                return declaration.get(1).getText().equalsIgnoreCase(methodName);
298        }

300        /**
301         * Parses a function call string.
302         * 
303         * @return {@link Optional}, empty if the given entity is not a CALL
304         *         FUNCTION statement otherwise containing a
305         *         {@link FunctionCallInfo}
306         * @throws CheckException
307         *             if the given statement can not be parsed
308         */
309        public Optional<FunctionCallInfo> getFunctionCallInfo(ShallowEntity callFunctionStatement) throws CheckException {
310                if (callFunctionStatement.getType() != EShallowEntityType.STATEMENT) {
311                        return Optional.empty();
312                }
313                List<IToken> tokens = callFunctionStatement.ownStartTokens();
314                if (tokens.size() < 3 || !TokenStreamUtils.startsWith(tokens, ETokenType.CALL, ETokenType.FUNCTION)) {
315                        return Optional.empty();
316                }
317                return Optional.of(new FunctionCallInfo(tokens));
318        }
