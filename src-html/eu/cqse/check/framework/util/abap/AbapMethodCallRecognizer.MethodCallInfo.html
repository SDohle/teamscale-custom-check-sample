043/**
044 * Tries to parse ABAP Method/Function calls
045 */
046public class AbapMethodCallRecognizer {
047
048        /** Token types that start a section in a parameter list */
049        final EnumSet<ETokenType> PARAMETER_TOKEN_TYPES = EnumSet.of(EXPORTING, IMPORTING, RECEIVING, CHANGING, EXCEPTIONS);
050
051        /** Tokens that terminate a section of a parameter list. */
052        private static final EnumSet<ETokenType> PARAMETER_TERMINATING_TOKENS = EnumSet.of(EXPORTING, IMPORTING, RECEIVING,
053                        CHANGING, EXCEPTIONS, EQ, DOT, RPAREN);
054
055        /** The tokens that get parsed */
056        private List<IToken> tokens;
057        /** Current position of the parser in the token list */
058        private int currentPosition = 0;
059        /** Details on the method call. Gets filled during the parse process. */
060        private MethodCallInfo details = new MethodCallInfo();
061
062        /** Constructor. */
063        private AbapMethodCallRecognizer(List<IToken> tokens) {
064                this.tokens = tokens;
065        }
066
067        /**
068         * Start parsing
069         */
070        public static Optional<MethodCallInfo> parse(List<IToken> tokens) {
071                AbapMethodCallRecognizer recognizer = new AbapMethodCallRecognizer(tokens);
072                return recognizer.parseMethodCall();
073        }
074
075        /** Parses the method call. */
076        private Optional<MethodCallInfo> parseMethodCall() {
077                if (tokens.isEmpty() || tokens.size() < 2) {
078                        return Optional.empty();
079                }
080                if (isTokenType(currentPosition, PERFORM)) {
081                        // PERFORM is the 'CALL' keyword for FORMs
082                        return parsePerformCall(tokens);
083                        // we should distinguish between FORM calls and METHOD calls in the
084                        // calldetails at some point.
085                }
086                if (EnumSet.of(ARROW, EQ).contains(tokens.get(currentPosition + 1).getType())) {
087                        if (!parseAssignee()) {
088                                return Optional.empty();
089                        }
090                }
091                // CALL and METHOD are optional
092                skipOptionalTokens(CALL, METHOD);
093                if (tokens.size() <= currentPosition || !isTokenType(currentPosition, IDENTIFIER)) {
094                        return Optional.empty();
095                }
096                if (!parseMethodName()) {
097                        return Optional.empty();
098                }
099                if (details.methodName.equalsIgnoreCase("CAST") || details.methodName.equalsIgnoreCase("CONV")) {
100                        // this is an ABAP cast or type conversion (looks like a function
101                        // call, but is none)
102                        return Optional.empty();
103                }
104                int positionBeforeParameters = currentPosition;
105                // without checking whether parseParameters actually advances, "x=y"
106                // would be recognized as method call
107                if (!parseParameters() || currentPosition == positionBeforeParameters) {
108                        return Optional.empty();
109                }
110                if (!isTokenType(currentPosition, DOT)) {
111                        return Optional.empty();
112                }
113                return Optional.of(details);
114        }
115
116        /**
117         * Advance the current position by the given distance. Returns whether the
118         * new position is still in the range of the token list.
119         */
120        private boolean advance(int distance) {
121                currentPosition += distance;
122                return currentPosition < tokens.size();
123        }
124
125        /**
126         * Skips tokens at the currentPosition while the current token is of the
127         * given types.
128         */
129        private void skipOptionalTokens(ETokenType optionalType, ETokenType... optionalTypesRest) {
130                EnumSet<ETokenType> types = EnumSet.of(optionalType, optionalTypesRest);
131                while (types.contains(tokens.get(currentPosition).getType())) {
132                        if (!advance(1)) {
133                                return;
134                        }
135                }
136        }
137
138        /**
139         * Parses the name of the called method
140         */
141        private boolean parseMethodName() {
142                if (isTokenType(currentPosition + 1, EQGT)) {
143                        details.methodContainerName = tokens.get(currentPosition).getText();
144                        details.isStaticCall = true;
145                        details.methodName = tokens.get(currentPosition + 2).getText();
146                        if (!advance(3)) {
147                                return false;
148                        }
149                } else if (isTokenType(currentPosition + 1, ARROW)) {
150                        details.methodContainerName = "";
151                        details.isStaticCall = false;
152                        while (isTokenType(currentPosition + 1, ARROW)) {
153                                details.methodContainerName += tokens.get(currentPosition).getText();
154                                if (!advance(2)) {
155                                        return false;
156                                }
157                                if (isTokenType(currentPosition + 1, ARROW)) {
158                                        details.methodContainerName += "->";
159                                }
160                        }
161                        details.methodName = tokens.get(currentPosition).getText();
162                        if (!advance(1)) {
163                                return false;
164                        }
165                } else {
166                        details.methodName = tokens.get(currentPosition).getText();
167                        if (!advance(1)) {
168                                return false;
169                        }
170                }
171                return true;
172        }
173
174        /**
175         * Returns whether the token at the given position is of the given type.
176         */
177        private boolean isTokenType(int position, ETokenType type) {
178                CCSMAssert.isTrue(tokens.size() > position,
179                                "Out of bounds of the token list (" + position + " >= " + tokens.size() + ")");
180                return tokens.get(position).getType() == type;
181        }
182
183        /**
184         * Tries to parse a parameter list of a method call. Returns false if the
185         * parsing failed.
186         */
187        private boolean parseParameters() {
188                if (!isTokenType(currentPosition, LPAREN)) {
189                        if (!parseParameterDeclarations(details)) {
190                                return false;
191                        }
192                        return true;
193                }
194                // functional call
195                // can be: meth(), meth(f), meth (p=f), or meth (p1=f1 ... pn=fn)
196                if (!advance(1)) {// skip the LPAREN "("
197                        return false;
198                }
199                if (skipMandatoryToken(RPAREN)) {
200                        // meth(), we are done
201                        return true;
202                } else if (isTokenType(currentPosition, IDENTIFIER) && isTokenType(currentPosition + 1, RPAREN)) {
203                        // just one unnamed parameter meth(f)
204                        details.unnamedParameters.add(tokens.get(currentPosition).getText());
205                        if (!advance(2)) {
206                                return false;
207                        }
208                } else if (isTokenType(currentPosition + 1, EQ)) {
209                        // meth (p=f), or meth (p1=f1 ... pn=fn)
210                        parseParameterListInto(details.namedParameters);
211                        if (!skipMandatoryToken(RPAREN)) {
212                                return false;
213                        }
214                } else {
215                        if (!parseParameterDeclarations(details)) {
216                                return false;
217                        }
218                        if (!skipMandatoryToken(RPAREN)) {
219                                return false;
220                        }
221                }
222                return true;
223        }
224
225        /**
226         * Skips the next token if it is of the given type. Returns whether the type
227         * was as expected.
228         */
229        private boolean skipMandatoryToken(ETokenType expectedType) {
230                // check token type and skip if it is correct
231                if (!isTokenType(currentPosition, expectedType) || !advance(1)) {
232                        return false;
233                }
234                return true;
235        }
236
237        /**
238         * Parses the name of an assignee.
239         */
240        private boolean parseAssignee() {
241                // stores the beginning of identifiers that consist of multiple
242                // tokens. e.g., a->b->c = d->e->f(g, h).
243                String identifierStart = "";
244                while (isTokenType(currentPosition + 1, ARROW)) {
245                        identifierStart += tokens.get(currentPosition).getText() + "->";
246                        if (!advance(2)) {
247                                return false;
248                        }
249                }
250                if (isTokenType(currentPosition + 1, EQ)) {
251                        details.assignee = identifierStart + tokens.get(currentPosition).getText();
252                        if (!advance(2)) {
253                                return false;
254                        }
255                }
256                return true;
257        }
258
259        /**
260         * Parses a PERFORM call
261         */
262        private Optional<MethodCallInfo> parsePerformCall(List<IToken> tokens) {
263                int currentPosition = 0;
264                MethodCallInfo details = new MethodCallInfo();
265                if (tokens.isEmpty() || tokens.size() < 2) {
266                        return Optional.empty();
267                }
268                if (isTokenType(currentPosition, PERFORM)) {
269                        currentPosition++; // PERFORM is mandatory
270                } else {
271                        return Optional.empty();
272                }
273                if (isTokenType(currentPosition, IDENTIFIER)) {
274                        details.methodName = tokens.get(currentPosition).getText();
275                        currentPosition++;
276                } else {
277                        return Optional.empty();
278                }
279                while (tokens.size() > currentPosition && !isTokenType(currentPosition, DOT)) {
280                        if (EnumSet.of(USING, CHANGING).contains(tokens.get(currentPosition).getType())) {
281                                // USING and CHANGING can be ignored, everything is parsed into
282                                // one parameter list in ABAP
283                                currentPosition++;
284                        } else {
285                                // must be an identifier
286                                details.unnamedParameters.add(tokens.get(currentPosition).getText());
287                                currentPosition++;
288                        }
289                }
290                if (isTokenType(currentPosition, DOT)) {
291                        return Optional.of(details);
292                }
293                return Optional.empty(); // DOT is missing
294        }
295
296        /**
297         * Parses a parameter declarations map such as "EXPORTING x=y IMPORTING a=b
298         * c=d ..." Returns the position of the next token that is not part of the
299         * parameter list. Returns -1 if the parameter list could not be parsed.
300         *
301         * @param details
302         *            the MethodCallInfo details where parsed data is inserted
303         * @return next token position or -1 in case of a parse error
304         */
305        private boolean parseParameterDeclarations(MethodCallInfo details) {
306                try {
307                        while (PARAMETER_TOKEN_TYPES.contains(tokens.get(currentPosition).getType())) {
308                                ETokenType sectionType = tokens.get(currentPosition).getType();
309                                if (!advance(1)) {
310                                        return false;
311                                }
312                                switch (sectionType) {
313                                case EXPORTING:
314                                        parseParameterListInto(details.namedParameters);
315                                        break;
316                                case IMPORTING:
317                                        parseParameterListInto(details.importingParameters);
318                                        break;
319                                case CHANGING:
320                                        parseParameterListInto(details.changingParameters);
321                                        break;
322                                case RECEIVING:
323                                        if (!TokenStreamUtils.tokenTypesAt(tokens, currentPosition, IDENTIFIER, EQ, IDENTIFIER)) {
324                                                throw new IllegalArgumentException(
325                                                                "Did not find pattern \"x = y\" in receiving parameter at position " + currentPosition
326                                                                                + " .");
327                                        }
328                                        String formalParameterName = tokens.get(currentPosition).getText();
329                                        String actualParameterReferece = tokens.get(currentPosition + 2).getText();
330                                        details.namedReturnParameter = new Pair<>(formalParameterName, actualParameterReferece);
331                                        if (!advance(3)) {
332                                                return false;
333                                        }
334                                        break;
335                                case EXCEPTIONS:
336                                        parseParameterListInto(details.exceptions);
337                                        break;
338                                default:
339                                        CCSMAssert.fail("List of parameterTokenTypes is not in sync with switch cases.");
340                                }
341                        }
342                } catch (IllegalArgumentException e) {
343                        // syntax error while parsing a parameter list.
344                        return false;
345                }
346                return true;
347        }

351        /**
352         * Parses multiple instances of the pattern "x = y" and inserts them into
353         * the given parameter list. Returns the position after the last pattern
354         * instance. The currentPosition after this method's execution does not
355         * contain an IDENTIFIER.
356         */
357        private void parseParameterListInto(Map<String, String> parameterList)
358                        throws IllegalArgumentException {
359
360                while (isTokenType(currentPosition, IDENTIFIER)) {
361                        String formalParameterName = tokens.get(currentPosition).getText();
362                        String actualParameterReference;
363                        if (!isTokenType(currentPosition + 1, EQ)) {
364                                throw new IllegalArgumentException(
365                                                "Did not find pattern \"x = y\" in parameter list element at position " + currentPosition
366                                                                + " of tokenStream " + tokens.toString() + ".");
367                        }
368                        if (isTokenType(currentPosition + 2, LT)) {
369                                if (!isTokenType(currentPosition + 4, GT)) {
370                                        throw new IllegalArgumentException("Could not parse element as actual parameter at "
371                                                        + currentPosition + " of tokenStream " + tokens.toString() + ".");
372                                }
373                                actualParameterReference = "<" + tokens.get(currentPosition + 3) + ">";
374                                currentPosition += 5;
375                        } else {
376                                actualParameterReference = scanUntilParameterTerminatingToken();
377                        }
378                        parameterList.put(formalParameterName, actualParameterReference);
379                }
380        }
381
382        /**
383         * Advances until the next {@link #PARAMETER_TERMINATING_TOKENS} token and
384         * returns the text of all skipped tokens as String value.
385         */
386        private String scanUntilParameterTerminatingToken() {
387                String actualParameterReference;
388                int nextTokenIndex = TokenStreamUtils.findFirstTopLevel(
389                                tokens.subList(currentPosition + 2, tokens.size()), PARAMETER_TERMINATING_TOKENS,
390                                Collections.singletonList(LPAREN), Collections.singletonList(RPAREN));
391                if (nextTokenIndex == TokenStreamUtils.NOT_FOUND) {
392                        throw new IllegalArgumentException("Did not find closing token of parameter at position "
393                                        + currentPosition + " of tokenStream " + tokens.toString() + ".");
394                }
395                nextTokenIndex += currentPosition + 2;
396                if (isTokenType(nextTokenIndex, EQ)) {
397                        nextTokenIndex--; // this is "IDENTIFIER ="
398                }
399                actualParameterReference = TokenStreamTextUtils
400                                .concatTokenTexts(tokens.subList(currentPosition + 2, nextTokenIndex), " ");
401                currentPosition = nextTokenIndex;
402                return actualParameterReference;
403        }

405        /**
406         * Data class wrapping information on one method call (method name,
407         * parameter, ...)
408         */
409        public static class MethodCallInfo {
410
411                /** Name of the called method */
412                private String methodName;
413
414                /**
415                 * Name of the class or identifier of the instance that should contain
416                 * the method @see {@link #isStaticCall}
417                 */
418                private String methodContainerName;
419
420                /**
421                 * Whether this is a static method call: (call with => in ABAP) or an
422                 * instance method call (call with -> in ABAP)
423                 */
424                private boolean isStaticCall;
425                /**
426                 * Ordered list of unnamed parameters. The list contains the names of
427                 * the actual parameters in the caller method.
428                 */
429                private List<String> unnamedParameters = new ArrayList<String>();
430                /**
431                 * named parameters (EXPORTING in ABAP). Formal parameter name is mapped
432                 * to actual parameter name
433                 */
434                private Map<String, String> namedParameters = new HashMap<String, String>();
435                /**
436                 * The caller variable that is assigned with the return value of the
437                 * function call. E.g., "x" in "x = foo(...).".
438                 */
439                private String assignee;
440                /**
441                 * IMPORTING parameters. Formal parameter name is mapped to actual
442                 * parameter name
443                 */
444                private Map<String, String> importingParameters = new HashMap<String, String>();
445                /**
446                 * CHANGING parameters. Formal parameter name is mapped to actual
447                 * parameter name
448                 */
449                private Map<String, String> changingParameters = new HashMap<String, String>();
450                /**
451                 * The RECEIVING parameter (if present in this call).
452                 * Pair(formalParameterName, actualParameterName)
453                 */
454                private Pair<String, String> namedReturnParameter;
455
456                /**
457                 * The EXCEPTIONS parameter of the ABAP Call. Formal parameter name is
458                 * mapped to actual parameter name
459                 */
460                private Map<String, String> exceptions = new HashMap<String, String>();
461
462                /** @see #methodName */
463                public String getMethodName() {
464                        return methodName;
465                }
466
467                /** @see #methodContainerName */
468                public String getMethodContainerName() {
469                        return methodContainerName;
470                }
471
472                /** @see #isStaticCall */
473                public boolean isStaticCall() {
474                        return isStaticCall;
475                }
476
477                /** @see #unnamedParameters */
478                public List<String> getUnnamedParameters() {
479                        return unnamedParameters;
480                }
481
482                /** @see #namedParameters */
483                public Map<String, String> getNamedParameters() {
484                        return namedParameters;
485                }
486
487                /** @see #assignee */
488                public String getAssignee() {
489                        return assignee;
490                }
491
492                /** @see #importingParameters */
493                public Map<String, String> getImportingParameters() {
494                        return importingParameters;
495                }
496
497                /** @see #changingParameters */
498                public Map<String, String> getChangingParameters() {
499                        return changingParameters;
500                }
501
502                /** @see #namedReturnParameter */
503                public Pair<String, String> getNamedReturnParameter() {
504                        return namedReturnParameter;
505                }
506
507                /** @see #exceptions */
508                public Map<String, String> getExceptions() {
509                        return exceptions;
510                }
511
512        }
