004import static eu.cqse.check.framework.scanner.ETokenType.DOT;
005import static eu.cqse.check.framework.scanner.ETokenType.EQ;
006import static eu.cqse.check.framework.scanner.ETokenType.EXCEPTIONS;
007import static eu.cqse.check.framework.scanner.ETokenType.EXCEPTION_TABLE;
008import static eu.cqse.check.framework.scanner.ETokenType.EXPORTING;
009import static eu.cqse.check.framework.scanner.ETokenType.IMPORTING;
010import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
011import static eu.cqse.check.framework.scanner.ETokenType.PARAMETER_TABLE;
012import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
013import static eu.cqse.check.framework.scanner.ETokenType.TABLES;
014import static eu.cqse.check.framework.shallowparser.TokenStreamUtils.NOT_FOUND;
015
016import java.util.Arrays;
017import java.util.Collections;
018import java.util.EnumSet;
019import java.util.List;
020import java.util.Map;
021import java.util.Optional;
022import java.util.TreeMap;
023
024import eu.cqse.check.framework.core.CheckException;
025import eu.cqse.check.framework.scanner.ETokenType;
026import eu.cqse.check.framework.scanner.IToken;
027import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
028import eu.cqse.check.framework.shallowparser.TokenStreamUtils;

030/**
031 * Parses ABAP function calls and wraps information on the call, currently only
032 * called function name and exporting parameters.
033 */
034public class FunctionCallInfo {
035
036        /**
037         * Index of token where the function name starts (third token after 'CALL'
038         * and 'FUNCTION')
039         */
040        private static final int NAME_START_TOKEN = 2;
041
042        /** Key word OTHERS for specifying default exceptions */
043        private static final String OTHERS = "OTHERS";
044
045        /**
046         * Set of {@link ETokenType}s for delimiters of parameter sections, the
047         * occurrence of such a token indicates that a parameter section ends before
048         * this token.
049         */
050        private static final EnumSet<ETokenType> PARAMETER_SECTION_DELIMITERS = EnumSet.of(EXPORTING, IMPORTING, TABLES,
051                        CHANGING, EXCEPTIONS, PARAMETER_TABLE, EXCEPTION_TABLE, DOT);
052
053        /** Closing {@link ETokenType}s of nested method calls */
054        private static final List<ETokenType> CLOSING_TOKENS = Arrays.asList(RPAREN);
055
056        /** Opening {@link ETokenType}s of nested method calls */
057        private static final List<ETokenType> OPENING_TOKENS = Arrays.asList(LPAREN);
058
059        /**
060         * Constant for error codes which are not specified in EXCEPTIONS section
061         */
062        private static final int UNSPECIFIED_ERROR_CODE = -1;
063
064        /**
065         * Name of the function, in the case of dynamic call of the function this
066         * holds the identifier name
067         */
068        private final String functionName;
069
070        /**
071         * EXPORTING parameters. Formal parameter name is mapped to actual parameter
072         * name
073         */
074        private final Map<String, List<IToken>> exportingParameters = new TreeMap<>(String.CASE_INSENSITIVE_ORDER);
075
076        /**
077         * EXEPTIONS specification - exception name is mapped to assigned token.
078         */
079        private final Map<String, Integer> exceptionsSpecifiction = new TreeMap<>(String.CASE_INSENSITIVE_ORDER);
080
081        /**
082         * Tokens to parse
083         */
084        private final List<IToken> tokens;
085
086        /**
087         * Constructor
088         * 
089         * @param tokens
090         *            tokens of the function call must start with CALL FUNCTION
091         * @throws CheckException
092         *             if parsing fails
093         */
094        /* package */ FunctionCallInfo(List<IToken> tokens) throws CheckException {
095                this.tokens = tokens;
096                functionName = parseFunctionName();
097                parseExportingParameters();
098                parseExceptions();
099        }
100
101        /**
102         * Parses the function name.
103         */
104        private String parseFunctionName() {
105                int endOfName = TokenStreamUtils.firstTokenOfType(tokens,
106                                PARAMETER_SECTION_DELIMITERS.toArray(new ETokenType[PARAMETER_SECTION_DELIMITERS.size()]));
107
108                List<IToken> calledFunctionTokens = tokens.subList(NAME_START_TOKEN, endOfName);
109
110                if (isStaticFunctionCall(calledFunctionTokens)) {
111                        return AbapCheckUtils.getUnquotedTextForCharcterLiteral(calledFunctionTokens.get(0));
112                }
113
114                return TokenStreamTextUtils.concatTokenTexts(calledFunctionTokens);
115        }
116
117        /**
118         * Checks if the given tokens refer to a static function call
119         * 
120         * @param functionNameTokens
121         *            tokens of the called function name
122         * @return <code>true</code> if the function is called statically, e.g. only
123         *         a character literal is stated
124         */
125        private static boolean isStaticFunctionCall(List<IToken> functionNameTokens) {
126                return functionNameTokens.size() == 1 && functionNameTokens.get(0).getType() == ETokenType.CHARACTER_LITERAL;
127        }
128
129        /**
130         * Parses the EXPORTING section and fills {@link #exportingParameters}.
131         * 
132         * @throws CheckException
133         *             in case {@link #tokens} are not well-formed
134         */
135        private void parseExportingParameters() throws CheckException {
136                List<IToken> exportingSectionTokens = getParameterSectionTokens(EXPORTING);
137                if (exportingSectionTokens.isEmpty()) {
138                        return;
139                }
140                if (exportingSectionTokens.size() < 2 || exportingSectionTokens.get(1).getType() != EQ) {
141                        throw new CheckException("Unable to parse CALL FUNCTION: EXPORTING section of " + functionName
142                                        + " does not start with 'param ='. (line " + tokens.get(0).getLineNumber() + ")");
143                }
144                int nextEq = 1;
145                while (nextEq != NOT_FOUND) {
146                        IToken formalParameter = exportingSectionTokens.get(nextEq - 1);
147                        int actualStart = nextEq + 1;
148                        nextEq = TokenStreamUtils.findFirstTopLevel(exportingSectionTokens, nextEq + 1, EnumSet.of(EQ),
149                                        OPENING_TOKENS, CLOSING_TOKENS);
150                        int actuelEnd;
151                        if (nextEq == NOT_FOUND) {
152                                actuelEnd = exportingSectionTokens.size();
153                        } else {
154                                actuelEnd = nextEq - 1;
155                        }
156                        exportingParameters.put(formalParameter.getText(), exportingSectionTokens.subList(actualStart, actuelEnd));
157                }
158        }
159
160        /**
161         * Parses the EXCEPTIONS section, the normal format is
162         * <code>exc1 = n1 exc2 = n2 ... [OTHERS =n_others]</code> where exec1 exec2
163         * ... refer to name of non-class-based exceptions and n1, n2, ..., n_others
164         * refers to the error code which must be an integer literal within
165         * [0..65535]. See
166         * http://help.sap.com/abapdocu_751/en/abapcall_function_parameter.htm#!
167         * ABAP_ADDITION_6@6@
168         * 
169         * However there is also the obsolete short form of
170         * <code>exc1 exc2 ..</code> which equivalent to
171         * <code>exc1 = 1 exc2 = 1 ...</code>. In case of the old form
172         * {@link #UNSPECIFIED_ERROR_CODE} is set as error value for the exception
173         * name to be able to distinguish between actually specified error codes or
174         * the obsolete short form (which should be avoided). See also
175         * http://help.sap.com/abapdocu_751/en/abapcall_function_exc_short_form.htm
176         * 
177         * @throws CheckException
178         *             in case {@link #tokens} are not well-formed
179         */
180        private void parseExceptions() throws CheckException {
181                List<IToken> sectionTokens = getParameterSectionTokens(EXCEPTIONS);
182                int currentIndex = 0;
183                while (currentIndex < sectionTokens.size()) {
184                        String exceptionName = sectionTokens.get(currentIndex).getText();
185                        if (currentIndex + 1 == sectionTokens.size() || sectionTokens.get(currentIndex + 1).getType() != EQ) {
186                                // obsolete short form is used
187                                exceptionsSpecifiction.put(exceptionName, UNSPECIFIED_ERROR_CODE);
188                                currentIndex += 1;
189                                continue;
190                        }
191                        currentIndex += 2;
192                        IToken errorCodeToken = sectionTokens.get(currentIndex);
193                        if (errorCodeToken.getType() != ETokenType.INTEGER_LITERAL) {
194                                throw new CheckException("Other than integer literal set for exception " + exceptionName);
195                        }
196                        exceptionsSpecifiction.put(exceptionName, Integer.valueOf(errorCodeToken.getText()));
197                        currentIndex += 1;
198                }
199        }
200
201        /**
202         * Gets the tokens of the parameter section which is introduced by an token
203         * of the given sectionType.
204         * 
205         * @param sectionType
206         *            {@link ETokenType} of the introducing token of the section
207         * @return tokens of the parameter section without the introducing token or
208         *         an empty list if the section does not occur.
209         * @throws CheckException
210         *             in case {@link #tokens} are not well-formed
211         */
212        private List<IToken> getParameterSectionTokens(ETokenType sectionType) throws CheckException {
213                int sectionStart = TokenStreamUtils.findFirstTopLevel(tokens, sectionType, OPENING_TOKENS, CLOSING_TOKENS);
214                if (sectionStart == NOT_FOUND) {
215                        return Collections.emptyList();
216                }
217                sectionStart++;
218                int sectionEnd = TokenStreamUtils.findFirstTopLevel(tokens, sectionStart + 1, PARAMETER_SECTION_DELIMITERS,
219                                OPENING_TOKENS, CLOSING_TOKENS);
220                if (sectionEnd == NOT_FOUND) {
221                        throw new CheckException("Unable to parse CALL FUNCTION: end token for EXPORTING section not found.");
222                }
223                return tokens.subList(sectionStart, sectionEnd);
224        }
225
226        /** see {@link #functionName} */
227        public String getFunctionName() {
228                return functionName;
229        }
230
231        /**
232         * Gets the token which is passed to the given exporting parameter
233         * 
234         * @return an {@link Optional} holding the single token which is passed to
235         *         the given formal parameter. If the given exporting parameter is
236         *         not set or a list of tokens is passed, the result is empty.
237         */
238        public Optional<IToken> getPassedExportingToken(String formalParamterName) {
239                List<IToken> passed = exportingParameters.get(formalParamterName);
240                if (passed == null || passed.size() != 1) {
241                        return Optional.empty();
242                }
243                return Optional.of(passed.get(0));
244        }
245
246        /**
247         * Checks if an error code is set for the given exception name, this is the
248         * case if a value not equal to {@value AbapCheckUtils#SY_SUBRC_NO_ERROR} is
249         * specified in the EXECPTIONS section or, if the value is not specified
250         * explicitly, the value for OTHERS is specified and not equal to
251         * {@value AbapCheckUtils#SY_SUBRC_NO_ERROR}.
252         * 
253         * @return <code>true</code> if an exception with this name specified and a
254         *         value other than {@value AbapCheckUtils#SY_SUBRC_NO_ERROR} is
255         *         assigned or, if the exception is not specified explicitly but
256         *         OTHERS is specified to be not
257         *         {@value AbapCheckUtils#SY_SUBRC_NO_ERROR}
258         */
259        public boolean isSettingErrorCodeForException(String exceptionName) {
260                Integer errorCode = exceptionsSpecifiction.get(exceptionName);
261                if (errorCode == null) {
262                        if (OTHERS.equals(exceptionName)) {
263                                return false;
264                        }
265                        return isSettingErrorCodeForException(OTHERS);
266                }
267                if (errorCode == AbapCheckUtils.SY_SUBRC_NO_ERROR) {
268                        return false;
269                }
270                return true;
271        }
