006package eu.cqse.check.framework.util.filter;
007
008import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
009import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
010
011import java.util.Arrays;
012import java.util.Collection;
013import java.util.HashMap;
014import java.util.List;
015import java.util.Map;
016
017import eu.cqse.check.framework.scanner.IToken;
018import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
019import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
020import eu.cqse.check.framework.util.LanguageFeatureParser;
021
022/**
023 * Filter for entities which are annotated with one of a given set of
024 * annotations.
025 */
026public class JavaAnnotatedEntityFilter {
027
028        private final Map<String, List<String>> filteringAnnotations = new HashMap<>();
029
030        /**
031         * Adds multiple annotations without any parameter.
032         *
033         * @param filteringAnnotations
034         *            List of filtering annotations without any parameters, or where the
035         *            parameters do not matter.
036         */
037        public void addAnnotations(Collection<String> filteringAnnotations) {
038                for (String annotation : filteringAnnotations) {
039                        addAnnotation(annotation);
040                }
041        }
042
043        /**
044         * Add a filtering annotation with specific parameters.
045         */
046        public void addAnnotation(String annotation, String... parameters) {
047                this.filteringAnnotations.put(annotation, Arrays.asList(parameters));
048        }
049
050        /**
051         * Add a filtering annotation which has specific parameters. Also different
052         * variants for the same annotation with the same parameters can be supplied.
053         * E.g. "SuppressWarnings" or "java.lang.SuppressWarnings"
054         */
055        public void addAnnotationAndVariants(Collection<String> annotationVariants, String... parameters) {
056                for (String variant : annotationVariants) {
057                        addAnnotation(variant, parameters);
058                }
059        }
060
061        /**
062         * Gets the annotations of the given entity in a map, where the key is the name
063         * of the annotation and the value are the tokens of potential parameters.
064         */
065        public Map<String, List<IToken>> getAnnotations(ShallowEntity entity) {
066                Map<String, List<IToken>> annotationMap = new HashMap<>();
067                List<ShallowEntity> annotations = LanguageFeatureParser.JAVA.getAnnotations(entity, true);
068                for (ShallowEntity annotation : annotations) {
069                        List<IToken> parameters = TokenStreamUtils.tokensBetween(annotation.ownStartTokens(), LPAREN, RPAREN);
070                        annotationMap.put(annotation.getName(), parameters);
071                }
072
073                return annotationMap;
074        }
075
076        /**
077         * Checks if the filtering annotation matches any of the annotations of the
078         * entity.
079         */
080        private boolean matchesFilteringAnnotation(String filteringAnnotation,
081                        Map<String, List<IToken>> entityAnnotations) {
082                if (entityAnnotations.containsKey(filteringAnnotation)) {
083                        List<String> filteringParameters = filteringAnnotations.get(filteringAnnotation);
084                        if (filteringParameters.size() > 0) {
085                                return matchesFilteringParameters(filteringParameters, entityAnnotations.get(filteringAnnotation));
086                        }
087                        return true;
088                }
089                return false;
090        }
091
092        /**
093         * Check if the filtering parameters match the actual parameters of the
094         * annotation.
095         */
096        private static boolean matchesFilteringParameters(List<String> filteringParameters, List<IToken> parameters) {
097                return filteringParameters.stream().anyMatch(filteringParameter -> parameters.stream()
098                                .anyMatch(parameter -> parameter.getText().equals(filteringParameter)));
099        }
100
101        /** Determines whether the given entity should be filtered. */
102        public final boolean isFiltered(ShallowEntity entity) {
103                Map<String, List<IToken>> entityAnnotations = getAnnotations(entity);
104
105                for (String filteringAnnotation : filteringAnnotations.keySet()) {
106                        if (matchesFilteringAnnotation(filteringAnnotation, entityAnnotations)) {
107                                return true;
108                        }
109                }
110
111                return false;
112        }
113}
