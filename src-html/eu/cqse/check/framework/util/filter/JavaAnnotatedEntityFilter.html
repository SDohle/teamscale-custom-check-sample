006package eu.cqse.check.framework.util.filter;
007
008import java.util.HashSet;
009import java.util.List;
010import java.util.Set;
011
012import org.conqat.lib.commons.assertion.CCSMAssert;
013
014import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
015import eu.cqse.check.framework.util.LanguageFeatureParser;
016
017/**
018 * Filter for entities which are annotated with one of a given set of
019 * annotations.
020 */
021public class JavaAnnotatedEntityFilter {
022
023        /** Set of annotations that mark an entity as filtered. */
024        private final Set<String> filteringAnnotations;
025
026        /** Constructor. */
027        public JavaAnnotatedEntityFilter(Set<String> filteringAnnotations) {
028                CCSMAssert.isNotNull(filteringAnnotations);
029                this.filteringAnnotations = new HashSet<String>(filteringAnnotations);
030        }
031
032        /**
033         * Determines if the entity annotation matches the filtering annotation
034         * contained in the {@link #filteringAnnotations}.
035         */
036        protected boolean matchesFilteringAnnotation(String filteringAnnotation, ShallowEntity entityAnnotation) {
037                return entityAnnotation.getName().equals(filteringAnnotation);
038        }
039
040        /** Determines whether the given entity should be filtered. */
041        public final boolean isFiltered(ShallowEntity entity) {
042                List<ShallowEntity> annotationsOnEntity = LanguageFeatureParser.JAVA.getAnnotations(entity, true);
043
044                return this.filteringAnnotations.stream().anyMatch(filteringAnnotation -> annotationsOnEntity.stream()
045                                .anyMatch(entityAnnotation -> matchesFilteringAnnotation(filteringAnnotation, entityAnnotation)));
046        }
047}
