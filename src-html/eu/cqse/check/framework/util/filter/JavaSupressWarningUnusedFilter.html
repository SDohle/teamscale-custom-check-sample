006package eu.cqse.check.framework.util.filter;
007
008import java.util.Collections;
009
010import eu.cqse.check.framework.scanner.IToken;
011import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
012
013/**
014 * Filter for entities annotated with a SupressWarnings("unused") annotation.
015 */
016public class JavaSupressWarningUnusedFilter extends JavaAnnotatedEntityFilter {
017
018        /** Constructor. */
019        public JavaSupressWarningUnusedFilter() {
020                super(Collections.singleton("SuppressWarnings"));
021        }
022
023        /**
024         * Checks if the given SuppressedWarning annotation entity has the 'unused'
025         * warning specified.
026         */
027        private static boolean isSuppressWarningsUnused(ShallowEntity entity) {
028                for (IToken token : entity.ownStartTokens()) {
029                        if (token.getText().replace("\"", "").equals("unused")) {
030                                return true;
031                        }
032                }
033                return false;
034        }
035
036        /** {@inheritDoc} */
037        @Override
038        protected boolean matchesFilteringAnnotation(String filteringAnnotation, ShallowEntity entityAnnotation) {
039                return super.matchesFilteringAnnotation(filteringAnnotation, entityAnnotation)
040                                && isSuppressWarningsUnused(entityAnnotation);
041        }
042
043}
