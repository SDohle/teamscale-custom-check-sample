008package eu.cqse.check.framework.util.tokens;
009
010import java.util.ArrayList;
011import java.util.List;
012import java.util.Set;
013
014import org.conqat.lib.commons.assertion.CCSMAssert;
015import org.conqat.lib.commons.collections.CollectionUtils;
016
017import eu.cqse.check.framework.scanner.ETokenType;
018import eu.cqse.check.framework.scanner.ETokenType.ETokenClass;
019import eu.cqse.check.framework.scanner.IToken;
020
021/**
022 * A pattern that can be applied like a regular expression to a token stream.
023 * 
024 * This class is not fully redundant to TokenTypePattern for two reasons:
025 * <ul>
026 * <li>It allows matching nested structures, e.g. parentheses, with
027 * {@link #skipNested(Object, Object, boolean)}
028 * <li>It allows named capture groups, something which is not currently possible
029 * with Java 7: http://bugs.java.com/bugdatabase/view_bug.do?bug_id=8013252
030 * </ul>
031 * 
032 * TODO (BH): I think there is still lots of potential for improvement in this
033 * class. Many of the matchers seem somewhat redundant. However, I would rather
034 * wait for the next round of improvements, as then the code should be easier to
035 * read. (Note to self: another full review ahead :))
036 */
037public class TokenPattern {
038
039        /**
040         * The parts of the pattern that are matched in sequence.
041         */
042        private final List<TokenPatternBase> submatchers = new ArrayList<>();
043
044        /**
045         * Matches the pattern at all positions in the given token stream and
046         * returns the found matches. This also finds overlapping matches.
047         */
048        public List<TokenPatternMatch> findAll(List<IToken> tokens) {
049                List<TokenPatternMatch> matches = new ArrayList<TokenPatternMatch>();
050                // TODO (BH): This looks extremely expensive (quadratic?), as you check
051                // at every position. Can't you skip ahead if a pattern matched? Or do
052                // you also intend to find overlapping matches? If so, please document.
053                // TODO (FS) I'm not sure if we need the overlapping. I documented it
054                // for now, but I'd have to take a closer look at the actual patterns to
055                // be sure we can make it non-overlapping
056                TokenStream stream = new TokenStream(tokens, 0);
057                for (int i = 0; i < tokens.size(); i++) {
058                        stream.setPosition(i);
059                        TokenPatternMatch match = matchesLocally(stream);
060                        if (match != null) {
061                                matches.add(match);
062                        }
063                }
064                return matches;
065        }
066
067        /**
068         * Matches the pattern at all positions in the given token stream and
069         * returns the first found match or <code>null</code> if no match happened.
070         */
071        public TokenPatternMatch findFirstMatch(List<IToken> tokens) {
072                for (int i = 0; i < tokens.size(); i++) {
073                        TokenStream stream = new TokenStream(tokens, i);
074                        TokenPatternMatch match = matchesLocally(stream);
075                        if (match != null) {
076                                return match;
077                        }
078                }
079                return null;
080        }
081
082        /**
083         * Returns <code>true</code> if the pattern matches at least once in the
084         * given token stream.
085         */
086        public boolean matchesAnywhere(List<IToken> tokens) {
087                return findFirstMatch(tokens) != null;
088        }
089
090        /**
091         * Matches the pattern once at the current position of the given token
092         * stream and records the results in the given match.
093         * 
094         * @return <code>true</code> if the pattern matched or <code>false</code> if
095         *         the given match should be discarded.
096         */
097        private TokenPatternMatch matchesLocally(TokenStream stream) {
098                return new SequencePattern(CollectionUtils.toArray(submatchers, TokenPatternBase.class)).matchesLocally(stream);
099                // TODO (BH): Actually, I do not understand why the submatchers
100                // have to handle the "match" as well. It seems that in case of
101                // groups, you just handle update of the match here. Why not
102                // also the the unnamed match groups? Then the submatchers would
103                // be much simpler. Just must return a boolean (match, nomatch).
104                // TODO (FS) we need to pass the match down to collect the group
105                // info (now moved to TokenPatternBase). I'd like to remove it from
106                // the interface, but then I'd have to store it in an instance field
107                // and the patterns would no longer be thread safe (stored as static
108                // constants now most of the time). So TokenPatternMatch is sort of
109                // the analogy to Java's Matcher class
110        }
111
112        /**
113         * Specifies that the last added subpattern should be appended to the group
114         * with the given index. The group index may be any integer number. Group
115         * indices are not required to be ordered or continuous.
116         */
117        public TokenPattern group(int groupIndex) {
118                CCSMAssert.isFalse(submatchers.isEmpty(),
119                                "You must specify a pattern part before you can assign a group to it.");
120                submatchers.get(submatchers.size() - 1).setGroupIndex(groupIndex);
121                return this;
122        }
123
124        /**
125         * Adds a new pattern part that consumes anything until it encounters one of
126         * the given match terms. The stop token matching one of the given terms is
127         * also consumed.
128         * 
129         * A group applied to this pattern will only contain the stop token.
130         * 
131         * @param matchTerms
132         *            a list of match terms which must match in order. These may be
133         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
134         *            of them, as well as {@link TokenPattern}s.
135         */
136        public TokenPattern skipTo(Object... matchTerms) {
137                TokenPatternBase[] matchers = convertMatchTerms(matchTerms);
138                submatchers.add(new SkipPattern(matchers));
139                submatchers.add(new AlternativePattern(matchers));
140                return this;
141        }
142
143        /**
144         * Adds a new pattern part that consumes nested structures.
145         * 
146         * @param openTerm
147         *            a match term that identifies the opening braces. May be an
148         *            instance of {@link ETokenType}, {@link ETokenClass}, or a set
149         *            of them, as well as an {@link TokenPattern}.
150         * @param closeTerm
151         *            a match term that identifies the closing braces. May be an
152         *            instance of {@link ETokenType}, {@link ETokenClass}, or a set
153         *            of them, as well as an {@link TokenPattern}.
154         * @param optional
155         *            if this is <code>false</code>, the nested structure must be
156         *            present or the pattern will not match. If this is
157         *            <code>true</code>, this pattern may return an empty match.
158         */
159        public TokenPattern skipNested(Object openTerm, Object closeTerm, boolean optional) {
160                TokenPatternBase openMatcher = convertMatchTerm(openTerm);
161                TokenPatternBase closeMatcher = convertMatchTerm(closeTerm);
162                submatchers.add(new SkipNestedPattern(openMatcher, closeMatcher, optional));
163                return this;
164        }
165
166        /**
167         * Adds a new pattern part that matches the given terms zero or more times
168         * in the given order.
169         * 
170         * @param matchTerms
171         *            a list of match terms which must match in order. These may be
172         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
173         *            of them, as well as {@link TokenPattern}s.
174         */
175        public TokenPattern repeated(Object... matchTerms) {
176                TokenPatternBase[] matchers = convertMatchTerms(matchTerms);
177                submatchers.add(new RepeatedPattern(matchers));
178                return this;
179        }
180
181        /**
182         * Adds a new pattern part that matches the given terms exactly once in the
183         * given order.
184         * 
185         * @param matchTerms
186         *            a list of match terms which must match in order. These may be
187         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
188         *            of them, as well as {@link TokenPattern}s.
189         */
190        public TokenPattern sequence(Object... matchTerms) {
191                TokenPatternBase[] matchers = convertMatchTerms(matchTerms);
192                submatchers.add(new SequencePattern(matchers));
193                return this;
194        }
195
196        /**
197         * Adds a new pattern part that matches the end of the token stream (i.e.
198         * there are no more tokens to match).
199         */
200        public TokenPattern endOfStream() {
201                submatchers.add(new ZeroLengthTokenPatternBase() {
202
203                        @Override
204                        public boolean conditionApplies(TokenStream stream) {
205                                return stream.isExhausted();
206                        }
207                });
208                return this;
209        }
210
211        /**
212         * Adds a new pattern part that matches the beginning of the token stream
213         * (i.e. no tokens have been consumed so far).
214         */
215        public TokenPattern beginningOfStream() {
216                submatchers.add(new ZeroLengthTokenPatternBase() {
217
218                        @Override
219                        public boolean conditionApplies(TokenStream stream) {
220                                return stream.isAtBeginning();
221                        }
222                });
223                return this;
224        }
225
226        /**
227         * Adds a new pattern part that matches anything but the beginning of the
228         * token stream (i.e. at least one token has been consumed so far).
229         */
230        public TokenPattern notAtBeginningOfStream() {
231                submatchers.add(new ZeroLengthTokenPatternBase() {
232
233                        @Override
234                        public boolean conditionApplies(TokenStream stream) {
235                                return !stream.isAtBeginning();
236                        }
237                });
238                return this;
239        }
240
241        /**
242         * Adds a new pattern part that matches if the last matched token is not
243         * followed by a token that matches the given term. This pattern part does
244         * not consume any tokens.
245         * 
246         * @param matchTerm
247         *            May be an instance of {@link ETokenType}, {@link ETokenClass},
248         *            or a set of them, as well as a {@link TokenPattern}.
249         */
250        public TokenPattern notFollowedBy(Object matchTerm) {
251                TokenPatternBase matcher = convertMatchTerm(matchTerm);
252                submatchers.add(new NotFollowedByPattern(matcher));
253                return this;
254        }
255
256        /**
257         * Adds a new pattern part that matches if the last matched token is not
258         * preceded by a token that matches the given term. This pattern part does
259         * not consume any tokens.
260         * 
261         * NOTE: passing a {@link TokenPattern} to this function will not result in
262         * an exception, but it will not yield the intuitive result. The given
263         * pattern will be matched against the token stream, starting with the token
264         * preceding the current token. It is therefore discouraged to pass a
265         * {@link TokenPattern} to this function as it makes understanding the
266         * pattern difficult.
267         * 
268         * @param matchTerm
269         *            May be an instance of {@link ETokenType}, {@link ETokenClass},
270         *            or a set of them, but usually not a {@link TokenPattern}.
271         */
272        public TokenPattern notPrecededBy(Object matchTerm) {
273                TokenPatternBase matcher = convertMatchTerm(matchTerm);
274                submatchers.add(new NotPrecededByPattern(matcher));
275                return this;
276        }
277
278        /**
279         * Adds a new pattern part that tries the given match terms one after the
280         * other and stops as soon as one of them matches.
281         * 
282         * @param matchTerms
283         *            a list of match terms which must match in order. These may be
284         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
285         *            of them, as well as {@link TokenPattern}s.
286         */
287        public TokenPattern alternative(Object... matchTerms) {
288                TokenPatternBase[] matchers = convertMatchTerms(matchTerms);
289                submatchers.add(new AlternativePattern(matchers));
290                return this;
291        }
292
293        /**
294         * Adds a new pattern part that matches the given terms zero times or once
295         * in the given order.
296         * 
297         * @param matchTerms
298         *            a list of match terms which must match in order. These may be
299         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
300         *            of them, as well as {@link TokenPattern}s.
301         */
302        public TokenPattern optional(Object... matchTerms) {
303                TokenPatternBase[] matchers = convertMatchTerms(matchTerms);
304                submatchers.add(new OptionalPattern(matchers));
305                return this;
306        }
307
308        /** Converts the given match terms to {@link TokenPatternBase}s. */
309        private TokenPatternBase[] convertMatchTerms(Object[] matchTerms) {
310                CCSMAssert.isFalse(matchTerms.length == 0, "You must provide at least one match term.");
311                TokenPatternBase[] matchers = new TokenPatternBase[matchTerms.length];
312                for (int i = 0; i < matchTerms.length; ++i) {
313                        matchers[i] = convertMatchTerm(matchTerms[i]);
314                }
315                return matchers;
316        }
317
318        /** Converts a match term to a matcher. */
319        private TokenPatternBase convertMatchTerm(final Object matchTerm) {
320                if (matchTerm instanceof TokenPattern) {
321                        return new TokenPatternBase() {
322
323                                @Override
324                                public TokenPatternMatch matchesLocally(TokenStream stream) {
325                                        TokenPattern pattern = (TokenPattern) matchTerm;
326                                        int beforePosition = stream.getPosition();
327                                        TokenPatternMatch match = pattern.matchesLocally(stream);
328                                        if (match == null) {
329                                                stream.setPosition(beforePosition);
330                                                return null;
331                                        }
332                                        return match;
333                                }
334                        };
335                }
336
337                if (matchTerm instanceof ETokenType) {
338                        return new SingleTokenPatternBase() {
339                                @Override
340                                public boolean matchesToken(IToken token) {
341                                        return token.getType() == (ETokenType) matchTerm;
342                                }
343                        };
344                }
345
346                if (matchTerm instanceof ETokenClass) {
347                        return new SingleTokenPatternBase() {
348                                @Override
349                                public boolean matchesToken(IToken token) {
350                                        return token.getType().getTokenClass() == (ETokenClass) matchTerm;
351                                }
352                        };
353                }
354
355                if (matchTerm instanceof Set<?>) {
356                        final Set<?> set = (Set<?>) matchTerm;
357                        return new SingleTokenPatternBase() {
358                                @Override
359                                public boolean matchesToken(IToken token) {
360                                        ETokenType type = token.getType();
361                                        return set.contains(type) || set.contains(type.getTokenClass());
362                                }
363                        };
364                }
365
366                throw new AssertionError("Unsupported match term of type " + matchTerm.getClass());
367        }
