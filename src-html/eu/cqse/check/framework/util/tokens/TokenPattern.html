021/**
022 * A pattern that can be applied like a regular expression to a token stream.
023 *
024 * This class is not fully redundant to TokenTypePattern for two reasons:
025 * <ul>
026 * <li>It allows matching nested structures, e.g. parentheses, with
027 * {@link #skipNested(Object, Object, boolean)}
028 * <li>It allows named capture groups, something which is not currently possible
029 * with Java 7: http://bugs.java.com/bugdatabase/view_bug.do?bug_id=8013252
030 * </ul>
031 *
032 * TODO (BH): I think there is still lots of potential for improvement in this
033 * class. Many of the matchers seem somewhat redundant. However, I would rather
034 * wait for the next round of improvements, as then the code should be easier to
035 * read. (Note to self: another full review ahead :))
036 */
037public class TokenPattern {
038
039        /**
040         * The parts of the pattern that are matched in sequence.
041         */
042        private final List<TokenPatternBase> submatchers = new ArrayList<>();
043
044        /** {@inheritDoc} */
045        @Override
046        public String toString() {
047                return "submatchers " + submatchers.toString();
048        }
049
050        /**
051         * Matches the pattern at all positions in the given token stream and
052         * returns the found matches. This also finds overlapping matches.
053         */
054        public List<TokenPatternMatch> findAll(List<IToken> tokens) {
055                List<TokenPatternMatch> matches = new ArrayList<TokenPatternMatch>();
056                // TODO (BH): This looks extremely expensive (quadratic?), as you check
057                // at every position. Can't you skip ahead if a pattern matched? Or do
058                // you also intend to find overlapping matches? If so, please document.
059                // TODO (FS) I'm not sure if we need the overlapping. I documented it
060                // for now, but I'd have to take a closer look at the actual patterns to
061                // be sure we can make it non-overlapping
062                TokenStream stream = new TokenStream(tokens, 0);
063                for (int i = 0; i < tokens.size(); i++) {
064                        stream.setPosition(i);
065                        TokenPatternMatch match = matchesLocally(stream);
066                        if (match != null) {
067                                matches.add(match);
068                        }
069                }
070                return matches;
071        }
072
073        /**
074         * Matches the pattern at all positions in the given token stream and
075         * returns the first found match or <code>null</code> if no match happened.
076         */
077        public TokenPatternMatch findFirstMatch(List<IToken> tokens) {
078                for (int i = 0; i < tokens.size(); i++) {
079                        TokenStream stream = new TokenStream(tokens, i);
080                        TokenPatternMatch match = matchesLocally(stream);
081                        if (match != null) {
082                                return match;
083                        }
084                }
085                return null;
086        }
087
088        /**
089         * Returns <code>true</code> if the pattern matches at least once in the
090         * given token stream.
091         */
092        public boolean matchesAnywhere(List<IToken> tokens) {
093                return findFirstMatch(tokens) != null;
094        }
095
096        /**
097         * Matches the pattern once at the current position of the given token
098         * stream and records the results in the given match.
099         *
100         * @return <code>true</code> if the pattern matched or <code>false</code> if
101         *         the given match should be discarded.
102         */
103        private TokenPatternMatch matchesLocally(TokenStream stream) {
104                return new SequencePattern(CollectionUtils.toArray(submatchers, TokenPatternBase.class)).matchesLocally(stream);
105                // TODO (BH): Actually, I do not understand why the submatchers
106                // have to handle the "match" as well. It seems that in case of
107                // groups, you just handle update of the match here. Why not
108                // also the the unnamed match groups? Then the submatchers would
109                // be much simpler. Just must return a boolean (match, nomatch).
110                // TODO (FS) we need to pass the match down to collect the group
111                // info (now moved to TokenPatternBase). I'd like to remove it from
112                // the interface, but then I'd have to store it in an instance field
113                // and the patterns would no longer be thread safe (stored as static
114                // constants now most of the time). So TokenPatternMatch is sort of
115                // the analogy to Java's Matcher class
116        }
117
118        /**
119         * Specifies that the last added subpattern should be appended to the group
120         * with the given index. The group index may be any integer number. Group
121         * indices are not required to be ordered or continuous.
122         */
123        public TokenPattern group(int groupIndex) {
124                CCSMAssert.isFalse(submatchers.isEmpty(),
125                                "You must specify a pattern part before you can assign a group to it.");
126                submatchers.get(submatchers.size() - 1).setGroupIndex(groupIndex);
127                return this;
128        }
129
130        /**
131         * Adds a new pattern part that consumes anything until it encounters one of
132         * the given match terms. The stop token matching one of the given terms is
133         * also consumed.
134         *
135         * A group applied to this pattern will only contain the stop token.
136         *
137         * @param matchTerms
138         *            a list of match terms which must match in order. These may be
139         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
140         *            of them, as well as {@link TokenPattern}s.
141         */
142        public TokenPattern skipTo(Object... matchTerms) {
143                TokenPatternBase[] matchers = convertMatchTerms(matchTerms);
144                submatchers.add(new SkipPattern(matchers));
145                submatchers.add(new AlternativePattern(matchers));
146                return this;
147        }
148
149        /**
150         * Adds a new pattern part that consumes nested structures.
151         *
152         * @param openTerm
153         *            a match term that identifies the opening braces. May be an
154         *            instance of {@link ETokenType}, {@link ETokenClass}, or a set
155         *            of them, as well as an {@link TokenPattern}.
156         * @param closeTerm
157         *            a match term that identifies the closing braces. May be an
158         *            instance of {@link ETokenType}, {@link ETokenClass}, or a set
159         *            of them, as well as an {@link TokenPattern}.
160         * @param optional
161         *            if this is <code>false</code>, the nested structure must be
162         *            present or the pattern will not match. If this is
163         *            <code>true</code>, this pattern may return an empty match.
164         */
165        public TokenPattern skipNested(Object openTerm, Object closeTerm, boolean optional) {
166                TokenPatternBase openMatcher = convertMatchTerm(openTerm);
167                TokenPatternBase closeMatcher = convertMatchTerm(closeTerm);
168                submatchers.add(new SkipNestedPattern(openMatcher, closeMatcher, optional));
169                return this;
170        }
171
172        /**
173         * Adds a new pattern part that matches the given terms zero or more times
174         * in the given order.
175         *
176         * @param matchTerms
177         *            a list of match terms which must match in order. These may be
178         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
179         *            of them, as well as {@link TokenPattern}s.
180         */
181        public TokenPattern repeated(Object... matchTerms) {
182                TokenPatternBase[] matchers = convertMatchTerms(matchTerms);
183                submatchers.add(new RepeatedPattern(matchers));
184                return this;
185        }
186
187        /**
188         * Adds a new pattern part that matches the given terms exactly once in the
189         * given order.
190         *
191         * @param matchTerms
192         *            a list of match terms which must match in order. These may be
193         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
194         *            of them, as well as {@link TokenPattern}s.
195         */
196        public TokenPattern sequence(Object... matchTerms) {
197                TokenPatternBase[] matchers = convertMatchTerms(matchTerms);
198                submatchers.add(new SequencePattern(matchers));
199                return this;
200        }

202        /**
203         * Adds a new pattern part that matches the end of the token stream (i.e.
204         * there are no more tokens to match).
205         */
206        public TokenPattern endOfStream() {
207                submatchers.add(new ZeroLengthTokenPatternBase() {
208
209                        @Override
210                        public boolean conditionApplies(TokenStream stream) {
211                                return stream.isExhausted();
212                        }
213
214                        /** {@inheritDoc} */
215                        @Override
216                        public String toString() {
217                                return "ZeroLength is exhausted";
218                        }
219                });
220                return this;
221        }

230                        @Override

244        /**
245         * Adds a new pattern part that matches anything but the beginning of the

248        public TokenPattern notAtBeginningOfStream() {
249                submatchers.add(new ZeroLengthTokenPatternBase() {
250
251                        @Override
252                        public boolean conditionApplies(TokenStream stream) {

254                        }
255
256                        /** {@inheritDoc} */
257                        @Override

259                                return "ZeroLength not at beginning";

261                });
262                return this;

265        /**
266         * Adds a new pattern part that matches if the last matched token is not
267         * followed by a token that matches the given term. This pattern part does
268         * not consume any tokens.
269         *
270         * @param matchTerm
271         *            May be an instance of {@link ETokenType}, {@link ETokenClass},
272         *            or a set of them, as well as a {@link TokenPattern}.
273         */
274        public TokenPattern notFollowedBy(Object matchTerm) {
275                TokenPatternBase matcher = convertMatchTerm(matchTerm);
276                submatchers.add(new NotFollowedByPattern(matcher));
277                return this;
278        }
279
280        /**
281         * Adds a new pattern part that matches if the last matched token is not
282         * preceded by a token that matches the given term. This pattern part does
283         * not consume any tokens.
284         *
285         * NOTE: passing a {@link TokenPattern} to this function will not result in
286         * an exception, but it will not yield the intuitive result. The given
287         * pattern will be matched against the token stream, starting with the token
288         * preceding the current token. It is therefore discouraged to pass a
289         * {@link TokenPattern} to this function as it makes understanding the
290         * pattern difficult.
291         *
292         * @param matchTerm
293         *            May be an instance of {@link ETokenType}, {@link ETokenClass},
294         *            or a set of them, but usually not a {@link TokenPattern}.
295         */
296        public TokenPattern notPrecededBy(Object matchTerm) {
297                TokenPatternBase matcher = convertMatchTerm(matchTerm);
298                submatchers.add(new NotPrecededByPattern(matcher));
299                return this;
300        }
301
302        /**
303         * Adds a new pattern part that tries the given match terms one after the
304         * other and stops as soon as one of them matches.
305         *
306         * @param matchTerms
307         *            a list of match terms which must match in order. These may be
308         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
309         *            of them, as well as {@link TokenPattern}s.
310         */
311        public TokenPattern alternative(Object... matchTerms) {
312                TokenPatternBase[] matchers = convertMatchTerms(matchTerms);
313                submatchers.add(new AlternativePattern(matchers));
314                return this;
315        }
316
317        /**
318         * Adds a new pattern part that matches the given terms zero times or once
319         * in the given order.
320         *
321         * @param matchTerms
322         *            a list of match terms which must match in order. These may be
323         *            instances of {@link ETokenType}, {@link ETokenClass}, or sets
324         *            of them, as well as {@link TokenPattern}s.
325         */
326        public TokenPattern optional(Object... matchTerms) {
327                TokenPatternBase[] matchers = convertMatchTerms(matchTerms);
328                submatchers.add(new OptionalPattern(matchers));
329                return this;
330        }
331
332        /** Converts the given match terms to {@link TokenPatternBase}s. */
333        private TokenPatternBase[] convertMatchTerms(Object[] matchTerms) {
334                CCSMAssert.isFalse(matchTerms.length == 0, "You must provide at least one match term.");
335                TokenPatternBase[] matchers = new TokenPatternBase[matchTerms.length];
336                for (int i = 0; i < matchTerms.length; ++i) {
337                        matchers[i] = convertMatchTerm(matchTerms[i]);
338                }
339                return matchers;
340        }
341
342        /** Converts a match term to a matcher. */
343        private TokenPatternBase convertMatchTerm(final Object matchTerm) {
344                if (matchTerm instanceof TokenPattern) {
345                        return new TokenPatternBase() {
346
347                                @Override
348                                public TokenPatternMatch matchesLocally(TokenStream stream) {
349                                        TokenPattern pattern = (TokenPattern) matchTerm;
350                                        int beforePosition = stream.getPosition();
351                                        TokenPatternMatch match = pattern.matchesLocally(stream);
352                                        if (match == null) {
353                                                stream.setPosition(beforePosition);
354                                                return null;
355                                        }
356                                        return match;
357                                }
358
359                                /** {@inheritDoc} */
360                                @Override
361                                public String toString() {
362                                        return matchTerm.toString();
363                                }
364                        };
365                }
366
367                if (matchTerm instanceof ETokenType) {
368                        return new SingleTokenPatternBase() {
369                                @Override
370                                public boolean matchesToken(IToken token) {
371                                        return token.getType() == (ETokenType) matchTerm;
372                                }
373
374                                /** {@inheritDoc} */
375                                @Override
376                                public String toString() {
377                                        return matchTerm.toString();
378                                }
379                        };
380                }
381
382                if (matchTerm instanceof ETokenClass) {
383                        return new SingleTokenPatternBase() {
384                                @Override
385                                public boolean matchesToken(IToken token) {
386                                        return token.getType().getTokenClass() == (ETokenClass) matchTerm;
387                                }
388
389                                /** {@inheritDoc} */
390                                @Override
391                                public String toString() {
392                                        return matchTerm.toString();
393                                }
394                        };
395                }
396
397                if (matchTerm instanceof Set<?>) {
398                        final Set<?> set = (Set<?>) matchTerm;
399                        return new SingleTokenPatternBase() {
400                                @Override
401                                public boolean matchesToken(IToken token) {
402                                        ETokenType type = token.getType();
403                                        return set.contains(type) || set.contains(type.getTokenClass());
404                                }
405
406                                /** {@inheritDoc} */
407                                @Override
408                                public String toString() {
409                                        return matchTerm.toString();
410                                }
411                        };
412                }
413
414                throw new AssertionError("Unsupported match term of type " + matchTerm.getClass());
415        }
