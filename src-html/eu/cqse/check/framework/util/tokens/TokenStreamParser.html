008package eu.cqse.check.framework.util.tokens;
009
010import java.util.ArrayList;
011import java.util.EnumSet;
012import java.util.List;
013import java.util.Set;
014
015import org.conqat.lib.commons.assertion.CCSMAssert;
016
017import eu.cqse.check.framework.scanner.ETokenType;
018import eu.cqse.check.framework.scanner.IToken;
019
020/**
021 * Allows parsing a token stream from left to right. Does not support
022 * backtracking, i.e. it is only possible to advance the parser forward, not
023 * backward.
024 */
025public class TokenStreamParser {
026
027        /** The token stream. */
028        private final List<IToken> tokens;
029
030        /** The current position of the parser. */
031        private int currentToken = 0;
032
033        /**
034         * Constructor.
035         * 
036         * @param tokens
037         *            The tokens to parse.
038         * @param filteredTokens
039         *            A set of token types. All tokens of these types are filtered
040         *            out before parsing, i.e. they are simply skipped.
041         */
042        public TokenStreamParser(List<IToken> tokens, Set<ETokenType> filteredTokens) {
043                this(filter(tokens, filteredTokens));
044        }
045
046        /** Constructor. */
047        public TokenStreamParser(List<IToken> tokens) {
048                this.tokens = tokens;
049        }
050
051        /**
052         * Returns a view of the given token list that does not contain any tokens
053         * of the given types.
054         */
055        private static List<IToken> filter(List<IToken> tokens, Set<ETokenType> filteredTokens) {
056                List<IToken> filtered = new ArrayList<IToken>();
057                for (IToken token : tokens) {
058                        if (!filteredTokens.contains(token.getType())) {
059                                filtered.add(token);
060                        }
061                }
062                return filtered;
063        }
064
065        /**
066         * Consumes all sequential tokens of the given types and returns their
067         * texts.
068         */
069        public List<String> consumeAnyOf(Set<ETokenType> types) {
070                List<String> texts = new ArrayList<String>();
071                while (isAnyOf(types)) {
072                        texts.add(currentText());
073                        currentToken++;
074                }
075                return texts;
076        }
077
078        /**
079         * Consumes one token of the given type and returns its text or
080         * <code>null</code> if the current token is not of any of those types. This
081         * method only advances the parser if the current token has one of the given
082         * types.
083         */
084        public String consumeOneOrZeroOf(Set<ETokenType> types) {
085                if (isAnyOf(types)) {
086                        String text = currentText();
087                        currentToken++;
088                        return text;
089                }
090                return null;
091        }
092
093        /**
094         * Consumes all sequential tokens that are not of the given types and
095         * returns their texts.
096         */
097        public List<String> consumeAnyExcept(Set<ETokenType> types) {
098                List<String> texts = new ArrayList<String>();
099                while (!isAnyOf(types)) {
100                        if (currentType() == null) {
101                                break;
102                        }
103                        texts.add(currentText());
104                        currentToken++;
105                }
106                return texts;
107        }
108
109        /**
110         * Skips balanced braces as long as they only contain the allowed inner
111         * types. Checks that the current token is of the opening type.
112         * 
113         * Skipping may be interrupted prematurely if unexpected tokens are
114         * encountered. In this case, the parser is left pointing to the bad token.
115         * 
116         * @return <code>true</code> if the skipping succeeded and
117         *         <code>false</code> if there were invalid tokens inside the
118         *         balanced braces or the braces were not balanced.
119         */
120        public boolean skipBalanced(ETokenType openingType, ETokenType closingType, Set<ETokenType> allowedInnerTypes) {
121                CCSMAssert.isFalse(allowedInnerTypes.contains(openingType),
122                                "The given inner token types contain the opening token type");
123                CCSMAssert.isFalse(allowedInnerTypes.contains(closingType),
124                                "The given inner token types contain the closing token type");
125
126                if (currentType() != openingType) {
127                        return false;
128                }
129                consumeOneOrZeroOf(EnumSet.of(openingType));
130
131                int openCount = 1;
132                while (openCount > 0) {
133                        consumeAnyOf(allowedInnerTypes);
134                        if (consumeOneOrZeroOf(EnumSet.of(openingType)) != null) {
135                                openCount++;
136                        } else if (consumeOneOrZeroOf(EnumSet.of(closingType)) != null) {
137                                openCount--;
138                        } else {
139                                return false;
140                        }
141                }
142                return true;
143        }
144
145        /** Consumes tokens as long as they alternate between the two type sets. */
146        public List<String> consumeAlternating(Set<ETokenType> typeSet1, Set<ETokenType> typeSet2) {
147                List<String> list = new ArrayList<String>();
148                while (true) {
149                        String token1 = consumeOneOrZeroOf(typeSet1);
150                        if (token1 == null) {
151                                break;
152                        }
153                        list.add(token1);
154                        String token2 = consumeOneOrZeroOf(typeSet2);
155                        if (token2 == null) {
156                                break;
157                        }
158                        list.add(token2);
159                }
160                return list;
161        }
162
163        /**
164         * Returns <code>true</code> if the current token has any of the given
165         * types.
166         */
167        public boolean isAnyOf(Set<ETokenType> types) {
168                return types.contains(currentType());
169        }
170
171        /** Returns <code>true</code> if all tokens have been consumed. */
172        public boolean isDone() {
173                return currentToken >= tokens.size();
174        }
175
176        /**
177         * Returns the text of the current token or <code>null</code> if all tokens
178         * have been consumed.
179         */
180        public String currentText() {
181                if (isDone()) {
182                        return null;
183                }
184                return tokens.get(currentToken).getText();
185        }
186
187        /**
188         * Returns the type of the current token or <code>null</code> if all tokens
189         * have been consumed.
190         */
191        public ETokenType currentType() {
192                if (isDone()) {
193                        return null;
194                }
195                return tokens.get(currentToken).getType();
196        }
197
198        /**
199         * Returns the number of tokens consumed so far.
200         */
201        public int getConsumedTokenCount() {
202                return currentToken;
203        }
204
205        /** {@inheritDoc} */
206        @Override
207        public String toString() {
208                return "TokenStreamParser " + tokens.subList(currentToken, tokens.size());
209        }
210}
