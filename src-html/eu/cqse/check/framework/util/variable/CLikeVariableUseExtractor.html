017package eu.cqse.check.framework.util.variable;
018
019import java.util.ArrayList;
020import java.util.EnumSet;
021import java.util.List;
022
023import eu.cqse.check.framework.scanner.ETokenType;
024import eu.cqse.check.framework.scanner.IToken;
025import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
026
027/**
028 * A variable use extractor extracts uses of variables from token streams.
029 * Thereby it differentiates between fields and local variables and considers
030 * shadowing of variables.
031 * 
032 * @author $Author: heinemann $
033 * @version $Rev: 55645 $
034 * @ConQAT.Rating RED Hash: B119CB3B696C8E2CE00FB2444F7B7E9D
035 */
036public class CLikeVariableUseExtractor {
037
038        /** The token type of the operator that is used to access fields. */
039        private final ETokenType accessOperator;
040
041        /**
042         * All token types that indicate that the previous token is no variable.
043         */
044        private final EnumSet<ETokenType> noVariableSuccessorTypes;
045
046        /** Constructor. */
047        public CLikeVariableUseExtractor(ETokenType accessOperator,
048                        EnumSet<ETokenType> noVariableSuccessorTypes) {
049                this.accessOperator = accessOperator;
050                this.noVariableSuccessorTypes = noVariableSuccessorTypes;
051        }
052
053        /**
054         * Extracts all uses of the given variable name from the given tokens.
055         * 
056         * @param isField
057         *            whether the given variable name is a field of a type
058         * @param isShadowed
059         *            whether the given variable name is shadowed by another
060         *            variable with the same name
061         */
062        public List<Integer> extractVariableUses(List<IToken> tokens,
063                        String variableName, boolean isField, boolean isShadowed) {
064                return extractFilteredVariableUses(tokens, variableName, isField,
065                                isShadowed, AcceptingVariableUseFilter.INSTANCE);
066        }
067
068        /**
069         * Extracts all uses of the given variable name from the given tokens that
070         * read its value.
071         * 
072         * @param isField
073         *            whether the given variable name is a field of a type
074         * @param isShadowed
075         *            whether the given variable name is shadowed by another
076         *            variable with the same name
077         */
078        public List<Integer> extractVariableReads(List<IToken> tokens,
079                        String variableName, boolean isField, boolean isShadowed) {
080                return extractFilteredVariableUses(tokens, variableName, isField,
081                                isShadowed, VariableReadFilter.INSTANCE);
082        }
083
084        /**
085         * Extracts all uses of the given variable name from the given tokens that
086         * change its value.
087         * 
088         * @param isField
089         *            whether the given variable name is a field of a type
090         * @param isShadowed
091         *            whether the given variable name is shadowed by another
092         *            variable with the same name
093         */
094        public List<Integer> extractVariableWrites(List<IToken> tokens,
095                        String variableName, boolean isField, boolean isShadowed) {
096                return extractFilteredVariableUses(tokens, variableName, isField,
097                                isShadowed, VariableWriteFilter.INSTANCE);
098        }
099
100        /**
101         * Extracts all uses of the given variable name from the given tokens.
102         * 
103         * @param isField
104         *            whether the given variable name is a field of a type
105         * @param isShadowed
106         *            whether the given variable name is shadowed by another
107         *            variable with the same name
108         */
109        public List<Integer> extractFilteredVariableUses(List<IToken> tokens,
110                        String variableName, boolean isField, boolean isShadowed,
111                        IVariableUseFilter filter) {
112                List<Integer> uses = TokenStreamTextUtils.findAll(tokens, variableName);
113                return filterUses(tokens, uses, isField, isShadowed, filter);
114        }
115
116        /**
117         * Filters then given variable uses based on whether they are fields,
118         * shadowed and the given custom filter.
119         */
120        // TODO (LH) Please choose better name for uses parameter and mention in
121        // comment what this is exactly. Probably indexes in some list?
122        public List<Integer> filterUses(List<IToken> tokens, List<Integer> uses,
123                        boolean isField, boolean isShadowed, IVariableUseFilter filter) {
124                List<Integer> filteredUses = new ArrayList<>();
125                for (int index : uses) {
126                        if (isVariableUse(tokens, index, isField, isShadowed)
127                                        && filter.isFiltered(tokens, index, isField)) {
128                                filteredUses.add(index);
129                        }
130                }
131                return filteredUses;
132        }
133
134        /**
135         * Returns whether the element at the given index is any kind of variable
136         * use.
137         */
138        private boolean isVariableUse(List<IToken> tokens, int index,
139                        boolean isField, boolean isShadowed) {
140                if (index < tokens.size() - 1 && noVariableSuccessorTypes
141                                .contains(tokens.get(index + 1).getType())) {
142                        return false;
143                }
144
145                if (isField) {
146                        return isFieldUse(tokens, index, isShadowed);
147                }
148                return isLocalUse(tokens, index, isShadowed);
149        }
150
151        /**
152         * Returns whether the variable use at the given index is the use of the
153         * field. If it is not shadowed, it is definitely a field, otherwise we
154         * check if it is prefixed by the field access operator.
155         */
156        private boolean isFieldUse(List<IToken> tokens, int index,
157                        boolean isShadowed) {
158                return !isShadowed || (index > 1
159                                && accessOperator.equals(tokens.get(index - 1).getType()));
160        }
161
162        /**
163         * Returns whether the variable use at the given index is the use of the
164         * local variable. If it is shadowed by another local variable it is not,
165         * otherwise we check if it is prefixed by the field access operator. If it
166         * is not, it is the local variable.
167         */
168        private boolean isLocalUse(List<IToken> tokens, int index,
169                        boolean isShadowed) {
170                return !isShadowed && (index <= 1
171                                || !accessOperator.equals(tokens.get(index - 1).getType()));
172        }
173}
