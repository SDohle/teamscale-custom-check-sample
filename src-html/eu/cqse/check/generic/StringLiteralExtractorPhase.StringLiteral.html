006package eu.cqse.check.generic;
007
008import java.util.ArrayList;
009import java.util.EnumSet;
010import java.util.List;
011import java.util.Objects;
012import java.util.Optional;
013import java.util.Set;
014
015import org.conqat.lib.commons.assertion.CCSMAssert;
016import org.conqat.lib.commons.collections.CollectionUtils;
017import org.conqat.lib.commons.string.StringUtils;
018
019import eu.cqse.check.CheckTextRegionLocation;
020import eu.cqse.check.framework.core.CheckException;
021import eu.cqse.check.framework.core.phase.ECodeViewOption;
022import eu.cqse.check.framework.core.phase.IExtractedValue;
023import eu.cqse.check.framework.core.phase.IGlobalExtractionPhase;
024import eu.cqse.check.framework.core.phase.ITokenElementContext;
025import eu.cqse.check.framework.scanner.ELanguage;
026import eu.cqse.check.framework.scanner.ETokenType;
027import eu.cqse.check.framework.scanner.IToken;
028import eu.cqse.check.generic.StringLiteralExtractorPhase.StringLiteral;
029
030/**
031 * Extraction phase for the {@link RedundantStringLiteralCheck} that extracts
032 * all string literals.
033 */
034public class StringLiteralExtractorPhase implements IGlobalExtractionPhase<StringLiteral, CheckTextRegionLocation> {
035
036        /**
037         * Strings that are ignored because they are used for technical reasons in the
038         * code. Examples are suppress warnings annotations in Java.
039         */
040        private static final Set<String> IGNORED_STRINGS = CollectionUtils.asHashSet("all", "allDeprecation", "allJavadoc",
041                        "assertIdentifier", "boxing", "cast", "charConcat", "conditionAssign", "constructorName", "dep",
042                        "deprecation", "discouraged", "divzero", "empty", "emptyBlock", "enumSwitch", "fallthrough", "fieldHiding",
043                        "finalBound", "finally", "forbidden", "hiding", "incomplete", "indirectStatic", "intfAnnotation",
044                        "intfNonInherited", "javadoc", "localHiding", "maskedCatchBlocks", "nls", "noEffectAssign", "none", "null",
045                        "nullDereference", "over", "overrides", "paramAssign", "path", "pkgDefaultMethod", "raw", "rawtypes",
046                        "restriction", "semicolon", "serial", "specialParamHiding", "static", "staticReceiver", "super", "suppress",
047                        "synthetic", "syntheticAccess", "tasks", "typeHiding", "unchecked", "unnecessaryElse", "unqualified",
048                        "unused", "unusedArgument", "unusedImport", "unusedLabel", "unusedLocal", "unusedPrivate", "unusedThrown",
049                        "uselessTypeCheck", "varargsCast", "warningToken");
050
051        @Override
052        public List<StringLiteral> extract(ITokenElementContext fileContext) throws CheckException {
053                List<StringLiteral> result = new ArrayList<>();
054                for (IToken token : fileContext.getTokens(ECodeViewOption.UNFILTERED)) {
055                        if (token.getType() == ETokenType.STRING_LITERAL) {
056                                Optional<String> value = extractStringValue(token.getText());
057                                if (value.isPresent()) {
058                                        result.add(new StringLiteral(fileContext.getUniformPath(), value.get(), token.getOffset(),
059                                                        token.getEndOffset(), token.getLineNumber() + 1));
060                                }
061                        }
062                }
063                return result;
064        }
065
066        /**
067         * Extracts the actual string value. This consists of removing quotes and also
068         * handles special prefixes, such as '@' in C# and 'L' in C++.
069         */
070        private static Optional<String> extractStringValue(String text) {
071                if (text.length() < 2) {
072                        return Optional.empty();
073                }
074
075                int start = 1;
076                if (text.charAt(0) != '"' && text.charAt(0) != '\'') {
077                        start += 1;
078                }
079
080                text = text.substring(start, text.length() - 1);
081
082                if (StringUtils.isEmpty(text) || IGNORED_STRINGS.contains(text)) {
083                        return Optional.empty();
084                }
085
086                return Optional.of(text);
087        }
088
089        /** {@inheritDoc} */
090        @Override
091        public StringLiteral createValue(String uniformPath, String value, CheckTextRegionLocation additionalInformation) {
092                CCSMAssert.isTrue(uniformPath.equals(additionalInformation.uniformPath),
093                                "uniform paths must be equal: " + uniformPath + " vs. " + additionalInformation.uniformPath);
094                return new StringLiteral(value, additionalInformation);
095        }
096
097        @Override
098        public boolean needsAccessUniformPathByValue() {
099                return true;
100        }
101
102        @Override
103        public Set<ELanguage> getLanguages() {
104                return EnumSet.complementOf(EnumSet.of(ELanguage.SIMULINK));
105        }
106
107        /** Value class for the extractor. */
108        public static class StringLiteral implements IExtractedValue<CheckTextRegionLocation> {
109
110                private final String literalValue;
111
112                private final CheckTextRegionLocation location;
113
114                private StringLiteral(String uniformPath, String literalValue, int startOffset, int endOffset, int startLine) {
115                        this.literalValue = literalValue;
116                        int linesInLiteral = StringUtils.countLines(literalValue);
117                        location = new CheckTextRegionLocation(uniformPath, startOffset, endOffset, startLine,
118                                        startLine + (linesInLiteral - 1));
119                }
120
121                private StringLiteral(String literalValue, CheckTextRegionLocation location) {
122                        this.literalValue = literalValue;
123                        this.location = location;
124                }
125
126                @Override
127                public String getUniformPath() {
128                        return location.uniformPath;
129                }
130
131                @Override
132                public String getValue() {
133                        return literalValue;
134                }
135
136                /**
137                 * Returns the location of this literal (contains uniform path, unfiltered-token
138                 * start/end offset, and start line)
139                 */
140                public CheckTextRegionLocation getLiteralLocation() {
141                        return location;
142                }
143
144                @Override
145                public CheckTextRegionLocation getAdditionalInformation() {
146                        return getLiteralLocation();
147                }
148
149                /** {@inheritDoc} */
150                @Override
151                public int hashCode() {
152                        return Objects.hash(literalValue, location);
153                }
154
155                /** {@inheritDoc} */
156                @Override
157                public boolean equals(Object obj) {
158                        if (this == obj) {
159                                return true;
160                        }
161                        if (obj == null || getClass() != obj.getClass()) {
162                                return false;
163                        }
164                        StringLiteral other = (StringLiteral) obj;
165                        return Objects.equals(literalValue, other.literalValue) && Objects.equals(location, other.location);
166                }
167        }
168}
