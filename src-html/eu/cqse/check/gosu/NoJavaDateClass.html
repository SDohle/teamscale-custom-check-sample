001package eu.cqse.check.gosu;
002
003import eu.cqse.check.base.EntityFindingCheckBase;
004import eu.cqse.check.framework.core.ACheck;
005import eu.cqse.check.framework.core.CheckGroupNames;
006import eu.cqse.check.framework.core.ECheckParameter;
007import eu.cqse.check.framework.scanner.ELanguage;
008import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
009
010/** {@ConQAT.Doc} */
011@ACheck(name = "Avoid the Java Date class", description = "Use Gosu DateTime object "
012                + "instead of Java Date as they expose the java date functions using "
013                + "operators like \">\" and \"<\" for simple comparisons. Gosu supports "
014                + "the java getTime() function \"as Number\" to give the millisecs like the "
015                + "getTime function.", groupName = CheckGroupNames.BAD_PRACTICE, languages = {
016                                ELanguage.GOSU }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
017public class NoJavaDateClass extends EntityFindingCheckBase {
018
019        /** {@inheritDoc} */
020        @Override
021        protected String getFindingMessage(ShallowEntity entity) {
022                return "Use Gosu DateTime instead of the Java Date class";
023        }
024
025        /** {@inheritDoc} */
026        @Override
027        protected String getXPathSelectionString() {
028                return "//META[subtype('uses') and name-matches('java\\.util\\.Date')]";
029        }
030}
