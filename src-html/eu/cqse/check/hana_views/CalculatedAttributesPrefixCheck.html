008package eu.cqse.check.hana_views;
009
010import java.util.regex.Pattern;
011
012import eu.cqse.check.framework.core.ACheck;
013import eu.cqse.check.framework.core.option.ACheckOption;
014import eu.cqse.check.framework.scanner.ELanguage;
015import eu.cqse.check.xml.AttributeRegexCheckBase;
016
017/** {@ConQAT.Doc} */
018@ACheck(name = CalculatedAttributesPrefixCheck.CHECK_NAME, description = ""
019                + "Calculated attributes must be properly prefixed to meet the naming conventions.", groupName = IHanaViewCheckGroups.IDENTIFIER_PREFIX, languages = {
020                                ELanguage.HANA_VIEW })
021public class CalculatedAttributesPrefixCheck extends AttributeRegexCheckBase {
022
023        /** Name of this check */
024        /* package */ static final String CHECK_NAME = "Prefix of calculated attributes";
025
026        /** {@ConQAT.Doc} */
027        @ACheckOption(name = CHECK_NAME + " - required prefix", description = ""
028                        + "Prefix required for calculated attributes.")
029        private String prefix = "CA_";
030
031        /** Constructor. */
032        public CalculatedAttributesPrefixCheck() {
033                super(HanaViewsCheckUtil.ELEMENTS_CALCULATED_ATTRIBUTE, HanaViewsCheckUtil.ATTRIBUTE_ID);
034        }
035
036        /** {@inheritDoc} */
037        @Override
038        protected String getFindingsMessage() {
039                return "Calculated attribute names must start with '" + prefix + "'";
040        }
041
042        /** {@inheritDoc} */
043        @Override
044        protected Pattern getPattern() {
045                return getPrefixPattern(prefix);
046        }
047
048        /** {@inheritDoc} */
049        @Override
050        protected boolean showPatternInFindingsMessage() {
051                return false;
052        }
053}
