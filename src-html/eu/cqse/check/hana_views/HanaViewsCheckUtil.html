008package eu.cqse.check.hana_views;
009
010import java.util.Set;
011
012import org.conqat.lib.commons.collections.CollectionUtils;
013import org.conqat.lib.commons.string.StringUtils;
014
015import eu.cqse.check.framework.core.ICheckContext;
016
017/**
018 * Utility methods and constants used in HANA View model checks.
019 */
020public class HanaViewsCheckUtil {
021
022        /**
023         * XML element name, refers to measure fields (calculated and restricted
024         * measures)
025         */
026        public static final String ELEMENT_MEASURE = "measure";
027
028        /** XML element name, refers to attribute fields */
029        private static final String ELEMENT_VIEW_ATTRIBUTE = "viewAttribute";
030
031        /**
032         * XML element name, refers to calculated measures (child of
033         * {@link #ELEMENT_MEASURE})
034         */
035        public static final String ELEMENT_CALCULATED_MEASURES = "calculatedMeasures";
036
037        /**
038         * XML element name, refers to restricted measures (child of
039         * {@link #ELEMENT_MEASURE})
040         */
041        public static final String ELEMENT_RESTRICTED_MEASURES = "restrictedMeasures";
042
043        /** XML element name, refers to attributes which are used in a join */
044        public static final String ELEMENT_JOIN_ATTRIBUTE = "joinAttribute";
045
046        /** XML element names, refers to calculated attributes */
047        public static final Set<String> ELEMENTS_CALCULATED_ATTRIBUTE = CollectionUtils.asHashSet("calculatedAttribute",
048                        "calculatedViewAttribute");
049
050        /** XML attribute name, refers to the specified name of an attribute */
051        public static final String ATTRIBUTE_ID = "id";
052
053        /**
054         * XML attribute name, refers to hidden attributes which are not specified
055         * by the developer
056         */
057        public static final String ATTRIBUTE_HIDDEN = "hidden";
058
059        /** XML attribute name, refers to the referenced name of an attribute */
060        public static final String ATTRIBUTE_NAME = "name";
061
062        /** Set of XML elements denoting fields */
063        public static final Set<String> FIELD_ELEMENTS = CollectionUtils.asHashSet(ELEMENT_VIEW_ATTRIBUTE, ELEMENT_MEASURE);
064
065        /**
066         * Constant for a regular expression on allowed identifiers (field names and
067         * object names). Only a-z, A-Z, _, 0-9 is allowed in identifier names, but
068         * no special characters
069         */
070        public static final String ALLOWED_IDENTIFIER_NAME_REGEX = "^[a-zA-Z_0-9]+$";
071
072        /**
073         * Gets the name of the repository object from the check context, this
074         * refers to the file name of the object.
075         */
076        public static String getRepositoryObjectName(ICheckContext context) {
077                String fileName = StringUtils.getLastPart(context.getUniformPath(), '/');
078                return StringUtils.removeLastPart(fileName, '.');
079        }
080
081}
