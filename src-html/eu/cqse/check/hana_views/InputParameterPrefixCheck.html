008package eu.cqse.check.hana_views;
009
010import eu.cqse.check.framework.core.ACheck;
011import eu.cqse.check.framework.core.option.ACheckOption;
012import eu.cqse.check.framework.scanner.ELanguage;
013
014/** {@ConQAT.Doc} */
015@ACheck(name = InputParameterPrefixCheck.CHECK_NAME, description = ""
016                + "Input paramters must be properly prefixed to meet the naming conventions.", groupName = IHanaViewCheckGroups.IDENTIFIER_PREFIX, languages = {
017                                ELanguage.HANA_VIEW })
018public class InputParameterPrefixCheck extends VariableNamingCheckBase {
019
020        /** Name of this check */
021        /* package */ static final String CHECK_NAME = "Prefix of input parameters";
022
023        /** {@ConQAT.Doc} */
024        @ACheckOption(name = CHECK_NAME
025                        + " - required prefix", description = "Prefix required for input parameters.")
026        private String parameterPrefix = "IP_";
027
028        /** {@inheritDoc} */
029        @Override
030        protected boolean isFinding(String id, boolean isParameter) {
031                return isParameter && !id.startsWith(parameterPrefix);
032        }
033
034        /** {@inheritDoc} */
035        @Override
036        protected String getFindingMessage() {
037                return "Input parameter names must start with '" + parameterPrefix + "'";
038        }
039}
