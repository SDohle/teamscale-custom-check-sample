008package eu.cqse.check.hana_views;
009
010import java.io.IOException;
011import java.util.HashSet;
012import java.util.Set;
013
014import org.conqat.lib.commons.xml.XMLUtils;
015import org.xml.sax.Attributes;
016import org.xml.sax.SAXException;
017import org.xml.sax.helpers.DefaultHandler;
018
019import eu.cqse.check.framework.core.ACheck;
020import eu.cqse.check.framework.core.CheckException;
021import eu.cqse.check.framework.core.phase.ECodeViewOption;
022import eu.cqse.check.framework.scanner.ELanguage;
023import eu.cqse.check.xml.XmlCheckBase;
024import eu.cqse.check.xml.XmlCheckHandlerBase;
025import eu.cqse.check.xml.XmlElementAttributesCheckHandlerBase;
026
027/**
028 * Checks if a calculated attribute is used as join attribute.
029 */
030@ACheck(name = "Joins on calculated attributes", description = ""
031                + "Joins should not be done over calculated attributes.", groupName = IHanaViewCheckGroups.PERFORMANCE, languages = {
032                                ELanguage.HANA_VIEW })
033public class JoinOnCalculatedAttributesCheck extends XmlCheckBase {
034
035        /**
036         * Set of calculated attributes in the analyzed file. Is initialized in
037         * {@link #preProcess()}.
038         */
039        private Set<String> calculatedAttributes = null;
040
041        /**
042         * {@inheritDoc}
043         */
044        @Override
045        protected void preProcess() throws CheckException {
046                calculatedAttributes = new HashSet<>();
047                try {
048                        XMLUtils.parseSAX(
049                                        XMLUtils.toInputSource(context.getTextContent(ECodeViewOption.ETextViewOption.FILTERED_CONTENT)),
050                                        new DefaultHandler() {
051                                                /** {@inheritDoc} */
052                                                @Override
053                                                public void startElement(String uri, String localName, String qName, Attributes attributes) {
054                                                        if (HanaViewsCheckUtil.ELEMENTS_CALCULATED_ATTRIBUTE.contains(localName)) {
055                                                                calculatedAttributes.add(attributes.getValue(HanaViewsCheckUtil.ATTRIBUTE_ID));
056                                                        }
057                                                }
058                                        });
059                } catch (SAXException | IOException e) {
060                        throw new CheckException("Unable to parse " + context.getUniformPath(), e);
061                }
062        }
063
064        /** {@inheritDoc} */
065        @Override
066        public XmlCheckHandlerBase getHandler() {
067                return new XmlElementAttributesCheckHandlerBase(HanaViewsCheckUtil.ELEMENT_JOIN_ATTRIBUTE) {
068                        @Override
069                        protected void checkAttributes(Attributes attributes) {
070                                String joinAttribute = attributes.getValue(HanaViewsCheckUtil.ATTRIBUTE_NAME);
071                                if (calculatedAttributes.contains(joinAttribute)) {
072                                        createFinding("Join on calculated attribute '" + joinAttribute + "'.");
073                                }
074
075                        }
076                };
077        }
078
079}
