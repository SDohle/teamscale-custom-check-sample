008package eu.cqse.check.hana_views;
009
010import org.xml.sax.Attributes;
011
012import eu.cqse.check.framework.core.ACheck;
013import eu.cqse.check.framework.core.option.ACheckOption;
014import eu.cqse.check.framework.scanner.ELanguage;
015import eu.cqse.check.xml.XmlCheckBase;
016import eu.cqse.check.xml.XmlCheckHandlerBase;
017import eu.cqse.check.xml.XmlElementAttributesCheckHandlerBase;
018
019/**
020 * {@ConQAT.Doc}
021 */
022@ACheck(name = ScriptedCalculationViewNamingCheck.CHECK_NAME, description = ""
023                + "Scripted calculation views must be named accoring to naming convention.", groupName = IHanaViewCheckGroups.OBJECT_NAMING, languages = {
024                                ELanguage.HANA_VIEW })
025public class ScriptedCalculationViewNamingCheck extends XmlCheckBase {
026
027        /** Name of the check */
028        /* package */ static final String CHECK_NAME = "Naming of scripted calculation views";
029
030        /** {@ConQAT.Doc} */
031        @ACheckOption(name = CHECK_NAME
032                        + " - required naming pattern", description = "Regular expression for the naming pattern.")
033        String namingPatternRegex = ".*_SQL$";
034
035        /**
036         * XML element name, refers to a calculation view, this is the main element
037         * in calculation view files
038         */
039        private static final String ELEMENT_CALCULATION_VIEW = "calculationView";
040
041        /**
042         * XML attribute name, refers to the type of e.g. a calculation view
043         */
044        private static final String ATTRIBUTE_XSI_TYPE = "xsi:type";
045
046        /**
047         * Expected value of {@link #ATTRIBUTE_XSI_TYPE} for scripted calculation
048         * views
049         */
050        private static final String VALUE_XSI_TYPE_SCRIPTED_CALCULATION_VIEWS = "Calculation:SqlScriptView";
051
052        /** {@inheritDoc} */
053        @Override
054        public XmlCheckHandlerBase getHandler() {
055                return new XmlElementAttributesCheckHandlerBase(ELEMENT_CALCULATION_VIEW) {
056
057                        @Override
058                        protected void checkAttributes(Attributes attributes) {
059                                String attributeValue = attributes.getValue(ATTRIBUTE_XSI_TYPE);
060                                if (VALUE_XSI_TYPE_SCRIPTED_CALCULATION_VIEWS.equals(attributeValue)) {
061                                        checkFileExtension();
062                                }
063
064                        }
065
066                        protected void checkFileExtension() {
067
068                                String objectName = HanaViewsCheckUtil.getRepositoryObjectName(context);
069                                if (!objectName.matches(namingPatternRegex)) {
070                                        createElementFinding("Scripted calculation view '" + objectName + "' violates naming pattern '"
071                                                        + namingPatternRegex + "'");
072                                }
073                        }
074
075                };
076        }
077
078        /**
079         * 
080         */
081
082}
