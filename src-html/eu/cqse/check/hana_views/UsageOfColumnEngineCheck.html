008package eu.cqse.check.hana_views;
009
010import java.util.EnumMap;
011import java.util.HashSet;
012import java.util.List;
013import java.util.Set;
014
015import eu.cqse.check.framework.core.ACheck;
016import eu.cqse.check.framework.core.option.ACheckOption;
017import eu.cqse.check.framework.scanner.ELanguage;
018import eu.cqse.check.framework.scanner.ETokenType;
019import eu.cqse.check.framework.scanner.IToken;
020import eu.cqse.check.framework.scanner.ScannerUtils;
021import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
022import eu.cqse.check.xml.XmlCheckBase;
023import eu.cqse.check.xml.XmlCheckHandlerBase;
024import eu.cqse.check.xml.XmlElementValueCheckHandlerBase;
025
026/**
027 * Checks if expressions of column engine are used. By setting
028 * {@link #relevantTokenTypes} the check can be restricted to expressions which
029 * contain the specified token types.
030 */
031@ACheck(name = UsageOfColumnEngineCheck.CHECK_NAME, description = "Usage of Column Engine should be avoided, "
032                + "SQL Engine should be used instead.", languages = {
033                                ELanguage.HANA_VIEW }, groupName = IHanaViewCheckGroups.PERFORMANCE)
034public class UsageOfColumnEngineCheck extends XmlCheckBase {
035
036        /** Name of this check */
037        /* package */ static final String CHECK_NAME = "Usage of column engine";
038
039        /** XML Element name, refers to formula expressions */
040        private static final String ELEMENT_FORMULA = "formula";
041
042        /** XML Attribute name, refers to expression language */
043        private static final String ATTRIBUTE_EXPRESSION_LANGUAGE = "expressionLanguage";
044
045        /**
046         * Expected value of {@link #ATTRIBUTE_EXPRESSION_LANGUAGE} for expression
047         * language 'Column Engine'
048         */
049        private static final String LANGUAGE_COLUMN_ENGINE = "COLUMN_ENGINE";
050
051        /** {@ConQAT.Doc} */
052        @ACheckOption(name = CHECK_NAME
053                        + " - relevant tokens", description = "Comma separted list of relevant token types. This would restrict the check to only"
054                                        + "report findings for expressions containing one of the specified token types. If nothing is specified here, "
055                                        + "any usage of Column Engine is reported.")
056        private Set<String> relevantTokenTypeNames = new HashSet<String>();
057
058        /**
059         * {@link EnumMap} of relevant tokens, build from
060         * {@link #relevantTokenTypeNames} in {@link #initialize()}. It is a map to
061         * show the token type as as specified in the
062         * {@link #relevantTokenTypeNames}, since usually this will be specified in
063         * lower case.
064         */
065        private EnumMap<ETokenType, String> relevantTokenTypes = new EnumMap<ETokenType, String>(ETokenType.class);
066
067        /** {@inheritDoc} */
068        @Override
069        public void initialize() {
070                for (String tokenTypeName : relevantTokenTypeNames) {
071                        relevantTokenTypes.put(ETokenType.valueOf(tokenTypeName.toUpperCase()), tokenTypeName);
072                }
073        }
074
075        /** {@inheritDoc} */
076        @Override
077        public XmlCheckHandlerBase getHandler() {
078                return new XmlElementValueCheckHandlerBase(ELEMENT_FORMULA) {
079
080                        @Override
081                        protected void checkValue(String value) {
082                                if (!isColumnEngineLanguage()) {
083                                        return;
084                                }
085                                if (relevantTokenTypes.isEmpty()) {
086                                        createFinding("Usage of column engine expression.");
087                                }
088
089                                // value contains an expression, using JavaScript
090                                // scanner to get tokens from it (JavaScript accepts almost any
091                                // expression)
092                                List<IToken> tokens = ScannerUtils.getTokens(value, ELanguage.JAVASCRIPT);
093
094                                for (ETokenType tokenType : relevantTokenTypes.keySet()) {
095                                        if (TokenStreamUtils.containsAny(tokens, tokenType)) {
096                                                createFinding("Usage of column engine expression containing '"
097                                                                + relevantTokenTypes.get(tokenType) + "'.");
098                                        }
099                                }
100                        }
101
102                        /**
103                         * Check if language attribute at parent is COLUMN_ENGINE
104                         */
105                        private boolean isColumnEngineLanguage() {
106                                return LANGUAGE_COLUMN_ENGINE.equalsIgnoreCase(getParentAttributeValue(ATTRIBUTE_EXPRESSION_LANGUAGE));
107                        }
108                };
109        }
110}
