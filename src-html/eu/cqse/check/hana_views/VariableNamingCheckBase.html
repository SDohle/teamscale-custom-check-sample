008package eu.cqse.check.hana_views;
009
010import org.xml.sax.Attributes;
011
012import eu.cqse.check.xml.XmlCheckBase;
013import eu.cqse.check.xml.XmlCheckHandlerBase;
014import eu.cqse.check.xml.XmlElementAttributesCheckHandlerBase;
015
016/**
017 * Base class for checks of naming (XML attribute 'id') of variables (XML
018 * element 'variable').
019 */
020public abstract class VariableNamingCheckBase extends XmlCheckBase {
021
022        /** XML element, refers to a variable definition */
023        private static final String ELEMENT_VARIABLE = "variable";
024
025        /**
026         * XML attribute, refers to the parameter flag of {@link #ELEMENT_VARIABLE}
027         * to indicate input parameter variables
028         */
029        private static final String ATTRIBUTE_PARAMETER = "parameter";
030
031        /** {@inheritDoc} */
032        @Override
033        public XmlCheckHandlerBase getHandler() {
034                return new XmlElementAttributesCheckHandlerBase(ELEMENT_VARIABLE) {
035                        @Override
036                        protected void checkAttributes(Attributes attributes) {
037                                boolean isParameter = false;
038                                String parameter = attributes.getValue(ATTRIBUTE_PARAMETER);
039                                if (parameter != null) {
040                                        isParameter = Boolean.parseBoolean(parameter);
041                                }
042                                String id = attributes.getValue(HanaViewsCheckUtil.ATTRIBUTE_ID);
043                                if (isFinding(id, isParameter)) {
044                                        createFinding(getFindingMessage() + " ('" + id + "' does not match)");
045                                }
046                        }
047                };
048        }
049
050        /**
051         * Performs the check of the variable name.
052         * 
053         * @param isParameter
054         *            flag to indicate if the variable is an input parameter (XML
055         *            attribute 'parameter' is set):
056         */
057        protected abstract boolean isFinding(String id, boolean isParameter);
058
059        /** Gets the finding message. */
060        protected abstract String getFindingMessage();
061}
