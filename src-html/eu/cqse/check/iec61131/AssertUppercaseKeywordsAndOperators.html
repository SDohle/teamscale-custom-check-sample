008package eu.cqse.check.iec61131;
009
010import java.util.List;
011
012import eu.cqse.check.base.EntityTokenCheckBase;
013import eu.cqse.check.framework.core.ACheck;
014import eu.cqse.check.framework.core.CheckException;
015import eu.cqse.check.framework.core.CheckGroupNames;
016import eu.cqse.check.framework.core.ECheckParameter;
017import eu.cqse.check.framework.core.EFindingEnablement;
018import eu.cqse.check.framework.scanner.ELanguage;
019import eu.cqse.check.framework.scanner.IToken;
020
021/**
022 * {@ConQAT.Doc}
023 */
024@ACheck(name = "Assert uppercase keywords and operators", description = "All keywords and operators in Structured Text should be written in uppercase.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
025                ELanguage.IEC61131 }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
026public class AssertUppercaseKeywordsAndOperators extends EntityTokenCheckBase {
027
028        /** {@inheritDoc} */
029        @Override
030        public String getXPathSelectionString() {
031                return "//*";
032        }
033
034        /** {@inheritDoc} */
035        @Override
036        public void processTokens(List<IToken> tokens) throws CheckException {
037                for (IToken token : tokens) {
038                        if (!token.getType().isKeyword() && !token.getType().isOperator()) {
039                                continue;
040                        }
041                        String text = token.getText().trim();
042                        if (!isUpperCase(text)) {
043                                String prefix = "Keyword";
044                                if (token.getType().isOperator()) {
045                                        prefix = "Operator";
046                                }
047                                createFinding(
048                                                prefix + " should be written in uppercase (`" + text + "` -> `" + text.toUpperCase() + "`)",
049                                                token);
050                        }
051                }
052        }
053
054        /**
055         * <p>
056         * Checks whether the String contains only uppercase characters and '_').
057         * </p>
058         *
059         * Some examples:
060         *
061         * <pre>
062         * isUpperCase("STRING") = true
063         * isUpperCase("string") = false
064         * isUpperCase("sTRING") = false
065         * isUpperCase(" STRING ") = false
066         * isUpperCase("") = false
067         * isUpperCase("  ") = false
068         * isUpperCase(null) = false
069         * </pre>
070         *
071         * @return true if the String contains only uppercase letters and is not empty
072         *         and not null
073         */
074        public static boolean isUpperCase(String text) {
075                if (text == null || text.length() == 0) {
076                        return false;
077                }
078                return text.equals(text.toUpperCase());
079        }
080}
