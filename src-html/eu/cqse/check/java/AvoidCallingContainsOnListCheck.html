017package eu.cqse.check.java;
018
019import java.util.Arrays;
020import java.util.HashMap;
021import java.util.HashSet;
022import java.util.List;
023import java.util.Map;
024import java.util.Set;
025
026import org.conqat.lib.commons.string.StringUtils;
027
028import eu.cqse.check.base.EntityCheckBase;
029import eu.cqse.check.framework.core.ACheck;
030import eu.cqse.check.framework.core.CheckException;
031import eu.cqse.check.framework.core.CheckGroupNames;
032import eu.cqse.check.framework.core.ECheckParameter;
033import eu.cqse.check.framework.core.EFindingEnablement;
034import eu.cqse.check.framework.scanner.ELanguage;
035import eu.cqse.check.framework.scanner.ETokenType;
036import eu.cqse.check.framework.scanner.IToken;
037import eu.cqse.check.framework.shallowparser.SubTypeNames;
038import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
039import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
040import eu.cqse.check.framework.typetracker.ScopedTypeLookup;
041import eu.cqse.check.framework.typetracker.TypedVariable;
042
043/**
044 * Detects calls to contains() on java.util.List.
045 */
046@ACheck(name = "Calling contains on list", description = "Calling contains() on lists is inefficient. Use hashsets instead.", groupName = CheckGroupNames.PERFORMANCE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
047                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE, ECheckParameter.TYPE_RESOLUTION })
048public class AvoidCallingContainsOnListCheck extends EntityCheckBase {
049
050        /** String constant for message for finding */
051        private static final String FINDING_MESSAGE = "Calling contains() on lists is inefficient.";
052
053        /**
054         * Holds all imported types as a map from short type name to full qualified
055         * type name.
056         */
057        private Map<String, String> typeImports;
058
059        /** A set of Java classes that are list types. */
060        public static final Set<String> LIST_TYPES = new HashSet<String>(Arrays.asList("java.util.List",
061                        "java.util.AbstractList", "java.util.AbstractSequentialList", "java.util.ArrayList",
062                        "javax.management.AttributeList", "java.util.concurrency.CopyOnWriteArrayList", "java.util.LinkedList",
063                        "javax.management.relation.RoleList", "javax.management.relation.RoleUnresolvedList", "java.util.Stack",
064                        "java.util.Vector"));
065
066        /** {@inheritDoc} */
067        @Override
068        protected String getXPathSelectionString() {
069                return "//STATEMENT";
070        }
071
072        /** {@inheritDoc} */
073        @Override
074        public void execute() throws CheckException {
075                typeImports = new HashMap<String, String>();
076                List<ShallowEntity> importEntities = select("/META[subtype('" + SubTypeNames.IMPORT + "')]");
077                for (ShallowEntity entity : importEntities) {
078                        String qualifiedName = entity.getName();
079                        typeImports.put(StringUtils.getLastPart(qualifiedName, '.'), qualifiedName);
080                }
081
082                super.execute();
083        }
084
085        /** {@inheritDoc} */
086        @Override
087        protected void processEntity(ShallowEntity entity) throws CheckException {
088                List<IToken> tokens = entity.ownStartTokens();
089                List<Integer> indices = TokenStreamUtils.firstTokenOfTypeSequences(tokens, 0, ETokenType.DOT,
090                                ETokenType.IDENTIFIER, ETokenType.LPAREN);
091                ScopedTypeLookup lookup = context.getTypeResolution().getTypeLookup(entity);
092
093                for (int index : indices) {
094                        if (index == 0) {
095                                continue;
096                        }
097
098                        IToken variableNameOrMember = tokens.get(index - 1);
099                        IToken containsCallToken = tokens.get(index + 1);
100                        if (containsCallToken.getText().equals("contains")) {
101                                processVariableNameOrMember(tokens, lookup, variableNameOrMember, containsCallToken, index);
102                        }
103
104                }
105        }
106
107        /**
108         * Given that we have a contains call, further processing is carried out
109         * here to decide if the variable name or member on which the call was made
110         * is or returns a list.
111         */
112        private void processVariableNameOrMember(List<IToken> tokens, ScopedTypeLookup lookup, IToken variableNameOrMember,
113                        IToken containsCallToken, int index) throws CheckException {
114                switch (variableNameOrMember.getType()) {
115                case IDENTIFIER:
116                        TypedVariable typedVariable = lookup.getTypeInfo(variableNameOrMember.getText());
117                        if (typedVariable != null && isListType(typedVariable.getTypeName())) {
118                                createFinding(FINDING_MESSAGE, containsCallToken);
119                        }
120                        break;
121
122                case RPAREN:
123                        checkForSubListCall(tokens, containsCallToken, index - 2);
124                        break;
125
126                default:
127                        break;
128                }
129        }
130
131        /**
132         * Checks if the method call contains is made from the subList method of a
133         * list type. If so, a finding is created.
134         */
135        private void checkForSubListCall(List<IToken> tokens, IToken containsCallToken, int searchStartIndex)
136                        throws CheckException {
137                if (searchStartIndex < 0) {
138                        return;
139                }
140                int indexMatchingLparen = TokenStreamUtils.findMatchingOpeningToken(tokens, searchStartIndex, ETokenType.LPAREN,
141                                ETokenType.RPAREN);
142                if (indexMatchingLparen < 2) {
143                        return;
144                }
145
146                if (tokens.get(indexMatchingLparen - 2).getType() == ETokenType.DOT
147                                && tokens.get(indexMatchingLparen - 1).getText().equals("subList")) {
148                        createFinding(FINDING_MESSAGE, containsCallToken);
149                }
150        }
151
152        /**
153         * Determines if the given type name (short class name or full qualified) is
154         * a list type.
155         */
156        private boolean isListType(String typeName) {
157                if (LIST_TYPES.contains(typeName)) {
158                        return true;
159                }
160                String qualifiedName = typeImports.get(typeName);
161                return qualifiedName != null && LIST_TYPES.contains(qualifiedName);
162        }
163}
