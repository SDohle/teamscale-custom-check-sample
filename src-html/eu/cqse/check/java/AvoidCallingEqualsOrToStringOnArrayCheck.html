047/**
048 * Detects direct and implicit calls of equals()/toString() on arrays.
049 */
050@ACheck(name = "Calling equals()/toString() on array", description = AvoidCallingEqualsOrToStringOnArrayCheck.DESCRIPTION, groupName = CheckGroupNames.CORRECTNESS, defaultEnablement = EFindingEnablement.YELLOW, languages = {
051                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE, ECheckParameter.TYPE_RESOLUTION })
052public class AvoidCallingEqualsOrToStringOnArrayCheck extends EntityCheckBase {
053
054        /** Description of this finding. */
055        protected static final String DESCRIPTION = "Calling `equals()` on an array does not check the contents of the array "
056                        + "for equality, but its memory addresses. "
057                        + "In the same way calling `toString()` on an array will not return the String representation of the content of the "
058                        + "array, but the memory address. Therefore this is likely a bug. ";
059
060        /** A set of the invalid method calls on arrays that produce a finding. */
061        private static final Set<String> INVALID_METHOD_CALLS_ON_ARRAY = CollectionUtils.asUnmodifiableHashSet("equals",
062                        "toString");
063
064        /** A set of valid method calls on arrays that don't return an array. */
065        private static final Set<String> VALID_METHOD_CALLS_ON_ARRAY = CollectionUtils.asUnmodifiableHashSet("length",
066                        "getClass", "hashCode", "iterator");
067
068        /** A set of methods returning an array. */
069        private static final Set<String> METHODS_WITH_RETURNTYPE_ARRAY = CollectionUtils.asUnmodifiableHashSet("copyOf",
070                        "copyOfRange", "getStackTrace", "values");
071
072        /** A set with names of types which have methods returning an array. */
073        private static final Set<String> TYPE_NAMES = CollectionUtils.asUnmodifiableHashSet("Throwable", "Exception",
074                        "Error");
075
076        /** Name of the class reference to manipulate arrays. */
077        private static final String CLASS_REFERENCE = "Arrays";
078
079        /** Set of Equality and Relational Operators. */
080        private static final Set<ETokenType> EQUALITY_AND_RELATIONAL_OPERATORS = CollectionUtils.asUnmodifiableHashSet(EQEQ,
081                        NOTEQ, GT, LT, GTEQ, LTEQ);
082
083        /** {@inheritDoc} */
084        @Override
085        protected String getXPathSelectionString() {
086                return "//STATEMENT";
087        }
088
089        /** {@inheritDoc} */
090        @Override
091        protected void processEntity(ShallowEntity entity) throws CheckException {
092                ScopedTypeLookup typeLookup = context.getTypeResolution(ECodeViewOption.FILTERED).getTypeLookup(entity);
093
094                List<IToken> tokens = entity.ownStartTokens();
095
096                checkForDirectCallOnVariableNameOrMember(typeLookup, tokens);
097
098                checkForDirectCallOnMethod(tokens);
099
100                if (isImplicitCallOfToString(tokens)) {
101                        processImplicitCallOfToString(typeLookup, tokens);
102                }
103        }
104
105        /**
106         * Checks if an equals() or toString() call was made on a variable name or
107         * member.
108         */
109        private void checkForDirectCallOnVariableNameOrMember(ScopedTypeLookup typeLookup, List<IToken> tokens)
110                        throws CheckException {
111                List<Integer> indices = TokenStreamUtils.firstTokenOfTypeSequences(tokens, 0, IDENTIFIER, DOT, IDENTIFIER,
112                                LPAREN);
113                for (int index : indices) {
114                        IToken variableNameOrMember = tokens.get(index);
115                        IToken methodInvocationToken = tokens.get(index + 2);
116                        if (INVALID_METHOD_CALLS_ON_ARRAY.contains(methodInvocationToken.getText())) {
117                                processVariableNameOrMemberForDirectCall(typeLookup, variableNameOrMember, methodInvocationToken);
118                        }
119                }
120        }
121
122        /**
123         * Given that we have an equals() or toString() call, further processing is
124         * carried out here to decide if the variable name or member on which the call
125         * was made on is an array.
126         */
127        private void processVariableNameOrMemberForDirectCall(ScopedTypeLookup typeLookup, IToken variableNameOrMember,
128                        IToken methodInvocationToken) throws CheckException {
129                TypedVariable typedVariable = typeLookup.getTypeInfo(variableNameOrMember.getText());
130                if (typedVariable != null && typedVariable.getTypeName().endsWith("[]")) {
131                        createFinding("Direct call of `" + methodInvocationToken.getText() + "()` on array is likely a bug.",
132                                        methodInvocationToken);
133                }
134        }
135
136        /**
137         * Checks if an equals() or toString() call was made on a method.
138         */
139        private void checkForDirectCallOnMethod(List<IToken> tokens) throws CheckException {
140                List<Integer> indices = TokenStreamUtils.firstTokenOfTypeSequences(tokens, 1, DOT, IDENTIFIER, LPAREN);
141                for (int index : indices) {
142                        IToken methodInvocationToken = tokens.get(index + 1);
143                        if (METHODS_WITH_RETURNTYPE_ARRAY.contains(methodInvocationToken.getText())) {
144                                if (isSurroundedWithToStringOrModified(tokens, index - 1)) {
145                                        continue;
146                                }
147                                IToken classReferenceOrTypeName = tokens.get(index - 1);
148                                int indexOfSucceedingMethodCall = TokenStreamUtils.firstTokenOfTypeSequence(tokens, index + 1, DOT,
149                                                IDENTIFIER, LPAREN);
150                                IToken succeedingMethodInvocationToken = tokens.get(indexOfSucceedingMethodCall + 1);
151                                if (INVALID_METHOD_CALLS_ON_ARRAY.contains(succeedingMethodInvocationToken.getText())) {
152                                        processMethodCallForDirectCall(methodInvocationToken, succeedingMethodInvocationToken,
153                                                        classReferenceOrTypeName);
154                                }
155                        }
156                }
157        }
158
159        /**
160         * Checks if the method call on which one of the methods from
161         * {@link #INVALID_METHOD_CALLS_ON_ARRAY} is called on, returns an array
162         * {@value #METHODS_WITH_RETURNTYPE_ARRAY}.
163         */
164        private void processMethodCallForDirectCall(IToken methodInvocationToken, IToken succeedingMethodInvocationToken,
165                        IToken classReferenceOrTypeName) throws CheckException {
166                if (isOfTypeThrowableOrSubclass(classReferenceOrTypeName.getText())
167                                || isStaticArraysCall(classReferenceOrTypeName.getText())
168                                || isEnum(classReferenceOrTypeName.getText())) {
169                        createFinding(
170                                        "Direct call of `" + succeedingMethodInvocationToken.getText()
171                                                        + "()` on array returned by call to '" + classReferenceOrTypeName.getText() + "."
172                                                        + methodInvocationToken.getText() + "()' is likely a bug.",
173                                        succeedingMethodInvocationToken);
174                }
175        }
176
177        /**
178         * Checks if the variable name or member on which a method is called on contains
179         * one of the keywords {@value #TYPE_NAMES} in its type name.
180         */
181        private static boolean isOfTypeThrowableOrSubclass(String typeName) {
182                for (String keyword : TYPE_NAMES) {
183                        if (typeName.endsWith(keyword)) {
184                                return true;
185                        }
186                }
187                return false;
188        }
189
190        /**
191         * Checks if the module {@link #CLASS_REFERENCE} was used to modify an object.
192         */
193        private static boolean isStaticArraysCall(String typeName) {
194                return typeName.equals(CLASS_REFERENCE);
195        }
196
197        private static boolean isEnum(String typeName) {
198                return typeName.length() > 1 && Character.isUpperCase(typeName.charAt(1));
199        }
200
201        /**
202         * Decides whether the implicit call of toString() was made as an initialization
203         * of a variable, after a return statement or as a parameter for a method call.
204         */
205        private void processImplicitCallOfToString(ScopedTypeLookup typeLookup, List<IToken> tokens) throws CheckException {
206                if (TokenStreamUtils.contains(tokens, EQ)) {
207                        List<IToken> tokensAfterEq = TokenStreamUtils.tokensBetween(tokens, EQ, SEMICOLON);
208                        checksForUseOfMethod(typeLookup, tokensAfterEq);
209                } else if (TokenStreamUtils.contains(tokens, RETURN)) {
210                        List<IToken> tokensAfterReturn = TokenStreamUtils.tokensBetween(tokens, ETokenType.RETURN, SEMICOLON);
211                        checksForUseOfMethod(typeLookup, tokensAfterReturn);
212                } else {
213                        // extract all parameter token lists, i.e. tokens between parentheses
214                        int start = TokenStreamUtils.firstTokenOfType(tokens, LPAREN);
215                        while (start != NOT_FOUND) {
216                                int end = TokenStreamUtils.findMatchingClosingToken(tokens, start + 1, LPAREN, RPAREN);
217                                if (end == NOT_FOUND) {
218                                        break;
219                                }
220
221                                List<IToken> methodParameterTokens = tokens.subList(start + 1, end);
222                                if (!methodParameterTokens.isEmpty()) {
223                                        checkParametersOfMethod(typeLookup, methodParameterTokens);
224                                }
225
226                                start = TokenStreamUtils.firstTokenOfType(tokens, end + 1, LPAREN);
227                        }
228
229                }
230        }
231
232        /**
233         * Checks whether the initialization/return statement was realized with a method
234         * returning an object and therefore the parameters of the used method need to
235         * be checked for the implicit call of toString() or if the plus operator was
236         * used directly.
237         */
238        private void checksForUseOfMethod(ScopedTypeLookup typeLookup, List<IToken> tokensAfterEqualsOrReturn)
239                        throws CheckException {
240                List<Integer> indices = TokenStreamUtils.firstTokenOfTypeSequences(tokensAfterEqualsOrReturn, 0, IDENTIFIER,
241                                LPAREN);
242                boolean isNoMethodCall = false;
243                int lastIndex = 0;
244                for (int i = 0; i < indices.size(); i++) {
245                        int matchingClosingToken = TokenStreamUtils.findMatchingClosingToken(tokensAfterEqualsOrReturn,
246                                        indices.get(i) + 2, LPAREN, RPAREN);
247                        if (TokenStreamUtils.containsSequence(tokensAfterEqualsOrReturn, lastIndex, indices.get(i), STRING_LITERAL,
248                                        PLUS)
249                                        || (i < indices.size() - 1 && i != 0 && TokenStreamUtils.containsSequence(tokensAfterEqualsOrReturn,
250                                                        matchingClosingToken, indices.get(i + 1), PLUS, STRING_LITERAL))) {
251                                isNoMethodCall = true;
252                                break;
253                        }
254                        List<IToken> methodParameterTokens = tokensAfterEqualsOrReturn.subList(indices.get(i) + 2,
255                                        matchingClosingToken);
256                        checkParametersOfMethod(typeLookup, methodParameterTokens);
257                        lastIndex = matchingClosingToken;
258                }
259                if (isNoMethodCall || indices.isEmpty()) {
260                        processVariableNameOrMemberForImplicitCall(typeLookup, tokensAfterEqualsOrReturn);
261                        processMethodCallForImplicitCall(typeLookup, tokensAfterEqualsOrReturn);
262                }
263        }
264
265        /**
266         * Checks every parameter of a method if a implicit call of toString() was made.
267         */
268        private void checkParametersOfMethod(ScopedTypeLookup typeLookup, List<IToken> tokens) throws CheckException {
269                List<List<IToken>> splittedTokensWithinMethod = TokenStreamUtils.split(tokens, COMMA);
270                for (List<IToken> currentList : splittedTokensWithinMethod) {
271                        if (isImplicitCallOfToString(currentList)) {
272                                processVariableNameOrMemberForImplicitCall(typeLookup, currentList);
273                                processMethodCallForImplicitCall(typeLookup, currentList);
274                        }
275                }
276        }
277
278        /**
279         * Checks if one or more of the tokens in the given list is of type array. If
280         * the array was modified with one of the methods from the module Arrays which
281         * return an array, the finding will be created in
282         * {@link #processMethodCallForImplicitCall(ScopedTypeLookup, List)}. If the
283         * array is used as parameter for any other method call no finding will be
284         * created.
285         */
286        private void processVariableNameOrMemberForImplicitCall(ScopedTypeLookup typeLookup, List<IToken> tokens)
287                        throws CheckException {
288                for (int i = 0; i < tokens.size(); i++) {
289                        TypedVariable typedVariable = typeLookup.getTypeInfo(tokens.get(i).getText());
290                        if (typedVariable == null || !typedVariable.getTypeName().endsWith("[]")) {
291                                continue;
292                        }
293                        if ((!isSurroundedWithToStringOrModified(tokens, i) || i == 0)
294                                        && (!isSingleElement(tokens, i) || i == tokens.size() - 1)) {
295                                createFinding("Implicit call to `toString()` on array `" + typedVariable.getVariableName()
296                                                + "` is likely a bug.", tokens.get(i));
297                        }
298                }
299        }
300
301        /**
302         * Checks if the whole array is used, or only one single element of it. As well
303         * as if a method was used on the array which doesn't return an array. We allow
304         * toString() here, so only the finding for the direct call of toString() will
305         * create a finding.
306         */
307        private static boolean isSingleElement(List<IToken> tokens, int i) {
308                if (i + 4 < tokens.size() - 1 && i + 4 > 0 && tokens.get(i + 4 - 1).getType() == LPAREN) {
309                        int matchingClosingToken = TokenStreamUtils.findMatchingClosingToken(tokens, i + 4, LPAREN, RPAREN);
310                        i = matchingClosingToken;
311                }
312                if (i < tokens.size() - 2 && (VALID_METHOD_CALLS_ON_ARRAY.contains(tokens.get(i + 2).getText())
313                                || tokens.get(i + 2).getText().equals("toString")
314                                || EQUALITY_AND_RELATIONAL_OPERATORS.contains(tokens.get(i + 1).getType()))) {
315                        return true;
316                }
317                if (i < tokens.size() - 1 && tokens.get(i + 1).getType() == LBRACK) {
318                        return true;
319                }
320                return false;
321        }
322
323        /**
324         * Checks if the array is used as a parameter in a method or if it is modified
325         * with one of the methods from the module Arrays which return an array.
326         */
327        private static boolean isSurroundedWithToStringOrModified(List<IToken> tokens, int i) {
328                if (i > 1 && (TokenStreamUtils.containsSequence(tokens, i - 2, i, IDENTIFIER, LPAREN)
329                                || METHODS_WITH_RETURNTYPE_ARRAY.contains(tokens.get(i - 2).getText()))) {
330                        return true;
331                }
332                if (i > 3 && tokens.get(i - 2).getText().equals("this")
333                                && (TokenStreamUtils.containsSequence(tokens, i - 4, i - 2, IDENTIFIER, LPAREN)
334                                                || METHODS_WITH_RETURNTYPE_ARRAY.contains(tokens.get(i - 4).getText()))) {
335                        return true;
336                }
337                return false;
338        }
339
340        /**
341         * Checks if one or more of the tokens in the given list is a method which
342         * returns an array. If the method which returns an array is used as a parameter
343         * in a method no finding will be created.
344         */
345        private void processMethodCallForImplicitCall(ScopedTypeLookup typeLookup, List<IToken> tokens)
346                        throws CheckException {
347                for (int i = 0; i < tokens.size() - 2; i++) {
348                        IToken variableNameOrMember = tokens.get(i);
349                        IToken methodInvocationToken = tokens.get(i + 2);
350                        TypedVariable typedVariable = typeLookup.getTypeInfo(variableNameOrMember.getText());
351                        String typeName;
352                        if (typedVariable != null) {
353                                typeName = typedVariable.getTypeName();
354                        } else {
355                                typeName = variableNameOrMember.getText();
356                        }
357                        if ((!isOfTypeThrowableOrSubclass(typeName) || !isStaticArraysCall(typeName) || !isEnum(typeName))
358                                        && ((typedVariable == null && !typeName.equals(CLASS_REFERENCE))
359                                                        || !METHODS_WITH_RETURNTYPE_ARRAY.contains(methodInvocationToken.getText())
360                                                        || !isMethod(typeLookup, methodInvocationToken))) {
361                                continue;
362                        }
363                        if ((!isSurroundedWithToStringOrModified(tokens, i) || i == 0)
364                                        && (!isSingleElement(tokens, i) || i == tokens.size() - 1)) {
365                                createFinding(
366                                                "Implicit call to `toString()` on array returned by call to '" + variableNameOrMember.getText()
367                                                                + "." + methodInvocationToken.getText() + "()' is likely a bug.",
368                                                methodInvocationToken);
369                        }
370
371                }
372        }
373
374        /** Checks if the given methodInvocationToken is a method and no variable. */
375        private static boolean isMethod(ScopedTypeLookup typeLookup, IToken methodInvocationTokenText) {
376                return typeLookup.getTypeInfo(methodInvocationTokenText.getText()) == null;
377        }
378
379        /** Checks if a String concatenation was made using the + operator. */
380        private static boolean isImplicitCallOfToString(List<IToken> tokens) {
381                return TokenStreamUtils.containsAll(tokens, STRING_LITERAL, PLUS);
382        }
