040/** Checks whether 'Thread.sleep' was called inside a test method. */
041@ACheck(description = AvoidCallingThreadSleepInTestsCheck.CHECK_DESCRIPTION, groupName = BAD_PRACTICE, languages = {
042                JAVA }, name = AvoidCallingThreadSleepInTestsCheck.CHECK_NAME, parameters = { ABSTRACT_SYNTAX_TREE,
043                                TYPE_RESOLUTION })
044public class AvoidCallingThreadSleepInTestsCheck extends EntityCheckBase {
045
046        /** The check's name. */
047        public static final String CHECK_NAME = "Avoid calling `Thread.sleep` inside test methods";
048
049        /** The check's description. */
050        public static final String CHECK_DESCRIPTION = "Calling 'Thread.sleep' in tests is discouraged, as it introduces unpredictable test results. Consider using libraries such as [Awaitility](https://github.com/awaitility/awaitility) instead.";
051
052        /**
053         * The mapping from the simple name of an imported class to its fully qualified
054         * one.
055         */
056        private Map<String, String> typeImports;
057
058        /** The name of the 'Thread' class. */
059        private static final String THREAD_CLASS_NAME = "Thread";
060
061        /** The name of the 'sleep' method. */
062        private static final String SLEEP_METHOD_NAME = "sleep";
063
064        /** {@ConQAT.Doc} */
065        @ACheckOption(name = CHECK_NAME
066                        + ": Test class name pattern", description = "The naming pattern for test classes. Default is \".*Test$\".")
067        private String testClassNamePattern = ".*Test$";
068
069        /** {@inheritDoc} */
070        @Override
071        protected String getXPathSelectionString() {
072                return "//TYPE[" + NAME + "('" + testClassNamePattern + "')]";
073        }
074
075        /** {@inheritDoc} */
076        @Override
077        protected void processEntity(ShallowEntity classEntity) throws CheckException {
078                List<ShallowEntity> testMethods = classEntity.getChildrenOfType(METHOD).stream()
079                                .filter(entity -> isTestMethod(entity)).collect(Collectors.toList());
080
081                for (ShallowEntity testMethod : testMethods) {
082                        List<ShallowEntity> statements = testMethod.getChildrenOfType(STATEMENT);
083                        for (ShallowEntity statement : statements) {
084                                processCandidateStatement(statement, testMethod.getName());
085                        }
086                }
087        }
088
089        /**
090         * Considers a single statement shallow entity found inside a JUnit test method
091         * and creates a finding if it is a 'Thread.sleep' call.
092         * 
093         * @param statement
094         *            the shallow entity to be processed
095         * @throws CheckException
096         */
097        private void processCandidateStatement(ShallowEntity statement, String methodName) throws CheckException {
098                List<IToken> tokens = statement.ownStartTokens();
099                List<Integer> indices = TokenStreamUtils.firstTokenOfTypeSequences(tokens, 0, DOT, IDENTIFIER, LPAREN);
100
101                for (int index : indices) {
102                        if (index < 1) {
103                                // should not be the case unless there is a compilation error
104                                continue;
105                        }
106                        IToken callerToken = tokens.get(index - 1);
107                        IToken methodNameToken = tokens.get(index + 1);
108
109                        if (callerToken.getText().equals(THREAD_CLASS_NAME)
110                                        && methodNameToken.getText().equals(SLEEP_METHOD_NAME)) {
111                                createFinding("`Thread.sleep` called inside test method `" + methodName + "`", statement);
112                        }
113                }
114        }
115
116        /** {@inheritDoc} */
117        @Override
118        public void execute() throws CheckException {
119                processTypeImports();
120                super.execute();
121        }
122
123        /**
124         * Processes the type imports in the current file and initializes the type im
125         * port map.
126         * 
127         * @throws CheckException
128         */
129        private void processTypeImports() throws CheckException {
130                typeImports = new HashMap<>();
131                List<ShallowEntity> typeImportEntities = select("/META[subtype('" + IMPORT + "')]");
132
133                for (ShallowEntity typeImportEntity : typeImportEntities) {
134                        String qualifiedName = typeImportEntity.getName();
135                        typeImports.put(StringUtils.getLastPart(qualifiedName, '.'), qualifiedName);
136                }
137        }
138
139        /**
140         * Checks if a given entity is a JUnit test method.
141         * 
142         * @param method
143         *            the shallow entity to be checked
144         * @return whether the given method is a JUnit test method
145         */
146        private boolean isTestMethod(ShallowEntity method) {
147                CCSMAssert.isTrue(method.getType() == METHOD, "Expected method entity, but got other.");
148                return LanguageFeatureParser.JAVA.getAnnotations(method).stream()
149                                .anyMatch(annotation -> isTestAnnotation(annotation));
150        }
151
152        /**
153         * Checks if a given shallow entity represents a Java annotation.
154         * 
155         * @param annotation
156         *            the shallow entity to be checked
157         * @return whether the given entity is a Java annotation
158         */
159        private boolean isTestAnnotation(ShallowEntity annotation) {
160                CCSMAssert.isTrue(annotation.getSubtype().equals(ANNOTATION), "Expected annotation entity, but got other.");
161                String annotationName = annotation.getName();
162
163                if (!TEST_ANNOTATION_SIMPLE_NAMES.contains(annotationName)) {
164                        return false;
165                }
166
167                String fullyQualifiedAnnotationName = typeImports.get(annotationName);
168
169                if (fullyQualifiedAnnotationName == null
170                                || !TEST_ANNOTATION_QUALIFIED_NAMES.contains(fullyQualifiedAnnotationName)) {
171                        return false;
172                }
173
174                return true;
175        }
