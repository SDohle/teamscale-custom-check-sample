006package eu.cqse.check.java;
007
008import java.util.Set;
009
010import org.conqat.lib.commons.collections.CollectionUtils;
011
012import eu.cqse.check.framework.core.ACheck;
013import eu.cqse.check.framework.core.CheckGroupNames;
014import eu.cqse.check.framework.core.ECheckParameter;
015import eu.cqse.check.framework.core.EFindingEnablement;
016import eu.cqse.check.framework.scanner.ELanguage;
017
018/**
019 * Checks, whether results of specific methods of the Immutable BigDecimal or
020 * BigInteger are ignored.
021 */
022@ACheck(name = "Avoid ignoring method results of immutable BigDecimal and BigInteger methods", description = "Certain methods of BigDecimal or BigInteger require making use of their results otherwise the method call has no effect.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.RED, languages = {
023                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE, ECheckParameter.TYPE_RESOLUTION })
024public class AvoidIgnoringBigDecimalsAndBigIntegersMethodsResultsCheck
025                extends AvoidIgnoringImmutablesMethodsResultsCheckBase {
026        private static final String FINDING_MESSAGE = "Avoid ignoring method results of immutable BigDecimal and BigInteger methods";
027
028        /** {@inheritDoc} */
029        @Override
030        protected Set<String> getImmutableTypeNames() {
031                return CollectionUtils.asHashSet("BigDecimal", "BigInteger");
032        }
033
034        /** {@inheritDoc} */
035        @Override
036        protected Set<String> getImmutablesMethodsNames() {
037                return CollectionUtils.asHashSet("add", "subtract", "multiply", "divide");
038        }
039
040        /** {@inheritDoc} */
041        @Override
042        protected String getFindingMessage() {
043                return FINDING_MESSAGE;
044        }
045
046}
