006package eu.cqse.check.java;
007
008import java.util.Collections;
009import java.util.HashSet;
010import java.util.List;
011import java.util.Set;
012
013import org.conqat.lib.commons.collections.UnmodifiableList;
014
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckImplementationBase;
017import eu.cqse.check.framework.core.phase.ECodeViewOption;
018import eu.cqse.check.framework.scanner.ETokenType;
019import eu.cqse.check.framework.scanner.IToken;
020import eu.cqse.check.framework.shallowparser.SubTypeNames;
021import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
022import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
023import eu.cqse.check.framework.shallowparser.framework.ShallowEntityTraversalUtils;
024import eu.cqse.check.framework.typetracker.ScopedTypeLookup;
025import eu.cqse.check.framework.typetracker.TypedVariable;
026
027/**
028 * Base class for checking Immutables in Java for ignored methods results.
029 */
030public abstract class AvoidIgnoringImmutablesMethodsResultsCheckBase extends CheckImplementationBase {
031
032        private Set<String> relevantVariableNames;
033
034        /** {@inheritDoc} */
035        @Override
036        public void execute() throws CheckException {
037                collectRelevantVariables();
038                if (relevantVariableNames.isEmpty()) {
039                        return;
040                }
041                List<ShallowEntity> statements = ShallowEntityTraversalUtils.listEntitiesOfType(
042                                Collections.singleton(context.getRootEntity(getCodeViewOption())), EShallowEntityType.STATEMENT);
043                for (ShallowEntity statement : statements) {
044                        if (relevantVariableNames.contains(statement.getName())) {
045                                processEntity(statement);
046                        }
047                }
048        }
049
050        /**
051         * Collects all relevant Attributes, Variables and their names in two select
052         * commands.
053         */
054        private void collectRelevantVariables() throws CheckException {
055                relevantVariableNames = new HashSet<>();
056                for (ShallowEntity entity : select("//ATTRIBUTE")) {
057                        relevantVariableNames.add(entity.getName());
058                }
059                List<ShallowEntity> relevantLocalVariables = select(
060                                "//STATEMENT[subtype('" + SubTypeNames.LOCAL_VARIABLE + "')]");
061                for (ShallowEntity entity : relevantLocalVariables) {
062                        relevantVariableNames.add(entity.getName());
063                }
064        }
065
066        /** Processes a single selected entity. */
067        protected void processEntity(ShallowEntity entity) throws CheckException {
068                UnmodifiableList<IToken> entityTokens = entity.includedTokens();
069                for (int i = 0; i < entityTokens.size(); i++) {
070                        // if the token after the current does not exist, it cannot be a
071                        // potentially problematic method, as the current is a DOT.
072                        if ((i + 1) >= entityTokens.size()) {
073                                return;
074                        }
075                        // checks if the statement looks similar to someVariable = var.someMethod()
076                        // which would be okay.
077                        if (i >= 3) {
078                                ETokenType indexTwoEarlierTokenType = entityTokens.get(i - 2).getType();
079                                String indexThreeEarlierTokenText = entityTokens.get(i - 3).getText();
080                                if (indexTwoEarlierTokenType == ETokenType.EQ && isRelevantVariable(indexThreeEarlierTokenText)) {
081                                        return;
082                                }
083                        }
084                        IToken currentToken = entityTokens.get(i);
085                        IToken nextToken = entityTokens.get(i + 1);
086                        if (currentToken.getType() == ETokenType.DOT && isRelevantMethod(nextToken.getText())
087                                        && isCheckedImmutable(entity)) {
088                                createFindingOnFirstLine(getFindingMessage(), entity);
089                        }
090                }
091        }
092
093        /**
094         * Checks whether the entity is referring to a local variable or an attribute
095         * and whether it is to be considered in the check.
096         */
097        private boolean isCheckedImmutable(ShallowEntity entity) throws CheckException {
098                ScopedTypeLookup entitiesTypeResolution = context.getTypeResolution(ECodeViewOption.FILTERED)
099                                .getTypeLookup(entity);
100                TypedVariable typedVariable = entitiesTypeResolution.getTypeInfo(entity.getName());
101                if (typedVariable == null) {
102                        return false;
103                }
104                return getImmutableTypeNames().contains(typedVariable.getTypeName());
105        }
106
107        /** Checks whether the given name appears in the defined Methods. */
108        private boolean isRelevantMethod(String name) {
109                return getImmutablesMethodsNames().contains(name);
110        }
111
112        private boolean isRelevantVariable(String name) {
113                return relevantVariableNames.contains(name);
114        }
115
116        /**
117         * Getter for the Types of Attributes and Local Variables to be checked, e.g.
118         * String or BigInteger.
119         */
120        protected abstract Set<String> getImmutableTypeNames();
121
122        /**
123         * Getter for the Methods the checked Immutables enable (and are checked for
124         * useless usages).
125         */
126        protected abstract Set<String> getImmutablesMethodsNames();
127
128        /** Getter for the Finding Message. */
129        protected abstract String getFindingMessage();
130}
