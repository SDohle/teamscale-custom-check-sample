009import java.util.Set;
010
011import org.conqat.lib.commons.collections.CollectionUtils;
012import org.conqat.lib.commons.collections.UnmodifiableList;
013
014import eu.cqse.check.base.EntityCheckBase;
015import eu.cqse.check.framework.core.ACheck;
016import eu.cqse.check.framework.core.CheckException;
017import eu.cqse.check.framework.core.CheckGroupNames;
018import eu.cqse.check.framework.core.ECheckParameter;

022import eu.cqse.check.framework.scanner.ETokenType;
023import eu.cqse.check.framework.scanner.IToken;
024import eu.cqse.check.framework.shallowparser.SubTypeNames;
025import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
026import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
027import eu.cqse.check.framework.typetracker.ScopedTypeLookup;
028
029/**
030 * Checks for deprecated hash algorithms based on SHA-1 and Message-Digest that
031 * are now considered insecure.
032 */
033@ACheck(name = "Avoid insecure SHA-1 and Message-Digest hash algorithms", description = "SHA-1 and Message-Digest hash algorithms should not be used in secure contexts.", groupName = CheckGroupNames.SECURITY, defaultEnablement = EFindingEnablement.RED, languages = {
034                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE, ECheckParameter.TYPE_RESOLUTION })
035public class AvoidInsecureHashAlgorithmsCheck extends EntityCheckBase {
036        /**
037         * Finding Message that is called when a class using an insecure hash algorithm
038         * is used.
039         */
040        private static final String FINDING_MESSAGE_INSECURE_CLASS = "Avoid usage of classes based on insecure hash algorithms like SHA1 and MD5.";
041
042        /** Finding Message that is called when an insecure hash algorithm is used */
043        private static final String FINDING_MESSAGE_INSECURE_ALGORITHM = "Avoid usage of insecure hash algorithms like SHA1 and MD5.";
044
045        /** Set of Classes that enable insecure as well as secure hash algorithms */
046        private static final Set<String> USABLE_HASH_ALGORITHM_CLASSES = CollectionUtils.asHashSet("Hashing",
047                        "KeyGenerator", "DigestUtils", "Mac", "KeyPairGenerator", "KeyFactory", "AlgorithmParameters",
048                        "AlgorithmParameterGenerator", "MessageDigest", "Signature");
049
050        /**
051         * Set of Classes that enable only insecure hash algorithms or are generally
052         * considered insecure
053         */
054        private static final Set<String> ALL_HASH_ALGORITHM_CLASSES = CollectionUtils.asHashSet("Hashing", "KeyGenerator",
055                        "DigestUtils", "Mac", "KeyPairGenerator", "KeyFactory", "AlgorithmParameters",
056                        "AlgorithmParameterGenerator", "MessageDigest", "Signature", "ShaPasswordEncoder", "Md5PasswordEncoder",
057                        "LdapShaPasswordEncoder", "Md4PasswordEncoder", "MessageDigestPasswordEncoder", "NoOpPasswordEncoder",
058                        "StandardPasswordEncoder");
059
060        /** Set of hash algorithms considered insecure */
061        private static final Set<String> INSECURE_HASH_ALGORITHMS = CollectionUtils.asHashSet("\"SHA1\"", "\"MD5\"",
062                        "\"MD2\"", "\"MD4\"", "\"MD6\"", "\"HAVAL-128\"", "\"HMAC-MD5\"", "\"DSA\"", "\"RIPEMD\"", "\"RIPEMD-128\"",
063                        "\"RIPMED-160\"", "\"HMACRIPEMD160\"");
064
065        /** Set of variables with the value of a problematic hash algorithm */
066        private final Set<ShallowEntity> insecureVariables = new HashSet<>();
067
068        @Override
069        protected String getXPathSelectionString() {
070                return "//STATEMENT | //ATTRIBUTE";
071        }
072
073        @Override
074        protected void processEntity(ShallowEntity entity) throws CheckException {
075
076                UnmodifiableList<IToken> entityTokens = entity.includedTokens();
077                if (entityTokens.isEmpty()) {
078                        return;
079                }
080                String firstTokenText = entityTokens.get(0).getText();
081                // Checks for a) hash Algorithms that can still be used in secure contexts,
082                // b) deprecated hash algorithms
083                // c) attributes and local variables pointing to insecure hash algorithms
084                if (USABLE_HASH_ALGORITHM_CLASSES.contains(firstTokenText)) {
085                        checkForInsecureAlgorithmReference(entity, entityTokens);
086                } else if (ALL_HASH_ALGORITHM_CLASSES.contains(firstTokenText)) {
087                        createFindingOnFirstLine(FINDING_MESSAGE_INSECURE_CLASS, entity);
088                } else if (entity.getSubtype().equals(SubTypeNames.LOCAL_VARIABLE)
089                                || entity.getType().equals(EShallowEntityType.ATTRIBUTE)) {
090                        for (IToken token : entityTokens) {
091                                if (token.getType() == ETokenType.STRING_LITERAL
092                                                && INSECURE_HASH_ALGORITHMS.contains(token.getText())) {
093                                        insecureVariables.add(entity);
094                                }
095                        }
096                }
097        }
098
099        /**
100         * Checks whether the class references an insecure algorithm directly, by local
101         * variable or by attribute
102         */
103        private void checkForInsecureAlgorithmReference(ShallowEntity entity, UnmodifiableList<IToken> entityTokens)
104                        throws CheckException {
105                ScopedTypeLookup entitiesTypeResolution = context.getTypeResolution(ECodeViewOption.FILTERED)
106                                .getTypeLookup(entity);
107
108                for (IToken token : entityTokens) {
109                        String maybeHashAlgorithm = token.getText();
110                        if (INSECURE_HASH_ALGORITHMS.parallelStream().anyMatch(maybeHashAlgorithm::contains)) {
111                                createFindingOnFirstLine(FINDING_MESSAGE_INSECURE_ALGORITHM, entity);
112                        }
113                        if (entitiesTypeResolution.containsVariable(maybeHashAlgorithm) && insecureVariables
114                                        .contains(entitiesTypeResolution.getTypeInfo(maybeHashAlgorithm).getDeclaringEntity())) {
115                                createFindingOnFirstLine(FINDING_MESSAGE_INSECURE_ALGORITHM, entity);
116
117                        }
118
119                }
120        }
121
