044/**
045 * {@ConQAT.Doc}
046 */
047@ACheck(name = "Avoid reassigning parameter value", description = "Reassigning values to method parameters is not recommended. Consider to use temporary local variables instead.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.OFF, languages = {
048                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
049public class AvoidReassigningParametersCheck extends EntityCheckBase {
050
051        /** Assignment operators in Java */
052        private static final EnumSet<ETokenType> ASSIGNMENT_OPERATORS = EnumSet.of(EQ, MINUSEQ, MINUSMINUS, PLUSEQ,
053                        PLUSPLUS, PLUSEQ, MINUSEQ, MULTEQ, DIVEQ, ANDEQ, OREQ, XOREQ, MODEQ, LSHIFTEQ, RSHIFTEQ, URSHIFTEQ);
054        /**
055         * Assignment operators in Java that can be followed by the assigned variable.
056         */
057        private static final EnumSet<ETokenType> UNARY_PREFIX_OPERATORS = EnumSet.of(MINUSMINUS, PLUSPLUS);
058
059        /** {@inheritDoc} */
060        @Override
061        protected String getXPathSelectionString() {
062                return "//METHOD";
063        }
064
065        /** {@inheritDoc} */
066        @Override
067        protected void processEntity(ShallowEntity entity) throws CheckException {
068                UnmodifiableList<IToken> methodStartTokens = entity.ownStartTokens();
069                Set<String> parameterNames = TokenStreamUtils.findAll(methodStartTokens, EnumSet.of(IDENTIFIER)).stream()
070                                .map(methodStartTokens::get).map(IToken::getText).collect(Collectors.toSet());
071                List<ShallowEntity> primitiveStatements = ShallowParsingUtils.listPrimitiveStatements(entity.getChildren());
072
073                // checks whether statement contains change of variable value that is part
074                // of method parameters
075                for (ShallowEntity statement : primitiveStatements) {
076                        UnmodifiableList<IToken> statementTokens = statement.ownStartTokens();
077                        List<Integer> assignmentIndices = TokenStreamUtils.findAll(statementTokens, ASSIGNMENT_OPERATORS);
078                        for (int assignmentIndex : assignmentIndices) {
079                                ETokenType assignmentType = statementTokens.get(assignmentIndex).getType();
080                                if (UNARY_PREFIX_OPERATORS.contains(assignmentType) && assignmentIndex + 1 < statementTokens.size()) {
081                                        checkIfVariableEqualsToIdentifier(assignmentIndex + 1, statementTokens, parameterNames);
082                                }
083                                if (assignmentIndex - 1 >= 0) {
084                                        checkIfVariableEqualsToIdentifier(assignmentIndex - 1, statementTokens, parameterNames);
085                                }
086                        }
087                }
088        }
089
090        /**
091         * Checks whether token at the given index is a parameter from the given set of
092         * parameter names.
093         */
094        private void checkIfVariableEqualsToIdentifier(int assignedVariableTokenIndex,
095                        UnmodifiableList<IToken> statementTokens, Set<String> parameterNames) throws CheckException {
096                if (assignedVariableTokenIndex - 1 >= 0
097                                && statementTokens.get(assignedVariableTokenIndex - 1).getType() == ETokenType.DOT) {
098                        // this can't be a parameter. It must be a field of some object.
099                        return;
100                }
101                IToken assignedVariableToken = statementTokens.get(assignedVariableTokenIndex);
102                String variableName = assignedVariableToken.getText();
103                if (assignedVariableToken.getType() == IDENTIFIER && parameterNames.contains(variableName)) {
104                        createFinding("Reassigned value of the method parameter `" + variableName + "`", assignedVariableToken);
105                }
106        }
107}
