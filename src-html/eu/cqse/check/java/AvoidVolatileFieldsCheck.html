003import static eu.cqse.check.framework.core.CheckGroupNames.BAD_PRACTICE;
004import static eu.cqse.check.framework.core.ECheckParameter.ABSTRACT_SYNTAX_TREE;

007import static eu.cqse.check.framework.scanner.ELanguage.JAVA;

009import eu.cqse.check.base.EntityCheckBase;
010import eu.cqse.check.framework.core.ACheck;
011import eu.cqse.check.framework.core.CheckException;
012import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
013
014/**
015 * Check that discourages the use of Java/C# 'volatile' keyword (only relevant for
016 * fields). This is equivalent to PMD's AvoidUsingVolatile, which is part of the
017 * 'controversial' ruleset, which is why this check is OFF by default.
018 */
019@ACheck(name = "Avoid using `volatile`", //
020                description = "Marking fields as `volatile` can be tricky as it requires a deep understanding of the underlying principles. "
021                                + "Instead, you can e.g. use a lock, the `synchronized` keyword (Java), or the `[MethodImpl(MethodImplOptions.Synchronized)]` attribute (C#), respectively.", //
022                groupName = BAD_PRACTICE, //
023                languages = { JAVA, CS }, defaultEnablement = OFF, parameters = { ABSTRACT_SYNTAX_TREE })
024public class AvoidVolatileFieldsCheck extends EntityCheckBase {
025        @Override
026        protected String getXPathSelectionString() {
027                return "//ATTRIBUTE[modifiers('volatile')]";
028        }
029
030        @Override
031        protected void processEntity(ShallowEntity entity) throws CheckException {
032                createFinding("Avoid marking `" + entity.getName() + "` as `volatile`", entity);
033        }
034}
