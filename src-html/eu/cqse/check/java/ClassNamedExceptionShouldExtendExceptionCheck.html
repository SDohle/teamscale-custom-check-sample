025/**
026 * {@ConQAT.Doc}
027 */
028@ACheck(name = "Class named like Exception should extend Exception or subclass", //
029                description = "If the name of a class has the suffix *Exception*, one would expect that it inherits from `Exception` or one of its subclasses. "
030                                + "Since naming choices like this can cause confusion, occurrences of such unexpected class names will be reported by this check. "
031                                + "To avoid false positives, no finding will be created if the name of the parent class contains the term *Exception*.\n"
032                                + "\n" + "*Incorrect examples (Java) *\n" + //
033                                "\n" + "```java\n" + //
034                                "class SomeException { } // unexpected, not inheriting from java.lang.Exception or subclass\n" + //
035                                "class OtherException extends Foo { } // unexpected, apparently no exception-related parent class\n" + //
036                                "```\n" + "\n" + "*Correct examples (Java) *\n" + //
037                                "\n" + "```java\n" + //
038                                "class SomeException extends Exception { } \n" + //
039                                "class OtherException extends SomeParentException { } \n" + //
040                                "```\n" + "\n", //
041                groupName = BAD_PRACTICE, defaultEnablement = YELLOW, languages = { JAVA, CS }, parameters = {
042                                ABSTRACT_SYNTAX_TREE })
043public class ClassNamedExceptionShouldExtendExceptionCheck extends EntityCheckBase {
044
045        /**
046         * Pattern that matches parent class declarations (Java & C#) containing
047         * 'Exception' or 'Throwable'.
048         */
049        private static final TokenPattern PARENT_CLASS_EXCEPTION_PATTERN = new TokenPattern().alternative(EXTENDS, COLON)
050                        .regex("\\w*([eE]xception)|Throwable\\w*");
051
052        /** {@inheritDoc} */
053        @Override
054        protected String getXPathSelectionString() {
055                return "//TYPE[subtype('" + SubTypeNames.CLASS + "')]";
056        }
057
058        @Override
059        protected void processEntity(ShallowEntity typeEntity) throws CheckException {
060
061                if (typeEntity.getName() == null || !StringUtils.endsWithIgnoreCase(typeEntity.getName(), "exception")) {
062                        return;
063                }
064                UnmodifiableList<IToken> classDefinitionTokens = typeEntity.ownStartTokens();
065                TokenPatternMatch match = PARENT_CLASS_EXCEPTION_PATTERN.findFirstMatch(classDefinitionTokens);
066                if (match == null) {
067                        createFinding("`" + typeEntity.getName() + "` should inherit from `Exception` or subclass",
068                                        classDefinitionTokens.get(TokenStreamUtils.firstTokenOfType(classDefinitionTokens, IDENTIFIER)));
069                }
070        }
071
072}
