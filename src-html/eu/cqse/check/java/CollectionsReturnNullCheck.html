008package eu.cqse.check.java;
009
010import java.util.Collection;
011import java.util.List;
012
013import org.conqat.lib.commons.string.StringUtils;
014
015import eu.cqse.check.base.EntityCheckBase;
016import eu.cqse.check.framework.core.ACheck;
017import eu.cqse.check.framework.core.CheckException;
018import eu.cqse.check.framework.core.ECheckParameter;
019import eu.cqse.check.framework.core.EFindingEnablement;
020import eu.cqse.check.framework.scanner.ELanguage;
021import eu.cqse.check.framework.scanner.IToken;
022import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
023import eu.cqse.check.framework.util.LanguageFeatureParser;
024
025/**
026 * Checks that methods returning arrays or collections do not return
027 * <code>null</code>, but instead return an empty array or collections.
028 */
029@ACheck(name = "Collection return value not null", languages = ELanguage.JAVA, groupName = "Bad practice", defaultEnablement = EFindingEnablement.OFF, description = ""
030                + "Methods returning arrays or collections may not return null, but must return an empty array or collections.", parameters = {
031                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
032public class CollectionsReturnNullCheck extends EntityCheckBase {
033
034        /**
035         * The names of all known classes and interfaces that implement the
036         * {@link Collection} interface.
037         */
038        private static final String[] COLLECTION_CLASS_NAMES = { "AbstractCollection", "AbstractList", "AbstractQueue",
039                        "AbstractSequentialList", "AbstractSet", "ArrayBlockingQueue", "ArrayDeque", "ArrayList", "AttributeList",
040                        "BeanContextServicesSupport", "BeanContextSupport", "ConcurrentLinkedDeque", "ConcurrentLinkedQueue",
041                        "ConcurrentSkipListSet", "CopyOnWriteArrayList", "CopyOnWriteArraySet", "DelayQueue", "EnumSet", "HashSet",
042                        "JobStateReasons", "LinkedBlockingDeque", "LinkedBlockingQueue", "LinkedHashSet", "LinkedList",
043                        "LinkedTransferQueue", "PriorityBlockingQueue", "PriorityQueue", "RoleList", "RoleUnresolvedList", "Stack",
044                        "SynchronousQueue", "TreeSet", "Vector", "BeanContext", "BeanContextServices", "BlockingDeque",
045                        "BlockingQueue", "Deque", "List", "NavigableSet", "Queue", "Set", "SortedSet", "TransferQueue",
046                        "Collection" };
047
048        /** {@inheritDoc} */
049        @Override
050        protected String getXPathSelectionString() {
051                return "//METHOD[not(subtype('constructor')) and not(subtype('static initializer')) and not(subtype('non-static initializer')) and not(subtype('lambda'))]";
052        }
053
054        /** {@inheritDoc} */
055        @Override
056        protected void processEntity(ShallowEntity entity) throws CheckException {
057                String returnType = LanguageFeatureParser.JAVA.getDetailedReturnType(entity);
058                if (!returnType.contains("[") && !StringUtils.startsWithOneOf(returnType, COLLECTION_CLASS_NAMES)) {
059                        // no array is returned
060                        return;
061                }
062
063                List<ShallowEntity> returnStatements = select(entity, "STATEMENT[name-matches('return')]");
064                for (ShallowEntity statement : returnStatements) {
065                        List<IToken> statementTokens = statement.includedTokens();
066                        if (statementTokens.size() == 3 && statementTokens.get(1).getText().equals("null")) {
067                                createFinding("Method '" + entity.getName() + "' returns null instead of empty collection.", statement);
068                        }
069                }
070        }
071
072}
