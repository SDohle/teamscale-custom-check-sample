043/**
044 * This check is about the encoding parameter in constructors like String. It
045 * creates findings in case there is no charset parameter
046 */
047@ACheck(name = ConstructorCallWithoutEncodingCheck.CHECK_NAME, description = "When transforming a byte sequence into a `String` the encoding must be considered. "
048                + "Using the default platform encoding may result in interoperability issues.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
049                                ELanguage.JAVA }, parameters = { ECheckParameter.TYPE_RESOLUTION,
050                                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
051public class ConstructorCallWithoutEncodingCheck extends EntityCheckBase {
052
053        /** The name of this service. */
054        public static final String CHECK_NAME = "Transformation of byte sequence into String must consider encoding";
055
056        /** Token type sequence that indicates a constructor call. */
057        private static final ETokenType[] CONSTRUCTOR_CALL_SEQUENCE = { ETokenType.NEW, ETokenType.IDENTIFIER,
058                        ETokenType.LPAREN };
059
060        /** The constructors to be checked. */
061        private static final Set<String> CONSTRUCTOR_NAMES = CollectionUtils.asHashSet("String", "InputStreamReader",
062                        "OutputStreamWriter", "FileReader");
063
064        /**
065         * A simple map to easily find the corresponding closing token type for various
066         * opening types.
067         */
068        private static final Map<ETokenType, ETokenType> MATCHING_TOKEN_TYPES = new HashMap<>();
069        static {
070                MATCHING_TOKEN_TYPES.put(ETokenType.LPAREN, ETokenType.RPAREN);
071                MATCHING_TOKEN_TYPES.put(ETokenType.LBRACE, ETokenType.RBRACE);
072                MATCHING_TOKEN_TYPES.put(ETokenType.LBRACK, ETokenType.RBRACK);
073        }
074
075        /** {@inheritDoc} */
076        @Override
077        protected String getXPathSelectionString() {
078                return "//STATEMENT | //ATTRIBUTE";
079        }
080
081        /** {@inheritDoc} */
082        @Override
083        protected void processEntity(ShallowEntity entity) throws CheckException {
084                processTokens(entity, entity.ownStartTokens());
085                if (entity.hasChildren()) {
086                        processTokens(entity, entity.ownEndTokens());
087                }
088        }
089
090        /** Processes the tokens of a statement entity. */
091        protected void processTokens(ShallowEntity entity, List<IToken> tokens) throws CheckException {
092                List<Integer> callIndices = TokenStreamUtils.firstTokenOfTypeSequences(tokens, 0, CONSTRUCTOR_CALL_SEQUENCE);
093
094                for (Integer callIndex : callIndices) {
095                        IToken constructorNameToken = tokens.get(callIndex + 1);
096                        String constructorName = constructorNameToken.getText();
097                        if (CONSTRUCTOR_NAMES.contains(constructorName)) {
098                                List<IToken> constructorArguments = getConstructorArgumentTokens(tokens, callIndex);
099                                ScopedTypeLookup tokenTypeLookup = context.getTypeResolution(ECodeViewOption.FILTERED)
100                                                .getTypeLookup(entity);
101
102                                switch (constructorName) {
103                                case "String":
104                                        stringCheck(tokenTypeLookup, constructorNameToken, constructorArguments);
105                                        break;
106                                case "InputStreamReader":
107                                case "OutputStreamWriter":
108                                        ioStreamReaderWriterCheck(constructorNameToken, constructorArguments);
109                                        break;
110                                case "FileReader":
111                                        createFinding(constructorNameToken);
112                                        break;
113                                default:
114                                        break;
115                                }
116                        }
117                }
118        }
119
120        /** Checks the String constructor. */
121        private void stringCheck(ScopedTypeLookup tokenTypeLookup, IToken constructorNameToken,
122                        List<IToken> constructorArguments) throws CheckException {
123                List<List<IToken>> parameters = extractParameters(constructorArguments);
124                int numberOfParameters = parameters.size();
125                if (numberOfParameters == 0) {
126                        return;
127                }
128                if (isParameterByteArray(tokenTypeLookup, parameters.get(0))) {
129                        if (numberOfParameters == 1 || numberOfParameters == 3) {
130                                createFinding(constructorNameToken);
131                        } else if (numberOfParameters == 2 || numberOfParameters == 4) {
132                                List<IToken> lastParameter = parameters.get(parameters.size() - 1);
133                                if (parameterIsNotCharsetOrString(tokenTypeLookup, lastParameter)) {
134                                        createFinding(constructorNameToken);
135                                }
136                        }
137                }
138        }
139
140        /** Checks the String constructor. */
141        private void ioStreamReaderWriterCheck(IToken constructorNameToken, List<IToken> constructorArguments)
142                        throws CheckException {
143                List<List<IToken>> parameters = extractParameters(constructorArguments);
144                int numberOfParameters = parameters.size();
145                if (numberOfParameters == 1) {
146                        createFinding(constructorNameToken);
147                }
148        }
149
150        /** Checks if a parameter is a charset or a string. */
151        private static boolean parameterIsNotCharsetOrString(ScopedTypeLookup tokenTypeLookup,
152                        List<IToken> parameterTokens) {
153                if (parameterTokens.size() == 1) {
154                        IToken parameterToken = parameterTokens.get(0);
155                        ETokenType parameterTokenType = parameterToken.getType();
156                        if (parameterTokenType.equals(ETokenType.IDENTIFIER)) {
157                                return identifierTypeNotStringOrCharset(parameterToken, tokenTypeLookup);
158                        } else if (!parameterTokenType.equals(ETokenType.STRING_LITERAL)) {
159                                return true;
160                        }
161                } else if (TokenStreamUtils.endsWith(parameterTokens,
162                                new ETokenType[] { ETokenType.DOT, ETokenType.IDENTIFIER })) {
163                        return identifierTypeNotStringOrCharset(parameterTokens.get(parameterTokens.size() - 1), tokenTypeLookup);
164                }
165                return false;
166        }
167
168        /** Returns true if an identifier is neither String nor Charset. */
169        private static boolean identifierTypeNotStringOrCharset(IToken identifier, ScopedTypeLookup tokenTypeLookup) {
170                TypedVariable typeInfo = tokenTypeLookup.getTypeInfo(identifier.getText());
171                if (typeInfo != null) {
172                        String typeName = typeInfo.getTypeName();
173                        return ((!typeName.equals("String")) && (!typeName.equals("Charset")));
174                }
175                return false;
176        }
177
178        /** Checks if a parameter is a byte array. */
179        private static boolean isParameterByteArray(ScopedTypeLookup tokenTypeLookup, List<IToken> parameterTokens) {
180                if (parameterTokens.size() == 1 && parameterTokens.get(0).getType().equals(ETokenType.IDENTIFIER)) {
181                        TypedVariable typeInfo = tokenTypeLookup.getTypeInfo(parameterTokens.get(0).getText());
182                        if (typeInfo != null && typeInfo.getTypeName().equals("byte[]")) {
183                                return true;
184                        }
185                }
186
187                if (parameterTokens.size() > 4) {
188                        int lastIndex = parameterTokens.size() - 1;
189                        if (endsWithMethodCall(parameterTokens, lastIndex, "getBytes")
190                                        || endsWithMethodCall(parameterTokens, lastIndex, "toByteArray")) {
191                                return true;
192                        }
193                        if (parameterTokens.size() >= 6) {
194                                return isCastToByteArray(parameterTokens) || (isByteArrayConstruction(parameterTokens)
195                                                && parameterTokens.get(lastIndex).getType().equals(ETokenType.RBRACE));
196                        }
197                }
198                return false;
199        }
200
201        /**
202         * Returns whether the parameter tokens end with a (parameter-less) method call
203         * to the given function.
204         */
205        private static boolean endsWithMethodCall(List<IToken> parameterTokens, int lastIndex, String methodName) {
206                return TokenStreamTextUtils.hasSequence(parameterTokens, lastIndex - 3, ".", methodName, "(", ")");
207        }
208
209        /** Returns whether the tokens indicate the construction of a new byte[]. */
210        private static boolean isByteArrayConstruction(List<IToken> parameterTokens) {
211                return TokenStreamTextUtils.hasSequence(parameterTokens, 0, "new", "byte", "[", "]", "{");
212        }
213
214        /** Returns whether the tokens indicate a cast to a byte[]. */
215        private static boolean isCastToByteArray(List<IToken> parameterTokens) {
216                return TokenStreamTextUtils.hasSequence(parameterTokens, 0, "(", "byte", "[", "]", ")");
217        }
218
219        /** Extracts the constructor argument tokens. */
220        private static List<IToken> getConstructorArgumentTokens(List<IToken> tokens, Integer callIndex) {
221                return TokenStreamUtils.tokensBetweenWithNesting(tokens, callIndex, ETokenType.LPAREN, ETokenType.RPAREN);
222        }
223
224        /** Extracts the constructor's parameters as lists of tokens. */
225        private static List<List<IToken>> extractParameters(List<IToken> constructorArgumentTokens) {
226                List<List<IToken>> parameters = new ArrayList<>();
227                List<IToken> buffer = new ArrayList<>();
228                int i = 0;
229                while (i < constructorArgumentTokens.size()) {
230                        IToken currentToken = constructorArgumentTokens.get(i);
231                        if (currentToken.getType().equals(ETokenType.LPAREN) || currentToken.getType().equals(ETokenType.LBRACE)) {
232                                buffer.add(currentToken);
233                                List<IToken> tokensBetweenWithNesting = TokenStreamUtils.tokensBetweenWithNesting(
234                                                constructorArgumentTokens, i, currentToken.getType(),
235                                                MATCHING_TOKEN_TYPES.get(currentToken.getType()));
236                                buffer.addAll(tokensBetweenWithNesting);
237                                i += tokensBetweenWithNesting.size() + 1;
238                                buffer.add(constructorArgumentTokens.get(i++));
239                        } else if (currentToken.getType().equals(ETokenType.COMMA)) {
240                                parameters.add(buffer);
241                                buffer = new ArrayList<>();
242                                i++;
243                        } else {
244                                buffer.add(currentToken);
245                                i++;
246                        }
247                }
248                if (!buffer.isEmpty()) {
249                        parameters.add(buffer);
250                }
251                return parameters;
252        }
253
254        /** Creates the finding for the specified constructor. */
255        private void createFinding(IToken constructorNameToken) throws CheckException {
256                createFinding("Transformation of byte sequence into `String` uses platform encoding", constructorNameToken);
257        }
