006package eu.cqse.check.java;
007
008import eu.cqse.check.base.EntityCheckBase;
009import eu.cqse.check.framework.core.ACheck;
010import eu.cqse.check.framework.core.CheckException;
011import eu.cqse.check.framework.core.CheckGroupNames;
012import eu.cqse.check.framework.core.ECheckParameter;
013import eu.cqse.check.framework.core.EFindingEnablement;
014import eu.cqse.check.framework.scanner.ELanguage;
015import eu.cqse.check.framework.scanner.ETokenType;
016import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
017
018/**
019 * Check for usage of continue or break in combination with a label.
020 */
021@ACheck(name = "Avoid continue and break with label", description = "The usage of continue or break in combination with a label is discouraged"
022                + " as it leads to code that is hard to read.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
023                                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
024public class ContinueOrBreakWithLabel extends EntityCheckBase {
025
026        /** {@inheritDoc} */
027        @Override
028        protected String getXPathSelectionString() {
029                return "//STATEMENT[name-matches('continue') or name-matches('break')]";
030        }
031
032        /**
033         * {@inheritDoc} Creates a finding if <code>continue</code> or
034         * <code>break</code> is followed by a label.
035         */
036        @Override
037        protected void processEntity(ShallowEntity entity) throws CheckException {
038                if (entity.includedTokens().size() < 2) {
039                        return;
040                }
041                if (entity.includedTokens().get(1).getType() == ETokenType.IDENTIFIER) {
042                        createFinding("Statement '" + entity.includedTokens().get(0).getType().toString().toLowerCase()
043                                        + "' should not be used in combination with a label.", entity);
044                }
045        }
046}
