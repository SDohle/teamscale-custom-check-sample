008package eu.cqse.check.java;
009
010import java.util.ArrayList;
011import java.util.List;
012
013import org.conqat.lib.commons.assertion.CCSMAssert;
014import org.conqat.lib.commons.string.StringUtils;
015
016import eu.cqse.check.base.EntityCheckBase;
017import eu.cqse.check.framework.core.ACheck;
018import eu.cqse.check.framework.core.CheckException;
019import eu.cqse.check.framework.core.CheckGroupNames;
020import eu.cqse.check.framework.core.ECheckParameter;
021import eu.cqse.check.framework.core.EFindingEnablement;
022import eu.cqse.check.framework.scanner.ELanguage;
023import eu.cqse.check.framework.scanner.ETokenType;
024import eu.cqse.check.framework.scanner.IToken;
025import eu.cqse.check.framework.shallowparser.SubTypeNames;
026import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
027import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
028import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
029
030/** Checks for correct usage of AssertJ. */
031@ACheck(name = "assertThat(..) calls should be followed by an assertion specification (excluding .as(..)", description = CorrectAssertThatAssertionsCheck.DESCRIPTION, groupName = CheckGroupNames.CORRECTNESS, defaultEnablement = EFindingEnablement.YELLOW, languages = {
032                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
033public class CorrectAssertThatAssertionsCheck extends EntityCheckBase {
034
035        /** Findings description for this check */
036        protected static final String DESCRIPTION = "Using assertJ all \"assertThat()\" calls should be followed by an assertion specification."
037                        + "The method \"as()\" should not be last in the method chain,"
038                        + "because the \"assertThat\" expression otherwise does not" + "have any effect.\n\n" + "Examples:\n"
039                        + "``` js \n" + "// Instead of this expression \n" + "assertThat(list.isEmpty()); \n" + "// write \n"
040                        + "assertThat(list).isEmpty(); \n" + "// or \n" + "assertThat(list.isEmpty()).isTrue(); \n\n"
041                        + "// Change this expression \n" + "assertThat(list).hasSize(5).as(\"This list should be size 5\"); \n"
042                        + "// to \n" + "assertThat(list).as(\"This list should be size 5\").hasSize(5); \n" + "```";
043
044        private boolean importsAssertJ = false;
045
046        private boolean staticallyImportsAssertJ = false;
047
048        @Override
049        public void execute() throws CheckException {
050                determineAssertJUsageFromImports();
051
052                if (importsAssertJ || staticallyImportsAssertJ) {
053                        super.execute();
054                }
055        }
056
057        /**
058         * This checks the imports and fills {@link #importsAssertJ} and
059         * {@link #staticallyImportsAssertJ}.
060         */
061        private void determineAssertJUsageFromImports() throws CheckException {
062                for (ShallowEntity topLevelEntity : context.getRootEntity(getCodeViewOption()).getChildren()) {
063                        if (topLevelEntity.getType() != EShallowEntityType.META) {
064                                continue;
065                        }
066
067                        if (SubTypeNames.IMPORT.equals(topLevelEntity.getSubtype())
068                                        && topLevelEntity.getName().startsWith("org.assertj.core.api")) {
069                                importsAssertJ = true;
070                        } else if (SubTypeNames.STATIC_IMPORT.equals(topLevelEntity.getSubtype())
071                                        && topLevelEntity.getName().startsWith("org.assertj.core.api")) {
072                                staticallyImportsAssertJ = true;
073                        }
074                }
075        }
076
077        @Override
078        protected String getXPathSelectionString() {
079                List<String> predicateParts = new ArrayList<>();
080                if (importsAssertJ) {
081                        predicateParts.add("name-matches('Assertions')");
082                }
083                if (staticallyImportsAssertJ) {
084                        predicateParts.add("name-matches('assertThat')");
085                }
086
087                CCSMAssert.isNotEmpty(predicateParts,
088                                "At least one prodicate expected, as otherwise execute() should not be called.");
089
090                return "//STATEMENT[" + StringUtils.concat(predicateParts, " or ") + "]";
091        }
092
093        @Override
094        protected void processEntity(ShallowEntity entity) throws CheckException {
095                List<IToken> tokens = entity.includedTokens();
096                if (tokens.get(0).getText().equals("Assertions")
097                                && (tokens.size() < 3 || !tokens.get(2).getText().equals("assertThat"))) {
098                        return;
099                }
100
101                int lastMethodTokenIndex = TokenStreamUtils.findMatchingOpeningToken(tokens,
102                                TokenStreamUtils.lastTokenOfType(tokens, ETokenType.RPAREN) - 1, ETokenType.LPAREN, ETokenType.RPAREN)
103                                - 1;
104
105                if (lastMethodTokenIndex < 0) {
106                        createFindingOnFirstLine("Illegal expression", entity);
107                        return;
108                }
109
110                String lastMethod = tokens.get(lastMethodTokenIndex).getText();
111                if (lastMethod.equals("as")) {
112                        createFindingOnFirstLine("\"as(..)\" should not be the last method in the chain", entity);
113                } else if (lastMethod.equals("assertThat")) {
114                        createFindingOnFirstLine("\"assertThat(..)\" should be followed by an assertion specification", entity);
115                }
116        }
117}
