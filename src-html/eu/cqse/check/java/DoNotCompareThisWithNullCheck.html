008package eu.cqse.check.java;
009
010import static eu.cqse.check.framework.scanner.ETokenType.EQEQ;
011import static eu.cqse.check.framework.scanner.ETokenType.NOTEQ;
012import static eu.cqse.check.framework.scanner.ETokenType.NULL_LITERAL;
013import static eu.cqse.check.framework.scanner.ETokenType.THIS;
014
015import java.util.EnumSet;
016import java.util.List;
017
018import eu.cqse.check.base.EntityTokenCheckBase;
019import eu.cqse.check.framework.core.ACheck;
020import eu.cqse.check.framework.core.CheckException;
021import eu.cqse.check.framework.core.CheckGroupNames;
022import eu.cqse.check.framework.core.ECheckParameter;
023import eu.cqse.check.framework.core.EFindingEnablement;
024import eu.cqse.check.framework.scanner.ELanguage;
025import eu.cqse.check.framework.scanner.ETokenType;
026import eu.cqse.check.framework.scanner.IToken;
027import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
028
029/**
030 * {@ConQAT.Doc}
031 * 
032 * @author $Author: heinemann $
033 * @version $Rev: 55467 $
034 * @ConQAT.Rating GREEN Hash: F261BCEF12F76C7CE8D68A61BA10601C
035 */
036@ACheck(name = "Do not compare this and null", description = "This must not be "
037                + "compared with null, because this cannot be null.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
038                                ELanguage.JAVA, ELanguage.XTEND }, parameters = {
039                                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
040public class DoNotCompareThisWithNullCheck extends EntityTokenCheckBase {
041
042        /** Compare operator token types. */
043        private static final EnumSet<ETokenType> COMPARE_OPERATORS = EnumSet
044                        .of(EQEQ, NOTEQ);
045
046        /** {@inheritDoc} */
047        @Override
048        protected String getXPathSelectionString() {
049                return "//METHOD[not(modifiers('static'))]//STATEMENT";
050        }
051
052        /**
053         * Checks whether the given tokens contain a comparison between this and
054         * null.
055         */
056        @Override
057        protected void processTokens(List<IToken> tokens) throws CheckException {
058                List<Integer> compareIndices = TokenStreamUtils.findAll(tokens, 1,
059                                tokens.size() - 1, COMPARE_OPERATORS);
060                for (int compareIndex : compareIndices) {
061                        IToken leftOperand = tokens.get(compareIndex - 1);
062                        IToken rightOperand = tokens.get(compareIndex + 1);
063                        ETokenType leftType = leftOperand.getType();
064                        ETokenType rightType = rightOperand.getType();
065                        if ((leftType == NULL_LITERAL && rightType == THIS)
066                                        || (leftType == THIS && rightType == NULL_LITERAL)) {
067                                createFinding("this must not be compared with null",
068                                                leftOperand, rightOperand);
069                                break;
070                        }
071                }
072        }
073}
