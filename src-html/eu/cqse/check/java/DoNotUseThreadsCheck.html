001package eu.cqse.check.java;
002
003import static eu.cqse.check.framework.core.CheckGroupNames.BAD_PRACTICE;
004import static eu.cqse.check.framework.core.ECheckParameter.ABSTRACT_SYNTAX_TREE;
005import static eu.cqse.check.framework.core.ECheckParameter.TYPE_RESOLUTION;
006import static eu.cqse.check.framework.core.EFindingEnablement.YELLOW;
007import static eu.cqse.check.framework.scanner.ELanguage.JAVA;
008import static eu.cqse.check.framework.scanner.ETokenType.EXTENDS;
009import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
010import static eu.cqse.check.framework.scanner.ETokenType.IMPLEMENTS;
011
012import java.util.List;
013
014import org.conqat.lib.commons.collections.UnmodifiableList;
015
016import eu.cqse.check.framework.core.ACheck;
017import eu.cqse.check.framework.core.CheckException;
018import eu.cqse.check.framework.core.CheckImplementationBase;
019import eu.cqse.check.framework.core.phase.ECodeViewOption;
020import eu.cqse.check.framework.scanner.ETokenType;
021import eu.cqse.check.framework.scanner.IToken;
022import eu.cqse.check.framework.shallowparser.SubTypeNames;
023import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
024import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
025import eu.cqse.check.framework.typetracker.ScopedTypeLookup;
026import eu.cqse.check.framework.util.tokens.TokenPattern;
027import eu.cqse.check.framework.util.tokens.TokenPatternMatch;
028
029@ACheck(name = DoNotUseThreadsCheck.FINDING_MESSAGE, description = DoNotUseThreadsCheck.DESCRIPTION, groupName = BAD_PRACTICE, defaultEnablement = YELLOW, languages = JAVA, parameters = {
030                ABSTRACT_SYNTAX_TREE, TYPE_RESOLUTION })
031public class DoNotUseThreadsCheck extends CheckImplementationBase {
032
033        /** {@value} */
034        protected static final String FINDING_MESSAGE = "Using threads is forbidden for Java EE containers";
035
036        /** Description of this finding */
037        protected static final String DESCRIPTION = "The J2EE specification explicitly forbids the use of threads for Java "
038                        + "EE containers. This rule raises a finding when: \n" + //
039                        "* a class inherits from `Thread`.\n" + //
040                        "* a class implements `Runnable`.\n" + //
041                        "* an interface extends `Runnable`.\n" + //
042                        "* the `start` method is called of an object that inherits from `Thread`.\n\n" + //
043                        "*Incorrect examples*\n\n" + //
044                        "```java\n" + //
045                        "Thread thread = new Thread(); thread.start(); // unexpected, executes `*Thread.start()`\n" + //
046                        "class MyThreadedClass extends Thread { } // unexpected, inherits from `Thread`\n" + //
047                        "class MyRunnableClass implements Runnable { } // unexpected, implements `Runnable`\n" + //
048                        "interface MyThreadedInterface extends Runnable { } // unexpected, extends `Runnable`\n```\n\n" + //
049                        "*Correct examples*\n\n" + //
050                        "```java\n" + "class MySequentialClass { } \n" + //
051                        "interface MySequentialInterface { } \n```\n";
052
053        /**
054         * Pattern that matches parent class or usage declarations (Java) containing
055         * 'Thread' or 'Runnable'
056         */
057        private static final TokenPattern USE_THREAD_PATTERN = new TokenPattern().alternative(EXTENDS, IMPLEMENTS)
058                        .regex("\\bThread\\b|\\bRunnable\\b");
059
060        /** Pattern to detect classes or interfaces. */
061        private static final String CLASS_OR_INTERFACE_PATH = "//TYPE[subtype('" + SubTypeNames.CLASS + "') | subtype('"
062                        + SubTypeNames.INTERFACE + "')]";
063
064        /** Pattern to detect statements. */
065        private static final String STATEMENT_PATH = "//STATEMENT";
066
067        /** Token sequence to look for in statements. */
068        private static final ETokenType[] RECOGNITION_SEQUENCE = { IDENTIFIER, ETokenType.DOT, IDENTIFIER,
069                        ETokenType.LPAREN };
070
071        /** Offset to extract the member name. */
072        private static final int MEMBER_NAME_OFFSET = 2;
073
074        /** {@inheritDoc} */
075        @Override
076        public void execute() throws CheckException {
077                for (ShallowEntity entity : select(CLASS_OR_INTERFACE_PATH)) {
078                        processClassEntity(entity);
079                }
080
081                for (ShallowEntity entity : select(STATEMENT_PATH)) {
082                        processStatementEntity(entity);
083                }
084        }
085
086        /**
087         * Recognizes class / interface declarations to create findings for threading.
088         */
089        private void processClassEntity(ShallowEntity entity) throws CheckException {
090                UnmodifiableList<IToken> tokens = entity.ownStartTokens();
091                TokenPatternMatch match = USE_THREAD_PATTERN.findFirstMatch(tokens);
092                if (match == null) {
093                        return;
094                }
095
096                String message = "`" + entity.getName() + "`";
097                if (entity.getSubtype().equals(SubTypeNames.CLASS)) {
098                        message += " should neither inherit from `Thread` nor implement `Runnable`";
099                } else {
100                        // must be an interface
101                        message += " should not extend `Runnable`";
102                }
103                createFinding(message, tokens.get(TokenStreamUtils.firstTokenOfType(tokens, IDENTIFIER)));
104        }
105
106        /**
107         * Recognizes method call statements to create findings for `*Thread.start()`.
108         */
109        private void processStatementEntity(ShallowEntity entity) throws CheckException {
110                ScopedTypeLookup typeLookup = context.getTypeResolution(ECodeViewOption.FILTERED).getTypeLookup(entity);
111                UnmodifiableList<IToken> tokens = entity.ownStartTokens();
112
113                List<Integer> indices = TokenStreamUtils.firstTokenOfTypeSequences(tokens, 0, RECOGNITION_SEQUENCE);
114                for (Integer index : indices) {
115                        String variableName = tokens.get(index).getText();
116                        String memberName = tokens.get(index + MEMBER_NAME_OFFSET).getText();
117
118                        if (memberName.equals("start") && typeLookup.containsVariable(variableName)
119                                        && typeLookup.getTypeInfo(variableName).getTypeName().endsWith("Thread")) {
120                                createFindingOnFirstLine("Thread.start() should not be used", entity);
121                        }
122                }
123        }
124
125}
