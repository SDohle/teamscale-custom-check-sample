010import java.util.List;

012import eu.cqse.check.base.EntityCheckBase;
013import eu.cqse.check.framework.core.ACheck;
014import eu.cqse.check.framework.core.CheckException;
015import eu.cqse.check.framework.scanner.IToken;

022/**
023 * Checks for usages of "e instanceof SomeException" within catch-blocks, which
024 * should be handled by explicit catch blocks instead.
025 */
026@ACheck(name = "Instance-of usage for exception within catch-block", //
027                description = "Instead of testing caught exceptions using `instanceof`, " + //
028                                "adding multiple catch blocks for the specific target exception types should be preferred instead, " + //
029                                "as this mitigates the risk of inadvertently catching higher-level exceptions. " + //
030                                "Often, this will also improve readability and make future changes easier.\n" + //
031                                "\n" + "*Incorrect example:*\n" + //
032                                "``` java\n" + //
033                                "try { something(); }\n" + //
034                                "catch (Exception e) { // Incorrect: Shared catch block for two exception types\n" + //
035                                "    if (e instanceof NullPointerException) {\n" + //
036                                "        handleNPE();\n" + //
037                                "    } \n" + "    if (e instanceof OtherException) {\n" + //
038                                "        handleOtherEvent();\n" + //
039                                "    }\n" + //
040                                "}\n" + //
041                                "```\n" + "\n" + "*Correct example:*\n" + //
042                                "``` java\n" + //
043                                "try { something(); }\n" + //
044                                "catch (NullPointerException e) {\n" + //
045                                "    handleNPE();\n" + //
046                                "}\n" + //
047                                "catch (OtherException e) { // Separate catch block for other exception\n" + //
048                                "    handleOtherEvent();\n" + //
049                                "}\n" + //
050                                "```", groupName = EXCEPTION_HANDLING, defaultEnablement = YELLOW, languages = { JAVA }, parameters = {
051                                                ABSTRACT_SYNTAX_TREE })
052public class ExceptionInstanceOfCheckInCatchCheck extends EntityCheckBase {
053
054        /** The group id of the instance-of token within the token pattern. */
055        private static final int INSTANCE_OF_GROUP_ID = 0;
056
057        /** {@inheritDoc} */
058        @Override
059        protected String getXPathSelectionString() {
060                return "//STATEMENT[subtype('" + CATCH + "')]";
061        }
062
063        @Override
064        protected void processEntity(ShallowEntity catchEntity) throws CheckException {
065                List<IToken> tokensInBlock = catchEntity.getAllTokens().subList(catchEntity.getStartTokenIndex(),
066                                catchEntity.getEndTokenIndex());
067
068                IToken exceptionVariableToken = LanguageFeatureParser.JAVA.getVariableNameFromCatchTokens(tokensInBlock);
069                if (exceptionVariableToken == null) {
070                        return;
071                }
072
073                String exceptionVariableName = exceptionVariableToken.getText();
074                List<TokenPatternMatch> instanceOfCheckMatches = new TokenPattern().regex(exceptionVariableName)
075                                .sequence(INSTANCEOF).group(INSTANCE_OF_GROUP_ID).regex("\\w*Exception\\w*").findAll(tokensInBlock);
076                for (TokenPatternMatch instanceOfCheckMatch : instanceOfCheckMatches) {
077                        MatchGroupElement instanceofElement = instanceOfCheckMatch.getMatchGroup(INSTANCE_OF_GROUP_ID).get(0);
078                        createFinding("`instanceof` check for `" + exceptionVariableName + "` within catch block",
079                                        instanceofElement.getTokens().get(0));
080                }
081        }
082
083}
