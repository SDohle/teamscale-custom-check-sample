017package eu.cqse.check.java;
018
019import java.util.List;
020
021import eu.cqse.check.base.EntityCheckBase;
022import eu.cqse.check.framework.core.ACheck;
023import eu.cqse.check.framework.core.CheckException;
024import eu.cqse.check.framework.core.CheckGroupNames;
025import eu.cqse.check.framework.core.ECheckParameter;
026import eu.cqse.check.framework.core.EFindingEnablement;
027import eu.cqse.check.framework.scanner.ELanguage;
028import eu.cqse.check.framework.scanner.ETokenType;
029import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
030import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
031
032/**
033 * Check for ignored JUnit tests.
034 */
035@ACheck(name = IgnoredJUnitTestCheck.CHECK_NAME, description = "JUnit tests should not be ignored or disabled for longer periods of time.", groupName = CheckGroupNames.TEST_CONVENTIONS, defaultEnablement = EFindingEnablement.YELLOW, languages = {
036                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
037public class IgnoredJUnitTestCheck extends EntityCheckBase {
038
039        /** The name of the check */
040        protected static final String CHECK_NAME = "Avoid ignored or disabled JUnit tests";
041
042        /** {@inheritDoc} */
043        @Override
044        protected String getXPathSelectionString() {
045                return "//META[subtype('annotation') and (name-matches('Ignore') or name-matches('Disabled'))]";
046        }
047
048        /** {@inheritDoc} */
049        @Override
050        protected void processEntity(ShallowEntity entity) throws CheckException {
051                // according to the syntax of the @Ignore or @Disabled annotation, this list can
052                // only contain one or zero elements
053                List<String> ignoreMessages = TokenStreamUtils.getTokenTextsByType(entity.includedTokens(),
054                                ETokenType.STRING_LITERAL);
055                if (ignoreMessages.isEmpty()) {
056                        createFindingOnFirstLine("Avoid disabled JUnit tests.", entity);
057                        return;
058                }
059                createFindingOnFirstLine("Avoid disabled JUnit tests (Disable message is: "
060                                + removeQuotesFromToken(ignoreMessages.get(0)) + ").", entity);
061        }
062
063        /**
064         * Returns a substring without the first and last element (quotes) of the passed
065         * String.
066         */
067        private static String removeQuotesFromToken(String token) {
068                return token.substring(1, token.length() - 1);
069        }
070}
