006package eu.cqse.check.java;
007
008import org.conqat.lib.commons.collections.UnmodifiableList;
009
010import eu.cqse.check.base.AvoidStringFormatInLogStatementsBase;
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckException;
013import eu.cqse.check.framework.core.CheckGroupNames;
014import eu.cqse.check.framework.core.ECheckParameter;
015import eu.cqse.check.framework.core.EFindingEnablement;
016import eu.cqse.check.framework.scanner.ELanguage;
017import eu.cqse.check.framework.scanner.ETokenType;
018import eu.cqse.check.framework.scanner.IToken;
019import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
020
021/**
022 * Checks, whether String.format oder String formatting is used in log
023 * statements in Java-Code.
024 */
025@ACheck(name = "Avoid String.format or String formatting in log statements (Java)", description = "Log statements should not include String.format or String formatting, because they can lead to unnecessary background operations even for disabled log levels.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.OFF, languages = {
026                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
027public class JavaAvoidStringFormatInLogStatementCheck extends AvoidStringFormatInLogStatementsBase {
028        /** The message of a String.format finding. */
029        private static final String FINDING_MESSAGE_STRING_FORMAT = "Avoid using String.format or *.format in log statements";
030
031        /** The message of a String formatting finding. */
032        private static final String FINDING_MESSAGE_STRING_FORMATTING = "Avoid using String formatting in log statements";
033
034        /** The String signaling a format call if it comes after a ETokenType.format */
035        private static final String FORMAT_METHOD_NAME = "format";
036
037        /** {@inheritDoc} */
038        @Override
039        protected void processEntity(ShallowEntity entity) throws CheckException {
040                UnmodifiableList<IToken> entityTokens = entity.includedTokens();
041                IToken lastToken = entityTokens.get(0);
042                for (IToken currentToken : entityTokens) {
043
044                        if (currentToken.getType() == ETokenType.ARROW) {
045                                return;
046                        }
047                        if ((lastToken.getType() == ETokenType.DOT && currentToken.getText().contains(FORMAT_METHOD_NAME))) {
048                                createFindingOnFirstLine(FINDING_MESSAGE_STRING_FORMAT, entity);
049
050                        }
051                        if (currentToken.getType() == ETokenType.PLUS) {
052                                createFindingOnFirstLine(FINDING_MESSAGE_STRING_FORMATTING, entity);
053                        }
054                        lastToken = currentToken;
055
056                }
057
058        }
059
060}
