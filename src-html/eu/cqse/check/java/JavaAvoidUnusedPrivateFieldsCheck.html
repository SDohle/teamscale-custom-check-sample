008package eu.cqse.check.java;
009
010import eu.cqse.check.clike.CLikeAvoidUnusedPrivateFieldsCheckBase;
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckGroupNames;
013import eu.cqse.check.framework.core.ECheckParameter;
014import eu.cqse.check.framework.core.EFindingEnablement;
015import eu.cqse.check.framework.scanner.ELanguage;
016import eu.cqse.check.framework.shallowparser.SubTypeNames;
017import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
018import eu.cqse.check.framework.util.LanguageFeatureParser;
019import eu.cqse.check.framework.util.filter.JavaSupressWarningUnusedFilter;
020
021/**
022 * {@ConQAT.Doc}
023 */
024@ACheck(name = "Avoid unused private fields (Java)", description = "This rule is reported when a "
025                + "private field in your code exists but is not used by any code path.", groupName = CheckGroupNames.UNUSED_CODE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
026                                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE,
027                                                ECheckParameter.TYPE_RESOLUTION })
028public class JavaAvoidUnusedPrivateFieldsCheck extends CLikeAvoidUnusedPrivateFieldsCheckBase {
029
030        /**
031         * Filter for fields having a SupressWarnings("unused") annotation.
032         */
033        private final JavaSupressWarningUnusedFilter annotatedFieldFilter = new JavaSupressWarningUnusedFilter();
034
035        /** Constructor. */
036        public JavaAvoidUnusedPrivateFieldsCheck() {
037                super(LanguageFeatureParser.JAVA);
038        }
039
040        /** {@inheritDoc} */
041        @Override
042        protected boolean isFiltered(ShallowEntity field) {
043                return annotatedFieldFilter.isFiltered(field);
044        }
045
046        /** {@inheritDoc} */
047        @Override
048        protected String getFieldSelectionXPath() {
049                // we ignore inner classes as they occur rarely and their field uses are
050                // hard to analyze
051                return "/TYPE/ATTRIBUTE[subtype('" + SubTypeNames.ATTRIBUTE
052                                + "') and modifiers('private') and not(name-matches('serialVersionUID') and tokens('static', 'final', 'long'))]";
053        }
054
055        /** {@inheritDoc} */
056        @Override
057        protected String getProcessedEntitiesXPath() {
058                return "descendant::METHOD | descendant::ATTRIBUTE | descendant::STATEMENT | descendant::META | descendant::META[subtype('"
059                                + SubTypeNames.CASE + "')] | descendant::TYPE[subtype('" + SubTypeNames.ANONYMOUS_CLASS + "')]";
060        }
061}
