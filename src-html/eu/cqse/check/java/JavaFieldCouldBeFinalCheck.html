017package eu.cqse.check.java;
018
019import java.util.HashSet;
020import java.util.List;
021import java.util.Set;
022
023import org.conqat.lib.commons.collections.CollectionUtils;
024import org.conqat.lib.commons.collections.Pair;
025
026import eu.cqse.check.clike.CLikeFieldUsageCheckBase;
027import eu.cqse.check.framework.core.ACheck;
028import eu.cqse.check.framework.core.CheckException;
029import eu.cqse.check.framework.core.CheckGroupNames;
030import eu.cqse.check.framework.core.ECheckParameter;
031import eu.cqse.check.framework.core.EFindingEnablement;
032import eu.cqse.check.framework.core.option.ACheckOption;
033import eu.cqse.check.framework.scanner.ELanguage;
034import eu.cqse.check.framework.scanner.ETokenType;
035import eu.cqse.check.framework.scanner.IToken;
036import eu.cqse.check.framework.shallowparser.SubTypeNames;
037import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
038import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
039import eu.cqse.check.framework.util.LanguageFeatureParser;
040import eu.cqse.check.framework.util.filter.JavaAnnotatedEntityFilter;
041
042/**
043 * {@ConQAT.Doc}
044 */
045@ACheck(name = "Field could be final (Java)", description = "Fields that can be made `final` should be marked as such to avoid accidental assignment to them.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
046                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE, ECheckParameter.TYPE_RESOLUTION })
047public class JavaFieldCouldBeFinalCheck extends CLikeFieldUsageCheckBase {
048
049        /** Description for check option */
050        private static final String CHECK_OPTION_DESCRIPTION = "List of Java annotations to exclude class fields from being reported as needing to be declared final";
051
052        /**
053         * Filter for fields with having an annotation contained in
054         * {@link #annotations}.
055         */
056        private JavaAnnotatedEntityFilter annotatedFieldFilter;
057
058        /** List of annotations to filter out fields. */
059        @ACheckOption(name = "Java annotations for non-final fields", description = CHECK_OPTION_DESCRIPTION)
060        private Set<String> annotations = CollectionUtils.asHashSet("Inject", "Autowired", "SpringBean", "EJB");
061
062        /** Constructor. */
063        public JavaFieldCouldBeFinalCheck() {
064                super(LanguageFeatureParser.JAVA);
065        }
066
067        /** {@inheritDoc} */
068        @Override
069        public void initialize() throws CheckException {
070                super.initialize();
071                annotatedFieldFilter = new JavaAnnotatedEntityFilter();
072                annotatedFieldFilter.addAnnotations(annotations);
073        }
074
075        /**
076         * Filter private fields that have been initialized at their points of
077         * declaration. These fields were initially processed before and there were no
078         * write occurrences into them.
079         */
080        private static Set<Pair<IToken, ShallowEntity>> filterFieldsWithAssignmentAtDeclaration(
081                        Set<Pair<IToken, ShallowEntity>> fields) {
082                Set<Pair<IToken, ShallowEntity>> result = new HashSet<>();
083                for (Pair<IToken, ShallowEntity> fieldTokenToParentEntity : fields) {
084                        ShallowEntity field = fieldTokenToParentEntity.getSecond();
085                        if (TokenStreamUtils.contains(field.ownEndTokens(), ETokenType.EQ)) {
086                                result.add(fieldTokenToParentEntity);
087                        }
088                }
089
090                return result;
091        }
092
093        /** {@inheritDoc} */
094        @Override
095        protected boolean isFiltered(ShallowEntity field) {
096                return annotatedFieldFilter.isFiltered(field);
097        }
098
099        /** {@inheritDoc} */
100        @Override
101        protected String getFieldSelectionXPath() {
102                return "//TYPE[subtype('" + SubTypeNames.CLASS + "')]/ATTRIBUTE[subtype('" + SubTypeNames.ATTRIBUTE
103                                + "') and anymodifier('private') and not(anymodifier('static', 'final'))]";
104        }
105
106        /** {@inheritDoc} */
107        @Override
108        protected String getFindingMessage(String fieldName) {
109                return "Field `" + fieldName + "` could be made final";
110        }
111
112        /** {@inheritDoc} */
113        @Override
114        protected String getProcessedEntitiesXPath() {
115                return "descendant::METHOD[not(subtype('" + SubTypeNames.CONSTRUCTOR + "') or subtype('"
116                                + SubTypeNames.STATIC_INITIALIZER + "'))] | descendant::STATEMENT";
117        }
118
119        @Override
120        protected Set<Pair<IToken, ShallowEntity>> filterFieldsWithFinding(
121                        Set<Pair<IToken, ShallowEntity>> unanalyzedFieldTokenToParentEntity,
122                        Set<Pair<IToken, ShallowEntity>> failedAnalyzedFieldTokenToParentEntity) {
123                Set<Pair<IToken, ShallowEntity>> filteredFailedFieldTokenToParentEntity = failedAnalyzedFieldTokenToParentEntity;
124                if (unanalyzedFieldTokenToParentEntity.equals(failedAnalyzedFieldTokenToParentEntity)) {
125                        filteredFailedFieldTokenToParentEntity = filterFieldsWithAssignmentAtDeclaration(
126                                        failedAnalyzedFieldTokenToParentEntity);
127                }
128
129                return super.filterFieldsWithFinding(unanalyzedFieldTokenToParentEntity,
130                                filteredFailedFieldTokenToParentEntity);
131        }
132
133        /** {@inheritDoc} */
134        @Override
135        protected boolean analyzeTokens(List<IToken> tokens, String fieldName, boolean isShadowed) {
136                return !languageFeatureParser.getVariableWritesFromTokens(tokens, fieldName, true, isShadowed).isEmpty();
137        }
138}
