008package eu.cqse.check.java;
009
010import static eu.cqse.check.framework.scanner.ETokenType.NEW;
011
012import java.util.Set;
013
014import org.conqat.lib.commons.collections.CollectionUtils;
015
016import eu.cqse.check.base.EntityCheckBase;
017import eu.cqse.check.framework.core.ACheck;
018import eu.cqse.check.framework.core.CheckException;
019import eu.cqse.check.framework.core.ECheckParameter;
020import eu.cqse.check.framework.core.EFindingEnablement;
021import eu.cqse.check.framework.scanner.ELanguage;
022import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
023
024/**
025 * Checks that types do not have the same name as types from
026 * <code>java.lang.*</code>.
027 */
028@ACheck(name = "Name from java.lang used", languages = ELanguage.JAVA, groupName = "Bad practice", defaultEnablement = EFindingEnablement.YELLOW, description = ""
029                + "Types must not have the same name as types from 'java.lang.*'.", parameters = {
030                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
031public class JavaLangNameCheck extends EntityCheckBase {
032
033        /** The names of all forbidden classes. */
034        private static final Set<String> FORBIDDEN_NAMES = CollectionUtils.asHashSet("Appendable", "AutoCloseable",
035                        "CharSequence", "Cloneable", "Comparable", "Iterable", "Readable", "Runnable",
036                        "Thread.UncaughtExceptionHandler", "Boolean", "Byte", "Character", "Character.Subset",
037                        "Character.UnicodeBlock", "Class", "ClassLoader", "ClassValue", "Compiler", "Double", "Enum", "Float",
038                        "InheritableThreadLocal", "Integer", "Long", "Math", "Number", "Object", "Package", "Process",
039                        "ProcessBuilder", "ProcessBuilder.Redirect", "Runtime", "RuntimePermission", "SecurityManager", "Short",
040                        "StackTraceElement", "StrictMath", "String", "StringBuffer", "StringBuilder", "System", "Thread",
041                        "ThreadGroup", "ThreadLocal", "Throwable", "Void", "Character.UnicodeScript",
042                        "ProcessBuilder.Redirect.Type", "Thread.State", "ArithmeticException", "ArrayIndexOutOfBoundsException",
043                        "ArrayStoreException", "ClassCastException", "ClassNotFoundException", "CloneNotSupportedException",
044                        "EnumConstantNotPresentException", "Exception", "IllegalAccessException", "IllegalArgumentException",
045                        "IllegalMonitorStateException", "IllegalStateException", "IllegalThreadStateException",
046                        "IndexOutOfBoundsException", "InstantiationException", "InterruptedException", "NegativeArraySizeException",
047                        "NoSuchFieldException", "NoSuchMethodException", "NullPointerException", "NumberFormatException",
048                        "ReflectiveOperationException", "RuntimeException", "SecurityException", "StringIndexOutOfBoundsException",
049                        "TypeNotPresentException", "UnsupportedOperationException", "AbstractMethodError", "AssertionError",
050                        "BootstrapMethodError", "ClassCircularityError", "ClassFormatError", "Error", "ExceptionInInitializerError",
051                        "IllegalAccessError", "IncompatibleClassChangeError", "InstantiationError", "InternalError", "LinkageError",
052                        "NoClassDefFoundError", "NoSuchFieldError", "NoSuchMethodError", "OutOfMemoryError", "StackOverflowError",
053                        "ThreadDeath", "UnknownError", "UnsatisfiedLinkError", "UnsupportedClassVersionError", "VerifyError",
054                        "VirtualMachineError", "Deprecated", "Override", "SafeVarargs", "SuppressWarnings", "Annotation",
055                        "ElementType", "RetentionPolicy", "AnnotationTypeMismatchException", "IncompleteAnnotationException",
056                        "AnnotationFormatError", "Documented", "Inherited", "Retention", "Target", "ClassFileTransformer",
057                        "Instrumentation", "ClassDefinition", "IllegalClassFormatException", "UnmodifiableClassException",
058                        "CallSite", "ConstantCallSite", "MethodHandle", "MethodHandleProxies", "MethodHandles",
059                        "MethodHandles.Lookup", "MethodType", "MutableCallSite", "SwitchPoint", "VolatileCallSite",
060                        "WrongMethodTypeException", "PhantomReference", "Reference", "ReferenceQueue", "SoftReference",
061                        "WeakReference", "AnnotatedElement", "GenericArrayType", "GenericDeclaration", "InvocationHandler",
062                        "Member", "ParameterizedType", "Type", "TypeVariable", "WildcardType", "", "AccessibleObject", "Array",
063                        "Constructor", "Field", "Method", "Modifier", "Proxy", "ReflectPermission", "InvocationTargetException",
064                        "MalformedParameterizedTypeException", "UndeclaredThrowableException", "GenericSignatureFormatError");
065
066        /** {@inheritDoc} */
067        @Override
068        protected String getXPathSelectionString() {
069                return "//TYPE";
070        }
071
072        /** {@inheritDoc} */
073        @Override
074        protected void processEntity(ShallowEntity entity) throws CheckException {
075                if (entity.includedTokens().get(0).getType() != NEW && FORBIDDEN_NAMES.contains(entity.getName())) {
076                        createFindingOnFirstLine("Not allowed to override names from java.lang: " + entity.getName(), entity);
077                }
078        }
079
080}
