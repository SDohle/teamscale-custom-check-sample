008package eu.cqse.check.java;
009
010import java.util.HashSet;
011import java.util.Set;
012
013import org.conqat.lib.commons.collections.CollectionUtils;
014
015import eu.cqse.check.base.UnwantedMethodCallsCheckBase;
016import eu.cqse.check.framework.core.ACheck;
017import eu.cqse.check.framework.core.CheckGroupNames;
018import eu.cqse.check.framework.core.ECheckParameter;
019import eu.cqse.check.framework.core.EFindingEnablement;
020import eu.cqse.check.framework.core.option.ACheckOption;
021import eu.cqse.check.framework.scanner.ELanguage;
022
023/**
024 * {@ConQAT.Doc}
025 */
026@ACheck(name = "Java unwanted method calls", description = UnwantedMethodCallsCheckBase.CHECK_DESCRIPTION, groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
027                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
028public class JavaUnwantedMethodCallsCheck extends UnwantedMethodCallsCheckBase {
029
030        /** {@ConQAT.Doc} */
031        @ACheckOption(name = "Java Unwanted Methods", description = "Names of static methods that should not be called.")
032        private Set<String> unwantedMethods = new HashSet<String>(
033                        CollectionUtils.asHashSet("System.exit", "System.gc", "System.out.println", "System.err.println"));
034
035        /** {@inheritDoc} */
036        @Override
037        protected Set<String> getUnwantedMethods() {
038                return unwantedMethods;
039        }
040
041}
