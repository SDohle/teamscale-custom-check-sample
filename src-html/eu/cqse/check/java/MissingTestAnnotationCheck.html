017package eu.cqse.check.java;
018
019import static eu.cqse.check.framework.scanner.ETokenType.ABSTRACT;
020import static eu.cqse.check.framework.scanner.ETokenType.PUBLIC;
021import static eu.cqse.check.framework.scanner.ETokenType.STATIC;
022
023import java.util.List;
024import java.util.Set;
025
026import org.conqat.lib.commons.collections.CollectionUtils;
027
028import eu.cqse.check.base.EntityCheckBase;
029import eu.cqse.check.framework.core.ACheck;
030import eu.cqse.check.framework.core.CheckException;
031import eu.cqse.check.framework.core.CheckGroupNames;
032import eu.cqse.check.framework.core.ECheckParameter;
033import eu.cqse.check.framework.core.EFindingEnablement;
034import eu.cqse.check.framework.core.option.ACheckOption;
035import eu.cqse.check.framework.core.xpath.functions.NameMatchesFunction;
036import eu.cqse.check.framework.scanner.ELanguage;
037import eu.cqse.check.framework.scanner.ETokenType;
038import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
039import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
040import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
041import eu.cqse.check.framework.typetracker.java.JavaTypeInfoExtractor;
042import eu.cqse.check.framework.util.LanguageFeatureParser;
043
044/**
045 * {@ConQAT.Doc}
046 */
047@ACheck(name = MissingTestAnnotationCheck.CHECK_NAME, description = MissingTestAnnotationCheck.DESCRIPTION, groupName = CheckGroupNames.TEST_CONVENTIONS, defaultEnablement = EFindingEnablement.RED, languages = {
048                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
049public class MissingTestAnnotationCheck extends EntityCheckBase {
050
051        /** The finding description */
052        static final String DESCRIPTION = "This public method in a JUnit 4 or 5 test class is not annotated."
053                        + "Is an annotation (`@Test`, `@BeforeEach`, `@After`, `@Rule`, `@Override`, `@RepeatedTest`, `@ParameterizedTest`) "
054                        + "missing or should the method have another access modifier?";
055
056        /** Name of the check. */
057        public static final String CHECK_NAME = "Annotation in JUnit test class missing";
058
059        /** Test annotation names in JUnit 4+ */
060        private static final Set<String> TEST_ANNOTATION_SIMPLE_NAMES = CollectionUtils.asHashSet("Test", "RepeatedTest",
061                        "ParameterizedTest");
062
063        /** Qualified Test annotation names in JUnit 4+ */
064        private static final Set<String> TEST_ANNOTATION_QUALIFIED_NAMES = CollectionUtils.asHashSet("org.junit.Test",
065                        "org.junit.jupiter.api.Test", "org.junit.jupiter.api.RepeatedTest");
066
067        /** {@ConQAT.Doc} */
068        @ACheckOption(name = CHECK_NAME
069                        + ": Test class name pattern", description = "The naming pattern for test classes. Default is \".*Test$\".")
070        private String testClassNamePattern = ".*Test$";
071
072        /** {@inheritDoc} */
073        @Override
074        protected String getXPathSelectionString() {
075                return "//TYPE[" + NameMatchesFunction.NAME + "('" + testClassNamePattern + "')]";
076        }
077
078        /** {@inheritDoc} */
079        @Override
080        protected void processEntity(ShallowEntity classEntity) throws CheckException {
081                if (!isJUnit4OrHigherTestClass(classEntity)) {
082                        // @Test annotation is only required in JUnit 4+
083                        return;
084                }
085
086                for (ShallowEntity child : classEntity.getChildrenOfType(EShallowEntityType.METHOD)) {
087                        if (classEntity.getName().equals(child.getName())) {
088                                // skip constructors
089                                continue;
090                        }
091
092                        processMethodInJUnit4OrHigherTestClass(child);
093                }
094        }
095
096        /** Check if the current test class is likely a JUnit 4+ test class. */
097        private static boolean isJUnit4OrHigherTestClass(ShallowEntity classEntity) {
098                if (!TokenStreamUtils.containsAny(classEntity.ownStartTokens(), ETokenType.EXTENDS)) {
099                        // class does not (directly or indirectly) inherit TestcaseBase ->
100                        // not a JUnit 3 test class
101                        return true;
102                }
103
104                for (ShallowEntity childEntity : classEntity.getChildrenOfType(EShallowEntityType.METHOD)) {
105                        if (isTestAnnotationPresent(childEntity)) {
106                                // at least one method in this class is annotated with a JUnit test annotation
107                                return true;
108                        }
109                }
110
111                return false;
112        }
113
114        /** Process the check on a method in a JUnit 4+ test class. */
115        private void processMethodInJUnit4OrHigherTestClass(ShallowEntity method) throws CheckException {
116                if (isPublicNonAbstractNonStaticParameterlessMethod(method)
117                                && !LanguageFeatureParser.JAVA.isAnnotated(method)) {
118                        // method is a test case in a JUnit 4+ class, but has no
119                        // annotations (such as @Test, @BeforeEach, @Rule, ...)
120                        // --> the method is likely a test case with a missing @Test
121                        // annotation (or should probably not be public)
122                        createFindingOnFirstLine("Method `" + method.getName() + "` is `public` but has no JUnit annotation",
123                                        method);
124                }
125        }
126
127        /** Check if the method is annotated with <code>@Test</code>. */
128        private static boolean isTestAnnotationPresent(ShallowEntity method) {
129                List<ShallowEntity> annotations = LanguageFeatureParser.JAVA.getAnnotations(method);
130                return annotations.stream().map(ShallowEntity::getName)
131                                .anyMatch(MissingTestAnnotationCheck::isTestAnnotationName);
132        }
133
134        /**
135         * Check if the method would be suitable as test case.
136         */
137        private static boolean isPublicNonAbstractNonStaticParameterlessMethod(ShallowEntity method) {
138                if (!TokenStreamUtils.containsAny(method.ownStartTokens(), PUBLIC)) {
139                        // method is not public
140                        return false;
141                }
142
143                if (TokenStreamUtils.containsAny(method.ownStartTokens(), ABSTRACT)) {
144                        // method is abstract
145                        return false;
146                }
147
148                if (TokenStreamUtils.containsAny(method.ownStartTokens(), STATIC)) {
149                        // method is static
150                        return false;
151                }
152
153                if (!new JavaTypeInfoExtractor().extractFromMethod(method).isEmpty()) {
154                        // method has parameters
155                        return false;
156                }
157
158                // method is public, non-abstract, non-static, and has 0 parameters
159                return true;
160        }
161
162        /** Check if the name of an annotation refers to a test annotation. */
163        private static boolean isTestAnnotationName(String annotationName) {
164                return TEST_ANNOTATION_SIMPLE_NAMES.contains(annotationName)
165                                || TEST_ANNOTATION_QUALIFIED_NAMES.contains(annotationName);
166        }
167}
