001package eu.cqse.check.java;
002
003import java.util.Set;
004
005import org.conqat.lib.commons.collections.CollectionUtils;
006
007import eu.cqse.check.base.EntityFindingCheckBase;
008import eu.cqse.check.framework.core.ACheck;
009import eu.cqse.check.framework.core.CheckException;
010import eu.cqse.check.framework.core.CheckGroupNames;
011import eu.cqse.check.framework.core.ECheckParameter;
012import eu.cqse.check.framework.scanner.ELanguage;
013import eu.cqse.check.framework.scanner.ETokenType;
014import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
015import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
016import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
017
018@ACheck(name = PublicClassThreadSafetyCheck.FINDING_MESSAGE, description = PublicClassThreadSafetyCheck.DESCRIPTION, groupName = CheckGroupNames.CORRECTNESS, languages = ELanguage.JAVA, parameters = ECheckParameter.ABSTRACT_SYNTAX_TREE)
019public class PublicClassThreadSafetyCheck extends EntityFindingCheckBase {
020
021        /** {@value} */
022        protected static final String FINDING_MESSAGE = "Non-thread-safe fields should not be static";
023
024        /** Description of this finding */
025        protected static final String DESCRIPTION = "Not all classes in the standard Java library were written to be thread-safe. "
026                        + "Using them in a multi-threaded manner is highly likely to cause data problems or exceptions at runtime. "
027                        + "This rule raises a finding when an instance of `Calender`, `DateFormat`, `XPath` or `SchemaFactory` is marked static.";
028
029        /** {@inheritDoc} */
030        @Override
031        protected String getFindingMessage(ShallowEntity entity) {
032                return FINDING_MESSAGE;
033        }
034
035        private static final Set<String> CONCERNED_INSTANCES = CollectionUtils.asUnmodifiableHashSet("DateFormat",
036                        "SimpleDateFormat", "Calendar", "GregorianCalendar", "SchemaFactory", "XPath");
037
038        /** {@inheritDoc} */
039        @Override
040        protected String getXPathSelectionString() {
041                return "//ATTRIBUTE[modifiers('static')]";
042        }
043
044        /** {@inheritDoc} */
045        @Override
046        protected void processEntity(ShallowEntity entity) throws CheckException {
047                int indexOfConcernedInstance = TokenStreamTextUtils.findFirst(entity.includedTokens(), CONCERNED_INSTANCES);
048                if (indexOfConcernedInstance == TokenStreamUtils.NOT_FOUND) {
049                        return;
050                }
051
052                boolean isGenericOrAssignmentValue = TokenStreamUtils.containsAny(entity.includedTokens(), 0,
053                                indexOfConcernedInstance, ETokenType.EQ, ETokenType.LT);
054                if (!isGenericOrAssignmentValue) {
055                        createFinding("Field `" + entity.getName() + "` should not be static", entity);
056                }
057        }
058
059}
