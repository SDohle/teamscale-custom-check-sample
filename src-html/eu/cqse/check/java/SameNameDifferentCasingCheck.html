040/**
041 * {@ConQAT.Doc}
042 */
043@ACheck(name = "Names must not only differ in casing", description = "Identifiers in the code must not differ only in their casing, as this can easily lead to confusion:  \n"
044                + "```java  \n" + "class Example {  \n" + "    String fileName;  \n"
045                + "    String filename; // Hard to distinguish from previous field  \n" + "}  \n"
046                + "```", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
047                                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
048public class SameNameDifferentCasingCheck extends CheckImplementationBase {
049
050        /**
051         * {@inheritDoc}
052         * 
053         * Groups all identifiers by their lower case name. For each identifier in each
054         * group, tries to find an identifier that differs in casing from it and reports
055         * a finding for that.
056         */
057        @Override
058        public void execute() throws CheckException {
059                List<IdentifierData> identifiers = findAllIdentifiers();
060
061                findClashesAmongCheckedTypes(identifiers, EnumSet.of(EIdentifierType.CLASS));
062                findClashesAmongCheckedTypes(identifiers, EnumSet.of(EIdentifierType.CONSTANT));
063                findClashesAmongCheckedTypes(identifiers, EnumSet.of(EIdentifierType.FIELD, EIdentifierType.LOCAL_VARIABLE,
064                                EIdentifierType.PARAMETER, EIdentifierType.METHOD));
065        }
066
067        /**
068         * Finds all identifiers that are declared in the analyzed file. Except
069         * constructors.
070         */
071        private List<IdentifierData> findAllIdentifiers() throws CheckException {
072                List<IdentifierData> identifiers = new ArrayList<>();
073
074                identifiers.addAll(extractIdentifiers(select("//TYPE"), EIdentifierType.CLASS));
075
076                List<ShallowEntity> localVariables = select(
077                                "//STATEMENT[count(node())=0 and subtype('" + SubTypeNames.LOCAL_VARIABLE + "')]");
078                identifiers.addAll(extractIdentifiers(localVariables, EIdentifierType.LOCAL_VARIABLE));
079
080                for (ShallowEntity field : select("//ATTRIBUTE[subtype('" + SubTypeNames.ATTRIBUTE + "')]")) {
081                        if (!StringUtils.isEmpty(field.getName())) {
082                                if (LanguageFeatureParser.JAVA.isConstant(field)) {
083                                        identifiers.add(new IdentifierData(EIdentifierType.CONSTANT, field.getName(), field));
084                                } else {
085                                        identifiers.add(new IdentifierData(EIdentifierType.FIELD, field.getName(), field));
086                                }
087                        }
088                }
089
090                for (ShallowEntity method : select("//METHOD")) {
091                        if (!StringUtils.isEmpty(method.getName()) && !method.getSubtype().equals(SubTypeNames.CONSTRUCTOR)) {
092                                identifiers.add(new IdentifierData(EIdentifierType.METHOD, method.getName(), method));
093                        }
094                        List<IToken> parameterTokens = ShallowParsingUtils.extractParameterNameTokens(method);
095                        for (IToken parameterToken : parameterTokens) {
096                                identifiers.add(new IdentifierData(EIdentifierType.PARAMETER, parameterToken.getText(), method));
097                        }
098                }
099                return identifiers;
100        }
101
102        /**
103         * Creates an {@link IdentifierData} object for each of the given entities.
104         */
105        private static List<IdentifierData> extractIdentifiers(List<ShallowEntity> entities, EIdentifierType type) {
106                return CollectionUtils.filterAndMap(entities, entity -> !StringUtils.isEmpty(entity.getName()),
107                                entity -> new IdentifierData(type, entity.getName(), entity));
108        }
109
110        /**
111         * Finds all clashes within the identifiers.
112         * 
113         * @param checkedTypes
114         *            only identifiers of these types will be considered.
115         */
116        private void findClashesAmongCheckedTypes(List<IdentifierData> identifiers, Set<EIdentifierType> checkedTypes)
117                        throws CheckException {
118                Set<Pair<IdentifierData, IdentifierData>> identifierPairsWithFindings = new HashSet<>();
119
120                Map<String, List<IdentifierData>> identifiersByCanonicalIdentifier = identifiers.stream()
121                                .filter(identifier -> checkedTypes.contains(identifier.getType()))
122                                .collect(groupingBy(identifier -> identifier.getName().toLowerCase()));
123                for (List<IdentifierData> identifiersWithSameCanonicalization : identifiersByCanonicalIdentifier.values()) {
124                        // Ensure that the findings produced are deterministic by fixing the order in
125                        // which identifiers are processed.
126                        List<IdentifierData> orderedIdentifiers = identifiersWithSameCanonicalization.stream()
127                                        .sorted(Comparator.comparing(IdentifierData::getType).thenComparing(IdentifierData::getName))
128                                        .collect(Collectors.toList());
129                        for (IdentifierData identifier : orderedIdentifiers) {
130                                Optional<IdentifierData> identifierWithDifferentCase = findPotentialClash(orderedIdentifiers,
131                                                identifier);
132                                if (!identifierWithDifferentCase.isPresent()) {
133                                        continue;
134                                }
135                                IdentifierData clashingIdentifier = identifierWithDifferentCase.get();
136
137                                // Has this pair of identifiers been reported as a finding already?
138                                if (identifierPairsWithFindings.contains(Pair.createPair(identifier, clashingIdentifier))
139                                                || identifierPairsWithFindings.contains(Pair.createPair(clashingIdentifier, identifier))) {
140                                        continue;
141                                }
142                                createFinding(identifier, clashingIdentifier);
143                                identifierPairsWithFindings.add(Pair.createPair(identifier, clashingIdentifier));
144                        }
145                }
146        }
147
148        /**
149         * Returns any found identifier which clashes with the currently examined
150         * identifier.
151         */
152        private static Optional<IdentifierData> findPotentialClash(List<IdentifierData> otherIdentifiers,
153                        IdentifierData currentIdentifier) {
154                return otherIdentifiers.stream()
155                                .filter(potentialClash -> !potentialClash.getName().equals(currentIdentifier.getName())).findAny();
156        }
157
158        /**
159         * Creates a finding with the given message at the given token.
160         */
161        private void createFinding(IdentifierData referenceData, IdentifierData otherData) throws CheckException {
162                PairList<IToken, IToken> pairList = new PairList<>();
163                // Assumes that the first token of each shallow entity is reasonably close to
164                // problematic identifier
165                pairList.add(getFirstToken(referenceData.getEntity()), getFirstToken(referenceData.getEntity()));
166                pairList.add(getFirstToken(otherData.getEntity()), getFirstToken(otherData.getEntity()));
167
168                createFindingForSiblings(StringUtils.capitalize(referenceData.getType().getDescription()) + " "
169                                + referenceData.getName() + " differs only in casing from "
170                                + otherData.getType().getDescription().toLowerCase() + " " + otherData.getName(), pairList);
171        }
172
173        private static IToken getFirstToken(ShallowEntity entity) {
174                return entity.getAllTokens().get(entity.getStartTokenIndex());
175        }
