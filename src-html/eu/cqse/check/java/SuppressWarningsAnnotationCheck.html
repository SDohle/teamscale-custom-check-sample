006package eu.cqse.check.java;
007
008import java.util.List;
009import java.util.Set;
010
011import org.conqat.lib.commons.collections.CollectionUtils;
012
013import eu.cqse.check.base.EntityFindingCheckBase;
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckGroupNames;
017import eu.cqse.check.framework.core.ECheckParameter;
018import eu.cqse.check.framework.core.EFindingEnablement;
019import eu.cqse.check.framework.core.option.ACheckOption;
020import eu.cqse.check.framework.scanner.ELanguage;
021import eu.cqse.check.framework.scanner.ETokenType;
022import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
023import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
024
025/**
026 * Check for SuppressWarnings annotation.
027 */
028@ACheck(name = SuppressWarningsAnnotationCheck.CHECK_NAME, description = "SuppressWarnings with these types should not be used.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
029                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
030public class SuppressWarningsAnnotationCheck extends EntityFindingCheckBase {
031        protected static final String CHECK_NAME = "Avoid using SuppressWarnings";
032
033        /** {@ConQAT.Doc} */
034        @ACheckOption(name = CHECK_NAME
035                        + " - Forbidden tokens", description = "The warnings that are not allowed to be suppressed. Separated by commas."
036                                        + " Default is \"all,serial,hiding\"."
037                                        + " If the parameter is empty, all suppressed warning tokens are forbidden.")
038        private final Set<String> forbiddenSuppressedWarnings = CollectionUtils.asHashSet("all", "serial", "hiding");
039
040        /** Used to construct a more specific Finding Message */
041        private String suppressedWarningInFinding;
042        private boolean allWarningsAreForbidden = false;
043
044        /** {@inheritDoc} */
045        @Override
046        protected String getXPathSelectionString() {
047                return "//META[subtype('annotation') and name-matches('SuppressWarnings')]";
048        }
049
050        /** {@inheritDoc} */
051        @Override
052        protected void processEntity(ShallowEntity entity) throws CheckException {
053                List<String> suppressedWarnings = TokenStreamUtils.getTokenTextsByType(entity.includedTokens(),
054                                ETokenType.STRING_LITERAL);
055                if (isForbiddenSuppressedWarningTypeUsed(forbiddenSuppressedWarnings, suppressedWarnings)) {
056                        createFinding(getFindingMessage(entity), entity);
057                }
058        }
059
060        /** {@inheritDoc} */
061        @Override
062        protected String getFindingMessage(ShallowEntity entity) {
063                if (allWarningsAreForbidden) {
064                        return "SuppressWarning should be avoided.";
065                }
066                return "Warning of type '" + suppressedWarningInFinding + "' should not be suppressed.";
067        }
068
069        /**
070         * Removes the first and last element of the passed String.
071         */
072        private static String removeQuotesFromToken(String token) {
073                return token.substring(1, token.length() - 1);
074        }
075
076        /**
077         * Checks whether at least one of the suppressed warnings is forbidden. Also
078         * sets the class attribute suppressedWarningInFinding to the concrete warning
079         * that is forbidden.
080         */
081        private boolean isForbiddenSuppressedWarningTypeUsed(Set<String> forbiddenWarnings,
082                        List<String> usedSuppressedWarnings) {
083                if (forbiddenWarnings.isEmpty()) {
084                        allWarningsAreForbidden = true;
085                        // usedSuppressedWarnings can't be empty
086                        return true;
087                }
088                for (String usedSuppressedWarning : usedSuppressedWarnings) {
089                        String usedSuppressedWarningType = removeQuotesFromToken(usedSuppressedWarning);
090                        if (forbiddenWarnings.contains(usedSuppressedWarningType)) {
091                                suppressedWarningInFinding = usedSuppressedWarningType;
092                                return true;
093                        }
094                }
095                return false;
096        }
097
098}
