008package eu.cqse.check.java;
009
010import org.conqat.lib.commons.assertion.CCSMAssert;
011
012import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
013import eu.cqse.check.framework.shallowparser.util.ShallowParsingUtils;
014
015/**
016 * Class for detecting suspicious methods, that are similar to a standard method
017 * of the Java language, e.g. toString or equals.
018 * 
019 * @author $Author: kinnen $
020 * @version $Rev: 53185 $
021 * @ConQAT.Rating GREEN Hash: 33E5CB8103D480F3E325185F7E181C28
022 */
023public class SuspiciousMethodDetector {
024
025        /** The method's name. */
026        private final String name;
027
028        /** The method's parameter count. */
029        private final int parameterCount;
030
031        /** Constructor. */
032        public SuspiciousMethodDetector(String name, int parameterCount) {
033                CCSMAssert.isNotNull(name);
034                CCSMAssert.isTrue(parameterCount >= 0,
035                                "method cannot have negative parameters");
036                this.name = name;
037                this.parameterCount = parameterCount;
038        }
039
040        /** @see #name */
041        public String getName() {
042                return name;
043        }
044
045        /** @see #parameterCount */
046        public int getParameterCount() {
047                return parameterCount;
048        }
049
050        /** Returns whether the given shallow entity is a suspicious method. */
051        public boolean isSuspicious(ShallowEntity methodEntity) {
052                boolean nameMatches = methodEntity.getName().equalsIgnoreCase(name);
053                if (ShallowParsingUtils.extractParameterNameTokens(methodEntity).size() == parameterCount) {
054                        return nameMatches && !methodEntity.getName().equals(name);
055                }
056                return nameMatches;
057        }
058}
