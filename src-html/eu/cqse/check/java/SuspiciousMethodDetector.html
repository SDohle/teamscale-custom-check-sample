008package eu.cqse.check.java;
009
010import org.conqat.lib.commons.assertion.CCSMAssert;
011
012import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
013import eu.cqse.check.framework.shallowparser.util.ShallowParsingUtils;
014
015/**
016 * Class for detecting suspicious methods, that are similar to a standard method
017 * of the Java language, e.g. toString or equals.
018 */
019public class SuspiciousMethodDetector {
020
021        /** The method's name. */
022        private final String name;
023
024        /** The method's parameter count. */
025        private final int parameterCount;
026
027        /** Constructor. */
028        public SuspiciousMethodDetector(String name, int parameterCount) {
029                CCSMAssert.isNotNull(name);
030                CCSMAssert.isTrue(parameterCount >= 0, "method cannot have negative parameters");
031                this.name = name;
032                this.parameterCount = parameterCount;
033        }
034
035        /** @see #name */
036        public String getName() {
037                return name;
038        }
039
040        /** @see #parameterCount */
041        public int getParameterCount() {
042                return parameterCount;
043        }
044
045        /** Returns whether the given shallow entity is a suspicious method. */
046        public boolean isSuspicious(ShallowEntity methodEntity) {
047                boolean nameMatches = methodEntity.getName().equalsIgnoreCase(name);
048                if (ShallowParsingUtils.extractParameterNameTokens(methodEntity).size() == parameterCount) {
049                        return nameMatches && !methodEntity.getName().equals(name);
050                }
051                return nameMatches;
052        }
053}
