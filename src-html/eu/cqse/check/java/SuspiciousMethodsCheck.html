008package eu.cqse.check.java;
009
010import java.util.Arrays;
011import java.util.List;
012
013import eu.cqse.check.base.EntityCheckBase;
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckGroupNames;
017import eu.cqse.check.framework.core.ECheckParameter;
018import eu.cqse.check.framework.core.EFindingEnablement;
019import eu.cqse.check.framework.scanner.ELanguage;
020import eu.cqse.check.framework.shallowparser.SubTypeNames;
021import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
022
023/**
024 * {@ConQAT.Doc}
025 */
026@ACheck(name = "Suspicious methods", description = "If methods are named similar to ones from Object, this might "
027                + "be a typing error or can lead to confusion. Those methods should be renamed.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
028                                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
029public class SuspiciousMethodsCheck extends EntityCheckBase {
030
031        /** A list of detectors for suspicious methods. */
032        private static final List<SuspiciousMethodDetector> METHOD_DETECTORS = Arrays.asList(
033                        new SuspiciousMethodDetector("equals", 1), new SuspiciousMethodDetector("hashCode", 0),
034                        new SuspiciousMethodDetector("toString", 0), new SuspiciousMethodDetector("compareTo", 1));
035
036        /** {@inheritDoc} */
037        @Override
038        protected String getXPathSelectionString() {
039                return "//METHOD[subtype('" + SubTypeNames.METHOD + "')]";
040        }
041
042        /** {@inheritDoc} */
043        @Override
044        protected void processEntity(ShallowEntity entity) throws CheckException {
045                for (SuspiciousMethodDetector detector : METHOD_DETECTORS) {
046                        if (detector.isSuspicious(entity)) {
047                                createFindingOnFirstLine("Suspicious method " + entity.getName(), entity);
048                                break;
049                        }
050                }
051        }
052}
