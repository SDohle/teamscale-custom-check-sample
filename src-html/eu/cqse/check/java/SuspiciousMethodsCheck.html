008package eu.cqse.check.java;
009
010import java.util.Arrays;
011import java.util.List;
012
013import eu.cqse.check.base.EntityCheckBase;
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckGroupNames;
017import eu.cqse.check.framework.core.ECheckParameter;
018import eu.cqse.check.framework.core.EFindingEnablement;
019import eu.cqse.check.framework.scanner.ELanguage;
020import eu.cqse.check.framework.shallowparser.SubTypeNames;
021import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
022
023/**
024 * {@ConQAT.Doc}
025 * 
026 * @author $Author: hummelb $
027 * @version $Rev: 54535 $
028 * @ConQAT.Rating GREEN Hash: 534B1AF5AC58626B365112C74E5F9E99
029 */
030@ACheck(name = "Suspicious methods", description = "If methods are named similar to ones from Object, this might "
031                + "be a typing error or can lead to confusion. Those methods should be renamed.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
032                                ELanguage.JAVA }, parameters = {
033                                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
034public class SuspiciousMethodsCheck extends EntityCheckBase {
035
036        /** A list of detectors for suspicious methods. */
037        private static final List<SuspiciousMethodDetector> METHOD_DETECTORS = Arrays
038                        .asList(new SuspiciousMethodDetector("equals", 1),
039                                        new SuspiciousMethodDetector("hashCode", 0),
040                                        new SuspiciousMethodDetector("toString", 0),
041                                        new SuspiciousMethodDetector("compareTo", 1));
042
043        /** {@inheritDoc} */
044        @Override
045        protected String getXPathSelectionString() {
046                return "//METHOD[subtype('" + SubTypeNames.METHOD + "')]";
047        }
048
049        /** {@inheritDoc} */
050        @Override
051        protected void processEntity(ShallowEntity entity) throws CheckException {
052                for (SuspiciousMethodDetector detector : METHOD_DETECTORS) {
053                        if (detector.isSuspicious(entity)) {
054                                createFindingOnFirstLine(
055                                                "Suspicious method " + entity.getName(), entity);
056                                break;
057                        }
058                }
059        }
060}
