006package eu.cqse.check.java;
007
008import java.util.Set;
009import java.util.regex.Pattern;
010
011import org.conqat.lib.commons.collections.CollectionUtils;
012
013import eu.cqse.check.base.EntityCheckBase;
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckGroupNames;
017import eu.cqse.check.framework.core.ECheckParameter;
018import eu.cqse.check.framework.core.EFindingEnablement;
019import eu.cqse.check.framework.core.option.ACheckOption;
020import eu.cqse.check.framework.scanner.ELanguage;
021import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
022import eu.cqse.check.framework.util.LanguageFeatureParser;
023
024/** {@AConQAT.Doc} */
025@ACheck(name = "Test class naming convention", description = "Test classes should be distinguishable from other classes by their name. Reports test classes whose name does not match a configured "
026                + "pattern.", groupName = CheckGroupNames.TEST_CONVENTIONS, defaultEnablement = EFindingEnablement.YELLOW, languages = {
027                                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
028public class TestClassNamingConventionCheck extends EntityCheckBase {
029
030        /** Name of the annotation that indicates a test method. */
031        private static final Set<String> TEST_ANNOTATION_SIMPLE_NAMES = CollectionUtils.asHashSet("Test",
032                        "ParameterizedTest", "RepeatedTest", "TestFactory");
033
034        private static final Set<String> TEST_ANNOTATION_QUALIFIED_NAMES = CollectionUtils.asHashSet("org.junit.Test",
035                        "org.junit.jupiter.api.Test", "org.junit.jupiter.api.RepeatedTest", "org.junit.jupiter.api.TestFactory");
036
037        /** {@AConQAT.Doc} */
038        @ACheckOption(name = "Test class naming convention: naming pattern", description = "The name of a test class has to match this regular expression.")
039        private String nameRegex = "^.*Test$";
040
041        /** The compiled name pattern. */
042        private Pattern namePattern;
043
044        /** {@inheritDoc} */
045        @Override
046        public void initialize() {
047                namePattern = Pattern.compile(nameRegex);
048        }
049
050        /** {@inheritDoc} */
051        @Override
052        protected void processEntity(ShallowEntity entity) throws CheckException {
053                if (hasTestMethods(entity)) {
054                        if (!namePattern.matcher(entity.getName()).matches()) {
055                                createFindingOnFirstLine(
056                                                "The class `" + entity.getName() + "` does not match the pattern `" + nameRegex + "`", entity);
057                        }
058                }
059        }
060
061        /** Returns whether the given type has test methods. */
062        private boolean hasTestMethods(ShallowEntity type) throws CheckException {
063                return select(type, "METHOD").stream().anyMatch(method -> LanguageFeatureParser.JAVA.getAnnotations(method)
064                                .stream().anyMatch(annotation -> isTestAnnotationName(annotation.getName())));
065        }
066
067        /** {@inheritDoc} */
068        @Override
069        protected String getXPathSelectionString() {
070                return "//TYPE";
071        }
072
073        /** Check if the name of an annotation refers to a test annotation. */
074        private static boolean isTestAnnotationName(String annotationName) {
075                return TEST_ANNOTATION_SIMPLE_NAMES.contains(annotationName)
076                                || TEST_ANNOTATION_QUALIFIED_NAMES.contains(annotationName);
077        }
078}
