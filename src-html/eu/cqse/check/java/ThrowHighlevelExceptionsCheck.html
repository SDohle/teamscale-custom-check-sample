017package eu.cqse.check.java;
018
019import java.util.List;
020import java.util.Set;
021import java.util.stream.Collectors;
022
023import org.conqat.lib.commons.collections.CollectionUtils;
024import org.conqat.lib.commons.string.StringUtils;
025
026import eu.cqse.check.base.EntityCheckBase;
027import eu.cqse.check.framework.core.ACheck;
028import eu.cqse.check.framework.core.CheckException;
029import eu.cqse.check.framework.core.CheckGroupNames;
030import eu.cqse.check.framework.core.ECheckParameter;
031import eu.cqse.check.framework.core.EFindingEnablement;
032import eu.cqse.check.framework.core.option.ACheckOption;
033import eu.cqse.check.framework.scanner.ELanguage;
034import eu.cqse.check.framework.scanner.IToken;
035import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
036import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
037import eu.cqse.check.framework.util.LanguageFeatureParser;
038
039/**
040 * {@AConQAT.Doc}
041 */
042@ACheck(name = "Throw of generic exception", description = "The exceptions java.lang.Throwable, java.lang.Error, java.lang.RuntimeException and java.lang.Exception must not be thrown by methods. Appropriate subclasses should be used instead.", groupName = CheckGroupNames.EXCEPTION_HANDLING, defaultEnablement = EFindingEnablement.RED, languages = {
043                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
044public class ThrowHighlevelExceptionsCheck extends EntityCheckBase {
045
046        /** {@ConQAT.Doc} */
047        @ACheckOption(name = "Throw of generic exception: Method annotation filter", description = "The check will be disabled if the method is annotated with one of these annotations (comma-separated list).")
048        private Set<String> filteredMethodAnnotations = CollectionUtils.asHashSet("@Override", "@Test");
049
050        /** {@inheritDoc} */
051        @Override
052        public void initialize() {
053                filteredMethodAnnotations = filteredMethodAnnotations.stream().map(annotationName -> annotationName.trim())
054                                .map(annotationName -> StringUtils.stripPrefix(annotationName, "@")).collect(Collectors.toSet());
055        }
056
057        /** {@inheritDoc} */
058        @Override
059        protected String getXPathSelectionString() {
060                return "//METHOD";
061        }
062
063        /** {@inheritDoc} */
064        @Override
065        protected void processEntity(ShallowEntity entity) throws CheckException {
066                IToken throwsToken = LanguageFeatureParser.JAVA.getThrowsToken(entity);
067                if (throwsToken == null) {
068                        return;
069                }
070                List<ShallowEntity> annotations = LanguageFeatureParser.JAVA.getAnnotations(entity);
071                if (annotations.stream().map(ShallowEntity::getName).anyMatch(filteredMethodAnnotations::contains)) {
072                        return;
073                }
074
075                for (List<IToken> thrownTypeTokens : LanguageFeatureParser.JAVA.getThrownTypesTokens(entity)) {
076                        String thrownType = TokenStreamTextUtils.concatTokenTexts(thrownTypeTokens);
077                        if (LanguageFeatureParser.JAVA.isGenericExceptionClass(thrownType)) {
078                                createFinding("Method throws generic exception of type '" + thrownType + "'", throwsToken,
079                                                thrownTypeTokens.get(thrownTypeTokens.size() - 1));
080                        }
081                }
082
083        }
084}
