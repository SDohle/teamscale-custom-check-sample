001package eu.cqse.check.java;
002
003import java.util.List;
004
005import org.conqat.lib.commons.assertion.CCSMAssert;
006import org.conqat.lib.commons.string.StringUtils;
007
008import eu.cqse.check.framework.core.ACheck;
009import eu.cqse.check.framework.core.CheckException;
010import eu.cqse.check.framework.core.CheckGroupNames;
011import eu.cqse.check.framework.core.CheckImplementationBase;
012import eu.cqse.check.framework.core.ECheckParameter;
013import eu.cqse.check.framework.core.option.ACheckOption;
014import eu.cqse.check.framework.scanner.ELanguage;
015import eu.cqse.check.framework.shallowparser.SubTypeNames;
016import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
017
018/**
019 * {@ConQAT.Doc}
020 */
021@ACheck(name = "Unit test classes should be located in the same package as the class under test", description = "Unit test classes should be located in the same package as the class under test.", groupName = CheckGroupNames.TEST_CONVENTIONS, languages = {
022                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
023public class UnitTestPackageMatchCheck extends CheckImplementationBase {
024
025        /** Xpath for packages */
026        private static final String X_PATH_PACKAGE = "/META[subtype('" + SubTypeNames.PACKAGE + "')]";
027        /** Xpath for imports */
028        private static final String X_PATH_IMPORTS = "/META[subtype('" + SubTypeNames.IMPORT + "')]";
029        /** Xpath pattern for test classes (the suffix still needs to be set) */
030        private static final String X_PATH_TEST_CLASSES_PATTERN = "/TYPE[subtype('" + SubTypeNames.CLASS
031                        + "') and name-matches('.*?%s')]";
032        /** Package separator in qualified class name. */
033        private static final char PACKAGE_SEPARATOR = '.';
034
035        /** {@ConQAT.Doc} */
036        @ACheckOption(name = "Test class name ending", description = "The class name suffix of test classes (e.g. \"Test\").")
037        private String testClassNameSuffix = "Test";
038
039        /** {@inheritDoc} */
040        @Override
041        public void execute() throws CheckException {
042                String xPathTestClasses = String.format(X_PATH_TEST_CLASSES_PATTERN, testClassNameSuffix);
043                List<ShallowEntity> testClasses = select(xPathTestClasses);
044
045                for (ShallowEntity testClassUnit : testClasses) {
046                        checkTestClass(testClassUnit);
047                }
048        }
049
050        /** Execute the check for the test class. */
051        private void checkTestClass(ShallowEntity testClassEntity) throws CheckException {
052                String expectedNameOfClassUnderTest = getNameOfExpectedClassUnderTest(testClassEntity.getName());
053                ShallowEntity packageEntity = getPackageEntity();
054                String testClassPackagePrefix = getPackagePrefixFromPackageEntity(packageEntity);
055
056                List<ShallowEntity> importStatements = select(X_PATH_IMPORTS);
057
058                for (ShallowEntity importStatement : importStatements) {
059                        if (isTestClassInWrongPackage(testClassPackagePrefix, expectedNameOfClassUnderTest, importStatement)) {
060                                createFinding(testClassEntity, getPackageNameFromImportStatement(importStatement));
061                                break;
062                        }
063                }
064        }
065
066        /**
067         * Get the package of a class (with suffix {@value #PACKAGE_SEPARATOR}) or an
068         * empty string if the class is in the default package.
069         * 
070         * @param packageEntity
071         *            entity of a package
072         */
073        private static String getPackagePrefixFromPackageEntity(ShallowEntity packageEntity) {
074                if (packageEntity != null) {
075                        return packageEntity.getName() + PACKAGE_SEPARATOR;
076                }
077                return StringUtils.EMPTY_STRING;
078        }
079
080        /**
081         * Get the package name of a class or an empty string if the class is in the
082         * default package.
083         * 
084         * @param importStatement
085         *            entity of an import statement
086         */
087        private static String getPackageNameFromImportStatement(ShallowEntity importStatement) {
088                if (importStatement != null) {
089                        return StringUtils.removeLastPart(importStatement.getName(), PACKAGE_SEPARATOR);
090                }
091                return StringUtils.EMPTY_STRING;
092        }
093
094        /** Create a finding for this check. */
095        private void createFinding(ShallowEntity testClassEntity, String expectedPackageName) throws CheckException {
096                createFindingOnFirstLine(
097                                "`" + testClassEntity.getName() + "` should be in package `" + expectedPackageName + "`",
098                                testClassEntity);
099        }
100
101        /** Get the package entity. May return null. */
102        private ShallowEntity getPackageEntity() throws CheckException {
103                List<ShallowEntity> packageDeclarations = select(X_PATH_PACKAGE);
104
105                if (!packageDeclarations.isEmpty()) {
106                        return packageDeclarations.get(0);
107                }
108
109                return null;
110        }
111
112        /**
113         * Check if the test class is supposed to be in another package.
114         * 
115         * @param testClassPackagePrefix
116         *            package name followed by {@value #PACKAGE_SEPARATOR} or empty (if
117         *            the class is in the default package)
118         * @param expectedSimpleNameOfClassUnderTest
119         *            expected simple name of the class under test
120         * @return true if the class under test is supposed to be in the wrong package
121         */
122        private static boolean isTestClassInWrongPackage(String testClassPackagePrefix,
123                        String expectedSimpleNameOfClassUnderTest, ShallowEntity importStatement) {
124                String qualifiedImportClassName = importStatement.getName();
125
126                if (expectedSimpleNameOfClassUnderTest
127                                .equals(StringUtils.getLastPart(qualifiedImportClassName, PACKAGE_SEPARATOR))) {
128                        // A class is imported which is supposed to be the class under
129                        // (unit) test. -> The test class is either in the wrong package or
130                        // the import is superfluous.
131                        return !qualifiedImportClassName.equals(testClassPackagePrefix + expectedSimpleNameOfClassUnderTest);
132                }
133                return false;
134        }
135
136        /**
137         * Derive the name of the class under test from the name of the test class.
138         * 
139         * @param testClassName
140         *            simple name of test class
141         * @return simple name of expected class under test
142         */
143        private String getNameOfExpectedClassUnderTest(String testClassName) {
144                CCSMAssert.isTrue(testClassName.endsWith(testClassNameSuffix), "Unexpected test class name");
145                return StringUtils.stripSuffix(testClassName, testClassNameSuffix);
146        }
147}
