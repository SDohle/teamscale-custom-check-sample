003import static eu.cqse.check.framework.core.CheckGroupNames.PERFORMANCE;
004import static eu.cqse.check.framework.core.ECheckParameter.ABSTRACT_SYNTAX_TREE;
005import static eu.cqse.check.framework.core.EFindingEnablement.YELLOW;
006import static eu.cqse.check.framework.scanner.ELanguage.JAVA;
007import static eu.cqse.check.framework.scanner.ETokenType.FLOATING_POINT_LITERAL;
008import static eu.cqse.check.framework.scanner.ETokenType.INTEGER_LITERAL;
009import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
010import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
011import static eu.cqse.check.framework.scanner.ETokenType.STRING_LITERAL;
012import static org.conqat.lib.commons.string.StringUtils.EMPTY_STRING;
013
014import java.util.HashMap;
015import java.util.Map;
016
017import eu.cqse.check.base.EntityCheckBase;
018import eu.cqse.check.framework.core.ACheck;
019import eu.cqse.check.framework.core.CheckException;
020import eu.cqse.check.framework.scanner.ETokenType;
021import eu.cqse.check.framework.scanner.IToken;
022import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
023import eu.cqse.check.framework.util.tokens.TokenPattern;
024import eu.cqse.check.framework.util.tokens.TokenPatternMatch;
025
026/**
027 * Java check looking for unnecessary constructor calls of BigInteger/BigDecimal
028 * objects for certain values (0, 1, 10). There are pre-defined constants for
029 * these values.
030 */
031@ACheck(name = "Unnecessary creation of pre-defined `BigInteger`/`BigDecimal` object", //
032                description = "There are a couple pre-defined BigInteger/BigDecimal constants for certain values:\n" + //
033                                "* `BigInteger.ZERO`, `BigInteger.ONE`, `BigInteger.TEN`\n" + //
034                                "* `BigDecimal.ZERO`, `BigDecimal.ONE`, `BigDecimal.TEN`\n" + //
035                                "\n" + "Recreating these constants is thus unnecessary and should be avoided for performance reasons.\n"
036                                + //
037                                "\n" + "*Examples:*\n" + //
038                                "``` java\n" + //
039                                "BigDecimal decimal = new BigDecimal(10);   // reference BigDecimal.TEN instead \n" + //
040                                "BigInteger myInt = new BigInteger(1);   // reference BigInteger.ONE instead \n" + //
041                                "BigInteger bi = new BigInteger(0);   // reference BigInteger.ZERO instead \n" + //
042                                "```", //
043                groupName = PERFORMANCE, languages = { JAVA }, defaultEnablement = YELLOW, parameters = {
044                                ABSTRACT_SYNTAX_TREE })
045public class UnnecessaryBigIntegerInstantiationCheck extends EntityCheckBase {
046
047        /**
048         * Maps the raw constructor values (eg. '0') to their string version ('ZERO'),
049         * to allow a helpful finding message.
050         */
051        private static final Map<String, String> RAW_VALUES_TO_STRING_VERSION = new HashMap<>();
052
053        /**
054         * Marks the type name token in {@link #UNNECESSARY_CONSTRUCTOR_PATTERN}
055         */
056        private final static int TYPE_NAME_TOKEN_GROUP = 1;
057
058        /**
059         * Marks the constructor argument token in
060         * {@link #UNNECESSARY_CONSTRUCTOR_PATTERN}
061         */
062        private final static int CONSTRUCTOR_ARGUMENT_TYPE_TOKEN_GROUP = 2;
063
064        /** Pattern to recognize constructor calls for 'BigInteger' or 'BigDecimal'. */
065        private static final TokenPattern UNNECESSARY_CONSTRUCTOR_PATTERN = new TokenPattern().sequence(ETokenType.NEW)
066                        .regex("BigInteger|BigDecimal").group(TYPE_NAME_TOKEN_GROUP).sequence(LPAREN)
067                        .alternative(STRING_LITERAL, INTEGER_LITERAL, FLOATING_POINT_LITERAL)
068                        .group(CONSTRUCTOR_ARGUMENT_TYPE_TOKEN_GROUP).sequence(RPAREN);
069
070        static {
071                RAW_VALUES_TO_STRING_VERSION.put("0", "ZERO");
072                RAW_VALUES_TO_STRING_VERSION.put("1", "ONE");
073                RAW_VALUES_TO_STRING_VERSION.put("10", "TEN");
074        }
075
076        @Override
077        protected String getXPathSelectionString() {
078                return "//STATEMENT";
079        }
080
081        @Override
082        protected void processEntity(ShallowEntity entity) throws CheckException {
083
084                for (TokenPatternMatch match : UNNECESSARY_CONSTRUCTOR_PATTERN.findAll(entity.ownStartTokens())) {
085
086                        IToken classNameToken = match.getMatchGroup(TYPE_NAME_TOKEN_GROUP).get(0).getTokens().get(0);
087                        String className = classNameToken.getText();
088                        IToken constructorArgumentToken = match.getMatchGroup(CONSTRUCTOR_ARGUMENT_TYPE_TOKEN_GROUP).get(0)
089                                        .getTokens().get(0);
090                        String cleanConstructorArgument = cleanConstructorArgument(constructorArgumentToken);
091                        if (RAW_VALUES_TO_STRING_VERSION.containsKey(cleanConstructorArgument)) {
092                                createFinding("Use `" + className + "." + RAW_VALUES_TO_STRING_VERSION.get(cleanConstructorArgument)
093                                                + "` instead of creating new object", classNameToken, constructorArgumentToken);
094                        }
095
096                }
097        }
098
099        /**
100         * Cleans the constructor argument by removing quotes and trailing 0s that don't
101         * change the actual value.
102         */
103        private static String cleanConstructorArgument(IToken argumentToken) {
104                return argumentToken.getText().replaceAll("[\"]|\\.0+", EMPTY_STRING);
105        }
106}
