017package eu.cqse.check.java;
018
019import java.util.List;
020import java.util.regex.Matcher;
021import java.util.regex.Pattern;
022
023import org.conqat.lib.commons.collections.CollectionUtils;
024import org.conqat.lib.commons.collections.ListMap;
025import org.conqat.lib.commons.string.StringUtils;
026
027import eu.cqse.check.framework.core.ACheck;
028import eu.cqse.check.framework.core.CheckException;
029import eu.cqse.check.framework.core.CheckGroupNames;
030import eu.cqse.check.framework.core.CheckImplementationBase;
031import eu.cqse.check.framework.core.ECheckParameter;
032import eu.cqse.check.framework.core.EFindingEnablement;
033import eu.cqse.check.framework.scanner.ELanguage;
034import eu.cqse.check.framework.scanner.ETokenType;
035import eu.cqse.check.framework.scanner.ETokenType.ETokenClass;
036import eu.cqse.check.framework.scanner.IToken;
037import eu.cqse.check.framework.shallowparser.SubTypeNames;
038import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
039import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
040
041/**
042 * Check for unused imports in Java.
043 *
044 * @author $Author: geissler $
045 * @version $Rev: $
046 * @ConQAT.Rating GREEN Hash: FD25FADFE90A402E741EE78E3BE4B628
047 */
048@ACheck(name = "Unused imports", description = "Unused imports are unnecessary.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
049                ELanguage.JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
050public class UnusedImportsCheck extends CheckImplementationBase {
051
052        /** Single star (marker for star imports). */
053        private static final String STAR = "*";
054
055        /**
056         * Regex that matches Links in JavaDoc comments. Group #1 will contain the
057         * type. Group #2 will contain parameter type(s) if any.
058         */
059        private final static Pattern COMMENT_LINK_PATTERN = Pattern
060                        .compile("@(?:see|link|linkplain|throws)\\s+"
061                                        + "([^ \\t#}(\\n]+)(?:[^ \\t}(\\n]*)(?:[(](.*?)[)])?");
062
063        /** {@inheritDoc} */
064        @Override
065        public void execute() throws CheckException {
066
067                // stores for simple type name the import entities
068                ListMap<String, ShallowEntity> imports = new ListMap<>();
069
070                // we know that import statements are top-level, so processing direct
071                // children is sufficient
072                context.getRootEntity().getChildren()
073                                .forEach(entity -> processEntity(entity, imports));
074
075                context.getTokens().stream()
076                                .filter(token -> token.getType()
077                                                .getTokenClass() == ETokenClass.COMMENT)
078                                .forEach(token -> processComment(token, imports));
079
080                createFindings(imports);
081        }
082
083        /** Recursively process all children entities. */
084        private static void processEntity(ShallowEntity entity,
085                        ListMap<String, ShallowEntity> imports) {
086                if (entity.getType() == EShallowEntityType.META
087                                && SubTypeNames.IMPORT.equals(entity.getSubtype())) {
088                        recordImport(entity, imports);
089                } else {
090                        markIdentifiersAsUsed(entity.includedTokens(), imports);
091                }
092        }
093
094        /** Marks all identifiers found in the given token stream as used types. */
095        private static void markIdentifiersAsUsed(List<IToken> tokens,
096                        ListMap<String, ShallowEntity> imports) {
097                tokens.stream()
098                                .filter(token -> token.getType() == ETokenType.IDENTIFIER)
099                                .forEach(token -> useType(token.getText(), imports));
100        }
101
102        /** Records an import statement. */
103        private static void recordImport(ShallowEntity entity,
104                        ListMap<String, ShallowEntity> imports) {
105                List<IToken> tokens = entity.includedTokens();
106                String typeName = tokens.get(tokens.size() - 2).getText();
107                if (!STAR.equals(typeName)) {
108                        imports.add(typeName, entity);
109                }
110        }
111
112        /** Processes a single comment token and marks included types. */
113        private static void processComment(IToken token,
114                        ListMap<String, ShallowEntity> imports) {
115                Matcher matcher = COMMENT_LINK_PATTERN.matcher(token.getText());
116                while (matcher.find()) {
117                        useType(matcher.group(1), imports);
118
119                        String parameterString = matcher.group(2);
120                        if (parameterString != null) {
121                                for (String parameter : parameterString.split(",")) {
122                                        useType(parameter.trim(), imports);
123                                }
124                        }
125                }
126        }
127
128        /** Marks a type as used by removing the import from the imports. */
129        private static void useType(String typeName,
130                        ListMap<String, ShallowEntity> imports) {
131                imports.removeCollection(typeName);
132        }
133
134        /** Create findings for unused imports. */
135        private void createFindings(ListMap<String, ShallowEntity> imports)
136                        throws CheckException {
137                for (String typeName : imports.getKeys()) {
138                        for (ShallowEntity entity : imports.getCollection(typeName)) {
139                                createFinding(
140                                                "Unused import: "
141                                                                + getFullyQualifiedNameFromImport(entity),
142                                                entity);
143                        }
144                }
145        }
146
147        /** Returns the fully qualified name for an import statement. */
148        private static String getFullyQualifiedNameFromImport(
149                        ShallowEntity entity) {
150                List<String> identifiers = CollectionUtils.filterAndMap(
151                                entity.includedTokens(),
152                                token -> token.getType() == ETokenType.IDENTIFIER,
153                                token -> token.getText());
154                return StringUtils.concat(identifiers, ".");
155        }
156}
