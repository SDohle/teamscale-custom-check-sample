008package eu.cqse.check.java;
009
010import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
011import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
012import static eu.cqse.check.framework.scanner.ETokenType.NEW;
013import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
014import static eu.cqse.check.framework.scanner.ETokenType.STRING_LITERAL;
015
016import java.util.Arrays;
017import java.util.EnumSet;
018import java.util.List;
019
020import eu.cqse.check.base.EntityTokenCheckBase;
021import eu.cqse.check.framework.core.ACheck;
022import eu.cqse.check.framework.core.CheckException;
023import eu.cqse.check.framework.core.CheckGroupNames;
024import eu.cqse.check.framework.core.ECheckParameter;
025import eu.cqse.check.framework.core.EFindingEnablement;
026import eu.cqse.check.framework.core.option.ACheckOption;
027import eu.cqse.check.framework.scanner.ELanguage;
028import eu.cqse.check.framework.scanner.ETokenType;
029import eu.cqse.check.framework.scanner.IToken;
030import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
031
032/**
033 * {@ConQAT.Doc}
034 * 
035 * @author $Author: heinemann $
036 * @version $Rev: 54700 $
037 * @ConQAT.Rating GREEN Hash: E4EAD208F449C13EDD21C64CBF6437FD
038 */
039@ACheck(name = "Unwanted constructor calls", description = "Constructors of some types, e.g. primitive "
040                + "wrappers and String must not be called, because they only create extra objects which take up "
041                + "more memory and reduce performance, without any other functional effect.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
042                                ELanguage.JAVA }, parameters = {
043                                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
044public class UnwantedConstructorCallsCheck extends EntityTokenCheckBase {
045
046        /** Token type sequence that indicates a constructor call. */
047        private static final ETokenType[] CONSTRUCTOR_CALL_SEQUENCE = { NEW,
048                        IDENTIFIER, LPAREN };
049
050        /**
051         * Token types of the first parameter of the String constructor that
052         * indicate a forbidden constructor call. This is either a string literal or
053         * the constructor without parameters (thus there is a closing parenthesis).
054         */
055        private static final EnumSet<ETokenType> FORBIDDEN_STRING_CONSTRUCTOR_PARAMETER_TOKENS = EnumSet
056                        .of(STRING_LITERAL, RPAREN);
057
058        /** {@ConQAT.Doc} */
059        @ACheckOption(name = "Unwanted constructor names", description = "All constructor names, that shall not be called.")
060        private final List<String> constructorNames = Arrays.asList("Boolean",
061                        "Byte", "Short", "Integer", "Long", "Float", "Double", "String");
062
063        /** {@inheritDoc} */
064        @Override
065        public String getXPathSelectionString() {
066                return "//STATEMENT | //ATTRIBUTE";
067        }
068
069        /** {@inheritDoc} */
070        @Override
071        public void processTokens(List<IToken> tokens) throws CheckException {
072                List<Integer> callIndices = TokenStreamUtils
073                                .findTokenTypeSequences(tokens, 0, CONSTRUCTOR_CALL_SEQUENCE);
074                for (int callIndex : callIndices) {
075                        if (isAllowedStringCall(tokens, callIndex)) {
076                                continue;
077                        }
078
079                        IToken typeToken = tokens.get(callIndex + 1);
080                        String typeName = typeToken.getText();
081                        for (String wrapperName : constructorNames) {
082                                if (wrapperName.equals(typeName)) {
083                                        createFinding(
084                                                        "Constructor of type " + wrapperName
085                                                                        + " must not be called explicitly",
086                                                        typeToken);
087                                        return;
088                                }
089                        }
090                }
091        }
092
093        /**
094         * Checks the special case of a String constructor call. If the given index
095         * is a call of the String constructor and its parameter is no string
096         * literal (but there is a parameter) the call is allowed and this method
097         * returns true. Otherwise false is returned.
098         */
099        private static boolean isAllowedStringCall(List<IToken> tokens,
100                        int callIndex) {
101                String typeName = tokens.get(callIndex + 1).getText();
102                int parameterIndex = callIndex + 3;
103
104                if (typeName.equals(String.class.getSimpleName())
105                                && parameterIndex < tokens.size()) {
106                        ETokenType parameterType = tokens.get(parameterIndex).getType();
107                        return !FORBIDDEN_STRING_CONSTRUCTOR_PARAMETER_TOKENS
108                                        .contains(parameterType);
109                }
110                return false;
111        }
112}
