008package eu.cqse.check.java;
009
010import java.util.ArrayList;
011import java.util.Arrays;
012import java.util.List;
013
014import eu.cqse.check.base.EntityTokenCheckBase;
015import eu.cqse.check.framework.core.ACheck;
016import eu.cqse.check.framework.core.CheckException;
017import eu.cqse.check.framework.core.CheckGroupNames;
018import eu.cqse.check.framework.core.ECheckParameter;
019import eu.cqse.check.framework.core.EFindingEnablement;
020import eu.cqse.check.framework.core.option.ACheckOption;
021import eu.cqse.check.framework.scanner.ELanguage;
022import eu.cqse.check.framework.scanner.IToken;
023import eu.cqse.check.framework.shallowparser.SubTypeNames;
024
025/**
026 * {@ConQAT.Doc}
027 * 
028 * @author $Author: hummelb $
029 * @version $Rev: 54535 $
030 * @ConQAT.Rating GREEN Hash: EC235EF1DAD42D64173EDEDB7F6093CB
031 */
032@ACheck(name = "Unwanted method calls", description = "Some methods like System.exit or System.gc might be unwanted, because they "
033                + "might introduce unwanted side effect or reduce performance.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
034                                ELanguage.JAVA }, parameters = {
035                                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
036public class UnwantedMethodCallsCheck extends EntityTokenCheckBase {
037
038        /** {@ConQAT.Doc} */
039        @ACheckOption(name = "Unwanted methods", description = "Names of methods that should not be called.")
040        private List<String> unwantedMethods = Arrays.asList("System.exit",
041                        "System.gc", "System.out.println", "System.err.println");
042
043        /** A list of recognizers for unwanted methods. */
044        private List<StaticMethodCallRecognizer> callRecognizers;
045
046        /** {@inheritDoc} */
047        @Override
048        public void initialize() {
049                callRecognizers = new ArrayList<>();
050                for (String unwantedMethod : unwantedMethods) {
051                        callRecognizers.add(new StaticMethodCallRecognizer(unwantedMethod));
052                }
053        }
054
055        /** {@inheritDoc} */
056        @Override
057        protected String getXPathSelectionString() {
058                return "//STATEMENT[subtype('" + SubTypeNames.SIMPLE_STATEMENT + "')]";
059        }
060
061        /** {@inheritDoc} */
062        @Override
063        protected void processTokens(List<IToken> tokens) throws CheckException {
064                for (StaticMethodCallRecognizer recognizer : callRecognizers) {
065                        List<Integer> callIndices = recognizer.findCallsInTokens(tokens);
066                        for (int callIndex : callIndices) {
067                                createFinding(
068                                                "Method " + recognizer.getFullQualifiedMethodName()
069                                                                + " must not be called",
070                                                tokens.get(callIndex),
071                                                tokens.get(callIndex + recognizer.getTokenLength()));
072                        }
073                }
074        }
075}
