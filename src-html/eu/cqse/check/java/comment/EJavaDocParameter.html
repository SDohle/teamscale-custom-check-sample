006package eu.cqse.check.java.comment;
007
008import static eu.cqse.check.framework.scanner.ETokenType.AT;
009import static eu.cqse.check.framework.scanner.ETokenType.EOL;
010import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
011import static eu.cqse.check.framework.scanner.ETokenType.JAVADOC_INLINE_TAG;
012import static eu.cqse.check.framework.scanner.ETokenType.LBRACE;
013import static eu.cqse.check.framework.scanner.ETokenType.NUMBER_WORD;
014import static eu.cqse.check.framework.scanner.ETokenType.RBRACE;
015import static eu.cqse.check.framework.scanner.ETokenType.REFERENCE;
016import static eu.cqse.check.framework.scanner.ETokenType.WORD;
017import static eu.cqse.check.framework.shallowparser.framework.EShallowEntityType.METHOD;
018import static eu.cqse.check.framework.shallowparser.framework.EShallowEntityType.TYPE;
019
020import java.util.EnumSet;
021import java.util.HashSet;
022import java.util.Set;
023import java.util.stream.Collectors;
024
025import org.conqat.lib.commons.collections.CollectionUtils;
026
027import eu.cqse.check.framework.scanner.ETokenType;
028import eu.cqse.check.framework.scanner.IToken;
029import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
030import eu.cqse.check.framework.util.LanguageFeatureParser;
031import eu.cqse.check.framework.util.tokens.SingleTokenPatternBase;
032import eu.cqse.check.framework.util.tokens.TokenPattern;
033
034/**
035 * Enumeration class for the different types of parameter for JavaDoc tags
036 */
037public enum EJavaDocParameter {
038
039        SINGLE_WORD_PARAMETER("single word"),
040        CLASS_REFERENCE_PARAMETER("class reference"),
041        TEXT_PARAMETER("text"),
042        OPTIONAL_TEXT_PARAMETER("optional text"),
043        REFERENCE_PARAMETER("reference"),
044        PARAMETER_PARAMETER("parameter name"),
045        OPTIONAL_REFERENCE_PARAMETER("optional reference");
046
047        /**
048         * {@link ETokenType#REFERENCE} {@link ETokenType#WORD}
049         * {@link ETokenType#NUMBER_WORD} {@link ETokenType#AT}
050         */
051        public static final EnumSet<ETokenType> TEXT_TOKEN_TYPES = EnumSet.of(REFERENCE, WORD, NUMBER_WORD, AT);
052
053        /**
054         * {@link ETokenType#EOL}
055         *
056         * @see #TEXT_TOKEN_TYPES
057         */
058        public static final EnumSet<ETokenType> TEXT_TOKEN_TYPES_AND_EOL = EnumSet.of(REFERENCE, WORD, NUMBER_WORD, AT,
059                        EOL);
060
061        /** {@link ETokenType#JAVADOC_INLINE_TAG} {@link ETokenType#LBRACE} */
062        public static final EnumSet<ETokenType> OPENING_TOKEN_TYPES = EnumSet.of(JAVADOC_INLINE_TAG, LBRACE);
063
064        /** {@link ETokenType#RBRACE} */
065        public static final EnumSet<ETokenType> CLOSING_TOKEN_TYPES = EnumSet.of(RBRACE);
066
067        /** TokenPattern skipping nested braces and inline tags */
068        public static final TokenPattern NESTED_BRACES = new TokenPattern().skipNested(OPENING_TOKEN_TYPES,
069                        CLOSING_TOKEN_TYPES, false);
070
071        /** name of parameter */
072        private final String parameterName;
073
074        /** Creates a JavaDocParameter */
075        EJavaDocParameter(String parameterName) {
076                this.parameterName = parameterName;
077        }
078
079        /** Returns pattern for checking */
080        public TokenPattern getPattern(ShallowEntity relatedEntity) {
081                TokenPattern parameterPattern = new TokenPattern();
082                switch (this) {
083                case CLASS_REFERENCE_PARAMETER:
084                        parameterPattern.sequence(EnumSet.of(WORD, NUMBER_WORD, REFERENCE));
085                        break;
086                case SINGLE_WORD_PARAMETER:
087                        parameterPattern.sequence(EnumSet.of(WORD, NUMBER_WORD));
088                        break;
089                case OPTIONAL_TEXT_PARAMETER:
090                case TEXT_PARAMETER:
091                        parameterPattern.repeatedAtLeastOnce(new TokenPattern().alternative(TEXT_TOKEN_TYPES, NESTED_BRACES))
092                                        .repeated(new TokenPattern().alternative(TEXT_TOKEN_TYPES_AND_EOL, NESTED_BRACES));
093                        break;
094                case OPTIONAL_REFERENCE_PARAMETER:
095                case REFERENCE_PARAMETER:
096                        parameterPattern.sequence(EnumSet.of(WORD, REFERENCE));
097                        break;
098                case PARAMETER_PARAMETER:
099                        Set<String> parameterNames = getParameterNames(relatedEntity);
100                        parameterPattern.optional(getStringMatcher("<")).alternative(getStringMatcher(parameterNames))
101                                        .optional(getStringMatcher(">"));
102                        break;
103                default:
104                        break;
105                }
106
107                if (isOptional()) {
108                        parameterPattern = new TokenPattern().optional(parameterPattern);
109                }
110                return parameterPattern;
111        }
112
113        /** Returns whether this parameter is optional */
114        public boolean isOptional() {
115                return name().startsWith("OPTIONAL");
116        }
117
118        /** @see #getStringMatcher(Set) */
119        private static SingleTokenPatternBase getStringMatcher(final String string) {
120                return getStringMatcher(CollectionUtils.asHashSet(string));
121        }
122
123        /** Returns SingleTokenPattern matches with given strings */
124        private static SingleTokenPatternBase getStringMatcher(final Set<String> strings) {
125                return new SingleTokenPatternBase() {
126                        @Override
127                        public boolean matchesToken(IToken token) {
128                                return strings.contains(token.getText());
129                        }
130
131                        /** {@inheritDoc} */
132                        @Override
133                        public String toString() {
134                                return strings.toString();
135                        }
136                };
137        }
138
139        /** Returns all parameter names of the related entity. */
140        private static Set<String> getParameterNames(ShallowEntity relatedEntity) {
141                if (relatedEntity.getType() == METHOD) {
142                        return LanguageFeatureParser.JAVA.getParameterTokens(relatedEntity).stream()
143                                        .filter(token -> IDENTIFIER == token.getType()).map(IToken::getText).collect(Collectors.toSet());
144                }
145                if (relatedEntity.getType() == TYPE) {
146                        return new HashSet<>(LanguageFeatureParser.JAVA.getGenericNames(relatedEntity));
147                }
148                return CollectionUtils.emptySet();
149        }
150
151        @Override
152        public String toString() {
153                return "<" + parameterName + ">";
154        }
155}
