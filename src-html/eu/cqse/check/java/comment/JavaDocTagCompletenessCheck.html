047/** {@ConQAT.Doc} */
048@ACheck(name = "Incorrect JavaDoc usage", description = "Incorrect used JavaDoc tags clutter comments and make them less usable", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
049                JAVA }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
050public class JavaDocTagCompletenessCheck extends CommentCheckBase {
051
052        /** All JavaDoc block tags (initialized statically). */
053        private static final Set<IJavaDocTag> BLOCK_TAGS = new HashSet<>();
054
055        /** All JavaDoc inline tags (initialized statically). */
056        private static final Set<IJavaDocTag> INLINE_TAGS = new HashSet<>();
057
058        /** All JavaDoc tags by name (initialized statically). */
059        private static final Map<String, IJavaDocTag> ALL_TAGS_BY_NAME = new HashMap<>();
060
061        static {
062                for (IJavaDocTag blockTag : EJavaDocBlockTag.values()) {
063                        BLOCK_TAGS.add(blockTag);
064                        ALL_TAGS_BY_NAME.put(blockTag.getName(), blockTag);
065                }
066                for (IJavaDocTag inlineTag : EJavaDocInlineTag.values()) {
067                        INLINE_TAGS.add(inlineTag);
068                        ALL_TAGS_BY_NAME.put("{" + inlineTag.getName(), inlineTag);
069                }
070        }
071
072        @Override
073        protected void processComment(IToken completeComment, int startLine, int endLine) throws CheckException {
074                if (completeComment.getType() != DOCUMENTATION_COMMENT) {
075                        return;
076                }
077                Optional<ShallowEntity> commentedEntity = getCommentedEntity(completeComment,
078                                context.getTokens(ECodeViewOption.FILTERED_PREPROCESSED));
079                if (!commentedEntity.isPresent()) {
080                        return;
081                }
082
083                List<IToken> commentTokens = ScannerUtils.getTokens(
084                                LanguageProperties.of(context.getLanguage()).getCommentContent(completeComment.getText()), JAVADOC);
085
086                checkComment(completeComment, new LinkedList<>(commentTokens), commentedEntity.get());
087        }
088
089        /**
090         * Checks the comment for misuses of JavaDoc tags and creates findings where
091         * necessary.
092         */
093        private void checkComment(IToken completeComment, List<IToken> commentTokens, ShallowEntity relatedEntity)
094                        throws CheckException {
095                SetMap<IToken, String> findingsByComment = collectFindings(completeComment, commentTokens, relatedEntity);
096                for (Entry<IToken, Set<String>> entry : findingsByComment) {
097                        IToken comment = entry.getKey();
098                        Set<String> findings = entry.getValue();
099                        createJavaDocFinding(StringUtils.concat(findings, "\n"), comment);
100                }
101        }
102
103        /**
104         * For all comments, collects and returns all findings related to the respective
105         * comment. The map structure is Comment -> Finding Messages.
106         */
107        private static SetMap<IToken, String> collectFindings(IToken completeComment, List<IToken> commentTokens,
108                        ShallowEntity relatedEntity) {
109                SetMap<IToken, String> findingsByComment = new SetMap<>();
110                while (!commentTokens.isEmpty()) {
111                        IToken token = commentTokens.remove(0);
112                        if (!isJavaDocTag(token)) {
113                                continue;
114                        }
115                        Optional<IJavaDocTag> optionalJavaDocTag = getByTagName(token);
116                        if (!optionalJavaDocTag.isPresent()) {
117                                continue;
118                        }
119                        IJavaDocTag javaDocTag = optionalJavaDocTag.get();
120                        int endIndex = computeEndIndex(token, commentTokens);
121                        if (endIndex == TokenStreamUtils.NOT_FOUND) {
122                                findingsByComment.add(completeComment, "`" + javaDocTag.getName() + "` - Unterminated inline tag");
123                                continue;
124                        }
125                        List<IToken> tagTokens = commentTokens.subList(0, endIndex);
126                        commentTokens = commentTokens.subList(tagTokens.size(), commentTokens.size());
127                        Optional<String> finding = createFindingMessage(token, tagTokens, relatedEntity);
128                        if (finding.isPresent()) {
129                                findingsByComment.add(completeComment, "`" + javaDocTag.getName() + "` - " + finding.get());
130                        }
131                }
132                return findingsByComment;
133        }
134
135        /** Creates and returns a finding for the given JavaDoc tag. */
136        private void createJavaDocFinding(String findingMessage, IToken completeComment) throws CheckException {
137                createFinding("Incorrect JavaDoc Tags: " + findingMessage, completeComment);
138        }
139
140        /**
141         * Returns the entity following the given comment token, i.e. the entity the
142         * comment references. META Entities are ignored.
143         */
144        private Optional<ShallowEntity> getCommentedEntity(IToken commentToken, List<IToken> filteredPreprocessedTokens)
145                        throws CheckException {
146                int index = filteredPreprocessedTokens.indexOf(commentToken) + 1;
147                if (index == filteredPreprocessedTokens.size()) {
148                        return Optional.empty();
149                }
150                IToken nextToken = filteredPreprocessedTokens.get(index);
151                return findEntity(nextToken, context.getAbstractSyntaxTree(ECodeViewOption.FILTERED_PREPROCESSED));
152
153        }
154
155        /**
156         * Searches recursively for and returns the entity following the given comment
157         * token, i.e. the entity the comment references. META Entities are ignored.
158         */
159        private static Optional<ShallowEntity> findEntity(IToken nextToken, List<ShallowEntity> shallowEntities) {
160                for (ShallowEntity entity : shallowEntities) {
161                        int index = entity.includedTokens().indexOf(nextToken);
162                        switch (index) {
163                        case -1:
164                                continue;
165                        case 0:
166                                if (entity.getType() == META) {
167                                        // This is a package entity
168                                        continue;
169                                }
170                                return Optional.of(entity);
171                        default:
172                                return findEntity(nextToken, entity.getChildren());
173                        }
174                }
175                return Optional.empty();
176        }
177
178        /** Returns JavaDocTag with matching tag name */
179        public static Optional<IJavaDocTag> getByTagName(IToken token) {
180                return Optional.ofNullable(ALL_TAGS_BY_NAME.get(token.getText()));
181        }
182
183        /** Returns whether the given token is a JavaDoc tag. */
184        public static boolean isJavaDocTag(IToken token) {
185                return EnumSet.of(JAVADOC_TAG, JAVADOC_INLINE_TAG).contains(token.getType());
186        }
187
188        /** Returns the end index of the given JavaDoc tag. */
189        public static int computeEndIndex(IToken tagToken, List<IToken> commentTokens) {
190                if (tagToken.getType() == JAVADOC_INLINE_TAG) {
191                        return TokenStreamUtils.findMatchingClosingToken(commentTokens, 0, OPENING_TOKEN_TYPES,
192                                        CLOSING_TOKEN_TYPES);
193                }
194                int index = TokenStreamUtils.firstTokenOfType(commentTokens, 0, JAVADOC_TAG);
195                if (index == NOT_FOUND) {
196                        return commentTokens.size();
197                }
198                return index;
199        }
200
201        /** Creates and returns the relevant finding message (if any) for this token. */
202        public static Optional<String> createFindingMessage(IToken token, List<IToken> commentTokens,
203                        ShallowEntity relatedEntity) {
204                IJavaDocTag javaDocTag = getByTagName(token).get();
205                if (token.getType() == JAVADOC_INLINE_TAG) {
206                        int nextInlineTag = TokenStreamUtils.firstTokenOfType(commentTokens, 0, JAVADOC_INLINE_TAG);
207                        if (nextInlineTag != NOT_FOUND) {
208                                return Optional.of("Nesting inline JavaDoc tags");
209                        }
210                }
211                if (conformsToExpectedPattern(javaDocTag, commentTokens, relatedEntity)) {
212                        return Optional.empty();
213                }
214                if (javaDocTag == EJavaDocBlockTag.PARAM && isValidParamSyntax(commentTokens)) {
215                        return Optional.of("Parameter " + commentTokens.get(0).getText() + " is not declared");
216                }
217                return Optional.of("Pattern expected but not observed: " + javaDocTag.toString());
218        }
219
220        /**
221         * Returns whether the given JavaDoc tag and the tokens following it conform to
222         * the pattern expected for that tag. This includes checking for parameter name
223         * matches (using the given related entity).
224         */
225        private static boolean conformsToExpectedPattern(IJavaDocTag tag, List<IToken> tagTokens,
226                        ShallowEntity relatedEntity) {
227                List<EJavaDocParameter> parameters = tag.getParameters();
228                if (parameters.isEmpty()) {
229                        return tagTokens.isEmpty();
230                } else if (allOptional(parameters) && tagTokens.isEmpty()) {
231                        return true;
232                }
233                TokenPattern pattern = new TokenPattern().beginningOfStream();
234                for (EJavaDocParameter parameter : parameters) {
235                        pattern.repeated(EOL).sequence(parameter.getPattern(relatedEntity)).repeated(EOL);
236                }
237                pattern.endOfStream();
238                return pattern.matchesAnywhere(tagTokens);
239        }
240
241        private static boolean allOptional(List<EJavaDocParameter> parameters) {
242                for (EJavaDocParameter parameter : parameters) {
243                        if (!parameter.isOptional()) {
244                                return false;
245                        }
246                }
247                return true;
248        }
249
250        /**
251         * Checks whether the syntax of the comment tokens is a valid parameter
252         * definition. If it is, this only leaves the option that the given parameter
253         * name must have been invalid.
254         */
255        private static boolean isValidParamSyntax(List<IToken> commentTokens) {
256                return TokenStreamUtils.hasTokenTypeSequence(commentTokens, 0, ETokenType.WORD, ETokenType.WORD)
257                                || TokenStreamUtils.hasTokenTypeSequence(commentTokens, 0, ETokenType.WORD, ETokenType.EOL,
258                                                ETokenType.WORD);
259        }
