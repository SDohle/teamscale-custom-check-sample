008package eu.cqse.check.javascript;
009
010import org.conqat.lib.commons.collections.UnmodifiableList;
011
012import eu.cqse.check.base.EntityCheckBase;
013import eu.cqse.check.framework.core.ACheck;
014import eu.cqse.check.framework.core.CheckException;
015import eu.cqse.check.framework.core.ECheckParameter;
016import eu.cqse.check.framework.core.EFindingEnablement;
017import eu.cqse.check.framework.scanner.ELanguage;
018import eu.cqse.check.framework.scanner.ETokenType;
019import eu.cqse.check.framework.scanner.IToken;
020import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
021import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
022
023/**
024 * {@ConQAT.Doc}
025 *
026 * @author $Author: heinemann $
027 * @version $Rev: 53052 $
028 * @ConQAT.Rating GREEN Hash: 36B6986FA6015362DE5B1A5CE01B8565
029 */
030@ACheck(name = "Do not use == and !=", description = "The comparison operators == and != should be avoided. Their behaviour is inconsistent and the performance of === and !== is better.", groupName = "Javascript (built-in)", defaultEnablement = EFindingEnablement.YELLOW, languages = { ELanguage.JAVASCRIPT }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
031public class DoNotUseEqEqCheck extends EntityCheckBase {
032
033        /** {@inheritDoc} */
034        @Override
035        protected String getXPathSelectionString() {
036                return "//STATEMENT";
037        }
038
039        /** {@inheritDoc} */
040        @Override
041        protected void processEntity(ShallowEntity entity) throws CheckException {
042                UnmodifiableList<IToken> tokens = entity.ownStartTokens();
043                if (TokenStreamUtils.containsAny(tokens, ETokenType.EQEQ,
044                                ETokenType.NOTEQ)) {
045                        createFindingOnFirstLine(
046                                        "Do not use == or !=, use === or !== instead.", entity);
047                }
048        }
049}
