008package eu.cqse.check.javascript;
009
010import java.util.EnumSet;
011import java.util.List;
012
013import org.conqat.lib.commons.collections.UnmodifiableList;
014
015import eu.cqse.check.base.EntityCheckBase;
016import eu.cqse.check.framework.core.ACheck;
017import eu.cqse.check.framework.core.CheckException;
018import eu.cqse.check.framework.core.ECheckParameter;
019import eu.cqse.check.framework.core.EFindingEnablement;
020import eu.cqse.check.framework.scanner.ELanguage;
021import eu.cqse.check.framework.scanner.ETokenType;
022import eu.cqse.check.framework.scanner.IToken;
023import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
024import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
025
026/**
027 * {@ConQAT.Doc}
028 */
029@ACheck(name = "Do not use == and !=", description = "The comparison operators == and != should be avoided. Their behavior is inconsistent and the performance of === and !== is better.", groupName = "Javascript (built-in)", defaultEnablement = EFindingEnablement.YELLOW, languages = {
030                ELanguage.JAVASCRIPT }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
031public class DoNotUseEqEqCheck extends EntityCheckBase {
032
033        /** Set of tokens, which should be avoided while programming JavaScript. */
034        public static final EnumSet<ETokenType> UNACCEPTED_TOKENS = EnumSet.of(ETokenType.EQEQ, ETokenType.NOTEQ);
035
036        /** {@inheritDoc} */
037        @Override
038        protected String getXPathSelectionString() {
039                return "//STATEMENT";
040        }
041
042        /** {@inheritDoc} */
043        @Override
044        protected void processEntity(ShallowEntity entity) throws CheckException {
045                UnmodifiableList<IToken> tokens = entity.ownStartTokens();
046                List<Integer> indexes = TokenStreamUtils.findAll(tokens, UNACCEPTED_TOKENS);
047                for (Integer index : indexes) {
048                        createFinding("Do not use == or !=, use === or !== instead.", tokens.get(index));
049                }
050        }
051}
