017package eu.cqse.check.javascript;
018
019import static eu.cqse.check.framework.scanner.ETokenType.NEW;
020
021import java.util.HashSet;
022
023import eu.cqse.check.framework.core.ACheck;
024import eu.cqse.check.framework.core.CheckException;
025import eu.cqse.check.framework.core.ECheckParameter;
026import eu.cqse.check.framework.core.EFindingEnablement;
027import eu.cqse.check.framework.core.option.ACheckOption;
028import eu.cqse.check.framework.scanner.ELanguage;
029import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
030import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
031
032/**
033 * This check checks if JavaScript method calls are supported by (a) Internet
034 * Explorer in general and (b) - if enabled - Internet Explorer version 8. Since
035 * we cannot use type resolution in JavaScript, a heuristic is used to check
036 * suspicious method calls. Specifically, this check will create a finding if
037 * the JS method is called with the expected number of arguments and is no
038 * JQuery or Closure method call.
039 */
040@ACheck(name = "Internet Explorer compatibility check", description = "Some method calls on global objects are not be supported by all browsers. Please refer to online resources like the [Mozilla Developer Network]((https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference)) for details.", groupName = "Cross-Browser Compatibility", defaultEnablement = EFindingEnablement.YELLOW, languages = {
041                ELanguage.JAVASCRIPT }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
042public class InternetExplorerCompatCheck extends JavascriptMethodCompatibilityCheckBase {
043
044        private static final HashSet<String> INCOMPATIBLE_CONSTRUCTORS = new HashSet<>();
045
046        /** {@ConQAT.Doc} */
047        @ACheckOption(name = "Check method compatibility for IE8", description = "Checks if method calls of global objects are available on Internet Explorer 8.")
048        private boolean checkIE8Compatibiltiy;
049
050        static {
051                INCOMPATIBLE_CONSTRUCTORS.add("Event");
052        }
053
054        @Override
055        protected void processEntity(ShallowEntity entity) throws CheckException {
056                super.processEntity(entity);
057
058                for (String constructor : INCOMPATIBLE_CONSTRUCTORS) {
059                        for (int tokenIndex : TokenStreamTextUtils.findAllSequences(entity.includedTokens(), 0, NEW, "new",
060                                        constructor, "(")) {
061                                createFinding("Constructor for `" + constructor + "` will crash in Internet Explorer",
062                                                entity.includedTokens().get(tokenIndex));
063                        }
064                }
065        }
066
067        /** Fills the method map with candidates that may create a finding. */
068        @Override
069        protected void buildMethodMap() {
070
071                // String methods
072                addSuspiciousMethod("startsWith", 1, 2);
073                addSuspiciousMethod("endsWith", 1, 2);
074                addSuspiciousMethod("includes", 1, 2);
075
076                // Number methods
077                addSuspiciousMethod("isFinite", 1);
078                addSuspiciousMethod("isNaN", 1);
079                addSuspiciousMethod("isInteger", 1);
080                addSuspiciousMethod("isSafeInteger", 1);
081
082                // Arrays
083                addSuspiciousMethod("copyWithin", 2, 3);
084                addSuspiciousMethod("find", 1, 2);
085                addSuspiciousMethod("findIndex", 1, 2);
086
087                if (checkIE8Compatibiltiy) {
088                        addIE8Methods();
089                }
090        }
091
092        /** {@inheritDoc} */
093        @Override
094        protected String getFindingMessage(String methodName) {
095                return "`" + methodName + "()` is not available on Internet Explorer";
096        }
097
098        /**
099         * Adds method calls that are not available in Internet Explorer 8 to the
100         * exclusion map.
101         */
102        private void addIE8Methods() {
103                // String methods
104                addSuspiciousMethod("trim", NO_ARGS);
105
106                // Date methods
107                addSuspiciousMethod("toISOString", NO_ARGS);
108
109                // Arrays
110                addSuspiciousMethod("every", 1, 2);
111                addSuspiciousMethod("isArray", 1);
112                addSuspiciousMethod("reduce", 1, 2);
113                addSuspiciousMethod("reduceRight", 1, 2);
114                addSuspiciousMethod("some", 1, 2);
115                addSuspiciousMethod("forEach", 1, 2);
116                addSuspiciousMethod("map", 1, 2);
117                addSuspiciousMethod("filter", 1, 2);
118        }
119
120}
