017package eu.cqse.check.javascript;
018
019import java.util.HashMap;
020import java.util.List;
021import java.util.Map;
022
023import eu.cqse.check.base.EntityCheckBase;
024import eu.cqse.check.framework.core.CheckException;
025import eu.cqse.check.framework.scanner.ETokenType;
026import eu.cqse.check.framework.scanner.IToken;
027import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
028import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
029
030/**
031 * Base class for checks that check for method calls that lack cross-browser
032 * compatibility. Since we cannot use type resolution in JavaScript, a heuristic
033 * is used to check suspicious method calls. Specifically, this check will
034 * create a finding if the JS method is called with the expected number of
035 * arguments and is no JQuery or Closure method call.
036 */
037public abstract class JavascriptMethodCompatibilityCheckBase extends EntityCheckBase {
038
039        /**
040         * Maps method names to their expected number of parameters (min and max).
041         */
042        private final Map<String, Integer[]> suspiciousMethods = new HashMap<>();
043
044        /** The value to indicate no method arguments. */
045        protected static final Integer[] NO_ARGS = new Integer[] {};
046
047        /** The identifier token for JQuery methods. */
048        private static final String JQUERY_IDENTIFIER = "$";
049
050        /** {@inheritDoc} */
051        @Override
052        public void initialize() throws CheckException {
053                super.initialize();
054                buildMethodMap();
055        }
056
057        /**
058         * Fills the method map with candidates that may create a finding. Use
059         * {@code addSuspicousMethod()} to insert a method.
060         */
061        protected abstract void buildMethodMap();
062
063        /**
064         * Adds a suspicious method to the exclusion map.
065         * 
066         * @param methodName
067         *            The name of the suspicious method.
068         * @param argsCountIntervall
069         *            The (optional) minimal and (required) maximal number of arguments
070         *            this method is expected to be called with. Use an empty array for
071         *            no expected method arguments.
072         */
073        protected void addSuspiciousMethod(String methodName, Integer... argsCountIntervall) {
074                suspiciousMethods.put(methodName, argsCountIntervall);
075        }
076
077        /** {@inheritDoc} */
078        @Override
079        protected String getXPathSelectionString() {
080                return "/STATEMENT";
081        }
082
083        /** {@inheritDoc} */
084        @Override
085        protected void processEntity(ShallowEntity entity) throws CheckException {
086                List<IToken> tokens = entity.includedTokens();
087                List<Integer> indices = TokenStreamUtils.firstTokenOfTypeSequences(tokens, 0, ETokenType.DOT,
088                                ETokenType.IDENTIFIER, ETokenType.LPAREN);
089
090                for (int index : indices) {
091
092                        int methodNameIndex = ++index;
093                        IToken token = tokens.get(methodNameIndex);
094                        String methodName = token.getText();
095
096                        if (shouldReportMethod(methodName, tokens, methodNameIndex)) {
097                                createFinding(getFindingMessage(methodName), token);
098                        }
099
100                }
101
102        }
103
104        /** @return a finding message for the given method. */
105        protected abstract String getFindingMessage(String methodName);
106
107        /**
108         * Determines whether the given method should be reported as finding.
109         * 
110         * @param methodName
111         *            the name of the method.
112         * @param tokens
113         *            the tokens of the current statement.
114         * @param methodNameIndex
115         *            the index of the method name within the tokens.
116         */
117        private boolean shouldReportMethod(String methodName, List<IToken> tokens, int methodNameIndex) {
118                return suspiciousMethods.containsKey(methodName) && matchesArgumentsCount(methodName, methodNameIndex, tokens)
119                                && !isClosureMethod(tokens, methodNameIndex) && !isJQueryMethod(tokens, methodNameIndex);
120        }
121
122        /**
123         * Checks whether the method call is part of a JQuery expression.
124         * 
125         * @param tokens
126         *            the tokens of the current statement.
127         * @param methodNameIndex
128         *            the index of the method name within the tokens.
129         */
130        private static boolean isJQueryMethod(List<IToken> tokens, int methodNameIndex) {
131                int openParenthesisCount = 0;
132                for (int i = methodNameIndex; i >= 0; i--) {
133                        IToken token = tokens.get(i);
134                        if (token.getType() == ETokenType.RPAREN) {
135                                openParenthesisCount++;
136                        } else if (token.getType() == ETokenType.LPAREN) {
137                                openParenthesisCount--;
138                        }
139
140                        if (token.getText().equals(JQUERY_IDENTIFIER) && openParenthesisCount == 0) {
141                                return true;
142                        }
143                }
144
145                return false;
146        }
147
148        /**
149         * Determines if the actual <b>number</b> of method arguments is within the
150         * expected range as defined in the exlusion map.
151         * 
152         * @param methodName
153         *            the name of the method.
154         * @param methodNameIndex
155         *            the index of the method name within the tokens.
156         * @param tokens
157         *            the tokens of the current statement.
158         */
159        private boolean matchesArgumentsCount(String methodName, int methodNameIndex, List<IToken> tokens) {
160                Integer[] expectedArgsCount = suspiciousMethods.get(methodName);
161
162                int actualArgs = countMethodArgs(tokens, methodNameIndex);
163
164                if (expectedArgsCount.length == 0) {
165                        return actualArgs == 0;
166                }
167
168                int minArgs = expectedArgsCount[0];
169                int maxArgs = minArgs;
170                if (expectedArgsCount.length > 1) {
171                        maxArgs = expectedArgsCount[1];
172                }
173                return actualArgs >= minArgs && actualArgs <= maxArgs;
174        }
175
176        /**
177         * Counts the number of arguments for the given method.
178         * 
179         * @param tokens
180         *            the tokens of the current statement.
181         * @param methodNameIndex
182         *            the index of the method name within the tokens.
183         */
184        private static int countMethodArgs(List<IToken> tokens, int methodNameIndex) {
185                int commaCount = 0;
186                int innerBracketsCounter = 0;
187
188                if (tokens.get(methodNameIndex + 2).getType() == ETokenType.RPAREN) {
189                        // Method is immediately followed by ')', so it has no arguments.
190                        return 0;
191                }
192
193                for (int j = methodNameIndex + 2; j < tokens.size(); j++) {
194                        IToken token = tokens.get(j);
195
196                        if (innerBracketsCounter == 0 && token.getType() == ETokenType.COMMA) {
197                                commaCount++;
198                                continue;
199                        }
200
201                        if (token.getType() == ETokenType.LPAREN) {
202                                innerBracketsCounter++;
203                        } else if (token.getType() == ETokenType.RPAREN) {
204                                innerBracketsCounter--;
205
206                                if (innerBracketsCounter < 0) {
207                                        break;
208                                }
209                        }
210                }
211
212                return commaCount + 1;
213        }
214
215        /**
216         * Determines if the current method call if part of a Google closure call, i.e.
217         * starts with 'goog.'
218         * 
219         * @param tokens
220         *            the tokens of the current statement.
221         * @param methodNameIndex
222         *            the index of the method name within the tokens.
223         */
224        private static boolean isClosureMethod(List<IToken> tokens, int methodNameIndex) {
225                int currentTokenIndex = methodNameIndex;
226                ETokenType currentType;
227
228                do {
229                        currentTokenIndex--;
230                        currentType = tokens.get(currentTokenIndex).getType();
231                        if (tokens.get(currentTokenIndex).getText().equals("goog"))
232                                return true;
233                } while (currentTokenIndex > 0 && (currentType == ETokenType.DOT || currentType == ETokenType.IDENTIFIER));
234
235                return false;
236        }
237}
