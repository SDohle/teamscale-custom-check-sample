017package eu.cqse.check.javascript;
018
019import java.util.HashMap;
020import java.util.List;
021import java.util.Map;
022import java.util.stream.Collectors;
023
024import eu.cqse.check.base.EntityCheckBase;
025import eu.cqse.check.framework.core.CheckException;
026import eu.cqse.check.framework.scanner.ETokenType;
027import eu.cqse.check.framework.scanner.IToken;
028import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
029import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
030
031/**
032 * Base class for checks that check for method calls that lack cross-browser
033 * compatibility. Since we cannot use type resolution in JavaScript, a heuristic
034 * is used to check suspicious method calls. Specifically, this check will
035 * create a finding if the JS method is called with the expected number of
036 * arguments and is no JQuery or Closure method call.
037 */
038public abstract class JavascriptMethodCompatibilityCheckBase extends EntityCheckBase {
039
040        /**
041         * Maps method names to their expected number of parameters (min and max).
042         */
043        private final Map<String, Integer[]> suspiciousMethods = new HashMap<>();
044
045        /** The value to indicate no method arguments. */
046        protected final static Integer[] NO_ARGS = new Integer[] {};
047
048        /** The identifier token for JQuery methods. */
049        private final static String JQUERY_IDENTIFIER = "$";
050
051        /** {@inheritDoc} */
052        @Override
053        public void initialize() throws CheckException {
054                super.initialize();
055                buildMethodMap();
056        }
057
058        /**
059         * Fills the method map with candidates that may create a finding. Use
060         * {@code addSuspicousMethod()} to insert a method.
061         */
062        protected abstract void buildMethodMap();
063
064        /**
065         * Adds a suspicious method to the exclusion map.
066         * 
067         * @param methodName
068         *            The name of the suspicious method.
069         * @param argsCountIntervall
070         *            The (optional) minimal and (required) maximal number of
071         *            arguments this method is expected to be called with. Use an
072         *            empty array for no expected method arguments.
073         */
074        protected void addSuspicousMethod(String methodName, Integer... argsCountIntervall) {
075                suspiciousMethods.put(methodName, argsCountIntervall);
076        }
077
078        /** {@inheritDoc} */
079        @Override
080        protected String getXPathSelectionString() {
081                return "//STATEMENT";
082        }
083
084        /** {@inheritDoc} */
085        @Override
086        protected void processEntity(ShallowEntity entity) throws CheckException {
087                List<IToken> tokens = entity.includedTokens();
088                List<Integer> indices = TokenStreamUtils.firstTokenOfTypeSequences(tokens, 0, ETokenType.DOT,
089                                ETokenType.IDENTIFIER, ETokenType.LPAREN);
090
091                for (int index : indices) {
092
093                        int methodNameIndex = ++index;
094                        IToken token = tokens.get(methodNameIndex);
095                        String methodName = token.getText();
096
097                        if (shouldReportMethod(methodName, tokens, methodNameIndex)) {
098                                createFinding(getFindingMessage(methodName), token);
099                        }
100
101                }
102
103        }
104
105        /** @return a finding message for the given method. */
106        protected abstract String getFindingMessage(String methodName);
107
108        /**
109         * Determines whether the given method should be reported as finding.
110         * 
111         * @param methodName
112         *            the name of the method.
113         * @param tokens
114         *            the tokens of the current statement.
115         * @param methodNameIndex
116         *            the index of the method name within the tokens.
117         */
118        private boolean shouldReportMethod(String methodName, List<IToken> tokens, int methodNameIndex) {
119                return suspiciousMethods.containsKey(methodName) && matchesArgumentsCount(methodName, methodNameIndex, tokens)
120                                && !isClosureMethod(tokens, methodNameIndex) && !isJQueryMethod(tokens, methodNameIndex);
121        }
122
123        /**
124         * Checks whether the method call is part of a JQuery expression.
125         * 
126         * @param tokens
127         *            the tokens of the current statement.
128         * @param methodNameIndex
129         *            the index of the method name within the tokens.
130         */
131        private static boolean isJQueryMethod(List<IToken> tokens, int methodNameIndex) {
132                int openParenthesisCount = 0;
133                for (int i = methodNameIndex; i >= 0; i--) {
134                        IToken token = tokens.get(i);
135                        if (token.getType() == ETokenType.RPAREN) {
136                                openParenthesisCount++;
137                        } else if (token.getType() == ETokenType.LPAREN) {
138                                openParenthesisCount--;
139                        }
140
141                        if (token.getText().equals(JQUERY_IDENTIFIER) && openParenthesisCount == 0) {
142                                return true;
143                        }
144                }
145
146                return false;
147        }
148
149        /**
150         * Determines if the actual <b>number</b> of method arguments is within the
151         * expected range as defined in the exlusion map.
152         * 
153         * @param methodName
154         *            the name of the method.
155         * @param methodNameIndex
156         *            the index of the method name within the tokens.
157         * @param tokens
158         *            the tokens of the current statement.
159         */
160        private boolean matchesArgumentsCount(String methodName, int methodNameIndex, List<IToken> tokens) {
161                Integer[] expectedArgsCount = suspiciousMethods.get(methodName);
162
163                int actualArgs = countMethodArgs(tokens, methodNameIndex);
164
165                if (expectedArgsCount.length == 0) {
166                        return actualArgs == 0;
167                }
168
169                int minArgs = expectedArgsCount[0];
170                int maxArgs = minArgs;
171                if (expectedArgsCount.length > 1) {
172                        maxArgs = expectedArgsCount[1];
173                }
174                return actualArgs >= minArgs && actualArgs <= maxArgs;
175        }
176
177        /**
178         * Counts the number of arguments for the given method.
179         * 
180         * @param tokens
181         *            the tokens of the current statement.
182         * @param methodNameIndex
183         *            the index of the method name within the tokens.
184         */
185        private static int countMethodArgs(List<IToken> tokens, int methodNameIndex) {
186                int commaCount = 0;
187                int innerBracketsCounter = 0;
188
189                if (tokens.get(methodNameIndex + 2).getType() == ETokenType.RPAREN) {
190                        // Method is immediately followed by ')', so it has no arguments.
191                        return 0;
192                }
193
194                for (int j = methodNameIndex + 2; j < tokens.size(); j++) {
195                        IToken token = tokens.get(j);
196
197                        if (innerBracketsCounter == 0 && token.getType() == ETokenType.COMMA) {
198                                commaCount++;
199                                continue;
200                        }
201
202                        if (token.getType() == ETokenType.LPAREN) {
203                                innerBracketsCounter++;
204                        } else if (token.getType() == ETokenType.RPAREN) {
205                                innerBracketsCounter--;
206
207                                if (innerBracketsCounter < 0) {
208                                        break;
209                                }
210                        }
211                }
212
213                return commaCount + 1;
214        }
215
216        /**
217         * Determines if the current method call if part of a Google closure call,
218         * i.e. starts with 'goog.'
219         * 
220         * @param tokens
221         *            the tokens of the current statement.
222         * @param methodNameIndex
223         *            the index of the method name within the tokens.
224         */
225        private static boolean isClosureMethod(List<IToken> tokens, int methodNameIndex) {
226                String tokenString = tokens.subList(0, methodNameIndex).stream().map(token -> token.getText())
227                                .collect(Collectors.joining());
228
229                // Match e.g. goog., goog.string. and goog.dom.classlist.
230                return tokenString.matches(".*goog\\.([a-z]*\\.){0,2}?$");
231        }
232}
