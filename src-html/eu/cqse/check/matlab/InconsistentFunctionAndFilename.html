001package eu.cqse.check.matlab;
002
003import java.nio.file.Paths;
004import java.util.List;
005
006import org.conqat.lib.commons.string.StringUtils;
007
008import eu.cqse.check.framework.core.ACheck;
009import eu.cqse.check.framework.core.CheckException;
010import eu.cqse.check.framework.core.CheckImplementationBase;
011import eu.cqse.check.framework.core.ECheckParameter;
012import eu.cqse.check.framework.scanner.ELanguage;
013import eu.cqse.check.framework.scanner.IToken;
014import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
015
016/**
017 * Checks that a name of a function is consistent with the name of the file in
018 * which the function is defined.
019 */
020@ACheck(name = InconsistentFunctionAndFilename.CHECK_NAME, description = InconsistentFunctionAndFilename.CHECK_NAME
021                + " Rename either the function or the filename to be consistent.", groupName = InconsistentFunctionAndFilename.GROUP_NAME, languages = ELanguage.MATLAB, parameters = {
022                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
023public class InconsistentFunctionAndFilename extends CheckImplementationBase {
024
025        /** The name of the check */
026        protected static final String CHECK_NAME = "The name of the first function defined in a file should be the same as the filename";
027
028        /** Name of group of checks. */
029        public static final String GROUP_NAME = "Statement conventions";
030
031        /** {@inheritDoc} */
032        @Override
033        public void execute() throws CheckException {
034                List<ShallowEntity> functions = select("//METHOD");
035                if (functions.size() <= 0) {
036                        // files that do not contain any function definition
037                        return;
038                }
039
040                ShallowEntity firstFunction = functions.get(0);
041                String functionName = firstFunction.getName();
042                IToken someToken = firstFunction.ownStartTokens().get(0);
043                String uniformPath = someToken.getOriginId();
044
045                String filename = Paths.get(uniformPath).getFileName().toString();
046                String filenameWithoutExtension = StringUtils.removeLastPart(filename, '.');
047                if (!functionName.equals(filenameWithoutExtension)) {
048                        createFindingOnFirstLine(CHECK_NAME, firstFunction);
049                }
050        }
051}
