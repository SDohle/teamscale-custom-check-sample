006package eu.cqse.check.matlab;
007
008import static eu.cqse.check.framework.scanner.ETokenType.EOL;
009import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
010import static eu.cqse.check.matlab.MatlabFunctionDeclaration.DECLARATION_RELEVANT_TOKEN_TYPES;
011
012import java.util.List;
013import java.util.stream.Collectors;
014
015import org.conqat.lib.commons.string.StringUtils;
016
017import eu.cqse.check.framework.scanner.ETokenType;
018import eu.cqse.check.framework.scanner.IToken;
019import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
020
021/**
022 * Utility class for Matlab checks.
023 */
024public class MatlabCheckUtils {
025
026        /**
027         * Given a list of tokens corresponding to a Matlab function, returns all tokens which are part of the corresponding function declaration.
028         * The tokens are filtered.
029         * {@link #filterDeclarationTokens(List)}
030         */
031        public static List<IToken> getFunctionDeclarationTokens(List<IToken> tokens, Integer functionTokenIndex) {
032                int functionDeclarationEndIndex = TokenStreamUtils.firstTokenOfType(tokens, functionTokenIndex, RPAREN, EOL);
033                List<IToken> declarationTokens = tokens.subList(functionTokenIndex, functionDeclarationEndIndex + 1);
034                return filterDeclarationTokens(declarationTokens);
035        }
036
037        /**
038         * Filters the declaration tokens to ensure only relevant tokens remain.
039         */
040        public static List<IToken> filterDeclarationTokens(List<IToken> declarationTokens) {
041                return declarationTokens.stream().filter(t -> DECLARATION_RELEVANT_TOKEN_TYPES.contains(t.getType()))
042                                .collect(Collectors.toList());
043        }
044
045        /** Returns whether a comment token contains only whitespace. */
046        public static boolean isEmptyCommentLine(IToken commentToken) {
047                return commentToken.getText().matches("%\\s*");
048        }
049
050        /** Returns whether a comment token contains at least one word character. */
051        public static boolean hasWords(IToken commentToken) {
052                return commentToken.getText().matches(".*\\w.*");
053        }
054
055        /**
056         * Returns the content of the comment. The prefixes and suffixes are stripped.
057         */
058        public static String getCommentContent(IToken commentToken) {
059                String content = commentToken.getText();
060                if (commentToken.getType() == ETokenType.TRADITIONAL_COMMENT) {
061                        content = StringUtils.stripPrefix(content, "%{");
062                        content = StringUtils.stripSuffix(content, "%}");
063                } else {
064                        content = StringUtils.stripPrefix(content, "%");
065                }
066                return content;
067        }
068
069        /** Returns the content without leading or trailing whitespace. */
070        public static String getTrimmedCommentContent(IToken commentToken) {
071                return getCommentContent(commentToken).trim();
072        }
073
074        /** Returns the content of the comment splitted by whitespace */
075        public static String[] getWordsSeperatedByWhiteSpace(IToken commentToken) {
076                return getTrimmedCommentContent(commentToken).split("\\s+");
077        }
078
079}
