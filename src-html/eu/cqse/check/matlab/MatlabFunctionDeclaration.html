006package eu.cqse.check.matlab;
007
008import static eu.cqse.check.framework.scanner.ETokenType.ARRAY_SEPARATOR;
009import static eu.cqse.check.framework.scanner.ETokenType.COMP;
010import static eu.cqse.check.framework.scanner.ETokenType.EQ;
011import static eu.cqse.check.framework.scanner.ETokenType.FUNCTION;
012import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
013import static eu.cqse.check.framework.scanner.ETokenType.LBRACK;
014import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
015import static eu.cqse.check.framework.scanner.ETokenType.RBRACK;
016import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
017
018import java.util.EnumSet;
019import java.util.List;
020import java.util.stream.Collectors;
021
022import org.conqat.lib.commons.collections.CollectionUtils;
023
024import eu.cqse.check.framework.scanner.ETokenType;
025import eu.cqse.check.framework.scanner.IToken;
026import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
027import eu.cqse.check.framework.util.tokens.TokenPattern;
028
029/**
030 * Represent a function declaration in matlab and provides useful methods.
031 */
032public class MatlabFunctionDeclaration {
033
034        /** Tokens relevant for a function delcaration */
035        /* package */ static final EnumSet<ETokenType> DECLARATION_RELEVANT_TOKEN_TYPES = EnumSet.of(FUNCTION, IDENTIFIER,
036                        EQ, LPAREN, RPAREN, LBRACK, RBRACK, ARRAY_SEPARATOR, COMP // COMP is the name of the tilde character ~
037        );
038
039        private final List<IToken> declaration;
040        private final List<IToken> output;
041        private final List<IToken> input;
042        private final IToken functionName;
043        private final int declarationLeftParenthesisIndex;
044
045        private MatlabFunctionDeclaration(List<IToken> functionDeclaration) {
046                this.declaration = functionDeclaration;
047                declarationLeftParenthesisIndex = TokenStreamUtils.firstTokenOfType(functionDeclaration, 0, ETokenType.LPAREN);
048                functionName = extractFunctionNameToken();
049                output = extractFunctionOutputTokens();
050                input = extractFunctionInputTokens();
051
052        }
053
054        /**
055         * Returns a new MatlabFunctionDeclaration out of the given tokens if they are a
056         * valid declaration otherwise null is returned
057         */
058        public static MatlabFunctionDeclaration of(List<IToken> functionDeclarationTokens) {
059                if (!isValidDeclaration(functionDeclarationTokens)) {
060                        return null;
061                }
062                return new MatlabFunctionDeclaration(functionDeclarationTokens);
063        }
064
065        /**
066         * Checks if given tokens correspond to a valid function declaration in matlab.
067         * The "parser" implemented in this function does not recognize class method
068         * declarations.
069         */
070        private static boolean isValidDeclaration(List<IToken> functionDeclarationTokens) {
071                TokenPattern argument = new TokenPattern().alternative(COMP, IDENTIFIER);
072                TokenPattern argumentList = new TokenPattern().sequence(argument).repeated(ARRAY_SEPARATOR, argument);
073                TokenPattern outputPattern = new TokenPattern().optional(LBRACK).optional(argumentList).optional(RBRACK);
074                TokenPattern inputPattern = new TokenPattern().sequence(LPAREN).optional(argumentList).sequence(RPAREN);
075                TokenPattern pattern = new TokenPattern().beginningOfStream().optional(FUNCTION).optional(outputPattern, EQ)
076                                .sequence(IDENTIFIER).optional(inputPattern).endOfStream();
077                return pattern.matchesAnywhere(functionDeclarationTokens);
078        }
079
080        /** Returns a list of tokens for the output arguments */
081        private List<IToken> extractFunctionOutputTokens() {
082                int eqIndex = TokenStreamUtils.firstTokenOfType(declaration, 0, ETokenType.EQ);
083                if (eqIndex != TokenStreamUtils.NOT_FOUND) {
084                        return getIdentifierBetween(0, eqIndex);
085                }
086                return CollectionUtils.emptyList();
087        }
088
089        /** Returns a list of tokens for the input arguments */
090        private List<IToken> extractFunctionInputTokens() {
091                if (declarationLeftParenthesisIndex != TokenStreamUtils.NOT_FOUND) {
092                        return getIdentifierBetween(declarationLeftParenthesisIndex, declaration.size());
093                }
094                return CollectionUtils.emptyList();
095        }
096
097        /** Returns all identifiers between start and end index */
098        private List<IToken> getIdentifierBetween(int startIndex, int endIndex) {
099                return TokenStreamUtils.findAll(declaration, startIndex, endIndex, EnumSet.of(IDENTIFIER)).stream()
100                                .map(declaration::get).collect(Collectors.toList());
101        }
102
103        /** Return the identifier token of the function name */
104        private IToken extractFunctionNameToken() {
105                if (declarationLeftParenthesisIndex != TokenStreamUtils.NOT_FOUND) {
106                        return declaration
107                                        .get(TokenStreamUtils.lastTokenOfType(declaration, 0, declarationLeftParenthesisIndex, IDENTIFIER));
108                }
109                return declaration.get(TokenStreamUtils.lastTokenOfType(declaration, IDENTIFIER));
110        }
111
112        public List<IToken> getFunctionOutputTokens() {
113                return output;
114        }
115
116        public List<IToken> getFunctionInputTokens() {
117                return input;
118        }
119
120        public IToken getFunctionNameToken() {
121                return functionName;
122        }
123
124        /** Returns the functionName as Text */
125        public String getFunctionNameText() {
126                return functionName.getText();
127        }
128
129        public List<IToken> getFunctionDeclarationTokens() {
130                return declaration;
131        }
132
133        @Override
134        public String toString() {
135                String representation = "";
136                representation += "==========\n";
137                representation += "function\t: " + getFunctionNameText() + "\n";
138                representation += "input\t:\n\t " + input.stream().map(IToken::getText).collect(Collectors.toList()) + "\n";
139                representation += "output\t:\n\t" + output.stream().map(IToken::getText).collect(Collectors.toList()) + "\n";
140                representation += "==========";
141                return representation;
142        }
143}
