006package eu.cqse.check.matlab;
007
008import java.util.HashSet;
009import java.util.Set;
010
011import org.conqat.lib.commons.collections.CollectionUtils;
012
013import eu.cqse.check.base.UnwantedMethodCallsCheckBase;
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.ECheckParameter;
016import eu.cqse.check.framework.core.EFindingEnablement;
017import eu.cqse.check.framework.core.option.ACheckOption;
018import eu.cqse.check.framework.scanner.ELanguage;
019
020/**
021 * Checks that unwanted method calls are not used.
022 * 
023 * In particular, the unwanted methods correspond to interactive code, function
024 * that require the user to do something (for example to press a key). Such code
025 * should not be present in production code.
026 */
027@ACheck(name = "Matlab unwanted method calls", description = UnwantedMethodCallsCheckBase.CHECK_DESCRIPTION, groupName = InconsistentFunctionAndFilename.GROUP_NAME, defaultEnablement = EFindingEnablement.YELLOW, languages = {
028                ELanguage.MATLAB }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
029public class MatlabUnwantedFunctionCallsCheck extends UnwantedMethodCallsCheckBase {
030
031        /** {@ConQAT.Doc} */
032        @ACheckOption(name = "Matlab Unwanted Methods", description = "Names of methods that should not be called, separated by comma.")
033        private Set<String> unwantedMethods = new HashSet<>(CollectionUtils.asHashSet("pause", "keyboard", "input"));
034
035        /** {@inheritDoc} */
036        @Override
037        protected Set<String> getUnwantedMethods() {
038                return unwantedMethods;
039        }
040
041        /** {@inheritDoc} */
042        @Override
043        protected String getXPathSelectionString() {
044                return "//STATEMENT";
045        }
046}
