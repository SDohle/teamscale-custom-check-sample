001package eu.cqse.check.matlab;
002
003import java.util.List;
004
005import eu.cqse.check.base.EntityCheckBase;
006import eu.cqse.check.framework.core.ACheck;
007import eu.cqse.check.framework.core.CheckException;
008import eu.cqse.check.framework.core.ECheckParameter;
009import eu.cqse.check.framework.scanner.ELanguage;
010import eu.cqse.check.framework.scanner.IToken;
011import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
012
013/**
014 * Checks for if statements with a complex condition.
015 */
016@ACheck(name = "Complex conditional expressions should be avoided", description = "Complex "
017                + "conditional expressions should be avoided. Introduce temporary logical variables instead.", groupName = InconsistentFunctionAndFilename.GROUP_NAME, languages = ELanguage.MATLAB, parameters = {
018                                ECheckParameter.ABSTRACT_SYNTAX_TREE })
019public class NoIfComplexCheck extends EntityCheckBase {
020
021        /**
022         * A numeric bound: an entity with more tokens than this value is deemed as
023         * 'complex'.
024         */
025        private static final int COMPLEX_TOKEN_LENGTH = 30;
026
027        /** Returns an XPath selection string that selects the if statements. */
028        @Override
029        protected String getXPathSelectionString() {
030                return "//STATEMENT[subtype('if')]";
031        }
032
033        /**
034         * Creates a finding for each entity that has more than COMPLEX_TOKEN_LENGTH
035         * tokens, e.g., {@value #COMPLEX_TOKEN_LENGTH} tokens.
036         **/
037        @Override
038        protected void processEntity(ShallowEntity entity) throws CheckException {
039                List<IToken> entityTokens = entity.ownStartTokens();
040                if (entityTokens.size() >= COMPLEX_TOKEN_LENGTH) {
041                        createFinding("Complex conditional expressions should be avoided.", entity);
042                }
043
044        }
045
046}
