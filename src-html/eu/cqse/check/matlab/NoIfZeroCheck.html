001package eu.cqse.check.matlab;
002
003import static eu.cqse.check.framework.scanner.ETokenType.COMMA;
004import static eu.cqse.check.framework.scanner.ETokenType.EOL;
005import static eu.cqse.check.framework.scanner.ETokenType.SEMICOLON;
006
007import java.util.EnumSet;
008import java.util.List;
009
010import eu.cqse.check.base.EntityCheckBase;
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckException;
013import eu.cqse.check.framework.core.ECheckParameter;
014import eu.cqse.check.framework.scanner.ELanguage;
015import eu.cqse.check.framework.scanner.ETokenType;
016import eu.cqse.check.framework.scanner.IToken;
017import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
018
019/**
020 * Checks for if statements with the condition being the 0 constant.
021 */
022@ACheck(name = "The conditional expression if 0 should be avoided", description = "The conditional expression if 0 should be avoided. Using the block comment feature of the editor is preferred.", groupName = InconsistentFunctionAndFilename.GROUP_NAME, languages = ELanguage.MATLAB, parameters = {
023                ECheckParameter.ABSTRACT_SYNTAX_TREE })
024public class NoIfZeroCheck extends EntityCheckBase {
025
026        /** All tokens that are valid statement separators. */
027        private static final EnumSet<ETokenType> STATEMENT_SEPARATORS = EnumSet.of(EOL, SEMICOLON, COMMA);
028
029        /** Returns an XPath selection string that selects the if statements. */
030        @Override
031        protected String getXPathSelectionString() {
032                return "//STATEMENT[subtype('if')]";
033        }
034
035        /**
036         * Creates a finding for each entity that has 0 as its condition.
037         **/
038        @Override
039        protected void processEntity(ShallowEntity entity) throws CheckException {
040                List<IToken> entityTokens = entity.ownStartTokens();
041                if (entityTokens.size() <= 3) {
042                        return;
043                }
044                // The 0th token has type IF
045                IToken token = entityTokens.get(1);
046                if (token.getType() != ETokenType.INTEGER_LITERAL || !token.getText().equals("0")) {
047                        return;
048                }
049                token = entityTokens.get(2);
050                if (STATEMENT_SEPARATORS.contains(token.getType())) {
051                        createFinding("The conditional expression if 0 should be avoided.", entity);
052                }
053
054        }
055
056}
