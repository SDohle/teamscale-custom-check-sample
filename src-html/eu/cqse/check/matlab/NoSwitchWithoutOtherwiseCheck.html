001package eu.cqse.check.matlab;
002
003import eu.cqse.check.base.EntityCheckBase;
004import eu.cqse.check.framework.core.ACheck;
005import eu.cqse.check.framework.core.CheckException;
006import eu.cqse.check.framework.core.ECheckParameter;
007import eu.cqse.check.framework.scanner.ELanguage;
008import eu.cqse.check.framework.scanner.ETokenType;
009import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
010import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
011
012/**
013 * Checks for switch statements without otherwise condition.
014 */
015@ACheck(name = "A switch statement should include the otherwise condition", description = "A switch statement should include the otherwise condition. Leaving the otherwise out is a common error, which can lead to unexpected results.", groupName = InconsistentFunctionAndFilename.GROUP_NAME, languages = ELanguage.MATLAB, parameters = {
016                ECheckParameter.ABSTRACT_SYNTAX_TREE })
017public class NoSwitchWithoutOtherwiseCheck extends EntityCheckBase {
018
019        /** Returns an XPath selection string that selects the switch statements. */
020        @Override
021        protected String getXPathSelectionString() {
022                return "//STATEMENT[subtype('switch')]";
023        }
024
025        /**
026         * Creates a finding for each entity that does not have a child of subtype
027         * otherwise.
028         **/
029        @Override
030        protected void processEntity(ShallowEntity entity) throws CheckException {
031                for (ShallowEntity child : entity.getChildren()) {
032                        if (TokenStreamUtils.firstTokenOfType(child.ownStartTokens(),
033                                        ETokenType.OTHERWISE) != TokenStreamUtils.NOT_FOUND) {
034                                return;
035                        }
036                }
037                createFinding("A switch statement should include the otherwise condition.", entity);
038        }
039
040}
