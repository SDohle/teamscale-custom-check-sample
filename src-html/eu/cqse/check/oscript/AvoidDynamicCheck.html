006package eu.cqse.check.oscript;
007
008import static eu.cqse.check.framework.core.CheckGroupNames.BAD_PRACTICE;
009import static eu.cqse.check.framework.core.ECheckParameter.ABSTRACT_SYNTAX_TREE;
010import static eu.cqse.check.framework.core.EFindingEnablement.OFF;
011import static eu.cqse.check.framework.scanner.ELanguage.OSCRIPT;
012import static eu.cqse.check.framework.scanner.ETokenType.DYNAMIC;
013import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
014import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
015import static eu.cqse.check.framework.scanner.ETokenType.OVERRIDE;
016import static eu.cqse.check.framework.shallowparser.SubTypeNames.PARAMETER;
017import static eu.cqse.check.framework.shallowparser.framework.EShallowEntityType.METHOD;
018
019import java.util.List;
020
021import org.conqat.lib.commons.collections.Pair;
022import org.conqat.lib.commons.collections.PairList;
023import org.conqat.lib.commons.collections.UnmodifiableList;
024import org.conqat.lib.commons.string.StringUtils;
025
026import eu.cqse.check.base.EntityCheckBase;
027import eu.cqse.check.framework.core.ACheck;
028import eu.cqse.check.framework.core.CheckException;
029import eu.cqse.check.framework.scanner.IToken;
030import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
031import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
032
033/**
034 * {@ConQAT.Doc}
035 */
036@ACheck(name = "Avoid using dynamics", description = "Dynamic typing should not be used, "
037                + "with the exception of overridden methods or attributes."
038                + "", groupName = BAD_PRACTICE, defaultEnablement = OFF, languages = {
039                                OSCRIPT }, parameters = { ABSTRACT_SYNTAX_TREE })
040public class AvoidDynamicCheck extends EntityCheckBase {
041
042        /** {@inheritDoc} */
043        @Override
044        protected String getXPathSelectionString() {
045                return "//*[(self::ATTRIBUTE or self::STATEMENT or self::METHOD) and anytoken('" + DYNAMIC
046                                + "') and not(anytoken('" + OVERRIDE + "'))]";
047        }
048
049        /** {@inheritDoc} */
050        @Override
051        protected void processEntity(ShallowEntity entity) throws CheckException {
052                PairList<String, String> namesAndTypes = getNamesAndTypesOfDynamics(entity);
053                for (Pair<String, String> namesAndType : namesAndTypes) {
054                        String name = namesAndType.getFirst();
055                        String type = namesAndType.getSecond();
056                        String message = StringUtils.capitalize(type) + " \"" + name + "\" should not be `Dynamic`";
057
058                        if (type.equals(PARAMETER)) {
059                                createFindingOnFirstLine(message, entity);
060                        } else {
061                                createFinding(message, entity);
062                        }
063                }
064        }
065
066        /**
067         * Returns a list of all identifier, which are preceded by a `Dynamic`. Normally
068         * this is only one, besides the case of a method where the method parameters
069         * are dynamics.
070         */
071        private static PairList<String, String> getNamesAndTypesOfDynamics(ShallowEntity entity) {
072                PairList<String, String> namesAndTypes = new PairList<String, String>();
073                UnmodifiableList<IToken> tokens = entity.ownStartTokens();
074                if (entity.getType().equals(METHOD)) {
075                        int parenIndex = TokenStreamUtils.firstTokenOfType(tokens, LPAREN);
076                        addNamesFromTokens(namesAndTypes, entity.getSubtype(), tokens, new Pair<Integer, Integer>(0, parenIndex));
077                        addNamesFromTokens(namesAndTypes, PARAMETER, tokens, new Pair<Integer, Integer>(parenIndex, tokens.size()));
078                } else {
079                        addNamesFromTokens(namesAndTypes, entity.getSubtype(), tokens,
080                                        new Pair<Integer, Integer>(0, tokens.size()));
081                }
082                return namesAndTypes;
083        }
084
085        /**
086         * Extracts and adds every name preceded by a `Dynamic` from the given tokens to
087         * the pairList with the given type.
088         */
089        private static void addNamesFromTokens(PairList<String, String> namesAndTypes, String type, List<IToken> tokens,
090                        Pair<Integer, Integer> limits) {
091                List<Integer> indices = TokenStreamUtils.allStartingIndicesOfTypeSequence(tokens, limits.getFirst(),
092                                limits.getSecond(), DYNAMIC, IDENTIFIER);
093                for (Integer index : indices) {
094                        namesAndTypes.add(tokens.get(index + 1).getText(), type);
095                }
096
097        }
098
099}
