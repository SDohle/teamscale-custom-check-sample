006package eu.cqse.check.plsql;
007
008import java.util.List;
009
010import eu.cqse.check.base.EntityCheckBase;
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckException;
013import eu.cqse.check.framework.core.CheckGroupNames;
014import eu.cqse.check.framework.core.ECheckParameter;
015import eu.cqse.check.framework.core.EFindingEnablement;
016import eu.cqse.check.framework.scanner.ELanguage;
017import eu.cqse.check.framework.scanner.ETokenType;
018import eu.cqse.check.framework.scanner.IToken;
019import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
020import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
021
022/** Checks for SELECT *. */
023@ACheck(name = "No SELECT *", description = "Never use 'SELECT * FROM xxx'&mdash;only select the columns you need.", groupName = CheckGroupNames.PERFORMANCE, languages = ELanguage.PLSQL, parameters = {
024                ECheckParameter.ABSTRACT_SYNTAX_TREE }, defaultEnablement = EFindingEnablement.YELLOW)
025public class NoSelectStarCheck extends EntityCheckBase {
026
027        /** {@inheritDoc} */
028        @Override
029        protected void processEntity(ShallowEntity entity) throws CheckException {
030                List<IToken> tokens = PlSqlCustomCheckUtils.extractSelect(entity);
031                if (TokenStreamUtils.startsWith(tokens, ETokenType.SELECT, ETokenType.MULT)) {
032                        createFinding("Never use 'SELECT * FROM xxx'.", tokens.get(1));
033                }
034        }
035
036        /** {@inheritDoc} */
037        @Override
038        protected String getXPathSelectionString() {
039                return PlSqlCustomCheckUtils.SELECT_OR_CURSOR_XPATH;
040        }
041}
