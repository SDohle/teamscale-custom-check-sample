006package eu.cqse.check.plsql;
007
008import java.util.List;
009
010import eu.cqse.check.framework.scanner.ETokenType;
011import eu.cqse.check.framework.scanner.IToken;
012import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
013import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
014
015/** Utility methods for PL/SQL custom checks. */
016public class PlSqlCustomCheckUtils {
017
018        /** XPath expression for SQL statements or CURSORs performing a SELECT. */
019        public static final String SELECT_OR_CURSOR_XPATH = "//STATEMENT[subtype('SQL') and name-matches('select')] | //ATTRIBUTE[subtype('cursor') and tokens('SELECT')]";
020
021        /** XPath expression for SQL statements or CURSORs. */
022        public static final String SQL_OR_CURSOR_XPATH = "//STATEMENT[subtype('SQL')] | //ATTRIBUTE[subtype('cursor')]";
023
024        /**
025         * Returns the entity's tokens, stripped by the cursor definition part, e.g.
026         * in case we have "CURSOR foo IS SELECT ..." just return the SELECT part.
027         * This is meant to be used in checks that select entities by
028         * {@link #SELECT_OR_CURSOR_XPATH} or {@link #SQL_OR_CURSOR_XPATH}.
029         */
030        public static List<IToken> extractSelect(ShallowEntity selectOrCursor) {
031                List<IToken> tokens = selectOrCursor.ownStartTokens();
032                if (selectOrCursor.getSubtype().equals("cursor")) {
033                        tokens = tokens.subList(TokenStreamUtils.firstTokenOfType(tokens, ETokenType.SELECT), tokens.size());
034                }
035                return tokens;
036        }
037}
