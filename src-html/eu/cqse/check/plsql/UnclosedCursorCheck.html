031/**
032 * Checks cursors that are not closed within the same procedure they are opened.
033 * This does currently not check that the cursor is closed on all possible
034 * branches through the code.
035 */
036@ACheck(name = "Unclosed cursors", description = "An opened cursor is not closed within the same procedure.", groupName = CheckGroupNames.BAD_PRACTICE, languages = ELanguage.PLSQL, parameters = {
037                ECheckParameter.ABSTRACT_SYNTAX_TREE }, defaultEnablement = EFindingEnablement.RED)
038public class UnclosedCursorCheck extends EntityCheckBase {
039
040        /** Either OPEN or CLOSE. */
041        private static final EnumSet<ETokenType> OPEN_CLOSE = EnumSet.of(OPEN, CLOSE);
042
043        /** {@inheritDoc} */
044        @Override
045        protected void processEntity(ShallowEntity method) throws CheckException {
046
047                List<IToken> tokens = method.includedTokens();
048                Map<String, IToken> openCursors = new HashMap<>();
049
050                for (int i : TokenStreamUtils.findAll(tokens, OPEN_CLOSE)) {
051                        IToken token = tokens.get(i);
052                        String name = tokens.get(i + 1).getText();
053                        switch (token.getType()) {
054                        case OPEN:
055                                openCursors.put(name, token);
056                                break;
057                        case CLOSE:
058                                openCursors.remove(name);
059                                break;
060                        default:
061                                CCSMAssert.fail("Token is neither OPEN nor CLOSE.");
062                        }
063                }
064
065                for (Entry<String, IToken> entry : openCursors.entrySet()) {
066                        createFinding("Cursor `" + entry.getKey() + "` opened but not closed in the same procedure",
067                                        entry.getValue());
068                }
069        }
070
071        /** {@inheritDoc} */
072        @Override
073        protected String getXPathSelectionString() {
074                return "//METHOD";
075        }
076}
