006package eu.cqse.check.python;

008import static eu.cqse.check.framework.scanner.ETokenType.DOT;
009import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
010import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
011import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
012import static eu.cqse.check.framework.shallowparser.TokenStreamUtils.allStartingIndicesOfTypeSequence;
013
014import java.util.HashSet;
015import java.util.List;
016import java.util.Optional;
017import java.util.Set;
018
019import org.conqat.lib.commons.collections.CollectionUtils;
020import org.conqat.lib.commons.collections.UnmodifiableList;
021import org.conqat.lib.commons.string.StringUtils;

023import eu.cqse.check.base.EntityCheckBase;
024import eu.cqse.check.framework.core.ACheck;
025import eu.cqse.check.framework.core.CheckException;
026import eu.cqse.check.framework.core.CheckGroupNames;
027import eu.cqse.check.framework.core.ECheckParameter;
028import eu.cqse.check.framework.core.EFindingEnablement;
029import eu.cqse.check.framework.scanner.ELanguage;
030import eu.cqse.check.framework.scanner.ETokenType;
031import eu.cqse.check.framework.scanner.IToken;
032import eu.cqse.check.framework.shallowparser.SubTypeNames;
033import eu.cqse.check.framework.shallowparser.TokenStreamTextUtils;
034import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
035import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
036import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
037
038/**
039 * @{AConQAT.Doc}
040 */
041@ACheck(name = "Do not access private class members", description = "Private class members and methods (name starting with underscore) must not be accessed from the outside."
042                + "", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
043                                ELanguage.PYTHON }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
044public class DoNotAccessPrivateClassMembersCheck extends EntityCheckBase {
045
046        /**
047         * Set of objects on which private member calls are allowed. These are
048         * heuristics, by convention the class instance is named self. cls is usually
049         * used for classmethods.
050         */
051        private static final Set<String> ALLOWED_OBJECTS = CollectionUtils.asHashSet("self", "cls");
052
053        /** The names of private members in Python starts with '_'. */
054        private static final String PRIVATE_MEMBER_SUFFIX = "_";
055
056        /**
057         * Token sequence of a member access. We expect a format of e.g. self._foo here.
058         */
059        private static final ETokenType[] MEMBER_ACCESS = new ETokenType[] { IDENTIFIER, DOT, IDENTIFIER };
060
061        /**
062         * The index of the identifier of the member see MEMBER_ACCESS.
063         */
064        private static final int MEMBER_IDENTIFIER_FOR_MEMBER_ACCESS = 2;
065
066        /** Chained method calls e.g. foo.bar()._private() */
067        private static final ETokenType[] CHAINED_MEMBER_ACCESS = new ETokenType[] { IDENTIFIER, LPAREN, RPAREN, DOT,
068                        IDENTIFIER };
069
070        /**
071         * The index of the identifier of the member see CHAINED_MEMBER_ACCESS.
072         */
073        private static final int MEMBER_IDENTIFIER_FOR_CHAINED_MEMBER_ACCESS = 4;
074
075        @Override
076        protected String getXPathSelectionString() {
077                return "//STATEMENT";
078        }
079
080        /** {@inheritDoc} */
081        @Override
082        protected void processEntity(ShallowEntity entity) throws CheckException {
083                UnmodifiableList<IToken> tokens = entity.includedTokens();
084                String containingMethodName = getContainingMethodName(entity);
085
086                Set<String> classNames = getClassNames(entity);
087                for (int indexOfFindingSequence : getStartingIndicesOfTypeSequence(tokens, MEMBER_ACCESS)) {
088                        checkForViolationAndCreateFinding(tokens, indexOfFindingSequence, MEMBER_IDENTIFIER_FOR_MEMBER_ACCESS,
089                                        classNames, containingMethodName);
090                }
091
092                for (int indexOfFindingSequence : getStartingIndicesOfTypeSequence(tokens, CHAINED_MEMBER_ACCESS)) {
093                        checkForViolationAndCreateFinding(tokens, indexOfFindingSequence,
094                                        MEMBER_IDENTIFIER_FOR_CHAINED_MEMBER_ACCESS, classNames, containingMethodName);
095                }
096
097        }
098
099        /** Returns the name of the method containing this call or null. */
100        private static String getContainingMethodName(ShallowEntity entity) {
101                while (entity != null) {
102                        if (entity.getType() == EShallowEntityType.METHOD && !StringUtils.isEmpty(entity.getName())) {
103                                return entity.getName();
104                        }
105                        entity = entity.getParent();
106                }
107                return null;
108        }
109
110        /**
111         * Tries to extract the class name of this statement. If the class has a super
112         * class, it tries also to extract the name of it.
113         */
114        private static Set<String> getClassNames(ShallowEntity statement) {
115                Set<String> classNames = new HashSet<>();
116                ShallowEntity parent = statement.getParent();
117                while (parent != null) {
118                        if (parent.getType().equals(EShallowEntityType.TYPE) && parent.getSubtype().equals(SubTypeNames.CLASS)) {
119                                getBaseClass(parent.ownStartTokens()).ifPresent(classNames::add);
120                                classNames.add(parent.getName());
121                        }
122                        parent = parent.getParent();
123                }
124                return classNames;
125        }
126
127        /** Tries to extract the base class of the given class definition. */
128        private static Optional<String> getBaseClass(UnmodifiableList<IToken> classDefinitonTokens) {
129                List<Integer> baseClassIndices = getStartingIndicesOfTypeSequence(classDefinitonTokens, LPAREN, IDENTIFIER,
130                                RPAREN);
131                if (baseClassIndices.size() == 1) {
132                        Integer baseClassIndex = baseClassIndices.get(0);
133                        if (baseClassIndex != TokenStreamUtils.NOT_FOUND) {
134                                return Optional.of(classDefinitonTokens.get(baseClassIndex + 1).getText());
135                        }
136                }
137                return Optional.empty();
138        }
139
140        private static List<Integer> getStartingIndicesOfTypeSequence(UnmodifiableList<IToken> tokens,
141                        ETokenType... pattern) {
142                return allStartingIndicesOfTypeSequence(tokens, 0, tokens.size() - 1, pattern);
143        }
144
145        private void checkForViolationAndCreateFinding(UnmodifiableList<IToken> tokens, int indexOfFindingSequence,
146                        int memberOffset, Set<String> classNames, String containingMethodName) throws CheckException {
147                String callPrefix = tokens.get(indexOfFindingSequence).getText();
148                String memberName = tokens.get(indexOfFindingSequence + memberOffset).getText();
149
150                if (containingMethodName != null && containingMethodName.equals(memberName)) {
151                        return;
152                }
153
154                if (!ALLOWED_OBJECTS.contains(callPrefix) && !classNames.contains(callPrefix)
155                                && memberName.startsWith(PRIVATE_MEMBER_SUFFIX)) {
156                        createFinding(
157                                        "Access of private member `"
158                                                        + TokenStreamTextUtils.concatTokenTexts(
159                                                                        tokens.subList(indexOfFindingSequence, indexOfFindingSequence + memberOffset + 1))
160                                                        + "`",
161                                        tokens.get(indexOfFindingSequence));
162                }
163        }
164
165}
