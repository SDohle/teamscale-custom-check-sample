006package eu.cqse.check.python;
007
008import static eu.cqse.check.framework.scanner.ETokenType.EQEQ;
009import static eu.cqse.check.framework.scanner.ETokenType.NONE;
010import static eu.cqse.check.framework.scanner.ETokenType.NOTEQ;
011
012import java.util.EnumSet;
013import java.util.List;
014
015import eu.cqse.check.base.EntityTokenPatternCheckBase;
016import eu.cqse.check.framework.core.ACheck;
017import eu.cqse.check.framework.core.CheckGroupNames;
018import eu.cqse.check.framework.core.ECheckParameter;
019import eu.cqse.check.framework.core.EFindingEnablement;
020import eu.cqse.check.framework.scanner.ELanguage;
021import eu.cqse.check.framework.scanner.ETokenType;
022import eu.cqse.check.framework.scanner.IToken;
023import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
024import eu.cqse.check.framework.util.tokens.TokenPattern;
025
026/**
027 * @{AConQAT.Doc}
028 */
029@ACheck(name = "Do not compare None with == or !=", description = "None should not be compared with == or !=. Instead 'is' or 'is not' should be "
030                + "used.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
031                                ELanguage.PYTHON }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
032public class DoNotCompareNoneWithOperatorCheck extends EntityTokenPatternCheckBase {
033
034        /** Set of operators that should not be used with None. */
035        private static final EnumSet<ETokenType> FORBIDDEN_OPERATORS = EnumSet.of(EQEQ, NOTEQ);
036
037        /**  */
038        /**
039         * The pattern the matches None comparisons with the forbidden operators.
040         */
041        private static final TokenPattern FINDING_PATTERN = new TokenPattern()
042                        .alternative(new TokenPattern().sequence(FORBIDDEN_OPERATORS, NONE),
043                                        new TokenPattern().sequence(NONE, FORBIDDEN_OPERATORS))
044                        .group(0);
045
046        /** {@inheritDoc} */
047        @Override
048        protected String getXPathSelectionString() {
049                return "//STATEMENT";
050        }
051
052        /** {@inheritDoc} */
053        @Override
054        protected TokenPattern getFindingPattern() {
055                return FINDING_PATTERN;
056        }
057
058        /** {@inheritDoc} */
059        @Override
060        protected String getFindingMessage(List<IToken> tokens) {
061                if (TokenStreamUtils.containsAll(tokens, EQEQ)) {
062                        return "`None` should not be compared with `==`, use `is` instead";
063                }
064                return "`None` should not be compared with `!=`, use `is not` instead";
065        }
066}
