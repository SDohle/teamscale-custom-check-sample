006package eu.cqse.check.python;
007
008import java.util.List;
009import java.util.Map.Entry;
010import java.util.regex.Pattern;
011
012import org.conqat.lib.commons.collections.ListMap;
013
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckGroupNames;
017import eu.cqse.check.framework.core.CheckImplementationBase;
018import eu.cqse.check.framework.core.ECheckParameter;
019import eu.cqse.check.framework.core.EFindingEnablement;
020import eu.cqse.check.framework.core.phase.ECodeViewOption;
021import eu.cqse.check.framework.scanner.ELanguage;
022import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
023
024/**
025 * {@AConQAT.Doc}
026 */
027@ACheck(name = "Do not import private members", description = "Private members prefixed with an underscore should not be imported.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.OFF, languages = {
028                ELanguage.PYTHON }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
029public class DoNotImportPrivateMembers extends CheckImplementationBase {
030
031        private static final Pattern PRIVATE_IMPORT_PATTERN = Pattern.compile("\\._");
032
033        /** {@inheritDoc} */
034        @Override
035        public void execute() throws CheckException {
036                ListMap<String, ShallowEntity> imports = PythonImportExtractor
037                                .extractImports(context.getRootEntity(ECodeViewOption.FILTERED));
038                for (Entry<String, List<ShallowEntity>> importEntry : imports) {
039                        createFindingsForPrivateImports(importEntry);
040                }
041        }
042
043        private void createFindingsForPrivateImports(Entry<String, List<ShallowEntity>> importEntry) throws CheckException {
044                String importStatement = importEntry.getKey();
045                if (PRIVATE_IMPORT_PATTERN.matcher(importStatement).find()) {
046                        for (ShallowEntity importEntity : importEntry.getValue()) {
047                                createFinding("Do not import private member `" + importStatement + "`", importEntity);
048                        }
049                }
050        }
051
052}
