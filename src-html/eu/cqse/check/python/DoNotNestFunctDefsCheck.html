006package eu.cqse.check.python;
007
008import eu.cqse.check.base.EntityFindingCheckBase;
009import eu.cqse.check.framework.core.ACheck;
010import eu.cqse.check.framework.core.CheckGroupNames;
011import eu.cqse.check.framework.core.ECheckParameter;
012import eu.cqse.check.framework.core.EFindingEnablement;
013import eu.cqse.check.framework.scanner.ELanguage;
014import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
015
016/**
017 * {@ConQAT.Doc}
018 */
019@ACheck(name = "Do not nest function definitions", description = "Function definitions should not be nested where top-level private"
020                + " functions can be used.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.OFF, languages = {
021                                ELanguage.PYTHON }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
022public class DoNotNestFunctDefsCheck extends EntityFindingCheckBase {
023
024        /** {@inheritDoc} */
025        @Override
026        protected String getFindingMessage(ShallowEntity entity) {
027                return String.format("Function `%s` nested in `%s`", entity.getName(), entity.getParent().getName());
028        }
029
030        /** {@inheritDoc} */
031        @Override
032        protected String getXPathSelectionString() {
033                return "//METHOD/METHOD";
034        }
035
036}
