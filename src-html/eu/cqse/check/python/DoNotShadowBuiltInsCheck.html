006package eu.cqse.check.python;
007
008import eu.cqse.check.framework.core.ACheck;
009import eu.cqse.check.framework.core.CheckException;
010import eu.cqse.check.framework.core.CheckGroupNames;
011import eu.cqse.check.framework.core.CheckImplementationBase;
012import eu.cqse.check.framework.core.ECheckParameter;
013import eu.cqse.check.framework.core.EFindingEnablement;
014import eu.cqse.check.framework.scanner.ELanguage;
015import eu.cqse.check.framework.scanner.IToken;
016import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
017import eu.cqse.check.framework.util.LanguageFeatureParser;
018
019/** {@AConQAT.Doc} */
020@ACheck(name = "Do not shadow built-ins", description = "Built-in names should not be used as identifier for variables, parameters, methods and "
021                + "types.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
022                                ELanguage.PYTHON }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
023public class DoNotShadowBuiltInsCheck extends CheckImplementationBase {
024
025        /** {@inheritDoc} */
026        @Override
027        public void execute() throws CheckException {
028                for (ShallowEntity type : select("//TYPE")) {
029                        analyzeName(type);
030                }
031
032                for (ShallowEntity method : select("//METHOD")) {
033                        analyzeMethod(method);
034                }
035
036                for (ShallowEntity entity : select("//STATEMENT | //ATTRIBUTE")) {
037                        analyzeEntity(entity);
038                }
039        }
040
041        /** Analyzes the name and parameters of the given method. */
042        private void analyzeMethod(ShallowEntity method) throws CheckException {
043                analyzeName(method);
044                analyzeEntity(method);
045        }
046
047        /** Analyzes the name of the given entity. */
048        private void analyzeName(ShallowEntity entity) throws CheckException {
049                String name = entity.getName();
050                if (name != null && LanguageFeatureParser.PYTHON.isBuiltInName(name)) {
051                        createFindingOnFirstLine(getMessage(name), entity);
052                }
053        }
054
055        /** Analyzes declared names of the given entity. */
056        private void analyzeEntity(ShallowEntity entity) throws CheckException {
057                for (IToken parameter : LanguageFeatureParser.PYTHON.getDeclaredVariableNames(entity)) {
058                        String name = parameter.getText();
059                        if (LanguageFeatureParser.PYTHON.isBuiltInName(name)) {
060                                createFinding(getMessage(name), parameter);
061                        }
062                }
063        }
064
065        /** Creates a finding message for the givnen built-in name. */
066        private static String getMessage(String name) {
067                return String.format("Built-in name '%s' must not be used!", name);
068        }
069}
