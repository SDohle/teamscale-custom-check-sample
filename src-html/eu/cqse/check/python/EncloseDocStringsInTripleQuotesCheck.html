006package eu.cqse.check.python;
007
008import static eu.cqse.check.framework.scanner.ETokenType.COLON;
009import static eu.cqse.check.framework.scanner.ETokenType.STRING_LITERAL;
010
011import java.util.List;
012import java.util.Optional;
013
014import org.conqat.lib.commons.collections.CollectionUtils;
015
016import eu.cqse.check.framework.core.ACheck;
017import eu.cqse.check.framework.core.CheckException;
018import eu.cqse.check.framework.core.CheckGroupNames;
019import eu.cqse.check.framework.core.CheckImplementationBase;
020import eu.cqse.check.framework.core.ECheckParameter;
021import eu.cqse.check.framework.core.EFindingEnablement;
022import eu.cqse.check.framework.scanner.ELanguage;
023import eu.cqse.check.framework.scanner.ETokenType;
024import eu.cqse.check.framework.scanner.ETokenType.ETokenClass;
025import eu.cqse.check.framework.scanner.IToken;
026import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
027import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
028
029/**
030 * {@AConQAT.Doc}
031 */
032@ACheck(name = "Enclose doc-strings in triple quotes", description = "Documentation strings should be enclosed in triple "
033                + "quotes.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
034                                ELanguage.PYTHON }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
035public class EncloseDocStringsInTripleQuotesCheck extends CheckImplementationBase {
036
037        /** {@inheritDoc} */
038        @Override
039        public void execute() throws CheckException {
040                List<IToken> comments = getPossibleDocStringsWithoutTripleQuote();
041
042                for (ShallowEntity entity : CollectionUtils.filter(select("//METHOD | // TYPE"), ShallowEntity::hasChildren)) {
043
044                        Optional<IToken> comment = findInvalidDocumentationComment(entity, comments);
045                        if (comment.isPresent()) {
046                                createFinding("Documentation strings have to be enclosed in triple quotes", comment.get());
047                        }
048                }
049        }
050
051        /**
052         * Finds the first invalid documentation comment for the given entity or
053         * empty if there is none.
054         */
055        private static Optional<IToken> findInvalidDocumentationComment(ShallowEntity entity, List<IToken> comments) {
056                Optional<IToken> lastColonToken = getFinalColonToken(entity);
057                if (!lastColonToken.isPresent()) {
058                        return Optional.empty();
059                }
060
061                List<ShallowEntity> childEntities = entity.getChildren();
062                if(childEntities.isEmpty()){
063                        return Optional.empty();
064                }
065                
066                ShallowEntity firstChildEntity = childEntities.get(0);
067                List<IToken> startOwnTokens = firstChildEntity.ownStartTokens();
068                if (startOwnTokens.isEmpty()) {
069                        // Occurs when dealing with lambda expressions with
070                        // single and simple statement.
071                        return Optional.empty();
072                }
073
074                IToken firstChildToken = startOwnTokens.get(0);
075                return comments.stream()
076                                .filter(comment -> isDocumentationComment(comment, lastColonToken.get(), firstChildToken)).findFirst()
077                                .filter(comment -> comment.getType() == STRING_LITERAL);
078        }
079
080        /**
081         * Returns whether the given comment token is documenting the entity that
082         * contains the given colon token in its declaration. A documentation
083         * comment must be in front of or the same token as the first child token.
084         */
085        private static boolean isDocumentationComment(IToken comment, IToken lastColonToken, IToken firstChildToken) {
086                return lastColonToken.getLineNumber() == comment.getLineNumber() - 1
087                                && comment.getEndOffset() <= firstChildToken.getEndOffset();
088        }
089
090        /**
091         * Returns the colon token that finishes the given entity.
092         */
093        private static Optional<IToken> getFinalColonToken(ShallowEntity entity) {
094                List<IToken> tokens = entity.ownStartTokens();
095                int colonIndex = TokenStreamUtils.lastTokenOfType(tokens, COLON);
096                if (colonIndex == TokenStreamUtils.NOT_FOUND) {
097                        return Optional.empty();
098                }
099                return Optional.of(tokens.get(colonIndex));
100        }
101
102        /**
103         * Returns all tokens that may be used as documentation string without
104         * triple quotes.
105         */
106        private List<IToken> getPossibleDocStringsWithoutTripleQuote() {
107                return CollectionUtils.filter(context.getUnfilteredTokens(),
108                                EncloseDocStringsInTripleQuotesCheck::mayBeDocStringWithoutTripleQuotes);
109        }
110
111        /**
112         * Returns whether the given token may be used as documentation string
113         * without triple quotes.
114         */
115        private static boolean mayBeDocStringWithoutTripleQuotes(IToken token) {
116                ETokenType type = token.getType();
117                return type == STRING_LITERAL || type.getTokenClass() == ETokenClass.COMMENT;
118        }
119}
