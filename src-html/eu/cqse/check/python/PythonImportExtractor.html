006package eu.cqse.check.python;
007
008import static eu.cqse.check.framework.scanner.ETokenType.AS;
009import static eu.cqse.check.framework.scanner.ETokenType.COMMA;
010import static eu.cqse.check.framework.scanner.ETokenType.DOT;
011import static eu.cqse.check.framework.scanner.ETokenType.FROM;
012import static eu.cqse.check.framework.scanner.ETokenType.IDENTIFIER;
013import static eu.cqse.check.framework.scanner.ETokenType.IMPORT;
014
015import java.util.List;
016
017import org.conqat.lib.commons.collections.ListMap;
018
019import eu.cqse.check.framework.core.CheckException;
020import eu.cqse.check.framework.core.CheckImplementationBase;
021import eu.cqse.check.framework.scanner.IToken;
022import eu.cqse.check.framework.shallowparser.SubTypeNames;
023import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
024import eu.cqse.check.framework.util.tokens.MatchGroupElement;
025import eu.cqse.check.framework.util.tokens.TokenPattern;
026import eu.cqse.check.framework.util.tokens.TokenPatternMatch;
027
028/**
029 * Utils to extract imported packages for python code.
030 */
031public class PythonImportExtractor {
032
033        /** xpath expression to extract import entities. */
034        private static final String IMPORT_ENTITY_XPATH = "//META[subtype('" + SubTypeNames.IMPORT + "') | subtype('"
035                        + SubTypeNames.FROM + "')]";
036
037        /**
038         * Pattern for imports like "import foo as bar, baz" (imports packages foo and
039         * baz, aliases are ignored)
040         */
041        private static final TokenPattern IMPORT_PATTERN = new TokenPattern().sequence(IMPORT)
042                        .repeatedAtLeastOnce(new TokenPattern().sequence(IDENTIFIER, new TokenPattern().repeated(DOT, IDENTIFIER))
043                                        .group(1).optional(new TokenPattern().sequence(AS, IDENTIFIER)).optional(COMMA));
044
045        /**
046         * Pattern for imports like "from foo import bar as baz" (imports foo.bar,
047         * aliases are ignored). This reuses {@link #IMPORT_PATTERN}.
048         */
049        private static final TokenPattern FROM_IMPORT_PATTERN = new TokenPattern().sequence(FROM)
050                        .sequence(IDENTIFIER, new TokenPattern().repeated(DOT, IDENTIFIER)).group(0).sequence(IMPORT_PATTERN);
051
052        /**
053         * Returns the imported modules and the according shallow entity from the given
054         * root entity.
055         */
056        public static ListMap<String, ShallowEntity> extractImports(ShallowEntity rootEntity) throws CheckException {
057                ListMap<String, ShallowEntity> imports = new ListMap<>();
058                for (ShallowEntity importEntity : CheckImplementationBase.select(rootEntity,
059                                PythonImportExtractor.IMPORT_ENTITY_XPATH)) {
060                        extractFromImportEntity(importEntity, imports);
061                }
062                return imports;
063        }
064
065        private static void extractFromImportEntity(ShallowEntity importEntity, ListMap<String, ShallowEntity> imports) {
066                List<IToken> importTokens = importEntity.ownStartTokens();
067                TokenPatternMatch importMatches = IMPORT_PATTERN.matchAtStartOf(importTokens);
068                if (importMatches != null) {
069                        for (MatchGroupElement member : importMatches.getMatchGroup(1)) {
070                                imports.add(member.concatTokenTexts(), importEntity);
071                        }
072                        return;
073                }
074                TokenPatternMatch fromImportMatches = FROM_IMPORT_PATTERN.matchAtStartOf(importTokens);
075                if (fromImportMatches != null) {
076                        String packageName = fromImportMatches.getMatchGroup(0).get(0).concatTokenTexts();
077                        for (String member : fromImportMatches.groupTexts(1)) {
078                                imports.add(packageName + "." + member, importEntity);
079                        }
080                }
081        }
082}
