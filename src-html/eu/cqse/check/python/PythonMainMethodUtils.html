006package eu.cqse.check.python;
007
008import java.util.List;
009import java.util.Map;
010
011import eu.cqse.check.framework.scanner.ETokenType;
012import eu.cqse.check.framework.scanner.IToken;
013import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
014import eu.cqse.check.framework.util.tokens.TokenPattern;
015
016/**
017 * In Python does not exist a main method, but if a module is called directly
018 * (not imported) its name is "__main__".
019 */
020public class PythonMainMethodUtils {
021
022        private static final String MODULE_NAME_PATTERN = "^__name__$";
023
024        private static final String MAIN_MODULE_NAME_PATTERN = "^\"__main__\"$";
025
026        /**
027         * Pattern for __name__ == "__main__" or "__main__" == __name__
028         */
029        private static final TokenPattern MAIN_METHOD_PATTERN = new TokenPattern().sequence(ETokenType.IF).alternative(
030                        new TokenPattern().regex(MODULE_NAME_PATTERN), ETokenType.EQEQ,
031                        new TokenPattern().regex(MAIN_MODULE_NAME_PATTERN), new TokenPattern().regex(MAIN_MODULE_NAME_PATTERN),
032                        ETokenType.EQEQ, new TokenPattern().regex(MODULE_NAME_PATTERN));
033
034        /**
035         * Traverses the parent hierarchy and returns true if a parent entity is the
036         * main "method" indicator.
037         * 
038         * @param entityInMainCache
039         *            this cache is only valid for a single file and must be reseted for
040         *            new files. It contains mappings from entities (by start token
041         *            index) to the result of this method.
042         */
043        public static boolean isInMainMethod(ShallowEntity entity, Map<Integer, Boolean> entityInMainCache) {
044                if (entity.getParent() == null) {
045                        entityInMainCache.put(entity.getStartTokenIndex(), false);
046                        return false;
047                }
048
049                ShallowEntity parent = entity.getParent();
050                int key = parent.getStartTokenIndex();
051                if (entityInMainCache.containsKey(key)) {
052                        return entityInMainCache.get(key);
053                }
054                if (isMainMethod(parent)) {
055                        entityInMainCache.put(key, true);
056                        return true;
057                }
058                return isInMainMethod(parent, entityInMainCache);
059        }
060
061        /**
062         * Checks if the entity indicates the main "method" eg if statement '__name__ ==
063         * "__main__"'
064         */
065        public static boolean isMainMethod(ShallowEntity entity) {
066                List<IToken> tokens = entity.includedTokens();
067                return MAIN_METHOD_PATTERN.matchAtStartOf(tokens) != null;
068        }
069
070}
