006package eu.cqse.check.python;

008import static eu.cqse.check.framework.scanner.ETokenType.COLON;
009import static eu.cqse.check.framework.scanner.ETokenType.ELIF;
010import static eu.cqse.check.framework.scanner.ETokenType.ELSE;
011import static eu.cqse.check.framework.scanner.ETokenType.FOR;
012import static eu.cqse.check.framework.scanner.ETokenType.IF;
013import static eu.cqse.check.framework.scanner.ETokenType.IN;
014import static eu.cqse.check.framework.scanner.ETokenType.LBRACE;
015import static eu.cqse.check.framework.scanner.ETokenType.LBRACK;
016import static eu.cqse.check.framework.scanner.ETokenType.LPAREN;
017import static eu.cqse.check.framework.scanner.ETokenType.RBRACE;
018import static eu.cqse.check.framework.scanner.ETokenType.RBRACK;
019import static eu.cqse.check.framework.scanner.ETokenType.RETURN;
020import static eu.cqse.check.framework.scanner.ETokenType.RPAREN;
021import static eu.cqse.check.framework.scanner.ETokenType.WHILE;
022
023import java.util.Arrays;
024import java.util.EnumMap;
025import java.util.EnumSet;
026import java.util.List;
027
028import org.conqat.lib.commons.collections.CollectionUtils;
029
030import eu.cqse.check.base.EntityTokenCheckBase;
031import eu.cqse.check.framework.core.ACheck;
032import eu.cqse.check.framework.core.CheckException;
033import eu.cqse.check.framework.core.CheckGroupNames;
034import eu.cqse.check.framework.core.ECheckParameter;
035import eu.cqse.check.framework.core.EFindingEnablement;
036import eu.cqse.check.framework.scanner.ELanguage;
037import eu.cqse.check.framework.scanner.ETokenType;
038import eu.cqse.check.framework.scanner.IToken;
039import eu.cqse.check.framework.shallowparser.NestingAwareTokenIterator;
040import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
041
042/**
043 * {@AConQAT.Doc}
044 */
045@ACheck(name = "Redundant parentheses check", description = "Reports redundant parentheses within if, while, for and return "
046                + "statements.", groupName = CheckGroupNames.BAD_PRACTICE, defaultEnablement = EFindingEnablement.YELLOW, languages = {
047                                ELanguage.PYTHON }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
048public class RedundantParenthesesCheck extends EntityTokenCheckBase {
049
050        /** The finding message that is reported for redundant parentheses. */
051        private static final String FINDING_MESSAGE = "Redundant parentheses";
052
053        /** Parentheses token types. */
054        private static final EnumSet<ETokenType> PARENTHESES = EnumSet.of(LPAREN, RPAREN);
055
056        /**
057         * Maps token types to a set of closing token types. The key token types
058         * indicate a part of a statement where redundant parentheses should be
059         * avoid. The expression that may be surrounded with parentheses ends at the
060         * first token type of the mapped set (if it is not nested within brackets
061         * or braces).
062         */
063        private static final EnumMap<ETokenType, EnumSet<ETokenType>> CLOSING_TYPES = new EnumMap<ETokenType, EnumSet<ETokenType>>(
064                        ETokenType.class);
065
066        static {
067                EnumSet<ETokenType> colonClosingType = EnumSet.of(COLON);
068                CLOSING_TYPES.put(IF, EnumSet.of(COLON, ELSE, RBRACK, RBRACE));
069                CLOSING_TYPES.put(ELIF, colonClosingType);
070                CLOSING_TYPES.put(WHILE, colonClosingType);
071                CLOSING_TYPES.put(FOR, EnumSet.of(IN));
072        }
073
074        /** {@inheritDoc} */
075        @Override
076        protected String getXPathSelectionString() {
077                return "//STATEMENT | //ATTRIBUTE";
078        }
079
080        /** {@inheritDoc} */
081        @Override
082        protected void processTokens(List<IToken> tokens) throws CheckException {
083                for (int index : TokenStreamUtils.findAll(tokens, EnumSet.of(IF, ELIF, WHILE, FOR))) {
084                        ETokenType type = tokens.get(index).getType();
085                        analyzeTokens(tokens, index, CLOSING_TYPES.get(type));
086                }
087
088                if (tokens.isEmpty()) {
089                        // May happen when there is a return statement containing
090                        // a lamba expression returning a simple expression.
091                        return;
092                }
093
094                // the value after return statements must never be surrounded
095                if (tokens.get(0).getType() == RETURN) {
096                        analyzeTokensForPossibleRedundantParenthesis(tokens.subList(1, tokens.size()));
097                }
098
099                checkForTopLevelRedundantParentheses(tokens);
100        }
101
102        /**
103         * Checks whether there are top-level redundant parentheses like in tuple
104         * assignments of the form (x,y) = foo().
105         */
106        private void checkForTopLevelRedundantParentheses(List<IToken> tokens) throws CheckException {
107                if (tokens.get(0).getType() == LPAREN) {
108                        int closingIndex = TokenStreamUtils.findMatchingClosingToken(tokens, 1, LPAREN, RPAREN);
109                        if (closingIndex != TokenStreamUtils.NOT_FOUND) {
110                                createFinding(FINDING_MESSAGE, tokens.get(0), tokens.get(closingIndex));
111                        }
112                }
113        }
114
115        /**
116         * Analyzes the tokens tokens from the given typeTokenOffset until the first
117         * occurrence of a non-nested given closing token for possible redundant
118         * braces.
119         */
120        private void analyzeTokens(List<IToken> tokens, int typeTokenOffset, EnumSet<ETokenType> closingTypes)
121                        throws CheckException {
122                int endIndex = TokenStreamUtils.findFirstTopLevel(tokens, typeTokenOffset + 1, closingTypes,
123                                Arrays.asList(LPAREN, LBRACK, LBRACE), Arrays.asList(RPAREN, RBRACK, RBRACE));
124                if (endIndex != -1) {
125                        // Avoids shortened versions of if and other statements e.g. in list comprehension
126                        analyzeTokensForPossibleRedundantParenthesis(tokens.subList(typeTokenOffset + 1, endIndex));
127                }
128        }
129
130        /**
131         * Analyzes the given tokens for possible redundant parenthesis. There are
132         * redundant parenthesis if no token exists, that is not nested within
133         * parenthesis.
134         */
135        private void analyzeTokensForPossibleRedundantParenthesis(List<IToken> tokens) throws CheckException {
136                if (tokens.isEmpty()) {
137                        return;
138                }
139
140                NestingAwareTokenIterator iterator = new NestingAwareTokenIterator(tokens, 0, Arrays.asList(LPAREN),
141                                Arrays.asList(RPAREN));
142                boolean foundTopLevelToken = false;
143                while (iterator.hasNext()) {
144                        ETokenType type = iterator.next().getType();
145                        if (iterator.isTopLevel() && !PARENTHESES.contains(type)) {
146                                foundTopLevelToken = true;
147                        }
148                }
149                if (!foundTopLevelToken) {
150                        createFinding(FINDING_MESSAGE, tokens.get(0), CollectionUtils.getLast(tokens));
151                }
152        }
153}
