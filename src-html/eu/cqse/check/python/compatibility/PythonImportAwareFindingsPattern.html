006package eu.cqse.check.python.compatibility;
007
008import java.util.Collections;
009import java.util.List;
010import java.util.function.Function;
011
012import org.conqat.lib.commons.collections.UnmodifiableSet;
013
014import eu.cqse.check.framework.scanner.IToken;
015import eu.cqse.check.framework.util.tokens.RegexPattern;
016import eu.cqse.check.framework.util.tokens.TokenPattern;
017import eu.cqse.check.framework.util.tokens.TokenPatternMatch;
018
019/**
020 * Allows to specify an import statement, if it is present in the file the
021 * pattern is not evaluated and no findings are created.
022 * 
023 * For example "long" is no longer available in Python 3, to use it you have to
024 * import "from past.builtins import long". If the import is present we do not
025 * want to create findings.
026 */
027public class PythonImportAwareFindingsPattern {
028
029        private final TokenPattern pattern;
030        /** If this import is present we do not create a finding. */
031        private final String importStatement;
032        private final Function<List<IToken>, String> getFindingsMessage;
033
034        public PythonImportAwareFindingsPattern(TokenPattern pattern, String importStatement, String findingsMessage) {
035                this(pattern, importStatement, token -> findingsMessage);
036        }
037
038        /** Creates a finding if the a token has this text. */
039        public PythonImportAwareFindingsPattern(String tokenText, String importStatement, String findingsMessage) {
040                this(new TokenPattern().sequence(new RegexPattern("^" + tokenText + "$")).group(0), importStatement,
041                                findingsMessage);
042        }
043
044        public PythonImportAwareFindingsPattern(TokenPattern pattern, String importStatement,
045                        Function<List<IToken>, String> getFindingsMessage) {
046                this.pattern = pattern;
047                this.importStatement = importStatement;
048                this.getFindingsMessage = getFindingsMessage;
049        }
050
051        /**
052         * Returns the matched tokens. If imports contains the import statement import
053         * an empty list is returned.
054         */
055        public List<TokenPatternMatch> getMatches(List<IToken> tokens, UnmodifiableSet<String> imports) {
056                if (imports.contains(importStatement)) {
057                        return Collections.emptyList();
058                }
059                return pattern.findAll(tokens);
060        }
061
062        /** Gets the finding message for this pattern. */
063        public String getFindingsMessage(List<IToken> tokens) {
064                return getFindingsMessage.apply(tokens);
065        }
066
067}
