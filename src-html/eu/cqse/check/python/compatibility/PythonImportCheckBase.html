006package eu.cqse.check.python.compatibility;
007
008import java.util.List;
009import java.util.Set;
010
011import org.conqat.lib.commons.collections.ListMap;
012
013import eu.cqse.check.framework.core.CheckException;
014import eu.cqse.check.framework.core.CheckImplementationBase;
015import eu.cqse.check.framework.core.phase.ECodeViewOption;
016import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
017import eu.cqse.check.python.PythonImportExtractor;
018
019/**
020 * Base class to create findings for unsupported imports.
021 */
022public abstract class PythonImportCheckBase extends CheckImplementationBase {
023
024        /** {@inheritDoc} */
025        @Override
026        public void execute() throws CheckException {
027                ListMap<String, ShallowEntity> imports = PythonImportExtractor
028                                .extractImports(context.getRootEntity(ECodeViewOption.FILTERED));
029                for (String incompatiblePackage : getIncompatibleModules()) {
030                        List<ShallowEntity> violatingEntities = imports.getCollectionOrEmpty(incompatiblePackage);
031                        for (ShallowEntity violatingEntity : violatingEntities) {
032                                createFinding(
033                                                getFindingsMessagePrefix() + " incompatible standard libraries `" + incompatiblePackage + "`",
034                                                violatingEntity);
035                        }
036                }
037        }
038
039        /**
040         * Gets the prefix for the findings message, either Python 2 or Python 3.
041         */
042        protected abstract String getFindingsMessagePrefix();
043
044        /**
045         * Gets incompatibve module names.
046         */
047        protected abstract Set<String> getIncompatibleModules();
048
049}
