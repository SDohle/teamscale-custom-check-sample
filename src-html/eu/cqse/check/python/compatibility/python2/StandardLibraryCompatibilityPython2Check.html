006package eu.cqse.check.python.compatibility.python2;
007
008import java.util.HashSet;
009import java.util.Set;
010
011import org.conqat.lib.commons.collections.CollectionUtils;
012
013import eu.cqse.check.framework.core.ACheck;
014import eu.cqse.check.framework.core.CheckGroupNames;
015import eu.cqse.check.framework.core.ECheckParameter;
016import eu.cqse.check.framework.core.EFindingEnablement;
017import eu.cqse.check.framework.scanner.ELanguage;
018import eu.cqse.check.python.compatibility.PythonCheckDescriptions;
019import eu.cqse.check.python.compatibility.PythonImportCheckBase;
020
021/**
022 * @{AConQAT.Doc}
023 */
024@ACheck(name = "Using Python 2 incompatible standard libraries", description = PythonCheckDescriptions.COPY_RIGHT_SUFFIX, groupName = CheckGroupNames.PYTHON_2_COMPATIBILITY, defaultEnablement = EFindingEnablement.OFF, languages = {
025                ELanguage.PYTHON }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
026public class StandardLibraryCompatibilityPython2Check extends PythonImportCheckBase {
027
028        private static final Set<String> INCOMPATIBLE_MODULES = new HashSet<>();
029
030        static {
031                // UserDict, UserList, UserString
032                INCOMPATIBLE_MODULES.addAll(
033                                CollectionUtils.asHashSet("collections.UserDict", "collections.UserList", "collections.UserString"));
034                // itertools: filterfalse, zip_longest
035                INCOMPATIBLE_MODULES.addAll(CollectionUtils.asHashSet("itertools.filterfalse", "itertools.zip_longest"));
036
037                INCOMPATIBLE_MODULES.addAll(CollectionUtils.asHashSet("sys.intern"));
038                // urllib
039                INCOMPATIBLE_MODULES.addAll(CollectionUtils.asHashSet("urllib.parse.urlparse", "urllib.parse.urlencode",
040                                "urllib.request.urlopen", "urllib.request.Request", "urllib.error.HTTPError"));
041        }
042
043        /** {@inheritDoc} */
044        @Override
045        protected String getFindingsMessagePrefix() {
046                return "Python 2";
047        }
048
049        /** {@inheritDoc} */
050        @Override
051        protected Set<String> getIncompatibleModules() {
052                return INCOMPATIBLE_MODULES;
053        }
054
055}
