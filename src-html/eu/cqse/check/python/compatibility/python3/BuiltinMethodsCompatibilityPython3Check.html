006package eu.cqse.check.python.compatibility.python3;
007
008import java.util.ArrayList;
009import java.util.List;
010
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckGroupNames;
013import eu.cqse.check.framework.core.ECheckParameter;
014import eu.cqse.check.framework.core.EFindingEnablement;
015import eu.cqse.check.framework.scanner.ELanguage;
016import eu.cqse.check.framework.scanner.ETokenType;
017import eu.cqse.check.framework.util.tokens.TokenPattern;
018import eu.cqse.check.python.compatibility.ImportAwarePythonCompatibiltiyCheckBase;
019import eu.cqse.check.python.compatibility.PythonCheckDescriptions;
020import eu.cqse.check.python.compatibility.PythonImportAwareFindingsPattern;
021
022/**
023 * @{AConQAT.Doc}
024 */
025@ACheck(name = "Python 3 incompatible built ins", description = BuiltinMethodsCompatibilityPython3Check.DESCRIPTION, groupName = CheckGroupNames.PYTHON_3_COMPATIBILITY, defaultEnablement = EFindingEnablement.OFF, languages = {
026                ELanguage.PYTHON }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
027public class BuiltinMethodsCompatibilityPython3Check extends ImportAwarePythonCompatibiltiyCheckBase {
028
029        /** Check description. */
030        protected static final String DESCRIPTION = PythonCheckDescriptions.COPY_RIGHT_SUFFIX;
031
032        /** {@inheritDoc} */
033        @Override
034        protected List<PythonImportAwareFindingsPattern> getFindingPatterns() {
035                List<PythonImportAwareFindingsPattern> patterns = new ArrayList<>();
036                patterns.add(new PythonImportAwareFindingsPattern(
037                                new TokenPattern().regex("^isinstance$").skipTo(ETokenType.COMMA).regex("^basestring$").group(0),
038                                "past.builtins.basestring", "`basestring` is not available in Python 3"));
039                patterns.add(createPattern("long", "past.builtins.long"));
040                patterns.add(createCallPattern("xrange", "past.builtins.xrange"));
041                patterns.add(createCallPattern("reduce", "functools.reduce"));
042                patterns.add(createCallPattern("raw_input", null));
043                patterns.add(createCallPattern("file", null));
044                patterns.add(createCallPattern("execfile", "past.builtins.execfile"));
045                patterns.add(createCallPattern("unichr", null));
046                patterns.add(createCallPattern("intern", "past.builtins.intern"));
047                patterns.add(createCallPattern("apply", "past.builtins.apply"));
048                patterns.add(createCallPattern("cmp", "past.builtins.cmp"));
049                patterns.add(createCallPattern("reload", "imp.reload"));
050                patterns.add(createCallPattern("unicode", null));
051                return patterns;
052        }
053
054        private static PythonImportAwareFindingsPattern createPattern(String builtIn, String allowedIfImported) {
055                return new PythonImportAwareFindingsPattern(builtIn, allowedIfImported,
056                                "`" + builtIn + "` is not supported in Python 3");
057        }
058
059        private static PythonImportAwareFindingsPattern createCallPattern(String methodName, String allowedIfImported) {
060                return new PythonImportAwareFindingsPattern(new TokenPattern().notPrecededBy(ETokenType.DOT)
061                                .regex("^" + methodName + "$").group(0).sequence(ETokenType.LPAREN).notFollowedBy(ETokenType.RPAREN)
062                                .skipTo(ETokenType.RPAREN).endOfStream(), allowedIfImported,
063                                "`" + methodName + "` is not available in Python 3");
064        }
065
066}
