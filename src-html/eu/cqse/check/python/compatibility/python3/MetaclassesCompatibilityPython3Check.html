006package eu.cqse.check.python.compatibility.python3;
007
008import java.util.List;
009
010import eu.cqse.check.base.EntityTokenPatternCheckBase;
011import eu.cqse.check.framework.core.ACheck;
012import eu.cqse.check.framework.core.CheckGroupNames;
013import eu.cqse.check.framework.core.ECheckParameter;
014import eu.cqse.check.framework.core.EFindingEnablement;
015import eu.cqse.check.framework.scanner.ELanguage;
016import eu.cqse.check.framework.scanner.ETokenType;
017import eu.cqse.check.framework.scanner.IToken;
018import eu.cqse.check.framework.util.tokens.TokenPattern;
019import eu.cqse.check.python.compatibility.PythonCheckDescriptions;
020import eu.cqse.check.python.compatibility.python2.MetaclassesCompatibilityPython2Check;
021
022/**
023 * @{AConQAT.Doc}
024 */
025@ACheck(name = "Metaclasses Python 3 incompatible", description = MetaclassesCompatibilityPython3Check.DESCRIPTION, groupName = CheckGroupNames.PYTHON_3_COMPATIBILITY, defaultEnablement = EFindingEnablement.OFF, languages = {
026                ELanguage.PYTHON }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
027public class MetaclassesCompatibilityPython3Check extends EntityTokenPatternCheckBase {
028
029        /** Check description. */
030        protected static final String DESCRIPTION = PythonCheckDescriptions.METACLASSES
031                        + PythonCheckDescriptions.COPY_RIGHT_SUFFIX;
032
033        /** {@inheritDoc} */
034        @Override
035        protected TokenPattern getFindingPattern() {
036                return new TokenPattern().regex("^__metaclass__$").group(0).sequence(ETokenType.EQ);
037        }
038
039        /** {@inheritDoc} */
040        @Override
041        protected String getFindingMessage(List<IToken> tokens) {
042                return MetaclassesCompatibilityPython2Check.FINDING_MESSAGE;
043        }
044
045        /** {@inheritDoc} */
046        @Override
047        protected String getXPathSelectionString() {
048                return "//ATTRIBUTE";
049        }
050
051}
