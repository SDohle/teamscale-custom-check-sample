008package eu.cqse.check.sqlscript;
009
010import java.util.List;
011
012import eu.cqse.check.base.EntityFindingCheckBase;
013import eu.cqse.check.framework.core.ACheck;
014import eu.cqse.check.framework.core.ECheckParameter;
015import eu.cqse.check.framework.scanner.ELanguage;
016import eu.cqse.check.framework.scanner.IToken;
017import eu.cqse.check.framework.shallowparser.SubTypeNames;
018import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
019
020/** {@ConQAT.Doc} */
021@ACheck(name = "Avoid cursors", description = "Cursors imply row-at-a-time processing. As a consequence,"
022                + "opportunities for optimizations by the SQL engine are missed. So you should consider replacing the use of"
023                + "cursors with loops, by SQL statements.\n\n"
024                + "See also: SAP HANA SQLScript Reference. Document Version: 1.0 - 2016-05-11. SAP SE, Walldorf, Germany, 2016. Chapter 13.8.", groupName = ISqlScriptCheckGroups.UNWANTED_SQL_SCRIPT_STATEMENTS, languages = {
025                                ELanguage.SQLSCRIPT }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
026public class AvoidCursorCheck extends EntityFindingCheckBase {
027
028        /** {@inheritDoc} */
029        @Override
030        protected String getFindingMessage(ShallowEntity entity) {
031                String message = "Usage of cursor";
032                List<IToken> tokens = entity.ownStartTokens();
033                if (tokens.size() > 2) {
034                        String identifier = tokens.get(2).getText();
035                        message += " '" + identifier + "'";
036                }
037                return message;
038        }
039
040        /** {@inheritDoc} */
041        @Override
042        protected String getXPathSelectionString() {
043                return "//ATTRIBUTE[subtype('" + SubTypeNames.CURSOR + "')]";
044        }
045
046}
