008package eu.cqse.check.sqlscript;
009
010import org.conqat.lib.commons.collections.UnmodifiableList;
011
012import eu.cqse.check.base.EntityCheckBase;
013import eu.cqse.check.framework.core.ACheck;
014import eu.cqse.check.framework.core.CheckException;
015import eu.cqse.check.framework.core.ECheckParameter;
016import eu.cqse.check.framework.scanner.ELanguage;
017import eu.cqse.check.framework.scanner.ETokenType;
018import eu.cqse.check.framework.scanner.IToken;
019import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
020import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
021
022/**
023 * {@ConQAT.Doc}
024 */
025@ACheck(name = "Missing LANGUAGE definition.", description = ""
026                + "The LANGUAGE keyword for defining the programming language is missing, "
027                + "thus it defaults to SQLSCRIPT. However, it is good practice to always "
028                + "specify the programming language. Add LANGUAGE SQLSCRIPT to the procedure's "
029                + "introduction statement.", groupName = ISqlScriptCheckGroups.BAD_PRACTICE, languages = {
030                                ELanguage.SQLSCRIPT }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
031public class MissingLanguageDefinitionCheck extends EntityCheckBase {
032
033        /** {@inheritDoc} */
034        @Override
035        protected String getXPathSelectionString() {
036                return "//METHOD";
037        }
038
039        /** {@inheritDoc} */
040        @Override
041        protected void processEntity(ShallowEntity entity) throws CheckException {
042                UnmodifiableList<IToken> tokens = entity.ownStartTokens();
043                if (!TokenStreamUtils.containsAny(tokens, ETokenType.LANGUAGE)) {
044                        createFinding("Missing LANGUAGE specification", tokens.get(0), tokens.get(tokens.size() - 1));
045                }
046        }
047}
