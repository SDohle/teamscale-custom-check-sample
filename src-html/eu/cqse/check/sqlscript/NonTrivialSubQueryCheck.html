008package eu.cqse.check.sqlscript;
009
010import java.util.EnumSet;
011
012import eu.cqse.check.framework.core.ACheck;
013import eu.cqse.check.framework.core.CheckException;
014import eu.cqse.check.framework.core.ECheckParameter;
015import eu.cqse.check.framework.scanner.ELanguage;
016import eu.cqse.check.framework.scanner.ETokenType;
017import eu.cqse.check.sqlscript.sqlstatement.SqlQuery;
018import eu.cqse.check.sqlscript.sqlstatement.SqlStatement;
019
020/** {@ConQAT.Doc} */
021@ACheck(name = "Non-trivial sub-queries", description = ""
022                + "Non-trivial sub-queries increase the complexity of SQL statemetns and should be better extracted as spearate queries where "
023                + "the result is stored in a table variable. This will result in easer to comprehend code since the names of the table variables "
024                + "convey the meaning of the query. Furhtermore, extracting sub-queries allows to easer reuse common expressions.\n\n"
025                + "Non-trivial sub-queries are queries which again contain a sub-query or a keyword (e.g. `WHERE`, `JOIN`, ...) that "
026                + "indicates a complex query.\n\n"
027                + "See also: SAP HANA SQLScript Reference. Document Version: 1.0 - 2016-05-11. SAP SE, Walldorf, Germany, 2016. Chapter 13.1 and 13.2.", groupName = ISqlScriptCheckGroups.COMPLEX_STATEMENT, languages = {
028                                ELanguage.SQLSCRIPT }, parameters = { ECheckParameter.ABSTRACT_SYNTAX_TREE })
029public class NonTrivialSubQueryCheck extends SqlStatementCheckBase {
030
031        /** Keywords that identify non-trivial queries */
032        private static final EnumSet<ETokenType> NON_TRIVIAL_TOKENS = EnumSet.of(ETokenType.WHERE, ETokenType.JOIN,
033                        ETokenType.GROUP, ETokenType.HAVING, ETokenType.UNION);
034
035        /**
036         * {@inheritDoc}
037         */
038        @Override
039        protected void checkSqlStatement(SqlStatement statement) throws CheckException {
040                for (SqlQuery subquery : getQueryToCheck(statement).getSubqueries()) {
041                        if (subquery.hasSubquery() || subquery.containsAnyToken(NON_TRIVIAL_TOKENS)) {
042                                createFinding("Non-trivial subquery", subquery);
043                        }
044                }
045        }
046
047        /**
048         * Gets the query which should be checked, for INSERT, REPLACE and UPSERT
049         * statements the first query (the INSERT / REPLACE / UPSERT itself) is ignored.
050         */
051        private static SqlQuery getQueryToCheck(SqlStatement statement) {
052                if (statement.isSubqueryTolerant() && statement.getQuery().getSubqueries().size() == 1) {
053                        return statement.getQuery().getSubqueries().get(0);
054                }
055                return statement.getQuery();
056        }
057
058}
