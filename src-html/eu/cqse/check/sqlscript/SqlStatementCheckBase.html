008package eu.cqse.check.sqlscript;
009
010import java.util.Collections;
011import java.util.HashSet;
012import java.util.List;
013
014import org.conqat.lib.commons.collections.CollectionUtils;
015
016import eu.cqse.check.base.EntityCheckBase;
017import eu.cqse.check.framework.core.CheckException;
018import eu.cqse.check.framework.shallowparser.SubTypeNames;
019import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
020import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
021import eu.cqse.check.sqlscript.sqlstatement.SqlQuery;
022import eu.cqse.check.sqlscript.sqlstatement.SqlStatement;
023
024/**
025 * Base class for checks of SQL statements (SELECT, INSERT, DELETE, ...), more
026 * precisely for {@link ShallowEntity}s of type
027 * {@link EShallowEntityType#STATEMENT} with subtype 'SQL'.
028 */
029public abstract class SqlStatementCheckBase extends EntityCheckBase {
030
031        /**
032         * {@link HashSet} holding statements which should be excluded, build from
033         * {@link #excludeStatements()} in {@link #initialize()}
034         */
035        private final HashSet<String> excludeStatements = new HashSet<String>();
036
037        /** {@inheritDoc} */
038        @Override
039        protected String getXPathSelectionString() {
040                return "//STATEMENT[subtype('" + SubTypeNames.SQL + "')]";
041        }
042
043        /** {@inheritDoc} */
044        @Override
045        public void initialize() {
046                for (String exclude : excludeStatements()) {
047                        excludeStatements.add(exclude.toLowerCase());
048                }
049        }
050
051        /** {@inheritDoc} */
052        @Override
053        protected List<ShallowEntity> filterEntities(List<ShallowEntity> entities) {
054                return CollectionUtils.filter(entities, entity -> !isExcluded(entity));
055        }
056
057        /**
058         * @return <code>true</code>, if the given entity is excluded according to
059         *         {@link #excludeStatements()}
060         */
061        private boolean isExcluded(ShallowEntity entity) {
062                return excludeStatements.contains(entity.getName());
063        }
064
065        /**
066         * @return List of statement names which should be excluded from this check
067         */
068        protected List<String> excludeStatements() {
069                return Collections.emptyList();
070        }
071
072        /** {@inheritDoc} */
073        @Override
074        protected void processEntity(ShallowEntity entity) throws CheckException {
075                checkSqlStatement(new SqlStatement(entity));
076        }
077
078        /**
079         * Executes the check on the given SQL statement.
080         */
081        protected abstract void checkSqlStatement(SqlStatement statement) throws CheckException;
082
083        /**
084         * Creates a finding at the given {@link SqlQuery}.
085         */
086        protected void createFinding(String message, SqlQuery query) throws CheckException {
087                createFinding(message, query.getBeginToken(), query.getEndToken());
088        }
089
090}
