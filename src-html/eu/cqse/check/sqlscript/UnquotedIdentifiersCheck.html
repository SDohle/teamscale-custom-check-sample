008package eu.cqse.check.sqlscript;
009
010import java.util.Set;
011
012import org.conqat.lib.commons.collections.SetMap;
013
014import eu.cqse.check.framework.core.ACheck;
015import eu.cqse.check.framework.core.CheckException;
016import eu.cqse.check.framework.core.CheckImplementationBase;
017import eu.cqse.check.framework.core.phase.ECodeViewOption;
018import eu.cqse.check.framework.scanner.ELanguage;
019import eu.cqse.check.framework.scanner.ETokenType.ETokenClass;
020import eu.cqse.check.framework.scanner.IToken;
021
022/**
023 * {@ConQAT.Doc}
024 */
025@ACheck(name = "Unquoted identfieres", description = ""
026                + "Internally unquoted identifiers are treated as uppercase. Thus it should be avoided to mix cases. This check "
027                + "will also produce findings for identifiers if the name itself is mixed cased (camel case). Camel-casing should be not used to seperate words in identifiers since "
028                + "this formatting might be lost when auto-formatting of the source code is applied (e.g. uppercase or lowercase all unquoted identifieres).\n\n"
029                + "See also: SAP HANA SQLScript Reference. Document Version: 1.0 - 2016-05-11. SAP SE, Walldorf, Germany, 2016. Chapter 5.1.", groupName = ISqlScriptCheckGroups.BAD_PRACTICE, languages = {
030                                ELanguage.SQLSCRIPT })
031public class UnquotedIdentifiersCheck extends CheckImplementationBase {
032
033        /**
034         * Maps the (normalized) identifier name to tokens, used to identify the tokens
035         * from the identifiers in {@link #normalizedToActualNamesMap} to create
036         * findings.
037         */
038        private final SetMap<String, IToken> identifierToTokenMap = new SetMap<>();
039
040        /**
041         * Maps the normalized identifier name (converted to upper case) to the actual
042         * identifier names
043         */
044        private final SetMap<String, String> normalizedToActualNamesMap = new SetMap<>();
045
046        /** {@inheritDoc} */
047        @Override
048        public void execute() throws CheckException {
049
050                identifierToTokenMap.clear();
051                normalizedToActualNamesMap.clear();
052
053                for (IToken token : context.getTokens(ECodeViewOption.FILTERED)) {
054                        if (token.getType().getTokenClass() == ETokenClass.IDENTIFIER) {
055                                checkIdentifierAndFillMaps(token);
056                        }
057
058                }
059                checkForMixedUsedUnquotedIdentifiers();
060        }
061
062        /**
063         * Checks the token which is supposed to be an identifier and creates a finding
064         * if it is malformatted, it also fills the maps {@link #identifierToTokenMap}
065         * and {@link #normalizedToActualNamesMap} which are used in
066         * {@link #checkForMixedUsedUnquotedIdentifiers()}.
067         */
068        private void checkIdentifierAndFillMaps(IToken token) throws CheckException {
069                String identifier = token.getText();
070                if (isDoubleQuoted(identifier)) {
071                        return;
072                }
073                if (isSingleQuoted(identifier)) {
074                        createFinding("Usage of single quotes at " + identifier, token);
075                        return;
076                }
077
078                addIdentifierToMaps(token);
079
080                if (isMixedCased(identifier)) {
081                        createFinding("Un-quoted identifier '" + identifier + "' with mixed upper and lower case letters", token);
082                }
083
084        }
085
086        /**
087         * Checks if there are identifiers which are not quoted but used in different
088         * casings.
089         */
090        private void checkForMixedUsedUnquotedIdentifiers() throws CheckException {
091                for (String normalizedName : identifierToTokenMap.getKeys()) {
092                        Set<String> actualNames = normalizedToActualNamesMap.getCollection(normalizedName);
093                        if (actualNames == null || actualNames.size() <= 1) {
094                                continue;
095                        }
096                        String findingsMessge = "Unquoted identifier is used in differnt casings " + actualNames;
097                        for (IToken token : identifierToTokenMap.getCollection(normalizedName)) {
098                                createFinding(findingsMessge, token);
099                        }
100                }
101        }
102
103        /**
104         * Adds the identifier of the given token to {@link #identifierToTokenMap} and
105         * {@link #normalizedToActualNamesMap}.
106         */
107        private void addIdentifierToMaps(IToken token) {
108                String normalizedIdentifier = token.getText().toUpperCase();
109                identifierToTokenMap.add(normalizedIdentifier, token);
110                normalizedToActualNamesMap.add(normalizedIdentifier, token.getText());
111        }
112
113        /**
114         * @return <code>true</code> if the identifier is quoted with double quotes.
115         */
116        private static boolean isDoubleQuoted(String identifier) {
117                return identifier.startsWith("\"") && identifier.endsWith("\"");
118        }
119
120        /**
121         * @return <code>true</code> if the identifier is quoted with single quotes.
122         */
123        private static boolean isSingleQuoted(String identifier) {
124                return identifier.startsWith("'") && identifier.endsWith("'");
125        }
126
127        /**
128         * @return <code>true</code> if identifier contains lower and upper case letters
129         */
130        private static boolean isMixedCased(String identifier) {
131                boolean containsLowerCaseLetters = !identifier.equals(identifier.toUpperCase());
132                boolean containsUperCaseLetters = !identifier.equals(identifier.toLowerCase());
133                return containsLowerCaseLetters && containsUperCaseLetters;
134        }
135}
