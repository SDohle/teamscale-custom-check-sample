008package eu.cqse.check.sqlscript.sqlstatement;
009
010import java.util.ArrayList;
011import java.util.EnumSet;
012import java.util.List;
013
014import eu.cqse.check.framework.scanner.ETokenType;
015import eu.cqse.check.framework.scanner.IToken;
016import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
017
018/** Analyzes an SQL query from an {@link SqlStatement}. */
019public class SqlQuery {
020
021        /** {@link SqlStatement} which this query is part of. */
022        private final SqlStatement statement;
023
024        /** Index of first token of {@link #statement} of this query */
025        private final int beginTokenIndex;
026
027        /** Index of last token of {@link #statement} of this query */
028        private final int endTokenIndex;
029
030        /** List of sub-queries */
031        private final List<SqlQuery> subqueries = new ArrayList<>();
032
033        /** The nesting level of this query inside {@link #statement}. */
034        private final int nestingLevel;
035
036        /** The maximum nesting depth of sub-queries starting from this query */
037        private int ownMaximumNestingDepth = 0;
038
039        /** Constructor. */
040        /* package */ SqlQuery(SqlStatement statement) {
041                this(statement, 0, statement.getTokens().size() - 1, 0);
042        }
043
044        /** Constructor. */
045        private SqlQuery(SqlQuery parent, int beginTokenIndex, int endTokenIndex) {
046                this(parent.statement, beginTokenIndex, endTokenIndex, parent.nestingLevel + 1);
047        }
048
049        /** Constructor. */
050        private SqlQuery(SqlStatement statement, int beginTokenIndex, int endTokenIndex, int nestingLevel) {
051                this.statement = statement;
052                this.beginTokenIndex = beginTokenIndex;
053                this.endTokenIndex = endTokenIndex;
054                this.nestingLevel = nestingLevel;
055                initializeSubqueries();
056        }
057
058        /**
059         * Initializes the {@link #subqueries} by searching for nested sub-queries.
060         */
061        private void initializeSubqueries() {
062                int searchIndex = beginTokenIndex + 1;
063                while (searchIndex <= endTokenIndex) {
064                        int subqueryBegin = findSubqueryBegin(searchIndex);
065                        if (subqueryBegin == TokenStreamUtils.NOT_FOUND) {
066                                return;
067                        }
068                        int subqueryEnd = findSubqueryEnd(subqueryBegin);
069                        searchIndex = subqueryEnd + 1;
070                        while (statement.getToken(subqueryBegin).getType() == ETokenType.LPAREN) {
071                                subqueryBegin++;
072                        }
073                        if (statement.getToken(subqueryBegin).getType() != ETokenType.SELECT) {
074                                continue;
075                        }
076                        while (statement.getToken(subqueryEnd).getType() == ETokenType.RPAREN) {
077                                subqueryEnd--;
078                        }
079                        SqlQuery subquery = new SqlQuery(this, subqueryBegin, subqueryEnd);
080                        if (subquery.ownMaximumNestingDepth >= ownMaximumNestingDepth) {
081                                ownMaximumNestingDepth = subquery.ownMaximumNestingDepth + 1;
082                        }
083                        subqueries.add(subquery);
084                }
085        }
086
087        /**
088         * Searches for the begin of a sub-query not before the given index. A begin
089         * is identified either by a left parenthesis or by a SELECT key word.
090         * 
091         * @return token index of the first token of the sub-query
092         */
093        private int findSubqueryBegin(int startTokenIndex) {
094                return TokenStreamUtils.firstTokenOfType(statement.getTokens(), startTokenIndex, endTokenIndex + 1,
095                                ETokenType.SELECT, ETokenType.LPAREN);
096        }
097
098        /**
099         * @return fist token of this query
100         */
101        public IToken getBeginToken() {
102                return statement.getToken(beginTokenIndex);
103        }
104
105        /**
106         * @return last token of this query
107         */
108        public IToken getEndToken() {
109                return statement.getToken(endTokenIndex);
110        }
111
112        /**
113         * Searches for the end of the sub-query within this query which starts at
114         * the given index. The end is identified as follows:
115         * <ul>
116         * <li>if the start token is a left parentheses, by the first right
117         * parentheses at the same nesting level.
118         * <li>otherwise the last token.
119         * </ul>
120         */
121        private int findSubqueryEnd(int subQueryBeginTokenIndex) {
122                if (statement.getTokens().get(subQueryBeginTokenIndex).getType() == ETokenType.LPAREN) {
123                        return TokenStreamUtils.findMatchingClosingToken(statement.getTokens(), subQueryBeginTokenIndex + 1,
124                                        ETokenType.LPAREN, ETokenType.RPAREN);
125                }
126                return endTokenIndex;
127        }
128
129        /**
130         * @return <code>true</code>, if query has a sub-query
131         */
132        public boolean hasSubquery() {
133                return !subqueries.isEmpty();
134        }
135
136        /**
137         * @return maximum nesting depth of this query relative to the containing
138         *         {@link SqlStatement}.
139         */
140        public int getMaximumNestingDepth() {
141                return ownMaximumNestingDepth + nestingLevel;
142        }
143
144        /**
145         * Collects regions nested deeper than the given threshold recursively,
146         * helper method for {@link SqlStatement#findSubqueriesDeeperThan(int)}
147         */
148        /* package */ void collectSubqueriesDeeperThan(int threshold, List<SqlQuery> result) {
149                if (nestingLevel > threshold) {
150                        result.add(this);
151                        return;
152                }
153                for (SqlQuery subquery : subqueries) {
154                        subquery.collectSubqueriesDeeperThan(threshold, result);
155                }
156        }
157
158        /** @see #subqueries */
159        public List<SqlQuery> getSubqueries() {
160                return subqueries;
161        }
162
163        /**
164         * @return <code>true</code> if the query contains any token of the given
165         *         {@link ETokenType}s, <code>false</code> otherwise.
166         */
167        public boolean containsAnyToken(EnumSet<ETokenType> tokenTypes) {
168                return TokenStreamUtils.containsAny(statement.getTokens(), beginTokenIndex, endTokenIndex + 1, tokenTypes);
169        }
170}
