008package eu.cqse.check.sqlscript.sqlstatement;
009
010import java.util.ArrayList;
011import java.util.EnumSet;
012import java.util.List;
013import java.util.Set;
014
015import org.conqat.lib.commons.collections.CollectionUtils;
016
017import eu.cqse.check.framework.core.CheckException;
018import eu.cqse.check.framework.scanner.ETokenType;
019import eu.cqse.check.framework.scanner.IToken;
020import eu.cqse.check.framework.shallowparser.SubTypeNames;
021import eu.cqse.check.framework.shallowparser.TokenStreamUtils;
022import eu.cqse.check.framework.shallowparser.framework.EShallowEntityType;
023import eu.cqse.check.framework.shallowparser.framework.ShallowEntity;
024
025/** Analyzes a SQL statement, especially if it contains sub-queries. */
026public class SqlStatement {
027
028        /**
029         * Constant for a set of SQL statements which typically hold a (non-trivial)
030         * sub-query and thus should be handled more tolerant in checks on
031         * sub-queries. These are INSERT, UPSERT and REPLACE.
032         */
033        public static final Set<String> SUBQUERY_TOLERANT_STATEMENTS = CollectionUtils.asHashSet("insert", "upsert",
034                        "replace");
035
036        /** Shallow entity of this statement */
037        private final ShallowEntity entity;
038
039        /** Query of this SQL statement, used for analyzing sub-queries. */
040        private final SqlQuery query;
041
042        /** Constructor. */
043        public SqlStatement(ShallowEntity entity) throws CheckException {
044                if (!(entity.getType() == EShallowEntityType.STATEMENT) && entity.getSubtype().equals(SubTypeNames.SQL)) {
045                        throw new CheckException(entity + " is not a SQL statement, thus can not be analyzed as SQL statement.");
046                }
047                this.entity = entity;
048                this.query = new SqlQuery(this);
049        }
050
051        /** @return tokens of this statement */
052        public List<IToken> getTokens() {
053                return entity.ownStartTokens();
054        }
055
056        /** @return name of this statement */
057        public String getStatementName() {
058                return entity.getName();
059        }
060
061        /**
062         * @return the token at the given index within this {@link SqlStatement}.
063         */
064        public IToken getToken(int index) {
065                return getTokens().get(index);
066        }
067
068        /** @see #query */
069        public SqlQuery getQuery() {
070                return query;
071        }
072
073        /**
074         * Finds all SELECT tokens which introduce a sub-query but are not preceded
075         * by a parentheses.
076         * 
077         * @return list of embedded SELECT tokens without preceding parentheses
078         */
079        public List<IToken> findSubquerySelectTokensWithoutParantheses() {
080                List<IToken> subqueriesWihtoutParantheses = new ArrayList<>();
081                for (int tokenIndex : findSelectTokenIndices()) {
082                        if (tokenIndex > 0 && getToken(tokenIndex - 1).getType() != ETokenType.LPAREN) {
083                                subqueriesWihtoutParantheses.add(getToken(tokenIndex));
084                        }
085                }
086                return subqueriesWihtoutParantheses;
087        }
088
089        /**
090         * Finds all occurrences of SELECT tokens within the given tokens for
091         * indices > 0 (ignores SELECT token at begin).
092         */
093        private List<Integer> findSelectTokenIndices() {
094                List<Integer> selectTokenIndicies = TokenStreamUtils.findAll(getTokens(), EnumSet.of(ETokenType.SELECT));
095                selectTokenIndicies.remove(Integer.valueOf(0));
096                return selectTokenIndicies;
097        }
098
099        /**
100         * Finds sub-queries which are nested deeper than the given threshold.
101         */
102        public List<SqlQuery> findSubqueriesDeeperThan(int threshold) {
103                List<SqlQuery> result = new ArrayList<>();
104                if (query.getMaximumNestingDepth() <= threshold) {
105                        return result;
106                }
107                query.collectSubqueriesDeeperThan(threshold, result);
108                return result;
109        }
110
111        /**
112         * @return <code>true</code>, if this {@link SqlStatement} is any statement
113         *         in {@link #SUBQUERY_TOLERANT_STATEMENTS} which should be handled
114         *         more tolerantly regarding sub-queries
115         */
116        public boolean isSubqueryTolerant() {
117                return SUBQUERY_TOLERANT_STATEMENTS.contains(getStatementName());
118        }
119}
