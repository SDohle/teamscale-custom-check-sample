001package eu.cqse.check.util.clang;
002
003import org.conqat.engine.core.core.ConQATException;
004import org.conqat.lib.commons.assertion.CCSMAssert;
005
006import eu.cqse.check.framework.core.CheckException;
007import eu.cqse.check.framework.core.phase.ECodeViewOption;
008import eu.cqse.check.framework.core.phase.ITokenElementContext;
009import eu.cqse.clang.CXCursor;
010import eu.cqse.clang.CXUnsavedFile;
011import eu.cqse.clang.Clang;
012import eu.cqse.clang.ClangJniLoader;
013import eu.cqse.clang.SWIGTYPE_p_CXTranslationUnitImpl;
014import eu.cqse.clang.SWIGTYPE_p_void;
015
016/**
017 * Wrapper for the CXIndex type in Clang. This wrapper implements disposing the
018 * index (freeing the memory) when this Object is garbage collected.
019 * <p>
020 * An "index" that consists of a set of translation units that would typically
021 * be linked together into an executable or library.
022 */
023public class ClangTranslationUnitWrapper {
024        private final SWIGTYPE_p_void clangIndex;
025        private final SWIGTYPE_p_CXTranslationUnitImpl mainTranslationUnit;
026
027        private ClangTranslationUnitWrapper(SWIGTYPE_p_void clangIndex,
028                        SWIGTYPE_p_CXTranslationUnitImpl mainTranslationUnit) {
029                this.clangIndex = clangIndex;
030                this.mainTranslationUnit = mainTranslationUnit;
031        }
032
033        /**
034         * Returns the current file parsed by clang.
035         * 
036         * The clang index might contain more than one file (header files) which maybe
037         * lead to more than one translation unit known to clang. This is the
038         * translation unit corresponding to the file referenced in the current
039         * {@link eu.cqse.check.framework.core.ICheckContext}.
040         */
041        public SWIGTYPE_p_CXTranslationUnitImpl getMainTranslationUnit() {
042                return mainTranslationUnit;
043        }
044
045        /**
046         * Creates a {@link ClangTranslationUnitWrapper} based on the given
047         * {@link ITokenElementContext}. This means the text content of the context is
048         * parsed by clang.
049         */
050        public static ClangTranslationUnitWrapper createForContext(ITokenElementContext context) throws ConQATException {
051                ClangJniLoader.ensureLoaded();
052                String text = null;
053                try {
054                        text = context.getTextContent(ECodeViewOption.ETextViewOption.UNFILTERED_CONTENT);
055                } catch (CheckException e) {
056                        throw new ConQATException("Could not retrieve text content for clang initialization.", e);
057                }
058                // the second parameter controls whether diagnostics are printed to stderr or
059                // not.
060                SWIGTYPE_p_void clangIndex = Clang.clang_createIndex(0, 0);
061                if (clangIndex == null) {
062                        CCSMAssert.fail("Failed to create a clang index");
063                }
064                CXUnsavedFile file = new CXUnsavedFile();
065                file.setFilename(context.getUniformPath());
066                file.setContents(text);
067                file.setLength(text.length());
068
069                SWIGTYPE_p_CXTranslationUnitImpl translationUnit = Clang.clang_parseTranslationUnit(clangIndex,
070                                context.getUniformPath(), null, 0, new CXUnsavedFile[] { file }, 1, 0);
071                if (translationUnit == null) {
072                        if (clangIndex != null) {
073                                Clang.clang_disposeIndex(clangIndex);
074                        }
075                        CCSMAssert.fail("Translation unit should not be null");
076                }
077                return new ClangTranslationUnitWrapper(clangIndex, translationUnit);
078        }
079
080        @Override
081        protected void finalize() throws Throwable {
082                if (mainTranslationUnit != null) {
083                        Clang.clang_disposeTranslationUnit(mainTranslationUnit);
084                }
085                if (clangIndex != null) {
086                        Clang.clang_disposeIndex(clangIndex);
087                }
088        }
089
090        /**
091         * Returns a new CXCursor pointing at the root node of the current translation
092         * unit ({@link #getMainTranslationUnit()}).
093         */
094        public CXCursor getMainRootCursor() {
095                return Clang.clang_getTranslationUnitCursor(getMainTranslationUnit());
096        }
097}
