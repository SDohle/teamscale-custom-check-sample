039        /*
040         * Flags for the {@link Clang#clang_getDiagnostic} method. Called with this
041         * flag, it will print the source location for errors and warnings including
042         * column numbers.
043         * 
044         * The magic numbers come from the C enum implementation (not exported properly
045         * by swig). Implement TS-19182 for fixing the export properly.
046         * https://github.com/llvm-mirror/clang/blob/release_80/include/clang-c/Index.h#
047         * L925 CXDiagnosticDisplayOptions.CXDiagnostic_DisplaySourceLocation = 0x01
048         * CXDiagnosticDisplayOptions.CXDiagnostic_DisplayColumn = 0x02
049         */
050        private static final int DIAGNOSTICS_FLAGS = 0x01 | 0x02;
051
052        /**
053         * Returns the direct children of the given cursor.
054         */
055        public static List<CXCursor> getDirectChildren(CXCursor cursor) {
056                final List<CXCursor> children = new ArrayList<>();
057                ClangBinding.visitChildren(cursor, new IClangCursorVisitor() {
058
059                        @Override
060                        public CXChildVisitResult visit(CXCursor cursor, CXCursor parent) {
061                                children.add(cursor);
062                                return CXChildVisitResult.CXChildVisit_Continue;
063                        }
064                });
065                return children;
066        }
067
068        /**
069         * Returns all children of the given cursor (recursive) that have the given
070         * {@link CXCursorKind}.
071         */
072        public static List<CXCursor> findChildrenWithKindRecursively(CXCursor cursor, final CXCursorKind cursorKind) {
073                final List<CXCursor> matchingCursors = new ArrayList<>();
074                ClangBinding.visitChildren(cursor, new IClangCursorVisitor() {
075                        @Override
076                        public CXChildVisitResult visit(CXCursor cursor, CXCursor parent) {
077                                if (cursor.getKind() == cursorKind) {
078                                        matchingCursors.add(cursor);
079                                }
080                                return CXChildVisitResult.CXChildVisit_Recurse;
081                        }
082                });
083                return matchingCursors;
084        }

086        /**
087         * Prints the diagnostics information ("errors and warnings" in Clang) generated
088         * during construction of the given translationUnit to the given
089         * {@link PrintStream}.
090         */
091        public static void debugPrintDiagnostics(SWIGTYPE_p_CXTranslationUnitImpl translationUnit, PrintStream out) {
092                long numDiagnostics = Clang.clang_getNumDiagnostics(translationUnit);
093                for (long i = 0; i < numDiagnostics; i++) {
094                        SWIGTYPE_p_void diag = Clang.clang_getDiagnostic(translationUnit, i);
095                        out.println(Clang.clang_getCString(Clang.clang_formatDiagnostic(diag, DIAGNOSTICS_FLAGS)));
096                }
097        }

107        /**
108         * Creates a {@link CheckTextRegionLocation} for the extent of the given clang
109         * cursor, if the given cursor is actually from the given file (and not some
110         * included header).
111         */
112        public static Optional<CheckTextRegionLocation> createTextRegionLocation(CXCursor cursor) {
113                CXSourceRange findingLocationExtent = Clang.clang_getCursorExtent(cursor);
114                CXSourceLocation findingLocationStart = Clang.clang_getRangeStart(findingLocationExtent);
115                if (Clang.clang_Location_isFromMainFile(findingLocationStart) == 0) {
116                        return Optional.empty();
117                }
118                ClangSpellingLocationProperties start = ClangBinding.getSpellingLocationProperties(findingLocationStart);
119                ClangSpellingLocationProperties end = ClangBinding
120                                .getSpellingLocationProperties(Clang.clang_getRangeEnd(findingLocationExtent));
121                return Optional.of(new CheckTextRegionLocation(start.getFile(), start.getOffset(), end.getOffset(),
122                                start.getLine(), end.getLine()));
123        }

125        /**
126         * Returns the code from the start to the end of the given {@link CXCursor}.
127         * This assumes that the cursors' TranslationUnit was created with the given
128         * code (same
129         * {@link eu.cqse.check.framework.core.phase.ECodeViewOption.ETextViewOption}
130         * text representation). Otherwise offsets will be wrong.
131         *
132         * Line breaks have been removed from the returned string (to simplify using it
133         * in user-visible messages).
134         *
135         * Just getting the name of the current cursor works for simple cases (with
136         * <code>Clang.clang_getCString(Clang.clang_getCursorSpelling(cursor))</code>),
137         * but if child is a more complex expression, this returns the empty string.
138         * Therefore, we get the text from the entire region of the cursor here.
139         */
140        public static Optional<String> getCompleteNodeText(CXCursor cursor, String mainFileContent) {
141                Optional<CheckTextRegionLocation> childLocation = createTextRegionLocation(cursor);
142                if (!childLocation.isPresent()) {
143                        return Optional.empty();
144                }
145
146                return Optional.of(StringUtils.replaceLineBreaks(
147                                mainFileContent.substring(childLocation.get().rawStartOffset, childLocation.get().rawEndOffset), ""));
148        }

150        /**
151         * Resolves the given type to its initial type definition.
152         *
153         * For example the type <code>X</code> in
154         * <code>typedef int A; typedef A B; typedef B X;</code> will be resolved to
155         * <code>int</code>.
156         */
157        public static CXType resolveTypedefs(CXType type) {
158                CXType resolvedType = type;
159                while (resolvedType.getKind() == CXTypeKind.CXType_Typedef) {
160                        resolvedType = Clang.clang_getTypedefDeclUnderlyingType(Clang.clang_getTypeDeclaration(resolvedType));
161                }
162                return resolvedType;
163        }

165        /**
166         * This is a hack to get the text of a binary operator. It is necessary since
167         * our clang JNI binding does not export the Clang BinaryOperator functions
168         * (e.g., clang::BinaryOperator::getOpcode).
169         * 
170         * This function throws an assertion error if called with a cursor that does not
171         * have kind CXCursor_BinaryOperator. Otherwise, it tries to returns the
172         * operator as String (e.g., "=" or "+"). The operator is normalized
173         * (leading/trailing whitespace is removed, linebreaks replaced by " ").
174         * 
175         * This won't work on cursors that were created by macros (we try to detect such
176         * situations and return an Optional.empty).
177         *
178         * @param operatorCursor
179         *            Cursor of kind {@link CXCursorKind#CXCursor_BinaryOperator}
180         * @param completeFileText
181         *            The text of the file used to create the current translation unit.
182         *            The operator text is extracted from this string based on the
183         *            offsets reported by clang.
184         * @return An Optional containing the text of the operator (e.g. "=" or "+") or
185         *         Optional.empty if the operator text can't be determined (e.g., macro
186         *         problems)
187         */
188        public static Optional<String> getBinaryOperatorText(CXCursor operatorCursor, String completeFileText) {
189                CXCursorKind kind = Clang.clang_getCursorKind(operatorCursor);
190                CCSMAssert.isTrue(Clang.clang_getCursorKind(operatorCursor) == CXCursorKind.CXCursor_BinaryOperator,
191                                () -> "Cursor must be of kind CXCursor_BinaryOperator to use this method. It was " + kind.toString()
192                                                + " instead.");
193                List<CXCursor> children = getDirectChildren(operatorCursor);
194                CXSourceLocation startLocation = Clang.clang_getRangeEnd(Clang.clang_getCursorExtent(children.get(0)));
195                CXSourceLocation endLocation = Clang.clang_getRangeStart(Clang.clang_getCursorExtent(children.get(1)));
196                if (Clang.clang_Location_isFromMainFile(startLocation) == 0
197                                || Clang.clang_Location_isFromMainFile(endLocation) == 0) {
198                        return Optional.empty();
199                }
200                ClangSpellingLocationProperties start = ClangBinding.getSpellingLocationProperties(startLocation);
201                ClangSpellingLocationProperties end = ClangBinding.getSpellingLocationProperties(endLocation);
202                if (start.getOffset() >= end.getOffset()) {
203                        // this was likely caused by a macro expansion
204                        return Optional.empty();
205                }
206
207                String operatorText = completeFileText.substring(start.getOffset(), end.getOffset());
208                operatorText = StringUtils.replaceLineBreaks(operatorText, " ").trim();
209                return Optional.of(operatorText);
210        }

212        /**
213         * Returns the cursor for a given token (if any found). This is the first cursor
214         * found in the tree with the exact same start/end offset as the given token.
215         * Depending on the structure of the AST and the offsets, this is not
216         * necessarily a leaf cursor.
217         */
218        public static Optional<CXCursor> getCursorForToken(IToken token, CXCursor rootCursor) {
219                return getCursorForStartEndOffset(token.getOffset(), token.getEndOffset() + 1, rootCursor);
220        }
221
222        /**
223         * Returns the cursor for a given start/end offset (if any found). This is the
224         * first cursor found in the tree with the exact start/end offset. Depending on
225         * the structure of the AST and the offsets, this is not necessarily a leaf
226         * cursor.
227         */
228        public static Optional<CXCursor> getCursorForStartEndOffset(int inclusiveStartOffset, int exclusiveEndOffset,
229                        CXCursor rootCursor) {
230                List<CXCursor> matches = new ArrayList<>();
231                ClangBinding.visitChildren(rootCursor, new IClangCursorVisitor() {
232                        @Override
233                        public CXChildVisitResult visit(CXCursor cursor, CXCursor parent) {
234                                CXSourceRange extent = Clang.clang_getCursorExtent(cursor);
235                                int cursorStartOffset = ClangBinding.getSpellingLocationProperties(Clang.clang_getRangeStart(extent))
236                                                .getOffset();
237                                int cursorEndOffset = ClangBinding.getSpellingLocationProperties(Clang.clang_getRangeEnd(extent))
238                                                .getOffset();
239
240                                if (cursorStartOffset > inclusiveStartOffset) {
241                                        return CXChildVisitResult.CXChildVisit_Break;
242                                }
243                                if (cursorEndOffset < exclusiveEndOffset) {
244                                        return CXChildVisitResult.CXChildVisit_Continue;
245                                }
246                                if (cursorStartOffset < inclusiveStartOffset || cursorEndOffset > exclusiveEndOffset) {
247                                        return CXChildVisitResult.CXChildVisit_Recurse;
248                                }
249
250                                matches.add(cursor);
251                                return CXChildVisitResult.CXChildVisit_Break;
252                        }
253                });
254                return matches.stream().findAny();
255        }
