025/**
026 * Util methods for using the libclang API
027 */
028public class ClangUtils {
029
030        /**
031         * Languages for which we enable parsing with clang.
032         */
033        public static final EnumSet<ELanguage> CLANG_ENABLED_LANGUAGES = EnumSet.of(ELanguage.CPP, ELanguage.OBJECTIVE_C);
034
035        /*
036         * Flags for the {@link Clang#clang_getDiagnostic} method. Called with this
037         * flag, it will print the source location for errors and warnings including
038         * column numbers.
039         * 
040         * The magic numbers come from the C enum implementation (not exported properly
041         * by swig). Implement TS-19182 for fixing the export properly.
042         * https://github.com/llvm-mirror/clang/blob/release_80/include/clang-c/Index.h#
043         * L925 CXDiagnosticDisplayOptions.CXDiagnostic_DisplaySourceLocation = 0x01
044         * CXDiagnosticDisplayOptions.CXDiagnostic_DisplayColumn = 0x02
045         */
046        private static final int DIAGNOSTICS_FLAGS = 0x01 | 0x02;
047
048        /**
049         * Returns the direct children of the given cursor.
050         */
051        public static List<CXCursor> getDirectChildren(CXCursor cursor) {
052                final List<CXCursor> children = new ArrayList<>();
053                ClangBinding.visitChildren(cursor, new IClangCursorVisitor() {
054
055                        @Override
056                        public CXChildVisitResult visit(CXCursor cursor, CXCursor parent) {
057                                children.add(cursor);
058                                return CXChildVisitResult.CXChildVisit_Continue;
059                        }
060                });
061                return children;
062        }
063
064        /**
065         * Returns all children of the given cursor (recursive) that have the given
066         * {@link CXCursorKind}.
067         */
068        public static List<CXCursor> findChildrenWithKindRecursively(CXCursor cursor, final CXCursorKind cursorKind) {
069                final List<CXCursor> matchingCursors = new ArrayList<>();
070                ClangBinding.visitChildren(cursor, new IClangCursorVisitor() {
071                        @Override
072                        public CXChildVisitResult visit(CXCursor cursor, CXCursor parent) {
073                                if (cursor.getKind() == cursorKind) {
074                                        matchingCursors.add(cursor);
075                                }
076                                return CXChildVisitResult.CXChildVisit_Recurse;
077                        }
078                });
079                return matchingCursors;
080        }
081
082        /**
083         * Prints the diagnostics information ("errors and warnings" in Clang) generated
084         * during construction of the given translationUnit to the given
085         * {@link PrintStream}.
086         */
087        public static void debugPrintDiagnostics(SWIGTYPE_p_CXTranslationUnitImpl translationUnit, PrintStream out) {
088                long numDiagnostics = Clang.clang_getNumDiagnostics(translationUnit);
089                for (long i = 0; i < numDiagnostics; i++) {
090                        SWIGTYPE_p_void diag = Clang.clang_getDiagnostic(translationUnit, i);
091                        out.println(Clang.clang_getCString(Clang.clang_formatDiagnostic(diag, DIAGNOSTICS_FLAGS)));
092                }
093        }
094
095        /**
096         * Prints the AST of the given translation unit to System.out.
097         */
098        public static void debugPrintTranslationUnitAST(SWIGTYPE_p_CXTranslationUnitImpl translationUnit) {
099                CXCursor topLevelCursor = Clang.clang_getTranslationUnitCursor(translationUnit);
100                ClangBinding.visitChildren(topLevelCursor, new PrintVisitor(new CursorPrinter(translationUnit), System.out));
101        }
102
103        /**
104         * Creates a {@link CheckTextRegionLocation} for the extent of the given clang
105         * cursor, if the given cursor is actually from the given file (and not some
106         * included header).
107         */
108        public static Optional<CheckTextRegionLocation> createTextRegionLocation(String uniformPath, CXCursor cursor) {
109                CXSourceRange findingLocationExtent = Clang.clang_getCursorExtent(cursor);
110                ClangSpellingLocationProperties start = ClangBinding
111                                .getSpellingLocationProperties(Clang.clang_getRangeStart(findingLocationExtent));
112                if (!start.getFile().equals(uniformPath)) {
113                        return Optional.empty();
114                }
115
116                ClangSpellingLocationProperties end = ClangBinding
117                                .getSpellingLocationProperties(Clang.clang_getRangeEnd(findingLocationExtent));
118                return Optional.of(new CheckTextRegionLocation(uniformPath, start.getOffset(), end.getOffset(), start.getLine(),
119                                end.getLine()));
120        }
121
122        /**
123         * Returns the code from the start to the end of the given {@link CXCursor}.
124         * This assumes that the cursors' TranslationUnit was created with the given
125         * code (same
126         * {@link eu.cqse.check.framework.core.phase.ECodeViewOption.ETextViewOption}
127         * text representation). Otherwise offsets will be wrong.
128         *
129         * Line breaks have been removed from the returned string (to simplify using it
130         * in user-visible messages).
131         *
132         * Just getting the name of the current cursor works for simple cases (with
133         * <code>Clang.clang_getCString(Clang.clang_getCursorSpelling(cursor))</code>),
134         * but if child is a more complex expression, this returns the empty string.
135         * Therefore, we get the text from the entire region of the cursor here.
136         */
137        public static Optional<String> getCompleteNodeText(CXCursor cursor, ICheckContext context, String mainFileContent) {
138                Optional<CheckTextRegionLocation> childLocation = createTextRegionLocation(context.getUniformPath(), cursor);
139                if (!childLocation.isPresent()) {
140                        return Optional.empty();
141                }
142
143                return Optional.of(StringUtils.replaceLineBreaks(
144                                mainFileContent.substring(childLocation.get().rawStartOffset, childLocation.get().rawEndOffset), ""));
145        }
146}
