005import java.util.List;
006import java.util.Map;
007import java.util.Optional;

009import org.conqat.lib.commons.assertion.CCSMAssert;
010import org.conqat.lib.commons.collections.CollectionUtils;
011
012import eu.cqse.check.util.clang.ClangUtils;
013import eu.cqse.clang.CXCursor;

015import eu.cqse.clang.CXType;

018/**
019 * Enumeration of the essential types defined by MISRA C2012 Section D.1
020 * 
021 * The essential-type concept in MISRA is a simplification of the C type system
022 * used in the definition of several misra guidelines. It does not apply to C++,
023 * only to C (MISRA Cpp2008 defines a different concept of "underlying types"
024 * which seems to be incompatible).
025 * 
026 * Do not modify the items of this enum. They are copied verbatim from the MISRA
027 * definition.
028 * 
029 * The ordering of the types is important (defines the rank and width of the
030 * types in the same category).
031 */
032public enum EEssentialType {
033
034        BOOL(EEssentialTypeCategory.BOOLEAN),
035
036        CHAR(EEssentialTypeCategory.CHARACTER),
037
038        SIGNED_CHAR(EEssentialTypeCategory.SIGNED),
039
040        SIGNED_SHORT(EEssentialTypeCategory.SIGNED),
041
042        SIGNED_INT(EEssentialTypeCategory.SIGNED),
043
044        SIGNED_LONG(EEssentialTypeCategory.SIGNED),
045
046        SIGNED_LONG_LONG(EEssentialTypeCategory.SIGNED),
047
048        UNSIGNED_CHAR(EEssentialTypeCategory.UNSIGNED),
049
050        UNSIGNED_SHORT(EEssentialTypeCategory.UNSIGNED),
051
052        UNSIGNED_INT(EEssentialTypeCategory.UNSIGNED),
053
054        UNSIGNED_LONG(EEssentialTypeCategory.UNSIGNED),
055
056        UNSIGNED_LONG_LONG(EEssentialTypeCategory.UNSIGNED),
057
058        /**
059         * All instances of NAMED_ENUM are considered different (MISRA C2012 D.5).
060         */
061        NAMED_ENUM(EEssentialTypeCategory.ENUM),
062
063        FLOAT(EEssentialTypeCategory.FLOATING),
064
065        DOUBLE(EEssentialTypeCategory.FLOATING),
066
067        LONG_DOUBLE(EEssentialTypeCategory.FLOATING);
068
069        private final EEssentialTypeCategory category;
070        /**
071         * Map from the basic type names used in clang to the {@link EEssentialType}
072         * enum items. This list does not include {@link EEssentialType#NAMED_ENUM}
073         * since the named-enum type names in clang start with "enum " but also contain
074         * the enum name.
075         */
076        private static final Map<String, EEssentialType> CLANG_TYPENAME_TO_ESSENTIAL_TYPE = CollectionUtils.asMap(
077                        // Essential-Type Category Boolean
078                        createPair("_bool", BOOL),
079                        // Essential-Type Category Character
080                        createPair("char", CHAR),
081                        // Essential-Type Category Signed
082                        // Note that all integer types except for char are signed by default.
083                        createPair("signed char", SIGNED_CHAR), //
084                        createPair("short", SIGNED_SHORT), createPair("signed short", SIGNED_SHORT), //
085                        createPair("int", SIGNED_INT), createPair("signed int", SIGNED_INT), //
086                        createPair("long", SIGNED_LONG), createPair("signed long", SIGNED_LONG),
087                        createPair("long long", SIGNED_LONG_LONG), createPair("signed long long", SIGNED_LONG_LONG),
088                        // Essential-Type Category Unsigned
089                        createPair("unsigned char", UNSIGNED_CHAR), //
090                        createPair("unsigned short", UNSIGNED_SHORT), //
091                        createPair("unsigned int", UNSIGNED_INT), //
092                        createPair("unsigned long", UNSIGNED_LONG), //
093                        createPair("unsigned long long", UNSIGNED_LONG_LONG),
094                        // Essential-Type Category Floating
095                        createPair("float", FLOAT), //
096                        createPair("double", DOUBLE), //
097                        createPair("long double", LONG_DOUBLE));
098
099        EEssentialType(EEssentialTypeCategory category) {
100                this.category = category;
101        }
102
103        /**
104         * Returns the highest-ranking type (and widest) of the given types. The given
105         * types must be of the same essential-type category.
106         */
107        public static EEssentialType getHighestRankType(EEssentialType type1, EEssentialType type2) {
108                CCSMAssert.isTrue(type1.category == type2.category,
109                                () -> "Essential types must be of same category for comparison (" + type1 + " vs " + type2 + ")");
110                if (type1.ordinal() > type2.ordinal()) {
111                        return type1;
112                }
113                return type2;
114        }
115
116        /**
117         * Determines the essential type of the given Cursor.
118         * <p>
119         * This implementation is based on Section D of Misra C2012.
120         */
121        public static Optional<EEssentialType> getEssentialTypeOfCursor(CXCursor cursor) {
122                CXCursorKind kind = Clang.clang_getCursorKind(cursor);
123                if (kind.equals(CXCursorKind.CXCursor_FirstExpr)) {
124                        CXCursor firstChild = ClangUtils.getDirectChildren(cursor).get(0);
125                        return getEssentialTypeOfCursor(firstChild);
126                } else if (kind.equals(CXCursorKind.CXCursor_BinaryOperator)) {
127                        // this is the implementation of the "Operations subject to the usual arithmetic
128                        // conversions (* / % + - & | ^)" section in C2012 D.6.
129                        // I did not test whether this works correctly for other binary operators such
130                        // as >> or =
131                        List<CXCursor> children = ClangUtils.getDirectChildren(cursor);
132                        Optional<EEssentialType> essentialLeft = getEssentialTypeOfCursor(children.get(0));
133                        Optional<EEssentialType> essentialRight = getEssentialTypeOfCursor(children.get(1));
134                        if (essentialLeft.isPresent() && essentialRight.isPresent()
135                                        && essentialLeft.get().category == essentialRight.get().category) {
136                                return Optional.of(getHighestRankType(essentialLeft.get(), essentialRight.get()));
137                        }
138                }
139
140                // use the "standard type" of the current cursor
141                CXType resolvedType = ClangUtils.resolveTypedefs(Clang.clang_getCursorType(cursor));
142                return fromClangResolvedType(resolvedType);
143        }
144
145        /**
146         * Returns the {@link EEssentialType} of the given {@link CXType}. This can't be
147         * done with {@link #valueOf(String)} since we can't be sure that types in Clang
148         * are named identical to the essential-type names in Misra.
149         */
150        public static Optional<EEssentialType> fromClangResolvedType(CXType resolvedType) {
151                String typeName = Clang.clang_getCString(Clang.clang_getTypeSpelling(resolvedType)).toLowerCase();
152                if (typeName.startsWith("enum ")) {
153                        return Optional.of(NAMED_ENUM);
154                }
155                return Optional.ofNullable(CLANG_TYPENAME_TO_ESSENTIAL_TYPE.get(typeName));
156        }
157
158        /**
159         * Returns whether this essential type is wider (needs more bits to be stored)
160         * than the given type.
161         */
162        public boolean isWiderThan(EEssentialType other) {
163                if (category != other.category) {
164                        return false;
165                }
166                return ordinal() > other.ordinal();
167        }
168
169        /**
170         * Category of this essential type (defined in Misra C:2012)
171         */
172        public EEssentialTypeCategory getCategory() {
173                return category;
174        }
175}
