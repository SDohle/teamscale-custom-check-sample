008package eu.cqse.check.xml;
009
010import java.util.HashMap;
011import java.util.Map;
012import java.util.Set;
013
014import org.xml.sax.Attributes;
015
016/**
017 * Base class for handlers to check an XML element.
018 */
019public abstract class XmlElementCheckHandlerBase extends XmlCheckHandlerBase {
020
021        /** The names of the XML Elements to be handled */
022        protected final Set<String> xmlElementNames;
023
024        /**
025         * The name of the ancestor XML element, within the elements in
026         * {@link #xmlElementNames} must be nested. If <code>null</code>, any
027         * Occurrence of the {@link #xmlElementNames} is handled.
028         */
029        protected final String requiredAncestorXmlElementName;
030
031        /**
032         * Nesting level of {@link #requiredAncestorXmlElementName}, 0 if not within
033         * an element of {@link #requiredAncestorXmlElementName}
034         */
035        private int ancestorLevel = 0;
036
037        /** The attributes of the current element. */
038        private Map<String, String> currentAttributes;
039
040        /** The attributes of the parent element. */
041        private Map<String, String> parentAttributes;
042
043        /**
044         * Buffer for the value string. Also indicates if parsing is within one of
045         * {@link #xmlElementNames}, if not {@link #valueBuffer} is
046         * <code>null</code>.
047         */
048        private StringBuilder valueBuffer = null;
049
050        /** Constructor. */
051        public XmlElementCheckHandlerBase(String parentXmlElementName, Set<String> xmlElementNames) {
052                this.requiredAncestorXmlElementName = parentXmlElementName;
053                this.xmlElementNames = xmlElementNames;
054        }
055
056        /** {@inheritDoc} */
057        @Override
058        public void startElement(String uri, String localName, String qName, Attributes attributes) {
059                updateAttributes(attributes);
060                if (requiredAncestorXmlElementName != null && localName.equals(requiredAncestorXmlElementName)) {
061                        ancestorLevel++;
062                        return;
063                }
064
065                if (isWithinRequiredAncestor() && xmlElementNames.contains(localName)) {
066                        valueBuffer = new StringBuilder();
067                        checkAttributes(attributes);
068                }
069        }
070
071        /**
072         * Updates attributes of current and parent element. Copying the values is
073         * required since {@link Attributes} is only working when within current
074         * element.
075         */
076        private void updateAttributes(Attributes attributes) {
077                parentAttributes = currentAttributes;
078                currentAttributes = new HashMap<String, String>();
079                for (int i = 0; i < attributes.getLength(); i++) {
080                        currentAttributes.put(attributes.getLocalName(i), attributes.getValue(i));
081                }
082        }
083
084        /** {@inheritDoc} */
085        @Override
086        public void endElement(String uri, String localName, String qName) {
087                if (requiredAncestorXmlElementName != null && localName.equals(requiredAncestorXmlElementName)) {
088                        ancestorLevel--;
089                        return;
090                }
091
092                if (isWithinRequiredAncestor() && xmlElementNames.contains(localName)) {
093                        checkValue(valueBuffer.toString());
094                        valueBuffer = null;
095                }
096        }
097
098        /** {@inheritDoc} */
099        @Override
100        public void characters(char[] ch, int start, int length) {
101                if (valueBuffer != null) {
102                        valueBuffer.append(ch, start, length);
103                }
104        }
105
106        /**
107         * @return true, if the current element has a ancestor of
108         *         {@link #requiredAncestorXmlElementName} or if no
109         *         {@link #requiredAncestorXmlElementName} is specified.
110         */
111        private boolean isWithinRequiredAncestor() {
112                return requiredAncestorXmlElementName == null || ancestorLevel > 0;
113        }
114
115        /**
116         * Gets an attribute of the parent element.
117         */
118        protected String getParentAttributeValue(String name) {
119                return parentAttributes.get(name);
120        }
121}
