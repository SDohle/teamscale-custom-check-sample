017package org.conqat.engine.commons.findings.location;
018
019import java.util.ArrayList;
020import java.util.Collections;
021import java.util.Comparator;
022import java.util.List;
023import java.util.Set;
024import java.util.regex.Matcher;
025import java.util.regex.Pattern;
026
027import org.conqat.lib.commons.algo.Diff;
028import org.conqat.lib.commons.algo.Diff.Delta;
029import org.conqat.lib.commons.region.Region;
030import org.conqat.lib.commons.region.SimpleRegion;
031import org.conqat.lib.commons.string.LineOffsetConverter;

033/**
034 * This class is used for adjusting the offsets used in locations (i.e.
035 * subclasses of {@link ElementLocation} for text that is slightly modified. The
036 * main use-case is the update of locations where the local (adjusted) text has
037 * different line ending, different content due to keyword expansion, or minor
038 * local modifications compared to the text on which the analysis was executed
039 * (original text).
040 *
041 * Both the original and adjusted text may have arbitrary line endings.
042 *
043 * The implementation is based on a token diff, which can lead to minor
044 * deviations for offsets that are not aligned with token boundaries. A
045 * character diff would be more precise, but is too performance and memory
046 * intensive for large files.
047 */
048public class LocationAdjuster {
049
050        /**
051         * If the number of tokens in the adjusted region differs by the tokens in the
052         * original region by more than this factor, the mapping is counted as wrong.
053         */
054        private static final double LOSS_FACTOR = 2;
055
056        /** Maximal number of tokens in the diff we accept. */
057        private static final int MAX_DIFF_SIZE = 5000;
058
059        /**
060         * Pattern defining tokens for the diff. Matches either alphanumeric strings
061         * (typical identifiers), or single non-whitespace characters.
062         */
063        private static final Pattern TOKEN_PATTERN = Pattern.compile("[a-zA-Z0-9_]+|\\S");
064
065        /** The tokens of the original string. */
066        private final List<AdjusterToken> originalTokens;
067
068        /**
069         * Adjusted tokens corresponding to the {@link #originalTokens}. If there is no
070         * corresponding token, this list contains null at the index. If the content
071         * could not be matched/adjusted at all (too many differences), this field is
072         * null.
073         */
074        private final List<AdjusterToken> mappedAdjustedTokens;
075
076        /** Line offset converted for the original text. */
077        private final LineOffsetConverter originalLineOffsetConverter;
078
079        /** Line offset converted for the adjusted text. */
080        private final LineOffsetConverter adjustedLineOffsetConverter;
081
082        /**
083         * Optional uniform path to correct element locations of
084         * {@link #adjustLocation(ElementLocation)}. Null if no correction will be
085         * performed.
086         */
087        private String adjustedUniformPath;
088
089        /**
090         * Constructor.
091         *
092         * @param originalText
093         *            the text for which the input locations have been created, i.e. the
094         *            text from the analysis. May be <code>null</code>, in which case no
095         *            adjustment is performed.
096         * @param adjustedText
097         *            the text for which the locations should be adjusted, i.e. the
098         *            local text.
099         * @param adjustedUniformPath
100         *            the adjusted uniform path for adjusted findings.
101         */
102        public LocationAdjuster(String originalText, String adjustedText, String adjustedUniformPath) {
103                if (originalText == null) {
104                        originalText = adjustedText;
105                }
106                this.adjustedUniformPath = adjustedUniformPath;
107                originalLineOffsetConverter = new LineOffsetConverter(originalText);
108                adjustedLineOffsetConverter = new LineOffsetConverter(adjustedText);
109                originalTokens = toTokens(originalText);
110                mappedAdjustedTokens = calculateMappedAdjustedTokens(adjustedText, originalTokens);
111        }
112
113        /**
114         * Constructor.
115         *
116         * @param originalText
117         *            the text for which the input locations have been created, i.e. the
118         *            text from the analysis.
119         * @param adjustedText
120         *            the text for which the locations should be adjusted, i.e. the
121         *            local text.
122         */
123        public LocationAdjuster(String originalText, String adjustedText) {
124                this(originalText, adjustedText, null);
125        }
126
127        /**
128         * Calculates the #mappedAdjustedTokens based on original tokens and adjusted
129         * text. May return null if adjustment is not possible due too many changes.
130         */
131        private static List<AdjusterToken> calculateMappedAdjustedTokens(String adjustedText,
132                        List<AdjusterToken> originalTokens) {
133
134                List<AdjusterToken> adjustedTokens = toTokens(adjustedText);
135                Delta<AdjusterToken> delta = Diff.computeDelta(originalTokens, adjustedTokens, MAX_DIFF_SIZE);
136
137                if (delta.getSize() >= MAX_DIFF_SIZE) {
138                        return null;
139                }
140
141                return calculateMappedAdjustedTokensFromDelta(delta, originalTokens, adjustedTokens);
142        }
143
144        /**
145         * Calculates the #mappedAdjustedTokens based on original tokens, adjusted text,
146         * and delta.
147         */
148        private static List<AdjusterToken> calculateMappedAdjustedTokensFromDelta(Delta<AdjusterToken> delta,
149                        List<AdjusterToken> originalTokens, List<AdjusterToken> adjustedTokens) {
150                List<AdjusterToken> mappedAdjustedTokens = new ArrayList<>(
151                                Collections.nCopies(originalTokens.size(), (AdjusterToken) null));
152                int originalIndex = 0;
153                int adjustedIndex = 0;
154                for (int i = 0; i < delta.getSize(); ++i) {
155                        int position = delta.getPosition(i);
156                        if (position > 0) {
157                                position -= 1;
158
159                                while (adjustedIndex < position) {
160                                        mappedAdjustedTokens.set(originalIndex++, adjustedTokens.get(adjustedIndex++));
161                                }
162                                adjustedIndex += 1;
163                        } else {
164                                position = -position - 1;
165
166                                while (originalIndex < position) {
167                                        mappedAdjustedTokens.set(originalIndex++, adjustedTokens.get(adjustedIndex++));
168                                }
169                                originalIndex += 1;
170                        }
171                }
172
173                while (originalIndex < originalTokens.size()) {
174                        mappedAdjustedTokens.set(originalIndex++, adjustedTokens.get(adjustedIndex++));
175                }
176
177                return mappedAdjustedTokens;
178        }
179
180        /** Splits a string into tokens. */
181        private static List<AdjusterToken> toTokens(String s) {
182                List<AdjusterToken> tokens = new ArrayList<>();
183                Matcher matcher = TOKEN_PATTERN.matcher(s);
184                while (matcher.find()) {
185                        tokens.add(new AdjusterToken(matcher.group(), matcher.start()));
186                }
187                return tokens;
188        }
189
190        /**
191         * Maps a zero-based offset range (both inclusive) to the adjusted string.
192         * Returns <code>null</code> if the region could not be approximately mapped.
193         */
194        public Region getAdjustedRegion(int originalStartOffset, int originalEndOffset) {
195
196                if (mappedAdjustedTokens == null) {
197                        return null;
198                }
199
200                Region originalIndexRegion = findOriginalIndexRegion(originalStartOffset, originalEndOffset);
201                if (originalIndexRegion.isEmpty()) {
202                        return null;
203                }
204
205                int numOriginalTokens = originalIndexRegion.getLength();
206                int numAdjustedTokens = 0;
207
208                AdjusterToken firstAdjustedToken = null;
209                AdjusterToken lastAdjustedToken = null;
210                for (int i = originalIndexRegion.getStart(); i <= originalIndexRegion.getEnd(); ++i) {
211                        AdjusterToken adjustedToken = mappedAdjustedTokens.get(i);
212                        if (adjustedToken != null) {
213                                numAdjustedTokens += 1;
214                                if (firstAdjustedToken == null) {
215                                        firstAdjustedToken = adjustedToken;
216                                }
217                                lastAdjustedToken = adjustedToken;
218                        }
219                }
220
221                if (firstAdjustedToken == null || lastAdjustedToken == null
222                                || LOSS_FACTOR * numAdjustedTokens < numOriginalTokens) {
223                        return null;
224                }
225
226                return new Region(firstAdjustedToken.startOffset, lastAdjustedToken.endOffset);
227        }
228
229        /**
230         * Returns the region of indexes in the {@link #originalTokens} contained in the
231         * given offsets.
232         */
233        private Region findOriginalIndexRegion(int originalStartOffset, int originalEndOffset) {
234                AdjusterToken searchToken = new AdjusterToken(null, originalStartOffset, originalEndOffset);
235                int originalStartTokenIndex = Collections.binarySearch(originalTokens, searchToken,
236                                AdjusterToken.COMPARE_BY_START_OFFSET);
237                if (originalStartTokenIndex < 0) {
238                        originalStartTokenIndex = -originalStartTokenIndex - 1;
239                }
240
241                int originalEndTokenIndex = Collections.binarySearch(originalTokens, searchToken,
242                                AdjusterToken.COMPARE_BY_END_OFFSET);
243                if (originalEndTokenIndex < 0) {
244                        // we want insertion point -1
245                        originalEndTokenIndex = -originalEndTokenIndex - 2;
246                }
247
248                return new Region(originalStartTokenIndex, originalEndTokenIndex);
249        }

251        /**
252         * Returns a new location with adjusted offsets (if necessary). Returns
253         * <code>null</code> if the location cannot be mapped to the adjusted text.
254         */
255        public ElementLocation adjustLocation(ElementLocation location) {
256                if (location instanceof TextRegionLocation) {
257                        return adjustLocation((TextRegionLocation) location);
258                }
259
260                // other locations do not have offsets, if the uniform path should not
261                // be adjusted simply return the original location.
262                if (adjustedUniformPath == null) {
263                        return location;
264                }
265
266                if (location instanceof QualifiedNameLocation) {
267                        return new QualifiedNameLocation(((QualifiedNameLocation) location).getQualifiedName(),
268                                        location.getLocation(), this.adjustedUniformPath);
269                }
270
271                return new ElementLocation(location.getLocation(), adjustedUniformPath);
272        }
273
274        /**
275         * Returns a new location with adjusted offsets (if necessary). Returns
276         * <code>null</code> if the cannot be mapped to the adjusted text.
277         */
278        public TextRegionLocation adjustLocation(TextRegionLocation location) {
279                int startOffset = location.getRawStartOffset();
280                int endOffset = location.getRawEndOffset();
281                if (startOffset < 0) {
282                        startOffset = originalLineOffsetConverter.getOffset(location.getRawStartLine());
283                }
284                if (endOffset < 0) {
285                        endOffset = originalLineOffsetConverter.getOffset(location.getRawEndLine() + 1) - 1;
286                }
287
288                Region adjustedOffsets = getAdjustedRegion(startOffset, endOffset);
289
290                if (adjustedOffsets == null || adjustedOffsets.isEmpty()) {
291                        return null;
292                }
293
294                String uniformPath = location.getUniformPath();
295                if (this.adjustedUniformPath != null) {
296                        uniformPath = adjustedUniformPath;
297                }
298
299                int newStartOffset = adjustedOffsets.getStart();
300                int newEndOffset = adjustedOffsets.getEnd();
301                return new TextRegionLocation(location.getLocation(), uniformPath, newStartOffset, newEndOffset,
302                                adjustedLineOffsetConverter.getLine(newStartOffset), adjustedLineOffsetConverter.getLine(newEndOffset));
303        }
304
305        /**
306         * Adjusts the location of a single line. This only respects the token part of a
307         * line, i.e. leading and trailing whitespace of a line will be ignored. This
308         * method is robust w.r.t lines numbers that are out of the range of the
309         * original text. In case of such an invalid line, the line is logged as error
310         * to the given logger and <code>null</code> is returned.
311         *
312         * @param line
313         *            the one-based line number of be adjusted.
314         * @param invalidLines
315         *            used for collecting invalid lines.
316         * 
317         * @return the one-based lines encoded as a region, as a line may map to
318         *         multiple lines after changing. This may also return null, if no
319         *         non-empty lines could be found that correspond to the input line
320         *         after adjustment.
321         */
322        public SimpleRegion adjustLine(int line, Set<Integer> invalidLines) {
323                if (!originalLineOffsetConverter.isValidLine(line) || !originalLineOffsetConverter.isValidLine(line + 1)) {
324                        invalidLines.add(line);
325                        return null;
326                }
327                int originalStartOffset = originalLineOffsetConverter.getOffset(line);
328                int originalEndOffset = originalLineOffsetConverter.getOffset(line + 1) - 1;
329                Region adjustedOffsets = getAdjustedRegion(originalStartOffset, originalEndOffset);
330                if (adjustedOffsets == null) {
331                        return null;
332                }
333
334                int adjustedStartLine = adjustedLineOffsetConverter.getLine(adjustedOffsets.getStart());
335                int adjustedEndLine = adjustedLineOffsetConverter.getLine(adjustedOffsets.getEnd());
336                return new SimpleRegion(adjustedStartLine, adjustedEndLine);
337        }
338
339        /** Returns the line count of the original text */
340        public int getOriginalLineCount() {
341                return originalLineOffsetConverter.getLineCount();
342        }
343
344        /** Simple token representation used in location adjustment. */
345        private static class AdjusterToken {
346
347                /** Compares by start offset. */
348                private static final Comparator<AdjusterToken> COMPARE_BY_START_OFFSET = new Comparator<AdjusterToken>() {
349                        @Override
350                        public int compare(AdjusterToken token1, AdjusterToken token2) {
351                                return token1.startOffset - token2.startOffset;
352                        }
353                };
354
355                /** Compares by end offset. */
356                private static final Comparator<AdjusterToken> COMPARE_BY_END_OFFSET = new Comparator<AdjusterToken>() {
357                        @Override
358                        public int compare(AdjusterToken token1, AdjusterToken token2) {
359                                return token1.endOffset - token2.endOffset;
360                        }
361                };
362
363                /** The text content. */
364                private final String text;
365
366                /** The start offset in the text. */
367                private final int startOffset;
368
369                /** The inclusive end offset in the text. */
370                private final int endOffset;
371
372                /** Constructor. */
373                public AdjusterToken(String text, int startOffset) {
374                        this(text, startOffset, startOffset + text.length() - 1);
375                }
376
377                /** Constructor. */
378                public AdjusterToken(String text, int startOffset, int endOffset) {
379                        this.text = text;
380                        this.startOffset = startOffset;
381                        this.endOffset = endOffset;
382                }
383
384                /** {@inheritDoc} */
385                @Override
386                public boolean equals(Object obj) {
387                        return (obj instanceof AdjusterToken) && ((AdjusterToken) obj).text.equals(text);
388                }
389
390                /** {@inheritDoc} */
391                @Override
392                public int hashCode() {
393                        return text.hashCode();
394                }
395        }
