017package org.conqat.engine.commons.util;
018
019import java.io.IOException;
020import java.lang.reflect.Field;
021import java.lang.reflect.Modifier;
022import java.util.Optional;
023
024import org.conqat.engine.core.core.ConQATException;
025import org.conqat.lib.commons.collections.CollectionUtils;
026import org.conqat.lib.commons.string.StringUtils;
027
028import com.fasterxml.jackson.annotation.JsonAutoDetect;
029import com.fasterxml.jackson.annotation.JsonIgnore;
030import com.fasterxml.jackson.annotation.JsonInclude;
031import com.fasterxml.jackson.annotation.PropertyAccessor;
032import com.fasterxml.jackson.core.JsonParser;
033import com.fasterxml.jackson.core.JsonProcessingException;
034import com.fasterxml.jackson.core.type.TypeReference;
035import com.fasterxml.jackson.databind.DeserializationFeature;
036import com.fasterxml.jackson.databind.JsonNode;
037import com.fasterxml.jackson.databind.ObjectMapper;
038
039/**
040 * Utility code for dealing with JSON. This uses Jackson for serialization and
041 * deserialization to JSON.
042 */
043public class JsonUtils {
044
045        /**
046         * The shared jackson object mapper instance. Once configured the object is
047         * thread-safe.
048         */
049        private static final ObjectMapper OBJECT_MAPPER = new ObjectMapper();
050
051        static {
052                OBJECT_MAPPER.registerModule(new ColorSerializationModule());
053                OBJECT_MAPPER.setVisibility(PropertyAccessor.ALL, JsonAutoDetect.Visibility.NONE);
054                OBJECT_MAPPER.setVisibility(PropertyAccessor.FIELD, JsonAutoDetect.Visibility.ANY);
055                OBJECT_MAPPER.setSerializationInclusion(JsonInclude.Include.NON_NULL);
056                OBJECT_MAPPER.enable(DeserializationFeature.FAIL_ON_TRAILING_TOKENS);
057                OBJECT_MAPPER.enable(JsonParser.Feature.ALLOW_UNQUOTED_FIELD_NAMES);
058                OBJECT_MAPPER.enable(JsonParser.Feature.ALLOW_SINGLE_QUOTES);
059                OBJECT_MAPPER.disable(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES);
060        }
061
062        /**
063         * Serializes the given object to JSON.
064         */
065        public static String serializeToJSON(Object object) {
066                try {
067                        return OBJECT_MAPPER.writeValueAsString(object);
068                } catch (JsonProcessingException e) {
069                        throw new JsonSerializationException(e);
070                }
071        }
072
073        /**
074         * Serializes the given object to JSON as UTF-8 encoded byte array.
075         */
076        public static byte[] serializeToJSONByteArray(Object object) {
077                try {
078                        return OBJECT_MAPPER.writeValueAsBytes(object);
079                } catch (JsonProcessingException e) {
080                        throw new JsonSerializationException(e);
081                }
082        }
083
084        /** @see #OBJECT_MAPPER */
085        public static ObjectMapper getObjectMapper() {
086                return OBJECT_MAPPER;
087        }
088
089        /**
090         * Wrapper exception for {@link JsonProcessingException} as the other is a
091         * checked exception and we don't want to check it everywhere.
092         */
093        public static class JsonSerializationException extends RuntimeException {
094                JsonSerializationException(Throwable cause) {
095                        super(cause);
096                }
097        }
098
099        /**
100         * Deserializes a JSON string.
101         *
102         * @throws ConQATException
103         *             if the input string could not be parsed as JSON or the class is
104         *             not constructible.
105         */
106        public static <T> T deserializeFromJson(String json, Class<T> expectedClass) throws ConQATException {
107                return safeConvert(objectMapper -> objectMapper.readValue(json, expectedClass));
108        }
109
110        /**
111         * Deserializes a JSON string.
112         *
113         * @throws ConQATException
114         *             if the input string could not be parsed as JSON or the class is
115         *             not constructible.
116         */
117        public static <T> T deserializeFromJson(String json, TypeReference expectedType) throws ConQATException {
118                return safeConvert(objectMapper -> objectMapper.readValue(json, expectedType));
119        }
120
121        /**
122         * Deserializes a JSON string into a {@link JsonNode}.
123         *
124         * @throws ConQATException
125         *             if the input string could not be parsed as JSON or the class is
126         *             not constructible.
127         */
128        public static JsonNode deserializeFromJson(String json) throws ConQATException {
129                return safeConvert(objectMapper -> objectMapper.readTree(json));
130        }
131
132        /**
133         * Deserializes a JSON string. Throws an error if the resulting object is null
134         * or contains an attribute that is null.
135         *
136         * @throws ConQATException
137         *             if the input string could not be parsed as JSON, or the class is
138         *             not constructible, or the resulting object or one of its
139         *             attributes would be null.
140         */
141        public static <T> T deserializeFromJsonWithNullCheck(String json, Class<T> expectedClass) throws ConQATException {
142                T parsedObject = deserializeFromJson(json, expectedClass);
143                return NullableFieldValidator.ensureAllFieldsNonNull(parsedObject, json);
144        }
145
146        /**
147         * Returns the json value or {@code null} for the given key, while trying 1. a
148         * lowercase version of the key (e.g. 'project'), and 2. a capitalized version
149         * of the lowercase key ('Project');
150         */
151        public static Optional<JsonNode> getWithCapitalizedOrLowercaseKey(JsonNode jsonNode, String key) {
152                key = key.toLowerCase();
153                if (jsonNode.has(key)) {
154                        return Optional.of(jsonNode.get(key));
155                }
156                return Optional.ofNullable(jsonNode.get(StringUtils.capitalize(key)));
157        }
158
159        /** Returns whether a string is a valid json or not. */
160        public static boolean isValidJson(String json) {
161                try {
162                        OBJECT_MAPPER.readTree(json);
163                        return true;
164                } catch (IOException e) {
165                        return false;
166                }
167        }
168
169        /** Reformats a JSON string to be pretty printed. */
170        public static String prettyPrintJSON(String json) throws ConQATException {
171                return safeConvert(objectMapper -> {
172                        JsonNode jsonInstance = objectMapper.readTree(json);
173                        return objectMapper.writerWithDefaultPrettyPrinter().writeValueAsString(jsonInstance);
174                });
175        }
176
177        /** Reformats a JSON string to be pretty printed. */
178        public static String serializeToJSONPrettyPrinted(Object object) throws ConQATException {
179                return safeConvert(objectMapper -> objectMapper.writerWithDefaultPrettyPrinter().writeValueAsString(object));
180        }
181
182        /**
183         * Safely converts json to an object by calling the given supplier and wrapping
184         * any thrown exceptions in {@link ConQATException}.
185         */
186        private static <T> T safeConvert(CollectionUtils.FunctionWithException<ObjectMapper, T, IOException> convertAction)
187                        throws ConQATException {
188                try {
189                        return convertAction.apply(OBJECT_MAPPER);
190                } catch (JsonProcessingException e) {
191                        throw new ConQATException("Input was invalid JSON.", e);
192                } catch (Throwable t) {
193                        throw new ConQATException("Trouble during JSON processing: " + t.getMessage(), t);
194                }
195        }
196
197        /**
198         * Wrapper exception for {@link JsonProcessingException} as the other is a
199         * checked exception and we don't want to check it everywhere.
200         */
201        public static class JsonDeserializationException extends RuntimeException {
202                public JsonDeserializationException(String message, Throwable cause) {
203                        super(message, cause);
204                }
205        }
206
207        /**
208         * @return whether the given field is not serialized, i.e. skipped in JSON
209         *         responses to the client.
210         */
211        public static boolean isNotSerialized(Field field) {
212
213                int fieldModifiers = field.getModifiers();
214
215                // Jackson excludes transient and static fields by default.
216                if (Modifier.isStatic(fieldModifiers) || Modifier.isTransient(fieldModifiers)) {
217                        return true;
218                }
219                // JsonIgnore indicates to skip this field
220                if (field.isAnnotationPresent(JsonIgnore.class)) {
221                        return true;
222                }
223                // This is either due to dynamically inserted fields created by Jacoco during
224                // test execution or to anonymous inner classes
225                if (field.getName().contains("$")) {
226                        return true;
227                }
228
229                return false;
230        }
