017package org.conqat.engine.commons.util;
018
019import java.lang.reflect.Field;
020import java.util.ArrayDeque;
021import java.util.ArrayList;
022import java.util.Deque;
023import java.util.List;
024
025import javax.annotation.Nullable;
026
027import org.apache.commons.lang3.builder.ReflectionToStringBuilder;
028import org.conqat.engine.core.core.ConQATException;
029import org.conqat.lib.commons.collections.IdentityHashSet;
030import org.conqat.lib.commons.reflect.ReflectionUtils;
031import org.conqat.lib.commons.string.StringUtils;
032
033/**
034 * Validator that checks whether all values of an object are non-null unless
035 * explicitly specified as {@link Nullable}.
036 */
037public class NullableFieldValidator {
038
039        /**
040         * Ensures that the given object is not null and that all its fields are set
041         * (throws {@link ConQATException} otherwise). Arrays and other iterables are
042         * checked recursively, as well as all fields of the given object (but only if
043         * their type is defined in ConQAT/Teamscale packages).
044         *
045         * If you get a false positive for a field from this method, annotate the field
046         * as {@link Nullable}.
047         */
048        public static <T> T ensureAllFieldsNonNull(T parsedObject, String queryContent) throws ConQATException {
049                if (StringUtils.isEmpty(queryContent)) {
050                        // Returns null, but that's okay for empty queries
051                        return parsedObject;
052                }
053
054                try {
055                        return ensureAllFieldsNonNull(parsedObject);
056                } catch (ConQATException e) {
057                        throw new ConQATException(e.getMessage() + "\nQuery: " + queryContent, e);
058                }
059        }
060
061        /**
062         * Ensures that the given object is not null and that all its fields are set
063         * (throws {@link ConQATException} otherwise). Arrays and other iterables are
064         * checked recursively, as well as all fields of the given object (but only if
065         * their type is defined in ConQAT/Teamscale packages).
066         *
067         * If you get a false positive for a field from this method, annotate the field
068         * as {@link Nullable}.
069         */
070        public static <T> T ensureAllFieldsNonNull(T parsedObject) throws ConQATException {
071                if (parsedObject == null) {
072                        throw createNullObjectException("<null>", "whole object");
073                }
074
075                Deque<Object> objectsToCheck = new ArrayDeque<>();
076                if (ReflectionUtils.isIterable(parsedObject)) {
077                        objectsToCheck.addAll(checkIterableAndGetContainedFields(parsedObject, "<object in given iterable>"));
078                } else {
079                        objectsToCheck.push(parsedObject);
080                }
081
082                IdentityHashSet<Object> visitedObjects = new IdentityHashSet<>();
083                while (!objectsToCheck.isEmpty()) {
084                        Object objectToCheck = objectsToCheck.pop();
085                        if (!visitedObjects.add(objectToCheck)) {
086                                // We have already checked this object
087                                continue;
088                        }
089                        objectsToCheck.addAll(checkObjectAndGetReferencedFields(objectToCheck));
090                }
091                return parsedObject;
092        }
093
094        /** Create a null object exception. */
095        private static <T> ConQATException createNullObjectException(T object, String checkedPart) {
096                return new ConQATException("The query contained input that would have been deserialized to null (" + checkedPart
097                                + ").\nResulting object: " + ReflectionToStringBuilder.toString(object)
098                                + " - Missing @Nullable annotation?");
099        }
100
101        /** Checks a single object and returns the fields it references. */
102        private static List<Object> checkObjectAndGetReferencedFields(Object objectToCheck) throws ConQATException {
103                List<Object> referencedObjects = new ArrayList<>();
104                for (Field field : ReflectionUtils.getAllFields(objectToCheck.getClass())) {
105                        if (JsonUtils.isNotSerialized(field) || fieldMaySkipNullCheck(field)) {
106                                continue;
107                        }
108
109                        field.setAccessible(true);
110                        try {
111                                Object referencedObject = field.get(objectToCheck);
112                                if (referencedObject == null) {
113                                        throw createNullObjectException(objectToCheck, "at field " + field.getName());
114                                }
115
116                                if (ReflectionUtils.isIterable(referencedObject)) {
117                                        referencedObjects.addAll(checkIterableAndGetContainedFields(referencedObject, field.getName()));
118                                        // We don't need to add the container object to referenced
119                                        // objects
120                                        continue;
121                                }
122
123                                // Check for correct package. This can only be done down here, if we included it
124                                // in fieldMaySkipNullCheck() we would also skip over iterable Java core types
125                                if (field.getType().getPackage() != null
126                                                && StringUtils.containsOneOf(field.getType().getPackage().getName(), "teamscale", "conqat")) {
127                                        referencedObjects.add(referencedObject);
128                                }
129                        } catch (IllegalArgumentException | IllegalAccessException e) {
130                                throw new ConQATException("Could not access field", e);
131                        }
132                }
133                return referencedObjects;
134        }
135
136        /** Check */
137        private static List<Object> checkIterableAndGetContainedFields(Object arrayOrIterable, String checkedField)
138                        throws ConQATException {
139                List<Object> containedObjects = new ArrayList<>();
140                for (Object containedObject : ReflectionUtils.asIterable(arrayOrIterable)) {
141                        if (containedObject == null) {
142                                throw createNullObjectException(arrayOrIterable, "at field " + checkedField);
143                        }
144                        containedObjects.add(containedObject);
145                }
146                return containedObjects;
147        }
148
149        /**
150         * Whether a field is relevant for the null check, i.e. it (a) can be null, and
151         * (b) is not expected to actually be null.
152         */
153        private static boolean fieldMaySkipNullCheck(Field field) {
154                return field.getType().isPrimitive() || field.isAnnotationPresent(Nullable.class)
155                                || field.getName().contains("$");
156        }
157}
