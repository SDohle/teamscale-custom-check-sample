017package org.conqat.engine.index.shared;
018
019import java.io.Serializable;
020import java.util.Arrays;
021import java.util.Comparator;
022import java.util.Objects;
023
024import org.conqat.engine.commons.util.JsonUtils;
025import org.conqat.engine.core.core.ConQATException;
026import org.conqat.lib.commons.assertion.CCSMAssert;
027import org.conqat.lib.commons.io.ByteArrayUtils;
028import org.conqat.lib.commons.js_export.ExportToJavaScript;
029import org.conqat.lib.commons.string.StringUtils;
030
031import com.fasterxml.jackson.annotation.JsonCreator;
032import com.fasterxml.jackson.annotation.JsonProperty;
033
034/**
035 * Immutable class describing a single commit by its branch name and a
036 * timestamp. The timestamp must be unique within the branch. They are
037 * comparable by timestamp, where equal timestamps are resolved alphabetically
038 * by the branch name.
039 *
040 * <strong>This class is used for communication with IDE clients (via the
041 * {@link org.conqat.engine.service.shared.client.IdeServiceClient}), so special
042 * care has to be taken when changing its signature!</strong>
043 */
044@ExportToJavaScript
045public class CommitDescriptor implements Serializable, Comparable<CommitDescriptor> {
046
047        /**
048         * Name of the branch if no branch is specified.
049         * 
050         * @see org.conqat.engine.persistence.store.hist.HistoryAccessOption#NO_BRANCH_NAME
051         */
052        // HistoryAccessOption is not accessible in all project where this class is
053        // used as external source file
054        @SuppressWarnings("javadoc")
055        private static final String NO_BRANCH_NAME = "##no-branch##";
056
057        /** Version for serialization. */
058        private static final long serialVersionUID = 1L;
059
060        /** Special value to indicate the HEAD of a branch. */
061        public static final String HEAD_TIMESTAMP = "HEAD";
062
063        /**
064         * A comparator for comparison by timestamps. Null commits are ordered to the
065         * front.
066         * 
067         * If there are commits with identical timestamps, we order them by comparing
068         * the branch names to guarantee a stable order. This should only happen in
069         * special cases since teamscale ensures that no timestamp is given to two
070         * commits.
071         */
072        public static final Comparator<CommitDescriptor> BY_TIMESTAMP_COMPARATOR = Comparator.nullsFirst(
073                        Comparator.comparingLong(CommitDescriptor::getTimestamp).thenComparing(CommitDescriptor::getBranchName));
074
075        /** The name of the JSON property name for {@link #branchName}. */
076        private static final String BRANCH_NAME_PROPERTY = "branchName";
077
078        /** The name of the JSON property name for {@link #timestamp}. */
079        private static final String TIMESTAMP_PROPERTY = "timestamp";
080
081        /** The name of the branch. */
082        @JsonProperty(BRANCH_NAME_PROPERTY)
083        private final String branchName;
084
085        /** The timestamp on the branch. */
086        @JsonProperty(TIMESTAMP_PROPERTY)
087        private final long timestamp;
088
089        /**
090         * Constructor. <br/>
091         * use {@link CommitDescriptor#createUnbranchedDescriptor(long)} to create a
092         * unbranched commit descriptor.
093         */
094        @JsonCreator
095        public CommitDescriptor(@JsonProperty(BRANCH_NAME_PROPERTY) String branchName,
096                        @JsonProperty(TIMESTAMP_PROPERTY) long timestamp) {
097                CCSMAssert.isTrue(timestamp >= 0, "Timestamp must be >= 0 but is " + timestamp);
098                CCSMAssert.isNotNull(branchName);
099
100                this.branchName = branchName;
101                this.timestamp = timestamp;
102        }
103
104        /**
105         * Create a {@link CommitDescriptor} without branch specification. <br/>
106         * Should be used with caution.
107         */
108        public static CommitDescriptor createUnbranchedDescriptor(long timestamp) {
109                return new CommitDescriptor(NO_BRANCH_NAME, timestamp);
110        }
111
112        /**
113         * Create a copy of this commit descriptor with <code>timestamp - 1</code>.
114         * 
115         * Using this CommitDescriptor to store a new commit might overwrite an existing
116         * commit if the timestamp is already used.
117         */
118        public CommitDescriptor cloneWithDecrementedTimestamp() {
119                return new CommitDescriptor(this.branchName, this.timestamp - 1);
120        }
121
122        /**
123         * Create a copy of this commit descriptor with <code>timestamp + 1</code>.
124         * 
125         * Using this CommitDescriptor to store a new commit might overwrite an existing
126         * commit if the timestamp is already used.
127         */
128        public CommitDescriptor cloneWithIncrementedTimestamp() {
129                return new CommitDescriptor(this.branchName, this.timestamp + 1);
130        }
131
132        /**
133         * @see #branchName
134         */
135        public String getBranchName() {
136                return branchName;
137        }
138
139        /** Return true if no branch is specified. */
140        public boolean isUnbranched() {
141                return NO_BRANCH_NAME.equals(this.branchName);
142        }
143
144        /** Returns whether this commit is a head timestamp */
145        public boolean isHeadCommit() {
146                return timestamp == Long.MAX_VALUE;
147        }
148
149        /**
150         * @see #timestamp
151         */
152        public long getTimestamp() {
153                return timestamp;
154        }
155
156        /** {@inheritDoc} */
157        @Override
158        public boolean equals(Object obj) {
159                if (obj instanceof CommitDescriptor) {
160                        CommitDescriptor other = (CommitDescriptor) obj;
161                        return other.timestamp == timestamp && Objects.equals(this.branchName, other.branchName);
162                }
163                return false;
164        }
165
166        /** {@inheritDoc} */
167        @Override
168        public int hashCode() {
169                return Objects.hashCode(branchName) ^ Long.hashCode(timestamp);
170        }
171
172        /** {@inheritDoc} */
173        @Override
174        public int compareTo(CommitDescriptor other) {
175                if (timestamp == other.timestamp) {
176                        return branchName.compareTo(other.branchName);
177                }
178                return Long.compare(timestamp, other.timestamp);
179        }
180
181        /** Converts this to a JSON representation. */
182        public String toJson() {
183                return JsonUtils.serializeToJSON(this);
184        }
185
186        /** Reads a {@link CommitDescriptor} from a JSON string. */
187        public static CommitDescriptor fromJson(String json) throws ConQATException {
188                return JsonUtils.deserializeFromJsonWithNullCheck(json, CommitDescriptor.class);
189        }
190
191        /** {@inheritDoc} */
192        @Override
193        public String toString() {
194                return branchName + "@" + timestamp;
195        }
196
197        /** Returns a timestamp+branchName key/byte[] representation. */
198        public byte[] toTimestampBranchKey() {
199                return ByteArrayUtils.concat(ByteArrayUtils.longToByteArray(getTimestamp()),
200                                StringUtils.stringToBytes(getBranchName()));
201        }
202
203        /** Parses a commit descriptor from its toString() representation. */
204        public static CommitDescriptor fromStringRepresentation(String representation) {
205                int separatorPosition = representation.lastIndexOf("@");
206                CCSMAssert.isTrue(separatorPosition >= 0,
207                                () -> "Invalid string representation of commit descriptor: " + representation);
208                String branch = representation.substring(0, separatorPosition);
209                String timestamp = representation.substring(separatorPosition + 1);
210                return new CommitDescriptor(branch, Long.parseLong(timestamp));
211        }
212
213        /** Parses a timestamp+branchName key/byte[] representation. */
214        public static CommitDescriptor fromTimestampBranchKey(byte[] key) {
215                long timestamp = ByteArrayUtils.byteArrayToLong(Arrays.copyOf(key, Long.BYTES));
216                String branchName = StringUtils.bytesToString(Arrays.copyOfRange(key, Long.BYTES, key.length));
217                return new CommitDescriptor(branchName, timestamp);
218        }
219
220        /**
221         * Returns a branchName+timestamp key/byte[] representation. NOTE: This is
222         * *DANGEROUS* since keys generated with this function may cause unwanted branch
223         * names (prefixes of wanted branch names) to be returned from store scans
224         * (TS-16367).
225         */
226        public byte[] toBranchTimestampKey() {
227                return ByteArrayUtils.concat(StringUtils.stringToBytes(getBranchName()),
228                                ByteArrayUtils.longToByteArray(getTimestamp()));
229        }
230
231        /** Parses a branchN+timestamp key/byte[] representation. */
232        public static CommitDescriptor fromBranchTimestampKey(byte[] key) {
233                String branchName = StringUtils.bytesToString(Arrays.copyOf(key, key.length - Long.BYTES));
234                long timestamp = ByteArrayUtils.byteArrayToLong(Arrays.copyOfRange(key, key.length - Long.BYTES, key.length));
235                return new CommitDescriptor(branchName, timestamp);
236        }
237
238        /** Creates a commit descriptor for the latest revision on a branch */
239        public static CommitDescriptor latestOnBranch(String branchName) {
240                return new CommitDescriptor(branchName, Long.MAX_VALUE);
241        }
242
243        /**
244         * Returns a format used for service calls ("branch:timestamp", or "timestamp"
245         * if no branch is specified).
246         */
247        public String toServiceCallFormat() {
248                String result = branchName + ":";
249                if (NO_BRANCH_NAME.contentEquals(branchName)) {
250                        result = StringUtils.EMPTY_STRING;
251                }
252
253                if (timestamp == Long.MAX_VALUE) {
254                        return result + HEAD_TIMESTAMP;
255                }
256                return result + timestamp;
257        }
258}
