076        /**
077         * Tries to find an existing bare or cloned repository along the given URI (with
078         * "file" scheme).
079         * 
080         * @throws RepositoryException
081         *             if no repository could be found
082         */
083        public static Repository getExistingRepository(URI localGitRepoURI) throws RepositoryException {
084                File localGitRepo = new File(localGitRepoURI);
085                FileRepositoryBuilder builder = new FileRepositoryBuilder();
086                File parentFile = localGitRepo.getParentFile();
087                if (parentFile != null) {
088                        builder.addCeilingDirectory(parentFile);
089                }
090                builder.findGitDir(localGitRepo);
091                try {
092                        if (builder.getGitDir() != null) {
093                                return builder.build();
094                        }
095                        Optional<Repository> bareRepo = searchForBareRepositoryAlongPath(localGitRepo);
096                        if (bareRepo.isPresent()) {
097                                return bareRepo.get();
098                        }
099                        throw new RepositoryException(REPOSITORY_NOT_FOUND_EXCEPTION_MESSAGE + localGitRepo.getPath());
100                } catch (IOException e) {
101                        throw new RepositoryException(REPOSITORY_NOT_FOUND_EXCEPTION_MESSAGE + localGitRepo.getPath(), e);
102                }
103        }
104
105        /**
106         * Retrieves an existing bare repository along the given path. If there is none,
107         * empty is returned.
108         */
109        private static Optional<Repository> searchForBareRepositoryAlongPath(File repoPath) {
110                while (repoPath != null) {
111                        Repository repo = openBareRepository(repoPath);
112                        if (repo != null) {
113                                return Optional.of(repo);
114                        }
115                        repoPath = repoPath.getParentFile();
116                }
117                return Optional.empty();
118        }
119
120        /**
121         * Tries to open a bare repository at the given path. If it could not be opened
122         * <code>null</code> is returned.
123         */
124        private static Repository openBareRepository(File repoPath) {
125                if (!repoPath.exists()) {
126                        return null;
127                }
128                try {
129                        return Git.open(repoPath).getRepository();
130                } catch (IOException e) {
131                        return null;
132                }
133        }

135        /**
136         * Sets up a repository by cloning it. If a cloned version already exists it
137         * will be reused.
138         */
139        public static Repository cloneAndSetUpRepository(File localDirectory, URI location,
140                        TeamscaleGitCredentialsProvider credentials) throws RepositoryException {
141                if (localGitCloneExists(localDirectory)) {
142                        return GitUtils.getExistingRepository(localDirectory.toURI());
143                }
144                try {
145
146                        CloneCommand cloneCommand = Git.cloneRepository();
147                        configureCommand(location, credentials, cloneCommand);
148                        return cloneCommand.setBare(true).setCredentialsProvider(credentials).setURI(location.toString())
149                                        .setTimeout(INITIAL_CLONE_OPERATION_TIMEOUT_SECONDS).setDirectory(localDirectory).call()
150                                        .getRepository();
151                } catch (GitAPIException e) {
152                        // if the cloning failed there is still an empty local repository
153                        // left which needs to be removed
154                        if (localDirectory.exists()) {
155                                FileSystemUtils.deleteRecursively(localDirectory);
156                        }
157                        throw new RepositoryException(e);
158                }
159        }

161        /**
162         * Configures the given command by adding ssh credentials and configuring a
163         * timeeout.
164         */
165        public static void configureCommand(URI location, TeamscaleGitCredentialsProvider credentials,
166                        TransportCommand<?, ?> command) {
167                command.setTransportConfigCallback(new TransportConfigCallback() {
168                        @Override
169                        public void configure(Transport transport) {
170                                transport.setTimeout(REMOTE_OPERATION_TIMEOUT_SECONDS);
171                                if (credentials != null && !StringUtils.isEmpty(credentials.getSshPrivateKey())) {
172                                        SshTransport sshTransport = (SshTransport) transport;
173                                        sshTransport.setSshSessionFactory(createSshSessionFactory(location, credentials));
174                                }
175                        }
176
177                });
178        }

180        private static SshSessionFactory createSshSessionFactory(URI location,
181                        TeamscaleGitCredentialsProvider credentials) {
182                if (EFeatureToggle.JGIT_SSH_JSCH.isEnabled()) {
183                        return new JschSshSessionFactory(toURIish(location), credentials);
184                }
185                return new ApacheMinaSshSessionFactory(toURIish(location), credentials);
186        }

220        /** Returns the commit denoted by the given commit id/tag/head. */
221        public static RevCommit getCommit(Repository repository, String revisionBranchOrTag) throws RepositoryException {
222                try (RevWalk revWalk = new RevWalk(repository)) {
223                        Ref head = repository.findRef(revisionBranchOrTag);
224                        if (head != null) {
225                                return revWalk.parseCommit(head.getLeaf().getObjectId());
226                        }
227                        return revWalk.parseCommit(ObjectId.fromString(revisionBranchOrTag));
228                } catch (IOException e) {
229                        throw new RepositoryException(e);
230                }
231        }

233        /**
234         * Returns the id of the object described by the given path in the given
235         * revision. If no object is found an empty optional is returned.
236         */
237        public static Optional<ObjectId> getId(Repository repository, RevCommit commit, String path)
238                        throws RepositoryException {
239                try (TreeWalk walk = TreeWalk.forPath(repository, path, commit.getTree())) {
240
241                        if (walk == null) {
242                                return Optional.empty();
243                        }
244
245                        return Optional.of(walk.getObjectId(0));
246                } catch (IOException e) {
247                        throw new RepositoryException(e);
248                }
249        }

251        /**
252         * Creates a {@link DiffFormatter} to retrieve changes from the repository
253         */
254        public static DiffFormatter createDiffFormatter(Repository repository) {
255                DiffFormatter diffFormatter = new DiffFormatter(DisabledOutputStream.INSTANCE);
256                diffFormatter.setRepository(repository);
257                diffFormatter.setDiffComparator(RawTextComparator.DEFAULT);
258                diffFormatter.setDetectRenames(true);
259                return diffFormatter;
260        }

262        /** Creates a {@link TreeWalk} instance for the given repository */
263        public static TreeWalk createTreeWalk(Repository repository) {
264                TreeWalk treeWalk = new TreeWalk(repository);
265                treeWalk.setFilter(TreeFilter.ANY_DIFF);
266                treeWalk.setRecursive(true);
267                return treeWalk;

278        /**
279         * Returns a timestamp for the given revision in the given repository. If there
280         * is no commit with the given id empty is returned.
281         */
282        public static Optional<Long> getTimestampFromRevision(Repository repository, String revision) {
283                try (RevWalk revWalk = new RevWalk(repository)) {
284                        ObjectId commitId = repository.resolve(revision);
285                        RevCommit commit = revWalk.parseCommit(commitId);
286                        return Optional.of(commit.getCommitterIdent().getWhen().getTime());
287                } catch (IOException e) {
288                        return Optional.empty();
289                }
290        }

293         * Returns whether the given revision is a valid commit hash or HEAD reference.

295        public static boolean isCommitHash(String revision) {
296                return revision.equals(HEAD) || COMMIT_HASH_PATTERN.matcher(revision).matches();

299        /**
300         * Basic credentials provider for the jgit library that automatically accepts
301         * requests whether we trust a source. This must be done, as we have no
302         * possibility to ask the user. Additionaly this class carries information about
303         * a possible git ssh key that should be used, so that it can later be used for
304         * configuration.
305         */
306        public static final class TeamscaleGitCredentialsProvider extends UsernamePasswordCredentialsProvider {
307
308                /**
309                 * The private key that should be used with these credentials. May be
310                 * <code>null</code>
311                 */
312                private final String sshPrivateKey;
313
314                private TeamscaleGitCredentialsProvider(String username, String password, String sshPrivateKey) {
315                        super(username, password);
316                        this.sshPrivateKey = sshPrivateKey;
317                }
318
319                @Override
320                public boolean supports(CredentialItem... items) {
321                        for (CredentialItem item : items) {
322                                if (item instanceof CredentialItem.YesNoType) {
323                                        continue;
324                                }
325                                if (!super.supports(item)) {
326                                        return false;
327                                }
328                        }
329                        return true;
330                }
331
332                @Override
333                public boolean get(URIish uri, CredentialItem... items) throws UnsupportedCredentialItem {
334                        for (CredentialItem item : items) {
335                                if (item instanceof CredentialItem.YesNoType) {
336                                        // Always answer the question about whether we trust a source to true, we cannot
337                                        // ask the user.
338                                        ((CredentialItem.YesNoType) item).setValue(true);
339                                        continue;
340                                }
341                                if (super.get(uri, item)) {
342                                        continue;
343                                }
344                        }
345                        return true;
346                }
347
348                /** @see #sshPrivateKey */
349                public String getSshPrivateKey() {
350                        return sshPrivateKey;
351                }
352        }

354        /**
355         * Because the git library does not support the git@ url format, we fix it by
356         * converting it to a plain ssh url
357         */
358        public static String rewriteGitAtUrl(String url) {
359                if (url.startsWith("git@")) {
360                        return "ssh://" + url.replaceFirst(":", "/");
361                }
362                return url;
363        }

365        /**
366         * Determines the change type for a given entry. This method respects the mode
367         * flags for the old and new file (if present) and uses them to override the
368         * stored mode. For example, changing a symlink to a plain file would be treated
369         * as an {@link ChangeType#ADD}. Returns {@link Optional#empty()} if the entry
370         * was and still is not an actual file and should hence be ignored entirely.
371         */
372        public static Optional<ChangeType> determineChangeType(DiffEntry diff) {
373                boolean wasActualFile = isActualFile(diff.getOldMode());
374                boolean isActualFile = isActualFile(diff.getNewMode());
375                if (wasActualFile && isActualFile) {
376                        return Optional.of(diff.getChangeType());
377                } else if (!wasActualFile && isActualFile) {
378                        return Optional.of(ChangeType.ADD);
379                } else if (wasActualFile && !isActualFile) {
380                        return Optional.of(ChangeType.DELETE);
381                }
382                return Optional.empty();
383        }

385        /**
386         * Returns whether a "file" with given file mode is an actual file and not,
387         * e.g., a symlink which is not tracked by Teamscale.
388         */
389        public static boolean isActualFile(FileMode mode) {
390                return isActualFile(mode.getBits());
391        }
392
393        /**
394         * Returns whether a "file" with the given mode bits is an actual file and not,
395         * e.g., a symlink which is not tracked by Teamscale.
396         */
397        public static boolean isActualFile(int modeBits) {
398                return FileMode.REGULAR_FILE.equals(modeBits) || FileMode.EXECUTABLE_FILE.equals(modeBits);
399        }
