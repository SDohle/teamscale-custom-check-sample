080        /**
081         * Tries to find an existing bare or cloned repository along the given URI (with
082         * "file" scheme).
083         * 
084         * @throws RepositoryException
085         *             if no repository could be found
086         */
087        public static Repository getExistingRepository(URI localGitRepoURI) throws RepositoryException {
088                File localGitRepo = new File(localGitRepoURI);
089                FileRepositoryBuilder builder = new FileRepositoryBuilder();
090                File parentFile = localGitRepo.getParentFile();
091                if (parentFile != null) {
092                        builder.addCeilingDirectory(parentFile);
093                }
094                builder.findGitDir(localGitRepo);
095                try {
096                        if (builder.getGitDir() != null) {
097                                return builder.build();
098                        }
099                        Optional<Repository> bareRepo = searchForBareRepositoryAlongPath(localGitRepo);
100                        if (bareRepo.isPresent()) {
101                                return bareRepo.get();
102                        }
103                        throw new RepositoryException(REPOSITORY_NOT_FOUND_EXCEPTION_MESSAGE + localGitRepo.getPath());
104                } catch (IOException e) {
105                        throw new RepositoryException(REPOSITORY_NOT_FOUND_EXCEPTION_MESSAGE + localGitRepo.getPath(), e);
106                }
107        }
108
109        /**
110         * Retrieves an existing bare repository along the given path. If there is none,
111         * empty is returned.
112         */
113        private static Optional<Repository> searchForBareRepositoryAlongPath(File repoPath) {
114                while (repoPath != null) {
115                        Repository repo = openBareRepository(repoPath);
116                        if (repo != null) {
117                                return Optional.of(repo);
118                        }
119                        repoPath = repoPath.getParentFile();
120                }
121                return Optional.empty();
122        }
123
124        /**
125         * Tries to open a bare repository at the given path. If it could not be opened
126         * <code>null</code> is returned.
127         */
128        private static Repository openBareRepository(File repoPath) {
129                if (!repoPath.exists()) {
130                        return null;
131                }
132                try {
133                        return Git.open(repoPath).getRepository();
134                } catch (IOException e) {
135                        return null;
136                }
137        }

139        /**
140         * Sets up a repository by cloning it. If a cloned version already exists it
141         * will be reused.
142         */
143        public static Repository cloneAndSetUpRepository(File localDirectory, URI location,
144                        TeamscaleGitCredentialsProvider credentials) throws RepositoryException {
145                if (localGitCloneExists(localDirectory)) {
146                        return GitUtils.getExistingRepository(localDirectory.toURI());
147                }
148                try {
149
150                        CloneCommand cloneCommand = Git.cloneRepository();
151                        configureCommand(location, credentials, cloneCommand);
152                        return cloneCommand.setBare(true).setCredentialsProvider(credentials).setURI(location.toString())
153                                        .setTimeout(INITIAL_CLONE_OPERATION_TIMEOUT_SECONDS).setDirectory(localDirectory).call()
154                                        .getRepository();
155                } catch (GitAPIException e) {
156                        // if the cloning failed there is still an empty local repository
157                        // left which needs to be removed
158                        if (localDirectory.exists()) {
159                                FileSystemUtils.deleteRecursively(localDirectory);
160                        }
161                        throw new RepositoryException(e);
162                }
163        }

165        /**
166         * Configures the given command by adding ssh credentials and configuring a
167         * timeeout.
168         */
169        public static void configureCommand(URI location, TeamscaleGitCredentialsProvider credentials,
170                        TransportCommand<?, ?> command) {
171                command.setTransportConfigCallback(new TransportConfigCallback() {
172                        @Override
173                        public void configure(Transport transport) {
174                                transport.setTimeout(REMOTE_OPERATION_TIMEOUT_SECONDS);
175                                if (credentials != null && !StringUtils.isEmpty(credentials.sshPrivateKey)) {
176                                        SshTransport sshTransport = (SshTransport) transport;
177                                        sshTransport.setSshSessionFactory(createSshSessionFactory(location, credentials));
178                                }
179                        }
180
181                });
182        }
183
184        private static SshSessionFactory createSshSessionFactory(URI location,
185                        TeamscaleGitCredentialsProvider credentials) {
186                SshSessionFactory sshSessionFactory = new JschConfigSessionFactory() {
187
188                        @Override
189                        protected JSch createDefaultJSch(FS fs) throws JSchException {
190                                JSch jsch = super.createDefaultJSch(fs);
191
192                                URIish wrappedUrl;
193                                try {
194                                        wrappedUrl = new URIish(location.toString());
195                                } catch (URISyntaxException e) {
196                                        throw new JSchException("Should never happen.", e);
197                                }
198                                Password password = new Password();
199                                credentials.get(wrappedUrl, password);
200                                jsch.addIdentity(location.toASCIIString(), credentials.sshPrivateKey.getBytes(), null,
201                                                new String(password.getValue()).getBytes());
202                                return jsch;
203                        }
204
205                        @Override
206                        protected void configure(Host hc, Session session) {
207                                // Nothing to do
208                        }
209
210                };
211                return sshSessionFactory;
212        }

238        /** Returns the commit denoted by the given commit id/tag/head. */
239        public static RevCommit getCommit(Repository repository, String revisionBranchOrTag) throws RepositoryException {
240                try (RevWalk revWalk = new RevWalk(repository)) {
241                        Ref head = repository.findRef(revisionBranchOrTag);
242                        if (head != null) {
243                                return revWalk.parseCommit(head.getLeaf().getObjectId());
244                        }
245                        return revWalk.parseCommit(ObjectId.fromString(revisionBranchOrTag));
246                } catch (IOException e) {
247                        throw new RepositoryException(e);
248                }
249        }

251        /**
252         * Returns the id of the object described by the given path in the given
253         * revision. If no object is found an empty optional is returned.
254         */
255        public static Optional<ObjectId> getId(Repository repository, RevCommit commit, String path)
256                        throws RepositoryException {
257                try (TreeWalk walk = TreeWalk.forPath(repository, path, commit.getTree())) {
258
259                        if (walk == null) {
260                                return Optional.empty();
261                        }
262
263                        return Optional.of(walk.getObjectId(0));
264                } catch (IOException e) {
265                        throw new RepositoryException(e);
266                }
267        }

269        /**
270         * Creates a {@link DiffFormatter} to retrieve changes from the repository
271         */
272        public static DiffFormatter createDiffFormatter(Repository repository) {
273                DiffFormatter diffFormatter = new DiffFormatter(DisabledOutputStream.INSTANCE);
274                diffFormatter.setRepository(repository);
275                diffFormatter.setDiffComparator(RawTextComparator.DEFAULT);
276                diffFormatter.setDetectRenames(true);
277                return diffFormatter;
278        }

280        /** Creates a {@link TreeWalk} instance for the given repository */
281        public static TreeWalk createTreeWalk(Repository repository) {
282                TreeWalk treeWalk = new TreeWalk(repository);
283                treeWalk.setFilter(TreeFilter.ANY_DIFF);
284                treeWalk.setRecursive(true);
285                return treeWalk;
286        }

296        /**
297         * Returns a timestamp for the given revision in the given repository. If there
298         * is no commit with the given id empty is returned.
299         */
300        public static Optional<Long> getTimestampFromRevision(Repository repository, String revision) {
301                try (RevWalk revWalk = new RevWalk(repository)) {
302                        ObjectId commitId = repository.resolve(revision);
303                        RevCommit commit = revWalk.parseCommit(commitId);
304                        return Optional.of(commit.getCommitterIdent().getWhen().getTime());
305                } catch (IOException e) {
306                        return Optional.empty();
307                }
308        }

310        /**
311         * Returns whether the given revision is a valid commit hash or HEAD reference.
312         */
313        public static boolean isCommitHash(String revision) {
314                return revision.equals(HEAD) || COMMIT_HASH_PATTERN.matcher(revision).matches();
315        }

317        /**
318         * Basic credentials provider for the jgit library that automatically accepts
319         * requests whether we trust a source. This must be done, as we have no
320         * possibility to ask the user. Additionaly this class carries information about
321         * a possible git ssh key that should be used, so that it can later be used for
322         * configuration.
323         */
324        public static final class TeamscaleGitCredentialsProvider extends UsernamePasswordCredentialsProvider {
325
326                /**
327                 * The private key that should be used with these credentials. May be
328                 * <code>null</code>
329                 */
330                private final String sshPrivateKey;
331
332                private TeamscaleGitCredentialsProvider(String username, String password, String sshPrivateKey) {
333                        super(username, password);
334                        this.sshPrivateKey = sshPrivateKey;
335                }
336
337                @Override
338                public boolean supports(CredentialItem... items) {
339                        for (CredentialItem item : items) {
340                                if (item instanceof CredentialItem.YesNoType) {
341                                        continue;
342                                }
343                                if (!super.supports(item)) {
344                                        return false;
345                                }
346                        }
347                        return true;
348                }
349
350                @Override
351                public boolean get(URIish uri, CredentialItem... items) throws UnsupportedCredentialItem {
352                        for (CredentialItem item : items) {
353                                if (item instanceof CredentialItem.YesNoType) {
354                                        // Always answer the question about whether we trust a source to true, we cannot
355                                        // ask the user.
356                                        ((CredentialItem.YesNoType) item).setValue(true);
357                                        continue;
358                                }
359                                if (super.get(uri, item)) {
360                                        continue;
361                                }
362                        }
363                        return true;
364                }
365
366                /** @see #sshPrivateKey */
367                public String getSshPrivateKey() {
368                        return sshPrivateKey;
369                }
370        }

372        /**
373         * Because the git library does not support the git@ url format, we fix it by
374         * converting it to a plain ssh url
375         */
376        public static String rewriteGitAtUrl(String url) {
377                if (url.startsWith("git@")) {
378                        return "ssh://" + url.replaceFirst(":", "/");
379                }
380                return url;
381        }
