017package org.conqat.engine.index.shared;
018
019import java.io.Serializable;
020
021import javax.annotation.Nullable;
022
023import org.conqat.lib.commons.js_export.ExportToJavaScript;
024
025import com.fasterxml.jackson.annotation.JsonCreator;
026import com.fasterxml.jackson.annotation.JsonProperty;
027
028/**
029 * Descriptor for a single project.
030 */
031@ExportToJavaScript
032public class ProjectInfo implements Serializable {
033
034        /** Serial version UID. */
035        private static final long serialVersionUID = 1;
036
037        /** The name of the JSON property name for {@link #id}. */
038        private static final String ID_PROPERTY = "id";
039
040        /** The name of the JSON property name for {@link #name}. */
041        private static final String NAME_PROPERTY = "name";
042
043        /** The name of the JSON property name for {@link #alias}. */
044        private static final String ALIAS_PROPERTY = "alias";
045
046        /** The name of the JSON property name for {@link #description}. */
047        private static final String DESCRIPTION_PROPERTY = "description";
048
049        /** The name of the JSON property name for {@link #creationTimestamp}. */
050        private static final String CREATION_TIMESTAMP_PROPERTY = "creationTimestamp";
051
052        /** ID of the project */
053        @JsonProperty(ID_PROPERTY)
054        private final String id;
055
056        /**
057         * An alias ID that can be used to access this project. This may be null if the
058         * project has no alias.
059         */
060        @JsonProperty(ALIAS_PROPERTY)
061        @Nullable
062        private String alias;
063
064        /** The project name. */
065        @JsonProperty(NAME_PROPERTY)
066        private String name;
067
068        /** The description, may be null. */
069        @JsonProperty(DESCRIPTION_PROPERTY)
070        @Nullable
071        private String description;
072
073        /** The creation time of the project. */
074        @JsonProperty(CREATION_TIMESTAMP_PROPERTY)
075        private long creationTimestamp;
076
077        /** Whether this project is in the process of being deleted. */
078        @JsonProperty("deleting")
079        private boolean deleting = false;
080
081        /** Whether this project is in the process of being re-analyzed. */
082        @JsonProperty("reanalyzing")
083        private boolean reanalyzing = false;
084
085        /** Constructor. */
086        @JsonCreator
087        public ProjectInfo(@JsonProperty(ID_PROPERTY) String id, @JsonProperty(NAME_PROPERTY) String name,
088                        @JsonProperty(ALIAS_PROPERTY) String alias, @JsonProperty(DESCRIPTION_PROPERTY) String description,
089                        @JsonProperty(CREATION_TIMESTAMP_PROPERTY) long creationTimestamp) {
090                this.id = id;
091                this.name = name;
092                this.alias = alias;
093                this.description = description;
094                this.creationTimestamp = creationTimestamp;
095        }
096
097        /** Returns the project id. */
098        public String getId() {
099                return id;
100        }
101
102        /** Returns {@link #alias}. */
103        public String getAlias() {
104                return alias;
105        }
106
107        /** Sets {@link #alias}. */
108        public void setAlias(String alias) {
109                this.alias = alias;
110        }
111
112        /**
113         * Returns the {@link #alias} if present. Otherwise the {@link #id} is returned.
114         * Can be used in case the REST API or UI code expects the alias to shadow the
115         * id.
116         */
117        public String getProjectAliasOrId() {
118                if (alias != null) {
119                        return alias;
120                }
121                return id;
122        }
123
124        /** Returns {@link #name}. */
125        public String getName() {
126                return name;
127        }
128
129        /** Sets {@link #name}. */
130        public void setName(String name) {
131                this.name = name;
132        }
133
134        /** Returns {@link #description}. */
135        public String getDescription() {
136                return description;
137        }
138
139        /** Sets {@link #description}. */
140        public void setDescription(String description) {
141                this.description = description;
142        }
143
144        /** Returns {@link #creationTimestamp}. */
145        public long getCreationTimestamp() {
146                return creationTimestamp;
147        }
148
149        /** Sets the creation timestamp. */
150        public void setCreationTimestamp(long creationTimestamp) {
151                this.creationTimestamp = creationTimestamp;
152        }
153
154        /** Returns {@link #deleting}. */
155        public boolean isDeleting() {
156                return deleting;
157        }
158
159        /** @see #isReanalyzing */
160        public boolean isReanalyzing() {
161                return reanalyzing;
162        }
163
164        /**
165         * Returns whether this project is currently in the process of either being
166         * reanalyzed or deleted.
167         */
168        public boolean isDeletingOrReanalyzing() {
169                return isDeleting() || isReanalyzing();
170        }
171
172        /** @see #deleting */
173        public void setDeleting(boolean deleting) {
174                this.deleting = deleting;
175        }
176
177        /** @see #isReanalyzing */
178        public void setReanalyzing(boolean reanalyzing) {
179                this.reanalyzing = reanalyzing;
180        }
181}
