018package org.conqat.engine.index.shared.tests;
019
020import java.io.Serializable;
021import java.time.Duration;
022import java.util.Arrays;
023import java.util.List;
024import java.util.Optional;
025import java.util.regex.Pattern;
026
027import javax.annotation.Nullable;
028
029import org.conqat.engine.sourcecode.coverage.TestUniformPathUtils;
030import org.conqat.lib.commons.assessment.Assessment;
031import org.conqat.lib.commons.assessment.ETrafficLightColor;
032import org.conqat.lib.commons.js_export.ExportToJavaScript;
033import org.conqat.lib.commons.string.StringUtils;
034import org.conqat.lib.commons.uniformpath.RelativeUniformPath;
035import org.conqat.lib.commons.uniformpath.UniformPath;
036
037import com.fasterxml.jackson.annotation.JsonProperty;
038
039/** Representation of a single test (method) execution. */
040@ExportToJavaScript
041public class TestExecution implements Serializable {
042
043        private static final long serialVersionUID = 1L;
044
045        /**
046         * The uniform path of the test (method) that was executed. This is an absolute
047         * (i.e. hierarchical) reference which identifies the test uniquely in the scope
048         * of the Teamscale project. It may (but is not required to) correspond to the
049         * path of some automated test case source code known to Teamscale. If the test
050         * was parameterized, this path is expected to reflect the parameter in some
051         * manner.
052         */
053        @JsonProperty("uniformPath")
054        private final String uniformPath;
055
056        /** Duration of the execution in seconds. */
057        @JsonProperty("durationSeconds")
058        private final double durationSeconds;
059
060        /** The actual execution result state. */
061        @JsonProperty("result")
062        private final ETestExecutionResult result;
063
064        /**
065         * Optional message given for test failures (normally contains a stack trace).
066         * For skipped and ignored tests the message will hold the message of skipped
067         * and ignored tests. e.g. "Flickers see TS-12345". May be {@code null}.
068         */
069        @JsonProperty("message")
070        @Nullable
071        private final String message;
072
073        private TestExecution(RelativeUniformPath uniformPath, double durationSeconds, ETestExecutionResult result,
074                        String message) {
075                this.uniformPath = uniformPath.toString();
076                this.durationSeconds = durationSeconds;
077                this.result = result;
078                this.message = message;
079        }
080
081        /** @see #uniformPath */
082        public String getUniformPath() {
083                return uniformPath;
084        }
085
086        /** @see #durationSeconds */
087        public double getDurationMillis() {
088                return durationSeconds * 1000.0;
089        }
090
091        /** @see #durationSeconds */
092        public double getDurationSeconds() {
093                return durationSeconds;
094        }
095
096        /** @see #result */
097        public ETestExecutionResult getResult() {
098                return result;
099        }
100
101        /** @see #message */
102        public Optional<String> getMessage() {
103                return Optional.ofNullable(message);
104        }
105
106        /** {@inheritDoc} */
107        @Override
108        public String toString() {
109                return "TestExecution [name=" + uniformPath + ", durationSeconds=" + durationSeconds + ", result=" + result
110                                + ", message=" + message + "]";
111        }
112
113        /**
114         * Returns whether something went wrong during execution of this test (either
115         * direct test failure or error during execution).
116         */
117        public boolean isFailure() {
118                return result == ETestExecutionResult.ERROR || result == ETestExecutionResult.FAILURE;
119        }
120
121        /** Create an assessment for this test result. */
122        public Assessment createAssessment() {
123                switch (result) {
124                case PASSED:
125                        return new Assessment(ETrafficLightColor.GREEN);
126                case IGNORED:
127                        return new Assessment(ETrafficLightColor.YELLOW);
128                case SKIPPED:
129                        return new Assessment(ETrafficLightColor.YELLOW);
130                case ERROR:
131                        return new Assessment(ETrafficLightColor.RED);
132                case FAILURE:
133                        return new Assessment(ETrafficLightColor.RED);
134                default:
135                        return new Assessment(ETrafficLightColor.UNKNOWN);
136                }
137        }
138
139        /** Builder for creating {@link TestExecution} instances. */
140        public static class Builder {
141                private static final Pattern UNESCAPED_SLASH = Pattern.compile("(?<!\\\\)/");
142
143                @JsonProperty("uniformPath")
144                private RelativeUniformPath uniformPath;
145                @JsonProperty("durationInSeconds")
146                private double durationInSeconds = 0;
147                @JsonProperty("result")
148                private ETestExecutionResult result;
149                @JsonProperty("failureMessage")
150                private String failureMessage;
151
152                private Builder(RelativeUniformPath uniformPath) {
153                        this.uniformPath = uniformPath;
154                }
155
156                /**
157                 * Creates a builder for a fully qualified test class name and test method name.
158                 * This also supports (unescaped) slashes as separator.
159                 */
160                public static Builder fromClassAndName(String fullyQualifiedClassName, String testName) {
161                        // replace all slashes by a dot, so both dots and slashes build hierarchy
162                        fullyQualifiedClassName = UNESCAPED_SLASH.matcher(fullyQualifiedClassName).replaceAll(".");
163                        return fromPathAndName(Arrays.asList(fullyQualifiedClassName.split("\\.")), testName);
164                }
165
166                /**
167                 * Creates a builder for a path and test name.
168                 */
169                public static Builder fromPathAndName(String path, String testName) {
170                        return fromPathAndName(StringUtils.splitWithEscapeCharacter(path, "/"), testName);
171                }
172
173                /**
174                 * Creates a builder for a relative uniform path.
175                 */
176                public static Builder fromPath(RelativeUniformPath path) {
177                        return new Builder(path);
178                }
179
180                /**
181                 * Creates a builder for a path and test name.
182                 */
183                private static Builder fromPathAndName(List<String> pathSegments, String testName) {
184                        return fromPath(RelativeUniformPath.of(pathSegments).addSuffix(UniformPath.escapeSegment(testName)));
185                }
186
187                /** Sets the failure message. Trims leading whitespace on each line. */
188                public Builder setFailureMessage(String failureMessage) {
189                        if (failureMessage != null) {
190                                failureMessage = StringUtils.removeWhitespaceAtBeginningOfLine(failureMessage.trim());
191                                if (!failureMessage.isEmpty()) {
192                                        this.failureMessage = failureMessage;
193                                }
194                        }
195                        return this;
196                }
197
198                /** Sets the test duration. */
199                public Builder setDuration(Duration duration) {
200                        this.durationInSeconds = duration.toMillis() / 1000d;
201                        return this;
202                }
203
204                /** Sets the duration. */
205                public Builder setDurationInSeconds(double durationInSeconds) {
206                        this.durationInSeconds = durationInSeconds;
207                        return this;
208                }
209
210                /** Sets the execution result. */
211                public Builder setResult(ETestExecutionResult result) {
212                        this.result = result;
213                        return this;
214                }
215
216                /** Creates a new {@link TestExecution} instance from this builder. */
217                public TestExecution build() {
218                        return new TestExecution(uniformPath, durationInSeconds, result, failureMessage);
219                }
220
221                /**
222                 * Returns the absolute uniform path of this test execution including test
223                 * namespace/path and test name.
224                 */
225                public RelativeUniformPath getAbsolutePath() {
226                        return uniformPath;
227                }
228        }
229
230        /** Compute the test execution path. */
231        public UniformPath toUniformPath() {
232                return TestUniformPathUtils.convertToUniformPath(uniformPath);
233        }
234}
