017package org.conqat.engine.service.shared;
018
019import java.net.MalformedURLException;
020import java.net.URI;
021import java.net.URISyntaxException;
022import java.net.URL;
023import java.util.Optional;
024import java.util.function.Function;
025
026import org.conqat.lib.commons.string.StringUtils;
027
028import com.google.common.net.UrlEscapers;
029
030/**
031 * Utility methods for services.
032 */
033public class ServiceUtils {
034
035        /** Encodes the given URL (but not slashes). */
036        public static String encodeURL(String url) {
037                return UrlEscapers.urlFragmentEscaper().escape(url);
038        }
039
040        /** Uses Java API to check if the server address is valid */
041        public static boolean isValidServerAddress(String address) {
042                try {
043                        // Check whether the given address is both a valid URL and a valid
044                        // URI.
045                        new URL(address);
046                        new URI(address);
047                } catch (URISyntaxException e) {
048                        return false;
049                } catch (MalformedURLException e) {
050                        return false;
051                }
052                return true;
053        }
054
055        /**
056         * Produces a specific error message if the identifier is invalid (e.g.,
057         * contains linebreaks, invisible unicode chars, or is empty). The error message
058         * contains the (truncated) text if it is not empty.
059         * 
060         * @param identifierText
061         *            the text of the identifier (e.g., "x" or "foo")
062         * @param identifierDescription
063         *            a description of what this identifier should be (e.g., "Group
064         *            name" or "Id"). This is used at the beginning of the error message
065         *            and therefore should start with an uppercase letter.
066         */
067        public static Optional<String> getErrorMessageForInvalidIdentifiers(String identifierText,
068                        String identifierDescription) {
069                Function<String, String> truncate = identifier -> "\""
070                                + StringUtils.removeAll(StringUtils.truncate(identifier, 9, "..."), "\n") + "\"";
071                if (identifierText.contains("\n")) {
072                        return Optional.of(identifierDescription + " " + truncate.apply(identifierText) + " contains linebreaks");
073                }
074                if (identifierText.matches(".*\\p{C}.*")) {
075                        return Optional.of(identifierDescription + " " + truncate.apply(identifierText)
076                                        + " contains invisible Unicode control characters");
077                }
078                if (identifierText.equals(StringUtils.EMPTY_STRING)) {
079                        return Optional.ofNullable(identifierDescription + " is empty");
080                }
081                return Optional.empty();
082        }
083
084        /** Ensures that the given URL ends in a slash. */
085        public static String padSlash(String url) {
086                if (!url.endsWith("/")) {
087                        url += "/";
088                }
089                return url;
090        }
091}
