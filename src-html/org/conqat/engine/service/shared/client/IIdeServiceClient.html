017package org.conqat.engine.service.shared.client;
018
019import java.util.List;
020
021import org.conqat.engine.index.shared.CommitDescriptor;
022import org.conqat.engine.index.shared.ContainerInfo;
023import org.conqat.engine.index.shared.TrackedFinding;
024import org.conqat.engine.service.shared.data.BranchesInfo;
025import org.conqat.engine.service.shared.data.FindingBlacklistInfo;
026import org.conqat.engine.service.shared.data.ProjectMapping;
027import org.conqat.engine.service.shared.data.ServiceApiInfo;
028
029/**
030 * An interface for methods used by the IDE service client
031 */
032public interface IIdeServiceClient {
033
034        /**
035         * Retrieves the branches of the specified project
036         */
037        BranchesInfo getBranches(String projectName, ServerDetails serverDetails) throws ServiceCallException;
038
039        /**
040         * Fetches the content of this resource.
041         *
042         * @return content or null if there is no content on the server.
043         */
044        String getContent(CommitDescriptor commitDescriptor, String uniformPath, String projectName,
045                        ServerDetails serverDetails) throws ServiceCallException;
046
047        /**
048         * Service call to retrieve container info.
049         *
050         * @return <code>null</code> if uniformPath is not an existing directory inside
051         *         the project.
052         */
053        ContainerInfo getContainerInfo(String uniformPath, String projectName, CommitDescriptor commitDescriptor,
054                        ServerDetails serverDetails) throws ServiceCallException;
055
056        /**
057         * Returns the findings for the element with the given uniform path starting
058         * from a baseline or <code>null</code> if the uniform path does not exist.
059         */
060        List<TrackedFinding> getFindings(String uniformPath, String projectName, String baseline,
061                        CommitDescriptor commitDescriptor, boolean includeChangedCodeFindings, ServerDetails serverDetails)
062                        throws ServiceCallException;
063
064        /**
065         * Returns the findings for the given uniform path and all its sub paths
066         * starting from baseline or <code>null</code> if the uniform path does not
067         * exist.
068         */
069        List<TrackedFinding> getFindingsRecursive(String uniformPath, String projectName, String baseline,
070                        boolean includeChangedCodeFindings, ServerDetails serverDetails, CommitDescriptor commitDescriptor,
071                        boolean fetchAll) throws ServiceCallException;
072
073        /**
074         * Blacklists a finding in a project.
075         *
076         * @return String success if blacklisting was successful otherwise an exception
077         *         is thrown
078         */
079        String addFindingBlacklistInfo(String projectName, FindingBlacklistInfo findingBlacklistInfo, String branchName,
080                        ServerDetails serverDetails) throws ServiceCallException;
081
082        /**
083         * Service call to retrieve projects from a server.
084         *
085         * @return A String array with the project names on the given server or
086         *         <code>null</code> if the server url is not found (404).
087         */
088        String[] getProjectList(ServerDetails serverDetails) throws ServiceCallException;
089
090        /**
091         * Retrieves the Service API info from a running Teamscale instance.
092         *
093         * @param serverDetails
094         *            Server configuration details.
095         * @return
096         */
097        default ServiceApiInfo getServiceAPIInfo(ServerDetails serverDetails) throws ServiceCallException {
098                return new ServiceApiInfo(EServiceApiVersion.CURRENT_VERSION.getVersionNumber(),
099                                EServiceApiVersion.VERSION_1.getVersionNumber(), "");
100        }
101
102        /**
103         * Calculates project mappings for the given project.
104         *
105         * @param serverDetails
106         *            Details of some Teamscale server
107         * @param paths
108         *            List of relative file paths
109         * @param teamscaleProject
110         *            Some project on server
111         * @param branchName
112         *            Name of current VCS branch
113         */
114        default ProjectMapping[] calculateProjectMappings(ServerDetails serverDetails, List<String> paths,
115                        String teamscaleProject, String branchName) throws ServiceCallException {
116                return new ProjectMapping[0];
117        }
118}
