037        /** The id/url of com.teamscale.service.findings.FindingBlacklistService. */
038        public static final String FINDING_BLACKLIST_SERVICE_NAME = "finding-blacklist";
039
040        /**
041         * The factor used for extending the timeout when fetching findings.
042         * <p>
043         * It has to be a bit longer than usual (twice as long), because even though the
044         * server will only send back a limited number of findings, it first has to load
045         * all of them from the database for filtering purposes. This can take a long
046         * time when there is a large number of findings for a given path.
047         */
048        private static final int GET_FINDINGS_TIMEOUT_FACTOR = 2;
049
050        /**
051         * After how many characters do we want to cut off the path parameter of a
052         * mappings request in case the server only supports GET (instead of POST)
053         */
054        private static final int MAPPINGS_GET_REQUEST_PATH_CUTOFF_LENGTH = 6000;
055

093        @Override
094        @SuppressWarnings({ "unchecked" })
095        public List<TrackedFinding> getFindings(String uniformPath, String projectIdOrAlias, String baseline,
096                        CommitDescriptor commitDescriptor, boolean includeChangedCodeFindings, ServerDetails serverDetails)
097                        throws ServiceCallException {
098                ListMap<String, String> options = ListMap.of(ServiceClientConstants.RECURSIVE_PARAMETER,
099                                Boolean.FALSE.toString());
100                if (commitDescriptor != null) {
101                        options.add(ServiceClientConstants.TIMESTAMP_PARAMETER_NAME, commitDescriptor.toServiceCallFormat());
102                }
103                if (baseline != null) {
104                        options.add(ServiceClientConstants.BASELINE_PARAMETER, baseline);
105                        options.add(ServiceClientConstants.INCLUDE_CHANGED_CODE_FINDINGS_PARAMETER,
106                                        Boolean.toString(includeChangedCodeFindings));
107                }
108                serverDetails = serverDetails.withTimeOut(serverDetails.getTimeoutSeconds() * GET_FINDINGS_TIMEOUT_FACTOR);
109                return ServiceClientCallable.withHttpClient(getHttpClient(serverDetails),
110                                ServiceClientCallable.get(List.class, ServiceClientCallable.getLegacyApiUri("findings", serverDetails,
111                                                projectIdOrAlias, uniformPath, options)));
112        }

116        public List<TrackedFinding> getFindingsRecursive(String uniformPath, String projectIdOrAlias, String baseline,
117                        boolean includeChangedCodeFindings, ServerDetails serverDetails, CommitDescriptor commitDescriptor,
118                        boolean fetchAll) throws ServiceCallException {
119                ListMap<String, String> options = ListMap.of(ServiceClientConstants.RECURSIVE_PARAMETER,
120                                Boolean.TRUE.toString());
121                if (commitDescriptor != null) {
122                        options.add(ServiceClientConstants.TIMESTAMP_PARAMETER_NAME, commitDescriptor.toServiceCallFormat());
123                }
124                if (baseline != null) {

126                        options.add(ServiceClientConstants.INCLUDE_CHANGED_CODE_FINDINGS_PARAMETER,
127                                        String.valueOf(includeChangedCodeFindings));
128                }
129                if (fetchAll) {
130                        options.add("all", "true");
131                }
132                serverDetails = serverDetails.withTimeOut(serverDetails.getTimeoutSeconds() * GET_FINDINGS_TIMEOUT_FACTOR);
133                return ServiceClientCallable.withHttpClient(getHttpClient(serverDetails),
134                                ServiceClientCallable.get(List.class, ServiceClientCallable.getLegacyApiUri("findings", serverDetails,
135                                                projectIdOrAlias, uniformPath, options)));
136        }

144        @Override
145        public String addFindingBlacklistInfo(String projectIdOrAlias, FindingBlacklistInfo findingBlacklistInfo,
146                        String branchName, ServerDetails serverDetails) throws ServiceCallException {
147                int apiVersion = getServiceAPIInfo(serverDetails).getApiVersion();
148                if (apiVersion <= EServiceApiVersion.VERSION_6.getVersionNumber()) {
149                        // old server which does not know about blacklist types
150                        findingBlacklistInfo = findingBlacklistInfo.cloneWithNewType(null);
151                }
152                String url = ServiceClientUtils.getServiceUrl(serverDetails, projectIdOrAlias, FINDING_BLACKLIST_SERVICE_NAME);
153                url += findingBlacklistInfo.getFindingId();
154                url += ServiceClientUtils.createOptionString("t",
155                                CommitDescriptor.latestOnBranch(branchName).toServiceCallFormat());
156                try (CloseableHttpClient httpClient = ServiceClientUtils.getHttpClient(serverDetails)) {
157                        return ServiceClientUtils.executeRequest(httpClient,
158                                        ServiceClientUtils.createPut(url, findingBlacklistInfo), String.class);
159                } catch (IOException e) {
160                        throw new ServiceCallException(ServiceClientUtils.PROBLEM_CLOSING_HTTP_CLIENT_MESSAGE, e);
161                }
162        }

173        /**
174         * Tries to calculate the project and prefix mapping for the given files and
175         * optionally project.
176         * <p>
177         * To prevent 214: URI too large Errors the paths are normally sent via POST
178         * request. In case the server does not support this (APIVersion < 7), it
179         * collects as many paths as fit into MAPPINGS_GET_REQUEST_PATH_CUTOFF_LENGTH
180         * and then sends them via GET
181         *
182         * @param serverDetails
183         *            Server connection information
184         * @param paths
185         *            A list of relative file paths
186         * @param projectIdOrAlias
187         *            The project id for which to search the mapping. Can be
188         *            <code>null</code> if the project is not known.
189         * @param branch
190         *            The repository branch name.
191         */
192        @Override
193        public ProjectMapping[] calculateProjectMappings(ServerDetails serverDetails, List<String> paths,
194                        String projectIdOrAlias, String branch) throws ServiceCallException {
195                ListMap<String, String> options = new ListMap<>();
196                if (!StringUtils.isEmpty(projectIdOrAlias)) {
197                        options.add("project", projectIdOrAlias);
198                }
199                if (!StringUtils.isEmpty(branch)) {
200                        options.add("branch", branch);
201                }
202
203                String pathSeparator = ",";
204                String joinedPaths = String.join(pathSeparator, paths);
205
206                // old server can only handle GET requests
207                int apiVersion = getServiceAPIInfo(serverDetails).getApiVersion();
208                if (apiVersion < EServiceApiVersion.VERSION_7.getVersionNumber()) {
209                        // to prevent '414: URI too large' we cut off long paths
210                        if (joinedPaths.length() > MAPPINGS_GET_REQUEST_PATH_CUTOFF_LENGTH) {
211                                int lastCommaBeforeCutoff = joinedPaths.lastIndexOf(pathSeparator,
212                                                MAPPINGS_GET_REQUEST_PATH_CUTOFF_LENGTH);
213                                joinedPaths = joinedPaths.substring(0, lastCommaBeforeCutoff);
214                        }
215                        options.add("file", joinedPaths);
216
217                        return ServiceClientCallable.withHttpClient(getHttpClient(serverDetails),
218                                        ServiceClientCallable.get(ProjectMapping[].class, ServiceClientCallable.getLegacyGlobalApiUri(
219                                                        "project-mapping", serverDetails, Pair.createPair("file", joinedPaths))));
220                }
221
222                String url = ServiceClientUtils.getServiceUrl(serverDetails, null, "project-mapping");
223                String optionsString = ServiceClientUtils.createOptionString(options);
224
225                try (CloseableHttpClient httpClient = getHttpClient(serverDetails)) {
226                        HttpPost post = ServiceClientUtils.createPostForJsonContent(url + optionsString, joinedPaths);
227                        return ServiceClientUtils.executeRequest(httpClient, post, ProjectMapping[].class);
228                } catch (IOException e) {
229                        throw new ServiceCallException(ServiceClientUtils.PROBLEM_CLOSING_HTTP_CLIENT_MESSAGE, e);
230                }
231        }

269        @Override
270        public Set<String> getFileExtensionsForProject(String projectName, ServerDetails serverDetails)
271                        throws ServiceCallException {
272
273                int apiVersion = getServiceAPIInfo(serverDetails).getApiVersion();
274                if (apiVersion <= EServiceApiVersion.VERSION_7.getVersionNumber()) {
275                        return Collections.emptySet();
276                }
277
278                try {
279                        String[] extensions = JsonUtils
280                                        .deserializeFromJson(
281                                                        ServiceClientCallable.withHttpClient(getHttpClient(serverDetails),
282                                                                        ServiceClientCallable.get(String.class, ServiceClientCallable.getNewApiUri(
283                                                                                        "file-extensions", serverDetails, projectName, "", ListMap.of()))),
284                                                        String[].class);
285                        return CollectionUtils.asHashSet(extensions);
286                } catch (ConQATException e) {
287                        throw new ServiceCallException("Could not deserialize JSON", e);
288                }
289        }

291        /**
292         * Returns all commits that were found for the given revision. An empty list is
293         * returned if the revision was not found, more than one commit might be
294         * returned in case of multiple repositories.
295         */
296        @SuppressWarnings("unchecked")
297        public List<CommitDescriptor> getCommitsForRevision(String projectIdOrAlias, ServerDetails serverDetails,
298                        String revision) throws ServiceCallException {
299                List<CommitDescriptor> commits = ServiceClientCallable.withHttpClient(getHttpClient(serverDetails),
300                                ServiceClientCallable.get(List.class, ServiceClientCallable.getLegacyApiUri(
301                                                "repository-timestamp-by-revision", serverDetails, projectIdOrAlias, revision, ListMap.of())));
302                if (commits == null) {
303                        return Collections.emptyList();
304                }
305                return commits;
306        }
