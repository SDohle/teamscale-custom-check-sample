001package org.conqat.engine.service.shared.client;
002
003import java.io.IOException;
004
005import org.apache.http.client.HttpClient;
006import org.apache.http.client.methods.HttpGet;
007import org.apache.http.impl.client.CloseableHttpClient;
008import org.conqat.lib.commons.collections.ListMap;
009
010/**
011 * Interface to enable testing service calls with provided responses. For an
012 * example see
013 * {@link org.conqat.engine.service.shared.client.IdeServiceClientTest}.
014 */
015@FunctionalInterface
016public interface ServiceClientCallable<T> {
017
018        /**
019         * Helper method for starting a service call using a given
020         * {@link CloseableHttpClient}
021         */
022        static <T> T withHttpClient(CloseableHttpClient client, ServiceClientCallable<T> callable)
023                        throws ServiceCallException {
024                try (CloseableHttpClient c = client) {
025                        return callable.call(c);
026                } catch (IOException e) {
027                        throw new ServiceCallException(
028                                        ServiceClientUtils.PROBLEM_CLOSING_HTTP_CLIENT_MESSAGE.concat(": " + e.getMessage()), e);
029                }
030        }
031
032        /** Overridden by implementors, actually starts the service call */
033        T call(HttpClient client) throws ServiceCallException;
034
035        /**
036         * Equivalent of
037         * {@link ServiceClientUtils#uniformPathGetCall(String, Class, String, String, ServerDetails, ListMap)}
038         */
039        static <R> ServiceClientCallable<R> uniformPathGetCall(String serviceName, Class<R> resultClass, String uniformPath,
040                        String projectName, ServerDetails serverDetails, ListMap<String, String> options) {
041                return client -> {
042                        HttpGet httpGet = new HttpGet(ServiceClientUtils.getServiceUrl(serverDetails, projectName, serviceName)
043                                        + ServiceClientUtils.encodeUniformPath(uniformPath)
044                                        + ServiceClientUtils.createOptionString(options));
045
046                        return ServiceClientUtils.executeRequest(client, httpGet, resultClass);
047                };
048        }
049
050        /**
051         * Equivalent of
052         * {@link ServiceClientUtils#uniformPathGetCall(String, Class, String, String, ServerDetails, ListMap)}
053         * without any options.
054         */
055        static <R> ServiceClientCallable<R> uniformPathGetCall(String serviceName, Class<R> resultClass, String uniformPath,
056                        String projectName, ServerDetails serverDetails) {
057                return uniformPathGetCall(serviceName, resultClass, uniformPath, projectName, serverDetails, new ListMap<>());
058        }
059}
