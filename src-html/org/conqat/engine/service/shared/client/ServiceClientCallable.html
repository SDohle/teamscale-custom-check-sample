001package org.conqat.engine.service.shared.client;
002
003import java.io.IOException;
004
005import org.apache.http.client.HttpClient;
006import org.apache.http.client.methods.HttpGet;
007import org.apache.http.impl.client.CloseableHttpClient;
008import org.conqat.engine.service.shared.ServiceUtils;
009import org.conqat.lib.commons.collections.ListMap;
010import org.conqat.lib.commons.collections.Pair;
011import org.conqat.lib.commons.string.StringUtils;
012
013/**
014 * Interface to enable testing service calls with provided responses. For an
015 * example see
016 * {@link org.conqat.engine.service.shared.client.IdeServiceClientTest}.
017 */
018@FunctionalInterface
019public interface ServiceClientCallable<T> {
020
021        /** Overridden by implementors, actually starts the service call */
022        T call(HttpClient client) throws ServiceCallException;
023
024        /**
025         * Helper method for starting a service call using a given
026         * {@link CloseableHttpClient}
027         */
028        static <T> T withHttpClient(CloseableHttpClient client, ServiceClientCallable<T> callable)
029                        throws ServiceCallException {
030                try (CloseableHttpClient c = client) {
031                        return callable.call(c);
032                } catch (IOException e) {
033                        throw new ServiceCallException(
034                                        ServiceClientUtils.PROBLEM_CLOSING_HTTP_CLIENT_MESSAGE.concat(": " + e.getMessage()), e);
035                }
036        }
037
038        /**
039         * @return a {@link ServiceClientCallable} for a HTTP GET on the given URI.
040         */
041        static <R> ServiceClientCallable<R> get(Class<R> resultClass, String uri) {
042                return client -> {
043                        HttpGet httpGet = new HttpGet(uri);
044                        return ServiceClientUtils.executeRequest(client, httpGet, resultClass);
045                };
046        }
047
048        static String getLegacyApiUri(String serviceName, ServerDetails serverDetails, String projectName,
049                        String uniformPath, ListMap<String, String> options) {
050                return ServiceClientUtils.getServiceUrl(serverDetails, projectName, serviceName) //
051                                + ServiceUtils.encodeUniformPath(uniformPath) //
052                                + ServiceClientUtils.createOptionString(options);
053        }
054
055        static String getNewApiUri(String serviceName, ServerDetails serverDetails, String projectName, String uniformPath,
056                        ListMap<String, String> options) {
057                return StringUtils.ensureEndsWith(serverDetails.getUrl(), "/") //
058                                + "api/projects/" + projectName + "/" //
059                                + serviceName + "/" //
060                                + ServiceUtils.encodeUniformPath(uniformPath) //
061                                + ServiceClientUtils.createOptionString(options);
062        }
063
064        static String getLegacyGlobalApiUri(String serviceName, ServerDetails serverDetails, Pair<?, ?>... options) {
065                return ServiceClientUtils.getServiceUrl(serverDetails, null, serviceName) //
066                                + ServiceClientUtils.createOptionString(toOptionsMap(options));
067        }
068
069        static String getLegacyApiUri(String serviceName, ServerDetails serverDetails, String projectName,
070                        Pair<?, ?>... options) {
071                return ServiceClientUtils.getServiceUrl(serverDetails, projectName, serviceName) //
072                                + ServiceClientUtils.createOptionString(toOptionsMap(options));
073        }
074
075        /**
076         * @return the given options as a {@link ListMap} or {@code null}, if no options
077         *         are given.
078         */
079        static ListMap<String, String> toOptionsMap(Pair<?, ?>... options) {
080                if (options.length == 0) {
081                        return null;
082                }
083                ListMap<String, String> optionsMap = new ListMap<>();
084                for (Pair<?, ?> option : options) {
085                        optionsMap.add(option.getFirst().toString(), option.getSecond().toString());
086                }
087                return optionsMap;
088        }
089}
