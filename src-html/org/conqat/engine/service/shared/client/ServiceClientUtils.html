094        /**
095         * Performs a simple call using GET without a uniform path but with optional
096         * options.
097         */
098        public static <T> T simpleGetCall(String serviceName, Class<T> resultClass, String projectName,
099                        ServerDetails serverDetails, Pair<?, ?>... options) throws ServiceCallException {
100                ListMap<String, String> optsNew = null;
101                if (options.length > 0) {
102                        optsNew = new ListMap<>();
103                        for (Pair<?, ?> option : options) {
104                                optsNew.add(option.getFirst().toString(), option.getSecond().toString());
105                        }
106                }
107                HttpGet httpGet = new HttpGet(
108                                getServiceUrl(serverDetails, projectName, serviceName) + createOptionString(optsNew));
109                return performCall(resultClass, serverDetails, httpGet);
110        }

191        /**
192         * Performs a simple blocking call with the given request and uniform path that
193         * is appended to the URL. Returns the element or null.
194         */
195        public static <T> T performCall(Class<T> resultClass, ServerDetails serverDetails, HttpRequestBase httpRequest)
196                        throws ServiceCallException {
197                try (CloseableHttpClient client = getHttpClient(serverDetails)) {
198                        return executeRequest(client, httpRequest, resultClass);
199                } catch (IOException e) {
200                        throw new ServiceCallException(PROBLEM_CLOSING_HTTP_CLIENT_MESSAGE.concat(": " + e.getMessage()), e);
201                }
202        }

207         * values, where each odd numbered parameter is a parameter name and each even
208         * one is the corresponding value.

214        /**
215         * Creates an encoded URL options string starting with <code>?</code> from the
216         * given options.
217         */
218        public static String createOptionString(ListMap<String, String> options) {
219                StringBuilder sb = new StringBuilder();
220                if (options != null) {
221                        String separator = "?";
222                        for (String option : options.getKeys()) {
223                                List<String> values = options.getCollection(option);
224                                for (String value : values) {
225                                        value = ServiceUtils.encodeQueryParameter(value);
226                                        sb.append(separator).append(option).append('=').append(value);
227                                        separator = "&";
228                                }
229                        }
230                }
231                return sb.toString();
232        }

246        /**
247         * Sets up the HTTP client and returns it. The caller of this method is
248         * responsible for closing the client when it is no longer needed.
249         */
250        public static CloseableHttpClient getHttpClient(ServerDetails serverDetails) throws ServiceCallException {
251                try {
252                        SSLContext sslContext = SSLContexts.createDefault();
253                        sslContext.init(null, new TrustManager[] { new TrustAllCertificatesManager() }, new SecureRandom());
254                        SSLConnectionSocketFactory socketFactory = new SSLConnectionSocketFactory(sslContext);
255                        CredentialsProvider credentialsProvider = new BasicCredentialsProvider();
256                        credentialsProvider.setCredentials(AuthScope.ANY,
257                                        new UsernamePasswordCredentials(serverDetails.getUsername(), serverDetails.getPassword()));
258                        RequestConfig.Builder requestBuilder = RequestConfig.custom();
259                        int timeoutMilliseconds = serverDetails.getTimeoutSeconds() * 1000;
260                        requestBuilder = requestBuilder.setConnectTimeout(timeoutMilliseconds);
261                        requestBuilder = requestBuilder.setSocketTimeout(timeoutMilliseconds);
262                        requestBuilder = requestBuilder.setConnectionRequestTimeout(timeoutMilliseconds);
263                        return HttpClients.custom().setSSLSocketFactory(socketFactory)
264                                        .setDefaultRequestConfig(requestBuilder.build()).setDefaultCredentialsProvider(credentialsProvider)
265                                        .build();
266                } catch (KeyManagementException e) {
267                        throw new ServiceCallException("Error creating HTTP client: " + e.getMessage(), e);
268                }
269        }

271        /** Constructs the URL of a service. */
272        public static String getServiceUrl(ServerDetails serverDetails, String projectName, String serviceName) {
273                String url = serverDetails.getUrl();
274                url = StringUtils.ensureEndsWith(url, "/");
275
276                if (projectName != null) {
277                        url += StringUtils.ensureEndsWith(ServiceConstants.LEGACY_PROJECT_PREFIX + projectName, "/");
278                }
279                url = StringUtils.ensureEndsWith(url, "/");
280
281                url += serviceName;
282                url = StringUtils.ensureEndsWith(url, "/");
283
284                return url;
285        }

287        /** Constructs the URL of a global service, e.g. for re-analyzing a project */
288        public static String getGlobalServiceUrl(ServerDetails serverDetails, String serviceName, String projectName) {
289                String url = serverDetails.getUrl();
290                url = StringUtils.ensureEndsWith(url, "/");
291                url += serviceName;
292                if (projectName != null) {
293                        url = StringUtils.ensureEndsWith(url, "/");
294                        url += projectName;
295                }
296                return url;
297        }

332        /** Sets the header and content to perform a Json POST / PUT */
333        private static <T extends HttpEntityEnclosingRequestBase> T setHeaderForJsonRequest(T request, Object data) {
334                request.setHeader("Accept", EMimeType.PLAIN.getType());
335                request.setHeader("Content-type", EMimeType.JSON.getType());
336                request.setEntity(new ByteArrayEntity(JsonUtils.serializeToJSONByteArray(data)));
337                return request;
338        }

340        /**
341         * Completes a PUT or POST request by fixing headers and appending the payload.
342         */
343        /* package */ static <T extends HttpEntityEnclosingRequest> T completePutOrPost(Object data, T putOrPost) {
344                putOrPost.setHeader("Accept", EMimeType.XML.getType() + "," + EMimeType.PLAIN.getType());
345
346                if (data instanceof String) {
347                        putOrPost.setHeader("Content-type", EMimeType.PLAIN.getType());
348                        putOrPost.setEntity(new ByteArrayEntity(StringUtils.stringToBytes((String) data)));
349                } else {
350                        putOrPost.setHeader("Content-type", EMimeType.XML.getType());
351                        putOrPost.setEntity(
352                                        new ByteArrayEntity(StringUtils.stringToBytes(XmlSerializationUtils.serializeToXML(data))));
353                }
354                return putOrPost;
355        }

357        /**
358         * Executes a request and handles the response. Returns null in case of a "not
359         * found" HTTP return code.
360         */
361        public static <T> T executeRequest(HttpClient client, HttpRequestBase request, Class<T> resultClass)
362                        throws ServiceCallException {
363                if (resultClass != String.class) {
364                        request.setHeader("Accept", EMimeType.XML.getType());
365                }
366                try {
367                        HttpResponse response = client.execute(request, createPreemptiveAuthContext(request));
368                        int statusCode = response.getStatusLine().getStatusCode();
369
370                        switch (statusCode) {
371                        case HttpStatus.SC_OK:
372                                return deriveResponseOnSuccess(resultClass, response);
373                        case HttpStatus.SC_NO_CONTENT:
374                                return null;
375                        case HttpStatus.SC_NOT_FOUND:
376                                return null;
377                        default:
378                                String responseBody = EntityUtils.toString(response.getEntity());
379                                throw new ServiceCallException(
380                                                "Service returned with status code " + statusCode + ". "
381                                                                + response.getStatusLine().getReasonPhrase() + "\n" + responseBody,
382                                                statusCode, responseBody);
383                        }
384                } catch (IOException | IllegalArgumentException e) {
385                        // IllegalArgumentException raised when the server port is illegal. Note that
386                        // this will also catch IllegalArgumentExceptions thrown for other reasons.
387                        throw new ServiceCallException("Service call " + request.getURI() + " failed:" + e.getMessage(), e);
388                }
389        }

391        /**
392         * Derive the appropriate type of response to a successful service request
393         */
394        @SuppressWarnings("unchecked")
395        private static <T> T deriveResponseOnSuccess(Class<T> resultClass, HttpResponse response) throws IOException {
396                if (resultClass == String.class) {
397                        return (T) EntityUtils.toString(response.getEntity());
398                }
399                if (resultClass == byte[].class) {
400                        return (T) EntityUtils.toByteArray(response.getEntity());
401                }
402
403                return XmlSerializationUtils.deserializeFromXML(EntityUtils.toString(response.getEntity()), resultClass);
404        }

406        /**
407         * Creates a HTTP context for preemptive basic authentication, i.e. without
408         * waiting for a server authentication request.</a>.
409         */
410        public static HttpClientContext createPreemptiveAuthContext(HttpRequestBase request) {
411                URI uri = request.getURI();
412                AuthCache authCache = new BasicAuthCache();
413                authCache.put(new HttpHost(uri.getHost(), uri.getPort(), uri.getScheme()), new BasicScheme());
414                HttpClientContext context = HttpClientContext.create();
415                context.setAuthCache(authCache);
416                return context;
417        }

419        /**
420         * Creates a multi-part entity that uploads the given files under the given
421         * request parameter.
422         */
423        public static HttpEntity createMultiPartEntity(String parameterName, List<Resource> files) {
424                MultipartEntityBuilder multipartEntityBuilder = MultipartEntityBuilder.create();
425                for (Resource file : files) {
426                        FormBodyPart bodyPart = FormBodyPartBuilder.create(parameterName,
427                                        new ByteArrayBody(file.getAsByteArray(), ContentType.APPLICATION_OCTET_STREAM, file.getName()))
428                                        .build();
429                        multipartEntityBuilder.addPart(bodyPart);
430                }
431                return multipartEntityBuilder.build();
432        }

434        /**
435         * Creates a multi-part entity that uploads the given files under the given
436         * request parameter.
437         */
438        public static HttpEntity createMultiPartEntityForFiles(String parameterName, List<File> files) {
439                MultipartEntityBuilder multipartEntityBuilder = MultipartEntityBuilder.create();
440                for (File file : files) {
441                        FormBodyPart bodyPart = FormBodyPartBuilder
442                                        .create(parameterName, new FileBody(file, ContentType.APPLICATION_OCTET_STREAM)).build();
443                        multipartEntityBuilder.addPart(bodyPart);
444                }
445                return multipartEntityBuilder.build();
446        }
