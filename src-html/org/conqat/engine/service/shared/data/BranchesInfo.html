017package org.conqat.engine.service.shared.data;
018
019import java.util.ArrayList;
020import java.util.Collections;
021import java.util.HashMap;
022import java.util.List;
023import java.util.Map;
024
025import org.apache.commons.lang3.builder.ToStringBuilder;
026import org.apache.commons.lang3.builder.ToStringStyle;
027import org.conqat.engine.index.shared.GitUtils;
028import org.conqat.lib.commons.js_export.ExportToJavaScript;
029
030import com.fasterxml.jackson.annotation.JsonProperty;
031
032/**
033 * Transport class for the branches of a project.
034 *
035 * <strong>This class is used for communication with IDE clients (via the @link
036 * {@link org.conqat.engine.service.shared.client.IdeServiceClient}, special
037 * care has to be taken when changing its signature!</strong>
038 */
039@ExportToJavaScript
040public class BranchesInfo {
041
042        /**
043         * Name of the default branch. This is called "main" branch for history reasons
044         * and we decided to keep as is, as we otherwise need migrations and new
045         * versions for the IDE plugins.
046         */
047        @JsonProperty("mainBranchName")
048        private final String mainBranchName;
049
050        /**
051         * The names of all available active branches, including the default branch, but
052         * excluding deleted and anonymous branches.
053         */
054        @JsonProperty("branchNames")
055        private final List<String> branchNames = new ArrayList<>();
056
057        /** The names of deleted/inactive branches. */
058        @JsonProperty("deletedBranches")
059        private final List<String> deletedBranches = new ArrayList<>();
060
061        /** A list of anonymous branches or an empty list if non exist. */
062        @JsonProperty("anonymousBranches")
063        private final List<String> anonymousBranches = new ArrayList<>();
064
065        /**
066         * This maps aliases to branch names. The aliases are displayed in the UI,
067         * instead of the branch name.
068         */
069        @JsonProperty("aliases")
070        private final Map<String, String> aliases = new HashMap<>();
071
072        /** Empty constructor for serialization purposes */
073        public BranchesInfo() {
074                this("", Collections.emptyList(), Collections.emptyList());
075        }
076
077        public BranchesInfo(String defaultBranchName, List<String> branchNames, List<String> deletedBranches) {
078                this.mainBranchName = defaultBranchName;
079
080                copyNonAnonymousBranches(branchNames, this.branchNames);
081                copyNonAnonymousBranches(deletedBranches, this.deletedBranches);
082        }
083
084        /**
085         * Copies all non-anonymous branches from source to target. Anonymous branches
086         * are stored in {@link #anonymousBranches}.
087         */
088        private void copyNonAnonymousBranches(List<String> sourceBranches, List<String> targetBranches) {
089                for (String branchName : sourceBranches) {
090                        if (branchName.startsWith(GitUtils.ANONYMOUS_BRANCH_NAME)) {
091                                anonymousBranches.add(branchName);
092                        } else {
093                                targetBranches.add(branchName);
094                        }
095                }
096        }
097
098        /** @see #aliases */
099        public void addBranchAlias(String branch, String alias) {
100                aliases.put(branch, alias);
101        }
102
103        /** @see #mainBranchName */
104        public String getDefaultBranchName() {
105                return mainBranchName;
106        }
107
108        /** Returns the union of all branch names. */
109        public List<String> getAllBranchNames() {
110                List<String> allBranchNames = new ArrayList<>(branchNames);
111
112                // prevent NPE for deserialized objects (https://jira.cqse.eu/browse/TS-18489)
113                if (deletedBranches != null) {
114                        allBranchNames.addAll(deletedBranches);
115                }
116                if (anonymousBranches != null) {
117                        allBranchNames.addAll(anonymousBranches);
118                }
119                return allBranchNames;
120        }
121
122        @Override
123        public String toString() {
124                return ToStringBuilder.reflectionToString(this, ToStringStyle.NO_CLASS_NAME_STYLE);
125        }
126}
