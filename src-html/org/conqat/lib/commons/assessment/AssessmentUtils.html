017package org.conqat.lib.commons.assessment;
018
019import java.awt.Color;
020import java.text.DecimalFormat;
021import java.text.NumberFormat;
022
023import org.conqat.lib.commons.color.ECCSMColor;
024
025/**
026 * Utility methods for dealing with ratings.
027 */
028public class AssessmentUtils {
029
030        /** Number format that adds + or - as necessary. */
031        public static final NumberFormat PLUS_MINUS_NUMBERFORMAT = new DecimalFormat("+#;-#");
032
033        /** Returns the color used for visualizing a traffic light color. */
034        public static Color getColor(ETrafficLightColor color) {
035                switch (color) {
036                case RED:
037                        return ECCSMColor.RED.getColor();
038                case YELLOW:
039                        return ECCSMColor.YELLOW.getColor();
040                case GREEN:
041                        return ECCSMColor.GREEN.getColor();
042                case BASELINE:
043                        return ECCSMColor.LIGHT_BLUE.getColor();
044
045                case UNKNOWN:
046                default:
047                        return ECCSMColor.DARK_GRAY.getColor();
048                }
049        }
050
051        /**
052         * Compares two {@link Assessment}s returning which is "better" than another
053         * one.
054         * 
055         * @return the value 0 if o1 is equal to o2; a value less than 0 if o1 is
056         *         worse than o2; and a value greater than 0 if o1 is better than
057         *         o2.
058         */
059        public static int compareAssessments(Assessment o1, Assessment o2) {
060                double red1 = o1.getColorFrequency(ETrafficLightColor.RED) / (double) o1.getSize();
061                double red2 = o2.getColorFrequency(ETrafficLightColor.RED) / (double) o2.getSize();
062                double yellow1 = o1.getColorFrequency(ETrafficLightColor.YELLOW) / (double) o1.getSize();
063                double yellow2 = o2.getColorFrequency(ETrafficLightColor.YELLOW) / (double) o2.getSize();
064
065                // chosen by a close look at the sorting result.
066                double yFactor = 0.4;
067                return Double.compare(red1 + yFactor * yellow1, red2 + yFactor * yellow2);
068        }
069
070        /**
071         * Calculates a delta string for two given assessments. Only colors RED,
072         * YELLOW and GREEN.
073         */
074        public static String calculateAndFormatDelta(Assessment oldAssessment, Assessment newAssessment) {
075                Assessment delta = newAssessment.deepClone();
076                delta.subtract(oldAssessment);
077                StringBuilder builder = new StringBuilder();
078                builder.append("Red: ").append(PLUS_MINUS_NUMBERFORMAT.format(delta.getColorFrequency(ETrafficLightColor.RED)));
079                builder.append(", Yellow: ")
080                                .append(PLUS_MINUS_NUMBERFORMAT.format(delta.getColorFrequency(ETrafficLightColor.YELLOW)));
081                builder.append(", Green: ")
082                                .append(PLUS_MINUS_NUMBERFORMAT.format(delta.getColorFrequency(ETrafficLightColor.GREEN)));
083                return builder.toString();
084        }
085}
