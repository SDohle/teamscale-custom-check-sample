017package org.conqat.lib.commons.assessment;
018
019import org.conqat.lib.commons.js_export.ExportToJavaScript;
020
021/**
022 * Enum for traffic light colors.
023 * 
024 * Note that the order is relevant: The first Color (RED) is considered the most
025 * dominant color (see {@link Assessment#getDominantColor()}).
026 * 
027 * The mapping to actual colors is defined in
028 * {@link AssessmentUtils#getColor(ETrafficLightColor)}, so typically this
029 * method should be adjusted when a new constant is introduced here.
030 */
031@ExportToJavaScript
032public enum ETrafficLightColor {
033
034        /** Red signals errors or incompleteness. */
035        RED("#FF3333"),
036
037        /**
038         * Orange is an intermediate state between {@link #RED} and {@link #YELLOW}.
039         */
040        ORANGE("#FF9966"),
041
042        /** Yellow signals warning or lack of control. */
043        YELLOW("#FFFF33"),
044
045        /** Green signals the absence of errors or correctness. */
046        GREEN("#00CC33"),
047
048        /**
049         * Baseline indicates a baseline entry, i.e. there has been no change compared
050         * to a given baseline and thus no color applies.
051         */
052        BASELINE("#DFEAF4"),
053
054        /** This is used if no information is available. */
055        UNKNOWN;
056
057        /** Display text of the color (name with normal casing) */
058        private final String displayText;
059
060        /** Short, one-character display text of the color (e.g., G, Y, R) */
061        private final String shortDisplayText;
062
063        /** Hex representation of the color */
064        private final String hexValue;
065
066        /**
067         * This is hex value is used if no color is associated with the
068         * ETrafficLightColor.
069         */
070        private static final String HEX_VALUE_BLACK = "#000000";
071
072        /**
073         * Constructor for ETrafficLightColors that do not have an associated color.
074         */
075        private ETrafficLightColor() {
076                this(HEX_VALUE_BLACK);
077        }
078
079        /** Constructor for ETrafficLightColors that have an associated color. */
080        private ETrafficLightColor(String hexValue) {
081                this.displayText = this.name().substring(0, 1) + this.name().substring(1).toLowerCase();
082                this.shortDisplayText = this.name().substring(0, 1);
083                this.hexValue = hexValue;
084        }
085
086        /** @see #displayText */
087        public String getDisplayText() {
088                return displayText;
089        }
090
091        /** @see #shortDisplayText */
092        public String getShortDisplayText() {
093                return shortDisplayText;
094        }
095
096        /** @see #hexValue */
097        public String getHexValue() {
098                return hexValue;
099        }
100
101        /**
102         * Returns the more dominant color, which is the enum literal with smaller index
103         * (as they are sorted by dominance).
104         */
105        public static ETrafficLightColor getDominantColor(ETrafficLightColor color1, ETrafficLightColor color2) {
106                if (color2.ordinal() < color1.ordinal()) {
107                        return color2;
108                }
109                return color1;
110        }
111}
