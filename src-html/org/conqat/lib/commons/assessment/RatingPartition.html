017package org.conqat.lib.commons.assessment;
018
019import java.util.List;
020import java.util.regex.Matcher;
021import java.util.regex.Pattern;
022
023import org.conqat.lib.commons.enums.EnumUtils;
024import org.conqat.lib.commons.string.StringUtils;
025
026/**
027 * A single partition as used in the {@link PartitionedRating}.
028 */
029public class RatingPartition {
030
031        /** The tag used to mark individual partitions. */
032        public static final String PARTITION_TAG = "@ConQAT.Partition";
033
034        /** Pattern used for parsing of the partition tags. */
035        private static final Pattern PARTITION_PATTERN = Pattern.compile(
036                        ".*" + PARTITION_TAG + " +(" + Rating.COLORS_REGEX + ") +Hash: *([a-fA-F0-9]*)( .*)?",
037                        Pattern.CASE_INSENSITIVE);
038
039        /** The stored rating color. */
040        private ETrafficLightColor storedRating;
041
042        /** The hash including both the partition and the hash. */
043        private String hash;
044
045        /** The name of the region. */
046        private final String name;
047
048        /** The start line in the file. */
049        private int startLine;
050
051        /** The end line in the file. */
052        private int endLine;
053
054        /**
055         * The content of this partition (used mostly for calculation of hash
056         * values).
057         */
058        private List<String> content;
059
060        /** Constructor. */
061        /* package */ RatingPartition(ETrafficLightColor rating, String hash, String name) {
062                this.storedRating = rating;
063                this.hash = hash;
064                this.name = name;
065        }
066
067        /** Sets the remaining information stored in this partition. */
068        /* package */void setLinesAndContent(int startLine, int endLine, List<String> content) {
069                this.startLine = startLine;
070                this.endLine = endLine;
071                this.content = content;
072        }
073
074        /** Returns name. */
075        public String getName() {
076                return name;
077        }
078
079        /** Returns 0-based start line (inclusive). */
080        public int getStartLine() {
081                return startLine;
082        }
083
084        /** Returns 0-based end line (inclusive). */
085        public int getEndLine() {
086                return endLine;
087        }
088
089        /**
090         * Returns the rating stored in the partition. This may be null if no stored
091         * rating exists.
092         */
093        public ETrafficLightColor getStoredRating() {
094                return storedRating;
095        }
096
097        /**
098         * Returns the actual rating determined based on the stored value and a
099         * comparison of hash codes.
100         */
101        public ETrafficLightColor getRating() {
102                if (storedRating == null || !hash.equalsIgnoreCase(getExpectedHash())) {
103                        return ETrafficLightColor.RED;
104                }
105
106                return getStoredRating();
107        }
108
109        /** Calculates and returns the expected hash. */
110        private String getExpectedHash() {
111                return new Rating(content, storedRating).getExpectedHashString();
112        }
113
114        /** Sets the rating and updates the hash value accordingly. */
115        public void setStoredRatingAndUpdateHash(ETrafficLightColor color) {
116                this.storedRating = color;
117                this.hash = getExpectedHash();
118        }
119
120        /**
121         * Processes a partition tag and returns the rating partition. Returns null
122         * if no tag was found and processed.
123         */
124        /* package */static RatingPartition processPartitionTag(String line) {
125                Matcher matcher = PARTITION_PATTERN.matcher(line);
126                if (!matcher.matches()) {
127                        return null;
128                }
129
130                ETrafficLightColor color = EnumUtils.valueOfIgnoreCase(ETrafficLightColor.class, matcher.group(1));
131
132                String partitionName = matcher.group(3);
133                if (StringUtils.isEmpty(partitionName)) {
134                        partitionName = StringUtils.EMPTY_STRING;
135                }
136
137                return new RatingPartition(color, matcher.group(2), partitionName.trim());
138        }
139
140        /** Returns the tag representing this partition. */
141        /* package */String getTag() {
142                ETrafficLightColor rating = storedRating;
143                if (rating == null) {
144                        rating = ETrafficLightColor.RED;
145                }
146
147                return PARTITION_TAG + " " + rating.name() + " Hash: " + hash + " " + getName();
148        }
149}
