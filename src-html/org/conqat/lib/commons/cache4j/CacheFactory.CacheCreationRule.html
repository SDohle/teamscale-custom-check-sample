020import java.io.IOException;
021import java.lang.ref.WeakReference;
022import java.util.ArrayList;
023import java.util.LinkedList;

025import java.util.regex.Matcher;
026import java.util.regex.Pattern;

029import org.conqat.lib.commons.assertion.CCSMAssert;
030import org.conqat.lib.commons.cache4j.backend.ECachingStrategy;
031import org.conqat.lib.commons.cache4j.backend.ICacheBackend;

037/**
038 * The central factory used for managing and creating caches. The type of cache
039 * is determined based on the name of the cache (possibly based on the class
040 * name). For this, there is a list of rules with regular expressions. The first
041 * rule, whose expression matches the name is used to create the cache. If no
042 * rule matches, the default rule is used.
043 * <p>
044 * The rules can also be loaded from a configuration file, where each line
045 * contains a rule (or a comment starting with '#'). Each rule has the format
046 * "PATTERN -&gt; THREAD:STRATEGY:PARAMETER", where PATTERN is the pattern used
047 * for matching, THREAD is a constant from {@link ECacheThreadSupport}, STRATEGY
048 * is a constant from {@link ECachingStrategy}, and PARAMETER is a parameter for
049 * the strategy.
050 * <p>
051 * This factory also manages links to all caches created (via this factory).
052 * These links are used for generating statistics. To not avoid garbage
053 * collection of caches (e.g. when many caches are created only temporarily), we
054 * use weak references which to not hinder the GC from collecting a cache.
055 */
056public class CacheFactory {
057
058        /**
059         * Number of creations after which the list of caches is cleaned from unused
060         * caches.
061         */
062        private static final int CLEANUP_INTERVAL = 100;
063
064        /** Pattern used for parsing rules. */
065        private static final Pattern RULE_PATTERN = Pattern.compile("([^\\s]+)\\s*->\\s*([^\\s:]+):([^\\s:]+):([0-9]+)");
066
067        /** The singleton instance. */
068        private static final CacheFactory INSTANCE = new CacheFactory();
069
070        /** Stores all caches created so far. */
071        private final List<CacheInfo> caches = new ArrayList<CacheInfo>();
072
073        /**
074         * Stores creation rules. As we insert both front and back, this is a linked
075         * list.
076         */
077        private final List<CacheCreationRule> rules = new LinkedList<CacheCreationRule>();
078
079        /** The default rule that is used if no other tule matches. */
080        private CacheCreationRule defaultRule = new CacheCreationRule("", ECacheThreadSupport.NONE, ECachingStrategy.OFF,
081                        0);
082
083        /** Counts the number of caches created so far. */
084        private int creationCount = 0;
085
086        /** Hidden constructor. */
087        private CacheFactory() {
088                // nothing to do
089        }
090
091        /** Returns the singleton instance. */
092        public static CacheFactory getInstance() {
093                return INSTANCE;
094        }
095
096        /** Sets the default rule. */
097        public void setDefaultRule(CacheCreationRule rule) {
098                CCSMAssert.isNotNull(rule);
099                defaultRule = rule;
100        }
101
102        /**
103         * Loads the caching rules from a UTF-8 encoded file. See the class comment
104         * for a description of the file format.
105         */
106        public void loadCacheConfiguration(File file) throws IOException, CacheRuleParsingException {
107                loadCacheConfiguration(FileSystemUtils.readFileUTF8(file));
108        }
109
110        /**
111         * Loads the caching rules from a string. See the class comment for a
112         * description of the file format.
113         */
114        public void loadCacheConfiguration(String config) throws CacheRuleParsingException {
115                loadCacheConfiguration(StringUtils.splitLines(config));
116        }
117
118        /**
119         * Loads the caching rules from individual lines. See the class comment for
120         * a description of the file format.
121         */
122        public void loadCacheConfiguration(String[] lines) throws CacheRuleParsingException {
123                clearRules();
124
125                for (int lineNumber = 0; lineNumber < lines.length; ++lineNumber) {
126                        String line = lines[lineNumber];
127
128                        line = line.trim();
129                        if (line.isEmpty() || line.startsWith("#")) {
130                                continue;
131                        }
132
133                        try {
134                                addRuleBack(parseRule(line));
135                        } catch (CacheRuleParsingException e) {
136                                e.setLine(lineNumber + 1);
137                                throw e;
138                        }
139                }
140        }
141
142        /** Parses a rule from a configuration line. */
143        private static CacheCreationRule parseRule(String line) throws CacheRuleParsingException {
144                Matcher matcher = RULE_PATTERN.matcher(line);
145
146                if (!matcher.matches()) {
147                        throw new CacheRuleParsingException("Line does not follow definition pattern!");
148                }
149
150                String pattern = matcher.group(1);
151
152                ECacheThreadSupport threadSupport = EnumUtils.valueOfIgnoreCase(ECacheThreadSupport.class, matcher.group(2));
153                if (threadSupport == null) {
154                        throw new CacheRuleParsingException("Unknown thread support: " + matcher.group(2));
155                }
156
157                ECachingStrategy cachingStrategy = EnumUtils.valueOfIgnoreCase(ECachingStrategy.class, matcher.group(3));
158                if (cachingStrategy == null) {
159                        throw new CacheRuleParsingException("Unknown caching strategy: " + matcher.group(3));
160                }
161
162                try {
163                        return new CacheCreationRule(pattern, threadSupport, cachingStrategy, Integer.parseInt(matcher.group(4)));
164                } catch (PatternSyntaxException e) {
165                        throw new CacheRuleParsingException("Could not parse pattern: " + e.getMessage(), e);
166                } catch (NumberFormatException e) {
167                        throw new CacheRuleParsingException("Could not parse parameter: " + e.getMessage(), e);
168                }
169        }
170
171        /** Removes all rules. */
172        public void clearRules() {
173                rules.clear();
174        }
175
176        /** Adds a rule to the beginning (i.e. it is checked before all others). */
177        public void addRuleFront(CacheCreationRule rule) {
178                rules.add(0, rule);
179        }
180
181        /** Adds a rule to the end (i.e. it is checked after all other rules). */
182        public void addRuleBack(CacheCreationRule rule) {
183                rules.add(rule);
184        }
185
186        /** Returns a cache based on the provided name and the configured rules. */
187        public static <K, V, X extends Exception> ICache<K, V, X> obtainCache(String name,
188                        IParameterizedFactory<V, K, X> factory) {
189                return getInstance().obtainCacheByRules(name, factory);
190        }
191
192        /** Returns a cache based on the provided name and the configured rules. */
193        private synchronized <K, V, X extends Exception> ICache<K, V, X> obtainCacheByRules(String name,
194                        IParameterizedFactory<V, K, X> factory) {
195                CacheCreationRule usedRule = defaultRule;
196                for (CacheCreationRule rule : rules) {
197                        if (rule.matches(name)) {
198                                usedRule = rule;
199                                break;
200                        }
201                }
202
203                ICache<K, V, X> cache = usedRule.createCache(name, factory);
204
205                creationCount += 1;
206                if ((creationCount % CLEANUP_INTERVAL) == 0) {
207                        cleanCaches();
208                }
209
210                caches.add(new CacheInfo(cache, usedRule));
211                return cache;
212        }
213
214        /** Clears the list of caches from all non-existing ones. */
215        private void cleanCaches() {
216                List<CacheInfo> copy = new ArrayList<CacheInfo>(caches);
217                caches.clear();
218                for (CacheInfo cache : copy) {
219                        if (cache.isAlive()) {
220                                caches.add(cache);
221                        }
222                }
223        }
224
225        /** Returns a cache for the given class. */
226        public static <K, V, X extends Exception> ICache<K, V, X> obtainCache(Class<?> clazz,
227                        IParameterizedFactory<V, K, X> factory) {
228                return obtainCache(clazz.getName(), factory);
229        }
230
231        /**
232         * Clears all data from all caches managed by cache4j.
233         * 
234         * @param allThreads
235         *            if this is true, cached data will be cleared in all threads.
236         *            Otherwise, only cached data from the current thread will be
237         *            removed (if possible).
238         */
239        public void clearAllCachedData(boolean allThreads) {
240                for (CacheInfo cache : new ArrayList<CacheInfo>(caches)) {
241                        cache.clearAllCachedData(allThreads);
242                }
243        }
244
245        /**
246         * Returns a string containing statistics on all existing caches in CSV
247         * format.
248         */
249        public String getStatistics() {
250                StringBuilder sb = new StringBuilder();
251
252                sb.append("Name;Thread Support;Cache Strategy; Strategy Parameter; Hits; Misses; Miss Cost Millis"
253                                + StringUtils.CR);
254                for (CacheInfo cache : new ArrayList<CacheInfo>(caches)) {
255                        cache.appendStatistics(sb);
256                }
257
258                return sb.toString();
259        }
260
261        /**
262         * Clears the list of caches used for reporting. This method is only used
263         * for testing.
264         */
265        /* package */void clearCaches() {
266                caches.clear();
267        }
268
269        /** A single rule used for cache creation. */
270        public static final class CacheCreationRule {
271
272                /** The pattern used to match the cache name. */
273                private final Pattern pattern;
274
275                /** The required thread support. */
276                private final ECacheThreadSupport threadSupport;
277
278                /** The caching strategy used. */
279                private final ECachingStrategy cachingStrategy;
280
281                /** The parameter used for the caching strategy. */
282                private final int cachingStrategyParameter;
283
284                /** Constructor. */
285                public CacheCreationRule(Pattern pattern, ECacheThreadSupport threadSupport, ECachingStrategy cachingStrategy,
286                                int cachingStrategyParameter) {
287                        this.pattern = pattern;
288                        this.threadSupport = threadSupport;
289                        this.cachingStrategy = cachingStrategy;
290                        this.cachingStrategyParameter = cachingStrategyParameter;
291                }
292
293                /** Constructor. */
294                public CacheCreationRule(String pattern, ECacheThreadSupport threadSupport, ECachingStrategy cachingStrategy,
295                                int cachingStrategyParameter) throws PatternSyntaxException {
296                        this(Pattern.compile(pattern), threadSupport, cachingStrategy, cachingStrategyParameter);
297                }
298
299                /** Returns whether the rules matches the given name. */
300                public boolean matches(String name) {
301                        return pattern.matcher(name).matches();
302                }
303
304                /** Creates the cache from this rule. */
305                @SuppressWarnings({ "unchecked", "rawtypes" })
306                public <K, V, X extends Exception> ICache<K, V, X> createCache(String name,
307                                IParameterizedFactory<V, K, X> factory) {
308                        return threadSupport.createCache(name, factory,
309                                        (ICacheBackend) cachingStrategy.getBackend(cachingStrategyParameter));
310                }
311        }
312
313        /** Class for storing information on created caches. */
314        private static final class CacheInfo {
315
316                /** Reference to the cache. */
317                private final WeakReference<ICache<?, ?, ?>> cacheRef;
318
319                /** The rule from which the cache was created. */
320                private final CacheCreationRule creationRule;
321
322                /** Constructor. */
323                public CacheInfo(ICache<?, ?, ?> cache, CacheCreationRule creationRule) {
324                        cacheRef = new WeakReference<ICache<?, ?, ?>>(cache);
325                        this.creationRule = creationRule;
326                }
327
328                /** Clears all cached data. */
329                public void clearAllCachedData(boolean allThreads) {
330                        ICache<?, ?, ?> cache = cacheRef.get();
331                        if (cache == null) {
332                                return;
333                        }
334
335                        cache.clear(allThreads);
336                }
337
338                /** Returns whether the cache is still in existence. */
339                public boolean isAlive() {
340                        return cacheRef.get() != null;
341                }
342
343                /**
344                 * Appends statistics on this cache in CSV format (including new line)
345                 */
346                public void appendStatistics(StringBuilder sb) {
347                        ICache<?, ?, ?> cache = cacheRef.get();
348                        if (cache == null) {
349                                return;
350                        }
351
352                        sb.append(cache.getName() + ";" + creationRule.threadSupport.name() + ";"
353                                        + creationRule.cachingStrategy.name() + ";" + creationRule.cachingStrategyParameter + ";"
354                                        + cache.getHits() + ";" + cache.getMisses() + ";" + cache.getMissCostMillis() + StringUtils.CR);
355                }
356        }
