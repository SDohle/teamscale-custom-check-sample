017package org.conqat.lib.commons.collections;
018
019import java.util.Collection;
020import java.util.HashMap;
021import java.util.List;
022import java.util.Map;
023
024import org.conqat.lib.commons.assertion.CCSMAssert;
025import org.conqat.lib.commons.error.NeverThrownRuntimeException;
026import org.conqat.lib.commons.factory.IFactory;
027
028/**
029 * A collection map deals with a map of collections, i.e. each key can store
030 * multiple elements. Depending on the collection implementation chosen, this
031 * can also be interpreted as a multi map.
032 * <p>
033 * If you deal with the basic case of {@link Map}s and {@link List}s, use the
034 * {@link ListMap}, which is much easier to apply.
035 * 
036 * @param <K>
037 *            the key type.
038 * @param <V>
039 *            the value type (i.e. the values stored in the collections).
040 * @param <C>
041 *            the collection type, which is made explicit at the interface.
042 */
043public class CollectionMap<K, V, C extends Collection<V>> {
044
045        /** The underlying map. */
046        private final Map<K, C> map;
047
048        /** The factory used for creating collections. */
049        private final IFactory<C, NeverThrownRuntimeException> collectionFactory;
050
051        /** Constructor using a {@link HashMap} as underlying map. */
052        public CollectionMap(IFactory<C, NeverThrownRuntimeException> collectionFactory) {
053                this(new HashMap<K, C>(), collectionFactory);
054        }
055
056        /** Constructor. */
057        public CollectionMap(Map<K, C> map, IFactory<C, NeverThrownRuntimeException> collectionFactory) {
058                CCSMAssert.isNotNull(map);
059                CCSMAssert.isNotNull(collectionFactory);
060
061                this.map = map;
062                this.collectionFactory = collectionFactory;
063        }
064
065        /**
066         * Returns the collection stored under the given key (or null). Modifying
067         * the collection will directly affect this object.
068         */
069        public C getCollection(K key) {
070                return map.get(key);
071        }
072
073        /**
074         * Adds a value to the collection associated with the given key.
075         * 
076         * @return <code>true</code> if the collection associated with the given key
077         *         changed as a result of the call.
078         */
079        public boolean add(K key, V value) {
080                return getOrCreateCollection(key).add(value);
081        }
082
083        /**
084         * Adds all values to the collection associated with the given key.
085         * 
086         * @return <code>true</code> if the collection associated with the given key
087         *         changed as a result of the call.
088         */
089        public boolean addAll(K key, Collection<? extends V> values) {
090                return getOrCreateCollection(key).addAll(values);
091        }
092
093        /**
094         * Returns the collection stored under the given key (or creates a new one).
095         */
096        /* package */C getOrCreateCollection(K key) {
097                C collection = map.get(key);
098                if (collection == null) {
099                        collection = collectionFactory.create();
100                        map.put(key, collection);
101                }
102                return collection;
103        }
104
105        /** Adds all elements from another collection map. */
106        public void addAll(CollectionMap<K, V, C> other) {
107                for (K key : other.getKeys()) {
108                        C collection = other.getCollection(key);
109                        if (collection != null) {
110                                addAll(key, collection);
111                        }
112                }
113        }
114
115        /** Returns whether an element is contained. */
116        public boolean contains(K key, V value) {
117                C collection = map.get(key);
118                if (collection == null) {
119                        return false;
120                }
121                return collection.contains(value);
122        }
123
124        /**
125         * Removes an element.
126         * 
127         * @return true if an element was removed as a result of this call.
128         */
129        public boolean remove(K key, V value) {
130                C collection = map.get(key);
131                if (collection == null) {
132                        return false;
133                }
134                return collection.remove(value);
135        }
136
137        /**
138         * Check if a (possibly empty) collection is present for a given key.
139         */
140        public boolean containsCollection(K key) {
141                return map.containsKey(key);
142        }
143
144        /**
145         * Removes the collection stored for a key.
146         * 
147         * @return true if a collection was removed as a result of this call.
148         */
149        public boolean removeCollection(K key) {
150                return map.remove(key) != null;
151        }
152
153        /** Get the keys. */
154        public UnmodifiableSet<K> getKeys() {
155                return CollectionUtils.asUnmodifiable(map.keySet());
156        }
157
158        /** Return all values from all collections. */
159        public C getValues() {
160                C result = collectionFactory.create();
161                for (C values : map.values()) {
162                        result.addAll(values);
163                }
164                return result;
165        }
166
167        /**
168         * Returns the number of key-value mappings in this map.
169         */
170        public int size() {
171                return map.size();
172        }
173
174        /** Return the total count of values over all collections. */
175        public int getValueCount() {
176                int result = 0;
177                for (C values : map.values()) {
178                        result += values.size();
179                }
180                return result;
181        }
182
183        /** Clears the underlying map and thus all contents. */
184        public void clear() {
185                map.clear();
186        }
187
188        /**
189         * Converts the {@link CollectionMap} to a map with arrays
190         * 
191         * @param type
192         *            Type of the target array
193         */
194        public Map<K, V[]> collectionsToArrays(Class<V> type) {
195                Map<K, V[]> map = new HashMap<K, V[]>();
196                for (K key : getKeys()) {
197                        map.put(key, CollectionUtils.toArray(getCollection(key), type));
198                }
199                return map;
200        }
201
202        /** {@inheritDoc} */
203        @Override
204        public String toString() {
205                return map.toString();
206        }
207}
