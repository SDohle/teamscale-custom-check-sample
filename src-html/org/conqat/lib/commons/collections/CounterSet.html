017package org.conqat.lib.commons.collections;
018
019import java.io.PrintWriter;
020import java.io.Serializable;
021import java.util.Collection;
022import java.util.Comparator;
023import java.util.Iterator;
024import java.util.LinkedHashMap;
025import java.util.List;
026import java.util.Map;
027import java.util.Map.Entry;
028
029import org.conqat.lib.commons.js_export.ExportAsType;
030import org.conqat.lib.commons.js_export.ExportToJavaScript;
031
032import com.fasterxml.jackson.annotation.JsonProperty;
033
034/**
035 * This class manages a set of counters (i.e. is a mapping from some key objects
036 * to integers). As the implementation is based on hash maps, key objects must
037 * provide suitable hash keys.
038 */
039@ExportToJavaScript
040public class CounterSet<E> implements Serializable, Iterable<Pair<E, Integer>> {
041
042        /** Version used for serialization. */
043        private static final long serialVersionUID = 1;
044
045        /** The underlying map. */
046        @JsonProperty("map")
047        @ExportAsType("!Object<string, number>")
048        protected final Map<E, Integer> map = new LinkedHashMap<>();
049
050        /** Stores total value. */
051        @JsonProperty("total")
052        protected int total = 0;
053
054        /** Constructs an empty {@link CounterSet}. */
055        public CounterSet() {
056                // nothing to do
057        }
058
059        /**
060         * Constructs a new {@link CounterSet} from the given keys. Initializes all keys
061         * with 1.
062         */
063        public CounterSet(Collection<E> keys) {
064                incAll(keys);
065        }
066
067        /** Constructs a CounterSet with one value. */
068        public CounterSet(E key, int value) {
069                inc(key, value);
070        }
071
072        /**
073         * Add the given increment to an element. If the element was not present before,
074         * it is interpreted as if it was present with value 0. Returns the new value.
075         *
076         * @param key
077         *            the key of the counter to increment.
078         * @param increment
079         *            the increment.
080         */
081        public int inc(E key, int increment) {
082                Integer value = map.get(key);
083                int newValue;
084                if (value == null) {
085                        newValue = increment;
086                } else {
087                        newValue = value + increment;
088                }
089                map.put(key, newValue);
090
091                // update total sum
092                total += increment;
093                return getValue(key);
094        }
095
096        /**
097         * Same as <code>inc(key, 1)</code>.
098         *
099         * @see #inc(Object, int)
100         */
101        public int inc(E key) {
102                return inc(key, 1);
103        }
104
105        /**
106         * Add the given increment to the given keys. If a key was not present before,
107         * it is interpreted as if it was present with value 0.
108         *
109         * @param keys
110         *            the keys of the counter to increment.
111         * @param increment
112         *            the increment.
113         */
114        public void incAll(Collection<E> keys, int increment) {
115                for (E key : keys) {
116                        inc(key, increment);
117                }
118        }
119
120        /** Increments the given elements by 1 */
121        public void incAll(Collection<E> keys) {
122                for (E key : keys) {
123                        inc(key);
124                }
125        }
126
127        /**
128         * Adds the given {@link CounterSet} to this {@link CounterSet} by incrementing
129         * all keys contained from other.
130         */
131        public void add(CounterSet<E> other) {
132                for (E key : other.getKeys()) {
133                        inc(key, other.getValue(key));
134                }
135        }
136
137        /**
138         * Removes the second CounterSet from the first one and returns a new
139         * CounterSet.
140         */
141        public static <E> CounterSet<E> removeSecondFromFirst(CounterSet<E> first, CounterSet<E> second) {
142                CounterSet<E> merged = new CounterSet<>();
143                for (E key : CollectionUtils.unionSet(first.getKeys(), second.getKeys())) {
144                        merged.inc(key, first.getValue(key));
145                        merged.inc(key, -second.getValue(key));
146                        if (merged.getValue(key) == 0) {
147                                merged.remove(key);
148                        }
149                }
150                return merged;
151        }
152
153        /**
154         * Remove the entry with the given key, i.e. sets its value to 0. In case the
155         * entry does not exist, nothing happens.
156         */
157        public void remove(E key) {
158                total -= getValue(key);
159                map.remove(key);
160        }
161
162        /** Removes all entries with the given keys. */
163        public void removeAll(Collection<E> keys) {
164                for (E key : keys) {
165                        remove(key);
166                }
167        }
168
169        /** Clears the counter set. */
170        public void clear() {
171                map.clear();
172                total = 0;
173        }
174
175        /**
176         * Checks if an element is stored in the array.
177         */
178        public boolean contains(E key) {
179                return map.containsKey(key);
180        }
181
182        /**
183         * Get the value for an element. If the the element is not stored in the counter
184         * <code>0</code> is returned.
185         */
186        public int getValue(E key) {
187                Integer value = map.get(key);
188                if (value == null) {
189                        return 0;
190                }
191                return value;
192        }
193
194        /**
195         * Returns the set of all elements used a keys for counters.
196         */
197        public UnmodifiableSet<E> getKeys() {
198                return CollectionUtils.asUnmodifiable(map.keySet());
199        }
200
201        /** Returns a list of all keys ordered by their value ascending */
202        public List<E> getKeysByValueAscending() {
203                return CollectionUtils.sort(getKeys(), new Comparator<E>() {
204                        @Override
205                        public int compare(E key1, E key2) {
206                                return map.get(key1).compareTo(map.get(key2));
207                        }
208                });
209        }
210
211        /** Returns a list of all keys ordered by their value descending */
212        public List<E> getKeysByValueDescending() {
213                return CollectionUtils.reverse(getKeysByValueAscending());
214        }
215
216        /** Get total sum of all elements. */
217        public int getTotal() {
218                return total;
219        }
220
221        /** Returns a collection of all values */
222        public Collection<Integer> values() {
223                return map.values();
224        }
225
226        /** {@inheritDoc} */
227        @Override
228        public String toString() {
229                return map.toString();
230        }
231
232        /**
233         * Prints the distribution of values (ascending or descending) to System.out,
234         * where each value is printed on a separate line in the form &lt;key&gt; :
235         * &lt;value&gt;.
236         *
237         * <p>
238         * <i>Example:</i><br>
239         *
240         * foo : 4 <br>
241         * bar : 2
242         * </p>
243         */
244        public void printValueDistribution(boolean ascending) {
245                printValueDistribution(new PrintWriter(System.out), ascending);
246        }
247
248        /**
249         * Prints the distribution of values (ascending or descending) to the given
250         * stream, where each value is printed on a separate line in the form
251         * &lt;key&gt; : &lt;value&gt;.
252         *
253         * <p>
254         * <i>Example:</i><br>
255         *
256         * foo : 4 <br>
257         * bar : 2
258         * </p>
259         */
260        public void printValueDistribution(PrintWriter writer, boolean ascending) {
261                List<E> keys = null;
262                if (ascending) {
263                        keys = getKeysByValueAscending();
264                } else {
265                        keys = getKeysByValueDescending();
266                }
267                for (E key : keys) {
268                        writer.print(String.valueOf(key));
269                        writer.print(" : ");
270                        writer.print(getValue(key));
271                        writer.println();
272                }
273                writer.flush();
274        }
275
276        /** {@inheritDoc} */
277        @Override
278        public boolean equals(Object obj) {
279                if (obj instanceof CounterSet) {
280                        @SuppressWarnings("rawtypes")
281                        CounterSet other = (CounterSet) obj;
282                        return map.equals(other.map);
283                }
284                return false;
285        }
286
287        /** {@inheritDoc} */
288        @Override
289        public int hashCode() {
290                return map.hashCode();
291        }
292
293        /** Returns the values as {@link Map}. */
294        public Map<E, Integer> toMap() {
295                return new LinkedHashMap<>(map);
296        }
297
298        /** Returns the values as a {@link Map}, but omits keys where the value is 0. */
299        public Map<E, Integer> toMapWithoutZeroEntries() {
300                Map<E, Integer> map = toMap();
301                map.keySet().removeIf(key -> map.get(key) == 0);
302                return map;
303        }
304
305        /** {@inheritDoc} */
306        @Override
307        public Iterator<Pair<E, Integer>> iterator() {
308                return new Iterator<Pair<E, Integer>>() {
309                        private Iterator<Map.Entry<E, Integer>> delegate = map.entrySet().iterator();
310
311                        /** {@inheritDoc} */
312                        @Override
313                        public boolean hasNext() {
314                                return delegate.hasNext();
315                        }
316
317                        /** {@inheritDoc} */
318                        @Override
319                        public Pair<E, Integer> next() {
320                                Entry<E, Integer> next = delegate.next();
321                                if (next == null) {
322                                        return null;
323                                }
324                                return new Pair<>(next.getKey(), next.getValue());
325                        }
326                };
327        }
328}
