017package org.conqat.lib.commons.collections;
018
019import java.util.ArrayList;
020import java.util.List;
021import java.util.Map;
022
023import org.conqat.lib.commons.error.NeverThrownRuntimeException;
024import org.conqat.lib.commons.factory.IFactory;
025
026/**
027 * Manages a map of lists, i.e. each key can store multiple elements.
028 * 
029 * @param <K>
030 *            the key type.
031 * @param <V>
032 *            the value type (i.e. the values stored in the collections).
033 */
034public class ListMap<K, V> extends CollectionMap<K, V, List<V>> {
035
036        /**
037         * The factory used for list creation. We have to make this static, so is
038         * can be used in the calls to super(), but this also means we can not use
039         * the generic parameters of this class. While we could use an anonymous
040         * class in the constructor to circumvent this, we would create an instance
041         * of this factory for each instance for {@link ListMap}. To avoid this, we
042         * use the static but untyped variant here.
043         */
044        private static final IFactory<List<?>, NeverThrownRuntimeException> factory = new IFactory<List<?>, NeverThrownRuntimeException>() {
045                @Override
046                public List<?> create() {
047                        return new ArrayList<Object>();
048                }
049        };
050
051        /** Create new hashed list map. */
052        @SuppressWarnings({ "unchecked", "rawtypes" })
053        public ListMap() {
054                // This cast is not nice, but it works
055                super((IFactory) factory);
056        }
057
058        /** Create new hashed list map with a specified map. */
059        @SuppressWarnings({ "unchecked", "rawtypes" })
060        public ListMap(Map<K, List<V>> map) {
061                // This cast is not nice, but it works
062                super(map, (IFactory) factory);
063        }
064
065        /** Copy constructor. */
066        public ListMap(ListMap<K, V> other) {
067                this();
068                addAll(other);
069        }
070
071}
