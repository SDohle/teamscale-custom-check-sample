047/**
048 * A list for storing pairs in a specific order.
049 */
050public class PairList<S, T> implements Serializable, Iterable<Pair<S, T>> {
051
052        /** Version used for serialization. */
053        private static final long serialVersionUID = 1;
054
055        /** The current size. */
056        private int size = 0;
057
058        /** The array used for storing the S. */
059        private Object[] firstElements;
060
061        /** The array used for storing the T. */
062        private Object[] secondElements;
063
064        /** Constructor. */
065        public PairList() {
066                this(16);
067        }
068
069        /** Constructor. */
070        public PairList(int initialCapacity) {
071                if (initialCapacity < 1) {
072                        initialCapacity = 1;
073                }
074                firstElements = new Object[initialCapacity];
075                secondElements = new Object[initialCapacity];
076        }
077
078        /** Copy constructor. */
079        public PairList(PairList<S, T> other) {
080                this(other.size);
081                addAll(other);
082        }
083
084        /**
085         * Constructor to convert a map into a pair list.
086         */
087        public PairList(Map<S, T> map) {
088                this(map.size());
089                for (Entry<S, T> entry : map.entrySet()) {
090                        add(entry.getKey(), entry.getValue());
091                }
092        }
093
094        /** Returns whether the list is empty. */
095        public boolean isEmpty() {
096                return size == 0;
097        }
098
099        /** Returns the size of the list. */
100        public int size() {
101                return size;
102        }
103
104        /** Add the given pair to the list. */
105        public void add(S first, T second) {
106                ensureSpace(size + 1);
107                firstElements[size] = first;
108                secondElements[size] = second;
109                ++size;
110        }
111
112        /** Adds all pairs from another list. */
113        public void addAll(PairList<S, T> other) {
114                // we have to store this in a local var, as other.size may change if
115                // other == this
116                int otherSize = other.size;
117
118                ensureSpace(size + otherSize);
119                for (int i = 0; i < otherSize; ++i) {
120                        firstElements[size] = other.firstElements[i];
121                        secondElements[size] = other.secondElements[i];
122                        ++size;
123                }
124        }
125
126        /** Make sure there is space for at least the given amount of elements. */
127        protected void ensureSpace(int space) {
128                if (space <= firstElements.length) {
129                        return;
130                }
131
132                Object[] oldFirst = firstElements;
133                Object[] oldSecond = secondElements;
134                int newSize = firstElements.length * 2;
135                while (newSize < space) {
136                        newSize *= 2;
137                }
138
139                firstElements = new Object[newSize];
140                secondElements = new Object[newSize];
141                System.arraycopy(oldFirst, 0, firstElements, 0, size);
142                System.arraycopy(oldSecond, 0, secondElements, 0, size);
143        }
144
145        /** Returns the first element at given index. */
146        @SuppressWarnings("unchecked")
147        public S getFirst(int i) {
148                checkWithinBounds(i);
149                return (S) firstElements[i];
150        }
151
152        /**
153         * Checks whether the given <code>i</code> is within the bounds. Throws an
154         * exception otherwise.
155         */
156        private void checkWithinBounds(int i) {
157                if (i < 0 || i >= size) {
158                        throw new IndexOutOfBoundsException("Out of bounds: " + i);
159                }
160        }
161
162        /** Sets the first element at given index. */
163        public void setFirst(int i, S value) {
164                checkWithinBounds(i);
165                firstElements[i] = value;
166        }
167
168        /** Returns the second element at given index. */
169        @SuppressWarnings("unchecked")
170        public T getSecond(int i) {
171                checkWithinBounds(i);
172                return (T) secondElements[i];
173        }
174
175        /** Sets the first element at given index. */
176        public void setSecond(int i, T value) {
177                checkWithinBounds(i);
178                secondElements[i] = value;
179        }
180
181        /** Creates a new list containing all first elements. */
182        @SuppressWarnings("unchecked")
183        public List<S> extractFirstList() {
184                List<S> result = new ArrayList<S>(size + 1);
185                for (int i = 0; i < size; ++i) {
186                        result.add((S) firstElements[i]);
187                }
188                return result;
189        }
190
191        /** Creates a new list containing all second elements. */
192        @SuppressWarnings("unchecked")
193        public List<T> extractSecondList() {
194                List<T> result = new ArrayList<T>(size + 1);
195                for (int i = 0; i < size; ++i) {
196                        result.add((T) secondElements[i]);
197                }
198                return result;
199        }
200
201        /**
202         * Swaps the pairs of this list. Is S and T are different types, this will
203         * be extremely dangerous.
204         */
205        public void swapPairs() {
206                Object[] temp = firstElements;
207                firstElements = secondElements;
208                secondElements = temp;
209        }
210
211        /** Swaps the entries located at indexes i and j. */
212        public void swapEntries(int i, int j) {
213                S tmp1 = getFirst(i);
214                T tmp2 = getSecond(i);
215                setFirst(i, getFirst(j));
216                setSecond(i, getSecond(j));
217                setFirst(j, tmp1);
218                setSecond(j, tmp2);
219        }
220
221        /** Clears this list. */
222        public void clear() {
223                size = 0;
224        }
225
226        /** Removes the last element of the list. */
227        public void removeLast() {
228                CCSMAssert.isTrue(size > 0, "Size must be positive!");
229                size -= 1;
230        }
231
232        /** {@inheritDoc} */
233        @Override
234        public String toString() {
235                StringBuffer result = new StringBuffer();
236                result.append('[');
237                for (int i = 0; i < size; i++) {
238                        if (i != 0) {
239                                result.append(',');
240                        }
241                        result.append('(');
242                        result.append(String.valueOf(firstElements[i]));
243                        result.append(',');
244                        result.append(String.valueOf(secondElements[i]));
245                        result.append(')');
246                }
247                result.append(']');
248                return result.toString();
249        }
250
251        /** {@inheritDoc} */
252        @Override
253        public int hashCode() {
254                int prime = 31;
255                int hash = size;
256                hash = prime * hash + HashCodeUtils.hashArrayPart(firstElements, 0, size);
257                return prime * hash + HashCodeUtils.hashArrayPart(secondElements, 0, size);
258        }
259
260        /** {@inheritDoc} */
261        @Override
262        public boolean equals(Object obj) {
263                if (this == obj) {
264                        return true;
265                }
266                if (!(obj instanceof PairList)) {
267                        return false;
268                }
269
270                PairList<?, ?> other = (PairList<?, ?>) obj;
271                if (size != other.size) {
272                        return false;
273                }
274                for (int i = 0; i < size; i++) {
275                        if (!Objects.equals(firstElements[i], other.firstElements[i])
276                                        || !Objects.equals(secondElements[i], secondElements[i])) {
277                                return false;
278                        }
279                }
280                return true;
281        }
282
283        /** {@inheritDoc} */
284        @Override
285        public Iterator<Pair<S, T>> iterator() {
286                return new Iterator<Pair<S, T>>() {
287                        int index = 0;
288
289                        /** {@inheritDoc} */
290                        @Override
291                        public boolean hasNext() {
292                                return index < size;
293                        }
294
295                        /** {@inheritDoc} */
296                        @Override
297                        public Pair<S, T> next() {
298                                checkWithinBounds(index);
299                                int oldIndex = index;
300                                index++;
301                                return createPairForIndex(oldIndex);
302                        }
303                };
304        }
305
306        /**
307         * Returns a non-parallel stream of {@link Pair}s from this list.
308         * <p>
309         * NOTE: the underlying {@link Spliterator} implementation does currently
310         * not support splitting. Thus, trying to use the returned stream in
311         * parallel execution will fail.
312         *
313         * @see PairListSpliterator#trySplit()
314         */
315        public Stream<Pair<S, T>> stream() {
316                return StreamSupport.stream(new PairListSpliterator(), false);
317        }
318
319        /**
320         * Converts this {@link PairList} into a {@link Map}.
321         *
322         * If the first elements contain duplicates (according to
323         * {@link Object#equals(Object)}, an {@link IllegalStateException} is
324         * thrown. If the first elements may have duplicates, use
325         * {@link #toMap(BinaryOperator)} instead.
326         */
327        public Map<S, T> toMap() {
328                return stream().collect(Collectors.toMap(Pair::getFirst, Pair::getSecond));
329        }
330
331        /**
332         * Converts this {@link PairList} into a {@link Map}.
333         *
334         * If the first elements contain duplicates (according to
335         * {@link Object#equals(Object)}, the respective second elements are merged
336         * using the provided merging function..
337         */
338        public Map<S, T> toMap(BinaryOperator<T> mergeFunction) {
339                return stream().collect(Collectors.toMap(Pair::getFirst, Pair::getSecond, mergeFunction));
340        }
341
342        /**
343         * Creates a pair from the values at the given index in this list.
344         *
345         * We suppress unchecked cast warnings since the PairList stores all
346         * elements as plain Objects.
347         */
348        @SuppressWarnings("unchecked")
349        private Pair<S, T> createPairForIndex(int index) {
350                return new Pair<S, T>((S) firstElements[index], (T) secondElements[index]);
351        }
352
353        /**
354         * Spliterator for the {@link PairList}.
355         */
356        private final class PairListSpliterator implements Spliterator<Pair<S, T>> {
357
358                /**
359                 * The next element to process when {@link #tryAdvance(Consumer)} is
360                 * called.
361                 */
362                private int nextElementToProcess;
363
364                /** {@inheritDoc} */
365                @Override
366                public int characteristics() {
367                        // the Spliterator is not subsized since we don't support splitting
368                        // at the moment
369                        return Spliterator.IMMUTABLE | Spliterator.ORDERED | Spliterator.SIZED;
370                }
371
372                /** {@inheritDoc} */
373                @Override
374                public long estimateSize() {
375                        return size - nextElementToProcess;
376                }
377
378                /** {@inheritDoc} */
379                @Override
380                public boolean tryAdvance(Consumer<? super Pair<S, T>> action) {
381                        if (nextElementToProcess == size) {
382                                return false;
383                        }
384
385                        action.accept(createPairForIndex(nextElementToProcess));
386                        nextElementToProcess++;
387                        return true;
388                }
389
390                /**
391                 * {@inheritDoc}
392                 *
393                 * We return <code>null</code> to indicate that this {@link Spliterator}
394                 * cannot be split. If parallel execution is required, this method must
395                 * be implemented and the subsized characteristic should be adhered to.
396                 */
397                @Override
398                public Spliterator<Pair<S, T>> trySplit() {
399                        return null;
400                }
401        }

403        /**
404         * Collects a stream of pairs into a {@link PairList}.
405         */
406        private static class PairListCollector<S, T> implements Collector<Pair<S, T>, PairList<S, T>, PairList<S, T>> {
407
408                /** {@inheritDoc} */
409                @Override
410                public Supplier<PairList<S, T>> supplier() {
411                        return PairList::new;
412                }
413
414                /** {@inheritDoc} */
415                @Override
416                public BiConsumer<PairList<S, T>, Pair<S, T>> accumulator() {
417                        return (list, pair) -> list.add(pair.getFirst(), pair.getSecond());
418                }
419
420                /** {@inheritDoc} */
421                @Override
422                public BinaryOperator<PairList<S, T>> combiner() {
423                        return (list1, list2) -> {
424                                list1.addAll(list2);
425                                return list1;
426                        };
427                }
428
429                /** {@inheritDoc} */
430                @Override
431                public Function<PairList<S, T>, PairList<S, T>> finisher() {
432                        return list -> list;
433                }
434
435                /** {@inheritDoc} */
436                @Override
437                public Set<java.util.stream.Collector.Characteristics> characteristics() {
438                        return EnumSet.of(Characteristics.IDENTITY_FINISH);
439                }
440
441        }
442
443        /**
444         * Returns a collector for collecting a stream of pairs into a
445         * {@link PairList}.
446         */
447        public static <S, T> PairListCollector<S, T> toPairList() {
448                return new PairListCollector<>();
449        }
450
451        /**
452         * Creates a {@link PairList} that consists of pairs of values taken from
453         * the two collections. I.e. entry i in the resulting {@link PairList} will
454         * consists of the pair (a, b) where a is the entry at index i in the first
455         * collection and b is the entry at index i of the second collection.
456         *
457         * Both collections must be of the same size. The order of insertion into
458         * the new {@link PairList} is determined by the order imposed by the
459         * collections' iterators.
460         */
461        public static <S, T> PairList<S, T> zip(Collection<S> firstValues, Collection<T> secondValues) {
462                CCSMAssert.isTrue(firstValues.size() == secondValues.size(),
463                                "Can only zip together collections of the same size.");
464                PairList<S, T> result = new PairList<>(firstValues.size());
465                Iterator<S> firstIterator = firstValues.iterator();
466                Iterator<T> secondIterator = secondValues.iterator();
467                while (firstIterator.hasNext()) {
468                        result.add(firstIterator.next(), secondIterator.next());
469                }
470                return result;
471        }
472
473        /**
474         * For each element pair in this list, calls the given consumer with the
475         * first and second value from the pair as the only arguments.
476         */
477        public void forEach(BiConsumer<S, T> consumer) {
478                for (int i = 0; i < size; i++) {
479                        consumer.accept(getFirst(i), getSecond(i));
480                }
481        }
482
483        /** Returns a reversed copy of this list. */
484        public PairList<S, T> reverse() {
485                PairList<S, T> reversed = new PairList<>();
486                for (int i = size() - 1; i >= 0; i--) {
487                        reversed.add(getFirst(i), getSecond(i));
488                }
489                return reversed;
490        }
491
492        /**
493         * Returns a newly allocated array containing all of the elements in this
494         * PairList as an array of Pair<S,T>s.
495         */
496        @SuppressWarnings("unchecked")
497        public Pair<S, T>[] toArray() {
498                Pair<S, T>[] result = new Pair[size];
499                for (int i = 0; i < size; i++) {
500                        result[i] = new Pair<>((S) firstElements[i], (T) secondElements[i]);
501                }
502                return result;
503        }
504
505        /**
506         * Sorts the PairList according to the given comparator.
507         */
508        public void sort(Comparator<Pair<S, T>> comparator) {
509                Pair<S, T>[] sorted = this.toArray();
510                Arrays.sort(sorted, comparator);
511                for (int i = 0; i < sorted.length; i++) {
512                        setFirst(i, sorted[i].getFirst());
513                        setSecond(i, sorted[i].getSecond());
514                }
515        }

563        /**
564         * Returns an empty list (immutable). This list is serializable. This is
565         * inspired by the Java Collections.emptyList() method.

567         * <p>
