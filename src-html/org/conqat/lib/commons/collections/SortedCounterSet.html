017package org.conqat.lib.commons.collections;
018
019import java.util.ArrayList;
020import java.util.Collections;
021import java.util.Comparator;
022import java.util.List;
023
024/**
025 * A counter set supporting ranged access to its values, i.e. it is possible to
026 * query the sum of values for all keys in a given range. Inserting items works
027 * in (amortized) constant time, but the range query is potentially expensive.
028 * 
029 * @author Florian Deissenboeck
030 */
031public class SortedCounterSet<E extends Comparable<? super E>> extends CounterSet<E> {
032
033        /**
034         * The comparator used to define the order of the elements. This may be
035         * <code>null</code>.
036         */
037        private final Comparator<? super E> comparator;
038
039        /**
040         * Create new counter array that orders its elements by the natural order.
041         */
042        public SortedCounterSet() {
043                comparator = null;
044        }
045
046        /**
047         * Create new counter array with specific comparator to define order.
048         */
049        public SortedCounterSet(Comparator<? super E> comparator) {
050                this.comparator = comparator;
051        }
052
053        /**
054         * Obtain the sum of all values in a certain range of elements. This
055         * operation runs in time <code>O(n*log(n))</code> where <code>n</code> is
056         * the size of this set, as the list of items is sorted each time.
057         * 
058         * @param firstElement
059         *            the first element to include. If the element is not present in
060         *            the list, the smallest element greater than this one is used.
061         * @param lastElement
062         *            the last element to include. If the element is not present in
063         *            the list, the largest element smaller than this one is used.
064         */
065        public int getRangeSum(E firstElement, E lastElement) {
066
067                List<E> elementList = new ArrayList<E>(getKeys());
068
069                // if the list is empty the sum must be zero
070                if (elementList.isEmpty()) {
071                        return 0;
072                }
073
074                // this uses natural ordering if compartor is null
075                Collections.sort(elementList, comparator);
076
077                // determine first index in the list
078                int firstIndex;
079                // this uses natural ordering if compartor is null
080                firstIndex = Collections.binarySearch(elementList, firstElement, comparator);
081                // see API documentation of Collections.binarySearch
082                if (firstIndex < 0) {
083                        firstIndex = (firstIndex + 1) * (-1);
084                }
085
086                // determine last index in the lst
087                int lastIndex;
088                lastIndex = Collections.binarySearch(elementList, lastElement, comparator);
089
090                // see API documentation of Collections.binarySearch
091                if (lastIndex < 0) {
092                        lastIndex = (lastIndex + 2) * (-1);
093                }
094
095                int sum = 0;
096                // iterate over the range and add values
097                for (int i = firstIndex; i <= lastIndex; i++) {
098                        sum += getValue(elementList.get(i));
099                }
100
101                return sum;
102        }
103}
