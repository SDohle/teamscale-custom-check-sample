017package org.conqat.lib.commons.datamining;
018
019import java.util.Set;
020
021/**
022 * A frequent item set consists of a set of items and an associated support
023 * value between 0..1 indicating in what fraction of all baskets the frequent
024 * item set occurs.
025 */
026public class FrequentItemSet<T> {
027
028        /** The items */
029        private final Set<T> items;
030
031        /** The confidence */
032        private final double support;
033
034        /** Constructor */
035        public FrequentItemSet(Set<T> items, double support) {
036                this.items = items;
037                this.support = support;
038        }
039
040        /** Returns the items. */
041        public Set<T> getItems() {
042                return items;
043        }
044
045        /** Returns the support [0..1]. */
046        public double getSupport() {
047                return support;
048        }
049
050        /** {@inheritDoc} */
051        @Override
052        public String toString() {
053                return items.toString() + "(" + support + ")";
054        }
055}
