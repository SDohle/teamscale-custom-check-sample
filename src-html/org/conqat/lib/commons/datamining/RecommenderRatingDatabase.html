017package org.conqat.lib.commons.datamining;
018
019import java.io.Serializable;
020import java.util.HashSet;
021import java.util.Set;
022
023import org.conqat.lib.commons.collections.CollectionUtils;
024import org.conqat.lib.commons.collections.ListMap;
025import org.conqat.lib.commons.collections.UnmodifiableSet;
026
027/**
028 * Binary rating database. Rating is either 'like' or 'unrated'.
029 * 
030 * TODO (BH): Do not understand the comment. The terms 'rating' and 'unrated'
031 * are not used once in the code. TODO(BH): I do not like the term "rating"
032 * here, as in this library we mean by rating the ConQAT rating (RED, GREEN,
033 * YELLOW). TODO (BH): The entire class seems to be just a thin wrapper around a
034 * ListMap (maybe better a SetMap, as getter returns set?) with changed names
035 * and a static calculation method. What is the benefit of using this class over
036 * just returning a SetMap from the static method?
037 * 
038 * TODO(LH) The term rating is a fixed term in the area of recommendation
039 * system. Therefore I would really like to keep it. I used the more specific
040 * term 'recommender rating' to distinguish to LeVD rating.
041 * 
042 * TODO (LH) Also I'd like to keep this wrapper as the methods have speeking
043 * names and thus improving comprehensibility IMHO.
044 */
045public class RecommenderRatingDatabase<T> implements Serializable {
046
047        /** Serial ID */
048        private static final long serialVersionUID = 1L;
049
050        /** The data */
051        private final ListMap<IRecommenderUser, T> data = new ListMap<IRecommenderUser, T>();
052
053        /** Returns all users contained in this database. */
054        public UnmodifiableSet<IRecommenderUser> getUsers() {
055                return CollectionUtils.asUnmodifiable(data.getKeys());
056        }
057
058        /** Adds the given users to this rating database. */
059        public void add(IRecommenderUser user, Set<T> likedItems) {
060                data.addAll(user, likedItems);
061        }
062
063        /** Removes the user completely */
064        public void remove(IRecommenderUser user) {
065                data.removeCollection(user);
066        }
067
068        /**
069         * Returns the items liked by the given user. If no entry for this user is
070         * found an IllegalArgumentException is thrown.
071         */
072        public Set<T> getLikedItems(IRecommenderUser user) {
073                if (!data.containsCollection(user)) {
074                        throw new IllegalArgumentException("No such user: " + user);
075                }
076                return new HashSet<T>(data.getCollection(user));
077        }
078
079        /**
080         * Constructs a new {@link RecommenderRatingDatabase} from the given
081         * shopping baskets
082         */
083        public static <T> RecommenderRatingDatabase<T> fromShoppingBaskets(Set<Set<T>> shoppingBaskets) {
084                RecommenderRatingDatabase<T> ratingDatabase = new RecommenderRatingDatabase<T>();
085                for (Set<T> basket : shoppingBaskets) {
086                        ratingDatabase.add(new ShoppingBasketUser<T>(basket), basket);
087                }
088                return ratingDatabase;
089        }
090
091        /** {@inheritDoc} */
092        @Override
093        public String toString() {
094                return data.toString();
095        }
096
097}
