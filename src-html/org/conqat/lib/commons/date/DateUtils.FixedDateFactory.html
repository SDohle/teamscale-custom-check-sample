045        /** The number of milliseconds per second */
046        public static final long MILLIS_PER_SECOND = 1000;
047
048        /** The number of milliseconds per minute */
049        public static final long MILLIS_PER_MINUTE = 1000 * 60;
050
051        /** The number of milliseconds per hour */
052        public static final long MILLIS_PER_HOUR = MILLIS_PER_MINUTE * 60;

054        /** The number of milliseconds per day */
055        public static final long MILLIS_PER_DAY = MILLIS_PER_HOUR * 24;

057        /** Simple date format, which matches the UI format */

067        /** Returns the latest date in a collection of dates */
068        public static Date getLatest(Collection<Date> dates) {
069                if (dates.isEmpty()) {
070                        return null;
071                }
072                return Collections.max(dates);
073        }
074
075        /** Returns the earliest date in a collection of dates */
076        public static Date getEarliest(Collection<Date> dates) {
077                if (dates.isEmpty()) {
078                        return null;
079                }
080                return Collections.min(dates);
081        }
082
083        /**
084         * Returns the earlier of two dates, or null, if one of the dates is null
085         */
086        public static Date min(Date d1, Date d2) {
087                if (d1 == null || d2 == null) {
088                        return null;
089                }
090
091                if (d1.compareTo(d2) < 0) {
092                        return d1;
093                }
094                return d2;
095        }
096
097        /** Returns the later of two dates or null, if one of the dates is null. */
098        public static Date max(Date d1, Date d2) {
099                if (d1 == null || d2 == null) {
100                        return null;
101                }
102
103                if (d2.compareTo(d1) > 0) {
104                        return d2;
105                }
106                return d1;
107        }

109        /**
110         * Retrieves the date which is exactly the given number of days before now.
111         */
112        public static Date getDateBefore(int days) {
113                Calendar calendar = Calendar.getInstance();
114                calendar.add(Calendar.DAY_OF_YEAR, -days);
115                return calendar.getTime();
116        }

118        /**
119         * Returns the current time as a {@link Date} object. This is preferred to
120         * directly calling the {@link Date#Date()} constructor, as this method provides
121         * a central entry point that allows the notion of time to be tweaked (e.g. for
122         * testing, when we want a certain date to be returned).
123         *
124         * The behavior of this method can be affected either by calling the
125         * {@link #testcode_setNowFactory(IFactory)} method, or by providing a fixed
126         * date in format "yyyyMMddHHmmss" as the system property
127         * "org.conqat.lib.commons.date.now".
128         */
129        public static synchronized Date getNow() {
130                if (nowFactory == null) {
131                        String property = System.getProperty("org.conqat.lib.commons.date.now");
132                        if (property == null) {
133                                nowFactory = new CurrentDateFactory();
134                        } else {
135                                try {
136                                        nowFactory = new FixedDateFactory(new SimpleDateFormat("yyyyMMddHHmmss").parse(property));
137                                } catch (ParseException e) {
138                                        // fail hard in case of misconfiguration
139                                        throw new RuntimeException("Invalid date string provided via system property: " + property, e);
140                                }
141                        }
142                }
143                return nowFactory.create();
144        }

146        /**
147         * {@link #getNow()} truncated to the begin of day.
148         */
149        public static Date getNowWithoutTime() {
150                return truncateToBeginOfDay(getNow());

189        /** A factory returning a fixed date. */
190        public static class FixedDateFactory implements IFactory<Date, NeverThrownRuntimeException> {
191
192                /** The date used as now. */
193                private final Date now;
194
195                /** Constructor. */
196                public FixedDateFactory(Date now) {
197                        this.now = now;
198                }
199
200                /** {@inheritDoc} */
201                @Override
202                public Date create() throws NeverThrownRuntimeException {
203                        return (Date) now.clone();
204                }
205        }
206
207        /** A factory returning the current date. */
208        public static class CurrentDateFactory implements IFactory<Date, NeverThrownRuntimeException> {
209
210                /** {@inheritDoc} */
211                @Override
212                public Date create() throws NeverThrownRuntimeException {
213                        return new Date();
214                }
215        }

218        public static Date incrementByOneDay(Date date) {

222        /**
223         * Returns a normalized version of the given date. Normalization is done by
224         * removing all time-of-day information.
225         */
226        public static Date truncateToBeginOfDay(Date date) {
227                if (date == null) {
228                        return null;
229                }
230
231                synchronized (YYYY_MM_DD_FORMAT) {
232                        String normalized = YYYY_MM_DD_FORMAT.format(date);
233                        try {
234                                return YYYY_MM_DD_FORMAT.parse(normalized);
235                        } catch (ParseException e) {
236                                throw new AssertionError("ParseException.", e);
237                        }
238                }
239        }

245                synchronized (YYYY_MM_DD_FORMAT) {
246                        return YYYY_MM_DD_FORMAT.format(new Date(timestamp));
247                }

265        /**
266         * Returns the difference between two dates in the given time unit.
267         */
268        public static long diff(Date earlier, Date later, TimeUnit timeUnit) {
269                long diffInMillies = later.getTime() - earlier.getTime();
270                return timeUnit.convert(diffInMillies, TimeUnit.MILLISECONDS);
271        }

280        /** Create a new date and add <code>days</code> to it. */
281        public static Date addDaysToDate(Date date, int days) {
282                Calendar calendar = Calendar.getInstance();
283                calendar.setTime(date);
284                calendar.add(Calendar.DAY_OF_YEAR, days);
285                return calendar.getTime();
286        }

288        /** Create a new date and subtracts <code>days</code> from it. */
289        public static Date subtractDaysFromDate(Date date, int days) {
290                return addDaysToDate(date, -days);
291        }

293        /**
294         * Get an ordered list of dates that includes <code>startDate</code>, date
295         * points in between start and end date and the <code>endDate</code> if
296         * <code>includeEndDate</code> is true.
297         */
298        public static List<Date> getDatePointsInTimespan(Date startDate, Date endDate, int daysBetweenDatePoints,
299                        boolean includeEndDate) {
300                CCSMAssert.isNotNull(startDate);
301                CCSMAssert.isNotNull(endDate);
302                CCSMAssert.isFalse(startDate.after(endDate), "startDate is after endDate");
303                CCSMAssert.isTrue(daysBetweenDatePoints > 0, "daysBetweenDatePoints must be > 0");
304
305                List<Date> result = new ArrayList<>();
306                result.add(startDate);
307
308                Date currentDate = startDate;
309
310                while (true) {
311                        currentDate = addDaysToDate(currentDate, daysBetweenDatePoints);
312
313                        if (!currentDate.before(endDate)) {
314                                break;
315                        }
316
317                        result.add(currentDate);
318                }
319
320                if (includeEndDate) {
321                        result.add(endDate);
322                }
323
324                return result;
325        }

338        /**
339         * Create a date.
340         *
341         * @param month
342         *            zero-based index
343         */
344        public static Date createDate(int day, int month, int year) {
345                if (month == 12) {
346                        throw new IllegalArgumentException("month is zero-based, use the Calendar instances");
347                }
348
349                Calendar calendar = Calendar.getInstance();
350                calendar.set(year, month, day);
351                return DateUtils.truncateToBeginOfDay(calendar.getTime());
352        }

354        /**
355         * Takes a <code>timestamp</code> in milliseconds, creates a date based on this
356         * timestamp and returns the UI formatted String representation of this date.
357         * This used a fixed English locale.
358         */
359        public static String getUiFormattedDateString(long milliseconds) {
360                Date date = new Date(milliseconds);
361                synchronized (MMM_DD_YYYY_HH_MM_FORMAT) {
362                        return MMM_DD_YYYY_HH_MM_FORMAT.format(date);
363                }
364        }

366        /**
367         * Takes a <code>timestamp</code> in milliseconds and returns formatted string
368         * representation of date using given pattern. The pattern format is the one
369         * used in Java {@link DateTimeFormatter}. Uses English locale.
370         */
371        public static String formatDate(long timestamp, String pattern) {
372                LocalDateTime localDateTime = LocalDateTime.ofInstant(Instant.ofEpochMilli(timestamp), ZoneId.systemDefault());
373                DateTimeFormatter formatter = DateTimeFormatter.ofPattern(pattern, Locale.ENGLISH);
374                return localDateTime.format(formatter);
375        }

377        /** Formats a duration in a human-readable way (e.g. 1h 20min 5s 123ms). */
378        public static String formatDurationHumanReadable(Duration duration) {
379                long milliseconds = duration.toMillis() % 1000;
380                long seconds = (duration.toMillis() / 1000) % 60;
381                long minutes = (duration.toMillis() / 1000 / 60) % 60;
382                long hours = (duration.toMillis() / 1000 / 60 / 60);
383
384                StringBuilder builder = new StringBuilder();
385                if (hours > 0) {
386                        builder.append(hours).append("h ");
387                }
388                if (hours > 0 || minutes > 0) {
389                        builder.append(minutes).append("min ");
390                }
391                if (hours > 0 || minutes > 0 || seconds > 0) {
392                        builder.append(seconds).append("s ");
393                }
394                builder.append(milliseconds).append("ms");
395                return builder.toString();
396        }
