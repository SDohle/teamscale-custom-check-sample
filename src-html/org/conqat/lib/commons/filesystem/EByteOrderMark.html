017package org.conqat.lib.commons.filesystem;
018
019import java.util.Arrays;
020
021import org.conqat.lib.commons.assertion.CCSMAssert;
022import org.conqat.lib.commons.io.ByteArrayUtils;
023
024/**
025 * Enumeration of the UTF byte order marks (BOM). The actual values are taken
026 * from http://unicode.org/faq/utf_bom.html
027 * <p>
028 * The order of the values in this enum is chosen such that BOMs that are a
029 * prefix of other BOMs are at the end, i.e. UTF-32 is before UTF-16. This way
030 * we can check the BOM prefix in the order of the enum values' appearance.
031 * 
032 * @author hummelb
033 */
034public enum EByteOrderMark {
035
036        /** UTF-32 with big endian encoding. */
037        UTF_32BE("UTF-32BE", new byte[] { 0x00, 0x00, (byte) 0xFE, (byte) 0xFF }),
038
039        /** UTF-32 with little endian encoding. */
040        UTF_32LE("UTF-32LE", new byte[] { (byte) 0xFF, (byte) 0xFE, 0x00, 0x00 }),
041
042        /** UTF-16 with big endian encoding. */
043        UTF_16BE("UTF-16BE", new byte[] { (byte) 0xFE, (byte) 0xFF }),
044
045        /** UTF-16 with little endian encoding. */
046        UTF_16LE("UTF-16LE", new byte[] { (byte) 0xFF, (byte) 0xFE }),
047
048        /**
049         * UTF-8. Note that for UTF-8 the endianess is not relevant and that the BOM
050         * is optional.
051         */
052        UTF_8_BOM("UTF-8", new byte[] { (byte) 0xEF, (byte) 0xBB, (byte) 0xBF });
053
054        /** The maximal length of a BOM. */
055        public static final int MAX_BOM_LENGTH = 4;
056
057        /** The name of the encoding */
058        private final String encoding;
059
060        /** The byte order mark. */
061        private final byte[] bom;
062
063        /** Constructor. */
064        private EByteOrderMark(String encoding, byte[] bom) {
065                this.encoding = encoding;
066                CCSMAssert.isTrue(bom.length <= MAX_BOM_LENGTH, "Inconsistent max BOM length!");
067                this.bom = bom;
068        }
069
070        /** Returns the encoding. */
071        public String getEncoding() {
072                return encoding;
073        }
074
075        /**
076         * Returns the byte order mark. This returns a copy, so the array may be
077         * modified.
078         */
079        public byte[] getBOM() {
080                return Arrays.copyOf(bom, bom.length);
081        }
082
083        /** Returns the size of the BOM in bytes. */
084        public int getBOMLength() {
085                return bom.length;
086        }
087
088        /**
089         * This method checks the start of the provided data array to find a BOM. If
090         * a BOM is found, the corresponding enum value is returned. Otherwise,
091         * <code>null</code> is returned. If possible, the provided data should at
092         * least be of size {@value #MAX_BOM_LENGTH}. Otherwise the encoding might
093         * not be detected correctly. However, the method also works with shorter
094         * arrays (e.g. if a file consists of only 3 bytes).
095         */
096        public static EByteOrderMark determineBOM(byte[] data) {
097                for (EByteOrderMark bom : values()) {
098                        if (ByteArrayUtils.isPrefix(bom.bom, data)) {
099                                return bom;
100                        }
101                }
102                return null;
103        }
104}
