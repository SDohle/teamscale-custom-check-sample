111        /**
112         * Copy an input stream to an output stream. This does <em>not</em> close the
113         * streams.
114         *
115         * @param input
116         *            input stream
117         * @param output
118         *            output stream
119         * @return number of bytes copied
120         * @throws IOException
121         *             if an IO exception occurs.
122         */
123        public static int copy(InputStream input, OutputStream output) throws IOException {
124                byte[] buffer = new byte[1024];
125                int size = 0;
126                int len;
127                while ((len = input.read(buffer)) > 0) {
128                        output.write(buffer, 0, len);
129                        size += len;
130                }
131                return size;
132        }

134        /** Copy a file. This creates all necessary directories. */
135        @SuppressWarnings("resource")
136        public static void copyFile(File sourceFile, File targetFile) throws IOException {
137
138                if (sourceFile.getAbsoluteFile().equals(targetFile.getAbsoluteFile())) {
139                        throw new IOException("Can not copy file onto itself: " + sourceFile);
140                }
141
142                ensureParentDirectoryExists(targetFile);
143
144                FileChannel sourceChannel = null;
145                FileChannel targetChannel = null;
146                try {
147                        sourceChannel = new FileInputStream(sourceFile).getChannel();
148                        targetChannel = new FileOutputStream(targetFile).getChannel();
149                        sourceChannel.transferTo(0, sourceChannel.size(), targetChannel);
150                } finally {
151                        close(sourceChannel);
152                        close(targetChannel);
153                }
154        }

161        /**
162         * Copy all files specified by a file filter from one directory to another. This
163         * automatically creates all necessary directories.
164         *
165         * @param fileFilter
166         *            filter to specify file types. If all files should be copied, use
167         *            {@link FileOnlyFilter}.
168         * @return number of files copied
169         */
170        public static int copyFiles(File sourceDirectory, File targetDirectory, FileFilter fileFilter) throws IOException {
171                List<File> files = FileSystemUtils.listFilesRecursively(sourceDirectory, fileFilter);
172
173                int fileCount = 0;
174                for (File sourceFile : files) {
175                        if (sourceFile.isFile()) {
176                                String path = sourceFile.getAbsolutePath();
177                                int index = sourceDirectory.getAbsolutePath().length();
178                                String newPath = path.substring(index);
179                                File targetFile = new File(targetDirectory, newPath);
180                                copyFile(sourceFile, targetFile);
181                                fileCount++;
182                        }
183                }
184                return fileCount;
185        }

187        /**
188         * Create jar file from all files in a directory.
189         *
190         * @param filter
191         *            filter to specify file types. If all files should be copied, use
192         *            {@link FileOnlyFilter}.
193         * @return number of files added to the jar file
194         */
195        public static int createJARFile(File jarFile, File sourceDirectory, FileFilter filter) throws IOException {
196                JarOutputStream out = null;
197                int fileCount = 0;
198
199                try {
200                        out = new JarOutputStream(new FileOutputStream(jarFile));
201
202                        for (File file : FileSystemUtils.listFilesRecursively(sourceDirectory, filter)) {
203                                if (!file.isFile()) {
204                                        continue;
205                                }
206
207                                FileInputStream in = null;
208                                fileCount += 1;
209                                try {
210                                        // works for forward slashes only
211                                        String entryName = normalizeSeparators(
212                                                        file.getAbsolutePath().substring(sourceDirectory.getAbsolutePath().length() + 1));
213                                        out.putNextEntry(new ZipEntry(entryName));
214
215                                        in = new FileInputStream(file);
216                                        copy(in, out);
217                                        out.closeEntry();
218                                } finally {
219                                        close(in);
220                                }
221                        }
222                } finally {
223                        close(out);
224                }
225
226                return fileCount;
227        }

229        /**
230         * Returns a string describing the relative path to the given directory. If
231         * there is no relative path, as the directories do not share a common parent,
232         * the absolute path is returned.
233         *
234         * @param path
235         *            the path to convert to a relative path (must describe an existing
236         *            directory)
237         * @param relativeTo
238         *            the anchor (must describe an existing directory)
239         * @return a relative path
240         * @throws IOException
241         *             if creation of canonical pathes fails.
242         */
243        public static String createRelativePath(File path, File relativeTo) throws IOException {
244                CCSMAssert.isNotNull(path, "Path must not be null!");
245                CCSMAssert.isNotNull(relativeTo, "relativeTo must not be null!");
246
247                if (!path.isDirectory() || !relativeTo.isDirectory()) {
248                        throw new IllegalArgumentException("Both arguments must be existing directories!");
249                }
250                path = path.getCanonicalFile();
251                relativeTo = relativeTo.getCanonicalFile();
252
253                Set<File> parents = new HashSet<>();
254                File f = path;
255                while (f != null) {
256                        parents.add(f);
257                        f = f.getParentFile();
258                }
259
260                File root = relativeTo;
261                while (root != null && !parents.contains(root)) {
262                        root = root.getParentFile();
263                }
264
265                if (root == null) {
266                        // no common root, so use full path
267                        return path.getAbsolutePath();
268                }
269
270                String result = "";
271                while (!path.equals(root)) {
272                        result = path.getName() + "/" + result;
273                        path = path.getParentFile();
274                }
275                while (!relativeTo.equals(root)) {
276                        result = "../" + result;
277                        relativeTo = relativeTo.getParentFile();
278                }
279
280                return result;
281        }

283        /**
284         * Recursively delete directories and files. This method ignores the return
285         * value of delete(), i.e. if anything fails, some files might still exist.
286         */
287        public static void deleteRecursively(File directory) {
288
289                if (directory == null) {
290                        throw new IllegalArgumentException("Directory may not be null.");
291                }
292                File[] filesInDirectory = directory.listFiles();
293                if (filesInDirectory == null) {
294                        throw new IllegalArgumentException(directory.getAbsolutePath() + " is not a valid directory.");
295                }
296
297                for (File entry : filesInDirectory) {
298                        if (entry.isDirectory()) {
299                                deleteRecursively(entry);
300                        }
301                        entry.delete();
302                }
303                directory.delete();
304        }

432         * Lists the names of all simple files (i.e. no directories) next to an URL. For
433         * example for a file, this would return the names of all files in the same
434         * directory (including the file itself). Currently, this supports both file and
435         * jar URLs. The intended use-case is to list a set of files in a package via
436         * the class loader.
437         */
438        public static List<String> listFilesInSameLocationForURL(URL baseUrl, boolean includeSubfolders)
439                        throws IOException {
440
441                String protocol = baseUrl.getProtocol();
442
443                if ("file".equals(protocol)) {
444                        return listFilesForFileURL(baseUrl, includeSubfolders);
445                }
446
447                if ("jar".equals(protocol)) {
448                        return listFilesForJarURL(baseUrl, includeSubfolders);
449                }
450
451                throw new IOException("Unsupported protocol: " + protocol);
452        }

454        /**
455         * Lists the names of files for a JAR URL. If the includeSubfolders flag is set,
456         * all files from the Jar file will be listed, otherwise only the ones in the
457         * same location.
458         */
459        private static List<String> listFilesForJarURL(URL baseUrl, boolean includeSubfolders) throws IOException {
460                JarFile jarFile = null;
461                try {
462
463                        jarFile = new JarFile(FileSystemUtils.extractJarFileFromJarURL(baseUrl));
464
465                        List<String> names = new ArrayList<>();
466
467                        if (includeSubfolders) {
468                                jarFile.stream().forEach(entry -> {
469                                        if (!entry.isDirectory()) {
470                                                names.add(entry.getName());
471                                        }
472                                });
473                                return names;
474                        }
475
476                        // in JAR URLs we can rely on the separator being a slash
477                        String parentPath = StringUtils.getLastPart(baseUrl.getPath(), '!');
478                        parentPath = StringUtils.stripSuffix(parentPath, StringUtils.getLastPart(parentPath, UNIX_SEPARATOR));
479                        parentPath = StringUtils.stripPrefix(parentPath, "/");
480
481                        Enumeration<JarEntry> entriesEnumerator = jarFile.entries();
482                        while (entriesEnumerator.hasMoreElements()) {
483                                JarEntry entry = entriesEnumerator.nextElement();
484
485                                String simpleName = StringUtils.getLastPart(entry.getName(), UNIX_SEPARATOR);
486                                String entryPath = StringUtils.stripSuffix(entry.getName(), simpleName);
487
488                                if (!entry.isDirectory() && entryPath.equals(parentPath)) {
489                                        names.add(simpleName);
490                                }
491                        }
492                        return names;
493                } finally {
494                        FileSystemUtils.close(jarFile);
495                }
496        }

498        /**
499         * Lists the names of files within the given file URL. This will only include
500         * subfolder (recursively) if the includeSubfolders flag is set.
501         *
502         * @return a list of relative, separator-normalized file paths without slash
503         *         prefix, e.g. ['foo.java', 'subfolder/bar.java']
504         */
505        private static List<String> listFilesForFileURL(URL baseUrl, boolean includeSubfolders) throws IOException {
506                try {
507                        File directory = new File(baseUrl.toURI()).getParentFile();
508                        if (!directory.isDirectory()) {
509                                throw new IOException("Parent directory does not exist or is not readable for " + baseUrl);
510                        }
511
512                        if (includeSubfolders) {
513                                return map(listFilesRecursively(directory), file -> {
514                                        String relativeFilePath = StringUtils.stripPrefix(file.getAbsolutePath(),
515                                                        directory.getAbsolutePath());
516                                        relativeFilePath = FileSystemUtils.normalizeSeparators(relativeFilePath);
517                                        return StringUtils.stripPrefix(relativeFilePath, String.valueOf(UNIX_SEPARATOR));
518                                });
519                        }
520
521                        List<String> names = new ArrayList<>();
522                        for (File file : directory.listFiles()) {
523                                if (file.isFile()) {
524                                        names.add(file.getName());
525                                }
526                        }
527                        return names;
528                } catch (URISyntaxException e) {
529                        throw new IOException("Could not convert URL to valid file: " + baseUrl, e);
530                }
531        }

533        /**
534         * Extract all top-level classes in the given JAR and returns a list of their
535         * fully qualified class names. Inner classes are ignored.
536         */
537        public static List<String> listTopLevelClassesInJarFile(File jarFile) throws IOException {
538                List<String> result = new ArrayList<>();
539                PathBasedContentProviderBase provider = PathBasedContentProviderBase.createProvider(jarFile);
540                Collection<String> paths = provider.getPaths();
541                for (String path : paths) {
542                        if (path.endsWith(ClassPathUtils.CLASS_FILE_SUFFIX) && !path.contains("$")) {
543                                String fqn = StringUtils.removeLastPart(path, '.');
544                                fqn = fqn.replace(UNIX_SEPARATOR, '.');
545                                result.add(fqn);
546                        }
547                }
548                return result;
549        }

593        /**
594         * Constructs a file from a base file by appending several path elements.
595         * Insertion of separators is performed automatically as needed. This is similar
596         * to the constructor {@link File#File(File, String)} but allows to define
597         * multiple child levels.
598         *
599         * @param pathElements
600         *            list of elements. If this is empty, the parent is returned.
601         * @return the new file.
602         */
603        public static File newFile(File parentFile, String... pathElements) {
604                if (pathElements.length == 0) {
605                        return parentFile;
606                }
607
608                File child = new File(parentFile, pathElements[0]);
609
610                String[] remainingElements = new String[pathElements.length - 1];
611                System.arraycopy(pathElements, 1, remainingElements, 0, pathElements.length - 1);
612                return newFile(child, remainingElements);
613        }

692        public static byte[] readFileBinary(File file) throws FileNotFoundException, IOException {
693                FileInputStream in = new FileInputStream(file);
694
695                byte[] buffer = new byte[(int) file.length()];
696                ByteBuffer byteBuffer = ByteBuffer.wrap(buffer);
697
698                FileChannel channel = in.getChannel();
699                try {
700                        int readSum = 0;
701                        while (readSum < buffer.length) {
702                                int read = channel.read(byteBuffer);
703                                if (read < 0) {
704                                        throw new IOException("Reached EOF before entire file could be read!");
705                                }
706                                readSum += read;
707                        }
708                } finally {
709                        close(channel);
710                        close(in);
711                }
712
713                return buffer;
714        }

756        /**
757         * Extract entries of a ZipFile to a directory when the ZipFile is created
758         * externally. Note that this does not close the ZipFile, so the caller has to
759         * take care of this.
760         */
761        public static List<String> unzip(ZipFile zip, File targetDirectory, String prefix) throws IOException {
762                Enumeration<? extends ZipArchiveEntry> entries = zip.getEntries();
763                List<String> extractedPaths = new ArrayList<>();
764
765                while (entries.hasMoreElements()) {
766                        ZipArchiveEntry entry = entries.nextElement();
767                        if (entry.isDirectory()) {
768                                continue;
769                        }
770                        String fileName = entry.getName();
771                        if (!StringUtils.isEmpty(prefix)) {
772                                if (!fileName.startsWith(prefix)) {
773                                        continue;
774                                }
775                                fileName = StringUtils.stripPrefix(fileName, prefix);
776                        }
777
778                        try (InputStream entryStream = zip.getInputStream(entry)) {
779                                File file = new File(targetDirectory, fileName);
780                                ensureParentDirectoryExists(file);
781
782                                try (FileOutputStream outputStream = new FileOutputStream(file)) {
783                                        copy(entryStream, outputStream);
784                                }
785                        }
786                        extractedPaths.add(fileName);
787                }
788                return extractedPaths;
789        }

975        /**
976         * Returns a reader that wraps the given input stream. This method handles the
977         * BOM transparently. As the normal reader constructors can not deal with this,
978         * direct construction of readers is discouraged.
979         */
980        public static Reader streamReader(InputStream in, Charset encoding) throws IOException {
981                // we need marking to read the BOM mark
982                if (!in.markSupported()) {
983                        in = new BufferedInputStream(in);
984                }
985
986                in.mark(EByteOrderMark.MAX_BOM_LENGTH);
987                byte[] prefix = new byte[EByteOrderMark.MAX_BOM_LENGTH];
988
989                EByteOrderMark bom = null;
990                try {
991                        safeRead(in, prefix);
992                        bom = EByteOrderMark.determineBOM(prefix).orElse(null);
993                } catch (IOException e) {
994                        // just use provided encoding; keep BOM as null
995                }
996
997                in.reset();
998
999                if (bom != null) {
1000                        encoding = bom.getEncoding();
1001
1002                        // consume BOM
1003                        for (int i = 0; i < bom.getBOMLength(); ++i) {
1004                                in.read();
1005                        }
1006                }
1007
1008                return new InputStreamReader(in, encoding);
1009        }

1023        /**
1024         * Determines the root directory from a collection of files. The root directory
1025         * is the lowest common ancestor directory of the files in the directory tree.
1026         * <p>
1027         * This method does not require the input files to exist.
1028         *
1029         * @param files
1030         *            Collection of files for which root directory gets determined. This
1031         *            collection is required to contain at least 2 files. If it does
1032         *            not, an AssertionError is thrown.
1033         *
1034         * @throws AssertionError
1035         *             If less than two different files are provided whereas fully
1036         *             qualified canonical names are used for comparison.
1037         *
1038         * @throws IOException
1039         *             Since canonical paths are used for determination of the common
1040         *             root, and {@link File#getCanonicalPath()} can throw
1041         *             {@link IOException}s.
1042         *
1043         * @return Root directory, or null, if the files do not have a common root
1044         *         directory.
1045         */
1046        public static File commonRoot(Iterable<? extends File> files) throws IOException {
1047                // determine longest common prefix on canonical absolute paths
1048                Set<String> absolutePaths = new HashSet<>();
1049                for (File file : files) {
1050                        absolutePaths.add(file.getCanonicalPath());
1051                }
1052
1053                CCSMAssert.isTrue(absolutePaths.size() >= 2, "Expected are at least 2 files");
1054
1055                String longestCommonPrefix = StringUtils.longestCommonPrefix(absolutePaths);
1056
1057                // trim to name of root directory (remove possible equal filename
1058                // prefixes.)
1059                int lastSeparator = longestCommonPrefix.lastIndexOf(File.separator);
1060                if (lastSeparator > -1) {
1061                        longestCommonPrefix = longestCommonPrefix.substring(0, lastSeparator);
1062                }
1063
1064                if (StringUtils.isEmpty(longestCommonPrefix)) {
1065                        return null;
1066                }
1067
1068                return new File(longestCommonPrefix);
1069        }

1073         * compressed. Otherwise the stream is just handed through. Currently the
1074         * following compression methods are supported:
1075         * <ul>
1076         * <li>GZIP via {@link GZIPInputStream}</li>
1077         * </ul>

1079        public static InputStream autoDecompressStream(InputStream in) throws IOException {
1080                if (!in.markSupported()) {

1088                        return new GZIPInputStream(in);
1089                }
1090                return in;

1175        /**
1176         * Often file URLs are created the wrong way, i.e. without proper escaping
1177         * characters invalid in URLs. Unfortunately, the URL class allows this and the
1178         * Eclipse framework does it. See
1179         * http://weblogs.java.net/blog/2007/04/25/how-convert-javaneturl-javaiofile for
1180         * details.
1181         *
1182         * This method attempts to fix this problem and create a file from it.
1183         *
1184         * @throws AssertionError
1185         *             if cleaning up fails.
1186         */
1187        private static File fromURL(String url) {
1188
1189                // We cannot simply encode the URL this also encodes slashes and other
1190                // stuff. As a result, the file constructor throws an exception. As a
1191                // simple heuristic, we only fix the spaces.
1192                // The other route to go would be manually stripping of "file:" and
1193                // simply creating a file. However, this does not work if the URL was
1194                // created properly and contains URL escapes.
1195
1196                url = url.replace(StringUtils.SPACE, "%20");
1197                try {
1198                        return new File(new URI(url));
1199                } catch (URISyntaxException e) {
1200                        throw new AssertionError("The assumption is that this method is capable of "
1201                                        + "working with non-standard-compliant URLs, too. " + "Apparently it is not. Invalid URL: " + url
1202                                        + ". Ex: " + e.getMessage(), e);
1203                }

1206        /**
1207         * Returns whether a filename represents an absolute path.
1208         *
1209         * This method returns the same result, independent on which operating system it
1210         * gets executed. In contrast, the behavior of {@link File#isAbsolute()} is
1211         * operating system specific.
1212         */
1213        public static boolean isAbsolutePath(String filename) {
1214                // Unix and MacOS: absolute path starts with slash or user home
1215                if (filename.startsWith("/") || filename.startsWith("~")) {
1216                        return true;
1217                }
1218                // Windows and OS/2: absolute path start with letter and colon
1219                if (filename.length() > 2 && Character.isLetter(filename.charAt(0)) && filename.charAt(1) == ':') {
1220                        return true;
1221                }
1222                // UNC paths (aka network shares): start with double backslash
1223                if (filename.startsWith("\\\\")) {
1224                        return true;
1225                }
1226
1227                return false;
1228        }

1255         * of bytes and stores it at position <code>offset</code>. This method blocks
1256         * until input data is available, end of file is detected, or an exception is
1257         * thrown.
1258         *
1259         * The reason for this method is that {@link InputStream#read(byte[], int, int)}
1260         * may read less than the requested number of bytes, while this method ensures
1261         * the data is complete.
1262         *
1263         * @param in
1264         *            the stream to read from.
1265         * @param data
1266         *            the stream to read from.
1267         * @param offset
1268         *            the offset in the array where the first read byte is stored.
1269         * @param length
1270         *            the length of data read.
1271         * @throws IOException
1272         *             if reading the underlying stream causes an exception.
1273         * @throws EOFException
1274         *             if the end of file was reached before the requested data was
1275         *             read.
1276         */
1277        public static void safeRead(InputStream in, byte[] data, int offset, int length) throws IOException, EOFException {
1278                while (length > 0) {
1279                        int read = in.read(data, offset, length);
1280                        if (read < 0) {
1281                                throw new EOFException("Reached end of file before completing read.");
1282                        }
1283                        offset += read;
1284                        length -= read;

1375         * Returns a file that does not exist in the same directory of the given file.
1376         * It either returns the passed file (if not exiting) or appends a number
1377         * between the filename and the extension to ensure uniqueness, e.g.
1378         * path/to/file_1.ext if path/to/file.ext is passed but this file already
1379         * exists. The number is incremented until the file does not exist.
1380         */
1381        public static File getNonExistingFile(File file) {
1382                if (!file.exists()) {
1383                        return file;
1384                }
1385
1386                String extensionlessName = getFilenameWithoutExtension(file);
1387                int suffix = 0;
1388                String extension = FileSystemUtils.getFileExtension(file);
1389
1390                do {
1391                        file = new File(file.getParentFile(), extensionlessName + "_" + ++suffix + "." + extension);
1392                } while (file.exists());
1393
1394                return file;
1395        }

1397        /**
1398         * Converts the given human readable data size to the corresponding number of
1399         * bytes. For example "1 KB" is converted to 1024. Also supports Si units ("1
1400         * KiB" is converted to 1000).
1401         *
1402         * Commas are ignored and can be used as thousands separator. A dot is the
1403         * decimal separator. ("1.2KiB" is converted to 1200).
1404         *
1405         * Method implementation based on stackoverflow
1406         * https://stackoverflow.com/a/45860167
1407         */
1408        public static long parseDataSize(String dataSize) {
1409                String dataSizeWithoutComma = dataSize.replaceAll(",", "");
1410                int unitBeginIndex = StringUtils.indexOfMatch(dataSizeWithoutComma, DATA_SIZE_UNIT_START_PATTERN);
1411                if (unitBeginIndex == -1) {
1412                        return Long.parseLong(dataSizeWithoutComma);
1413                }
1414                double rawDataSize = Double.parseDouble(dataSizeWithoutComma.substring(0, unitBeginIndex));
1415                String unitString = dataSizeWithoutComma.substring(unitBeginIndex);
1416                int unitChar = unitString.charAt(0);
1417                int power = METRIC_SYSTEM_UNITS.indexOf(unitChar) + 1;
1418                boolean isSi = unitBeginIndex != -1 && unitString.length() >= 2 && unitString.charAt(1) == 'i';
1419
1420                int factor = 1024;
1421                if (isSi) {
1422                        factor = 1000;
1423                        if (StringUtils.stripSuffix(unitString, "B").length() != 2) {
1424                                throw new NumberFormatException("Malformed data size: " + dataSizeWithoutComma);
1425                        }
1426                } else if (power == 0) {
1427                        if (StringUtils.stripSuffix(unitString, "B").length() != 0) {
1428                                throw new NumberFormatException("Malformed data size: " + dataSizeWithoutComma);
1429                        }
1430                } else {
1431                        if (StringUtils.stripSuffix(unitString, "B").length() != 1) {
1432                                throw new NumberFormatException("Malformed data size: " + dataSizeWithoutComma);
1433                        }
1434                }
1435                return (long) (rawDataSize * Math.pow(factor, power));
1436        }
