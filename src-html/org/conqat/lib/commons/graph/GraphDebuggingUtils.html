008package org.conqat.lib.commons.graph;
009
010import java.util.ArrayList;
011import java.util.List;
012import java.util.Optional;
013
014import org.conqat.lib.commons.collections.IIdProvider;
015import org.conqat.lib.commons.collections.IdManager;
016import org.conqat.lib.commons.string.StringUtils;
017import org.conqat.lib.commons.visitor.IMeshWalker;
018import org.conqat.lib.commons.visitor.VisitorUtils;
019
020/**
021 * Useful methods for debugging arbitrary graphs.
022 */
023public class GraphDebuggingUtils {
024
025        /**
026         * Returns a string representation of the graph. The individual nodes are
027         * represented using their toString() method. This is useful for debugging.
028         * 
029         * @param root
030         *            the root node of the graph to visualize.
031         * @param walker
032         *            the mesh walker used to traverse the graph.
033         * @param optionalIdProvider
034         *            an optional IdProvider that provides integer IDs for all
035         *            Nodes. If it is empty, we will use an IdManager. The Ids are
036         *            assumed to be stable (the same on each query of the same
037         *            node).
038         */
039        public static <NodeT, X extends Exception> String toString(NodeT root, IMeshWalker<NodeT, X> walker,
040                        Optional<IIdProvider<Integer, NodeT>> optionalIdProvider) throws X {
041                return toString(root, walker, true, optionalIdProvider);
042        }
043
044        /**
045         * Returns a human-readable string representation of the graph. The
046         * individual nodes are only represented as integers. This is useful for
047         * debugging.
048         * 
049         * @param root
050         *            the root node of the graph to visualize.
051         * @param walker
052         *            the mesh walker used to traverse the graph.
053         * @param optionalIdProvider
054         *            an optional IdProvider that provides integer IDs for all
055         *            Nodes. If it is empty, we will use an IdManager. The Ids are
056         *            assumed to be stable (the same on each query of the same
057         *            node).
058         */
059        public static <NodeT, X extends Exception> String toShortString(NodeT root, IMeshWalker<NodeT, X> walker,
060                        Optional<IIdProvider<Integer, NodeT>> optionalIdProvider) throws X {
061                return toString(root, walker, false, optionalIdProvider);
062        }
063
064        /**
065         * Returns a string representation of the graph.
066         *
067         * @param root
068         *            the root node of the graph to visualize.
069         * @param walker
070         *            the mesh walker used to traverse the graph.
071         * @param detailled
072         *            if <code>true</code>, each node's toString() representation is
073         *            included in the output.
074         * @param optionalIdProvider
075         *            an optional IdProvider that provides integer IDs for all
076         *            Nodes. If it is empty, we will use an IdManager. The Ids are
077         *            assumed to be stable (the same on each query of the same
078         *            node).
079         */
080        private static <NodeT, X extends Exception> String toString(NodeT root, IMeshWalker<NodeT, X> walker,
081                        boolean detailled, Optional<IIdProvider<Integer, NodeT>> optionalIdProvider) throws X {
082                StringBuilder sb = new StringBuilder();
083                IIdProvider<Integer, NodeT> idManager = optionalIdProvider.orElse(new IdManager<NodeT>());
084                for (NodeT node : VisitorUtils.listAllDepthFirst(root, walker)) {
085                        sb.append(idManager.obtainId(node));
086
087                        if (detailled) {
088                                sb.append(": { ");
089                                sb.append(node.toString());
090                                sb.append(" }");
091                        }
092
093                        sb.append(" --> ");
094                        List<Integer> childIds = new ArrayList<Integer>();
095                        for (NodeT child : walker.getAdjacentElements(node)) {
096                                childIds.add(idManager.obtainId(child));
097                        }
098                        sb.append(StringUtils.concat(childIds, ", "));
099                        sb.append(StringUtils.CR);
100                }
101                return sb.toString();
102        }
103
104}
