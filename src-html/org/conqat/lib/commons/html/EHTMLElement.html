310        /** Constructor. */
311        private EHTMLElement(String name, EHTMLAttribute... attributes) {
312                this.name = name;
313                if (attributes.length > 0) {
314                        allowedAttributes = EnumSet.of(attributes[0], attributes);
315                } else {
316                        allowedAttributes = EnumSet.noneOf(EHTMLAttribute.class);
317                }
318        }

320        /** Returns whether the given attribute is allowed for the element. */
321        public boolean allowsAttribute(EHTMLAttribute attribute) {
322                return allowedAttributes.contains(attribute);
323        }
