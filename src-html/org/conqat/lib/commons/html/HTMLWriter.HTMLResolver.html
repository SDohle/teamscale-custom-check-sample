017package org.conqat.lib.commons.html;
018
019import static org.conqat.lib.commons.html.EHTMLAttribute.SRC;
020import static org.conqat.lib.commons.html.EHTMLAttribute.TYPE;
021import static org.conqat.lib.commons.html.EHTMLElement.SCRIPT;
022
023import java.io.File;
024import java.io.IOException;
025import java.io.OutputStream;
026import java.io.OutputStreamWriter;
027import java.io.PrintStream;
028import java.io.PrintWriter;
029import java.io.UnsupportedEncodingException;
030
031import org.conqat.lib.commons.filesystem.FileSystemUtils;
032import org.conqat.lib.commons.string.StringUtils;
033import org.conqat.lib.commons.xml.IXMLResolver;
034import org.conqat.lib.commons.xml.XMLWriter;
035
036/**
037 * This class is used for writing HTML.
038 */
039public class HTMLWriter extends XMLWriter<EHTMLElement, EHTMLAttribute> {
040
041        /** The CSS manager class used. */
042        private final CSSManagerBase cssManager;
043
044        /**
045         * Creates a new writer for HTML documents.
046         * 
047         * @param file
048         *            the file to write to.
049         * @param cssManager
050         *            the CSS manager used. If this is null, the class attributes
051         *            may be filled with simple strings.
052         */
053        public HTMLWriter(File file, CSSManagerBase cssManager) throws IOException {
054                this(new PrintStream(file, FileSystemUtils.UTF8_ENCODING), cssManager);
055        }
056
057        /**
058         * Creates a new writer for HTML documents.
059         * 
060         * @param stream
061         *            the stream to print to.
062         * @param cssManager
063         *            the CSS manager used. If this is null, the class attributes
064         *            may be filled with simple strings.
065         */
066        public HTMLWriter(OutputStream stream, CSSManagerBase cssManager) {
067                super(new PrintWriter(wrapStream(stream)), new HTMLResolver());
068                this.cssManager = cssManager;
069        }
070
071        // Fix comment finding?
072        /**
073         * Helper method for {@link #HTMLWriter(OutputStream, CSSManagerBase)} to
074         * deal with the exception.
075         */
076        private static OutputStreamWriter wrapStream(OutputStream stream) {
077                try {
078                        return new OutputStreamWriter(stream, FileSystemUtils.UTF8_ENCODING);
079                } catch (UnsupportedEncodingException e) {
080                        throw new AssertionError("UTF-8 should be supported!");
081                }
082        }
083
084        /**
085         * Creates a new writer for HTML documents.
086         * 
087         * @param writer
088         *            the writer to print to.
089         */
090        public HTMLWriter(PrintWriter writer, CSSManagerBase cssManager) {
091                super(writer, new HTMLResolver());
092                this.cssManager = cssManager;
093        }
094
095        /**
096         * This adds a default header for HTML files consisting of the XML header
097         * and a DOCTYPE of the xhtml frameset DTD.
098         * <p>
099         * XML version is set to "1.0", encoding provided by a parameter, and doc
100         * type definition to XHTML 1.0 Frameset.
101         */
102        public void addStdHeader(String encoding) {
103                addHeader("1.0", encoding);
104                addPublicDocTypeDefintion(EHTMLElement.HTML, "-//W3C//DTD XHTML 1.0 Frameset//EN",
105                                "http://www.w3.org/TR/xhtml1/DTD/xhtml1-frameset.dtd");
106        }
107
108        /**
109         * This adds a default header for HTML files consisting of the XML header
110         * and a DOCTYPE of the xhtml frameset DTD.
111         * <p>
112         * XML version is set to "1.0", encoding to "UTF-8", and doc type definition
113         * to XHTML 1.0 Frameset.
114         */
115        public void addStdHeader() {
116                addStdHeader(FileSystemUtils.UTF8_ENCODING);
117        }
118
119        /**
120         * {@inheritDoc}
121         * 
122         * Made this public here.
123         */
124        @Override
125        public void addRawString(String html) {
126                super.addRawString(html);
127        }
128
129        /**
130         * Adds a line separator with closing and open tag (see
131         * {@link #addNewLine()}.
132         */
133        public void addRawNewLine() {
134                addRawString(StringUtils.LINE_SEPARATOR);
135        }
136
137        /** Inserts a JavaScript block. */
138        public void insertJavaScript(String javaScript) {
139                openElement(SCRIPT, TYPE, "text/javascript");
140                addRawNewLine();
141                addRawString(javaScript);
142                addRawNewLine();
143                closeElement(SCRIPT);
144        }
145
146        /** Inserts a script tag that loads JavaScript from a separate file. */
147        public void addExternalJavaScript(String scriptFilePath) {
148                // this is required, as some browsers choke on a directly closed
149                // script element
150                insertEmptyElement(SCRIPT, SRC, scriptFilePath, TYPE, "text/javascript");
151        }
152
153        /**
154         * Inserts an empty element but ensures that it is not closed using the
155         * shorthand syntax (e.g. <code>&lt;div /&gt;</code>).
156         * 
157         * This is useful since some browsers choke on some shorthand elements, e.g.
158         * divs or JavaScript tags.
159         * 
160         * @see #openElement(EHTMLElement, Object...)
161         */
162        public void insertEmptyElement(EHTMLElement element, Object... attributes) {
163                openElement(element, attributes);
164                addText(StringUtils.EMPTY_STRING);
165                closeElement(element);
166        }
167
168        /** Inserts a non-breaking space. */
169        public void addNonBreakingSpace() {
170                addRawString("&nbsp;");
171        }
172
173        /**
174         * Adds an attribute to the currently open element but checks in addition if
175         * the attribute may be added at all.
176         * 
177         * @throws HTMLWriterException
178         *             if the attribute is not allowed for the current element.
179         */
180        @Override
181        public void addAttribute(EHTMLAttribute attribute, Object value) {
182                if (!getCurrentElement().allowsAttribute(attribute)) {
183                        throw new HTMLWriterException("Attribute " + attribute + " not allowed for element " + getCurrentElement());
184                }
185
186                if (attribute == EHTMLAttribute.STYLE) {
187                        assertCssDeclarationBlock(value);
188                        value = ((CSSDeclarationBlock) value).toInlineStyle();
189                } else if (cssManager != null && attribute == EHTMLAttribute.CLASS) {
190                        assertCssDeclarationBlock(value);
191                        value = cssManager.getCSSClassName((CSSDeclarationBlock) value);
192                }
193
194                super.addAttribute(attribute, value);
195        }
196
197        /**
198         * Asserts that the given value is a {@link CSSDeclarationBlock} and throws
199         * an exception otherwise.
200         */
201        private static void assertCssDeclarationBlock(Object value) {
202                if (!(value instanceof CSSDeclarationBlock)) {
203                        throw new HTMLWriterException("The argument for STYLE and CLASS attributes must be a "
204                                        + CSSDeclarationBlock.class.getSimpleName() + "!");
205                }
206        }
207
208        /** The resolver used for the {@link HTMLWriter}. */
209        public static class HTMLResolver implements IXMLResolver<EHTMLElement, EHTMLAttribute> {
210
211                /** {@inheritDoc} */
212                @Override
213                public String resolveAttributeName(EHTMLAttribute attribute) {
214                        return attribute.toString();
215                }
216
217                /** {@inheritDoc} */
218                @Override
219                public String resolveElementName(EHTMLElement element) {
220                        return element.toString();
221                }
222
223                /** {@inheritDoc} */
224                @Override
225                public Class<EHTMLAttribute> getAttributeClass() {
226                        return EHTMLAttribute.class;
227                }
228        }
229}
