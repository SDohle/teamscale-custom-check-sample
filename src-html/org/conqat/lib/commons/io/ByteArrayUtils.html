030/**
031 * Utility methods for dealing with raw byte arrays. This is located in the I/O
032 * package, as the typical application for these methods is binary I/O on byte
033 * array level.
034 */
035public class ByteArrayUtils {
036
037        /**
038         * Converts an integer value to a byte array. The returned array has a
039         * length of {@link Integer#BYTES}.
040         */
041        public static byte[] intToByteArray(int value) {
042                byte[] bytes = new byte[Integer.BYTES];
043                bytes[0] = (byte) (value >> 24);
044                bytes[1] = (byte) (value >> 16);
045                bytes[2] = (byte) (value >> 8);
046                bytes[3] = (byte) (value);
047                return bytes;
048        }
049
050        /**
051         * Converts a double value to a byte array. The returned array has a length
052         * of {@link Double#BYTES}
053         */
054        public static byte[] doubleToByteArray(double value) {
055                long longBits = Double.doubleToRawLongBits(value);
056                return ByteArrayUtils.longToByteArray(longBits);
057        }
058
059        /**
060         * Converts a long value to a byte array. The returned array has a length of
061         * {@link Long#BYTES}
062         */
063        public static byte[] longToByteArray(long value) {
064                byte[] bytes = new byte[Long.BYTES];
065                bytes[0] = (byte) (value >> 56);
066                bytes[1] = (byte) (value >> 48);
067                bytes[2] = (byte) (value >> 40);
068                bytes[3] = (byte) (value >> 32);
069                bytes[4] = (byte) (value >> 24);
070                bytes[5] = (byte) (value >> 16);
071                bytes[6] = (byte) (value >> 8);
072                bytes[7] = (byte) (value);
073                return bytes;
074        }
075
076        /**
077         * Converts a byte array to an integer value.
078         * 
079         * Overall, this method is only guaranteed to work if the input array was
080         * created by {@link #intToByteArray(int)}.
081         */
082        public static int byteArrayToInt(byte[] bytes) {
083                CCSMAssert.isTrue(bytes.length == Integer.BYTES, "bytes.length must be 4");
084                int value = 0;
085                value |= unsignedByte(bytes[0]) << 24;
086                value |= unsignedByte(bytes[1]) << 16;
087                value |= unsignedByte(bytes[2]) << 8;
088                value |= unsignedByte(bytes[3]);
089                return value;
090        }
091
092        /**
093         * Converts a byte array to a double value.
094         * 
095         * Overall, this method is only guaranteed to work if the input array was
096         * created by {@link #doubleToByteArray(double)}.
097         */
098        public static double byteArrayToDouble(byte[] value) {
099                long longBits = ByteArrayUtils.byteArrayToLong(value);
100                return Double.longBitsToDouble(longBits);
101        }
102
103        /**
104         * Converts a byte array to a long value.
105         * 
106         * Overall, this method is only guaranteed to work if the input array was
107         * created by {@link #longToByteArray(long)}.
108         */
109        public static long byteArrayToLong(byte[] bytes) {
110                CCSMAssert.isTrue(bytes.length == Long.BYTES, "bytes.length must be 8");
111                long value = 0l;
112                value |= unsignedByteAsLong(bytes[0]) << 56;
113                value |= unsignedByteAsLong(bytes[1]) << 48;
114                value |= unsignedByteAsLong(bytes[2]) << 40;
115                value |= unsignedByteAsLong(bytes[3]) << 32;
116                value |= unsignedByteAsLong(bytes[4]) << 24;
117                value |= unsignedByteAsLong(bytes[5]) << 16;
118                value |= unsignedByteAsLong(bytes[6]) << 8;
119                value |= unsignedByteAsLong(bytes[7]);
120                return value;
121        }
122
123        /**
124         * Decompresses a single byte[] using GZIP. A null input array will cause
125         * this method to return null.
126         * 
127         * @throws IOException
128         *             if the input array is not valid GZIP compressed data (as
129         *             created by {@link #compress(byte[])}).
130         */
131        public static byte[] decompress(byte[] value) throws IOException {
132                if (value == null) {
133                        return null;
134                }
135
136                ByteArrayOutputStream bos = new ByteArrayOutputStream(value.length);
137                ByteArrayInputStream bis = new ByteArrayInputStream(value);
138                GZIPInputStream gzis = new GZIPInputStream(bis);
139
140                FileSystemUtils.copy(gzis, bos);
141
142                // it does not matter if we close in case of exceptions, as these are
143                // in-memory resources
144                gzis.close();
145                bos.close();
146
147                return bos.toByteArray();
148        }
149
150        /**
151         * Compresses a single byte[] using GZIP. A null input array will cause this
152         * method to return null.
153         */
154        public static byte[] compress(byte[] value) {
155                if (value == null) {
156                        return null;
157                }
158
159                ByteArrayOutputStream bos = new ByteArrayOutputStream(value.length);
160                try {
161                        GZIPOutputStream gzos = new GZIPOutputStream(bos);
162                        gzos.write(value);
163
164                        // it does not matter if we close in case of exceptions, as this is
165                        // an in-memory resource
166                        gzos.close();
167                } catch (IOException e) {
168                        throw new AssertionError("Can not happen as we work in memory: " + e.getMessage());
169                }
170
171                return bos.toByteArray();
172        }

174        /** Returns whether the prefix is a prefix of the given key. */
175        public static boolean isPrefix(byte[] prefix, byte[] key) {
176                return isPrefix(prefix, key, 0);
177        }
178
179        /**
180         * Returns whether the <code>prefix</code> is a prefix of the given
181         * <code>key</code> when only looking at the part of <code>key</code>
182         * starting at <code>startIndex</code>.
183         */
184        public static boolean isPrefix(byte[] prefix, byte[] key, int startIndex) {
185
186                if (key.length - startIndex < prefix.length) {
187                        return false;
188                }
189                for (int i = 0; i < prefix.length; ++i) {
190                        if (prefix[i] != key[i + startIndex]) {
191                                return false;
192                        }
193                }
194                return true;
195        }

197        /** Returns true if a1 is (lexicographically) less than a2. */
198        public static boolean isLess(byte[] a1, byte[] a2, boolean resultIfEqual) {
199                int limit = Math.min(a1.length, a2.length);
200                for (int i = 0; i < limit; ++i) {
201                        if (unsignedByte(a1[i]) < unsignedByte(a2[i])) {
202                                return true;
203                        }
204                        if (unsignedByte(a1[i]) > unsignedByte(a2[i])) {
205                                return false;
206                        }
207                }
208
209                if (a1.length < a2.length) {
210                        return true;
211                }
212                if (a1.length > a2.length) {
213                        return false;
214                }
215
216                return resultIfEqual;
217        }

219        /** Returns the unsigned byte interpretation of the parameter. */
220        public static int unsignedByte(byte b) {

225        public static long unsignedByteAsLong(byte b) {

229        /** Returns the concatenation of the given arrays. */
230        public static byte[] concat(byte[]... arrays) {
231                return concat(Arrays.asList(arrays));
232        }
233
234        /** Returns the concatenation of the given arrays. */
235        public static byte[] concat(Iterable<byte[]> arrays) {
236                int length = 0;
237                for (byte[] array : arrays) {
238                        length += array.length;
239                }
240
241                byte[] result = new byte[length];
242                int start = 0;
243                for (byte[] array : arrays) {
244                        System.arraycopy(array, 0, result, start, array.length);
245                        start += array.length;
246                }
247                return result;
248        }

250        /**
251         * Creates a hex dump of the provided bytes. This is similar to output from
252         * hexdump tools and primarily used for debugging. The output string will
253         * contain in each line 16 bytes of data first printed as hex numbers and
254         * then as a string interpretation. Each line is also prefixed with an
255         * offset.
256         */
257        public static String hexDump(byte[] data) {
258                return hexDump(data, 16);
259        }
260
261        /**
262         * Creates a hex dump of the provided bytes. This is similar to output from
263         * hexdump tools and primarily used for debugging. The output string will
264         * contain in each line <code>width</code> bytes of data first printed as
265         * hex numbers and then as a string interpretation. Each line is also
266         * prefixed with an offset.
267         */
268        public static String hexDump(byte[] data, int width) {
269                CCSMAssert.isTrue(width >= 1, "Width must be positive!");
270
271                StringBuilder builder = new StringBuilder();
272                for (int i = 0; i < data.length; i += width) {
273                        hexDumpAppendLine(data, i, Math.min(data.length, i + width), width, builder);
274                }
275                return builder.toString();
276        }
277
278        /**
279         * Appends a single line to the hex dump for {@link #hexDump(byte[], int)}.
280         * The start is inclusive, the end is exclusive.
281         */
282        private static void hexDumpAppendLine(byte[] data, int startOffset, int endOffset, int width,
283                        StringBuilder builder) {
284                builder.append(String.format("%06d: ", startOffset));
285                for (int i = startOffset; i < endOffset; ++i) {
286                        builder.append(String.format("%02x ", data[i]));
287                }
288
289                if (endOffset - startOffset < width) {
290                        builder.append(StringUtils.fillString((width - (endOffset - startOffset)) * 3, StringUtils.SPACE_CHAR));
291                }
292
293                builder.append(StringUtils.SPACE_CHAR);
294                for (int i = startOffset; i < endOffset; ++i) {
295                        boolean isInPrintableAsciiRange = (33 <= data[i] && data[i] <= 126);
296                        if (isInPrintableAsciiRange) {
297                                builder.append((char) data[i]);
298                        } else {
299                                builder.append('.');
300                        }
301                }
302
303                builder.append(StringUtils.CR);
304        }

306        /**
307         * Returns whether the given bytes start with the
308         * <a href="http://en.wikipedia.org/wiki/Zip_%28file_format%29#File_headers"
309         * >magic bytes</a> that mark a ZIP file.
310         */
311        public static boolean startsWithZipMagicBytes(byte[] data) {
312                return isPrefix(new byte[] { 0x50, 0x4b, 0x03, 0x04 }, data);
313        }

315        /**
316         * Returns the first index in <code>searchIn</code> at or after the start
317         * index containing <code>searchFor</code> (or -1 if not found).
318         */
319        public static int indexOf(byte[] searchFor, byte[] searchIn, int startIndex) {
320                return indexOf(searchFor, searchIn, startIndex, searchIn.length);
321        }
322
323        /**
324         * Returns the first index in <code>searchIn</code> at or after the start
325         * index containing <code>searchFor</code> (or -1 if not found). endIndex is
326         * the index of the first byte that is not considered in the match
327         * (exclusive).
328         */
329        public static int indexOf(byte[] searchFor, byte[] searchIn, int startIndex, int endIndex) {
330                if (startIndex + searchFor.length >= endIndex) {
331                        return -1;
332                }
333
334                for (int i = startIndex; i <= endIndex - searchFor.length; ++i) {
335                        if (isPrefix(searchFor, searchIn, i)) {
336                                return i;
337                        }
338                }
339
340                return -1;
341        }
