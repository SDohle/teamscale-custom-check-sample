017package org.conqat.lib.commons.predicate;
018
019import java.util.ArrayList;
020import java.util.Collection;
021import java.util.List;
022import java.util.function.Predicate;
023import java.util.stream.Collectors;
024
025import org.conqat.lib.commons.collections.Pair;
026
027/**
028 * Utility methods for working with predicates.
029 */
030public class PredicateUtils {
031
032        /**
033         * Returns all input elements that are contained in the set described by the
034         * predicate.
035         * 
036         * @param <T>
037         *            Type in collection.
038         * 
039         * @deprecated use {@link #obtainContained(Collection, Predicate)} instead.
040         */
041        @Deprecated
042        public static <T> List<T> obtainContained(Collection<T> input, IPredicate<? super T> predicate) {
043                return input.stream().filter(predicate::isContained).collect(Collectors.toCollection(ArrayList::new));
044        }
045
046        /**
047         * Returns all input elements that are contained in the set described by the
048         * predicate.
049         * 
050         * @param <T>
051         *            Type in collection.
052         */
053        public static <T> List<T> obtainContained(Collection<T> input, Predicate<? super T> predicate) {
054                return input.stream().filter(predicate).collect(Collectors.toCollection(ArrayList::new));
055        }
056
057        /**
058         * Returns all input elements that are <b>not</b> contained in the set
059         * described by the predicate.
060         * 
061         * @deprecated use {@link #obtainNonContained(Collection, Predicate)}
062         *             instead.
063         */
064        @Deprecated
065        public static <T> List<T> obtainNonContained(Collection<T> input, IPredicate<? super T> predicate) {
066                return obtainContained(input, InvertingPredicate.create(predicate));
067        }
068
069        /**
070         * Efficiently splits the list of input elements into two lists.<br/>
071         * 
072         * The first consists of all input elements that are contained in the set
073         * described by the predicate.<br/>
074         * The second consists of all input elements that are <b>not</b> contained
075         * in the set described by the predicate.<br />
076         * 
077         * Efficiently in this case means that only one iteration through the list
078         * is needed to split into two lists.
079         * 
080         * @param input
081         *            the list of input elements.
082         * @param predicate
083         *            the {@link IPredicate} used to decide containment.
084         */
085        public static <T> Pair<List<T>, List<T>> splitContainedNonContained(Collection<T> input,
086                        IPredicate<? super T> predicate) {
087
088                List<T> contained = new ArrayList<T>();
089                List<T> nonContained = new ArrayList<T>();
090
091                for (T t : input) {
092                        if (predicate.isContained(t)) {
093                                contained.add(t);
094                        } else {
095                                nonContained.add(t);
096                        }
097                }
098                return new Pair<List<T>, List<T>>(contained, nonContained);
099        }
100
101        /**
102         * Returns all input elements that are <b>not</b> contained in the set
103         * described by the predicate.
104         */
105        public static <T> List<T> obtainNonContained(Collection<T> input, Predicate<? super T> predicate) {
106                return obtainContained(input, predicate.negate());
107        }
108
109        /**
110         * Returns a predicate containing everything.
111         * 
112         * @deprecated use {@link #createAlwaysTruePredicate()} instead.
113         */
114        @Deprecated
115        public static <T> IPredicate<T> createAllContainingPredicate() {
116                return t -> true;
117        }
118
119        /** Returns a predicate that always returns true. */
120        public static <T> Predicate<T> createAlwaysTruePredicate() {
121                return t -> true;
122        }
123}
