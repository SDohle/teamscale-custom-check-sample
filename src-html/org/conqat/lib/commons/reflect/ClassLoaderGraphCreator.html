017package org.conqat.lib.commons.reflect;
018
019import static org.conqat.lib.commons.string.StringUtils.CR;
020
021import java.util.ArrayList;
022import java.util.List;
023
024import org.conqat.lib.commons.assertion.CCSMAssert;
025import org.conqat.lib.commons.collections.IdentityHashSet;
026import org.conqat.lib.commons.collections.UniqueIdManager;
027import org.conqat.lib.commons.color.ECCSMColor;
028import org.conqat.lib.commons.string.StringUtils;
029
030/**
031 * Create a DOT graph that contains the provided object, their defining classes,
032 * the classes' class loaders and the parent class loaders up to the bootstrap
033 * class loader. This is very useful for debugging class loader-related bugs.
034 * 
035 * @author deissenb
036 */
037public class ClassLoaderGraphCreator {
038
039        /** The header for DOT files. */
040        public final static String HEADER = "digraph G {" + CR + "  edge [  fontname = \"Helvetica\"," + CR
041                        + "          color = \"#639CCE\", fontsize = 8 ];" + CR + "  node [  shape = polygon," + CR
042                        + "          sides = 4," + CR + "          color = \"#639CCE\"," + CR
043                        + "          fontname = \"Helvetica\"," + CR + "          fontsize    = 9," + CR + "          height=0.25];"
044                        + CR;
045
046        /** Manager to create unique node ids. */
047        private final UniqueIdManager<Object> idManager = new UniqueIdManager<Object>();
048
049        /** Objects included in the graph. */
050        private final IdentityHashSet<Object> objects = new IdentityHashSet<Object>();
051
052        /** Classes included in the graph. */
053        private final IdentityHashSet<Class<?>> classes = new IdentityHashSet<Class<?>>();
054
055        /** Class loaders. */
056        private final IdentityHashSet<ClassLoader> classLoaders = new IdentityHashSet<ClassLoader>();
057
058        /**
059         * Create new graph creator.
060         * 
061         * @param objects
062         *            the objects provided may be arbitrary objects or {@link Class}
063         *            -objects.
064         */
065        public ClassLoaderGraphCreator(Object... objects) {
066                for (Object object : objects) {
067                        if (object instanceof Class<?>) {
068                                addClass((Class<?>) object);
069                        } else {
070                                addObject(object);
071                        }
072                }
073        }
074
075        /** Add a object. */
076        @SuppressWarnings("null")
077        public void addObject(Object object) {
078                CCSMAssert.isFalse(object == null, "Object may not be null.");
079                objects.add(object);
080                addClass(object.getClass());
081        }
082
083        /** Add a class. */
084        public void addClass(Class<?> clazz) {
085                CCSMAssert.isFalse(clazz == null, "Class may not be null.");
086                classes.add(clazz);
087                classLoaders.addAll(getClassLoaders(clazz));
088        }
089
090        /**
091         * Determine class loader hierarchy of a class. This includes the
092         * <code>null</code> value to describe the bootstrap loader.
093         */
094        private static List<ClassLoader> getClassLoaders(Class<?> clazz) {
095                ArrayList<ClassLoader> loaders = new ArrayList<ClassLoader>();
096
097                ClassLoader loader = clazz.getClassLoader();
098
099                while (loader != null) {
100                        loaders.add(loader);
101                        loader = loader.getParent();
102                }
103
104                loaders.add(null);
105
106                return loaders;
107        }
108
109        /** Create graph. */
110        public String createClassLoaderGraph() {
111                StringBuilder builder = new StringBuilder();
112                builder.append(HEADER + CR);
113                builder.append(createVertices());
114                builder.append(createEdges());
115                builder.append("}" + CR);
116
117                return builder.toString();
118        }
119
120        /** Create edge. */
121        private String createEdges() {
122                StringBuilder result = new StringBuilder();
123
124                for (Object object : objects) {
125                        appendEdge(result, object, object.getClass());
126                }
127
128                for (Class<?> clazz : classes) {
129                        appendEdge(result, clazz, clazz.getClassLoader());
130                }
131
132                for (ClassLoader loader : classLoaders) {
133                        if (loader != null) {
134                                appendEdge(result, loader, loader.getParent());
135                        }
136                }
137
138                return result.toString();
139        }
140
141        /** Append edge if it was not added before. */
142        private void appendEdge(StringBuilder builder, Object start, Object end) {
143                String edge = makeId(start) + " -> " + makeId(end) + ";" + CR;
144                builder.append(edge);
145        }
146
147        /** Creates vertices. */
148        private String createVertices() {
149                StringBuilder result = new StringBuilder();
150
151                for (Object object : objects) {
152                        result.append(createVertex(object, ECCSMColor.GREEN));
153                }
154
155                for (Class<?> clazz : classes) {
156                        result.append(createVertex(clazz, ECCSMColor.RED));
157                }
158
159                for (ClassLoader loader : classLoaders) {
160                        result.append(createVertex(loader, ECCSMColor.BLUE));
161                }
162
163                return result.toString();
164        }
165
166        /** Creates vertex. */
167        private String createVertex(Object node, ECCSMColor color) {
168                StringBuilder result = new StringBuilder();
169                result.append(makeId(node));
170                result.append(" [label=\"" + makeLabel(node) + "\", ");
171                result.append(" color=\"" + color.getHTMLColorCode() + "\"");
172                result.append("];" + CR);
173                return result.toString();
174        }
175
176        /** Creates a label for a node. */
177        private String makeLabel(Object object) {
178                String result;
179                if (object == null) {
180                        result = "Bootstrap Loader";
181                } else if (object instanceof Class<?>) {
182                        // A class object
183                        result = ((Class<?>) object).getName();
184                } else if (object instanceof ClassLoader) {
185                        // A class loader object
186                        result = object.toString();
187                } else {
188                        // A "normal" object
189                        String toString = object.toString();
190
191                        // Deal with empty toStrings
192                        if (StringUtils.isEmpty(toString)) {
193                                toString = "instance of " + object.getClass().getName();
194                        }
195                        result = toString;
196                }
197
198                result = StringUtils.replaceLineBreaks(result, " ");
199                result = result.replace('"', '\'');
200
201                return result;
202        }
203
204        /** Determines unique id for node. */
205        private String makeId(Object object) {
206                return "id" + idManager.obtainId(object);
207        }
208}
