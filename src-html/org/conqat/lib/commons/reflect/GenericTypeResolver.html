017package org.conqat.lib.commons.reflect;
018
019import java.lang.reflect.Array;
020import java.lang.reflect.Field;
021import java.lang.reflect.GenericArrayType;
022import java.lang.reflect.Method;
023import java.lang.reflect.ParameterizedType;
024import java.lang.reflect.Type;
025import java.lang.reflect.TypeVariable;
026import java.util.HashMap;
027import java.util.Map;
028
029import org.conqat.lib.commons.assertion.CCSMAssert;
030
031/**
032 * This is a class for helping with the resolution of generic types in the
033 * context of reflection. Unfortunately there is no easy way to get the actual
034 * return type or parameter type of a method if some generic class is within the
035 * class hierarchy. This class handles the messy aspects of this.
036 * <p>
037 * The instances of this class are bound to single classes, for which they are
038 * constructed. They only work correctly when querying parameters originating
039 * from this class or one of its methods. Furthermore this class does not work,
040 * if the class the lookup is performed for is generic itself.
041 * <p>
042 * The error handling of this class is rather crude. If any of the assumptions
043 * (either specified above or we learned from playing with reflection) is not
044 * met, an exception is thrown (currently {@link IllegalStateException}).
045 */
046public class GenericTypeResolver {
047
048        /** The map for looking up generic parameters. */
049        private final Map<TypeVariable<?>, Class<?>> parameterLookup = new HashMap<TypeVariable<?>, Class<?>>();
050
051        /**
052         * Creates a new generic type resolver for the given class.
053         * 
054         * @throws IllegalArgumentException
055         *             if called for generic class.
056         */
057        public GenericTypeResolver(Class<?> clazz) {
058                if (clazz.getTypeParameters().length != 0) {
059                        throw new IllegalArgumentException("This only works for non-generic classes!");
060                }
061                fillParamMap(clazz);
062        }
063
064        /**
065         * Creates a new generic type resolver for the given field and resolver of
066         * the field's (non-generic) class or child class.
067         */
068        public GenericTypeResolver(Field field, GenericTypeResolver parentResolver) {
069                parameterLookup.putAll(parentResolver.parameterLookup);
070
071                fillInParameters(field.getType(), field.getGenericType());
072                fillParamMap(field.getType());
073        }
074
075        /**
076         * Initializes the generic parameter lookup table by comparing for each
077         * super class and interface the type parameters with the actual type
078         * arguments. This process then is repeated recursively. It is important to
079         * fill from the current class before going to the super class, as the super
080         * class may reference parameters used in this class.
081         */
082        private void fillParamMap(Class<?> clazz) {
083                Class<?> superClass = clazz.getSuperclass();
084                if (superClass != null) {
085                        Type superType = clazz.getGenericSuperclass();
086                        fillInParameters(superClass, superType);
087                        fillParamMap(superClass);
088                }
089
090                Class<?>[] interfaces = clazz.getInterfaces();
091                Type[] genericInterfaces = clazz.getGenericInterfaces();
092                check(interfaces.length == genericInterfaces.length, "Interface lists should be equally long!");
093                for (int i = 0; i < interfaces.length; ++i) {
094                        fillInParameters(interfaces[i], genericInterfaces[i]);
095                        fillParamMap(interfaces[i]);
096                }
097        }
098
099        /**
100         * Fill the generic parameter lookup map from an explicit (class, type) pair
101         * by comparing the type parameters with the actual type parameters (if the
102         * class is generic at all).
103         * 
104         * @param clazz
105         *            the class (potentially) containing type parameters.
106         * @param type
107         *            the corresponding (potentially) generic type.
108         */
109        private void fillInParameters(Class<?> clazz, Type type) {
110                if (type instanceof ParameterizedType) {
111                        Type[] actualTypeArguments = ((ParameterizedType) type).getActualTypeArguments();
112                        TypeVariable<?>[] typeParameters = clazz.getTypeParameters();
113                        check(actualTypeArguments.length == typeParameters.length,
114                                        "Type parameters and actual arguments should be equally long!");
115
116                        for (int i = 0; i < typeParameters.length; ++i) {
117                                parameterLookup.put(typeParameters[i], resolveGenericType(actualTypeArguments[i]));
118                        }
119                }
120        }
121
122        /**
123         * Returns the actual type from a (potentially) generic type. If the
124         * argument is a plain class, it is returned, otherwise a lookup in the
125         * internal generic parameter map is performed. For parameterized types
126         * (e.g. <code>List&lt;String&gt;</code> the raw type (here: List) is
127         * returned.
128         * <p>
129         * Note that this only works for return values and parameters of methods
130         * belonging to the class for which this instance was constructed for.
131         * Otherwise the behavior is undefined (either returning nonsense or
132         * throwing an exception).
133         * 
134         * @param genericType
135         *            a type such as returned from
136         *            {@link Method#getGenericReturnType()} or
137         *            {@link Method#getGenericParameterTypes()}.
138         */
139        public Class<?> resolveGenericType(Type genericType) {
140                if (genericType instanceof Class<?>) {
141                        return (Class<?>) genericType;
142                }
143                if (genericType instanceof TypeVariable<?>) {
144                        check(parameterLookup.containsKey(genericType), "All generic parameters should be bound.");
145                        return parameterLookup.get(genericType);
146                }
147                if (genericType instanceof ParameterizedType) {
148                        ParameterizedType pt = (ParameterizedType) genericType;
149                        return (Class<?>) pt.getRawType();
150                }
151                if (genericType instanceof GenericArrayType) {
152                        GenericArrayType gat = (GenericArrayType) genericType;
153                        return Array.newInstance(resolveGenericType(gat.getGenericComponentType()), 0).getClass();
154                }
155
156                check(false, "Generic types should be either concrete classes, type variables, or parametrized types: "
157                                + genericType.getClass());
158                return null; // this line is never reached
159        }
160
161        /**
162         * This is an assertion method to simplify changing the type of error
163         * handling. Many of the assertions rather document our assumptions about
164         * the JVM, than really checking errors.
165         */
166        private void check(boolean assumedCondition, String errorMessage) {
167                CCSMAssert.isTrue(assumedCondition, errorMessage);
168        }
169}
