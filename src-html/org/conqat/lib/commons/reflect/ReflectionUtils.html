057        /**
058         * Convert a String to an Object of the provided type. It supports
059         * conversion to primitive types and simple tests (char: use first character
060         * of string, boolean: test for values "true", "on", "1", "yes"). Enums are
061         * handled by the {@link EnumUtils#valueOfIgnoreCase(Class, String)} method.
062         * Otherwise it is checked if the target type has a constructor that takes a
063         * single string and it is invoked. For all other cases an exception is
064         * thrown, as no conversion is possible.
065         * 
066         * <i>Maintainer note</i>: Make sure this method is changed in accordance
067         * with method {@link #isConvertibleFromString(Class)}
068         * 
069         * @see #convertString(String, Class)
070         * 
071         * @param value
072         *            the string to be converted.
073         * @param targetType
074         *            the type of the resulting object.
075         * @return the converted object.
076         * @throws TypeConversionException
077         *             in the case that no conversion could be performed.
078         * @see #isConvertibleFromString(Class)
079         * 
080         */
081        @SuppressWarnings({ "unchecked", "rawtypes" })
082        public static <T> T convertString(String value, Class<T> targetType) throws TypeConversionException {
083
084                // value must be provided
085                if (value == null) {
086                        if (String.class.equals(targetType)) {
087                                return (T) StringUtils.EMPTY_STRING;
088                        }
089                        throw new TypeConversionException("Null value can't be converted to type '" + targetType.getName() + "'.");
090                }
091
092                if (targetType.equals(Object.class) || targetType.equals(String.class)) {
093                        return (T) value;
094                }
095                if (targetType.isPrimitive() || EJavaPrimitive.isWrapperType(targetType)) {
096                        return convertPrimitive(value, targetType);
097                }
098                if (targetType.isEnum()) {
099                        // we checked manually before
100                        Object result = EnumUtils.valueOfIgnoreCase((Class) targetType, value);
101                        if (result == null) {
102                                throw new TypeConversionException("'" + value + "' is no valid value for enum " + targetType.getName());
103                        }
104                        return (T) result;
105
106                }
107                if (targetType.equals(Color.class)) {
108                        Color result = ColorUtils.fromString(value);
109                        if (result == null) {
110                                throw new TypeConversionException("'" + value + "' is not a valid color!");
111                        }
112                        return (T) result;
113                }
114
115                // Check if the target type has a constructor taking a single string.
116                try {
117                        Constructor<T> c = targetType.getConstructor(String.class);
118                        return c.newInstance(value);
119                } catch (Exception e) {
120                        throw new TypeConversionException("No constructor taking one String argument found for type '" + targetType
121                                        + "' (" + e.getMessage() + ")", e);
122                }
123
124        }

133        /**
134         * This method checks if the provided type can be converted from a string.
135         * With respect to {@link #convertString(String, Class)} the semantics are
136         * the following: If this method returns <code>true</code> a particular
137         * string <i>may</i> be convertible to the target type. If this method
138         * returns <code>false</code>, a call to
139         * {@link #convertString(String, Class)} is guaranteed to result in a
140         * {@link TypeConversionException}. If a call to
141         * {@link #convertString(String, Class)} does not result in an exception, a
142         * call to this method is guaranteed to return <code>true</code>.
143         * <p>
144         * <i>Maintainer note</i>: Make sure this method is change in accordance
145         * with method {@link #convertString(String, Class)}
146         * 
147         * @see #convertString(String, Class)
148         */
149        public static boolean isConvertibleFromString(Class<?> targetType) {
150
151                if (targetType.equals(Object.class) || targetType.equals(String.class)) {
152                        return true;
153                }
154                if (targetType.isPrimitive() || EJavaPrimitive.isWrapperType(targetType)) {
155                        return true;
156                }
157                if (targetType.isEnum()) {
158                        return true;
159
160                }
161                if (targetType.equals(Color.class)) {
162                        return true;
163                }
164
165                try {
166                        targetType.getConstructor(String.class);
167                        return true;
168                } catch (SecurityException e) {
169                        // case is handled at method end
170                } catch (NoSuchMethodException e) {
171                        // case is handled at method end
172                }
173
174                return false;
175        }

215        /**
216         * Invoke a method with parameters.
217         * 
218         * @param method
219         *            the method to invoke
220         * @param object
221         *            the object the underlying method is invoked from
222         * @param parameterMap
223         *            this maps from the formal parameter of the method to the
224         *            parameter value
225         * @return the result of dispatching the method
226         * @throws IllegalArgumentException
227         *             if the method is an instance method and the specified object
228         *             argument is not an instance of the class or interface
229         *             declaring the underlying method (or of a subclass or
230         *             implementor thereof); if the number of actual and formal
231         *             parameters differ; if an unwrapping conversion for primitive
232         *             arguments fails; or if, after possible unwrapping, a
233         *             parameter value cannot be converted to the corresponding
234         *             formal parameter type by a method invocation conversion; if
235         *             formal parameters belong to different methods.
236         * 
237         * @throws IllegalAccessException
238         *             if this Method object enforces Java language access control
239         *             and the underlying method is inaccessible.
240         * @throws InvocationTargetException
241         *             if the underlying method throws an exception.
242         * @throws NullPointerException
243         *             if the specified object is null and the method is an instance
244         *             method.
245         * @throws ExceptionInInitializerError
246         *             if the initialization provoked by this method fails.
247         */
248        public static Object invoke(Method method, Object object, Map<FormalParameter, Object> parameterMap)
249                        throws IllegalArgumentException, IllegalAccessException, InvocationTargetException {
250
251                for (FormalParameter formalParameter : parameterMap.keySet()) {
252                        if (!formalParameter.getMethod().equals(method)) {
253                                throw new IllegalArgumentException("Parameters must belong to method.");
254                        }
255                }
256
257                Object[] parameters = obtainParameters(parameterMap);
258
259                return method.invoke(object, parameters);
260
261        }

279        /**
280         * Returns the wrapper class type for a primitive type (e.g.
281         * <code>Integer</code> for an <code>int</code>). If the given class is not
282         * a primitive, the class itself is returned.
283         * 
284         * @param clazz
285         *            the class.
286         * @return the corresponding class type.
287         */
288        public static Class<?> resolvePrimitiveClass(Class<?> clazz) {
289                if (!clazz.isPrimitive()) {
290                        return clazz;
291                }
292
293                EJavaPrimitive primitive = EJavaPrimitive.getForPrimitiveClass(clazz);
294                if (primitive == null) {
295                        throw new IllegalStateException("Did Java get a new primitive? " + clazz.getName());
296                }
297                return primitive.getWrapperClass();
298        }

300        /**
301         * Convert a String to an Object of the provided type. This only works for
302         * primitive types and wrapper types.
303         * 
304         * @param value
305         *            the string to be converted.
306         * @param targetType
307         *            the type of the resulting object.
308         * @return the converted object.
309         * @throws TypeConversionException
310         *             in the case that no conversion could be performed.
311         */
312        @SuppressWarnings("unchecked")
313        /* package */static <T> T convertPrimitive(String value, Class<T> targetType) throws TypeConversionException {
314
315                EJavaPrimitive primitive = EJavaPrimitive.getForPrimitiveOrWrapperClass(targetType);
316                if (primitive == null) {
317                        throw new IllegalArgumentException("Type '" + targetType.getName() + "' is not a primitive type!");
318                }
319
320                try {
321
322                        switch (primitive) {
323                        case BOOLEAN:
324                                boolean b = "1".equalsIgnoreCase(value) || "true".equalsIgnoreCase(value)
325                                                || "on".equalsIgnoreCase(value) || "yes".equalsIgnoreCase(value);
326                                return (T) Boolean.valueOf(b);
327
328                        case CHAR:
329                                return (T) Character.valueOf(value.charAt(0));
330
331                        case BYTE:
332                                return (T) Byte.valueOf(value);
333
334                        case SHORT:
335                                return (T) Short.valueOf(value);
336
337                        case INT:
338                                return (T) Integer.valueOf(value);
339
340                        case LONG:
341                                return (T) Long.valueOf(value);
342
343                        case FLOAT:
344                                return (T) Float.valueOf(value);
345
346                        case DOUBLE:
347                                return (T) Double.valueOf(value);
348
349                        default:
350                                throw new TypeConversionException("No conversion possible for " + primitive);
351                        }
352
353                } catch (NumberFormatException e) {
354                        throw new TypeConversionException("Value'" + value + "' can't be converted to type '" + targetType.getName()
355                                        + "' (" + e.getMessage() + ").", e);
356                }
357        }

372         *             if the typeName neither resolves to a primitive, nor to a

376                return resolveType(typeName, null);

379        /**
380         * Resolves the class object for a type name. Type name can be a primitive.
381         * For resolution, the given class loader is used.
382         * <p>
383         * While method <code>Class.forName(...)</code> resolves fully qualified
384         * names, it does not resolve primitives, e.g. "java.lang.Boolean" can be
385         * resolved but "boolean" cannot.
386         * 
387         * @param typeName
388         *            name of the type. For primitives case is ignored.
389         * 
390         * @param classLoader
391         *            the class loader used for loading the class. If this is null,
392         *            the caller class loader is used.
393         * 
394         * @throws ClassNotFoundException
395         *             if the typeName neither resolves to a primitive, nor to a
396         *             known class.
397         */
398        public static Class<?> resolveType(String typeName, ClassLoader classLoader) throws ClassNotFoundException {
399
400                EJavaPrimitive primitive = EJavaPrimitive.getPrimitiveIgnoreCase(typeName);
401
402                if (primitive != null) {
403                        return primitive.getClassObject();
404                }
405
406                if (classLoader == null) {
407                        return Class.forName(typeName);
408                }
409                return Class.forName(typeName, true, classLoader);
410        }

429        /**
430         * Obtain parameter array from parameter map.
431         */
432        private static Object[] obtainParameters(Map<FormalParameter, Object> parameterMap) {
433
434                ArrayList<FormalParameter> formalParameters = new ArrayList<FormalParameter>(parameterMap.keySet());
435
436                Collections.sort(formalParameters, comparator);
437
438                Object[] result = new Object[formalParameters.size()];
439
440                for (int i = 0; i < formalParameters.size(); i++) {
441                        result[i] = parameterMap.get(formalParameters.get(i));
442                }
443
444                return result;
445        }

447        /**
448         * Obtain the return type of method. This method deals with bridge methods
449         * introduced by generics. This works for methods without parameters only.
450         * 
451         * @param clazz
452         *            the class
453         * @param methodName
454         *            the name of the method.
455         * @return the return type
456         * @throws NoSuchMethodException
457         *             if the class doesn't contain the desired method
458         */
459        public static Class<?> obtainMethodReturnType(Class<?> clazz, String methodName) throws NoSuchMethodException {
460
461                // due to the potential presense of bridge methods we can't use
462                // Clazz.getMethod() and have to iterate over all methods.
463                for (Method method : clazz.getMethods()) {
464                        if (isValid(method, methodName)) {
465                                return method.getReturnType();
466                        }
467                }
468                // method not found
469                throw new NoSuchMethodException(
470                                "Class " + clazz.getName() + " doesn't have parameterless method named " + methodName);
471        }

473        /**
474         * Obtain the generic return type of method. This method deals with the gory
475         * details of bridge methods and generics. This works for methods without
476         * parameters only. This doesn't work for interfaces, arrays and enums.
477         * 
478         * @param clazz
479         *            the class
480         * @param methodName
481         *            the name of the method.
482         * @return the return type
483         * @throws NoSuchMethodException
484         *             if the class doesn't contain the desired method
485         */
486        public static Class<?> obtainGenericMethodReturnType(Class<?> clazz, String methodName)
487                        throws NoSuchMethodException {
488
489                if (clazz.isArray() || clazz.isEnum()) {
490                        throw new IllegalArgumentException("Doesn't work for arrays and enums.");
491                }
492                if (clazz.getTypeParameters().length != 0) {
493                        throw new IllegalArgumentException("Doesn't work for generic classes.");
494                }
495
496                for (Method method : clazz.getMethods()) {
497                        if (isValid(method, methodName)) {
498                                return new GenericTypeResolver(clazz).resolveGenericType(method.getGenericReturnType());
499                        }
500                }
501
502                // method not found
503                throw new NoSuchMethodException(
504                                "Class " + clazz.getName() + " doesn't have parameterless method named " + methodName);
505        }

515        /**
516         * Returns the value from the map, whose key is the best match for the given
517         * class. The best match is defined by the first match occurring in a breath
518         * first search of the inheritance tree, where the base class is always
519         * visited before the implemented interfaces. Interfaces are traversed in
520         * the order they are defined in the source file. The only exception is
521         * {@link Object}, which is considered only as the very last option.
522         * <p>
523         * As this lookup can be expensive (reflective iteration over the entire
524         * inheritance tree) the results should be cached if multiple lookups for
525         * the same class are expected.
526         * 
527         * 
528         * @param clazz
529         *            the class being looked up.
530         * @param classMap
531         *            the map to perform the lookup in.
532         * @return the best match found or <code>null</code> if no matching entry
533         *         was found. Note that <code>null</code> will also be returned if
534         *         the entry for the best matching class was <code>null</code>.
535         */
536        public static <T> T performNearestClassLookup(Class<?> clazz, Map<Class<?>, T> classMap) {
537                Queue<Class<?>> q = new LinkedList<Class<?>>();
538                q.add(clazz);
539
540                while (!q.isEmpty()) {
541                        Class<?> current = q.poll();
542                        if (classMap.containsKey(current)) {
543                                return classMap.get(current);
544                        }
545
546                        Class<?> superClass = current.getSuperclass();
547                        if (superClass != null && superClass != Object.class) {
548                                q.add(superClass);
549                        }
550
551                        for (Class<?> iface : current.getInterfaces()) {
552                                q.add(iface);
553                        }
554                }
555                return classMap.get(Object.class);
556        }

598        /**
599         * Obtains the version of a Java class file.
600         * 
601         * Class file major versions (from
602         * http://en.wikipedia.org/wiki/Class_(file_format)):
603         * 
604         * <pre>
605         * J2SE 7 = 51 
606         * J2SE 6.0 = 50
607         * J2SE 5.0 = 49
608         * JDK 1.4 = 48
609         * JDK 1.3 = 47
610         * JDK 1.2 = 46
611         * JDK 1.1 = 45
612         * </pre>
613         * 
614         * @param inputStream
615         *            stream to read class file from.
616         * @return the class file version or <code>null</code> if stream does not
617         *         contain a class file.
618         * @throws IOException
619         *             if an IO problem occurs.
620         */
621        public static Version obtainClassFileVersion(InputStream inputStream) throws IOException {
622                DataInputStream classfile = new DataInputStream(inputStream);
623                int magic = classfile.readInt();
624                if (magic != 0xcafebabe) {
625                        return null;
626                }
627                int minorVersion = classfile.readUnsignedShort();
628                int majorVersion = classfile.readUnsignedShort();
629
630                return new Version(majorVersion, minorVersion);
631        }

633        /**
634         * This method extracts the class file version from each class file in the
635         * provided jar.
636         * 
637         * @return the result maps from the class file to its version.
638         */
639        public static HashMap<String, Version> getClassFileVersions(File jarFile) throws IOException {
640
641                HashMap<String, Version> result = new HashMap<String, Version>();
642
643                JarFile jar = new JarFile(jarFile);
644                Enumeration<JarEntry> entries = jar.entries();
645
646                while (entries.hasMoreElements()) {
647                        JarEntry entry = entries.nextElement();
648                        if (!entry.isDirectory() && entry.getName().endsWith(".class")) {
649                                InputStream entryStream = jar.getInputStream(entry);
650                                Version version = obtainClassFileVersion(entryStream);
651                                result.put(entry.getName(), version);
652                                entryStream.close();
653                        }
654                }
655
656                jar.close();
657
658                return result;
659        }

685        /**
686         * Determines the most specific common base class. If one of the types is
687         * primitive or an interface, null is returned.
688         */
689        public static Class<?> determineCommonBase(Class<?> class1, Class<?> class2) {
690                List<Class<?>> hierarchy1 = getBaseHierarchy(class1);
691                List<Class<?>> hierarchy2 = getBaseHierarchy(class2);
692
693                if (hierarchy1.isEmpty() || hierarchy2.isEmpty() || !hierarchy1.get(0).equals(hierarchy2.get(0))) {
694                        return null;
695                }
696
697                int index = 0;
698                while (index < hierarchy1.size() && index < hierarchy2.size()
699                                && hierarchy1.get(index).equals(hierarchy2.get(index))) {
700                        index += 1;
701                }
702                if (index <= hierarchy1.size()) {
703                        index -= 1;
704                }
705
706                return hierarchy1.get(index);
707        }

709        /**
710         * Returns the hierarchy of base classes starting with {@link Object} up to
711         * the class itself.
712         */
713        public static List<Class<?>> getBaseHierarchy(Class<?> baseClass) {
714                List<Class<?>> hierarchy = ReflectionUtils.getSuperClasses(baseClass);
715                Collections.reverse(hierarchy);
716                hierarchy.add(baseClass);
717                return hierarchy;
718        }

720        /**
721         * Returns the set of all interfaces implemented by the given class. This
722         * includes also interfaces that are indirectly implemented as they are
723         * extended by an interfaces that is implemented by the given class. If the
724         * given class is an interface, it is included itself.
725         */
726        public static Set<Class<?>> getAllInterfaces(Class<?> baseClass) {
727                Queue<Class<?>> q = new LinkedList<Class<?>>();
728                q.add(baseClass);
729
730                Set<Class<?>> result = new HashSet<Class<?>>();
731                if (baseClass.isInterface()) {
732                        result.add(baseClass);
733                }
734
735                while (!q.isEmpty()) {
736                        for (Class<?> iface : q.poll().getInterfaces()) {
737                                if (result.add(iface)) {
738                                        q.add(iface);
739                                }
740                        }
741                }
742
743                return result;
744        }

746        /**
747         * Returns all fields declared for a class (all visibilities and also
748         * inherited from super classes). Note that multiple fields with the same
749         * name may exist due to redeclaration/shadowing in sub classes.
750         */
751        public static List<Field> getAllFields(Class<?> type) {
752                List<Field> fields = new ArrayList<>();
753                while (type != null) {
754                        fields.addAll(Arrays.asList(type.getDeclaredFields()));
755                        type = type.getSuperclass();
756                }
757                return fields;
758        }

774        /**
775         * Returns the field with the given name from the given type or null if none
776         * exists. If the field cannot be found in the given type, all super-classes
777         * are searched as well. Different from {@link Class#getField(String)}, this
778         * will also work for non-public fields.
779         */
780        public static Field getField(Class<?> type, String name) throws SecurityException {
781                while (type != null) {
782                        try {
783                                return type.getDeclaredField(name);
784                        } catch (NoSuchFieldException e) {
785                                type = type.getSuperclass();
786                        }
787                }
788                return null;
789        }
