059        /**
060         * Convert a String to an Object of the provided type. It supports conversion to
061         * primitive types and simple tests (char: use first character of string,
062         * boolean: test for values "true", "on", "1", "yes"). Enums are handled by the
063         * {@link EnumUtils#valueOfIgnoreCase(Class, String)} method. Otherwise it is
064         * checked if the target type has a constructor that takes a single string and
065         * it is invoked. For all other cases an exception is thrown, as no conversion
066         * is possible.
067         * 
068         * <i>Maintainer note</i>: Make sure this method is changed in accordance with
069         * method {@link #isConvertibleFromString(Class)}
070         * 
071         * @see #convertString(String, Class)
072         * 
073         * @param value
074         *            the string to be converted.
075         * @param targetType
076         *            the type of the resulting object.
077         * @return the converted object.
078         * @throws TypeConversionException
079         *             in the case that no conversion could be performed.
080         * @see #isConvertibleFromString(Class)
081         * 
082         */
083        @SuppressWarnings({ "unchecked", "rawtypes" })
084        public static <T> T convertString(String value, Class<T> targetType) throws TypeConversionException {
085
086                // value must be provided
087                if (value == null) {
088                        if (String.class.equals(targetType)) {
089                                return (T) StringUtils.EMPTY_STRING;
090                        }
091                        throw new TypeConversionException("Null value can't be converted to type '" + targetType.getName() + "'.");
092                }
093
094                if (targetType.equals(Object.class) || targetType.equals(String.class)) {
095                        return (T) value;
096                }
097                if (targetType.isPrimitive() || EJavaPrimitive.isWrapperType(targetType)) {
098                        return convertPrimitive(value, targetType);
099                }
100                if (targetType.isEnum()) {
101                        // we checked manually before
102                        Object result = EnumUtils.valueOfIgnoreCase((Class) targetType, value);
103                        if (result == null) {
104                                throw new TypeConversionException("'" + value + "' is no valid value for enum " + targetType.getName());
105                        }
106                        return (T) result;
107
108                }
109                if (targetType.equals(Color.class)) {
110                        Color result = ColorUtils.fromString(value);
111                        if (result == null) {
112                                throw new TypeConversionException("'" + value + "' is not a valid color!");
113                        }
114                        return (T) result;
115                }
116
117                // Check if the target type has a constructor taking a single string.
118                try {
119                        Constructor<T> c = targetType.getConstructor(String.class);
120                        return c.newInstance(value);
121                } catch (Exception e) {
122                        throw new TypeConversionException("No constructor taking one String argument found for type '" + targetType
123                                        + "' (" + e.getMessage() + ")", e);
124                }
125
126        }

135        /**
136         * This method checks if the provided type can be converted from a string. With
137         * respect to {@link #convertString(String, Class)} the semantics are the
138         * following: If this method returns <code>true</code> a particular string
139         * <i>may</i> be convertible to the target type. If this method returns
140         * <code>false</code>, a call to {@link #convertString(String, Class)} is
141         * guaranteed to result in a {@link TypeConversionException}. If a call to
142         * {@link #convertString(String, Class)} does not result in an exception, a call
143         * to this method is guaranteed to return <code>true</code>.
144         * <p>
145         * <i>Maintainer note</i>: Make sure this method is change in accordance with
146         * method {@link #convertString(String, Class)}
147         * 
148         * @see #convertString(String, Class)
149         */
150        public static boolean isConvertibleFromString(Class<?> targetType) {
151
152                if (targetType.equals(Object.class) || targetType.equals(String.class)) {
153                        return true;
154                }
155                if (targetType.isPrimitive() || EJavaPrimitive.isWrapperType(targetType)) {
156                        return true;
157                }
158                if (targetType.isEnum()) {
159                        return true;
160
161                }
162                if (targetType.equals(Color.class)) {
163                        return true;
164                }
165
166                try {
167                        targetType.getConstructor(String.class);
168                        return true;
169                } catch (SecurityException e) {
170                        // case is handled at method end
171                } catch (NoSuchMethodException e) {
172                        // case is handled at method end
173                }
174
175                return false;
176        }

215        /**
216         * Invoke a method with parameters.
217         * 
218         * @param method
219         *            the method to invoke
220         * @param object
221         *            the object the underlying method is invoked from
222         * @param parameterMap
223         *            this maps from the formal parameter of the method to the parameter
224         *            value
225         * @return the result of dispatching the method
226         * @throws IllegalArgumentException
227         *             if the method is an instance method and the specified object
228         *             argument is not an instance of the class or interface declaring
229         *             the underlying method (or of a subclass or implementor thereof);
230         *             if the number of actual and formal parameters differ; if an
231         *             unwrapping conversion for primitive arguments fails; or if, after
232         *             possible unwrapping, a parameter value cannot be converted to the
233         *             corresponding formal parameter type by a method invocation
234         *             conversion; if formal parameters belong to different methods.
235         * 
236         * @throws IllegalAccessException
237         *             if this Method object enforces Java language access control and
238         *             the underlying method is inaccessible.
239         * @throws InvocationTargetException
240         *             if the underlying method throws an exception.
241         * @throws NullPointerException
242         *             if the specified object is null and the method is an instance
243         *             method.
244         * @throws ExceptionInInitializerError
245         *             if the initialization provoked by this method fails.
246         */
247        public static Object invoke(Method method, Object object, Map<FormalParameter, Object> parameterMap)
248                        throws IllegalArgumentException, IllegalAccessException, InvocationTargetException {
249
250                for (FormalParameter formalParameter : parameterMap.keySet()) {
251                        if (!formalParameter.getMethod().equals(method)) {
252                                throw new IllegalArgumentException("Parameters must belong to method.");
253                        }
254                }
255
256                Object[] parameters = obtainParameters(parameterMap);
257
258                return method.invoke(object, parameters);
259
260        }

278        /**
279         * Returns the wrapper class type for a primitive type (e.g.
280         * <code>Integer</code> for an <code>int</code>). If the given class is not a
281         * primitive, the class itself is returned.
282         * 
283         * @param clazz
284         *            the class.
285         * @return the corresponding class type.
286         */
287        public static Class<?> resolvePrimitiveClass(Class<?> clazz) {
288                if (!clazz.isPrimitive()) {
289                        return clazz;
290                }
291
292                EJavaPrimitive primitive = EJavaPrimitive.getForPrimitiveClass(clazz);
293                if (primitive == null) {
294                        throw new IllegalStateException("Did Java get a new primitive? " + clazz.getName());
295                }
296                return primitive.getWrapperClass();
297        }

299        /**
300         * Convert a String to an Object of the provided type. This only works for
301         * primitive types and wrapper types.
302         * 
303         * @param value
304         *            the string to be converted.
305         * @param targetType
306         *            the type of the resulting object.
307         * @return the converted object.
308         * @throws TypeConversionException
309         *             in the case that no conversion could be performed.
310         */
311        @SuppressWarnings("unchecked")
312        /* package */static <T> T convertPrimitive(String value, Class<T> targetType) throws TypeConversionException {
313
314                EJavaPrimitive primitive = EJavaPrimitive.getForPrimitiveOrWrapperClass(targetType);
315                if (primitive == null) {
316                        throw new IllegalArgumentException("Type '" + targetType.getName() + "' is not a primitive type!");
317                }
318
319                try {
320
321                        switch (primitive) {
322                        case BOOLEAN:
323                                boolean b = "1".equalsIgnoreCase(value) || "true".equalsIgnoreCase(value)
324                                                || "on".equalsIgnoreCase(value) || "yes".equalsIgnoreCase(value);
325                                return (T) Boolean.valueOf(b);
326
327                        case CHAR:
328                                return (T) Character.valueOf(value.charAt(0));
329
330                        case BYTE:
331                                return (T) Byte.valueOf(value);
332
333                        case SHORT:
334                                return (T) Short.valueOf(value);
335
336                        case INT:
337                                return (T) Integer.valueOf(value);
338
339                        case LONG:
340                                return (T) Long.valueOf(value);
341
342                        case FLOAT:
343                                return (T) Float.valueOf(value);
344
345                        case DOUBLE:
346                                return (T) Double.valueOf(value);
347
348                        default:
349                                throw new TypeConversionException("No conversion possible for " + primitive);
350                        }
351
352                } catch (NumberFormatException e) {
353                        throw new TypeConversionException("Value'" + value + "' can't be converted to type '" + targetType.getName()
354                                        + "' (" + e.getMessage() + ").", e);
355                }
356        }

371         *             if the typeName neither resolves to a primitive, nor to a known

375                return resolveType(typeName, null);

378        /**
379         * Resolves the class object for a type name. Type name can be a primitive. For
380         * resolution, the given class loader is used.
381         * <p>
382         * While method <code>Class.forName(...)</code> resolves fully qualified names,
383         * it does not resolve primitives, e.g. "java.lang.Boolean" can be resolved but
384         * "boolean" cannot.
385         * 
386         * @param typeName
387         *            name of the type. For primitives case is ignored.
388         * 
389         * @param classLoader
390         *            the class loader used for loading the class. If this is null, the
391         *            caller class loader is used.
392         * 
393         * @throws ClassNotFoundException
394         *             if the typeName neither resolves to a primitive, nor to a known
395         *             class.
396         */
397        public static Class<?> resolveType(String typeName, ClassLoader classLoader) throws ClassNotFoundException {
398
399                EJavaPrimitive primitive = EJavaPrimitive.getPrimitiveIgnoreCase(typeName);
400
401                if (primitive != null) {
402                        return primitive.getClassObject();
403                }
404
405                if (classLoader == null) {
406                        return Class.forName(typeName);
407                }
408                return Class.forName(typeName, true, classLoader);
409        }

428        /**
429         * Obtain parameter array from parameter map.
430         */
431        private static Object[] obtainParameters(Map<FormalParameter, Object> parameterMap) {
432
433                ArrayList<FormalParameter> formalParameters = new ArrayList<>(parameterMap.keySet());
434
435                formalParameters.sort(comparator);
436
437                Object[] result = new Object[formalParameters.size()];
438
439                for (int i = 0; i < formalParameters.size(); i++) {
440                        result[i] = parameterMap.get(formalParameters.get(i));
441                }
442
443                return result;
444        }

446        /**
447         * Obtain the return type of method. This method deals with bridge methods
448         * introduced by generics. This works for methods without parameters only.
449         * 
450         * @param clazz
451         *            the class
452         * @param methodName
453         *            the name of the method.
454         * @return the return type
455         * @throws NoSuchMethodException
456         *             if the class doesn't contain the desired method
457         */
458        public static Class<?> obtainMethodReturnType(Class<?> clazz, String methodName) throws NoSuchMethodException {
459
460                // due to the potential presense of bridge methods we can't use
461                // Clazz.getMethod() and have to iterate over all methods.
462                for (Method method : clazz.getMethods()) {
463                        if (isValid(method, methodName)) {
464                                return method.getReturnType();
465                        }
466                }
467                // method not found
468                throw new NoSuchMethodException(
469                                "Class " + clazz.getName() + " doesn't have parameterless method named " + methodName);
470        }

472        /**
473         * Obtain the generic return type of method. This method deals with the gory
474         * details of bridge methods and generics. This works for methods without
475         * parameters only. This doesn't work for interfaces, arrays and enums.
476         * 
477         * @param clazz
478         *            the class
479         * @param methodName
480         *            the name of the method.
481         * @return the return type
482         * @throws NoSuchMethodException
483         *             if the class doesn't contain the desired method
484         */
485        public static Class<?> obtainGenericMethodReturnType(Class<?> clazz, String methodName)
486                        throws NoSuchMethodException {
487
488                if (clazz.isArray() || clazz.isEnum()) {
489                        throw new IllegalArgumentException("Doesn't work for arrays and enums.");
490                }
491                if (clazz.getTypeParameters().length != 0) {
492                        throw new IllegalArgumentException("Doesn't work for generic classes.");
493                }
494
495                for (Method method : clazz.getMethods()) {
496                        if (isValid(method, methodName)) {
497                                return new GenericTypeResolver(clazz).resolveGenericType(method.getGenericReturnType());
498                        }
499                }
500
501                // method not found
502                throw new NoSuchMethodException(
503                                "Class " + clazz.getName() + " doesn't have parameterless method named " + methodName);
504        }

514        /**
515         * Returns the value from the map, whose key is the best match for the given
516         * class. The best match is defined by the first match occurring in a breath
517         * first search of the inheritance tree, where the base class is always visited
518         * before the implemented interfaces. Interfaces are traversed in the order they
519         * are defined in the source file. The only exception is {@link Object}, which
520         * is considered only as the very last option.
521         * <p>
522         * As this lookup can be expensive (reflective iteration over the entire
523         * inheritance tree) the results should be cached if multiple lookups for the
524         * same class are expected.
525         * 
526         * 
527         * @param clazz
528         *            the class being looked up.
529         * @param classMap
530         *            the map to perform the lookup in.
531         * @return the best match found or <code>null</code> if no matching entry was
532         *         found. Note that <code>null</code> will also be returned if the entry
533         *         for the best matching class was <code>null</code>.
534         */
535        public static <T> T performNearestClassLookup(Class<?> clazz, Map<Class<?>, T> classMap) {
536                Queue<Class<?>> q = new LinkedList<>();
537                q.add(clazz);
538
539                while (!q.isEmpty()) {
540                        Class<?> current = q.poll();
541                        if (classMap.containsKey(current)) {
542                                return classMap.get(current);
543                        }
544
545                        Class<?> superClass = current.getSuperclass();
546                        if (superClass != null && superClass != Object.class) {
547                                q.add(superClass);
548                        }
549
550                        q.addAll(Arrays.asList(current.getInterfaces()));
551                }
552                return classMap.get(Object.class);
553        }

594        /**
595         * Obtains the version of a Java class file.
596         * 
597         * Class file major versions (from
598         * http://en.wikipedia.org/wiki/Class_(file_format)):
599         * 
600         * <pre>
601         * J2SE 7 = 51 
602         * J2SE 6.0 = 50
603         * J2SE 5.0 = 49
604         * JDK 1.4 = 48
605         * JDK 1.3 = 47
606         * JDK 1.2 = 46
607         * JDK 1.1 = 45
608         * </pre>
609         * 
610         * @param inputStream
611         *            stream to read class file from.
612         * @return the class file version or <code>null</code> if stream does not
613         *         contain a class file.
614         * @throws IOException
615         *             if an IO problem occurs.
616         */
617        public static Version obtainClassFileVersion(InputStream inputStream) throws IOException {
618                DataInputStream classfile = new DataInputStream(inputStream);
619                int magic = classfile.readInt();
620                if (magic != 0xcafebabe) {
621                        return null;
622                }
623                int minorVersion = classfile.readUnsignedShort();
624                int majorVersion = classfile.readUnsignedShort();
625
626                return new Version(majorVersion, minorVersion);
627        }

629        /**
630         * This method extracts the class file version from each class file in the
631         * provided jar.
632         * 
633         * @return the result maps from the class file to its version.
634         */
635        public static HashMap<String, Version> getClassFileVersions(File jarFile) throws IOException {
636
637                HashMap<String, Version> result = new HashMap<>();
638
639                JarFile jar = new JarFile(jarFile);
640                Enumeration<JarEntry> entries = jar.entries();
641
642                while (entries.hasMoreElements()) {
643                        JarEntry entry = entries.nextElement();
644                        if (!entry.isDirectory() && entry.getName().endsWith(".class")) {
645                                InputStream entryStream = jar.getInputStream(entry);
646                                Version version = obtainClassFileVersion(entryStream);
647                                result.put(entry.getName(), version);
648                                entryStream.close();
649                        }
650                }
651
652                jar.close();
653
654                return result;
655        }

680        /**
681         * Determines the most specific common base class. If one of the types is
682         * primitive or an interface, null is returned.
683         */
684        public static Class<?> determineCommonBase(Class<?> class1, Class<?> class2) {
685                List<Class<?>> hierarchy1 = getBaseHierarchy(class1);
686                List<Class<?>> hierarchy2 = getBaseHierarchy(class2);
687
688                if (hierarchy1.isEmpty() || hierarchy2.isEmpty() || !hierarchy1.get(0).equals(hierarchy2.get(0))) {
689                        return null;
690                }
691
692                int index = 0;
693                while (index < hierarchy1.size() && index < hierarchy2.size()
694                                && hierarchy1.get(index).equals(hierarchy2.get(index))) {
695                        index += 1;
696                }
697                if (index <= hierarchy1.size()) {
698                        index -= 1;
699                }
700
701                return hierarchy1.get(index);
702        }

704        /**
705         * Returns the hierarchy of base classes starting with {@link Object} up to the
706         * class itself.
707         */
708        public static List<Class<?>> getBaseHierarchy(Class<?> baseClass) {
709                List<Class<?>> hierarchy = ReflectionUtils.getSuperClasses(baseClass);
710                Collections.reverse(hierarchy);
711                hierarchy.add(baseClass);
712                return hierarchy;
713        }

716         * Returns the set of all interfaces implemented by the given class. This
717         * includes also interfaces that are indirectly implemented as they are extended
718         * by an interfaces that is implemented by the given class. If the given class
719         * is an interface, it is included itself.
720         */
721        public static Set<Class<?>> getAllInterfaces(Class<?> baseClass) {
722                Queue<Class<?>> q = new LinkedList<>();
723                q.add(baseClass);
724
725                Set<Class<?>> result = new HashSet<>();
726                if (baseClass.isInterface()) {
727                        result.add(baseClass);
728                }
729
730                while (!q.isEmpty()) {
731                        for (Class<?> iface : q.poll().getInterfaces()) {
732                                if (result.add(iface)) {
733                                        q.add(iface);
734                                }
735                        }
736                }
737
738                return result;
739        }

741        /**
742         * Returns all fields declared for a class (all visibilities and also inherited
743         * from super classes). Note that multiple fields with the same name may exist
744         * due to redeclaration/shadowing in sub classes.
745         */
746        public static List<Field> getAllFields(Class<?> type) {
747                List<Field> fields = new ArrayList<>();
748                while (type != null) {
749                        fields.addAll(Arrays.asList(type.getDeclaredFields()));
750                        type = type.getSuperclass();
751                }
752                return fields;
753        }

769        /**
770         * Returns the field with the given name from the given type or null if none
771         * exists. If the field cannot be found in the given type, all super-classes are
772         * searched as well. Different from {@link Class#getField(String)}, this will
773         * also work for non-public fields.
774         */
775        public static Field getField(Class<?> type, String name) throws SecurityException {
776                while (type != null) {
777                        try {
778                                return type.getDeclaredField(name);
779                        } catch (NoSuchFieldException e) {
780                                type = type.getSuperclass();
781                        }
782                }
783                return null;
784        }

786        /**
787         * Sets the value of a field on the given object.
788         * The field may also be private and/or final.
789         */
790        public static void setFieldValue(Object object, String fieldName, Object newValue) throws IOException {
791                Field field = ReflectionUtils.getField(object.getClass(), fieldName);
792                Field modifiersField;
793                try {
794                        modifiersField = Field.class.getDeclaredField("modifiers");
795                        modifiersField.setAccessible(true);
796                        modifiersField.setInt(field, field.getModifiers() & ~Modifier.FINAL);
797                        field.set(object, newValue);
798                } catch (NoSuchFieldException e) {
799                        throw new IOException("Failed to get modifiers", e);
800                } catch (IllegalAccessException e) {
801                        throw new IOException("Failed to access field", e);
802                }
803        }

813        /**
814         * Unified interface to iterable types (i.e. array or implements
815         * {@link Iterable}).
816         * 
817         * @throws IllegalArgumentException
818         *             if given object does not conform to one of the expected types
819         */
820        public static Iterable<?> asIterable(Object arrayOrIterable) {
821                if (arrayOrIterable instanceof Iterable) {
822                        return (Iterable<?>) arrayOrIterable;
823                }
824                if (arrayOrIterable.getClass().isArray()) {
825                        Class<?> componentType = arrayOrIterable.getClass().getComponentType();
826                        if (componentType.isArray() || componentType.isPrimitive()) {
827                                ArrayList<Object> arrayContent = new ArrayList<>();
828                                for (int i = 0; i < Array.getLength(arrayOrIterable) - 1; i++) {
829                                        arrayContent.add(Array.get(arrayOrIterable, i));
830                                }
831                                return arrayContent;
832                        }
833                        return Arrays.asList((Object[]) arrayOrIterable);
834                }
835                throw new IllegalArgumentException(
836                                "Array or Iterable expected, but was given " + arrayOrIterable.getClass().getCanonicalName());
837        }
