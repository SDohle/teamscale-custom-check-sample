001package org.conqat.lib.commons.resources;
002
003import static org.conqat.lib.commons.filesystem.FileSystemUtils.ensureParentDirectoryExists;
004import static org.conqat.lib.commons.filesystem.FileSystemUtils.readStreamUTF8;
005
006import java.io.File;
007import java.io.FileOutputStream;
008import java.io.IOException;
009import java.io.InputStream;
010import java.net.URLClassLoader;
011import java.util.Arrays;
012import java.util.List;
013import java.util.Objects;
014
015import org.conqat.lib.commons.filesystem.FileSystemUtils;
016import org.conqat.lib.commons.string.StringUtils;
017
018/**
019 * Represents a java resource which is identified by a {@link #contextClass} and
020 * a {@link #path}. The resource will be loaded using the same class loader that
021 * was used to load the {@link #contextClass}. The path consists of the class'
022 * package name plus the {@link #path}. A resource object is always guaranteed
023 * to exist meaning that instantiation of a non-existent resource will fail.
024 */
025public class Resource implements Comparable<Resource> {
026
027        /** Character that separates a resource' basename from the extension. */
028        private static final char RESOURCE_EXTENSION_SEPARATOR = '.';
029
030        /**
031         * The context class. The resource is expected below the class's package name
032         * and will be loaded using the same class loader that was used to load the
033         * class.
034         */
035        private final Class<?> contextClass;
036
037        /**
038         * A path including the resource name and extension relative to the package name
039         * of the #contextClass. The path is separated by forward slashes on all
040         * operating systems. The path must not navigate up with "..".
041         */
042        private final String path;
043
044        /**
045         * Creates an instance of a resource. If the resource does not exist and
046         * AssertionError is thrown. Check with {@link #exists(Class, String)} before
047         * creating an instance in case it might be missing.
048         */
049        public static Resource of(Class<?> contextClass, String path) {
050                boolean exists = contextClass.getResource(path) != null;
051                if (!exists) {
052                        ClassLoader contextClassLoader = Thread.currentThread().getContextClassLoader();
053                        String message = getAbsolutePath(contextClass, path) + " does not exist!";
054                        if (contextClassLoader instanceof URLClassLoader) {
055                                message += " Classpath: " + Arrays.toString(((URLClassLoader) contextClassLoader).getURLs());
056                        } else {
057                                message += " Context class loader is of type " + contextClassLoader.getClass();
058                        }
059                        throw new AssertionError(message);
060                }
061                return new Resource(contextClass, path);
062        }
063
064        /**
065         * Returns the resource's absolute path including the {@link #contextClass}'
066         * package name.
067         */
068        public static String getAbsolutePath(Class<?> contextClass, String path) {
069                if (path.startsWith(String.valueOf(FileSystemUtils.UNIX_SEPARATOR))) {
070                        // strip leading slash
071                        return path.substring(1);
072                }
073                return ResourceUtils.getPackageResourcePath(contextClass) + FileSystemUtils.UNIX_SEPARATOR + path;
074        }
075
076        /**
077         * Returns true is the resource exists.
078         */
079        public static boolean exists(Class<?> contextClass, String path) {
080                return contextClass.getResource(path) != null;
081        }
082
083        /** Visible for testing only. */
084        /* package */ Resource(Class<?> contextClass, String path) {
085                this.path = path;
086                this.contextClass = contextClass;
087        }
088
089        /** Returns the name of the resource including the extension. */
090        public String getName() {
091                return FileSystemUtils.getLastPathSegment(path);
092        }
093
094        /** Returns the resource's base name without extension. */
095        public String getBaseName() {
096                return StringUtils.removeLastPart(getName(), RESOURCE_EXTENSION_SEPARATOR);
097        }
098
099        /** Returns the resource's extension. */
100        public String getExtension() {
101                return StringUtils.getLastPart(getName(), RESOURCE_EXTENSION_SEPARATOR);
102        }
103
104        /**
105         * Returns true is the resource's extension matches one of the given extensions.
106         * In contrast to {@link #getExtension()} this method supports also extensions
107         * that contain dots.
108         */
109        public boolean hasExtension(String... extensions) {
110                return Arrays.stream(extensions)
111                                .anyMatch(extension -> getName().endsWith(RESOURCE_EXTENSION_SEPARATOR + extension));
112        }
113
114        /** Returns the resource's context class. */
115        public Class<?> getContextClass() {
116                return contextClass;
117        }
118
119        /**
120         * Returns the resource's path relative to the {@link #contextClass} including
121         * the resource's name.
122         */
123        public String getPath() {
124                return path;
125        }
126
127        /**
128         * Returns the resource's parent path. This is the {@link #path} without the
129         * resource's name.
130         */
131        public String getParentPath() {
132                return StringUtils.stripSuffix(path, getName());
133        }
134
135        /**
136         * Returns the resource's absolute path including the {@link #contextClass}'
137         * package name.
138         */
139        public String getAbsolutePath() {
140                return getAbsolutePath(contextClass, path);
141        }
142
143        /**
144         * Returns only a single path segment of the resource's path. If the index is
145         * negative the path segments are counted from the end of the path. E.g. for
146         * "base/sub/name.txt" 0=base, -1=name.txt, -2=1=sub.
147         */
148        public String getPathSegmentAt(int index) {
149                String[] pathSegments = FileSystemUtils.getPathSegments(path);
150                if (index < 0) {
151                        return pathSegments[pathSegments.length + index];
152                }
153                return pathSegments[index];
154        }
155
156        /**
157         * Get resource as stream.
158         */
159        public InputStream getAsStream() {
160                return contextClass.getResourceAsStream(path);
161        }
162
163        /**
164         * Returns the resource as a raw byte array.
165         */
166        public byte[] getAsByteArray() {
167                try {
168                        return FileSystemUtils.readStreamBinary(getAsStream());
169                } catch (IOException e) {
170                        throw ResourceException.newLoadFailed(this, e);
171                }
172        }
173
174        /**
175         * Get resource as string (assuming UTF-8 encoding) without normalized line
176         * separators.
177         */
178        public String getContentAsRawString() {
179                try {
180                        return readStreamUTF8(getAsStream());
181                } catch (IOException e) {
182                        throw ResourceException.newLoadFailed(this, e);
183                }
184        }
185
186        /**
187         * Get resource as string (assuming UTF-8 encoding) with normalized line
188         * separators.
189         */
190        public String getContent() {
191                return StringUtils.normalizeLineSeparatorsPlatformIndependent(getContentAsRawString());
192        }
193
194        /**
195         * Returns the content of the resource split into lines.
196         */
197        public List<String> getLines() {
198                return StringUtils.splitLinesAsList(getContentAsRawString());
199        }
200
201        /**
202         * Creates a temporary copy of the resource in the systems default temp file
203         * location. The file will be marked for deletion when the JVM shuts down.
204         */
205        public File getAsTmpFile() throws IOException {
206                File tempFile = File.createTempFile("test-data-", getName());
207                copyTo(tempFile);
208                tempFile.deleteOnExit();
209                return tempFile;
210        }
211
212        /**
213         * Copies the resource to the given file. Parent directories are created if they
214         * do not exist yet. If the file itself already exists it wil be overridden.
215         */
216        public File copyTo(File file) throws IOException {
217                ensureParentDirectoryExists(file);
218                try (FileOutputStream output = new FileOutputStream(file); InputStream input = getAsStream()) {
219                        FileSystemUtils.copy(input, output);
220                }
221                return file;
222        }
223
224        /**
225         * Unzips the resource into into the given folder under a folder with the same
226         * name as the resource's base name.
227         */
228        public void unzipTo(File tempDir) throws IOException {
229                FileSystemUtils.unzip(getAsStream(), tempDir);
230        }
231
232        @Override
233        public int compareTo(Resource other) {
234                return getAbsolutePath().compareTo(other.getAbsolutePath());
235        }
236
237        @Override
238        public String toString() {
239                return getName();
240        }
241
242        @Override
243        public boolean equals(Object o) {
244                if (this == o) {
245                        return true;
246                }
247                if (o == null || getClass() != o.getClass()) {
248                        return false;
249                }
250                Resource resource = (Resource) o;
251                return getAbsolutePath().equals(resource.getAbsolutePath());
252        }
253
254        @Override
255        public int hashCode() {
256                return Objects.hash(getAbsolutePath());
257        }
258}
