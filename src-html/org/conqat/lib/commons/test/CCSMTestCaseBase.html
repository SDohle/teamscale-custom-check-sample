017package org.conqat.lib.commons.test;
018
019import static org.junit.Assert.assertEquals;
020import static org.junit.Assert.assertTrue;
021import static org.junit.Assert.fail;
022
023import java.io.File;
024import java.io.IOException;
025import java.util.Collection;
026import java.util.HashSet;
027
028import org.conqat.lib.commons.filesystem.CanonicalFile;
029import org.conqat.lib.commons.filesystem.FileSystemUtils;
030
031/**
032 * Base class for test cases that access test data files. This class provides a
033 * simple mechanism for accessing test data files in a specified directory and
034 * provides statistics on test file usage and non-usage.
035 * <p>
036 * The test files a test case accesses must reside in the following location:
037 * 
038 * <pre>
039 *        test-data/&lt;Name of the package the test case resides in&gt;
040 * </pre>
041 * 
042 * For example if a test case is defined in package <code>demo.test</code> the
043 * test files it accesses must be located in directory
044 * <code>test-data/demo.test</code>.
045 */
046public abstract class CCSMTestCaseBase {
047
048        /** Tmp directory. */
049        private final static File TEST_TMP_ROOT_DIRECTORY = new File("test-tmp");
050
051        /** Test data manager for this test case. */
052        private final TestDataManager testDataManager = TestDataManager.getInstance(getClass());
053
054        /** Tmp directory. */
055        private final File tmpDirectory = new File(TEST_TMP_ROOT_DIRECTORY, getClass().getPackage().getName());
056
057        /** Default constructor */
058        public CCSMTestCaseBase() {
059                super();
060        }
061
062        /** Get a test file. (Useful for JUnit 4 test cases.) */
063        public static File useTestFile(Class<?> cls, String fileName) {
064                return TestDataManager.getInstance(cls).getTestFile(fileName);
065        }
066
067        /** Returns the tmp-directory for the given test class. */
068        public static File getTmpDirectory(Class<?> cls) {
069                return TestDataManager.getInstance(cls).getTmpDirectory();
070        }
071
072        /**
073         * Use test file. This method does not actually access the file, so no IO
074         * exception can be raised. This method uses a {@link TestDataManager} to
075         * log access to test data files.
076         * 
077         * @param filename
078         *            Name of the file
079         * @return the file.
080         */
081        protected File useTestFile(String filename) {
082                return testDataManager.getTestFile(filename, getClass());
083        }
084
085        /** Returns the test data directory for this test case. */
086        protected File getTestDataDirectory() {
087                return useTestFile(".");
088        }
089
090        /**
091         * Create a temporary file in a subdirectory of the test temp directory.
092         * Directories are created as needed.
093         * 
094         * @param filename
095         *            name of the file
096         * @param content
097         *            content
098         * @return the file
099         * @throws IOException
100         *             if an IO exception occurrs
101         */
102        protected File createTmpFile(String filename, String content) throws IOException {
103                File file = new File(tmpDirectory, filename);
104                FileSystemUtils.writeFile(file, content);
105                return file;
106        }
107
108        /** Get temporary directory. */
109        protected File getTmpDirectory() {
110                return tmpDirectory;
111        }
112
113        /** Delete temporary directory. */
114        protected void deleteTmpDirectory() {
115                if (tmpDirectory.isDirectory()) {
116                        FileSystemUtils.deleteRecursively(tmpDirectory);
117                }
118        }
119
120        /**
121         * Same as {@link #useTestFile(String)} but returns a {@link CanonicalFile}.
122         * If canonization fails, this makes the current test fail.
123         */
124        protected CanonicalFile useCanonicalTestFile(String filename) {
125                try {
126                        return new CanonicalFile(useTestFile(filename));
127                } catch (IOException e) {
128                        fail("Problem canonizing file: " + filename + ": " + e.getMessage());
129                        return null;
130                }
131        }
132
133        /**
134         * Same as {@link #createTmpFile(String, String)} but returns a
135         * {@link CanonicalFile}. If canonization fails, this makes the current test
136         * fail.
137         */
138        protected CanonicalFile createCanonicalTmpFile(String filename, String content) throws IOException {
139                return canonize(createTmpFile(filename, content));
140        }
141
142        /**
143         * Canonize file. If canonization fails, this makes the current test fail.
144         */
145        protected CanonicalFile canonize(File file) {
146                try {
147                        return new CanonicalFile(file);
148                } catch (IOException e) {
149                        fail("Problem canonizing file: " + file + ": " + e.getMessage());
150                        return null;
151                }
152        }
153
154        /**
155         * Asserts that the given collections equal, except for their order. Since
156         * we use a {@link HashSet} for this purpose, the items inside the
157         * collection must properly implement {@link #hashCode()} and
158         * {@link #equals(Object)}.
159         */
160        public static <T> void assertEqualsExceptOrder(String message, Collection<T> expected, Collection<T> actual) {
161                assertEquals(message, new HashSet<>(expected), new HashSet<>(actual));
162        }
163
164        /** Asserts that the delegate throws an exception of the given type. */
165        public static <T extends Throwable> void assertThrows(Class<T> exceptionClass, IExceptionAction action) {
166                try {
167                        action.perform();
168                        fail("Expected expection of type '" + exceptionClass.getName() + "'.");
169                } catch (Throwable e) {
170                        assertTrue("Expected expection of type '" + exceptionClass.getName() + "', but caught '"
171                                        + e.getClass().getName() + "'.", exceptionClass.isInstance(e));
172                }
173        }
174
175        /** Describes a delegate method that is able to throw an exception. */
176        @FunctionalInterface
177        public static interface IExceptionAction {
178                /** Performs the action. */
179                public void perform() throws Exception;
180        }
181
182}
