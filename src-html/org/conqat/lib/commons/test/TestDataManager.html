017package org.conqat.lib.commons.test;
018
019import java.io.File;
020import java.io.FileWriter;
021import java.io.IOException;
022import java.io.PrintWriter;
023import java.util.ArrayList;
024import java.util.Collections;
025import java.util.HashMap;
026import java.util.HashSet;
027import java.util.Map;
028
029import org.conqat.lib.commons.collections.SetMap;
030import org.conqat.lib.commons.filesystem.FileSystemUtils;
031import org.conqat.lib.commons.string.StringUtils;
032
033/**
034 * Support class for identifying unused test data files. This class provides a
035 * method to access test data files and logs which test cases access which test
036 * files. On every access to a file, access statistic for used and unused files
037 * are written to {@value #REPORT_DIRECTORY_NAME}.
038 * 
039 * @author Florian Deissenboeck
040 * @author Benjamin Hummel
041 */
042public class TestDataManager {
043
044        /** Test data directory. */
045        private final static File TEST_DATA_ROOT_DIRECTORY = new File("test-data");
046
047        /** Name of the directory to write reports to. */
048        public static final String REPORT_DIRECTORY_NAME = "test-tmp";
049
050        /** Map of all instances (which is indexed by managed directory). */
051        private static Map<File, TestDataManager> instances = new HashMap<>();
052
053        /** Returns the instance of the test data manager for the given class. */
054        public static TestDataManager getInstance(Class<?> clazz) {
055                File directory = new File(TEST_DATA_ROOT_DIRECTORY, clazz.getPackage().getName());
056                if (!instances.containsKey(directory)) {
057                        instances.put(directory, new TestDataManager(directory, clazz));
058                }
059                return instances.get(directory);
060        }
061
062        /** The set of unused files. */
063        private final HashSet<String> unusedFiles = new HashSet<>();
064
065        /** Storage for all test files used so far. */
066        private final SetMap<Class<?>, String> usedFiles = new SetMap<>();
067
068        /** The directory this manager works in. */
069        private final File directory;
070
071        /** The class this manager was instantiated for. */
072        private Class<?> testCaseClass;
073
074        /** Private constructor. */
075        private TestDataManager(File directory, Class<?> testClass) {
076                this.directory = directory;
077                this.testCaseClass = testClass;
078
079                if (!directory.exists() || !directory.isDirectory()) {
080                        return;
081                }
082
083                for (File file : directory.listFiles()) {
084                        if (file.isFile()) {
085                                unusedFiles.add(file.getName());
086                        }
087                }
088        }
089
090        /** Returns a file pointing to the tmp directory. */
091        public File getTmpDirectory() {
092                return new File(REPORT_DIRECTORY_NAME, getClass().getPackage().getName());
093        }
094
095        /**
096         * Marks the given file as used and returns the complete file (with
097         * directory).
098         */
099        public File getTestFile(String fileName) {
100                return getTestFile(fileName, testCaseClass);
101        }
102
103        /**
104         * Marks the given file as used and returns the complete file (with
105         * directory).
106         */
107        public File getTestFile(String fileName, Class<?> testCaseClass) {
108                usedFiles.add(testCaseClass, fileName);
109                unusedFiles.remove(fileName);
110                updateUsageReports();
111
112                return new File(directory, fileName);
113        }
114
115        /**
116         * Print a summary on used and unused test data files into a directory
117         * specific log file.
118         */
119        private void updateUsageReports() {
120                try {
121                        File baseDir = new File(REPORT_DIRECTORY_NAME);
122                        FileSystemUtils.ensureDirectoryExists(baseDir);
123                        String fname = directory.toString().replaceAll("[\\\\/]", "_");
124
125                        PrintWriter pw = new PrintWriter(new FileWriter(new File(baseDir, fname + "_usage.txt")));
126                        printUsedFiles(pw);
127                        pw.close();
128
129                        pw = new PrintWriter(new FileWriter(new File(baseDir, fname + "_unusage.txt")));
130                        printUnusedFiles(pw);
131                        pw.close();
132                } catch (IOException e) {
133                        // This is the best we can do (as we are in testing)
134                        e.printStackTrace();
135                }
136        }
137
138        /** Print a report on all files not used. */
139        public void printUnusedFiles(PrintWriter pw) {
140                pw.println("Unused files for directory " + directory + ": " + unusedFiles.size());
141                ArrayList<String> fileList = new ArrayList<>(unusedFiles);
142                Collections.sort(fileList);
143                for (String fileName : fileList) {
144                        pw.print("  ");
145                        pw.println(fileName);
146                }
147
148                pw.flush();
149        }
150
151        /** Print a report on all files used. */
152        public void printUsedFiles(PrintWriter pw) {
153                pw.println("Used files for directory " + directory);
154                for (Class<?> key : usedFiles.getKeys()) {
155                        pw.print("  ");
156                        pw.println(key);
157                        for (String fileName : usedFiles.getCollection(key)) {
158                                pw.print("      ");
159                                pw.println(fileName);
160                        }
161                }
162                pw.println();
163                pw.flush();
164        }
165
166        /** Read content of file from test-data folder as a string. */
167        public String readTestDataFile(String fileName) throws IOException {
168                String fileContent = FileSystemUtils.readFile(getTestFile(fileName));
169                return StringUtils.replaceLineBreaks(fileContent, "\n");
170        }
171
172}
